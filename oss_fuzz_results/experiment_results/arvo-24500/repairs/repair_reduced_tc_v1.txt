template <typename T>
void hb_vector_t<T>::resize(int new_size) {
    if (new_size == size_) {
        return;
    }

    // Allocate new buffer
    T* new_data = static_cast<T*>(malloc(new_size * sizeof(T)));
    if (!new_data) {
        // Handle out-of-memory error
        throw std::bad_alloc();
    }

    // Copy existing data to new buffer
    for (int i = 0; i < size_; ++i) {
        new_data[i] = data_[i];
    }

    // Free old buffer and update pointers
    free(data_);
    data_ = new_data;
    size_ = new_size;
    allocated_size_ = new_size;
}
