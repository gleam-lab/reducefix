#### 3.1 检查内存分配和释放

```cpp
template <typename T>
void hb_vector_t<T>::resize(int new_size) {
    if (new_size > alloc) {
        int old_alloc = alloc;
        alloc = std::max(alloc * 2, new_size); // 确保至少分配两倍的当前大小
        array = static_cast<T*>(realloc(array, alloc * sizeof(T)));
        if (!array) {
            throw std::bad_alloc();
        }
        // 初始化新分配的内存
        for (int i = old_alloc; i < alloc; ++i) {
            new (&array[i]) T(); // 使用 placement new 初始化对象
        }
    } else if (new_size < alloc) {
        // 释放多余的内存
        for (int i = new_size; i < alloc; ++i) {
            array[i].~T(); // 调用析构函数
        }
        array = static_cast<T*>(realloc(array, new_size * sizeof(T)));
        alloc = new_size;
    }
}
```
