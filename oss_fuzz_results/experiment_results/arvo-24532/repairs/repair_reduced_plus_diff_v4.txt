template <typename T>
void hb_vector_t<T>::resize(int new_size) {
    if (new_size < 0) {
        // Handle negative size error
        fprintf(stderr, "Error: Negative size requested\n");
        return;
    }

    if (new_size == _size) {
        // No change needed
        return;
    }

    T* new_array = nullptr;
    try {
        new_array = static_cast<T*>(malloc(new_size * sizeof(T)));
    } catch (...) {
        // Handle memory allocation failure
        fprintf(stderr, "Error: Memory allocation failed\n");
        return;
    }

    if (!new_array) {
        // Handle null pointer
        fprintf(stderr, "Error: Null pointer returned from malloc\n");
        return;
    }

    // Initialize new elements to default value
    for (int i = _size; i < new_size; ++i) {
        new (&new_array[i]) T(); // Placement new
    }

    // Copy existing elements to new array
    for (int i = 0; i < std::min(_size, new_size); ++i) {
        new_array[i] = _array[i];
    }

    // Destruct and free old elements
    for (int i = 0; i < _size; ++i) {
        _array[i].~T();
    }
    free(_array);

    _array = new_array;
    _size = new_size;
}
