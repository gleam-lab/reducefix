```cpp
template <typename T>
class hb_vector_t {
public:
    void push(const T& value);
private:
    T* data;
    int size;
    int capacity;

    void resize(int new_capacity);
};

template <typename T>
void hb_vector_t<T>::push(const T& value) {
    if (size == capacity) {
        resize(capacity * 2);
    }
    if (size >= capacity) { // 添加边界检查
        throw std::out_of_range("Vector overflow");
    }
    data[size++] = value;
}

template <typename T>
void hb_vector_t<T>::resize(int new_capacity) {
    if (new_capacity <= capacity) { // 添加边界检查
        throw std::invalid_argument("New capacity must be greater than current capacity");
    }
    T* new_data = new T[new_capacity];
    try {
        for (int i = 0; i < size; ++i) {
            new_data[i] = data[i];
        }
    } catch (...) {
        delete[] new_data;
        throw;
    }
    delete[] data;
    data = new_data;
    capacity = new_capacity;
}
```

通过这些修改，可以在调整大小时进行边界检查，并在发生越界时抛出适当的异常，从而避免 segmentation-fault。
