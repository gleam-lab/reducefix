def generated_reduce(input_bytes: bytes, test_fn, verbose: bool = True) -> bytes:
    """
    Format-aware test case reducer for ffmpeg
    Handles fuzzer-generated media container and codec bitstream files.
    Prioritizes aggressive prefix reduction and zero-filling based on analysis of common fuzzer patterns.
    """
    import time

    start_time = time.time()
    test_count = 0
    current = input_bytes

    if verbose:
        print("[ReduceFix] Starting reduction")
        print(f"[ReduceFix] Original size: {len(current)} bytes")

    # Edge case: empty or tiny input
    if len(current) <= 1:
        return current

    def run_test(candidate: bytes) -> bool:
        nonlocal test_count
        test_count += 1
        success = test_fn(candidate)
        if test_count % 100 == 0 and verbose:
            elapsed = time.time() - start_time
            print(f"  [ReduceFix] Progress: {test_count} tests, {len(candidate)} bytes, {elapsed:.1f}s")
        return success

    # ================================================================================
    # Stage 0: Aggressive prefix reduction - try keeping only first N%
    # ================================================================================
    if verbose:
        print("[ReduceFix] Stage 0: Aggressive prefix reduction (binary search min length)")
        print(f"  Starting size: {len(current)} bytes")

    # First try large fractions to get immediate wins
    initial_reduction_found = False
    for fraction in [0.5, 0.25, 0.125]:
        keep_len = int(len(current) * fraction)
        if keep_len == 0:
            continue
        candidate = current[:keep_len]
        if run_test(candidate):
            current = candidate
            initial_reduction_found = True
            if verbose:
                print(f"  [ReduceFix] ✓ Kept only {int(fraction*100)}% of file ({len(current)} bytes)")

    # Binary search for minimal prefix
    low, high = len(current) // 2, len(current)
    best_prefix_len = len(current)
    while low <= high:
        mid = (low + high) // 2
        candidate = current[:mid]
        if run_test(candidate):
            best_prefix_len = mid
            high = mid - 1
        else:
            low = mid + 1

    previous_size = len(current)
    current = current[:best_prefix_len]
    reduction_percent = (previous_size - len(current)) / previous_size * 100 if previous_size > 0 else 0
    if verbose:
        print(f"  [ReduceFix] Binary search: {previous_size} → {len(current)} bytes ({reduction_percent:.1f}% reduction)")
        print(f"[ReduceFix] Stage 0 complete: {input_bytes} → {len(current)} bytes ({(len(input_bytes)-len(current))/len(input_bytes)*100:.1f}% total)")

    # ================================================================================
    # Stage 1: Shrink long repetitive byte runs
    # ================================================================================
    if verbose:
        print("[ReduceFix] Stage 1: Shrink repetitive byte runs")
        print(f"  Starting size: {len(current)} bytes")

    changed = True
    while changed:
        changed = False
        i = 0
        n = len(current)
        while i < n:
            # Find run of identical bytes
            run_start = i
            while i < n and current[i] == current[run_start]:
                i += 1
            run_length = i - run_start

            # Only process runs >= 16 bytes
            if run_length >= 16:
                byte_val = current[run_start]
                # Binary search for minimum required length
                low_run, high_run = 1, run_length
                best_run_len = run_length
                while low_run <= high_run:
                    mid_run = (low_run + high_run) // 2
                    candidate_parts = [
                        current[:run_start],
                        bytes([byte_val]) * mid_run,
                        current[run_start + run_length:]
                    ]
                    candidate = b''.join(candidate_parts)
                    if run_test(candidate):
                        best_run_len = mid_run
                        high_run = mid_run - 1
                    else:
                        low_run = mid_run + 1
                if best_run_len < run_length:
                    current = current[:run_start] + bytes([byte_val]) * best_run_len + current[run_start + run_length:]
                    changed = True
                    break  # Restart scan from beginning
            # Continue after this run
        n = len(current)  # Update length after possible shrink

    if verbose:
        print(f"  [ReduceFix] Run shrinking complete: {len(current)} bytes")

    # ================================================================================
    # Stage 2: Remove fuzzer signatures and trailing padding
    # ================================================================================
    if verbose:
        print("[ReduceFix] Stage 2: Remove fuzzer signatures and trailing padding")
        print(f"  Starting size: {len(current)} bytes")

    signatures = [b'UZZ-T', b'UZZG', b'FUZZ']
    clean_candidate = current
    signature_removed = False

    # Scan for any signature
    for sig in signatures:
        pos = clean_candidate.find(sig)
        if pos != -1:
            # Remove from first signature occurrence to end
            clean_candidate = clean_candidate[:pos]
            signature_removed = True
            if verbose:
                print(f"  [ReduceFix] ✓ Removed signature '{sig.decode()}' at offset {pos}")
            break

    # Also remove long trailing runs of common padding bytes (0x60, 0x20, 0xFF)
    padding_bytes = [0x60, 0x20, 0xFF]
    final_trim = len(clean_candidate)
    for pad_byte in padding_bytes:
        run_len = 0
        for i in range(len(clean_candidate) - 1, max(-1, len(clean_candidate) - 100), -1):
            if clean_candidate[i] == pad_byte:
                run_len += 1
            else:
                break
        if run_len >= 16:
            trim_to = len(clean_candidate) - run_len
            if trim_to > 0:
                temp_candidate = clean_candidate[:trim_to]
                if run_test(temp_candidate):
                    clean_candidate = temp_candidate
                    if verbose:
                        print(f"  [ReduceFix] ✓ Trimmed trailing 0x{pad_byte:02x} run of {run_len} bytes")

    if signature_removed or len(clean_candidate) < len(current):
        current = clean_candidate

    if verbose:
        print(f"  [ReduceFix] Signature/padding removal: {len(current)} bytes")

    # ================================================================================
    # Stage 3: Aggressive zero-fill optimization
    # ================================================================================
    if verbose:
        print("[ReduceFix] Stage 3: Aggressive zero-fill optimization")
        print(f"  Starting size: {len(current)} bytes")

    # Try zeroing second half
    mid = len(current) // 2
    if mid > 0:
        candidate = current[:mid] + bytes(len(current) - mid)
        if run_test(candidate):
            current = candidate
            if verbose:
                print(f"  [ReduceFix] ✓ Zeroed second half ({mid} bytes zeroed)")

    # Try zeroing each quarter
    chunk_size = len(current) // 4
    if chunk_size > 0:
        for i in range(4):
            start = i * chunk_size
            end = start + chunk_size if i < 3 else len(current)
            candidate = current[:start] + bytes(end - start) + current[end:]
            if run_test(candidate):
                current = candidate
                if verbose:
                    print(f"  [ReduceFix] ✓ Zeroed quarter {i+1}/4 (offset {start}-{end})")

    # Try smaller blocks (512, 256)
    for block_size in [512, 256]:
        if block_size >= len(current):
            continue
        i = 0
        while i < len(current):
            end = min(i + block_size, len(current))
            candidate = current[:i] + bytes(end - i) + current[end:]
            if run_test(candidate):
                current = candidate
                # Do not increment i - recheck same position with new data
            else:
                i = end

    if verbose:
        print(f"  [ReduceFix] Zero-fill complete: {len(current)} bytes")

    # ================================================================================
    # Stage 4: Adaptive DDMin with region-aware chunking
    # ================================================================================
    if verbose:
        print("[ReduceFix] Stage 4: Adaptive DDMin reduction")
        print(f"  Starting size: {len(current)} bytes")

    def ddmin_pass(data: bytes, chunk_size: int) -> bytes:
        if len(data) <= chunk_size:
            return data
        result = data
        i = 0
        while i < len(result):
            end = min(i + chunk_size, len(result))
            candidate = result[:i] + result[end:]
            if len(candidate) > 0 and run_test(candidate):
                result = candidate
            else:
                i += chunk_size
        return result

    # Choose chunk sizes based on current file size
    size = len(current)
    if size < 1024:
        chunks = [64, 32, 16, 8, 4, 1]
    elif size < 5 * 1024:
        chunks = [256, 128, 64, 32, 16, 8, 4, 1]
    elif size < 20 * 1024:
        chunks = [512, 256, 128, 64, 32, 16, 8]
    else:
        chunks = [2048, 1024, 512, 256, 128, 64]

    if verbose:
        print(f"  [ReduceFix] Chunk sizes: {chunks}")

    for chunk_size in chunks:
        before = len(current)
        changed = True
        pass_count = 0
        while changed:
            pass_count += 1
            prev_len = len(current)
            current = ddmin_pass(current, chunk_size)
            changed = len(current) < prev_len
            if pass_count > 10:  # Prevent infinite loops
                break
        after = len(current)
        reduction = (before - after) / before * 100 if before > 0 else 0
        if verbose:
            action = "no change" if after == before else f"{after} bytes ({reduction:.1f}% reduction)"
            print(f"  [ReduceFix] Chunk {chunk_size}: {before} → {after} bytes ({action})")

    # ================================================================================
    # Final output
    # ================================================================================
    total_reduction = (len(input_bytes) - len(current)) / len(input_bytes) * 100
    total_time = time.time() - start_time
    if verbose:
        print(f"[ReduceFix] Complete: {len(input_bytes)} → {len(current)} bytes ({total_reduction:.1f}% reduction)")
        print(f"[ReduceFix] Total tests: {test_count}, total time: {total_time:.1f}s")

    return current
