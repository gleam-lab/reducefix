#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int n;
long long a[110], p[110], b[110], q[110], x;

long long gcd(long long u, long long v) {
    return v == 0 ? u : gcd(v, u % v);
}

long long lcm(long long u, long long v) {
    return u / gcd(u, v) * v;
}

bool check(long long v) {
    long long cnt = 0;
    for (int i = 1; i <= n; i++) {
        long long _lcm = lcm(a[i], b[i]);
        long long vm = max(0LL, (v / _lcm - 1) * min(p[i] * (_lcm / a[i]), q[i] * (_lcm / b[i])));
        cnt += vm;

        long long val = (v / _lcm - 1) * _lcm;
        long long ms = LLONG_MAX / 2;
        long long j = 0;
        while (j * a[i] <= v - val) {
            long long rem = (v - val - j * a[i]);
            long long tmp = (rem / b[i]) * q[i];
            if (rem % b[i]) tmp += q[i];
            ms = min(ms, j * p[i] + tmp);
            j++;
        }

        cnt += ms;
        if (cnt > x) return 0;
    }
    return 1;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    long long l = 0, r = 2e18;
    while (l + 1 < r) {
        long long mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }

    cout << l + 1;
    return 0;
}
