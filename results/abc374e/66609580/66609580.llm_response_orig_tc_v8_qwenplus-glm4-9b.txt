#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int gcd(long long a, long long b) {
    while (b) {
        long long t = b;
        b = a % b;
        a = t;
    }
    return a;
}

long long lcm(long long a, long long b) {
    return a / gcd(a, b) * b;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int n;
    long long x;
    cin >> n >> x;
    vector<pair<long long, pair<long long, long long>>> a(n);
    for (int i = 0; i < n; i++) {
        long long a_i, b_i, p_i, q_i;
        cin >> a_i >> p_i >> b_i >> q_i;
        a[i] = {lcm(a_i, b_i), {p_i, q_i}};
    }

    sort(a.begin(), a.end());

    long long k = 0;
    long long v = 0;
    for (auto& item : a) {
        long long l = item.first / x - k;
        if (l < 0) continue;
        long long t = v + item.second.first * l;
        k += item.second.second * l;
        long long g = gcd(t, x);
        x = x / g * t;
        v = x / item.first;
        if (x == v) break;
    }

    cout << x;
    return 0;
}
