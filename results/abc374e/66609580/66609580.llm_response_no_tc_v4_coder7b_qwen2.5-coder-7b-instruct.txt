#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

bool check(const vector<vector<int>>& machines, const vector<int>& limits, int budget, long long target) {
    int n = machines.size();
    for (int i = 0; i < n; ++i) {
        long long lcm_ab = lcm(machines[i][0], machines[i][2]);
        long long min_cost_per_unit = min(machines[i][1] * (lcm_ab / machines[i][0]), machines[i][3] * (lcm_ab / machines[i][2]));
        long long units_can_purchase = budget / min_cost_per_unit;
        if (units_can_purchase == 0) continue;
        long long total_units = units_can_purchase * lcm_ab / min_cost_per_unit;
        if (total_units < target) return false;
    }
    return true;
}

int main() {
    int n, x;
    cin >> n >> x;
    vector<vector<int>> machines(n, vector<int>(4));
    for (auto& m : machines) {
        cin >> m[0] >> m[1] >> m[2] >> m[3];
    }

    long long low = 0, high = 1e9, best = 0;
    while (low <= high) {
        long long mid = (low + high) / 2;
        if (check(machines, {}, x, mid)) {
            best = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << best << endl;
    return 0;
}
