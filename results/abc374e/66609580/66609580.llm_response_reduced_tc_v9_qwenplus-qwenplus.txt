#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll a, ll b) {
    return a / gcd(a, b) * b;
}

bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; ++i) {
        ll l = lcm(a[i], b[i]);
        ll full_cycles = v / l;
        ll contribution = full_cycles * min(p[i] * (l / a[i]), q[i] * (l / b[i]));
        cnt += contribution;
        
        if (cnt > x)
            return false;
            
        ll remaining = v - full_cycles * l;
        ll min_cost = numeric_limits<ll>::max();
        
        for (ll j = 0; j * a[i] <= remaining; ++j) {
            ll rem = remaining - j * a[i];
            ll k = (rem + b[i] - 1) / b[i]; // ceil division
            if (k < 0) k = 0;
            min_cost = min(min_cost, j * p[i] + k * q[i]);
        }
        
        cnt += min_cost;
        if (cnt > x)
            return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> x;
    
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll left = 0, right = 1;
    while (check(right)) {
        right <<= 1;
    }
    
    while (left + 1 < right) {
        ll mid = (left + right) >> 1;
        if (check(mid))
            left = mid;
        else
            right = mid;
    }
    
    cout << left;
    return 0;
}
