```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll INF = 1e18;

int n;
ll a[110], b[110], p[110], q[110], x;

inline bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; i++) {
        ll g = __gcd(a[i], b[i]);
        ll lcm = a[i] / g * b[i];
        
        // Full cycles
        ll full_cycles = v / lcm;
        ll rem = v % lcm;
        ll cost_full = full_cycles * min(p[i], q[i]);
        
        // Check if we can add this to total
        if (cnt > x - cost_full) return false;
        cnt += cost_full;
        
        // Partial cycle
        ll min_cost = INF;
        for (ll j = 0; j * a[i] <= rem; j++) {
            ll rem_after_a = rem - j * a[i];
            ll k = (rem_after_a + b[i] - 1) / b[i]; // ceil division
            if (k < 0) k = 0;
            min_cost = min(min_cost, j * p[i] + k * q[i]);
        }
        
        // Try swapping roles of a and b
        for (ll k = 0; k * b[i] <= rem; k++) {
            ll rem_after_b = rem - k * b[i];
            ll j = (rem_after_b + a[i] - 1) / a[i];
            if (j < 0) j = 0;
            min_cost = min(min_cost, j * p[i] + k * q[i]);
        }
        
        if (cnt > x - min_cost) return false;
        cnt += min_cost;
        
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll l = 0, r = 2e14; // Large enough upper bound
    
    while (l < r) {
        ll mid = (l + r) / 2;
        if (check(mid)) l = mid + 1;
        else r = mid;
    }
    
    // Final answer should be the largest value that doesn't exceed x
    cout << (check(l) ? l - 1 : l - 1);
    
    return 0;
}
```
