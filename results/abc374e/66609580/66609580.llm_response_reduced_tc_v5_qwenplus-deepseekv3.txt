```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
long long a[110], b[110], p[110], q[110], x;

inline bool check(long long v) {
    long long total_cost = 0;
    for (int i = 1; i <= n; ++i) {
        long long lcm_ab = a[i] / __gcd(a[i], b[i]) * b[i];
        long long full_cycles = v / lcm_ab;
        long long remaining = v % lcm_ab;
        
        // Cost for full cycles
        long long cost_full_cycle = min(p[i] * (lcm_ab / a[i]), q[i] * (lcm_ab / b[i]));
        long long cost_full = full_cycles * cost_full_cycle;
        
        // Cost for remaining time
        long long cost_remaining = LLONG_MAX;
        for (long long k = 0; k * a[i] <= remaining; ++k) {
            long long remaining_b = remaining - k * a[i];
            long long num_b = (remaining_b + b[i] - 1) / b[i];  // ceil(remaining_b / b[i])
            cost_remaining = min(cost_remaining, k * p[i] + num_b * q[i]);
        }
        
        total_cost += cost_full + cost_remaining;
        if (total_cost > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    long long left = 0, right = LLONG_MAX / 2;

        if (check(mid)) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }
    
    cout << left << '\n';
    return 0;
}
```
