```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll a, ll b) {
    return a / gcd(a, b) * b;
}

inline bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; ++i) {
        ll l = lcm(a[i], b[i]);
        ll full_cycles = v / l;
        ll cost_per_cycle = min(p[i] * (l / a[i]), q[i] * (l / b[i]));
        ll total_full_cost = full_cycles * cost_per_cycle;
        
        cnt += total_full_cost;
        if (cnt > x) return false;

        // Remaining time after full cycles
        ll rem = v % l;
        ll min_remaining = LLONG_MAX;

        // Try different numbers of item A within remaining time
        for (ll j = 0; j * a[i] <= rem; ++j) {
            ll cost_a = j * p[i];
            ll rem_time = rem - j * a[i];
            ll num_b = rem_time >= 0 ? (rem_time / b[i] + (rem_time % b[i] ? 1 : 0)) : 0;
            ll cost_b = num_b * q[i];
            min_remaining = min(min_remaining, cost_a + cost_b);
        }

        cnt += min_remaining;
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    
    cin >> n >> x;
    for (int i = 1; i <= n; ++i)
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    
    ll l = 0, r = 2e18;
    while (l + 1 < r) {
        ll mid = (l + r) >> 1;
        if (check(mid)) l = mid;
        else r = mid;
    }
    
    cout << l << "\n";
    return 0;
}
```
