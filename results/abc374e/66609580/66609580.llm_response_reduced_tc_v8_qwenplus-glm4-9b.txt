#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <numeric>  // For gcd
using namespace std;

long long gcd(long long a, long long b) {
    return b == 0 ? a : gcd(b, a % b);
}

inline long long lcm(long long a, long long b) {
    return a / gcd(a, b) * b;
}

inline long long min(long long a, long long b) {
    return a < b ? a : b;
}

inline long long max(long long a, long long b) {
    return a > b ? a : b;
}

bool check(long long v) {
    long long cnt = 0;
    for (int i = 1; i <= n; i++) {
        long long lcm_ab = lcm(a[i], b[i]);
        long long vm = max(0LL, (v / lcm_ab - 1) * min(p[i] * (lcm_ab / a[i]), q[i] * (lcm_ab / b[i])));
        cnt += vm;
        long long val = max(0LL, v / lcm_ab - 1) * lcm_ab;
        long long ms = LLONG_MAX >> 1;
        long long j = 0;
        while (j * a[i] + val <= v) {
            long long remaining = v - val - j * a[i];
            ms = min(ms, j * p[i] + max(0LL, (remaining / b[i]) * q[i]));
            j++;
        }
        cnt += ms;
        if (cnt > x) return 0;
    }
    return 1;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    long long l = 0, r = 1LL << 60;
    while (l + 1 < r) {
        long long mid = (l + r) >> 1;
        if (check(mid)) l = mid;
        else r = mid;
    }
    cout << l;
    return 0;
}
