#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll x, ll y) {
    return x / gcd(x, y) * y;
}

bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; ++i) {
        ll g = gcd(a[i], b[i]);
        ll l = a[i] / g * b[i];
        
        // Full cycles
        ll full_cycles = max(0LL, v / l - 1);
        if (full_cycles > 0) {
            ll cost_per_cycle = min(p[i] * (l / a[i]), q[i] * (l / b[i]));
            cnt += full_cycles * cost_per_cycle;
            if (cnt > x) return false;
        }
        
        // Remaining part after full cycles
        ll remaining = v - full_cycles * l;
        
        // Try all possible counts of A in the remaining part
        ll min_cost = INF;
        for (ll j = 0; j * a[i] <= remaining && j <= x / p[i]; ++j) {
            ll rem = remaining - j * a[i];
            ll k = rem >= 0 ? (rem + b[i] - 1) / b[i] : 0;
            ll cost = j * p[i] + k * q[i];
            min_cost = min(min_cost, cost);
            
            // Optimization: limit iterations
            if (j > 200) break;
        }
        
        cnt += min_cost;
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll l = 0, r = 1;
    while (check(r)) r <<= 1;
    
    while (l + 1 < r) {
        ll mid = (l + r) >> 1;
        if (check(mid)) l = mid;
        else r = mid;
    }
    
    cout << l << '\n';
    return 0;
}
