#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

long long gcd(long long x, long long y) {
    return y == 0 ? x : gcd(y, x % y);
}

long long lcm(long long x, long long y) {
    return x / gcd(x, y) * y;
}

bool check(long long v) {
    long long cnt = 0;
    for (int i = 1; i <= n; ++i) {
        long long _lcm = lcm(a[i], b[i]);
        long long vm = (v / _lcm - 1) * min(p[i] * (_lcm / a[i]), q[i] * (_lcm / b[i]));
        cnt += vm;
        if (cnt > x) return 0;
    }
    return 1;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    long long l = 0, r = 2e18;
    while (l + 1 < r) {
        long long mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }
    cout << l;
    return 0;
}
