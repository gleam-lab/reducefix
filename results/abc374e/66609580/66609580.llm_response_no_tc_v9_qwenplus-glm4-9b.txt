#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int n;
long long a[110], b[110], p[110], q[110], x;

// Helper function to calculate the greatest common divisor (GCD)
auto gcd = [](long long a, long long b) {
    return b == 0 ? a : gcd(b, a % b);
};

// Function to check if a given value v is valid
bool check(long long v) {
    long long cnt = 0;
    for (int i = 0; i < n; i++) {
        long long lcm = (a[i] / gcd(a[i], b[i])) * b[i];
        long long vm = max(0LL, (v / lcm - 1) * min(p[i] * (lcm / a[i]), q[i] * (lcm / b[i])));
        cnt += vm;

        long long val = max(0LL, v / lcm - 1) * lcm;
        long long ms = LLONG_MAX >> 1;
        long long j = 0;
        while (j * a[i] <= v - val) {
            long long remainder = (v - val - j * a[i]);
            long long q_part = remainder / b[i];
            long long r_part = remainder % b[i];
            ms = min(ms, j * p[i] + q_part * q[i]);
            j++;
        }
        cnt += ms;
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n >> x;
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    long long left = 0, right = (1LL << 32) - 1;
    while (left < right) {
        long long mid = (left + right + 1) / 2;
        if (check(mid)) left = mid;
        else right = mid - 1;
    }
    cout << left;
    return 0;
}
