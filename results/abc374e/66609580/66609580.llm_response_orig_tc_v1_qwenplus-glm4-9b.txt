#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const long long MOD = 1e9 + 7;

struct Data {
    long long a, b, p, q;
};

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    
    int n;
    long long x;
    cin >> n >> x;
    
    vector<Data> data(n);
    for (int i = 0; i < n; ++i) {
        cin >> data[i].a >> data[i].p >> data[i].b >> data[i].q;
    }
    
    long long l = 0, r = 1e18;
    while (l + 1 < r) {
        long long mid = (l + r) / 2;
        long long cnt = 0;
        
        for (auto& d : data) {
            long long lcm = (d.a / __gcd(d.a, d.b)) * d.b;
            long long term1 = mid / lcm * min(d.p * (lcm / d.a), d.q * (lcm / d.b));
            long long term2 = (mid / lcm - 1);
            
            long long m;
            long long current = 0;
            for (long long j = 0; j * d.a + current <= mid; ++j) {
                long long next = j * d.a + current;
                long long part = max(0LL, (next - mid) / d.b);
                m = min(m, j * d.p + part * d.q);
                current = next;
            }
            
            cnt = (cnt + term1 + term2 + m) % MOD;
            if (cnt > x) break;
        }
        
        if (cnt > x) r = mid;
        else l = mid;
    }
    
    cout << l + 1 << "\n";
    return 0;
}
