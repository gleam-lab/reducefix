#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const ll INF = 1e18;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll x, ll y) {
    return x / gcd(x, y) * y;
}

// Computes the minimum cost to buy at least 'need' items from one vendor
ll compute_cost(ll a, ll p, ll b, ll q, ll need) {
    ll res = INF;
    // Try all possible counts of item A up to what we need
    for (ll cntA = 0; cntA <= need && cntA * a <= INF; cntA++) {
        ll remaining = max(0LL, need - cntA);
        ll cntB = (remaining + b - 1) / b;  // ceil division
        res = min(res, cntA * p + cntB * q);
    }
    return res;
}

inline bool check(ll v) {
    ll total = 0;
    for (int i = 1; i <= n; i++) {
        ll l = lcm(a[i], b[i]);
        ll chunk_size = l / a[i] * p[i] + l / b[i] * q[i];
        
        // Full chunks
        ll full_chunks = v / l;
        ll chunk_cost = full_chunks * chunk_size;
        total += chunk_cost;
        
        if (total > x) return false;
        
        // Remaining time
        ll rem_time = v % l;
        
        // Compute minimal cost to get enough items within remaining time
        ll max_items = 0;
        for (ll t = 0; t <= rem_time; t++) {
            ll items_a = t / a[i];
            ll items_b = (rem_time - t) / b[i];
            max_items = max(max_items, items_a + items_b);
        }
        
        ll additional_cost = compute_cost(a[i], p[i], b[i], q[i], max_items);
        total += additional_cost;
        
        if (total > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll low = 0, high = 2e14;
    while (low < high) {
        ll mid = (low + high + 1) / 2;
        if (check(mid)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }
    
    cout << low << endl;
}
