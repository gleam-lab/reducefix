```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll a, ll b) {
    return a / gcd(a, b) * b;
}

bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; ++i) {
        ll l = lcm(a[i], b[i]);
        ll full_cycles = v / l;
        
        // Calculate cost for full cycles
        ll full_cost = full_cycles * min(p[i] * (l / a[i]), q[i] * (l / b[i]));
        if (full_cost > x) return false;
        
        // Remaining time after full cycles
        ll rem_time = v - full_cycles * l;
        
        // Try all possible uses of a[i]
        ll min_single_cost = INF;
        for (ll j = 0; ; ++j) {
            ll time_a = j * a[i];
            if (time_a > rem_time) break;
            
            ll rem_after_a = rem_time - time_a;
            ll k = (rem_after_a + b[i] - 1) / b[i]; // ceil division
            
            ll total_cost = j * p[i] + k * q[i];
            min_single_cost = min(min_single_cost, total_cost);
        }
        
        cnt += full_cost + min_single_cost;
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> x;
    
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll l = 0, r = 2e18;
    while (l + 1 < r) {
        ll mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }
    
    cout << l << "\n";
}
```
