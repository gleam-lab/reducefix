#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int n;
ll a[110], b[110], p[110], q[110], x;

inline ll lcm(ll a, ll b) {
    return a / gcd(a, b) * b;
}

bool check(ll v) {
    ll cnt = 0;
    for (int i = 1; i <= n; ++i) {
        ll l = lcm(a[i], b[i]);
        ll full_cycles = v / l;
        
        // Full cycle cost
        ll full_cost = full_cycles * min(p[i] * (l / a[i]), q[i] * (l / b[i]));
        cnt += full_cost;
        if (cnt > x) return false;
        
        // Remaining part after full cycles
        ll rem = v % l;
        ll min_single = LLONG_MAX;
        
        // Try all possible number of a[i] usages in the remaining part
        for (ll j = 0; j * a[i] <= rem; ++j) {
            ll rem_after_a = rem - j * a[i];
            ll num_b = (rem_after_a + b[i] - 1) / b[i]; // ceil division
            min_single = min(min_single, j * p[i] + num_b * q[i]);
            
            // Optimization: early exit if we already exceed the best option
            if (min_single == 0) break;
        }
        
        cnt += min_single;
        if (cnt > x) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    ll l = 0, r = 1e18; // Use a sufficiently large upper bound
    while (l < r) {
        ll mid = (l + r) / 2;
        if (check(mid)) {
            l = mid + 1;
        } else {
            r = mid;
        }
    }
    
    // Final check to ensure l is the correct maximum value that doesn't exceed budget
    if (check(l)) {
        cout << l << '\n';
    } else {
        cout << l - 1 << '\n';
    }
    
    return 0;
}
