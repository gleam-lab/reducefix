#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate cost for this ingredient
        ld cost_per_a = (ld)p[i] / a[i];
        ld cost_per_b = (ld)q[i] / b[i];

        if (cost_per_a <= cost_per_b) {
            // Prefer type A first
            int count_a = mid / a[i];
            int remaining = mid % a[i];
            total_cost += (ld)count_a * p[i] + (ld)remaining * cost_per_b;
        } else {
            // Prefer type B first
            int count_b = mid / b[i];
            int remaining = mid % b[i];
            total_cost += (ld)count_b * q[i] + (ld)remaining * cost_per_a;
        }
    }

    return total_cost <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n); p.resize(n); b.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) >> 1;
        if (canMake(mid)) left = mid;
        else right = mid - 1;
    }

    cout << left;

    return 0;
}
