```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate cost per unit for both options
        ld costA = p[i] / (ld)a[i];
        ld costB = q[i] / (ld)b[i];

        if (costA < costB) {
            // Prefer using more of A
            int fullSetsA = mid / a[i];
            int remaining = mid % a[i];
            total_cost += fullSetsA * p[i];
            // Use B for the remainder
            total_cost += remaining * costB;
        } else {
            // Prefer using more of B
            int fullSetsB = mid / b[i];
            int remaining = mid % b[i];
            total_cost += fullSetsB * q[i];
            // Use A for the remainder
            total_cost += remaining * costA;
        }
    }

    return (int)(ceil(total_cost)) <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n); b.resize(n); p.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) >> 1;

        if (canMake(mid)) left = mid;
        else right = mid - 1;
    }

    cout << left;

    return 0;
}
```
