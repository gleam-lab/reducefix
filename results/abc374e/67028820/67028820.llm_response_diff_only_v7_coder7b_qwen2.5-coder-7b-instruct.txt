```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, q, p;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        int x = mid / a[i];
        total_cost += min(x, (mid + b[i] - 1) / b[i]) * p[i];
        x = mid / b[i];
        total_cost += min(x, (mid + a[i] - 1) / a[i]) * q[i];
    }

    return (int)(ceil(total_cost)) <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n); b.resize(n); p.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right) >> 1;

        if (canMake(mid)) left = mid + 1;
        else right = mid;
    }

    cout << left - 1;

    return 0;
}
```
