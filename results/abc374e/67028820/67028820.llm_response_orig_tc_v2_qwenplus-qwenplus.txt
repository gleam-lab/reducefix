### ✅ Fix Strategy

Instead of using floating points, we will:
- Use **binary search** as before.
- In `canMake(mid)`, avoid floating point by using **integer arithmetic only**, approximating comparisons via cross-multiplication.
- Compute required ingredients and their costs precisely using integer math.

We treat each ingredient type as a "resource" that provides some amount of base material at a cost. We want to make `mid` units of the final product, so we calculate how much of each resource we should use.

For each resource:
- It gives `a[i]` units of material per item at price `p[i]`
- Or `b[i]` units at price `q[i]`

We pick the better option based on **cost per unit material**, again using integer comparison without floats.

### ✅ Final Corrected Code

```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    int total_cost = 0;
    for (int i = 0; i < n; ++i) {
        // Compare cost per unit: p[i]/a[i] vs q[i]/b[i]
        // Use cross multiplication to avoid division
        if (p[i] * b[i] < q[i] * a[i]) {
            // a[i] is cheaper per unit
            int count = mid / a[i];
            int rem = mid % a[i];
            total_cost += count * p[i];
            total_cost += rem * (p[i] + (a[i] - 1)) / a[i]; // ceil(rem * p[i]/a[i])
        } else {
            // b[i] is cheaper or equal
            int count = mid / b[i];
            int rem = mid % b[i];
            total_cost += count * q[i];
            total_cost += rem * (q[i] + (b[i] - 1)) / b[i]; // ceil(rem * q[i]/b[i])
        }

        if (total_cost > x)
            return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    a.resize(n); p.resize(n); b.resize(n); q.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) / 2;
        if (canMake(mid))
            left = mid;
        else
            right = mid - 1;
    }

    cout << left << '\n';

    return 0;
}
```
