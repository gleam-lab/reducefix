#include <bits/stdc++.h>
using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

// Function to check if we can make at least 'mid' units of goods
bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        int temp = mid;
        ld s = p[i] / (ld)a[i]; // Cost per unit from source i
        ld t = q[i] / (ld)b[i]; // Cost per unit from source j

        if (s < t) {
            int x = min(mid / a[i], 1LL); // Maximum units we can take from source i without exceeding mid
            total_cost += x * p[i];
            temp -= x * a[i];
            total_cost += temp * t; // Add remaining cost from source j
        } else {
            int x = min(mid / b[i], 1LL); // Maximum units we can take from source j without exceeding mid
            total_cost += x * q[i];
            temp -= x * b[i];
            total_cost += temp * s; // Add remaining cost from source i
        }
    }

    return ceil(total_cost) <= x; // Check if total cost is within budget
}

signed main() {
    cin >> n >> x;
    a.resize(n), b.resize(n), p.resize(n), q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right) >> 1; // Midpoint calculation

        if (canMake(mid)) right = mid; // If we can afford 'mid', try for a smaller value
        else left = mid + 1; // Otherwise, increase 'mid'
    }

    cout << left << endl; // Output the minimum number of units that can be made

    return 0;
}

1. **Corrected `mid` Calculation**: In the binary search loop, the calculation of `mid` was slightly off (`(left + right + 1) >> 1`). It should be `(left + right) >> 1`.
