```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate how much we need of each ingredient in the patch
        int required_a = a[i];
        int required_b = b[i];
        ld price_p = p[i];
        ld price_q = q[i];

        // Strategy: Try to buy as many full patches as possible with cheaper combination,
        // then handle leftover with better price

        ld cost_per_patch = price_p + price_q;

        // Number of full patches we can make from "mid" items
        int num_patches = mid;

        // Total cost for full patches
        total_cost += num_patches * cost_per_patch;
    }

    // Use ceiling to ensure we don't accept solutions that exceed budget
    return ceil(total_cost) <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n); b.resize(n); p.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) >> 1;

        if (canMake(mid)) left = mid;
        else right = mid - 1;
    }

    cout << left;

    return 0;
}
```
