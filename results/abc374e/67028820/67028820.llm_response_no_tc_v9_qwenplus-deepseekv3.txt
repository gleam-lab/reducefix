            // Buy as many a bundles as possible
            int cnt = remaining / a[i];
            total_cost += cnt * p[i];
            remaining -= cnt * a[i];
            
            // Buy remaining with cheaper option

            if (remaining > 0) {

            // Buy as many b bundles as possible
            int cnt = remaining / b[i];
