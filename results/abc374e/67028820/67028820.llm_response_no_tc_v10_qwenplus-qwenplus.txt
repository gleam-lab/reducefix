#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate cost for this factory
        ld cost_per_a = (ld)p[i] / a[i];
        ld cost_per_b = (ld)q[i] / b[i];

        if (cost_per_a < cost_per_b) {
            // Prefer buying full sets of type A first
            int count = mid / a[i];
            int remainder = mid % a[i];
            total_cost += (ld)count * p[i] + (ld)remainder * cost_per_b;
        } else {
            // Prefer buying full sets of type B first
            int count = mid / b[i];
            int remainder = mid % b[i];
            total_cost += (ld)count * q[i] + (ld)remainder * cost_per_a;
        }
    }

    // Use a small epsilon to handle precision issues with floating point comparisons
    return (total_cost <= x + 1e-7);
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    a.resize(n); p.resize(n); b.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 2e14; // Increased upper bound to handle large possible answers

    while (left < right) {
        int mid = (left + right + 1) / 2;

        if (canMake(mid)) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }

    cout << left;

    return 0;
}
