```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // We need to make 'mid' units using machine i
        ld cost_per_a = p[i] / (ld)a[i];
        ld cost_per_b = q[i] / (ld)b[i];

        if (cost_per_a < cost_per_b) {
            // Prefer buying full sets of A first
            int num_full_a = mid / a[i];
            int remaining = mid % a[i];
            total_cost += num_full_a * p[i];
            total_cost += remaining * cost_per_a;
        } else {
            // Prefer buying full sets of B first
            int num_full_b = mid / b[i];
            int remaining = mid % b[i];
            total_cost += num_full_b * q[i];
            total_cost += remaining * cost_per_b;
        }
    }

    return total_cost <= x + 1e-9; // Handle floating point precision issues
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    a.resize(n); p.resize(n); b.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) / 2;

        if (canMake(mid)) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }

    cout << left;

    return 0;
}
```
