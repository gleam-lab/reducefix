```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        ld s = (ld)p[i] / a[i];
        ld t = (ld)q[i] / b[i];

        if (s < t) {
            int x = min(mid / a[i], q[i] / b[i]);
            total_cost += x * p[i];
            mid -= x * a[i];
        } else {
            int x = min(mid / b[i], p[i] / a[i]);
            total_cost += x * q[i];
            mid -= x * b[i];
        }
    }

    return total_cost <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n), b.resize(n), p.resize(n), q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right) >> 1;

        if (canMake(mid)) right = mid;
        else left = mid + 1;
    }

    cout << left << endl;

    return 0;
}
```
