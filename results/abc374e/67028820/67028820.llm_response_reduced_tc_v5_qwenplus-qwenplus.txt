```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate how many full sets of type A we can buy
        int countA = mid / a[i];
        int remaining = mid - countA * a[i];
        
        // Cost of buying countA sets of A and the rest with B
        ld cost = (ld)countA * p[i] + (ld)remaining * ((ld)b[i] / (ld)a[i]) * q[i];
        
        // Alternatively, try buying more B sets if they're cheaper
        int countB = mid / b[i];
        remaining = mid - countB * b[i];
        cost = min(cost, (ld)countB * q[i] + (ld)remaining * ((ld)a[i] / (ld)b[i]) * p[i]);
        
        total_cost += cost;
    }

    return total_cost <= x + 1e-8; // Allow small floating point error
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    a.resize(n); p.resize(n); b.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 1e10;
    while (left < right) {
        int mid = (left + right + 1) >> 1;

        if (canMake(mid)) left = mid;
        else right = mid - 1;
    }

    cout << left << "\n";

    return 0;
}
```
