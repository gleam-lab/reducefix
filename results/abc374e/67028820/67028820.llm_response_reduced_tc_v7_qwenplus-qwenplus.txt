```cpp
#include <bits/stdc++.h>

using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, p, q;

bool canMake(int mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        // Calculate how many full sets of A and B we can make
        int pieces = mid;

        ld costA = p[i] / (ld)a[i];  // cost per unit from A
        ld costB = q[i] / (ld)b[i];  // cost per unit from B

        if (costA <= costB) {
            // Prefer buying more from A
            int numAsets = pieces / a[i];
            int leftover = pieces % a[i];
            total_cost += numAsets * p[i];
            total_cost += leftover * costB;
        } else {
            // Prefer buying more from B
            int numBsets = pieces / b[i];
            int leftover = pieces % b[i];
            total_cost += numBsets * q[i];
            total_cost += leftover * costA;
        }
    }

    return total_cost <= x + 1e-8; // Allow for floating point imprecision
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    a.resize(n); b.resize(n); p.resize(n); q.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int left = 0, right = 2e14;
    while (left < right) {
        int mid = (left + right + 1) >> 1;

        if (canMake(mid)) left = mid;
        else right = mid - 1;
    }

    cout << left << "\n";

    return 0;
}
```
