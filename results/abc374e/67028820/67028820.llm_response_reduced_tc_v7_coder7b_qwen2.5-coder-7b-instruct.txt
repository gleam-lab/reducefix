```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define ld long double

int n, x;
vector<int> a, b, q, p;

bool canMake(ld mid) {
    ld total_cost = 0;
    for (int i = 0; i < n; i++) {
        ld s = p[i] / (ld)a[i];
        ld t = q[i] / (ld)b[i];
        if (s < t) {
            total_cost += ceil(mid / a[i]) * p[i];
        } else {
            total_cost += ceil(mid / b[i]) * q[i];
        }
    }
    return total_cost <= x;
}

signed main() {
    cin >> n >> x;
    a.resize(n), b.resize(n), p.resize(n), q.resize(n);
    for (int i = 0; i < n; i++) cin >> a[i] >> p[i] >> b[i] >> q[i];

    ld left = 0, right = 1e10;
    while (right - left > 1e-7) {
        ld mid = (left + right) / 2;
        if (canMake(mid)) left = mid;
        else right = mid;
    }

    cout << floor(left);

    return 0;
}
```
