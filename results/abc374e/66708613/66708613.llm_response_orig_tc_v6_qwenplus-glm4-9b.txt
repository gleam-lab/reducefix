#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

int main() {
    int N;
    ll X;
    cin >> N >> X;
    vector<pair<ll, pair<ll, ll>>> machines(N);

    for (int i = 0; i < N; ++i) {
        ll A, P, B, Q;
        cin >> A >> P >> B >> Q;
        machines[i] = {A, {P, B, Q}};
    }

    // Sort machines based on their A value
    sort(machines.begin(), machines.end(), [](const auto& a, const auto& b) {
        return a.first < b.first;
    });

    // Binary search for the minimum processing capacity
    ll low = 0, high = X;
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll cost = 0;
        bool possible = true;

        for (const auto& machine : machines) {
            ll A = machine.first;
            ll P = machine.second.first;
            ll B = machine.second.second;
            ll Q = machine.second.third;

            ll maxMachines = (mid - A + B - 1) / B; // Maximum machines that can be used
            ll minMachines = (mid - A + A - 1) / A; // Minimum machines that can be used

            if (maxMachines < minMachines) {
                possible = false;
                break;
            }

            ll machinesUsed = min(maxMachines, minMachines);
            ll costForMachines = machinesUsed * P;
            ll costForTime = (mid - machinesUsed * A) / machinesUsed * Q;
            cost += costForMachines + costForTime;
        }

        if (possible && cost <= X) {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << (low == X + 1 ? 0 : low) << endl;

    return 0;
}
