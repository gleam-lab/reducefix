#include<iostream>

using namespace std;
using ll = long long;
const int inf=1<<30;
const ll INF=1LL<<62;
using P = pair<ll,int>;
using PP = pair<int,P>; 
const ll MOD=998244353;
const int dy[]={-1,0,1,0};
const int dx[]={0,1,0,-1};

int main(){
    int N;
    ll X;
    cin>>N>>X;
    vector<ll> A(N),P(N),B(N),Q(N);
    
    for(int i=0;i<N;i++){
        cin>>A[i]>> P[i]>> B[i] >> Q[i];
    }

    auto f=[&](ll x){
        ll res=X;
        for(int i=0;i<N;i++){
            ll val=INF;
            for(ll s=0;s<=B[i];s++){
                ll t=(max(0LL,x-A[i]*s)+B[i]-1)/B[i];
                val=min(val,s*P[i]+t*Q[i]);
            }
            for(ll t=0;t<=A[i];t++){
                ll s=(max(0LL,x-B[i]*t)+A[i]-1)/A[i];
                val=min(val,t*Q[i]+s*P[i]);
            }
            if(val==INF){
                return false;
            }
            res-=val;
        }
        return res>=0;
    };

    ll ub=INF,lb=-INF; // lb is set to -INF to cover the case when no positive result is possible
    while(ub-lb>1){
        ll mid=(ub+lb)/2;
        if(f(mid)){
            lb=mid;
        }else{
            ub=mid;
        }
    }
   
    cout<<(lb == -INF ? 0 : lb)<<endl;
}
