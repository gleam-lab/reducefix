#include <iostream>
#include <vector>
#include <climits>
#include <algorithm>
using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

int N;
ll X;
vector<ll> A, P, B, Q;

bool canAchieve(ll target) {
    ll total_cost = 0;

    for (int i = 0; i < N; ++i) {
        ll min_cost = INF;

        // Try all possible numbers of A machines up to what might help
        for (ll a = 0; a <= 1000000 / A[i] + 1 && a * P[i] <= X; ++a) {
            ll remaining = max(0LL, target - a * A[i]);
            if (remaining == 0) {
                min_cost = min(min_cost, a * P[i]);
                continue;
            }

            if (B[i] == 0) continue; // Cannot fulfill with B machines

            ll b_needed = (remaining + B[i] - 1) / B[i];
            ll cost = a * P[i] + b_needed * Q[i];
            if (cost <= X)
                min_cost = min(min_cost, cost);
        }

        // Try all possible numbers of B machines
        for (ll b = 0; b <= 1000000 / B[i] + 1 && b * Q[i] <= X; ++b) {
            ll remaining = max(0LL, target - b * B[i]);
            if (remaining == 0) {
                min_cost = min(min_cost, b * Q[i]);
                continue;
            }

            if (A[i] == 0) continue;

            ll a_needed = (remaining + A[i] - 1) / A[i];
            ll cost = b * Q[i] + a_needed * P[i];
            if (cost <= X)
                min_cost = min(min_cost, cost);
        }

        if (min_cost == INF)
            return false;

        total_cost += min_cost;
        if (total_cost > X)
            return false;
    }

    return true;
}

int main() {
    cin >> N >> X;
    A.resize(N); P.resize(N); B.resize(N); Q.resize(N);

    for (int i = 0; i < N; ++i)
        cin >> A[i] >> P[i] >> B[i] >> Q[i];

    // Binary search on answer
    ll low = 0, high = 1LL << 60;
    ll ans = 0;

    while (low <= high) {
        ll mid = (low + high) / 2;
        if (canAchieve(mid)) {
            ans = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << ans << endl;
    return 0;
}
