    for(int i=0;i<N;i++){
        cin>>A[i]>> P[i]>> B[i] >> Q[i];
    }

        ll res=X;
        for(int i=0;i<N;i++){
            ll val=INF;
            for(ll s=0;s<=B[i];s++){//챈짤혺챈짖째s찾혗혪B[i]채쨩짙찾혖쩐찾혖짠채쩍쩔찾혖혛

            }
            res-=val;
        }
        return res>=0;
    };

            lb=mid;
        }else{
            ub=mid;
        }
    }

}
