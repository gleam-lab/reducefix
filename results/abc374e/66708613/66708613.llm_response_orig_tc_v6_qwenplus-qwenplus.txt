#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

const int inf = 1 << 30;
const ll INF = 1LL << 62;

struct Machine {
    ll A, P, B, Q;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<Machine> machines(N);
    for (int i = 0; i < N; ++i) {
        cin >> machines[i].A >> machines[i].P >> machines[i].B >> machines[i].Q;
    }

    // Binary search for the maximum achievable processing capacity
    auto canAchieve = [&](ll capacity) -> bool {
        ll totalCost = 0;
        for (const auto& m : machines) {
            ll min_cost = INF;

            // Try all possible counts of machine A within reasonable limits
            for (ll a = 0; a * m.A <= capacity + m.B - 1; ++a) {
                ll remaining = max(0LL, capacity - a * m.A);
                ll b = (remaining + m.B - 1) / m.B;
                ll cost = a * m.P + b * m.Q;
                min_cost = min(min_cost, cost);
                if (cost > X) break; // No need to try more if already over budget
            }

            // Try all possible counts of machine B within reasonable limits
            for (ll b = 0; b * m.B <= capacity + m.A - 1; ++b) {
                ll remaining = max(0LL, capacity - b * m.B);
                ll a = (remaining + m.A - 1) / m.A;
                ll cost = b * m.Q + a * m.P;
                min_cost = min(min_cost, cost);
                if (cost > X) break; // No need to try more if already over budget
            }

            if (min_cost > X) return false;
            totalCost += min_cost;
            if (totalCost > X) return false;
        }
        return true;
    };

    ll low = 0, high = 1e18;
    ll result = 0;
    while (low <= high) {
        ll mid = (low + high) / 2;
        if (canAchieve(mid)) {
            result = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << result << "\n";
    return 0;
}
