#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int N;
    ll X;
    cin >> N >> X;
    vector<pair<ll, pair<ll, ll>>> machines(N); // (processing time, cost per unit time, min cost per unit processing time)
    
    for (int i = 0; i < N; ++i) {
        ll a, p, b, q;
        cin >> a >> p >> b >> q;
        ll min_cost = p + b * q;
        machines[i] = {a, {min_cost, q}};
    }
    
    sort(machines.begin(), machines.end());
    
    // Binary Search for the minimum achievable processing capacity
    auto can_buy = [&](ll processing_capacity) {
        ll cost = 0;
        for (auto& [a, {min_cost, q}] : machines) {
            ll required = (processing_capacity + a - 1) / a; // ceil(processing_capacity / a)
            ll cost_needed = required * min_cost;
            if (cost_needed > cost) {
                cost = cost_needed;
            }
        }
        return cost <= X;
    };
    
    ll lb = 0, ub = 1e18; // Set upper bound to a large number
    while (ub - lb > 1) {
        ll mid = (ub + lb) / 2;
        if (can_buy(mid)) {
            ub = mid;
        } else {
            lb = mid;
        }
    }
    
    if (can_buy(lb)) {
        cout << lb << endl;
    } else {
        cout << 0 << endl;
    }
    
    return 0;
}
