#include<iostream>

using namespace std;
using ll = long long;
const int inf=1<<30;
const ll INF=1LL<<62;
using P = pair<ll,int>;
using PP = pair<int,P>; 
const ll MOD=998244353;
const int dy[]={-1,0,1,0};
const int dx[]={0,1,0,-1};

bool check(vector<ll>& A, vector<ll>& P, vector<ll>& B, vector<ll>& Q, ll X, ll target) {
    for(int i=0;i<A.size();i++) {
        ll total_cost = INF;
        for(ll s=0;s<=B[i];s++) {
            ll t = (max(0LL, target - A[i]*s) + B[i] - 1) / B[i];
            total_cost = min(total_cost, s * P[i] + t * Q[i]);
        }
        for(ll t=0;t<=A[i];t++) {
            ll s = (max(0LL, target - B[i]*t) + A[i] - 1) / A[i];
            total_cost = min(total_cost, t * Q[i] + s * P[i]);
        }
        if(total_cost > X) return false;
    }
    return true;
}

int main() {
    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);

    for(int i=0;i<N;i++) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    if(X == 0) {
        cout << 0 << endl;
        return 0;
    }

    ll lb = 0, ub = INF;
    while(lb < ub) {
        ll mid = (lb + ub + 1) / 2;
        if(check(A, P, B, Q, X, mid)) {
            lb = mid;
        } else {
            ub = mid - 1;
        }
    }

    cout << lb << endl;

    return 0;
}
