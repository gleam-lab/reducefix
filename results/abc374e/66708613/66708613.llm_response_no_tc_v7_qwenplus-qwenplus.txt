#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <cassert>
using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

// Function to check if processing capacity 'x' is achievable with budget X
auto is_possible = [](ll x, ll X, const vector<ll>& A, const vector<ll>& P,
                      const vector<ll>& B, const vector<ll>& Q, int N) {
    ll remaining = X;
    for (int i = 0; i < N; ++i) {
        ll min_cost = INF;

        // Try different numbers of type A machines and compute needed type B
        for (ll s = 0; s <= B[i]; ++s) {
            ll need_A = max(0LL, (x - A[i] * s + B[i] - 1) / B[i]);
            if (need_A <= A[i]) {
                ll cost = s * P[i] + need_A * Q[i];
                min_cost = min(min_cost, cost);
            }
        }

        // Try different numbers of type B machines and compute needed type A
        for (ll t = 0; t <= A[i]; ++t) {
            ll need_B = max(0LL, (x - B[i] * t + A[i] - 1) / A[i]);
            if (need_B <= B[i]) {
                ll cost = t * Q[i] + need_B * P[i];
                min_cost = min(min_cost, cost);
            }
        }

        if (min_cost == INF || remaining < min_cost) {
            return false;
        }
        remaining -= min_cost;
    }
    return true;
};

int main() {
    int N;
    ll X;
    cin >> N >> X;

    vector<ll> A(N), P(N), B(N), Q(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // Binary search on the maximum achievable processing capacity
    ll low = 0, high = 1LL << 60;
    while (low < high) {
        ll mid = (low + high + 1) / 2;
        if (is_possible(mid, X, A, P, B, Q, N)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }

    cout << low << endl;
    return 0;
}
