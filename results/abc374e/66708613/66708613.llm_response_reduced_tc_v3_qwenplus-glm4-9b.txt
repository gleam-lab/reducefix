#include <iostream>
#include <string>
#include <queue>

#include <assert.h>

#include <algorithm>
using namespace std;
using ll = long long;
const int inf = 1 << 30;
const ll INF = 1LL << 62;
using P = pair<ll, int>;
using PP = pair<int, P>;
const ll MOD = 998244353;
const int dy[] = {-1, 0, 1, 0};
const int dx[] = {0, 1, 0, -1};

int main() {
    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    
    for (int i = 0; i < N; i++) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    auto f = [&](ll x) {
        ll res = X;
        for (int i = 0; i < N; i++) {
            ll val = INF;
            // Calculate minimum cost for using B[i] machines
            for (ll s = 0; s <= B[i]; s++) {
                ll t = (max(0LL, x - A[i] * s) + B[i] - 1) / B[i];
                val = min(val, s * P[i] + t * Q[i]);
            }
            // Calculate minimum cost for using A[i] machines
            for (ll t = 0; t <= A[i]; t++) {
                ll s = (max(0LL, x - B[i] * t) + A[i] - 1) / A[i];
                val = min(val, t * Q[i] + s * P[i]);
            }
            // If no valid configuration is found
            if (val == INF) {
                return false;
            }
            res -= val; // Subtract the cost from the total budget
        }
        return res >= 0; // Check if the remaining budget is non-negative
    };

    ll ub = 1LL << 60, lb = 0; // lb is the minimum possible value
    while (ub - lb > 1) {
        ll mid = (ub + lb) / 2;
        if (f(mid)) {
            lb = mid;
        } else {
            ub = mid;
        }
    }
    
    cout << lb << endl; // Output the minimum processing capacity that can be achieved within budget X.
    
    return 0;
}
