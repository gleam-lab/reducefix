#include<iostream>
#include<vector>
#include<numeric>
#include<algorithm>
using namespace std;
using ll = long long;

const int inf = 1 << 30;
const ll INF = 1LL << 62;

// Main function
int main() {
    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);

    for(int i = 0; i < N; ++i) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // Function to check if processing capacity x can be achieved within budget X
    auto is_possible = [&](ll x) {
        ll remaining = X;
        for(int i = 0; i < N; ++i) {
            ll min_cost = INF;

            // Try all possible numbers of machine A used (s)
            for(ll s = 0; s <= B[i]; ++s) {
                ll a_total = A[i] * s;
                if(a_total >= x) {
                    min_cost = min(min_cost, s * P[i]);
                    continue;
                }
                ll needed_b = (x - a_total + B[i] - 1) / B[i];
                if(needed_b <= B[i]) {
                    min_cost = min(min_cost, s * P[i] + needed_b * Q[i]);
                }
            }

            // Try all possible numbers of machine B used (t)
            for(ll t = 0; t <= A[i]; ++t) {
                ll b_total = B[i] * t;
                if(b_total >= x) {
                    min_cost = min(min_cost, t * Q[i]);
                    continue;
                }
                ll needed_a = (x - b_total + A[i] - 1) / A[i];
                if(needed_a <= A[i]) {
                    min_cost = min(min_cost, t * Q[i] + needed_a * P[i]);
                }
            }

            if(min_cost > X || min_cost == INF) {
                return false;
            }
            remaining -= min_cost;
            if(remaining < 0) {
                return false;
            }
        }
        return true;
    };

    // Binary search for maximum achievable processing capacity
    ll low = 0, high = 1LL << 60;
    while(low < high) {
        ll mid = (low + high + 1) / 2;
        if(is_possible(mid)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }

    cout << low << endl;
    return 0;
}
