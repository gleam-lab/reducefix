def can_achieve(capacity):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Option 1: use only S_i machines
        if a > 0:
            num_s = (capacity + a - 1) // a  # ceil(capacity / a)
            min_cost = min(min_cost, num_s * p)
        
        # Option 2: use only T_i machines
        if b > 0:
            num_t = (capacity + b - 1) // b  # ceil(capacity / b)
            min_cost = min(min_cost, num_t * q)
        
        # Option 3: mix of both machines
        # Use j units of S_i and remaining with T_i
        if a > 0:
            max_s = (capacity + a - 1) // a
            for j in range(max_s + 1):
                processed_by_s = j * a
                remaining = max(0, capacity - processed_by_s)
                if remaining == 0:
                    cost = j * p
                else:
                    num_t = (remaining + b - 1) // b  # ceil(remaining / b)
                    cost = j * p + num_t * q
                min_cost = min(min_cost, cost)
        
        # Also try using T_i first then fill with S_i
        if b > 0:
            max_t = (capacity + b - 1) // b
            for j in range(max_t + 1):
                processed_by_t = j * b
                remaining = max(0, capacity - processed_by_t)
                if remaining == 0:
                    cost = j * q
                else:
                    num_s = (remaining + a - 1) // a  # ceil(remaining / a)
                    cost = j * q + num_s * p
                min_cost = min(min_cost, cost)
                
        total_cost += min_cost
        if total_cost > x:
            return False
    return total_cost <= x

n, x = map(int, input().split())
machines = []
for _ in range(n):
    a, p, b, q = map(int, input().split())
    machines.append((a, p, b, q))

# Binary search on the production capacity
lo, hi = 0, 2 * 10**7  # upper bound: even with cheapest machine (cost 1), X yen can buy X units

while lo < hi:
    mid = (lo + hi + 1) // 2
    if can_achieve(mid):
        lo = mid
    else:
        hi = mid - 1

print(lo)
