def min_cost_for_capacity(target):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Try all possible numbers of S_i machines
        # We need at least ceil(target / a) from S_i if we use only S_i
        max_s = target // a + 1 if a > 0 else 0
        for s_count in range(max_s + 1):
            products_from_s = s_count * a
            remaining = max(0, target - products_from_s)
            
            # Calculate how many T_i machines needed
            t_count = (remaining + b - 1) // b if b > 0 else float('inf')
            if b == 0:
                if remaining > 0:
                    continue  # Cannot fulfill with T_i
                else:
                    t_count = 0
            
            cost = s_count * p + t_count * q
            min_cost = min(min_cost, cost)
            
            # Early termination if cost already too high
            if cost > x:
                break
                
        if min_cost == float('inf'):
            return float('inf')  # Impossible to achieve target in this process
        
        total_cost += min_cost
        if total_cost > x:
            break
            
    return total_cost

# Read input
n, x = map(int, input().split())
machines = []
for _ in range(n):
    a, p, b, q = map(int, input().split())
    machines.append((a, p, b, q))

# Binary search on the production capacity
left, right = 0, 2 * 10**7  # Upper bound: max possible capacity

while left < right:
    mid = (left + right + 1) // 2
    if min_cost_for_capacity(mid) <= x:
        left = mid
    else:
        right = mid - 1

print(left)
