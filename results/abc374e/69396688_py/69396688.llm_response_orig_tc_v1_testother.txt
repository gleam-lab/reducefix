def can_achieve(capacity):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Maximum units needed for S_i to achieve at least `capacity`
        max_units_s = (capacity + a - 1) // a if a > 0 else float('inf')
        # Try all possible numbers of S_i machines (from 0 up to what's needed)
        for s_units in range(max_units_s + 1):
            products_from_s = s_units * a
            remaining = max(0, capacity - products_from_s)
            t_units = (remaining + b - 1) // b if b > 0 else 0
            cost = s_units * p + t_units * q
            if cost < min_cost:
                min_cost = cost
        total_cost += min_cost
        if total_cost > x:
            return False
    return total_cost <= x

# Read input
n, x = map(int, input().split())
machines = []
for _ in range(n):
    machines.append(list(map(int, input().split())))

# Binary search on the production capacity
lo = 0
hi = 2 * 10**7  # Upper bound: even with minimal per-machine output, budget allows this much

while lo < hi:
    mid = (lo + hi + 1) // 2
    if can_achieve(mid):
        lo = mid
    else:
        hi = mid - 1

print(lo)
