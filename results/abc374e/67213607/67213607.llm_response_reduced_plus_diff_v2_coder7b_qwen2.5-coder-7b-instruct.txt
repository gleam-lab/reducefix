#include<bits/stdc++.h>
using namespace std;

bool check(long long mid, vector<vector<int>>& v, int x) {
    long long sum = 0;
    for(auto &it : v) {
        int a = it[0], p = it[1], b = it[2], q = it[3];
        long long minn = LLONG_MAX;
        for(int j = 0; j <= b; ++j) {
            if((mid - j * a) <= 0) {
                minn = min(minn, static_cast<long long>(j * p));
            } else if((mid - j * a) % b == 0) {
                minn = min(minn, static_cast<long long>(j * p + (mid - j * a) / b * q));
            } else {
                minn = min(minn, static_cast<long long>(j * p + ((mid - j * a) / b + 1) * q));
            }
        }
        for(int j = 0; j <= a; ++j) {
            if((mid - j * b) <= 0) {
                minn = min(minn, static_cast<long long>(j * q));
            } else if((mid - j * b) % a == 0) {
                minn = min(minn, static_cast<long long>(j * q + (mid - j * b) / a * p));
            } else {
                minn = min(minn, static_cast<long long>(j * q + ((mid - j * b) / a + 1) * p));
            }
        }
        sum += minn;
    }
    return sum <= x;
}

int main() {
    int n, x;
    cin >> n >> x;
    vector<vector<int>> v(n, vector<int>(4));
    for(int i = 0; i < n; ++i) {
        cin >> v[i][0] >> v[i][1] >> v[i][2] >> v[i][3];
    }

    long long l = 0, r = 1e9, ans = 0;
    while(l <= r) {
        long long mid = l + (r - l) / 2;
        if(check(mid, v, x)) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    cout << ans << endl;
    return 0;
}
