#include <bits/stdc++.h>

bool isFeasible(int capacity) {
    long long totalCost = 0;

        int minCost = LLONG_MAX;

                cost += (remainingProducts / b[i]) * q[i];
            } else {
                cost += ((remainingProducts / b[i]) + 1) * q[i];
            }
            
            minCost = min(minCost, cost);
        }
        
        totalCost += minCost;
    }
    
    return totalCost <= x;
}
