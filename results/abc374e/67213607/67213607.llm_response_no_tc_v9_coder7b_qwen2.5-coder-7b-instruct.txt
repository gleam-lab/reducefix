        mid = (l + r) / 2;
        sum = 0;
        for (int i = 1; i <= n; ++i) {

            for (int j = 0; j <= b[i]; ++j) {
                if ((mid - j * a[i]) >= 0) {
                    minn = min(minn, j * p[i]);

                        int k = (mid - j * a[i]) / b[i];
                        minn = min(minn, j * p[i] + (k + 1) * q[i]);
                    }
                }
            }

                if ((mid - j * b[i]) >= 0) {
                    minn = min(minn, j * q[i]);

                        int k = (mid - j * b[i]) / a[i];
                        minn = min(minn, j * q[i] + (k + 1) * p[i]);
                    }
                }
            }
            sum += minn;
        }

        } else {

        }
    }

}
