```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    long long n, x;
    cin >> n >> x;
    vector<long long> a(n), p(n), b(n), q(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    long long l = 0, r = 2e18, ans = 0;

    while(l <= r) {
        long long mid = (l + r) / 2;
        long long cost = 0;

        for(int i = 0; i < n; i++) {
            long long min_cost = 1e18;

            // Try all possible number of type A pills
            for(long long j = 0; j <= b[i] && j * a[i] <= mid; j++) {
                long long remaining = mid - j * a[i];
                if(remaining <= 0) {
                    min_cost = min(min_cost, j * p[i]);
                } else {
                    long long k = (remaining + b[i] - 1) / b[i]; // ceil division
                    min_cost = min(min_cost, j * p[i] + k * q[i]);
                }
            }

            // Try all possible number of type B pills
            for(long long j = 0; j <= a[i] && j * b[i] <= mid; j++) {
                long long remaining = mid - j * b[i];
                if(remaining <= 0) {
                    min_cost = min(min_cost, j * q[i]);
                } else {
                    long long k = (remaining + a[i] - 1) / a[i]; // ceil division
                    min_cost = min(min_cost, j * q[i] + k * p[i]);
                }
            }

            cost += min_cost;
            if(cost > x) break; // Early exit if already too expensive
        }

        if(cost <= x) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << ans << endl;
}
```
