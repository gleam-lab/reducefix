#include<bits/stdc++.h>
using namespace std;

long long n, x, a[105], p[105], b[105], q[105], l, r, mid, ans, minn, sum;

int main() {
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    l = 1;
    r = 1000000005; // Set the upper limit to a safe large number
    while (l <= r) {
        mid = (l + r) / 2;
        sum = 0;
        for (int i = 1; i <= n; i++) {
            minn = 210000000000000000LL; // Ensure minn is long long
            for (int j = 0; j <= b[i]; j++) {
                long long cost = abs(mid - j * a[i]) / b[i] * q[i];
                if (cost < 0) cost = 0;
                minn = min(minn, j * p[i] + cost);
            }
            for (int j = 0; j <= a[i]; j++) {
                long long cost = abs(mid - j * b[i]) / a[i] * p[i];
                if (cost < 0) cost = 0;
                minn = min(minn, j * q[i] + cost);
            }
            sum += minn;
        }
        if (sum <= x) {
            ans = mid; // Update ans to mid when the condition is satisfied
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    cout << ans;
}
