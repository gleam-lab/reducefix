```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 105;
const long long INF = 210000000000000000;

int main() {
    long long n, x;
    cin >> n >> x;
    vector<vector<long long>> a(n + 1), p(n + 1), b(n + 1), q(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i][0] >> p[i][0] >> b[i][0] >> q[i][0];
        a[i].resize(b[i][0] + 1);
        p[i].resize(b[i][0] + 1);
        b[i].resize(a[i][0] + 1);
        q[i].resize(a[i][0] + 1);
        for (int j = 1; j <= b[i][0]; ++j) {
            cin >> a[i][j] >> p[i][j];
        }
        for (int j = 1; j <= a[i][0]; ++j) {
            cin >> b[i][j] >> q[i][j];
        }
    }

    long long l = 0, r = 1e9;
    while (l < r) {
        long long mid = (l + r + 1) / 2;
        long long sum = 0;
        for (int i = 1; i <= n; ++i) {
            long long minn = INF;
            for (int j = 0; j <= b[i][0]; ++j) {
                if ((mid - j * a[i][0]) >= 0) {
                    minn = min(minn, j * p[i][0] + (mid - j * a[i][0]) / a[i][1] * p[i][1]);
                }
                if ((mid - j * a[i][0]) % a[i][1] == 0) {
                    minn = min(minn, j * p[i][0] + (mid - j * a[i][0]) / a[i][1] * q[i][1]);
                } else {
                    minn = min(minn, j * p[i][0] + (mid - j * a[i][0]) / a[i][1] * q[i][1] + q[i][1]);
                }
            }
            for (int j = 0; j <= a[i][0]; ++j) {
                if ((mid - j * b[i][0]) >= 0) {
                    minn = min(minn, j * q[i][0] + (mid - j * b[i][0]) / b[i][1] * p[i][1]);
                }
                if ((mid - j * b[i][0]) % b[i][1] == 0) {
                    minn = min(minn, j * q[i][0] + (mid - j * b[i][0]) / b[i][1] * q[i][1]);
                } else {
                    minn = min(minn, j * q[i][0] + (mid - j * b[i][0]) / b[i][1] * q[i][1] + q[i][1]);
                }
            }
            sum += minn;
        }
        if (sum <= x) {
            l = mid;
        } else {
            r = mid - 1;
        }
    }
    cout << l << endl;
    return 0;
}
```
