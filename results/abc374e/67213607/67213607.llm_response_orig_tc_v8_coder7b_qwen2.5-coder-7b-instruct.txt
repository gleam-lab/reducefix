#include<bits/stdc++.h>
using namespace std;

long long n, x, a[105], p[105], b[105], q[105], l, r, mid, ans, minn, sum;

bool check(long long mid) {
    sum = 0;
    for (int i = 1; i <= n; ++i) {
        minn = LLONG_MAX;
        for (int j = 0; j <= b[i]; ++j) {
            if ((mid - j * a[i]) >= 0) {
                minn = min(minn, (mid - j * a[i]) / b[i] * q[i] + j * p[i]);
            }
        }
        for (int j = 0; j <= a[i]; ++j) {
            if ((mid - j * b[i]) >= 0) {
                minn = min(minn, (mid - j * b[i]) / a[i] * p[i] + j * q[i]);
            }
        }
        sum += minn;
    }
    return sum <= x;
}

int main() {
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    l = 0, r = 1000000005;
    while (l < r) {
        mid = l + (r - l) / 2;
        if (check(mid)) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid;
        }
    }
    cout << ans << endl;
    return 0;
}
