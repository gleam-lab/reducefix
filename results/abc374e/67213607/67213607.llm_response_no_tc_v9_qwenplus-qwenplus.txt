#include<bits/stdc++.h>
using namespace std;

const int MAX_N = 100;
long long n, x;
long long a[MAX_N + 5], p[MAX_N + 5], b[MAX_N + 5], q[MAX_N + 5];
long long l, r, mid, ans;

// Function to calculate minimum cost for one type of item to achieve at least 'need' quality
long long min_cost(long long need, long long count1, long long cost1, long long count2, long long cost2) {
    long long minn = LLONG_MAX;
    // Try all possible number of first items
    for (long long j = 0; j <= count1 && j * count2 <= 1e18; j++) {
        long long remaining = max(0LL, need - j * count2);
        long long k = remaining / count1;
        if (remaining % count1 != 0) k++;
        if (k < 0) continue;
        if (j * cost2 + k * cost1 < 0) continue; // Avoid overflow
        minn = min(minn, j * cost2 + k * cost1);
    }
    return minn;
}

bool is_possible(long long quality) {
    long long total_cost = 0;
    for (int i = 1; i <= n; i++) {
        long long cost1 = min_cost(quality, b[i], q[i], a[i], p[i]);
        long long cost2 = min_cost(quality, a[i], p[i], b[i], q[i]);
        total_cost += min(cost1, cost2);
        if (total_cost > x) return false;
    }
    return total_cost <= x;
}

int main() {
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    l = 0;
    r = 2000000000; // Larger upper bound based on problem constraints
    while (l <= r) {
        mid = (l + r) / 2;
        if (is_possible(mid)) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << ans;
    return 0;
}
