```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 105;
const long long INF = 2e18;

long long n, x, a[MAXN], p[MAXN], b[MAXN], q[MAXN], l, r, mid, ans, minn, sum;

bool check(long long m) {
    for (int i = 1; i <= n; ++i) {
        minn = INF;
        for (int j = 0; j <= b[i]; ++j) {
            long long cost = j * p[i];
            if ((m - j * a[i]) > 0) {
                cost += (m - j * a[i]) / b[i] * q[i];
                if ((m - j * a[i]) % b[i] != 0) {
                    cost += q[i];
                }
            }
            minn = min(minn, cost);
        }
        for (int j = 0; j <= a[i]; ++j) {
            long long cost = j * q[i];
            if ((m - j * b[i]) > 0) {
                cost += (m - j * b[i]) / a[i] * p[i];
                if ((m - j * b[i]) % a[i] != 0) {
                    cost += p[i];
                }
            }
            minn = min(minn, cost);
        }
        sum += minn;
    }
    return sum <= x;
}

int main() {
    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    l = 0, r = 1000000005;
    while (l <= r) {
        mid = (l + r) / 2;
        sum = 0;
        if (check(mid)) {
            ans = max(ans, mid);
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    cout << ans << endl;
    return 0;
}
```
