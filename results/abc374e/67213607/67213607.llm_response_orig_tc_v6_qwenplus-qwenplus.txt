#include <bits/stdc++.h>
using namespace std;

long long n, x;
long long a[105], p[105], b[105], q[105];
long long l, r, mid, ans, sum;

long long compute_min_cost(long long a_i, long long p_i, long long b_i, long long q_i, long long target) {
    long long minn = LLONG_MAX;
    
    // Try all possible number of type A pills
    for (long long j = 0; j <= b_i && j * a_i <= target + b_i; ++j) {
        long long remaining = target - j * a_i;
        if (remaining <= 0) {
            minn = min(minn, j * p_i);
        } else {
            long long k = (remaining + b_i - 1) / b_i; // ceiling division
            if (k >= 0 && k <= b_i)
                minn = min(minn, j * p_i + k * q_i);
        }
    }

    // Try all possible number of type B pills
    for (long long j = 0; j <= a_i && j * b_i <= target + a_i; ++j) {
        long long remaining = target - j * b_i;
        if (remaining <= 0) {
            minn = min(minn, j * q_i);
        } else {
            long long k = (remaining + a_i - 1) / a_i; // ceiling division
            if (k >= 0 && k <= a_i)
                minn = min(minn, j * q_i + k * p_i);
        }
    }

    return minn;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    l = 0;
    r = 1000000000LL * 200; // Max possible total pills: 1e9 per test case * 100 cases

    while (l <= r) {
        mid = (l + r) / 2;
        sum = 0;

        for (int i = 1; i <= n; ++i) {
            long long cost = compute_min_cost(a[i], p[i], b[i], q[i], mid);
            sum += cost;
            if (sum > x) break; // Optimization to exit early
        }

        if (sum <= x) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << ans << '\n';
}
