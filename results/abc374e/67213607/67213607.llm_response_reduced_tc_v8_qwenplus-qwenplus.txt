#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long n, x;
    cin >> n >> x;

    vector<long long> a(n), p(n), b(n), q(n);
    for (long long i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    auto is_possible = [&](long long target) -> bool {
        long long total_cost = 0;
        for (long long i = 0; i < n; ++i) {
            long long min_cost = numeric_limits<long long>::max();
            
            // Try all possible usages of the first item type
            for (long long j = 0; j <= b[i]; ++j) {
                long long remaining = target - j * a[i];
                if (remaining <= 0) {
                    min_cost = min(min_cost, j * p[i]);
                    break;
                }
                long long needed_second = (remaining + b[i] - 1) / b[i];
                total_cost += j * p[i] + needed_second * q[i];
                break;
            }

            // Try all possible usages of the second item type
            for (long long j = 0; j <= a[i]; ++j) {
                long long remaining = target - j * b[i];
                if (remaining <= 0) {
                    min_cost = min(min_cost, j * q[i]);
                    break;
                }
                long long needed_first = (remaining + a[i] - 1) / a[i];
                total_cost += j * q[i] + needed_first * p[i];
                break;
            }

            // Choose the minimum cost between the two strategies
            total_cost += min(
                [&]() {
                    long long res = numeric_limits<long long>::max();
                    for (long long j = 0; j <= b[i]; ++j) {
                        long long remaining = target - j * a[i];
                        if (remaining <= 0) {
                            res = min(res, j * p[i]);
                            break;
                        }
                        long long needed_second = (remaining + b[i] - 1) / b[i];
                        res = min(res, j * p[i] + needed_second * q[i]);
                    }
                    return res;
                }(),
                [&]() {
                    long long res = numeric_limits<long long>::max();
                    for (long long j = 0; j <= a[i]; ++j) {
                        long long remaining = target - j * b[i];
                        if (remaining <= 0) {
                            res = min(res, j * q[i]);
                            break;
                        }
                        long long needed_first = (remaining + a[i] - 1) / a[i];
                        res = min(res, j * q[i] + needed_first * p[i]);
                    }
                    return res;
                }()
            );
            
            if (total_cost > x) {
                return false;
            }
        }
        return total_cost <= x;
    };

    long long left = 0, right = 1e18;
    long long answer = 0;
    while (left <= right) {
        long long mid = (left + right) / 2;
        if (is_possible(mid)) {
            answer = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    cout << answer << '\n';
    return 0;
}
