ll f(ll a, ll b, ll p, ll q, ll x) {
    ll dp[10010] = {1e10};
    dp[0] = 0;
    for (int i = 1; i <= 10009; i++) {
        if (i - a >= 0) dp[i] = min(dp[i], dp[i - a] + p);
        if (i - b >= 0) dp[i] = min(dp[i], dp[i - b] + q);
    }
    for (int i = 10009; i >= 0; i--) dp[10009 - i] = min(dp[10009 - i], dp[i]);
    return dp[x];
}

    int n, x;
