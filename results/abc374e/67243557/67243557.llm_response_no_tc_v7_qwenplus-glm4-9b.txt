
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define rep(i, n) for (int i = 0; i < n; i++)

ll f(int a, int b, int p, int q, ll x) {
    vector<ll> dp(10010, 1e10);
    dp[0] = 0;
    for (int i = 1; i <= 10009; i++) {
        if (i >= a) dp[i] = min(dp[i], dp[i - a] + p);
        if (i >= b) dp[i] = min(dp[i], dp[i - b] + q);
    }
    dp[10010] = min(dp[10010], dp[10009]);
    for (int i = 10000; i >= 0; i--) {
        dp[10010] = min(dp[10010], dp[i]);
    }
    return dp[10010];
}

int main() {
    int n, x;
    cin >> n >> x;
    vector<int> a(n), b(n), p(n), q(n);
    rep(i, n) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    ll low = 0, high = 1e10;
    while (high - low > 1) {
        ll mid = (low + high) / 2;
        ll cost = 0;
        rep(i, n) {
            cost += f(a[i], b[i], p[i], q[i], mid);
        }
        if (cost <= x) low = mid;
        else high = mid;
    }
    cout << low << endl;
    return 0;
}
```

### Explanation:
1. **dp Array Initialization**: The dp array is initialized with a large value (1e10) because we are finding the minimum cost, and then the initial cost of 0 for buying 0 items.
2. **DP Calculation**: The dp array is updated based on the cost of buying `a[i]` or `b[i]` items, whichever is cheaper. This ensures that we always use the minimum cost for any quantity up to 10009.
3. **Binary Search**: The binary search is performed to find the maximum number of items that can be bought with the total cost not exceeding `x`.
4. **Cost Calculation**: For each midpoint in the binary search, the cost of buying items is calculated using the `f` function, and this cost is compared with `x` to decide whether to move the lower or upper bound of the binary search.
