    vector<ll> dp(x + 1, LLONG_MAX);

    for (ll i = 1; i <= x; ++i) {

        ll full_patterns = x / pattern_size;
        ll remainder = x % pattern_size;
        return full_patterns * pattern_cost + dp[remainder];
    }
    
    return dp[x];
}
