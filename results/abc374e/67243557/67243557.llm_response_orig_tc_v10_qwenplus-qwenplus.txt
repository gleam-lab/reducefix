#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<(n);i++)
using ll = long long;

// 配るDPで最小コストを計算
ll compute_min_cost(ll a, ll b, ll p, ll q, ll target) {
    // 周期の大きさ
    ll period = lcm(a, b);
    
    // DP配列 (最大周期分まで)
    vector<ll> dp(period + 1, 1e18);
    dp[0] = 0;

    // DP更新
    for (ll i = 0; i <= period; ++i) {
        if (i + a <= period) dp[i + a] = min(dp[i + a], dp[i] + p);
        if (i + b <= period) dp[i + b] = min(dp[i + b], dp[i] + q);
    }

    // 最小値を前から押し出し
    for (ll i = 1; i <= period; ++i) {
        dp[i] = min(dp[i], dp[i - 1]);
    }

    // 周期外の部分と周期内の部分を分ける
    ll full_periods = target / period;
    ll remainder = target % period;

    return full_periods * dp[period] + dp[remainder];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    ll x;
    cin >> n >> x;

    vector<ll> a(n), b(n), p(n), q(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    // 二分探索
    ll left = 0, right = 1e18;
    while (right - left > 1) {
        ll mid = (left + right) / 2;
        ll total_cost = 0;

        for (int i = 0; i < n; ++i) {
            total_cost += compute_min_cost(a[i], b[i], p[i], q[i], mid);
            if (total_cost > x) break;
        }

        if (total_cost <= x) {
            left = mid;
        } else {
            right = mid;
        }
    }

    cout << left << '\n';
    return 0;
}
