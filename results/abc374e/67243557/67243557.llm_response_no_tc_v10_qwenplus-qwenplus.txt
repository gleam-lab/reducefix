#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<(n);i++)
using ll = long long;

// 遅延評価DPで、x個作るための最小コストを求める関数
ll compute_min_cost(ll a, ll b, ll p, ll q, ll target) {
    // dp[i] = i個買うための最小コスト
    vector<ll> dp(target + 1, 1e18);
    dp[0] = 0;
    
    // 個数が大きい場合のために、ループ回数を抑える工夫が必要
    for (ll i = 0; i <= target; ++i) {
        if (i + a <= target) dp[i + a] = min(dp[i + a], dp[i] + p);
        if (i + b <= target) dp[i + b] = min(dp[i + b], dp[i] + q);
    }
    
    return dp[target];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    ll x;
    cin >> n >> x;
    
    vector<ll> a(n), p(n), b(n), q(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    // 二分探索：最大何個まで作れるか
    ll left = 0;
    ll right = 1e9; // 問題の制約に応じた上限
    
    while (left < right) {
        ll mid = (left + right + 1) / 2;
        ll total_cost = 0;
        
        for (int i = 0; i < n; ++i) {
            // 各工場でmid個生産するための最小コスト
            ll cost = compute_min_cost(a[i], b[i], p[i], q[i], mid);
            total_cost += cost;
            
            // オーバーフロー対策 or 上限超過
            if (total_cost > x) break;
        }
        
        if (total_cost <= x) left = mid;
        else right = mid - 1;
    }
    
    cout << left << '\n';
    
    return 0;
}
