#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<n;i++)
using ll = long long;
using Graph = vector<vector<int>>;

ll f(vector<int>& a, vector<int>& b, vector<int>& p, vector<int>& q, ll x){
  ll ans = 0;
  vector<ll> dp(x+1, 1e10);
  dp[0] = 0; // i個買う時の最小値
  rep(i,a.size()){
    if(a[i] <= x)dp[a[i]] = min(dp[a[i]], p[i]);
    if(b[i] <= x)dp[b[i]] = min(dp[b[i]], q[i]);
  }
  
  rep(i,x){
    if(dp[i] != 1e10){
      if(i + a[i] <= x)dp[i + a[i]] = min(dp[i + a[i]], dp[i] + p[i]);
      if(i + b[i] <= x)dp[i + b[i]] = min(dp[i + b[i]], dp[i] + q[i]);
    }
  }

  return dp[x];
}

int main() {
  int n, x;
  cin >> n >> x;
  vector<int> a(n), b(n), p(n), q(n);
  rep(i,n)cin >> a[i] >> p[i] >> b[i] >> q[i];
  ll now; // wの候補
  ll l = 0, r = 10000000000;
  while(r - l > 1){
    now = (r + l)/2;
    ll tmp = 0;
    rep(i,n){
      tmp += f(a, b, p, q, now);
    }
    if(tmp <= x)l = now;
    else r = now;
  }
  cout << l;
	return 0;
}
