#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<n;i++)
using ll = long long;
using Graph = vector<vector<int>>;

ll f(vector<ll>& a, vector<ll>& b, vector<ll>& p, vector<ll>& q, ll x){
  ll n = a.size();
  ll ans = 0;
  vector<ll> dp(x+1, 1e18); // Initialize with a large value
  dp[0] = 0; // Base case: cost of making 0 items is 0
  
  rep(i, n){
    for(int j = x; j >= 0; --j){
      if(j - a[i] >= 0) dp[j] = min(dp[j], dp[j-a[i]] + p[i]);
      if(j - b[i] >= 0) dp[j] = min(dp[j], dp[j-b[i]] + q[i]);
    }
  }

  for(int i = 0; i <= x; ++i){
    if(dp[i] != 1e18) ans = max(ans, i);
  }
  
  return ans;
}

int main() {
  int n, x;
  cin >> n >> x;
  vector<ll> a(n), b(n), p(n), q(n);
  rep(i,n)cin >> a[i] >> p[i] >> b[i] >> q[i];
  ll now; // wの候補
  ll l = 0, r = 10000000000LL;
  while(r - l > 1){
    now = (r + l)/2;
    ll tmp = 0;
    rep(i,n){
      tmp += f(a, b, p, q, now);
    }
    if(tmp <= x)l = now;
    else r = now;
  }
  cout << l;
	return 0;
}
