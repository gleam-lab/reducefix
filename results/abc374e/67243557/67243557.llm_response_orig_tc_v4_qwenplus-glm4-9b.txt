#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define INF 1e18
using ll = long long;

ll f(ll a, ll b, ll p, ll q, ll x){
  ll z = a*b;
  ll ans = INF;
  vector<ll> dp(x+1, INF);
  dp[0] = 0; // i氓聙聥猫虏路茫聛聠忙聶聜茫聛庐忙聹聙氓掳聫氓聙陇
  rep(i, x+1){
    if(i - a >= 0)dp[i] = min(dp[i], dp[i-a] + p);
    if(i - b >= 0)dp[i] = min(dp[i], dp[i-b] + q);
    if(dp[i] <= x)ans = min(ans, dp[i]);
  }
  return ans;
}

int main() {
  int n, x;
  cin >> n >> x;
  vector<int> a(n), b(n), p(n), q(n);
  rep(i,n)cin >> a[i] >> p[i] >> b[i] >> q[i];
  ll now; // w茫聛庐氓聙聶猫拢聹
  ll l = 0, r = x;
  while(r - l > 1){
    now = (r + l) / 2;
    ll tmp = f(a[0], b[0], p[0], q[0], now);
    rep(i, n)tmp += f(a[i], b[i], p[i], q[i], now - tmp);
    if(tmp <= x)l = now;
    else r = now;
  }
  cout << l;
	return 0;
}
