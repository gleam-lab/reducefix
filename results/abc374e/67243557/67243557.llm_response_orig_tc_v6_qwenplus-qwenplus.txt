#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<n;i++)
using ll = long long;

ll f(ll a, ll b, ll p, ll q, ll x) {
    // dp[i] = minimum cost to buy i items
    vector<ll> dp(2 * max(a, b) + 10, 1e18);
    dp[0] = 0;
    for (int i = 1; i < (int)dp.size(); ++i) {
        if (i >= a) dp[i] = min(dp[i], dp[i - a] + p);
        if (i >= b) dp[i] = min(dp[i], dp[i - b] + q);
    }

    // Find the cycle
    ll base = 0;
    ll period_len = 0;
    ll period_cost = 0;
    for (int i = 2; i < (int)dp.size(); ++i) {
        if (dp[i - 1] + dp[1] == dp[i]) {
            base = i - 1;
            period_len = i - base;
            period_cost = dp[i] - dp[base];
            break;
        }
    }

    if (period_len == 0) {
        // No cycle found, just return dp[x]
        return dp[x];
    } else {
        // Use cycle optimization
        ll full_cycles = (x - base) / period_len;
        ll remainder = (x - base) % period_len;
        return dp[base] + full_cycles * period_cost + dp[remainder];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    ll x;
    cin >> n >> x;
    vector<int> a(n), b(n), p(n), q(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    // Binary search for maximum number of snacks
    ll low = 0;
    ll high = 1e18;
    ll answer = 0;
    
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll total_cost = 0;
        
        for (int i = 0; i < n; ++i) {
            total_cost += f(a[i], b[i], p[i], q[i], mid);
            if (total_cost > x) break;
        }
        
        if (total_cost <= x) {
            answer = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    cout << answer << endl;
    return 0;
}
