#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<n;i++)
using ll = long long;

ll calculate_cost(ll a, ll b, ll p, ll q, ll cakes) {
    // Calculate minimum cost to get 'cakes' using a (cost p) and b (cost q)
    ll min_cost = LLONG_MAX;
    
    // Maximum number of a packages we might need
    ll max_a = cakes / a + 2;
    for (ll use_a = 0; use_a <= max_a; use_a++) {
        ll remaining = max(0LL, cakes - use_a * a);
        ll use_b = (remaining + b - 1) / b; // ceiling division
        ll total_cost = use_a * p + use_b * q;
        if (total_cost < min_cost) {
            min_cost = total_cost;
        }
        // Early exit if we can't improve further
        if (remaining <= 0) break;
    }
    return min_cost;
}

int main() {

    cin >> n >> x;
    vector<int> a(n), b(n), p(n), q(n);
    rep(i,n) cin >> a[i] >> p[i] >> b[i] >> q[i];
    
    ll left = 0, right = 1e18, answer = 0;
    while (left <= right) {
        ll mid = left + (right - left) / 2;
        ll total_cost = 0;
        
        for (int i = 0; i < n; i++) {
            total_cost += calculate_cost(a[i], b[i], p[i], q[i], mid);
            if (total_cost > x) break; // Early exit if already over budget
        }
        
        if (total_cost <= x) {
