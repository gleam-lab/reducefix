#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<n;i++)
using ll = long long;

// Function to compute the minimum cost to buy at least x candies of a single type using two packs
ll min_cost(ll a, ll b, ll p, ll q, ll x) {
    // dp[i] = minimum cost to get i candies
    vector<ll> dp(2 * (a + b), 1e18);
    dp[0] = 0;
    
    for (ll i = 0; i < 2 * (a + b); ++i) {
        if (i + a < 2 * (a + b)) dp[i + a] = min(dp[i + a], dp[i] + p);
        if (i + b < 2 * (a + b)) dp[i + b] = min(dp[i + b], dp[i] + q);
    }

    // Try all values >= x and find the minimum cost
    ll res = 1e18;
    for (ll i = x; i < 2 * (a + b); ++i) {
        res = min(res, dp[i]);
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    ll x;
    cin >> n >> x;
    
    vector<ll> a(n), b(n), p(n), q(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }
    
    // Binary search on the answer (number of candies we can buy)
    ll low = 0, high = 1e18;
    ll ans = 0;
    
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll total_cost = 0;
        
        for (int i = 0; i < n; ++i) {
            total_cost += min_cost(a[i], b[i], p[i], q[i], mid);
            if (total_cost > x) break;
        }
        
        if (total_cost <= x) {
            ans = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    cout << ans << '\n';
    return 0;
}
