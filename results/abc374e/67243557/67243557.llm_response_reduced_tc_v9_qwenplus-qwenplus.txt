#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<(n);i++)
using ll = long long;

ll compute_min_cost(ll a, ll b, ll p, ll q, ll required) {
    // We want to buy at least `required` number of items with minimum cost
    // Using DP for small cycle and math optimization for large cycles

    ll period = a * b; // The periodic cycle length
    vector<ll> dp(period + 1, 1e18);
    dp[0] = 0;

    // Classical unbounded knapsack up to one full period
    for (ll i = 0; i <= period; ++i) {
        if (i + a <= period) dp[i + a] = min(dp[i + a], dp[i] + p);
        if (i + b <= period) dp[i + b] = min(dp[i + b], dp[i] + q);
    }

    // Post-process to ensure we have the best value for all numbers up to period
    for (ll i = 1; i <= period; ++i) {
        dp[i] = min(dp[i], dp[i - 1]);
    }

    // For large values, use periodic properties
    ll full_cycles = required / period;
    ll remainder = required % period;

    return full_cycles * dp[period] + dp[remainder];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    ll x;
    cin >> n >> x;

    vector<ll> a(n), b(n), p(n), q(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    // Binary search on how many cookies we can bake
    ll low = 0, high = 1e18;
    ll answer = 0;

    while (low < high) {
        ll mid = (low + high + 1) / 2;
        ll total_cost = 0;

        for (int i = 0; i < n; ++i) {
            total_cost += compute_min_cost(a[i], b[i], p[i], q[i], mid);
            if (total_cost > x) break;
        }

        if (total_cost <= x) {
            answer = mid;
            low = mid;
        } else {
            high = mid - 1;
        }
    }

    cout << answer << endl;
    return 0;
}
