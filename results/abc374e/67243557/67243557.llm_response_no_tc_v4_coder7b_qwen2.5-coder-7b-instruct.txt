#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) (x).begin(),(x).end()
#define rall(x) (x).rbegin(),(x).rend()

typedef long long ll;

struct Machine{
    int type;
    int rate;
    int cost;
};

bool compare(Machine &a, Machine &b){
    return a.rate * b.cost < a.cost * b.rate;
}

int main(){
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n, x; cin >> n >> x;
    vector<Machine> s(n), t(n);
    rep(i, n){
        cin >> s[i].rate >> s[i].cost;
        s[i].type = 1;
    }
    rep(i, n){
        cin >> t[i].rate >> t[i].cost;
        t[i].type = 2;
    }

    sort(all(s));
    sort(all(t));

    vector<ll> dp(x+1, INT_MAX);
    dp[0] = 0;

    auto update = [&](vector<Machine> &machines) -> void {
        vector<ll> ndp(x+1, INT_MAX);
        ndp[0] = 0;
        for(auto &m : machines){
            for(int j = x; j >= m.cost; --j){
                ndp[j] = min(ndp[j], ndp[j-m.cost] + m.type);
            }
        }
        dp.swap(ndp);
    };

    update(s);
    update(t);

    int ans = 0;
    for(int i = 0; i <= x; ++i){
        if(dp[i] != INT_MAX) ans = max(ans, i);
    }

    cout << ans << endl;
    return 0;
}
