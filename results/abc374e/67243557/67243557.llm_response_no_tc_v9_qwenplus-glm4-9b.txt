
```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < n; i++)
#define INF 1e18

using ll = long long;

ll f(ll a, ll b, ll p, ll q, ll x) {
    ll dp[x + 1];
    memset(dp, INF, sizeof(dp));
    dp[0] = 0;

    for (ll j = 0; j <= x; j++) {
        if (j >= a) {
            dp[j] = min(dp[j], dp[j - a] + p);
        }
        if (j >= b) {
            dp[j] = min(dp[j], dp[j - b] + q);
        }
    }

    return dp[x];
}

int main() {
    ll n, x;
    cin >> n >> x;
    vector<ll> a(n), p(n), b(n), q(n);
    rep(i, n) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    ll l = 0, r = 1000000000;
    while (r - l > 1) {
        ll mid = (l + r) / 2;
        ll cost = 0;
        for (int i = 0; i < n; i++) {
            cost += f(a[i], b[i], p[i], q[i], mid);
        }
        if (cost <= x) {
            l = mid;
        } else {
            r = mid;
        }
    }

    cout << l << '\n';
    return 0;
}
```

### Explanation:
1. **Initialization and Memory Allocation**: I used `memset` to initialize the `dp` array with `INF` to ensure that all values are initially considered unreachable. This prevents any undefined behavior when accessing `dp[i]` where `i` might be zero.

2. **Dynamic Programming Calculation**: The `dp` array is calculated iteratively. For each possible `j` (number of items), it checks if buying more items with type `a` or `b` is cheaper than the previously computed cost.

3. **Binary Search**: The binary search is used to find the maximum number of items that can be bought within the given budget `x`. The function `f` computes the total cost for a given number of items, and the binary search narrows down to the highest feasible value.

4. **Edge Cases Handling**: The upper limit for the binary search is set to `1000000000` as per the problem statement. Adjustments can be made if a smaller range is known to be sufficient.
