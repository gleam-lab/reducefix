#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(ll i=0;i<n;i++)
using ll = long long;

ll f(ll a, ll b, ll p, ll q, ll x) {
    ll k = x / (a * b);
    ll z = x % (a * b);
    ll ans = k * p * b + k * q * a;

    // Calculate the additional cost for the remaining amount
    if (z >= a) {
        ans += min(p, q) * a;
        z -= a;
    }
    if (z >= b) {
        ans += min(p, q) * b;
        z -= b;
    }

    return ans;
}

int main() {
    ll n, x;
    cin >> n >> x;
    vector<pair<ll, ll>> a(n), p(n);
    rep(i, n) {
        cin >> a[i].first >> p[i].second >> a[i].second >> p[i].first;
    }

    // Sort the items by cost in descending order
    sort(p.begin(), p.end(), [](const pair<ll, pair<ll, ll>>& a, const pair<ll, pair<ll, ll>>& b) {
        return a.first > b.first;
    });

    ll low = 0, high = 1e18;
    while (low < high) {
        ll mid = (low + high + 1) / 2;
        ll total_cost = 0;
        ll count = 0;
        for (const auto& item : p) {
            ll max_items = mid / (item.second * item.first);
            total_cost += max_items * item.second * item.first;
            count += max_items;
            if (count * item.first >= x) break;
        }

        if (total_cost <= x) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }

    cout << low;
    return 0;
}
