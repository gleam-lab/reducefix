#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ll = long long;
using vi = vector<int>;

ll f(int a, int b, int p, int q, ll x) {
    ll z = x;
    ll ans = 0;
    vector<ll> dp(z + 1, 1e18); // Use 1e18 to represent infinity
    dp[0] = 0;

    for (ll j = 0; j < z; ++j) {
        if (j - a >= 0) dp[j] = min(dp[j], dp[j - a] + p);
        if (j - b >= 0) dp[j] = min(dp[j], dp[j - b] + q);
    }

    for (ll j = 0; j <= z; ++j) {
        if (dp[j] >= dp[z]) break; // Since we want minimum cost, we continue until dp[j] becomes greater than dp[z]
        ans += dp[j];
        if (j == z) break;
    }

    return ans;
}

int main() {
    int n, x;
    cin >> n >> x;
    vector<vi> items(n);
    for (int i = 0; i < n; ++i) {
        cin >> items[i][0] >> items[i][1] >> items[i][2] >> items[i][3];
    }

    ll l = 0, r = 1e18;
    while (l < r) {
        ll mid = l + (r - l) / 2;
        ll tmp = 0;
        for (const auto& item : items) {
            tmp += f(item[0], item[1], item[2], item[3], mid);
        }
        if (tmp <= x) l = mid + 1;
        else r = mid;
    }

    cout << l - 1 << endl; // Subtract 1 because r is updated to mid + 1
    return 0;
}
