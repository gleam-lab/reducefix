#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(long long i=0;i<(n);i++)
using ll = long long;

ll compute_min_cost(ll a, ll b, ll p, ll q, ll required) {
    // We want to buy at least 'required' number of items with minimum cost
    // Use dynamic programming up to a reasonable limit

    const ll MAX = 20000; // Increased buffer size to safely handle wraparound cases
    vector<ll> dp(MAX + 1, 1e18);
    dp[0] = 0;

    // Build DP table
    for (ll i = 1; i <= MAX; ++i) {
        if (i >= a) dp[i] = min(dp[i], dp[i - a] + p);
        if (i >= b) dp[i] = min(dp[i], dp[i - b] + q);
    }

    // Find the minimum cost to buy at least 'required' items
    ll result = 1e18;
    for (ll i = required; i <= MAX; ++i) {
        result = min(result, dp[i]);
    }
    return result;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    ll x;
    cin >> n >> x;
    
    vector<ll> a(n), b(n), p(n), q(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    // Binary search on the maximum number of items we can buy
    ll low = 0, high = 1e6; // Adjusted high based on problem constraints
    ll answer = 0;

    while (low <= high) {
        ll mid = (low + high) / 2;
        ll total_cost = 0;

        for (int i = 0; i < n; ++i) {
            ll cost = compute_min_cost(a[i], b[i], p[i], q[i], mid);
            total_cost += cost;
            if (total_cost > x) break; // Early exit if already over budget
        }

        if (total_cost <= x) {
            answer = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << answer << "\n";
    return 0;
}
