ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    ll best = LLONG_MAX;
    // Check both possible optimal directions
    for (int direction = 0; direction < 2; ++direction) {

                best = min(best, p * x + q * y);
            }

                best = min(best, p * x + q * y);

}
