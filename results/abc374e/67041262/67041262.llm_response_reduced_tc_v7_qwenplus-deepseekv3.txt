#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    ll best = LLONG_MAX;
    // Check all possible x values near the optimal points
    for (ll x = 0; x <= W / a + 1; x++) {
        ll rem = W - a * x;
        if (rem < 0) rem = 0;
        ll y = (rem + b - 1) / b;  // ceil(rem/b)
        best = min(best, p * x + q * y);
        
        // Also check the corresponding y points
        if (y > 0) {
            best = min(best, p * x + q * (y - 1));
        }
    }
    // Similarly check y values
    for (ll y = 0; y <= W / b + 1; y++) {
        ll rem = W - b * y;
        if (rem < 0) rem = 0;
        ll x = (rem + a - 1) / a;  // ceil(rem/a)
        best = min(best, p * x + q * y);
        
        if (x > 0) {
            best = min(best, p * (x - 1) + q * y);
        }
    }
    return best;
}

bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; i++) {
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]);
        sum += c;
        if (sum > X) return false;
    }
    return sum <= X;
}
