#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    ll cost_s = LLONG_MAX, cost_t = LLONG_MAX;
    for (ll x = 0; a * x <= W; ++x) {
        ll rem = W - a * x;
        ll y = (rem + b - 1) / b;
        cost_s = min(cost_s, p * x + q * y);
    }
    for (ll y = 0; b * y <= W; ++y) {
        ll rem = W - b * y;
        ll x = (rem + a - 1) / a;
        cost_t = min(cost_t, p * x + q * y);
    }
    return min(cost_s, cost_t);
}

bool can(ll W, int N, ll X, const vector<ll>& A, const vector<ll>& P, const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; ++i) {
        sum += min_cost(W, A[i], P[i], B[i], Q[i]);
        if (sum > X) return false;
    }
    return true;
}

        if (can(mid, N, X, A, P, B, Q)) lo = mid;
