#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

ll min_cost(ll weight, ll cost_per_unit, ll num_units, ll max_unit_weight, ll cost_per_excess_unit) {
    ll max_units = weight / num_units;
    ll min_cost = cost_per_unit * max_units + cost_per_excess_unit * (weight - max_units * num_units);
    ll min_cost_with_one_less_unit = LLONG_MAX;
    if (weight - num_units > 0) {
        min_cost_with_one_less_unit = cost_per_unit * (max_units - 1) + cost_per_excess_unit * (weight - (max_units - 1) * num_units);
    }
    return min(min_cost, min_cost_with_one_less_unit);
}

bool can(ll total_weight, int num_items, ll budget,
         const vector<ll>& item_weights, const vector<ll>& item_costs,
         const vector<ll>& max_unit_weights, const vector<ll>& max_unit_costs) {
    ll sum = 0;
    for (int i = 0; i < num_items; ++i) {
        ll cost = min_cost(total_weight, item_costs[i], item_weights[i], max_unit_weights[i], max_unit_costs[i]);
        sum += cost;
        if (sum > budget) return false;
    }
    return sum <= budget;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int num_items;
    ll budget;
    cin >> num_items >> budget;
    vector<ll> item_weights(num_items), item_costs(num_items),
                max_unit_weights(num_items), max_unit_costs(num_items);
    for (int i = 0; i < num_items; ++i) {
        cin >> item_weights[i] >> item_costs[i] >> max_unit_weights[i] >> max_unit_costs[i];
    }

    ll low = 0, high = 1e18;
    while (high - low > 1) {
        ll mid = (low + high) / 2;
        if (can(mid, num_items, budget, item_weights, item_costs, max_unit_weights, max_unit_costs)) {
            low = mid;
        } else {
            high = mid;
        }
    }
    cout << low << '\n';
    return 0;
}
