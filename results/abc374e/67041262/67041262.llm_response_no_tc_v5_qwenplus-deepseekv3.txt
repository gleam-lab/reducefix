#include <bits/stdc++.h>

ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    ll best = LLONG_MAX;
    
    // Try both strategies and their nearby values to find optimal
    // Strategy 1: prioritize x (a units)
    ll x_max = W / a;
    for (ll x = max(0LL, x_max - 100); x <= x_max + 100; x++) {
        if (x < 0) continue;
        ll rem = W - a * x;
        ll y = (rem <= 0 ? 0 : (rem + b - 1) / b);
        best = min(best, p * x + q * y);
    }
    
    // Strategy 2: prioritize y (b units)
    ll y_max = W / b;
    for (ll y = max(0LL, y_max - 100); y <= y_max + 100; y++) {

        ll rem = W - b * y;
        ll x = (rem <= 0 ? 0 : (rem + a - 1) / a);
        best = min(best, p * x + q * y);
    }
    
    return best;
}

bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    if (W == 0) return true;
    ll sum = 0;
    for (int i = 0; i < N; i++) {
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]);

}
