#include <bits/stdc++.h>

// Function to calculate the cost of processing with given parameters
ll min_cost(ll W, ll a, ll p, ll b, ll q){
    // Determine the optimal allocation between machines S and T
    ll best = LLONG_MAX;
    if (p * b < q * a) {
        // Prioritize using machine S
        ll x0 = W / a;            // Maximum units of S we can use
        for (ll dx = 0; dx <= b && x0 >= dx; dx++){
            ll x = x0 - dx;
            ll rem = W - a * x;
            ll y = (rem <= 0 ? 0 : (rem + b - 1) / b); // Calculate units of T needed
            best = min(best, p * x + q * y);
        }
    } else {
        // Prioritize using machine T
        ll y0 = W / b;
        for (ll dy = 0; dy <= a && y0 >= dy; dy++){
            ll y = y0 - dy;
            ll rem = W - b * y;
            ll x = (rem <= 0 ? 0 : (rem + a - 1) / a);
            best = min(best, p * x + q * y);
        }
    }
    return best;
}

// Function to check if a given production capacity is feasible within the budget
bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q){
    ll sum = 0;
    for (int i = 0; i < N; i++){
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]); // Calculate cost for current process
        sum += c;
        if (sum > X) return false; // If total cost exceeds budget, return false
    }
    return true; // Feasible if total cost does not exceed budget
}

        if (can(mid, N, X, A, P, B, Q)) lo = mid;
