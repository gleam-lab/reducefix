### Approaches
1. **Understanding the Problem**: The problem involves finding the maximum possible `W` such that the total cost of covering `W` units across all `N` items does not exceed `X`. Each item can be covered using two different methods with different costs and coverage rates.
2. **Binary Search**: We use binary search to efficiently determine the maximum `W`.
3. **Cost Calculation**: For each item, compute the minimum cost to cover `W` units by considering all possible combinations of the two methods (`x` and `y`). The range of possible values for `x` and `y` should be sufficiently large to ensure the optimal solution is found.
4. **Optimal Combination**: For each possible `x` (or `y`), compute the corresponding `y` (or `x`) required to cover the remaining units and keep track of the minimum cost.

### Solution Code

#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    ll best = LLONG_MAX;
    // Check both strategies and a reasonable range around the optimal point
    // Strategy 1: Prefer x (a's method)
    ll max_x = W / a + 2;
    for (ll x = 0; x <= max_x; ++x) {
        ll rem = W - a * x;
        ll y = max(0LL, rem <= 0 ? 0 : (rem + b - 1) / b);
        best = min(best, p * x + q * y);
        if (best == 0) break; // Early exit if cost is zero
    }
    // Strategy 2: Prefer y (b's method)
    ll max_y = W / b + 2;
    for (ll y = 0; y <= max_y; ++y) {
        ll rem = W - b * y;
        ll x = max(0LL, rem <= 0 ? 0 : (rem + a - 1) / a);
        best = min(best, p * x + q * y);
        if (best == 0) break; // Early exit if cost is zero
    }
    return best;
}

bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; ++i) {
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]);
        sum += c;
        if (sum > X) return false;
    }
    return sum <= X;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    ll lo = 0, hi = 1e18;
    while (lo < hi) {
        ll mid = lo + (hi - lo + 1) / 2;
        if (can(mid, N, X, A, P, B, Q)) {
            lo = mid;
        } else {
            hi = mid - 1;
        }
    }
    cout << lo << "\n";
    return 0;
}

### Explanation
1. **Binary Search Setup**: The binary search is initialized with `lo` as 0 and `hi` as a sufficiently large value (1e18). The loop continues until `lo` and `hi` converge.
2. **Cost Calculation**: For each mid value `W` during the binary search, the `can` function checks if the total cost to cover `W` units across all items is within the budget `X`.
3. **Optimal Cost for Each Item**: The `min_cost` function computes the minimum cost for each item by considering both strategies (preferring `x` or `y`) and checking a reasonable range around the optimal values to ensure the true minimum cost is found.
4. **Early Termination**: If the cost becomes zero, the loop terminates early to save computation time.
5. **Output**: The maximum `W` for which the total cost is within the budget is printed.

This approach ensures that all possible optimal combinations are considered, leading to the correct solution for all test cases.
