#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll min_cost(ll W, ll a, ll b, ll p, ll q) {
    // S 茫聛庐忙聳鹿茫聛聦氓聤鹿莽聨聡猫聣炉茫聛聭茫聜聦茫聛掳 x 茫聜聮氓陇職莽聰篓茫聛聴茫聙聛茫聛聺茫聛聠茫聛搂茫聛陋茫聛聭茫聜聦茫聛掳 y 茫聜聮氓陇職莽聰篓
    ll best = LLONG_MAX;
    // x茫聛庐莽炉聞氓聸虏茫聜聮猫篓聢莽庐聴
    ll x0 = ceil(W / static_cast<double>(a));
    for (ll x = x0; x <= (W + a - 1) / a; ++x) {
        ll rem = W - a * x;
        ll y = (rem < b ? 0 : rem / b + 1);
        best = min(best, p * x + q * y);
    }
    // y茫聛庐莽炉聞氓聸虏茫聜聮猫篓聢莽庐聴
    ll y0 = ceil(W / static_cast<double>(b));
    for (ll y = y0; y <= (W + b - 1) / b; ++y) {
        ll rem = W - b * y;
        ll x = (rem < a ? 0 : rem / a + 1);
        best = min(best, p * x + q * y);
    }
    return best;
}

bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; ++i) {
        ll c = min_cost(W, A[i], B[i], P[i], Q[i]);
        sum += c;
        if (sum > X) return false;
    }
    return sum <= X;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    ll lo = 0, hi = 1000000000000LL; // 盲赂聤茅聶聬茫聛炉氓聧聛氓聢聠氓陇搂茫聛聧茫聛聫
    while (hi - lo > 1) {
        ll mid = (lo + hi) / 2;
        if (can(mid, N, X, A, P, B, Q)) lo = mid;
        else hi = mid;
    }
    cout << lo << "\n";
    return 0;
}
