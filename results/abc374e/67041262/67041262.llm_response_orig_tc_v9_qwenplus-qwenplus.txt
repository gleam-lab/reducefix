#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    // Try both strategies: using more x or more y
    ll best = LLONG_MAX;

    // Strategy 1: Use mostly x (S)
    ll x0 = (W + a - 1) / a; // ceil division
    for (ll dx = max(0LL, x0 - 2); dx <= x0 + 2 && dx <= W / a + 2; dx++) {
        ll rem = max(0LL, W - a * dx);
        ll y = (rem + b - 1) / b;
        best = min(best, p * dx + q * y);
    }

    // Strategy 2: Use mostly y (T)
    ll y0 = (W + b - 1) / b; // ceil division
    for (ll dy = max(0LL, y0 - 2); dy <= y0 + 2 && dy <= W / b + 2; dy++) {
        ll rem = max(0LL, W - b * dy);
        ll x = (rem + a - 1) / a;
        best = min(best, p * x + q * dy);
    }

    return best;
}

bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; i++) {
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]);
        sum += c;
        if (sum > X) return false;
    }
    return sum <= X;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    ll lo = 0, hi = 1000000000000LL; // Upper bound is sufficiently large
    while (hi - lo > 1) {
        ll mid = (lo + hi) / 2;
        if (can(mid, N, X, A, P, B, Q)) lo = mid;
        else hi = mid;
    }
    cout << lo << "\n";

    return 0;
}
