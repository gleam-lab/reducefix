#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Calculates the minimum cost to get at least W units of a resource,
// using two types of packages:
// - Type 1: 'a' units per package, cost 'p' per package
// - Type 2: 'b' units per package, cost 'q' per package
ll min_cost(ll W, ll a, ll p, ll b, ll q) {
    // Try both strategies: favoring type 1 (S) or type 2 (T)

    ll best = LLONG_MAX;

    // Strategy 1: Favor type 1 packages
    {
        ll max_x = (W + a - 1) / a;  // Upper bound for x
        // Search in a window around the optimal x
        for (ll dx = -100; dx <= 100; dx++) {
            ll x = max_x + dx;
            if (x < 0) continue;
            ll rem = max(0LL, W - a * x);
            ll y = (rem + b - 1) / b;
            best = min(best, p * x + q * y);
        }
    }

    // Strategy 2: Favor type 2 packages
    {
        ll max_y = (W + b - 1) / b;  // Upper bound for y
        // Search in a window around the optimal y
        for (ll dy = -100; dy <= 100; dy++) {
            ll y = max_y + dy;
            if (y < 0) continue;
            ll rem = max(0LL, W - b * y);
            ll x = (rem + a - 1) / a;
            best = min(best, p * x + q * y);
        }
    }

    return best;
}

// Checks if it's possible to achieve at least W units with total cost <= X
bool can(ll W, int N, ll X,
         const vector<ll>& A, const vector<ll>& P,
         const vector<ll>& B, const vector<ll>& Q) {
    ll sum = 0;
    for (int i = 0; i < N; ++i) {
        ll c = min_cost(W, A[i], P[i], B[i], Q[i]);
        sum += c;
        if (sum > X) return false;
    }
    return sum <= X;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // Binary search on the answer
    ll lo = 0, hi = 1e18; // Use a large upper bound for safety
    while (hi - lo > 1) {
        ll mid = (lo + hi) / 2;
        if (can(mid, N, X, A, P, B, Q)) {
            lo = mid;
        } else {
            hi = mid;
        }
    }
    cout << lo << "\n";
    
    return 0;
}
