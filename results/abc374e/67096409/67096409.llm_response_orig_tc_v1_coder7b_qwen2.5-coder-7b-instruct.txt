#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Product {
    int a, p, b, q;
};

bool operator<(const Product& lhs, const Product& rhs) {
    return lhs.a * lhs.b > rhs.a * rhs.b || (lhs.a * lhs.b == rhs.a * rhs.b && lhs.p * lhs.q > rhs.p * rhs.q);
}

vector<Product> products;
vector<vector<long long>> dp;

long long cost(int i, int n) {
    long long d = n / (products[i].a * products[i].b), rem = n % (products[i].a * products[i].b);
    return dp[i][rem] + d * products[i].p * products[i].b;
}

bool feasible(long long n) {
    long long total_cost = 0;
    for (int i = 0; i < products.size(); ++i) {
        total_cost += cost(i, n);
        if (total_cost > n) return false;
    }
    return true;
}

int binarySearch() {
    int left = 0, right = 1e9;
    while (left <= right) {
        long long mid = left + (right - left) / 2;
        if (feasible(mid)) left = mid + 1;
        else right = mid - 1;
    }
    return left - 1;
}

int main() {
    int N, X;
    cin >> N >> X;

    products.resize(N);
    dp.resize(N, vector<long long>(N));

    for (int i = 0; i < N; ++i) {
        cin >> products[i].a >> products[i].p >> products[i].b >> products[i].q;
        if (products[i].a * products[i].q < products[i].b * products[i].p) swap(products[i].a, products[i].b), swap(products[i].p, products[i].q);
        dp[i][0] = 0;
        for (int j = 1; j < products[i].a * products[i].b; ++j) {
            dp[i][j] = min(dp[i][max(0, j - products[i].a)] + products[i].p, dp[i][max(0, j - products[i].b)] + products[i].q);
        }
    }

    cout << binarySearch() << endl;

    return 0;
}
