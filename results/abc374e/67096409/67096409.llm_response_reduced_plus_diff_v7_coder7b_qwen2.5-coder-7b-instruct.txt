#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Item {
    int a, p, b, q;
};

long long dp[100][10000];

bool canFitItems(const vector<Item>& items, int X) {
    int N = items.size();
    for (int i = 0; i < N; ++i) {
        int a = items[i].a, p = items[i].p, b = items[i].b, q = items[i].q;
        if (a * q < b * p) swap(a, b), swap(p, q);
        dp[i][0] = 0;
        for (int j = 1; j <= a * b; ++j) {
            dp[i][j] = min(j / a * p + dp[i][j % a], j / b * q + dp[i][j % b]);
        }
    }

    int left = 0, right = X;
    while (left < right) {
        int mid = (left + right) / 2;
        bool fits = true;
        for (int i = 0; i < N; ++i) {
            int a = items[i].a, p = items[i].p, b = items[i].b, q = items[i].q;
            int d = mid / (a * b), rem = mid % (a * b);
            if (dp[i][rem] + d * a * p > X) {
                fits = false;
                break;
            }
        }
        if (fits) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    return left > 0;
}

int main() {
    int N, X;
    cin >> N >> X;

    vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        cin >> items[i].a >> items[i].p >> items[i].b >> items[i].q;
        if (items[i].a * items[i].q < items[i].b * items[i].p) {
            swap(items[i].a, items[i].b);
            swap(items[i].p, items[i].q);
        }
    }

    if (canFitItems(items, X)) {
        cout << left - 1 << endl;
    } else {
        cout << 0 << endl;
    }

    return 0;
}
