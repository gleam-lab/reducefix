#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Item {
    int a, p, b, q;
};

vector<Item> items;
vector<vector<long long>> dp;

long long cost(int i, int n) {
    long long d = n / (items[i].a * items[i].b);
    long long rem = n % (items[i].a * items[i].b);
    return dp[i][rem] + d * items[i].p * items[i].b;
}

bool feasible(long long n) {
    long long total_cost = 0;
    for (int i = 0; i < items.size(); ++i) {
        total_cost += cost(i, n);
        if (total_cost > X) return false;
    }
    return true;
}

int main() {
    int N, X;
    cin >> N >> X;

    items.resize(N);
    dp.resize(N, vector<long long>(10001, LLONG_MAX));
    
    for (int i = 0; i < N; ++i) {
        cin >> items[i].a >> items[i].p >> items[i].b >> items[i].q;
        if (items[i].a * items[i].q < items[i].b * items[i].p) swap(items[i].a, items[i].b), swap(items[i].p, items[i].q);
        dp[i][0] = 0;
        for (int j = 1; j <= items[i].a * items[i].b; ++j) {
            dp[i][j] = min(dp[i][max(0, j - items[i].a)] + items[i].p, dp[i][max(0, j - items[i].b)] + items[i].q);
        }
    }

    int low = 0, high = 1e9;
    while (high - low > 1) {
        int mid = (low + high) / 2;
        if (feasible(mid)) low = mid;
        else high = mid - 1;
    }

    if (low != high && feasible(low + 1)) low++;
    cout << low << endl;

    return 0;
}
