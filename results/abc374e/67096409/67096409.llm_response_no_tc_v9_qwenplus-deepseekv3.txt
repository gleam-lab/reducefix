#include <iostream>
#include <algorithm>
using namespace std;

typedef struct { int a, p, b, q; } p;
p A[100];
long long dp[100][10001]; // Increased size to handle larger remainders

int N, X;

long long cost(int i, int n) {
    long long d = n / (A[i].a * A[i].b);
    long long rem = n % (A[i].a * A[i].b);
    return dp[i][rem] + d * (A[i].p * A[i].b + A[i].q * A[i].a);
}

bool f(int n) {
    long long total_cost = 0;
    for (int i = 0; i < N; i++) {
        total_cost += cost(i, n);
        if (total_cost > X) return false;
    }
    return true;
}

int main() {
    scanf("%d %d", &N, &X);

    for (int i = 0; i < N; i++) {
        scanf("%d %d %d %d", &A[i].a, &A[i].p, &A[i].b, &A[i].q);
        if (A[i].a * A[i].q < A[i].b * A[i].p) {
            swap(A[i].a, A[i].b);
            swap(A[i].p, A[i].q);
        }
        
        int ab = A[i].a * A[i].b;
        dp[i][0] = 0;
        for (int j = 1; j <= ab; j++) {
            long long option1 = (j >= A[i].a) ? dp[i][j - A[i].a] + A[i].p : LLONG_MAX;
            long long option2 = (j >= A[i].b) ? dp[i][j - A[i].b] + A[i].q : LLONG_MAX;
            dp[i][j] = min(option1, option2);
        }
    }

    int lo = 0, hi = 1e9;
    int answer = 0;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        if (f(mid)) {
            answer = mid;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    printf("%d\n", answer);
    return 0;
}
