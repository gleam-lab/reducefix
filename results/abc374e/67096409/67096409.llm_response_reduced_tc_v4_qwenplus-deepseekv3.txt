typedef struct { int a, p, b, q; } p;
p A[100];
long long dp[100][10000];

int N;
long long X;

long long cost(int i, long long n) {
    long long ab = (long long)A[i].a * A[i].b;
    long long d = n / ab;
    long long rem = n % ab;
    return dp[i][rem] + d * A[i].p * A[i].b;
}

bool f(long long n) {
    long long total_cost = 0;
    for (int i = 0; i < N; ++i) {
        total_cost += cost(i, n);
        if (total_cost > X) return false;
    }
    return true;
}
