```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

struct Item {
    int a, p, b, q;
};

bool canFill(int capacity, const vector<Item>& items, int X) {
    for (const auto& item : items) {
        long long count = capacity / (item.a * item.b);
        long long remainder = capacity % (item.a * item.b);
        long long totalCost = count * item.p * item.b + dp[item.a][item.b][remainder];
        if (totalCost > X) return false;
    }
    return true;
}

int main() {
    int N, X;
    cin >> N >> X;

    vector<Item> items(N);
    for (auto& item : items) {
        cin >> item.a >> item.p >> item.b >> item.q;
        if (item.a * item.q < item.b * item.p) swap(item.a, item.b), swap(item.p, item.q);
    }

    // Initialize DP table
    vector<vector<vector<long long>>> dp(101, vector<vector<long long>>(10001, vector<long long>(10001, LLONG_MAX)));
    for (int i = 0; i <= 100; ++i) {
        dp[i][0][0] = 0;
    }

    for (int i = 1; i <= 100; ++i) {
        for (int rem = 1; rem < i * i; ++rem) {
            dp[i][rem][0] = min(dp[i][rem][0], dp[i][max(0, rem - i)][i] + i * i * i * i);
            dp[i][0][rem] = min(dp[i][0][rem], dp[i][rem][max(0, rem - i)] + i * i * i * i);
        }
    }

    int left = 0, right = INT_MAX;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (canFill(mid, items, X)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }

    cout << left << endl;
    return 0;
}
```

This solution addresses the issues by using dynamic programming to precompute the minimum cost for filling up a given amount of space with each type of item. The `canFill` function checks if it's possible to fill the given capacity without exceeding the budget. The binary search is used to find the maximum possible capacity that can be filled within the budget.
