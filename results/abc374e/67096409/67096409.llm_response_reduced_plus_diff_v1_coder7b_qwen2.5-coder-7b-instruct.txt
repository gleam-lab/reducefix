#include <iostream>
#include <algorithm>
using namespace std;

typedef struct{int a,p,b,q;} item;
item items[100];
long long dp[100][10000];

int N,X;
long long cost(int i, int n){
    long long d=n/(items[i].a*items[i].b),rem=n%(items[i].a*items[i].b);
    return dp[i][rem]+d*items[i].p*items[i].b;
}
bool f(int n){
    long long ct=0;
    for(int i=0;i<N;i++){
        ct+=cost(i,n);
        if(ct>X)return false;
    }
    return true;
}

int main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    scanf("%d %d", &N,&X);

    for(int i=0,a,b,p,q;i<N;i++){
        scanf("%d %d %d %d", &a,&p,&b,&q);
        if(a*q<b*p)swap(a,b),swap(p,q);
        items[i] = {a,p,b,q};
        dp[i][0]=0;
        for(int j=1;j<a*b;j++){
            dp[i][j] = min(dp[i][max(0,j-a)]+p,dp[i][max(0,j-b)]+q);
        }
    }

    int lo=0,hi=X+1;
    while(hi-lo>1){
        int mid = (hi+lo)>>1;
        if(f(mid))lo=mid;
        else hi=mid-1;
    }

    printf("%d\n",lo);

}
