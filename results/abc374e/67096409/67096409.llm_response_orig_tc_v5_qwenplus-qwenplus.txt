### Solution:

We should use a more standard binary search pattern:
- Start with `lo = 0`, `hi = some upper bound` (say `1e9`)
- In each step, compute `mid = (lo + hi) / 2`
- If `f(mid)` is true (i.e., we can afford `mid` units), move `lo = mid`
- Else, move `hi = mid - 1`
- After the loop ends, `lo` will be the maximum affordable number of units

This avoids the extra check `if(lo != hi && f(lo+1)) lo++` which was likely causing the wrong answer.

#include <iostream>
#include <algorithm>
using namespace std;

typedef struct{int a,p,b,q;} p;
p A[100];
long long dp[100][10000];

int N,X;
long long cost(int i, int n){
    long long d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
    return dp[i][rem] + d * A[i].p * A[i].b;
}

bool f(int n){
    long long ct = 0;
    for(int i = 0; i < N; i++){
        ct += cost(i, n);
        if(ct > X) return false;
    }
    return true;
}

int main(){
    scanf("%d %d", &N, &X);

    for(int i = 0; i < N; i++){
        int a, b, p, q;
        scanf("%d %d %d %d", &a, &p, &b, &q);
        if(a * q < b * p) swap(a, b), swap(p, q);
        A[i] = {a, p, b, q};
        dp[i][0] = 0;
        for(int j = 1; j < a * b; j++){
            dp[i][j] = min(dp[i][max(0, j - a)] + p, dp[i][max(0, j - b)] + q);
        }
    }

    int lo = 0, hi = 1e9;
    int ans = 0;
    while (lo <= hi) {
        int mid = (lo + hi) >> 1;
        if (f(mid)) {
            ans = mid;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    printf("%d\n", ans);
}
