#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Item {
    int a, p, b, q;
};

bool canFit(const vector<Item>& items, int X, int totalItems) {
    long long currentCost = 0;
    for (int i = 0; i < totalItems; ++i) {
        int n = X / (items[i].a * items[i].b);
        int rem = X % (items[i].a * items[i].b);
        currentCost += n * items[i].p * items[i].b + dp[i][rem];
        if (currentCost > X) return false;
    }
    return true;
}

int main() {
    int N, X;
    cin >> N >> X;

    vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        cin >> items[i].a >> items[i].p >> items[i].b >> items[i].q;
        if (items[i].a * items[i].q < items[i].b * items[i].p) swap(items[i].a, items[i].b), swap(items[i].p, items[i].q);
    }

    const int MAX_N = 100, MAX_X = 10000;
    long long dp[MAX_N][MAX_X + 1];
    for (int i = 0; i < N; ++i) {
        dp[i][0] = 0;
        for (int j = 1; j <= items[i].a * items[i].b; ++j) {
            dp[i][j] = min(j >= items[i].a ? dp[i][j - items[i].a] + items[i].p : INT_MAX,
                          j >= items[i].b ? dp[i][j - items[i].b] + items[i].q : INT_MAX);
        }
    }

    int low = 0, high = X;
    while (low < high) {
        int mid = (low + high) / 2;
        if (canFit(items, mid, N)) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }

    cout << low << endl;
    return 0;
}
