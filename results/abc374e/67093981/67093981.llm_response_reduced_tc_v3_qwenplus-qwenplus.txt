#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

typedef struct {
    int a, p, b, q;
} Printer;

Printer A[100];
long long dp[100][10000];

int N, X;

// Calculate the minimum cost to print 'n' pages on printer i
long long cost(int i, int n) {
    Printer pr = A[i];
    int lcm = pr.a * pr.b;
    long long full_blocks = n / lcm;
    int remainder = n % lcm;
    return full_blocks * (long long)pr.p * pr.b + dp[i][remainder];
}

// Check if we can print 'n' pages within budget X
bool is_possible(int n) {
    long long total_cost = 0;
    for (int i = 0; i < N; i++) {
        total_cost += cost(i, n);
        if (total_cost > X)
            return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> X;

    // Initialize each printer and precompute DP table
    for (int i = 0; i < N; i++) {
        int a, p, b, q;
        cin >> a >> p >> b >> q;
        
        // Ensure a*q >= b*p (prefer first method for large batches)
        if (a * q < b * p) {
            swap(a, b);
            swap(p, q);
        }
        
        A[i] = {a, p, b, q};
        
        // Base case
        dp[i][0] = 0;
        
        // Fill DP table up to a*b (the LCM in worst case)
        for (int j = 1; j <= a * b; j++) {
            long long use_a = LLONG_MAX;
            long long use_b = LLONG_MAX;
            
            if (j >= a) use_a = dp[i][j - a] + p;
            if (j >= b) use_b = dp[i][j - b] + q;
            
            dp[i][j] = min(use_a, use_b);
        }
    }

    // Binary search for maximum number of pages
    int left = 0;
    int right = 1e9;
    int answer = 0;

    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (is_possible(mid)) {
            answer = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    cout << answer << endl;
    return 0;
}
