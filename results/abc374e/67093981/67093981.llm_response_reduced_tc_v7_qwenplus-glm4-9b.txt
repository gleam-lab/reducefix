#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

typedef struct { int a, p, b, q; } p;
struct Range {
    int lo, hi;
};

vector<p> A;
vector<Range> ranges;

long long cost(int i, int n) {
    int d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
    return ranges[i].lo * rem + d * A[i].p;
}

bool canAfford(int n) {
    long long totalCost = 0;
    for (int i = 0; i < A.size(); ++i) {
        int d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
        totalCost += ranges[i].lo * rem;
        if (totalCost > n) return false;
        totalCost += d * A[i].p;
        if (totalCost > n) return false;
    }
    return true;
}

int main() {
    int N, X;
    scanf("%d %d", &N, &X);

    A.resize(N);
    ranges.resize(N);

    for (int i = 0; i < N; ++i) {
        int a, p, b, q;
        scanf("%d %d %d %d", &a, &p, &b, &q);
        if (a * q < b * p) swap(a, b), swap(p, q);
        A[i] = {a, p, b, q};

        if (b > a) {
            ranges[i] = {0, (X + a - 1) / a};
        } else {
            ranges[i] = {(X + a - 1) / a, 0};
        }
    }

    sort(ranges.begin(), ranges.end(), [](const Range& l, const Range& r) {
        return l.hi < r.hi;
    });

    int lo = 0, hi = 1e9;
    while (hi - lo > 1) {
        int mid = (hi + lo) / 2;
        if (canAfford(mid)) {
            lo = mid;
        } else {
            hi = mid;
        }
    }

    if (canAfford(hi)) {
        lo = hi;
    }

    printf("%d\n", lo);

    return 0;
}
