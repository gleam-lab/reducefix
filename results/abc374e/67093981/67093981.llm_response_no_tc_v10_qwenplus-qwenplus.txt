#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

typedef struct { int a, p, b, q; } Machine;
Machine machines[100];
long long dp[100][10000];  // dp[i][j] = min cost to produce j units using machine i

int N, X;

// Calculate the minimum cost to produce 'n' units on machine i
long long cost(int i, int n) {
    int cycle = A[i].a * A[i].b;
    long long full_cycles = n / cycle;
    int remainder = n % cycle;
    return full_cycles * (A[i].p * A[i].b + A[i].q * A[i].a) - full_cycles * min(A[i].p, A[i].q) + dp[i][remainder];
}

// Total cost for producing 'n' units across all machines
bool isPossible(int n) {
    long long totalCost = 0;
    for (int i = 0; i < N; ++i) {
        totalCost += cost(i, n);
        if (totalCost > X)
            return false;
    }
    return true;
}

int main() {
    scanf("%d %d", &N, &X);

    for (int i = 0; i < N; ++i) {
        int a, p, b, q;
        scanf("%d %d %d %d", &a, &p, &b, &q);

        // Ensure a*p <= b*q
        if (a * q < b * p) {
            swap(a, b);
            swap(p, q);
        }

        machines[i] = {a, p, b, q};

        // Initialize DP for this machine
        dp[i][0] = 0;
        for (int j = 1; j <= a * b; ++j) {
            dp[i][j] = LLONG_MAX;
            if (j >= a)
                dp[i][j] = min(dp[i][j], dp[i][j - a] + p);
            if (j >= b)
                dp[i][j] = min(dp[i][j], dp[i][j - b] + q);
        }
    }

    // Binary search over max units possible
    int low = 0, high = 1e6;  // Adjusted upper bound to avoid overflow
    int answer = 0;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (isPossible(mid)) {
            answer = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    printf("%d\n", answer);
}
