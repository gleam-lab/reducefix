#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

typedef struct { int a, p, b, q; } Item;
Item A[100];
long long dp[100][10000];

int N, X;

// Calculate the minimum cost to obtain 'n' units of item i
long long cost(int i, int n) {
    if (n == 0) return 0;
    long long d = n / (1LL * A[i].a * A[i].b);
    long long rem = n % (1LL * A[i].a * A[i].b);
    return dp[i][rem] + d * 1LL * A[i].p * A[i].b + d * 1LL * A[i].q * A[i].a;
}

// Check if we can produce 'n' units within budget X
bool isPossible(int n) {
    long long totalCost = 0;
    for (int i = 0; i < N; i++) {
        totalCost += cost(i, n);
        if (totalCost > X)
            return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> X;

    for (int i = 0; i < N; i++) {
        int a, p, b, q;
        cin >> a >> p >> b >> q;
        
        // Ensure the more efficient option is selected (lower cost per unit)
        if (1LL * a * q < 1LL * b * p) {
            swap(a, b);
            swap(p, q);
        }
        
        A[i] = {a, p, b, q};
        
        // Precompute minimum cost for 0 to a*b units using dynamic programming
        dp[i][0] = 0;
        for (int j = 1; j <= a * b; j++) {
            dp[i][j] = LLONG_MAX;
            if (j >= a) dp[i][j] = min(dp[i][j], dp[i][j - a] + p);
            if (j >= b) dp[i][j] = min(dp[i][j], dp[i][j - b] + q);
        }
    }

    // Binary search for maximum number of units we can afford
    int lo = 0, hi = 1e9;
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (isPossible(mid))
            lo = mid + 1;
        else
            hi = mid;
    }

    // After loop, lo is the first value where isPossible(lo) is false
    // So return lo-1 if lo > 0, otherwise 0
    cout << (isPossible(lo) ? lo : lo - 1) << endl;

    return 0;
}
