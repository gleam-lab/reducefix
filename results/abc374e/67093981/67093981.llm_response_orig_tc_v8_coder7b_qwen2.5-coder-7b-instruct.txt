#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Item {
    int a, p, b, q;
};

long long cost(const vector<Item>& items, int i, int n) {
    int d = n / (items[i].a * items[i].b);
    int rem = n % (items[i].a * items[i].b);
    return d * items[i].p * items[i].b + cost(items, i, rem);
}

bool canFit(const vector<Item>& items, int X, int n) {
    long long totalCost = 0;
    for (int i = 0; i < items.size(); ++i) {
        totalCost += cost(items, i, n);
        if (totalCost > X) return false;
    }
    return true;
}

int binarySearch(const vector<Item>& items, int X, int left, int right) {
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (canFit(items, X, mid)) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return right;
}

int main() {
    int N, X;
    cin >> N >> X;

    vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        cin >> items[i].a >> items[i].p >> items[i].b >> items[i].q;
        if (items[i].a * items[i].q < items[i].b * items[i].p) {
            swap(items[i].a, items[i].b);
            swap(items[i].p, items[i].q);
        }
    }

    cout << binarySearch(items, X, 0, X) << endl;

    return 0;
}
