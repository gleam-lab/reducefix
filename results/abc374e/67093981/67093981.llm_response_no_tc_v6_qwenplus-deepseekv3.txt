#include <iostream>
#include <algorithm>
using namespace std;

typedef struct{int a,p,b,q;} p;
p A[100];
long long dp[100][10001]; // Increased size to handle a*b cases

int N,X;
long long cost(int i, int n){
    long long full_cycles = n / (A[i].a * A[i].b);
    long long rem = n % (A[i].a * A[i].b);
    return full_cycles * min(A[i].p * A[i].b, A[i].q * A[i].a) + dp[i][rem];
}

bool f(int n){
    long long total_cost = 0;
    for(int i=0;i<N;i++){
        total_cost += cost(i,n);
        if(total_cost > X) return false;
    }
    return true;
}

        dp[i][0] = 0;
        for(int j=1;j<=A[i].a*A[i].b;j++){

            long long option2 = (j >= A[i].b) ? dp[i][j-A[i].b] + A[i].q : LLONG_MAX;
            dp[i][j] = min(option1, option2);
        }

    int lo=0, hi=X; // More reasonable upper bound
