
```cpp
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

typedef struct {
    int a, p, b, q;
} p;

int A[100];
long long dp[100][10000];

void updateDP(int i, int j, long long val) {
    while (j < 10000) {
        dp[i][j] = min(dp[i][j], val);
        j += A[i].b;
    }
}

int main() {
    int N, X;
    long long min_cost = 1e18;

    cin >> N >> X;

    for (int i = 0; i < N; ++i) {
        cin >> A[i].a >> A[i].p >> A[i].b >> A[i].q;
        A[i].a = __gcd(A[i].a, A[i].b);
        A[i].b /= A[i].a;
        A[i].p *= A[i].a;
        updateDP(i, 0, 0);
    }

    for (int j = 1; j <= 10000; ++j) {
        long long cost = 0;
        for (int i = 0; i < N; ++i) {
            if (dp[i][j] == 0) {
                if (j % A[i].b == 0) {
                    int d = j / A[i].b;
                    cost += d * A[i].p * A[i].b;
                }
                cost += A[i].q;
                if (cost > X) break;
            }
        }
        min_cost = min(min_cost, cost);
    }

    cout << min_cost << endl;

    return 0;
}
```
