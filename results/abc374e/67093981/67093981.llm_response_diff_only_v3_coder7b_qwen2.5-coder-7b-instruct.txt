```cpp
#include <iostream>
using namespace std;

typedef struct {
    int a, p, b, q;
} p;
p A[100];
long long dp[100][10001]; // Ensure array size is 10001 to handle all possible remainders

int N, X;
long long cost(int i, int n) {
    long long d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
    return dp[i][rem] + d * A[i].p * A[i].b;
}

bool f(int n) {
    long long ct = 0;
    for (int i = 0; i < N; i++) {
        ct += cost(i, n);
        if (ct > X) return false;
    }
    return true;
}

int main() {
    scanf("%d %d", &N, &X);

    for (int i = 0, a, b, p, q; i < N; i++) {
        scanf("%d %d %d %d", &a, &p, &b, &q);
        if (a * q < b * p) swap(a, b), swap(p, q);
        A[i] = {a, p, b, q};
        dp[i][0] = 0;
        for (int j = 1; j <= a * b; j++) {
            dp[i][j] = min(dp[i][max(0, j - a)] + p, dp[i][max(0, j - b)] + q);
        }
    }

    int lo = 0, hi = 1e9;
    while (hi - lo > 1) {
        int mid = (hi + lo) >> 1;
        if (f(mid)) lo = mid;
        else hi = mid - 1;
    }

    if (lo != hi && f(hi)) lo = hi;
    printf("%d\n", lo);
}
```

### Explanation of Changes
1. **Array Size Adjustment**: The `dp` array was adjusted to have a size of 10001 to ensure it can handle all possible remainders when dividing by `a * b`.
2. **Correct Typo in Array Declaration**: The array `dp` was declared as `long long dp[100][10000]`, but since we need to accommodate all possible remainders from 0 to `a * b`, the correct size should be `10001`.

These changes address potential overflow issues and ensure the algorithm runs correctly within the given constraints.
