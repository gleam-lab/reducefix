```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Process {
    int A, P, B, Q;
};

int main() {
    int N, X;
    cin >> N >> X;
    
    vector<Process> processes(N);
    for (auto& p : processes) {
        cin >> p.A >> p.P >> p.B >> p.Q;
        if (p.A * p.Q > p.B * p.P) swap(p.A, p.B), swap(p.P, p.Q); // Ensure P is always cheaper per product processed
    }

    vector<vector<long long>> dp(N + 1, vector<long long>(X + 1, X + 1)); // Initialize DP table with a large value
    dp[0][0] = 0; // Base case: no machines used, no cost, production capacity is 0

    for (int i = 1; i <= N; ++i) {
        for (int j = 0; j <= X; ++j) {
            dp[i][j] = dp[i - 1][j]; // Option 1: Don't use this process at all
            for (int k = 0; k * processes[i - 1].A * processes[i - 1].B <= j; ++k) {
                dp[i][j] = min(dp[i][j], dp[i - 1][j - k * processes[i - 1].A * processes[i - 1].B] + 
                    static_cast<long long>(k * processes[i - 1].P * processes[i - 1].B));
            }
        }
    }

    int max_capacity = 0;
    for (int j = 0; j <= X; ++j) {
        int total_cost = 0;
        for (int i = 1; i <= N; ++i) {
            total_cost += dp[i][j];
        }
        if (total_cost <= X) {
            max_capacity = max(max_capacity, j / (processes[N - 1].A * processes[N - 1].B));
        }
    }

    cout << max_capacity << endl;
    return 0;
}
```
