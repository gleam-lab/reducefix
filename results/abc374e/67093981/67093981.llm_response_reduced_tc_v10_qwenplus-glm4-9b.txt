#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

typedef struct { int a, p, b, q; } p;
vector<p> A;
vector<vector<long long>> dp;

int N, X;
long long cost(int i, int n) {
    long long d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
    return dp[i][rem] + d * A[i].p * A[i].b;
}
bool f(int n) {
    long long ct = 0;
    for (int i = 0; i < A.size(); i++) {
        ct += cost(i, n);
        if (ct > X) return false;
    }
    return true;
}

int main() {
    scanf("%d %d", &N, &X);

    A.resize(N);
    dp.resize(N, vector<long long>(10000, 0));

    for (int i = 0, a, b, p, q; i < N; i++) {
        scanf("%d %d %d %d", &a, &p, &b, &q);
        A[i] = {a, p, b, q};
        dp[i][0] = 0;
        for (int j = 1; j <= a * b; j++) {
            dp[i][j] = min(dp[i][max(0, j - a)] + p, dp[i][max(0, j - b)] + q);
        }
    }

    int lo = 0, hi = 1e9;
    while (hi - lo > 1) {
        int mid = (hi + lo) / 2;
        if (f(mid)) lo = mid;
        else hi = mid - 1;
    }

    if (lo != hi && f(hi)) lo = hi;
    cout << lo << endl;

    return 0;
}
