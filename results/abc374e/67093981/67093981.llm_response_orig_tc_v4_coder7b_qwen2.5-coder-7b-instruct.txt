#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Item {
    int a, p, b, q;
};

vector<Item> items;
vector<vector<long long>> dp;

long long cost(int i, int n) {
    long long d = n / (items[i].a * items[i].b), rem = n % (items[i].a * items[i].b);
    return dp[i][rem] + d * items[i].p * items[i].b;
}

bool feasible(long long n) {
    long long totalCost = 0;
    for (int i = 0; i < items.size(); ++i) {
        totalCost += cost(i, n);
        if (totalCost > X) return false;
    }
    return true;
}

int main() {
    int N, X;
    cin >> N >> X;

    items.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> items[i].a >> items[i].p >> items[i].b >> items[i].q;
        if (items[i].a * items[i].q < items[i].b * items[i].p) swap(items[i].a, items[i].b), swap(items[i].p, items[i].q);
        dp.push_back(vector<long long>(items[i].a * items[i].b + 1));
        dp.back()[0] = 0;
        for (int j = 1; j <= items[i].a * items[i].b; ++j) {
            dp.back()[j] = min(j >= items[i].a ? dp.back()[j - items[i].a] + items[i].p : LLONG_MAX,
                              j >= items[i].b ? dp.back()[j - items[i].b] + items[i].q : LLONG_MAX);
        }
    }

    long long low = 0, high = X;
    while (low < high) {
        long long mid = (low + high + 1) / 2;
        if (feasible(mid)) low = mid;
        else high = mid - 1;
    }

    cout << low << endl;
    return 0;
}
