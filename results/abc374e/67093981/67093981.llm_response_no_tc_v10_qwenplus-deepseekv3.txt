int N,X;

long long cost(int i, int n) {
    int ab = A[i].a * A[i].b;
    long long full_cycles = n / ab;
    int rem = n % ab;
    
    // Cost is full_cycles * (optimal cost for ab items) + cost for remaining
    return full_cycles * dp[i][ab] + dp[i][rem];
}

bool f(int n) {

        total_cost += cost(i, n);
        if(total_cost > X) return false;
    }
    return true;
}
