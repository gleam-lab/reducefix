#include <iostream>
#include <algorithm>
using namespace std;

typedef struct{ int a, p, b, q; } p;
p A[100];
long long dp[100][10000];

int N, X;
long long cost(int i, int n){
    long long d = n / (A[i].a * A[i].b), rem = n % (A[i].a * A[i].b);
    return dp[i][rem] + d * (A[i].p * A[i].b);
}

bool canAchieve(int n){
    long long c = 0;
    for(int i = 0; i < N; i++){
        c += cost(i, n);
        if(c > X) return false;
    }
    return true;
}

int main(){
    scanf("%d %d", &N, &X);

    for(int i = 0, a, p, b, q; i < N; i++){
        scanf("%d %d %d %d", &a, &p, &b, &q);
        if(a * q < b * p) swap(a, b), swap(p, q);
        A[i] = {a, p, b, q};
        for(int j = 0; j < a * b; j++)
            dp[i][j] = min(dp[i][max(0, j - a)] + p, dp[i][max(0, j - b)] + q);
    }

    int lo = 0, hi = 1e9;
    while(hi - lo > 1){
        int mid = lo + (hi - lo) / 2;
        if(canAchieve(mid)) lo = mid;
        else hi = mid;
    }

    if(lo != hi && canAchieve(hi)) lo = hi;
    printf("%d\n", lo);
}
