#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Product {
    int a, p, b, q;
};

long long cost(const vector<Product>& products, int i, int n) {
    int d = n / (products[i].a * products[i].b);
    int rem = n % (products[i].a * products[i].b);
    return d * products[i].p * products[i].b + cost(products, i, rem);
}

bool canMeetBudget(const vector<Product>& products, int budget, int target) {
    long long totalCost = 0;
    for (int i = 0; i < target; ++i) {
        totalCost += cost(products, i % products.size(), target - i);
        if (totalCost > budget) return false;
    }
    return true;
}

int main() {
    int N, X;
    cin >> N >> X;

    vector<Product> products(N);
    for (int i = 0; i < N; ++i) {
        cin >> products[i].a >> products[i].p >> products[i].b >> products[i].q;
        if (products[i].a * products[i].q < products[i].b * products[i].p) {
            swap(products[i].a, products[i].b);
            swap(products[i].p, products[i].q);
        }
    }

    int low = 0, high = X + 1;
    while (low < high) {
        int mid = (low + high) / 2;
        if (canMeetBudget(products, X, mid)) {
            low = mid + 1;
        } else {
            high = mid;
        }
    }

    cout << low - 1 << endl;
    return 0;
}
