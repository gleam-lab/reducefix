#include <bits/stdc++.h>

const int N = 105;

int n;
int x;
int a[N], b[N], p[N], q[N];
int c[N];

bool check(int mid) {
    for (int i = 1; i <= n; i++) {
        c[i] = INT_MAX;
        for (int j = 0; j <= b[i]; j++) {
            c[i] = min(c[i], j * p[i] + q[i] * ((mid - j * a[i] + b[i] - 1) / b[i]));
        }
        for (int j = 0; j <= b[i]; j++) {
            c[i] = min(c[i], j * q[i] + p[i] * ((mid - j * b[i] + a[i] - 1) / a[i]));
        }
    }
    int ans = 0;
    for (int i = 1; i <= n; i++) {
        ans += c[i];
    }
    if (ans > x) return false;
    return true;
}

    }
    int l = 1, r = 1e9; // Adjust upper bound based on problem constraints
    while (l < r) {
        int mid = l + (r - l) / 2;
        if (check(mid)) {
            r = mid;
        } else {
            l = mid + 1;

    }

    return 0;
}
