```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 105;
int n, x;
int a[N], p[N], b[N], q[N];

int compute_min_cost(int A, int P, int B, int Q, int required) {
    int min_cost = LLONG_MAX;

    // Try all possible number of type A attacks
    for (int j = 0; j <= B; j++) {
        int damage_from_A = j * B;
        int remaining = max(0LL, required - damage_from_A);
        int num_p = (remaining + A - 1) / A; // ceiling division
        min_cost = min(min_cost, j * Q + num_p * P);
    }

    // Try all possible number of type B attacks
    for (int j = 0; j <= A; j++) {
        int damage_from_B = j * A;
        int remaining = max(0LL, required - damage_from_B);
        int num_q = (remaining + B - 1) / B; // ceiling division
        min_cost = min(min_cost, j * P + num_q * Q);
    }

    return min_cost;
}

bool is_possible(int days) {
    int total_cost = 0;
    for (int i = 1; i <= n; i++) {
        int required_damage = days;
        int cost = compute_min_cost(a[i], p[i], b[i], q[i], required_damage);
        total_cost += cost;
        if (total_cost > x)
            return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> x;

    for (int i = 1; i <= n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int low = 0;
    int high = 1e9;
    int answer = 0;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (is_possible(mid)) {
            answer = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << answer << "\n";
    return 0;
}
```
