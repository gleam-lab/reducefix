#include <bits/stdc++.h>
using namespace std;

#define ll long long

const int N = 105;
ll n, x;
ll a[N], b[N], p[N], q[N];
ll c[N];

bool check(ll mid) {
    memset(c, INT_MAX, sizeof(c));
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= b[i]; ++j) {
            if (j * p[i] + q[i] * ((mid - j * a[i] + b[i] - 1) / b[i]) < c[i])
                c[i] = j * p[i] + q[i] * ((mid - j * a[i] + b[i] - 1) / b[i]);
            if (j * q[i] + p[i] * ((mid - j * b[i] + a[i] - 1) / a[i]) < c[i])
                c[i] = j * q[i] + p[i] * ((mid - j * b[i] + a[i] - 1) / a[i]);
        }
    }
    ll ans = 0;
    for (int i = 1; i <= n; ++i) {
        ans += c[i];
    }
    return ans <= x;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    ll l = 0, r = x;
    while (l < r) {
        ll mid = (l + r) / 2;
        if (check(mid)) {
            r = mid;
        } else {
            l = mid + 1;
        }
    }

    cout << l << '\n';
    return 0;
}
