#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N=105;
int n;
int x;
int a[N],b[N],p[N],q[N];

int c[N];
bool check(int mid)
{
    for(int i=1;i<=n;i++)
    {
        c[i] = (mid/a[i]) * b[i] * p[i] + (mid%a[i]) * b[i] * q[i];
    }
    int ans=0;
    for(int i=1;i<=n;i++)
    {
        ans+=c[i];
    }
    if(ans>x) return false;
    return true;
}
signed main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i]>>p[i]>>b[i]>>q[i];
    }
    int l=0,r=1000000000000LL;
    while(l+1<r)
    {
        int mid=(l+r)/2;
        if(check(mid))
        {
            l=mid;
        }
        else 
        {
            r=mid;
        }
    }
    
    cout<<l;
    return 0;
}
