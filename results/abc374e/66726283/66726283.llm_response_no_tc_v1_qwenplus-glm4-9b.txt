#include <iostream>
#include <vector>
#include <climits>
using namespace std;
#define int long long

const int N = 105;
int n;
int x;
vector<int> a(N), p(N), b(N), q(N);

vector<int> c(N);

bool check(int mid) {
    for (int i = 0; i < n; i++) {
        int cost1 = INT_MAX;
        int cost2 = INT_MAX;

        for (int j = 0; j <= b[i]; j++) {
            cost1 = min(cost1, j * p[i] + (mid - j * a[i]) * q[i] / b[i]);
        }
        for (int j = 0; j <= b[i]; j++) {
            cost2 = min(cost2, j * q[i] + (mid - j * b[i]) * p[i] / a[i]);
        }

        c[i] = min(cost1, cost2);
    }

    int ans = 0;
    for (int i = 0; i < n; i++) {
        ans += c[i];
    }

    return ans <= x;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int l = 1, r = n;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid)) {
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << l;
    return 0;
}
