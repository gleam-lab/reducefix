```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 105;
int n;
int x;
int a[N], b[N], p[N], q[N];

int c[N];
bool check(int mid)
{
    for(int i = 1; i <= n; i++)
    {
        c[i] = LLONG_MAX;
        // Try all possible number of type A attacks (j)
        // and compute minimum cost to reach at least 'mid' damage
        for(int j = 0; j <= b[i]; j++)
        {
            int damage_from_A = j * a[i];
            int remaining_damage = max(0LL, mid - damage_from_A);
            int num_B = (remaining_damage + b[i] - 1) / b[i]; // ceil division
            int total_cost = j * p[i] + num_B * q[i];
            c[i] = min(c[i], total_cost);
        }

        // Try all possible number of type B attacks (j)
        // and compute minimum cost to reach at least 'mid' damage
        for(int j = 0; j <= a[i]; j++)
        {
            int damage_from_B = j * b[i];
            int remaining_damage = max(0LL, mid - damage_from_B);
            int num_A = (remaining_damage + a[i] - 1) / a[i]; // ceil division
            int total_cost = j * q[i] + num_A * p[i];
            c[i] = min(c[i], total_cost);
        }
    }

    int total_cost = 0;
    for(int i = 1; i <= n; i++)
    {
        total_cost += c[i];
    }

    return total_cost <= x;
}

signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> x;

    int max_damage = 0;
    for(int i = 1; i <= n; i++)
    {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
        max_damage += max(a[i], b[i]) * 100; // upper bound: max damage possible from each monster in 100 attacks
    }

    int l = 0, r = max_damage;
    int answer = 0;

    while(l <= r)
    {
        int mid = (l + r) / 2;
        if(check(mid))
        {
            answer = mid;
            l = mid + 1;
        }
        else
        {
            r = mid - 1;
        }
    }

    cout << answer << '\n';
    return 0;
}
```
