```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 105;
int n, x;
int a[N], p[N], b[N], q[N];

bool check(int mid) {
    int total = 0;
    for (int i = 1; i <= n; ++i) {
        int min_cost = numeric_limits<int>::max();
        // Try all possible number of type A attacks
        for (int j = 0; j <= b[i]; ++j) {
            int rem = mid - j * a[i];
            if (rem <= 0) {
                min_cost = min(min_cost, j * p[i]);
                continue;
            }
            int k = rem / b[i] + (rem % b[i] != 0);
            min_cost = min(min_cost, j * p[i] + k * q[i]);
        }

        // Try all possible number of type B attacks
        for (int j = 0; j <= a[i]; ++j) {
            int rem = mid - j * b[i];
            if (rem <= 0) {
                min_cost = min(min_cost, j * q[i]);
                continue;
            }
            int k = rem / a[i] + (rem % a[i] != 0);
            min_cost = min(min_cost, j * q[i] + k * p[i]);
        }

        total += min_cost;
        if (total > x)
            return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> x;

    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int l = 1, r = 1e18;
    int ans = 0;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid)) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << ans << '\n';
    return 0;
}
```
