#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N=105;
int n;
int x;
int a[N],b[N],p[N],q[N];

int c[N];
bool check(int mid)
{
    for(int i=1;i<=n;i++)
    {
        c[i]=INT_MAX;
        for(int j=0;j<=b[i];j++)
        {
            c[i]=min(c[i],j*p[i]+q[i]*(mid-j*a[i]+b[i]-1)/b[i]);
        }
        for(int j=0;j<=a[i];j++)
        {
            c[i]=min(c[i],j*q[i]+p[i]*(mid-j*b[i]+a[i]-1)/a[i]);
        }
    }
    int ans=0;
    for(int i=1;i<=n;i++)
    {
        ans+=c[i];
    }
    return ans <= x;
}

int32_t main()
{
    cin.tie(nullptr);
    cout.tie(nullptr);
    ios_base::sync_with_stdio(false);
    cin>>n;
    cin>>x;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i]>>p[i]>>b[i]>>q[i];
    }
    int l=0,r=1000000000;
    while(l < r)
    {
        int mid = (l + r) / 2;
        if(check(mid))
        {
            l = mid + 1;
        }
        else
        {
            r = mid;
        }
    }
    cout << l;
    return 0;
}
