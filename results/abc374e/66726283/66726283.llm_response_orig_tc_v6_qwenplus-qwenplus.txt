```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 105;
int n, x;
int a[N], p[N], b[N], q[N];

int compute_cost(int i, int k) {
    // k is the required total damage to monster i
    int min_cost = numeric_limits<int>::max();
    // Try all possible number of type A attacks (0 to b[i])
    for (int j = 0; j <= b[i] && j <= k; ++j++) {
        int remaining = max(0LL, k - j * a[i]);
        int needed_b = (remaining + b[i] - 1) / b[i];
        int cost = j * p[i] + needed_b * q[i];
        min_cost = min(min_cost, cost);
    }

    // Try all possible number of type B attacks (0 to a[i])
    for (int j = 0; j <= a[i] && j <= k; ++j++) {
        int remaining = max(0LL, k - j * b[i]);
        int needed_a = (remaining + a[i] - 1) / a[i];
        int cost = j * q[i] + needed_a * p[i];
        min_cost = min(min_cost, cost);
    }
    return min_cost;
}

bool is_possible(int k) {
    int total_cost = 0;
    for (int i = 1; i <= n; ++i) {
        total_cost += compute_cost(i, k);
        if (total_cost > x)
            return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> x;

    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int low = 1;
    int high = 1e18; // Large upper bound
    int answer = 0;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (is_possible(mid)) {
            answer = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    cout << answer;
    return 0;
}
```
