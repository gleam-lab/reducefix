```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 105;
int n, x;
int a[N], p[N], b[N], q[N];

int compute_cost(int i, int t) {
    // Try all possible number of type A attacks (j)
    int cost = LLONG_MAX;
    for (int j = 0; j <= b[i]; ++j) {
        int damage = j * a[i];
        int remaining = max(0LL, t - damage);
        int numB = (remaining + b[i] - 1) / b[i];
        cost = min(cost, j * p[i] + numB * q[i]);
    }
    
    for (int j = 0; j <= a[i]; ++j) {
        int damage = j * b[i];
        int remaining = max(0LL, t - damage);
        int numA = (remaining + a[i] - 1) / a[i];
        cost = min(cost, j * q[i] + numA * p[i]);
    }
    
    return cost;
}

bool can_defeat_all(int mid) {
    int total_cost = 0;
    for (int i = 1; i <= n; ++i) {
        int cost = compute_cost(i, mid);
        total_cost += cost;
        if (total_cost > x) return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> x;

    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int l = 1, r = 1e18;
    int answer = 0;

    while (l <= r) {
        int mid = (l + r) / 2;
        if (can_defeat_all(mid)) {
            answer = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << answer << '\n';
    return 0;
}
```
