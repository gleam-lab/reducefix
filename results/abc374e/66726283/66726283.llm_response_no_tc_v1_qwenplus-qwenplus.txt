```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 105;
int n, x;
int a[N], b[N], p[N], q[N];

bool check(int mid) {
    int total_cost = 0;
    for (int i = 1; i <= n; ++i) {
        int min_cost = LLONG_MAX;
        // Try all possible number of type A attacks
        for (int j = 0; j <= b[i]; ++j) {
            int damage_from_A = j * a[i];
            int remaining_health = max(0LL, mid - damage_from_A);
            int num_B_attacks = (remaining_health + b[i] - 1) / b[i]; // ceil division
            int cost = j * p[i] + num_B_attacks * q[i];
            min_cost = min(min_cost, cost);
        }

        // Try all possible number of type B attacks
        for (int j = 0; j <= a[i]; ++j) {
            int damage_from_B = j * b[i];
            int remaining_health = max(0LL, mid - damage_from_B);
            int num_A_attacks = (remaining_health + a[i] - 1) / a[i]; // ceil division
            int cost = j * q[i] + num_A_attacks * p[i];
            min_cost = min(min_cost, cost);
        }

        total_cost += min_cost;
        if (total_cost > x)
            return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> x;
    for (int i = 1; i <= n; ++i)
        cin >> a[i] >> p[i] >> b[i] >> q[i];

    int l = 0, r = 1e9;
    int answer = 0;

    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid)) {
            answer = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << answer << "\n";
    return 0;
}
```
