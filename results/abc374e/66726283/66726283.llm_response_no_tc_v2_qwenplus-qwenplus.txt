```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 105;
int n, x;
int a[N], b[N], p[N], q[N];

bool check(int total_days) {
    int total_cost = 0;
    for (int i = 1; i <= n; ++i) {
        int min_cost = LLONG_MAX;
        // Try all possible days for type A, rest are type B
        for (int j = 0; j <= b[i] && j <= total_days; ++j) {
            int remaining = total_days - j;
            int required_a = max(0LL, (a[i] * remaining - b[i] + a[i] - 1) / a[i]);
            int cost = j * q[i] + p[i] * max(0LL, required_a);
            min_cost = min(min_cost, cost);
        }
        for (int j = 0; j <= a[i] && j <= total_days; ++j) {
            int remaining = total_days - j;
            int required_b = max(0LL, (b[i] * remaining - a[i] + b[i] - 1) / b[i]);
            int cost = j * p[i] + q[i] * max(0LL, required_b);
            min_cost = min(min_cost, cost);
        }
        total_cost += min_cost;
        if (total_cost > x) return false;
    }
    return true;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    int l = 0, r = 1e6; // Binary search range
    int ans = 0;

    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid)) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    cout << ans << "\n";
    return 0;
}
```
