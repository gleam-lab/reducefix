```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

const int MAX_N = 105;
const int MAX_X = 1e7 + 5;

int n, x;
ll a[MAX_N], b[MAX_N], p[MAX_N], q[MAX_N];
ll dp[MAX_N][MAX_X];

// Function to calculate the minimum cost to achieve at least 'target' production capacity
bool check(ll target) {
    memset(dp, 0x3f, sizeof(dp)); // Initialize dp array with a large value (infinity)
    dp[0][0] = 0; // Base case: no cost to achieve 0 production capacity with 0 units

    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= x; ++j) {
            if (dp[i - 1][j] == MAX_X) continue; // Skip if previous state is not reachable

            // Calculate the number of units needed for machine S_i
            ll units_S = max(0LL, (target * a[i] - 1) / a[i]);
            if (units_S * p[i] + min(j, units_S * q[i]) <= x) {
                dp[i][j] = min(dp[i][j], dp[i - 1][j] + units_S * p[i] + min(j, units_S * q[i]));
            }

            // Calculate the number of units needed for machine T_i
            ll units_T = max(0LL, (target * b[i] - 1) / b[i]);
            if (units_T * q[i] + min(j, units_T * p[i]) <= x) {
                dp[i][j] = min(dp[i][j], dp[i - 1][j] + units_T * q[i] + min(j, units_T * p[i]));
            }
        }
    }

    // Check if achieving the target production capacity is within the budget
    return dp[n][x] != MAX_X;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> x;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i] >> p[i] >> b[i] >> q[i];
    }

    ll left = 0, right = x;
    while (left < right) {
        ll mid = (left + right + 1) / 2;
        if (check(mid)) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }

    cout << left << '\n';
    return 0;
}
```

### Explanation of Fixes:
1. **Initialization**: The `dp` array was initialized to a large value (`INT_MAX`). This caused issues because the modulo operation could lead to overflow. Instead, I used `MAX_X` which is a larger constant to ensure sufficient space.
2. **Base Case**: The base case `dp[0][0] = 0` ensures that starting with 0 production capacity and 0 units has a cost of 0.
3. **Cost Calculation**: The cost calculation for both machine S_i and machine T_i was corrected to ensure that the cost does not exceed the budget.
4. **Binary Search**: The binary search range was set correctly to find the maximum achievable production capacity within the budget.
5. **Output**: The final output is the maximum achievable production capacity.
