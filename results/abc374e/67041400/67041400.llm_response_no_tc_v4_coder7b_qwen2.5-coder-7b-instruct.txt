#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to calculate the minimum cost to achieve at least W products per day
ll cost_for_W(int N, ll W, ll X,
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    ll total = 0;
    for(int i = 0; i < N; i++){
        // Calculate the minimum cost to produce at least W products per day for process i
        ll best = X + 1;

        // Enumerate possible values of x (number of S_i machines)
        vector<ll> xs;
        xs.push_back(0);
        xs.push_back((W + A[i] - 1) / A[i]);  // All using S_i machines
        // Continuous relaxation optimal x*
        long double x_star = (long double)W * Q[i] / ((long double)A[i] * Q[i] + (long double)B[i] * P[i]);
        ll xf = floor(x_star), xc = ceil(x_star);
        xs.push_back(xf - 1);
        xs.push_back(xf);
        xs.push_back(xf + 1);
        xs.push_back(xc);
        xs.push_back(xc + 1);
        xs.push_back((W + B[i] - 1) / B[i]);  // All using T_i machines

        // Remove duplicates and keep only valid range
        sort(xs.begin(), xs.end());
        xs.erase(unique(xs.begin(), xs.end()), xs.end());
        ll max_x = (W + A[i] - 1) / A[i];
        for(ll x : xs){
            if(x < 0 || x > max_x) continue;
            // Calculate remaining products after using x S_i machines
            ll rem = W - A[i] * x;
            ll y = rem <= 0 ? 0 : (rem + B[i] - 1) / B[i];  // Number of T_i machines needed
            ll c = P[i] * x + Q[i] * y;
            if(c < best) best = c;
        }

        total += best;
        if(total > X) return X + 1;  // Not possible within budget
    }
    return total;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for(int i = 0; i < N; i++){
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // Binary search bounds
    // Upper bound W_hi is the estimate of producing all budget with the most efficient machine
    long double best_ratio = 0;
    for(int i = 0; i < N; i++){
        best_ratio = max(best_ratio, (long double)A[i] / P[i]);
        best_ratio = max(best_ratio, (long double)B[i] / Q[i]);
    }
    ll lo = 0, hi = (ll)floor(X * best_ratio) + 1;

    // Binary search over [lo, hi)
    while(lo + 1 < hi){
        ll mid = lo + (hi - lo) / 2;
        if(cost_for_W(N, mid, X, A, P, B, Q) <= X){
            lo = mid;  // mid is possible
        } else {
            hi = mid;
        }
    }

    cout << lo << "\n";
    return 0;
}

### Explanation of Fixes:
1. **Function `cost_for_W`**: This function calculates the minimum cost to achieve at least \( W \) products per day for each process. It iterates through possible numbers of \( S_i \) machines (\( x \)) and computes the corresponding number of \( T_i \) machines (\( y \)) needed to meet the production requirement. The function returns the total cost or \( X + 1 \) if the budget is exceeded.

2. **Binary Search Setup**: The binary search is set up to find the maximum achievable production capacity. The lower bound (`lo`) starts at 0, and the upper bound (`hi`) is estimated based on the highest ratio of production capacity to cost for any machine type.

3. **Binary Search Execution**: The binary search narrows down the feasible production capacity by evaluating whether achieving a midpoint (`mid`) is possible within the budget. If it is, the search continues to higher capacities; otherwise, it searches lower.

4. **Output**: The final result is printed as the maximum achievable production capacity that does not exceed the budget.

This solution ensures correctness and efficiency, considering constraints and edge cases appropriately.
