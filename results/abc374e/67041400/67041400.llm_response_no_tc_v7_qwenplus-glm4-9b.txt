#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// 忙聢禄茫聜聤氓聙陇茫聛聦 X+1 盲禄楼盲赂聤茫聛陋茫聜聣茫聙聦盲赂聧氓聫炉猫聝陆茫聙聧茫聛篓茫聛驴茫聛陋茫聛聶
ll cost_for_W(int N, ll W, ll X,
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    ll total = 0;
    for(int i = 0; i < N; i++){
        ll best = X + 1;

        vector<ll> xs;
        xs.push_back((W + A[i] - 1) / A[i]);  //茫聛娄 Si
        xs.push_back((W + B[i] - 1) / B[i]);  //茫聛娄 Ti

        // 茅聙拢莽露職莽路漏氓聮聦茫聛聥茫聜聣茫聛庐忙聹聙茅聛漏 x*
        long double x_star = (long double)W * Q[i] / ( (long double)A[i]*Q[i] + (long double)B[i]*P[i] );
        ll xf = floor(x_star), xc = ceil(x_star);
        xs.push_back(xf-1);
        xs.push_back(xf);
        xs.push_back(xc);
        xs.push_back(xc+1);

        sort(xs.begin(), xs.end());
        xs.erase(unique(xs.begin(), xs.end()), xs.end());

        for(ll x: xs){
            if(x < 0 || x > (W + A[i] - 1) / A[i]) continue;
            ll rem = W - A[i]*x;
            ll y = rem <= 0 ? 0 : (rem + B[i] - 1) / B[i];
            ll c = P[i]*x + Q[i]*y;
            if(c < best) best = c;
        }

        total += best;
        if(total > X) return X+1;  // 茫聛聯茫聜聦盲禄楼盲赂聤茫聛炉茫聝聽茫聝陋
    }
    return total;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for(int i = 0; i < N; i++){
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // 盲潞聦氓聢聠忙聨垄莽麓垄茫聛庐盲赂聤盲赂聥茅聶聬
    ll lo = 0, hi = (ll)1e15; // 盲赂聤茅聶聬茫聜聮氓陇搂茫聛聧茫聛聫猫篓颅氓庐職茫聛聶茫聜聥
    while(lo < hi - 1){
        ll mid = (lo + hi) / 2;
        if(cost_for_W(N, mid, X, A, P, B, Q) <= X){
            lo = mid;  // mid 茫聛聦氓聫炉猫聝陆
        } else {
            hi = mid;
        }
    }

    cout << lo << "\n";
    return 0;
}
