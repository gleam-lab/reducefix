// Function to calculate the minimum cost to achieve a given capability W
ll cost_for_W(int N, ll W, ll X,
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    ll total = 0;
    for(int i = 0; i < N; i++){
        // Minimum cost to achieve capability W at machine i
        ll best = X + 1;

        // Possible values of x to consider
        vector<ll> xs;
        xs.push_back(0);
        xs.push_back((W + A[i] - 1) / A[i]);  // All Si
        // Optimal x* from continuous relaxation
        long double x_star = (long double)W * Q[i] / ( (long double)A[i]*Q[i] + (long double)B[i]*P[i] );
        ll xf = floor(x_star), xc = ceil(x_star);
        xs.push_back(xf-1);
        xs.push_back(xf);
        xs.push_back(xf+1);
        xs.push_back(xc);
        xs.push_back(xc+1);
        xs.push_back((W + B[i] - 1) / B[i]);  // All Ti as 'x' (S is 0 machines)

        // Remove duplicates and keep only valid values
        sort(xs.begin(), xs.end());
        xs.erase(unique(xs.begin(), xs.end()), xs.end());
        ll max_x = (W + A[i] - 1) / A[i];
        for(ll x: xs){
            if(x < 0 || x > max_x) continue;
            // Calculate remaining capacity after using x machines
            ll rem = W - A[i]*x;
            ll y = rem <= 0 ? 0 : (rem + B[i] - 1) / B[i];
            ll c = P[i]*x + Q[i]*y;
            if(c < best) best = c;
        }

        total += best;
        if(total > X) return X+1;  // If budget exceeded, return early
    }
    return total;
}
