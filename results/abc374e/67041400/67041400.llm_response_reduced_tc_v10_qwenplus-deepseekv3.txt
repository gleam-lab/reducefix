#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll cost_for_W(int N, ll W, ll X,
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    if (W == 0) return 0;  // special case: 0 work requires 0 cost
    
    ll total = 0;
    for(int i = 0; i < N; i++){
        ll best = X + 1;

        vector<ll> xs;
        xs.push_back(0);
        xs.push_back((W + A[i] - 1) / A[i]);
        long double x_star = (long double)W * Q[i] / ((long double)A[i]*Q[i] + (long double)B[i]*P[i]);
        ll xf = floor(x_star), xc = ceil(x_star);
        for (ll dx = -1; dx <= 1; dx++) {
            if (xf + dx >= 0) xs.push_back(xf + dx);
            if (xc + dx >= 0) xs.push_back(xc + dx);
        }
        xs.push_back((W + B[i] - 1) / B[i]);

        sort(xs.begin(), xs.end());
        xs.erase(unique(xs.begin(), xs.end()), xs.end());
        
        for(ll x: xs){
            if(x < 0) continue;
            ll rem = W - A[i]*x;
            ll y = max(0LL, rem <= 0 ? 0 : (rem + B[i] - 1) / B[i]);
            ll c = P[i]*x + Q[i]*y;
            if(c < best) best = c;
        }

        total += best;
        if(total > X) return X+1;
    }
    return total;
}

    ll lo = 0, hi = (ll)floor(X * best_ratio) + 2;
