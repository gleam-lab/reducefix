// Function to calculate the minimum cost to achieve at least W capacity with given budget X
ll cost_for_W(int N, ll W, ll X, 
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    ll total_cost = 0;
    for(int i = 0; i < N; ++i){
        // Calculate the minimum cost to achieve at least W capacity for the i-th machine
        ll min_cost = X + 1;

        // Generate candidate values for x
        vector<ll> candidates = {0};
        candidates.push_back((W + A[i] - 1) / A[i]);  // All machines running at full speed
        // Optimal x from continuous relaxation
        long double x_star = (long double)W * Q[i] / ((long double)A[i] * Q[i] + (long double)B[i] * P[i]);
        ll x_floor = floor(x_star), x_ceil = ceil(x_star);
        candidates.insert(candidates.end(), {x_floor - 1, x_floor, x_floor + 1, x_ceil, x_ceil + 1});
        candidates.push_back((W + B[i] - 1) / B[i]);  // All machines running at full time

        // Remove duplicates and filter valid candidates
        sort(candidates.begin(), candidates.end());
        candidates.erase(unique(candidates.begin(), candidates.end()), candidates.end());
        ll max_x = (W + A[i] - 1) / A[i];

        for(ll x : candidates){
            if(x < 0 || x > max_x) continue;
            // Calculate remaining capacity after using x machines
            ll remaining_capacity = W - A[i] * x;
            ll y = remaining_capacity <= 0 ? 0 : (remaining_capacity + B[i] - 1) / B[i];
            ll current_cost = P[i] * x + Q[i] * y;
            if(current_cost < min_cost) min_cost = current_cost;
        }

        total_cost += min_cost;
        if(total_cost > X) return X + 1;  // Budget exceeded
    }
    return total_cost;
}

    long double best_ratio = 0;
    for(int i = 0; i < N; ++i){

        best_ratio = max(best_ratio, (long double)B[i] / Q[i]);
