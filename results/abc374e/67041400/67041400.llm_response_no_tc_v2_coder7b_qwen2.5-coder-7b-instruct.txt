#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to calculate the cost for achieving a specific production capacity W
ll cost_for_W(int N, ll W, ll X,
    const vector<ll>& A, const vector<ll>& P,
    const vector<ll>& B, const vector<ll>& Q)
{
    ll total = 0;
    for(int i = 0; i < N; i++){
        // Calculate the minimum cost to achieve at least W products per day for process i
        ll best_cost = X + 1;

        // Iterate over possible numbers of machines S_i and T_i
        for(ll s = 0; s * A[i] <= W; ++s){
            ll t = (W - s * A[i]) / B[i];
            ll current_cost = s * P[i] + t * Q[i];
            if(current_cost < best_cost){
                best_cost = current_cost;
            }
        }

        total += best_cost;
        if(total > X){
            return X + 1;  // Budget exceeded, impossible to reach this W
        }
    }
    return total;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    ll X;
    cin >> N >> X;
    vector<ll> A(N), P(N), B(N), Q(N);
    for(int i = 0; i < N; i++){
        cin >> A[i] >> P[i] >> B[i] >> Q[i];
    }

    // Binary search for the maximum achievable production capacity
    ll lo = 0, hi = X + 1;  // Initialize bounds

    // Perform binary search
    while(lo < hi){
        ll mid = lo + (hi - lo) / 2;
        if(cost_for_W(N, mid, X, A, P, B, Q) <= X){
            lo = mid + 1;  // Try for a higher production capacity
        } else {
            hi = mid;  // Lower production capacity
        }
    }

    cout << lo - 1 << "\n";  // The largest valid production capacity
    return 0;
}
