// Calculates the minimum cost to achieve at least W capacity across all machines
// Returns X+1 if it's not achievable within the budget
ll cost_for_W(int N, ll W, ll X,
              const vector<ll>& A, const vector<ll>& P,
              const vector<ll>& B, const vector<ll>& Q)
{

        ll best = X + 1;

        // Compute x* using Lagrange multipliers relaxation
        // Minimize P*x + Q*y subject to A*x + B*y >= W
        // Relaxing to real numbers: y = max(0, (W - A*x)/B)
        // So cost ~ f(x) = P*x + Q*(max(0, (W - A*x)/B))
        // For continuous minimization: balance marginal cost efficiency

        long double numerator = Q[i] * W;
        long double denominator = A[i] * Q[i] + B[i] * P[i];
        long double x_star = (numerator / denominator);
        ll x_floor = static_cast<ll>(floor(x_star));
        ll x_ceil = x_floor + 1;

        // Generate candidate x values around x_star
        vector<ll> xs = {x_floor - 1, x_floor, x_floor + 1, x_ceil - 1, x_ceil, x_ceil + 1};
