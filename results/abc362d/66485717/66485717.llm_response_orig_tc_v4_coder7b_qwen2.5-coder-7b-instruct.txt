#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define INF 1e18

struct Edge {
    int u, v, w;
    bool operator<(const Edge &other) const {
        return w < other.w;
    }
};

struct DSU {
    vector<int> parent, rank;
    DSU(int n) : parent(n+1), rank(n+1, 1) {
        for(int i = 1; i <= n; i++) parent[i] = i;
    }
    int find(int x) {
        if(parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    void unionSet(int x, int y) {
        int px = find(x), py = find(y);
        if(px == py) return;
        if(rank[px] > rank[py]) swap(px, py);
        parent[px] = py;
        rank[py] += rank[px];
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, m;
    cin >> n >> m;
    vector<ll> a(n+1), dist(n+1, INF);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    vector<vector<pair<int, ll>>> adj(n+1);

    for(int i = 1; i <= n; i++) cin >> a[i];

    for(int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    pq.push({a[1], 1});
    dist[1] = a[1];

    while(!pq.empty()) {
        auto [w, u] = pq.top(); pq.pop();
        if(w != dist[u]) continue;

        for(auto &[v, bw]: adj[u]) {
            if(dist[v] > dist[u] + bw + a[v]) {
                dist[v] = dist[u] + bw + a[v];
                pq.push({dist[v], v});
            }
        }
    }

    for(int i = 2; i <= n; i++) cout << dist[i] << " ";
    cout << endl;

    return 0;
}
