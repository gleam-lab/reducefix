### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

constexpr i64 INF = 1e18;

struct Edge {
    int to;
    i64 weight;
};

struct Node {
    int vertex;
    i64 distance;
    bool operator>(const Node& other) const {
        return distance > other.distance;
    }
};

void solve() {
    int N, M;
    cin >> N >> M;
    vector<i64> A(N + 1);
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }
    vector<vector<Edge>> g(N + 1);
    for (int i = 0; i < M; ++i) {
        int u, v;
        i64 b;
        cin >> u >> v >> b;
        g[u].push_back({v, b});
        g[v].push_back({u, b});
    }
    
    vector<i64> dist(N + 1, INF);
    dist[1] = A[1];
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    pq.push({1, dist[1]});
    
    while (!pq.empty()) {
        Node node = pq.top();
        pq.pop();
        int u = node.vertex;
        i64 d = node.distance;
        if (d > dist[u]) continue;
        for (const Edge& e : g[u]) {
            int v = e.to;
            i64 new_dist = d + e.weight + A[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({v, new_dist});
            }
        }
    }
    
    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " \n"[i == N];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```

1. **Initialization**: The distances array `dist` is initialized to a large value (INF) except for the starting vertex (vertex 1), which is set to its own weight `A[1]`.
2. **Priority Queue**: The priority queue (min-heap) is initialized with the starting vertex and its initial distance. The queue is used to process vertices in order of increasing current distance.
3. **Processing Nodes**: For each vertex processed, the algorithm checks all adjacent vertices. If a shorter path to an adjacent vertex is found (considering both edge weight and the adjacent vertex's weight), the distance is updated, and the vertex is added to the priority queue for further processing.
4. **Output**: After processing all vertices, the distances from vertex 1 to all other vertices are printed.

This approach ensures that the shortest path is found efficiently, leveraging the properties of Dijkstra's algorithm with a priority queue for optimal performance. The solution handles large input sizes within acceptable time limits due to its O(M log N) complexity.
