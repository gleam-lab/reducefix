};

        g[y].emplace_back(x, c + a[x]);
    }
    
    vector<i64> d(n + 1, inf);
    vector<bool> vis(n + 1, false);
    priority_queue<Node> pq;
    
    d[1] = a[1];
    pq.push({1, d[1]});
    
    while (!pq.empty()) {

        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        
        for (const auto& [y, cost] : g[x]) {
            if (d[x] + cost < d[y]) {
                d[y] = d[x] + cost;
                pq.push({y, d[y]});
            }
        }
    }
