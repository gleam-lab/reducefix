struct Node {
    int x;
    i64 w;
    bool operator < (const Node& u) const {
        return w > u.w;
    }
};
constexpr int N = 2e5 + 10;
priority_queue<Node> pq;
vector<pair<int, int>> g[N];
void solve2() {
    int n, m;
    cin >> n >> m;
    V<i64> a(n + 1);
    vector<i64> d(n + 1, 1e18);
    vector<bool> vis(n + 1, false);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }
    d[1] = a[1];
    pq.push({1, d[1]});
    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        for (const auto& [y, dw] : g[x]) {
            if (d[x] + dw + a[y] < d[y]) {
                d[y] = d[x] + dw + a[y];
                pq.push({y, d[y]});
            }
        }
    }
    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " \n"[i == n];
    }
}
INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);

        solve2();
