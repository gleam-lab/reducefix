#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2 * 1e5 + 10;
vector<pair<int, int>> adj[MAXN];
long long dist[MAXN];

void dijkstra(int start, int n) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    fill(dist, dist + n + 1, LLONG_MAX);
    dist[start] = 0;
    pq.push({0, start});
    
    while (!pq.empty()) {
        int u = pq.top().second;
        long long d = pq.top().first;
        pq.pop();
        
        if (d > dist[u])
            continue;
        
        for (auto &[v, w] : adj[u]) {
            if (dist[u] + w + v < dist[v]) {
                dist[v] = dist[u] + w + v;
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    int n, m;
    cin >> n >> m;
    
    for (int i = 1; i <= n; ++i) {
        cin >> dist[i];
    }
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    
    dijkstra(1, n);
    
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    
    return 0;
}
