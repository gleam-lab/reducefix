#include <bits/stdc++.h>
using namespace std;
#define INT long long 

template <class T>
using V = vector<T>;
template <class T>
using VV = vector<vector<T>>;

const int N = 3e5 + 10;
const INT inf = 1e10;

struct DSU {
private:
    V<int> fa, sz;
public:
    DSU(int n) : fa(n + 1), sz(n + 1, 1) {
        iota(fa.begin(), fa.end(), 0);
    }
    int find(int x) {
        if (fa[x] != x) {
            fa[x] = find(fa[x]);
        }
        return fa[x];
    }
    bool same(int x, int y) {
        return find(x) == find(y);
    }

    bool merge(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return false;
        if (sz[x] < sz[y]) swap(x, y);
        fa[y] = x;
        sz[x] += sz[y];
        return true;
    }
    int size(int x) {
        return sz[find(x)];
    }
};

struct Edge {
    int x, y, c;
    bool operator < (const Edge& other) const {
        return c == other.c ? x < other.x : c < other.c;
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    V<INT> a(n + 1);
    VV<Edge> g(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < m; ++i) {
        int u, v, c;
        cin >> u >> v >> c;
        g[u].push_back({v, c});
        g[v].push_back({u, c});
    }
    
    // Dijkstra's algorithm to find shortest paths from vertex 1
    priority_queue<Edge, VV<Edge>, greater<Edge>> pq;
    V<INT> dist(n + 1, inf);
    dist[1] = a[1];
    pq.push({1, 1, a[1]});
    while (!pq.empty()) {
        auto [u, len, cost] = pq.top();
        pq.pop();
        if (dist[u] < len) continue;
        for (auto& [v, w] : g[u]) {
            if (dist[u] + w + a[v] < dist[v]) {
                dist[v] = dist[u] + w + a[v];
                pq.push({v, dist[v], dist[v]});
            }
        }
    }
    
    // Output the minimum weights for paths to all vertices 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << (i == n ? "" : " ");
    }
    cout << endl;
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
