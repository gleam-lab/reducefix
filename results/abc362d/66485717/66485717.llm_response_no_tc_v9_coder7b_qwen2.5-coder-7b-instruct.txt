```cpp
#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define vi vector<int>
#define vvi vector<vi>

const int MAXN = 2 * 100010;
const LL INF = 1e18;

vvi adj(MAXN);
vi weight(MAXN);
vi dist(MAXN);

void dijkstra(int src) {
    priority_queue<pair<LL, int>, vector<pair<LL, int>>, greater<pair<LL, int>>> pq;
    fill(dist.begin(), dist.end(), INF);
    dist[src] = weight[src];
    pq.push({dist[src], src});

    while (!pq.empty()) {
        int u = pq.top().second;
        LL du = pq.top().first;
        pq.pop();

        if (du != dist[u])
            continue;

        for (auto &e : adj[u]) {
            int v = e.first;
            LL dv = e.second + weight[v];

            if (dist[u] + dv < dist[v]) {
                dist[v] = dist[u] + dv;
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        cin >> weight[i];

    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    dijkstra(1);

    for (int i = 2; i <= n; ++i)
        cout << dist[i] << " ";

    return 0;
}
```

### Explanation
1. **Graph Representation**: We use an adjacency list (`adj`) to represent the graph.
2. **Weight Initialization**: We store the weight of each vertex in the `weight` array.
3. **Dijkstra's Algorithm**: We implement Dijkstra's algorithm using a min-priority queue to efficiently find the shortest path from the source vertex (vertex 1) to all other vertices.
4. **Output**: We print the shortest distances from vertex 1 to all other vertices.
