#include <bits/stdc++.h>
using namespace std;
#define int long long 
template <class T>
using V = vector<T>;
struct DSU {
private:
    vector<int> fa, sz;
public:
    DSU(int n) : fa(n + 1), sz(n + 1, 1) {
        iota(fa.begin(), fa.end(), 0);
    }
    int find(int x) {
        int root = x;
        while (root != fa[root]) {
            root = fa[root];
        }
        while (x != root) {
            int nex = fa[x];
            fa[x] = fa[fa[x]];
            x = nex;
        }
        return root;
    }
    bool same(int x, int y) {
        return find(x) == find(y);
    }

    bool merge(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return false;
        if (sz[x] < sz[y]) swap(x, y);
        fa[y] = x;
        sz[x] += sz[y];
        return true;
    }
    int size(int x) {
        return sz[find(x)];
    }
};
struct Edge {
    int x, y, c;
    bool operator < (const Edge& other) const {
        return c < other.c;
    }
};
using i64 = long long;
void solve() {
    int n, m;
    cin >> n >> m;
    V<i64> a(n + 1, 0);
    vector<Edge> edges;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        edges.emplace_back(x, y, c);
    }
    sort(edges.begin(), edges.end());
    DSU dsu(n + 1);
    vector<i64> d(n + 1, LLONG_MAX);
    d[1] = a[1];
    int cnt = 0;
    for (const auto& edge : edges) {
        if (dsu.same(edge.x, edge.y)) continue;
        dsu.merge(edge.x, edge.y);
        cnt++;
        if (cnt == n - 1) break;
    }
    for (int i = 2; i <= n; ++i) {
        i64 min_weight = LLONG_MAX;
        for (const auto& edge : edges) {
            if (!dsu.same(edge.x, edge.y)) {
                min_weight = min(min_weight, d[edge.x] + edge.c + a[i]);
            }
        }
        d[i] = min_weight;
    }
    for (int i = 2; i <= n; ++i) {
        cout << (i == n ? d[i] : d[i] + a[i] - a[1]) << " ";
    }
}
int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    solve();
    return 0;
}
