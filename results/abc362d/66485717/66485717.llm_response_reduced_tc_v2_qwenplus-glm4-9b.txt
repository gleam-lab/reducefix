#include <bits/stdc++.h>
using namespace std;
#define int long long
#define pb push_back

using V = vector<int>;
using VII = vector<VI>;

int N, M;
VII G(N + 1);

struct DSU {
    V parent, rank;
    DSU(int n) : parent(n + 1), rank(n + 1, 0) {
        iota(parent.begin(), parent.end(), 0);
    }
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    void unionSets(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX != rootY) {
            if (rank[rootX] > rank[rootY]) {
                parent[rootY] = rootX;
            } else if (rank[rootX] < rank[rootY]) {
                parent[rootX] = rootY;
            } else {
                parent[rootY] = rootX;
                rank[rootX]++;
            }
        }
    }
};

void solve() {
    cin >> N >> M;
    V weight(N + 1);
    V dist(N + 1, LLONG_MAX);
    
    for (int i = 1; i <= N; ++i) {
        cin >> weight[i];
    }

    DSU dsu(N);

    for (int i = 0; i < M; ++i) {
        int u, v, c;
        cin >> u >> v >> c;
        G[u].emplace_back(v, c), G[v].emplace_back(u, c);
    }

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    
    pq.emplace(weight[1] + 0, 1);
    dist[1] = weight[1];

    while (!pq.empty()) {
        auto [d, node] = pq.top();
        pq.pop();

        if (dist[node] < d) continue;

        for (auto &edge : G[node]) {
            int nextNode = edge.first;
            long long newDist = d + edge.second + weight[nextNode];
            
            if (newDist < dist[nextNode]) {
                dist[nextNode] = newDist;
                pq.emplace(newDist, nextNode);
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << (i < N ? " " : "");
    }
    cout << '\n';
}

int main() {
    cin.tie(nullptr);
    cin.sync_with_stdio(false);
    solve();
    return 0;
}
