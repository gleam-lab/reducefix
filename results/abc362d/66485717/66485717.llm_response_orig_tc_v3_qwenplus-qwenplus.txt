```cpp
#include <bits/stdc++.h>
using namespace std;

using i64 = long long;

constexpr i64 INF = 1e18;

struct Node {
    int node;
    i64 dist;
    bool operator<(const Node& other) const {
        return dist > other.dist;
    }
};

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);

    int N, M;
    cin >> N >> M;

    vector<i64> A(N + 1);
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }

    vector<vector<pair<int, i64>>> graph(N + 1);
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        // The total weight of a path is the sum of vertex weights and edge weights.
        // So each edge contributes its own weight and the destination vertex weight.
        // We model edge from u to v as weight B_j + A_v
        graph[u].emplace_back(v, w + A[v]);
        graph[v].emplace_back(u, w + A[u]);
    }

    vector<i64> dist(N + 1, INF);
    priority_queue<Node> pq;

    dist[1] = A[1]; // Start at node 1, include A_1 since we're visiting it
    pq.push({1, dist[1]});

    while (!pq.empty()) {
        auto [u, d] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (const auto& [v, cost] : graph[u]) {
            if (dist[v] > dist[u] + cost) {
                dist[v] = dist[u] + cost;
                pq.push({v, dist[v]});
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " ";
    }
    cout << '\n';

    return 0;
}
```
