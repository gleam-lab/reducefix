#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define INF 1e18

vector<pair<ll, ll>> adj[200005];

ll dijkstra(ll start, ll end) {
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
    vector<ll> dist(200005, INF);
    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        ll u = pq.top().second;
        ll d = pq.top().first;
        pq.pop();

        if (u == end) break;

        if (d > dist[u]) continue;

        for (auto v : adj[u]) {
            ll v_id = v.first;
            ll v_dist = v.second;

            if (dist[v_id] > dist[u] + v_dist) {
                dist[v_id] = dist[u] + v_dist;
                pq.push({dist[v_id], v_id});
            }
        }
    }

    return dist[end];
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll n, m;
    cin >> n >> m;

    vector<ll> a(n + 1);
    for (ll i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    for (ll i = 0; i < m; ++i) {
        ll u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    for (ll i = 2; i <= n; ++i) {
        ll min_cost = a[1] + dijkstra(1, i);
        cout << min_cost << " ";
    }
    cout << endl;

    return 0;
}
