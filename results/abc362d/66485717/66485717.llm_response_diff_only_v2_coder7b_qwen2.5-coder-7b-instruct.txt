#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define PII pair<int, int>

struct Edge {
    int from, to, cost;
};

struct DisjointSetUnion {
    vector<int> parent, rank;

    DisjointSetUnion(int n) : parent(n + 1), rank(n + 1, 1) {
        for (int i = 1; i <= n; ++i)
            parent[i] = i;
    }

    int find(int u) {
        if (parent[u] != u)
            parent[u] = find(parent[u]);
        return parent[u];
    }

    void unite(int u, int v) {
        int pu = find(u), pv = find(v);
        if (pu == pv) return;
        if (rank[pu] < rank[pv])
            swap(pu, pv);
        parent[pv] = pu;
        if (rank[pu] == rank[pv])
            ++rank[pu];
    }
};

struct Graph {
    int n, m;
    vector<Edge> edges;
    vector<vector<PII>> adjList;

    Graph(int _n, int _m) : n(_n), m(_m), adjList(n + 1) {}

    void addEdge(int u, int v, int cost) {
        edges.push_back({u, v, cost});
        adjList[u].push_back({v, cost});
        adjList[v].push_back({u, cost});
    }

    vector<LL> dijkstra(int start) {
        vector<LL> dist(n + 1, LLONG_MAX);
        priority_queue<pair<LL, int>, vector<pair<LL, int>>, greater<pair<LL, int>>> pq;
        dist[start] = 0;
        pq.push({0, start});

        while (!pq.empty()) {
            auto [currentDist, u] = pq.top();
            pq.pop();

            if (currentDist > dist[u]) continue;

            for (auto [v, cost] : adjList[u]) {
                if (dist[u] + cost + edges[(u - 1) * n + v - u].cost < dist[v]) {
                    dist[v] = dist[u] + cost + edges[(u - 1) * n + v - u].cost;
                    pq.push({dist[v], v});
                }
            }
        }

        return dist;
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    Graph g(n, m);

    for (int i = 1; i <= m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        g.addEdge(u, v, b);
    }

    vector<LL> result(n + 1, LLONG_MAX);
    result[1] = a[1];

    for (int i = 2; i <= n; ++i) {
        vector<LL> dist = g.dijkstra(i);
        for (int j = 1; j <= n; ++j) {
            result[j] = min(result[j], dist[j] + a[j]);
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << result[i] << " ";
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();

    return 0;
}
