#include <bits/stdc++.h>
using namespace std;
#define INT long long
#define pb push_back

template <class T>
using V = vector<T>;
using E = pair<int, int>;

struct DSU {
    vector<int> parent, size;
    DSU(int n) : parent(n + 1), size(n + 1, 1) {
        for (int i = 0; i <= n; ++i) {
            parent[i] = i;
        }
    }
    int find(int u) {
        if (u != parent[u]) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }
    void unite(int u, int v) {
        int root_u = find(u);
        int root_v = find(v);
        if (root_u != root_v) {
            if (size[root_u] < size[root_v]) {
                swap(root_u, root_v);
            }
            parent[root_v] = root_u;
            size[root_u] += size[root_v];
        }
    }
    int getSize(int u) {
        return size[find(u)];
    }
};

void solve() {
    INT n, m;
    cin >> n >> m;
    V<INT> a(n + 1);
    V<E> edges;
    DSU dsu(n);

    for (INT i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    for (INT i = 0; i < m; ++i) {
        INT u, v, w;
        cin >> u >> v >> w;
        edges.pb({u, v, w});
        dsu.unite(u, v);
    }

    // Find the number of connected components
    V<INT> component_size(n + 1, 0);
    for (INT i = 1; i <= n; ++i) {
        component_size[dsu.find(i)]++;
    }

    // Calculate the minimum weight for each component
    sort(edges.begin(), edges.end(), [](const E& a, const E& b) {
        return a.second < b.second;
    });

    V<INT> min_weight(n + 1, LLONG_MAX);
    for (INT i = 1; i <= n; ++i) {
        min_weight[i] = LLONG_MAX;
    }

    for (const auto& [u, v, w] : edges) {
        INT root_u = dsu.find(u);
        INT root_v = dsu.find(v);
        if (root_u != root_v) {
            dsu.unite(u, v);
            INT new_component = dsu.getSize(root_u);
            INT total_weight = dsu.getSize(root_u) * a[root_u];
            min_weight[root_u] = min(min_weight[root_u], total_weight + w);
            min_weight[root_v] = min(min_weight[root_v], total_weight + w);
        }
    }

    // Output the minimum weights for each component
    for (INT i = 2; i <= n; ++i) {
        cout << min_weight[dsu.find(i)] << " ";
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
