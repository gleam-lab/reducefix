#include <bits/stdc++.h>
using namespace std;
#define int long long

template <class T>
using V = vector<T>;
template <class T>

struct Node {
    int x;
    long long w;
    bool operator < (const Node& u) const {
        return w > u.w;
    }
};

constexpr int N = 2e5 + 10;
constexpr long long inf = 1LL << 60;
priority_queue<Node> pq;
vector<pair<int, int>> g[N];

void solve() {
    int n, m;
    cin >> n >> m;
    V<int> a(n + 1, 0);
    V<long long> d(n + 1, inf);
    V<bool> vis(n + 1, false);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c + a[y]);
        g[y].emplace_back(x, c + a[x]);
    }
    pq.push({1, d[1] = a[1]});
    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        for (const auto& [y, dw] : g[x]) {
            if (d[x] + dw < d[y]) {
                d[y] = d[x] + dw;
                pq.push({y, d[y]});
            }
        }
    }
    for (int i = 2; i <= n; ++i) {
        cout << (d[i] == inf ? -1 : d[i]) << " \n"[i == n];
    }
}

INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    while (t--) {
        solve();
    }
    return 0;
}

### Explanation
1. **Graph Representation**: The graph is built using adjacency lists. Each edge from vertex `x` to `y` is stored with a weight that is the sum of the edge weight `c` and the vertex weight `a[y]` (or `a[x]` for the reverse direction).
2. **Dijkstra's Algorithm**: The priority queue is used to process vertices in order of increasing distance from the source vertex (vertex 1). The distance array `d` is initialized to `inf` (a very large number), except for the source vertex which is initialized to its vertex weight `a[1]`.
3. **Edge Cases Handling**: The algorithm checks if vertices are reachable. If a vertex is unreachable, it outputs `-1`; otherwise, it outputs the computed shortest path weight. This ensures correctness even when there are no edges (as in the failing test case).
4. **Efficiency**: The algorithm runs in O((N + M) log N) time due to the use of a priority queue, which is efficient for the given constraints. This ensures the solution is scalable and handles large inputs efficiently.
