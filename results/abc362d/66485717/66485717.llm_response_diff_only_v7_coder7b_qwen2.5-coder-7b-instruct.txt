#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>
#define vvi vector<vi>
#define pii pair<ll,ll>
#define mp make_pair
#define pb push_back
#define all(x) begin(x), end(x)

const ll INF = 1e18;

struct DSU {
    vi parent, rank;
    DSU(ll n) : parent(n+1), rank(n+1, 1) {
        iota(all(parent), 0);
    }
    ll find(ll u) {
        return parent[u] == u ? u : parent[u] = find(parent[u]);
    }
    void merge(ll u, ll v) {
        ll pu = find(u), pv = find(v);
        if(pu == pv) return;
        if(rank[pu] < rank[pv]) swap(pu, pv);
        parent[pv] = pu;
        rank[pu] += rank[pv];
    }
};

struct Graph {
    vi dist;
    vvi adj;
    Graph(ll n) : dist(n+1, INF), adj(n+1) {}
    void addEdge(ll u, ll v, ll w) {
        adj[u].pb({v, w});
        adj[v].pb({u, w});
    }
    void dijkstra(ll src) {
        priority_queue<pii, vector<pii>, greater<pii>> pq;
        pq.push({0, src});
        dist[src] = 0;
        while(!pq.empty()) {
            auto [w, u] = pq.top(); pq.pop();
            if(w > dist[u]) continue;
            for(auto &[v, wt]: adj[u]) {
                if(dist[u] + wt < dist[v]) {
                    dist[v] = dist[u] + wt;
                    pq.push({dist[v], v});
                }
            }
        }
    }
};

void solve() {
    ll n, m; cin >> n >> m;
    vi a(n+1); for(ll &i: a) cin >> i;
    Graph g(n);
    for(ll i = 0; i < m; i++) {
        ll u, v, w; cin >> u >> v >> w;
        g.addEdge(u, v, w);
    }
    g.dijkstra(1);
    for(ll i = 2; i <= n; i++) cout << g.dist[i] + a[i] << " ";
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    solve();
    return 0;
}
