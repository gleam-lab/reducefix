#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define mod 1000000007
#define vi vector<ll>
#define vvi vector<vi>

vvi adj;
vi dist, cost;
ll n, m;

void dijkstra(int src) {
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
    dist[src] = cost[src];
    pq.push({dist[src], src});
    
    while (!pq.empty()) {
        auto [dis, node] = pq.top();
        pq.pop();
        
        if (dis > dist[node])
            continue;
        
        for (auto &child : adj[node]) {
            int childNode = child.first;
            ll childCost = child.second;
            
            if (dist[node] + childCost + cost[childNode] < dist[childNode]) {
                dist[childNode] = dist[node] + childCost + cost[childNode];
                pq.push({dist[childNode], childNode});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    
    cin >> n >> m;
    cost.resize(n+1, 0);
    adj.resize(n+1);
    
    for (int i = 1; i <= n; i++)
        cin >> cost[i];
    
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    
    dist.assign(n+1, LLONG_MAX);
    dijkstra(1);
    
    for (int i = 2; i <= n; i++)
        cout << dist[i] << " ";
    
    return 0;
}
