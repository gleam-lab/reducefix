#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

struct Edge {
    int to;
    int weight;
};

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n + 1);
    vector<vector<Edge>> graph(n + 1);
    
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }
    
    // Use Dijkstra's algorithm to find the shortest path from vertex 1 to all other vertices
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    vector<long long> min_distance(n + 1, LLONG_MAX);
    
    pq.push({a[1], 1});
    min_distance[1] = a[1];
    
    while (!pq.empty()) {
        auto [current_distance, u] = pq.top();
        pq.pop();
        
        if (current_distance > min_distance[u]) continue;
        
        for (const auto& edge : graph[u]) {
            int v = edge.to;
            int weight = edge.weight;
            long long distance = current_distance + weight + a[v];
            
            if (distance < min_distance[v]) {
                min_distance[v] = distance;
                pq.push({distance, v});
            }
        }
    }
    
    // Output the minimum distance from vertex 1 to each vertex from 2 to n
    for (int i = 2; i <= n; ++i) {
        cout << min_distance[i] << " ";
    }
    cout << endl;
    
    return 0;
}
