#include <bits/stdc++.h>
using namespace std;

struct Node {

        return w > u.w;

constexpr int N = 2e5 + 10;
constexpr long long inf = 1LL << 60;
priority_queue<Node> pq;
vector<pair<int, int>> g[N];

void solve2() {
    int n, m;
    cin >> n >> m;
    V<int> a(n + 1, 0);
    vector<long long> d(n + 1, inf);
    vector<bool> vis(n + 1, false);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }
    pq.push({1, d[1] = a[1]});
    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        for (const auto& [y, dw] : g[x]) {
            if (d[x] + dw + a[y] < d[y]) {
                d[y] = d[x] + dw + a[y];
                pq.push({y, d[y]});
            }
        }
    }
    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " \n"[i == n];
    }
}

INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    while (t--) {
        solve2();
    }
    return 0;
}

2. **Priority Queue**: The `Node` struct uses a custom operator to ensure the priority queue processes nodes in order of increasing distance (`w`). This is crucial for Dijkstra's algorithm.
3. **Relaxation Step**: For each edge from node `x` to `y`, the algorithm checks if the path through `x` to `y` offers a shorter distance (`d[x] + edge weight + vertex weight of y`). If so, it updates the distance and adds `y` to the priority queue.
4. **Output**: The results for nodes 2 through `N` are printed, showing the minimum path weights from node 1 to each respective node.
