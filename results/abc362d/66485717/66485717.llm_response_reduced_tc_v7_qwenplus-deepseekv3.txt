2. **Priority Queue**: Use a min-heap (priority queue) to implement Dijkstra's algorithm efficiently. The priority queue should store nodes with their current distance, allowing us to always process the node with the smallest current distance next.
3. **Edge Cases**: Handle cases where there are no edges (like the failing test case) by ensuring the distance from vertex 1 to itself is set to its weight, and other vertices remain at infinity if unreachable.

### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

constexpr i64 INF = 9223372036854775807LL;

struct Node {
    int x;
    i64 w;
    bool operator < (const Node& u) const {
        return w > u.w;
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    vector<vector<pair<int, i64>>> g(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        i64 b;
        cin >> u >> v >> b;
        g[u].emplace_back(v, b);
        g[v].emplace_back(u, b);
    }
    vector<i64> d(n + 1, INF);
    vector<bool> vis(n + 1, false);
    priority_queue<Node> pq;
    d[1] = a[1];
    pq.push({1, d[1]});
    while (!pq.empty()) {
        auto node = pq.top();
        pq.pop();
        int x = node.x;
        if (vis[x]) continue;
        vis[x] = true;
        for (const auto& [y, b] : g[x]) {
            if (d[x] + b + a[y] < d[y]) {
                d[y] = d[x] + b + a[y];
                pq.push({y, d[y]});
            }
        }
    }
    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " \n"[i == n];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
