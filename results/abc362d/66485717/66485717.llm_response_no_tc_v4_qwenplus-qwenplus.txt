#include <bits/stdc++.h>
using namespace std;

using i64 = long long;

constexpr i64 INF = 1e18;

struct Node {
    int vertex;
    i64 cost;
    bool operator<(const Node& other) const {
        return cost > other.cost; // min-heap
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<i64> A(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> A[i];
    }

    vector<vector<pair<int, i64>>> graph(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        i64 b;
        cin >> u >> v >> b;
        graph[u].emplace_back(v, b);
        graph[v].emplace_back(u, b);
    }

    vector<i64> minPath(n + 1, INF);
    priority_queue<Node> pq;

    // Start from vertex 1
    minPath[1] = A[1]; // Only the weight of the starting vertex
    pq.push({1, minPath[1]});

    while (!pq.empty()) {
        auto [u, dist] = pq.top();
        pq.pop();

        if (dist > minPath[u]) continue;

        for (auto [v, edgeCost] : graph[u]) {
            i64 newCost = dist + edgeCost + A[v]; // path cost to v via u

            if (newCost < minPath[v]) {
                minPath[v] = newCost;
                pq.push({v, newCost});
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << minPath[i] << " ";
    }
    cout << "\n";
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    solve();
    return 0;
}
