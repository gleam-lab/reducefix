```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],a[200005],n,m;
void dij(){
   priority_queue<pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>>pq;
	pq.push({a[1],1});
	while(!pq.empty()){
		auto [d,u] = pq.top(); pq.pop();
		if(d > dis[u]) continue;
		for(auto [v,w]:G[u]){
			if(dis[u] + w + a[v] < dis[v]){
				dis[v] = dis[u] + w + a[v];
				pq.push({dis[v],v});
			}
		}
	}
}

signed main() {
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=m;i++){
		int u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		G[u].push_back({v,w});
		G[v].push_back({u,w});
	}
	memset(dis,0x3f,sizeof dis);
	dis[1]=a[1];
	dij();
	for(int i=2;i<=n;i++){
		cout<<dis[i]<<' ';
	}
	cout<<"\n";
	return 0;
}
```

### Explanation of Changes:
1. **Replaced `queue` with `priority_queue`**: This change ensures that we always expand the least costly node first, which is essential for Dijkstra's algorithm to work correctly.
2. **Removed the `vis` array**: Since the graph is connected and we start from vertex 1, there's no need to keep track of visited nodes using an additional array. If a node is already processed, its distance will not be updated again because it won't have a shorter path.
3. **Corrected the initialization of `dis`**: Initially, all distances should be set to infinity except for the starting node (vertex 1). In the original code, this was done incorrectly using `memset`.
