```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],a[200005],n,m,vis[200005];
void dij(int u){
   priority_queue<pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>>q;
	q.push({0,u});
	while(!q.empty()){
		auto [d,u] = q.top(); q.pop();
		if(d > dis[u]) continue; // Skip outdated distances
		if(vis[u]) continue; // Avoid revisiting nodes
		vis[u] = 1;
		for(auto [v,w]:G[u]){
			if(dis[v] > d + a[v] + w){
				dis[v] = d + a[v] + w;
				q.push({dis[v],v});
			}
		}
	}
}

signed main() {
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%lld",&a[i]);
	}
	for(int i=1;i<=m;i++){
		int u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		G[u].push_back({v,w});
		G[v].push_back({u,w});
	}
	memset(dis,0x3f,sizeof dis);
	dis[1] = a[1]; // Initialize distance to source node
	dij(1);
	for(int i=2;i<=n;i++){
		cout<<dis[i]<<' ';
	}
	cout<<"\n";
	return 0;
}
```

### Explanation of Changes:
1. **Priority Queue**: Replaced `queue` with `priority_queue` to ensure efficient access to the smallest element, which is crucial for Dijkstra's algorithm to work correctly.
2. **Distance Check**: Added a check `if(d > dis[u]) continue;` before processing further to skip outdated distances, optimizing the algorithm.
3. **Visited Array Update**: Moved `vis[u] = 1;` inside the loop where the node is processed to avoid revisiting nodes multiple times unnecessarily.
