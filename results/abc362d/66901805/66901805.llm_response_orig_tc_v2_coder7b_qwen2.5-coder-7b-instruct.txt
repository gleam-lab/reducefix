const int MAXN = 2e5 + 5;
vector<pair<int, int>> G[MAXN];
long long dis[MAXN], dis2[MAXN], a[MAXN], n, m, vis[MAXN];

void dij(int u) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, u});
    while (!pq.empty()) {
        auto [d, v] = pq.top(); pq.pop();
        if (vis[v]) continue;
        vis[v] = 1;
        for (auto [to, w] : G[v]) {
            if (dis[to] > d + a[to] + w) {
                dis[to] = d + a[to] + w;
                pq.push({dis[to], to});
            }
        }
    }
}
