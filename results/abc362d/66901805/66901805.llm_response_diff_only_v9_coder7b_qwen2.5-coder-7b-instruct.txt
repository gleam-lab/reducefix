#include<bits/stdc++.h>
using namespace std;
#define int long long 
const int INF = LLONG_MAX / 2;
vector<pair<int,int>> G[200005];
int dis[200005], a[200005], n, m;

void dij(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});
    while (!pq.empty()) {
        auto [cost, u] = pq.top(); pq.pop();
        if (cost > dis[u]) continue; // Skip outdated paths
        for (auto [v, w] : G[u]) {
            int new_cost = cost + a[v] + w;
            if (new_cost < dis[v]) {
                dis[v] = new_cost;
                pq.push({new_cost, v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    memset(dis, 0x3f, sizeof(dis));
    dis[1] = a[1];
    dij(1);

    for (int i = 2; i <= n; i++) {
        printf("%lld ", dis[i]);
    }
    printf("\n");
    return 0;
}
