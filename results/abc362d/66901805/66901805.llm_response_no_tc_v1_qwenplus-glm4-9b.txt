#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int, int>> G[200005];
int dis[200005], dis2[200005], a[200005], n, m;
void dijkstra(int src) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({a[src], src});
    dis[src] = a[src];
    while (!pq.empty()) {
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();
        if (dis[u] != d) continue;
        for (auto &edge : G[u]) {
            int v = edge.first;
            int weight = edge.second;
            if (dis[v] > dis[u] + weight) {
                dis[v] = dis[u] + weight;
                pq.push({dis[v], v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    memset(dis, 0x3f, sizeof dis);
    dijkstra(1);

}

### Explanation
1. **Priority Queue**: Changed the data structure from a queue to a priority queue to efficiently get the next vertex with the smallest distance.
2. **Dijkstra's Algorithm**: Implemented Dijkstra's algorithm correctly using a priority queue to always expand the least costly path first.
3. **Initialization**: Initialized `dis[1]` with `a[1]` since the weight of reaching vertex 1 is its own weight.
4. **Correct Output**: Outputs the distances for all vertices from 2 to N, as required.
