#include <bits/stdc++.h>
using namespace std;
#define int long long

const int INF = 1e18;

template<typename T>
class Dijkstra {
public:
    using pii = pair<T, int>;
    vector<vector<pii>> graph;
    vector<T> dist;
    int n;

    Dijkstra(int _n) : n(_n), graph(n + 1), dist(n + 1, INF) {}

    void addEdge(int u, int v, T w) {
        graph[u].push_back({w, v});
        graph[v].push_back({w, u}); // Remove this for directed graphs
    }

    void run(int s) {
        priority_queue<pii, vector<pii>, greater<pii>> pq;
        dist[s] = 0;
        pq.push({0, s});

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (dist[u] < d) continue;
            for (auto [v, w] : graph[u]) {
                if (dist[v] > dist[u] + w) {
                    dist[v] = dist[u] + w;
                    pq.push({dist[v], v});
                }
            }
        }
    }
};

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n, m;
    cin >> n >> m;
    vector<int> A(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> A[i];
    }

    Dijkstra<int> dijkstra(n);
    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        // Edge weight from u to v and vice versa is b
        dijkstra.addEdge(u, v, b);
    }

    // The total path weight includes the starting vertex's value
    // So we initialize distance to A[1], and modify the algorithm to include vertex weights
    dijkstra.dist[1] = A[1];
    
    // Custom Dijkstra implementation to handle vertex weights
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({dijkstra.dist[1], 1});

    while (!pq.empty()) {
        auto [curr_dist, u] = pq.top();
        pq.pop();
        
        if (curr_dist > dijkstra.dist[u]) continue;
        
        for (auto [w, v] : dijkstra.graph[u]) {
            int new_dist = curr_dist + w + A[v]; // Add edge weight and destination vertex weight
            
            if (new_dist < dijkstra.dist[v]) {
                dijkstra.dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << dijkstra.dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
