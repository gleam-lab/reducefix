#include<bits/stdc++.h>
using namespace std;
#define int long long

const int MAXN = 200005;
const int INF = 1e18;

vector<pair<int, int>> G[MAXN];
int dis[MAXN];
int a[MAXN];
bool vis[MAXN];

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    dis[start] = a[start];
    pq.push({dis[start], start});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (vis[u]) continue;
        vis[u] = true;

        for (auto [v, w] : G[u]) {
            int newDist = d + w + a[v];
            if (newDist < dis[v]) {
                dis[v] = newDist;
                pq.push({dis[v], v});
            }
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    for (int i = 1; i <= m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    fill(dis, dis + n + 1, INF);
    fill(vis, vis + n + 1, false);

    dijkstra(1);

    for (int i = 2; i <= n; i++) {
        cout << dis[i] << " ";
    }
    cout << "\n";

    return 0;
}
