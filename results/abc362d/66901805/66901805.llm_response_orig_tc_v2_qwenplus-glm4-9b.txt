#include <bits/stdc++.h>
using namespace std;
#define INF 1e18
#define int long long

vector<pair<int, int>> G[200005];
int n, m;
int dist[200005];

void dijkstra(int source) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, source});
    dist[source] = 0;

    while (!pq.empty()) {
        int current = pq.top().second;
        int currentDist = pq.top().first;
        pq.pop();

        if (currentDist > dist[current]) {
            continue;
        }

        for (auto &edge : G[current]) {
            int next = edge.first;
            int weight = edge.second + G[next][0].second; // assuming an initial edge weight of A[next]
            
            if (dist[next] > currentDist + weight) {
                dist[next] = currentDist + weight;
                pq.push({dist[next], next});
            }
        }
    }
}

    int A[n + 1];

It also considers the vertex weights by subtracting them from the path weights when outputting the result.
