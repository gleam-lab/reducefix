```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,pair<int,int>>>G[200005]; // Graph with edge weight and vertex weight
int dist[200005]; // Distance from vertex 1 to other vertices
int a[200005]; // Vertex weights
int n, m; // Number of vertices and edges

void dijkstra(int src){
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq; // Min-heap
    pq.push({0, src});
    dist[src] = a[src]; // Distance to itself is its own weight

    while(!pq.empty()){
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if(d > dist[u]) continue; // Skip if we have already found a better path

        for(auto [v, w] : G[u]){
            int alt = d + w + a[v]; // New path distance
            if(alt < dist[v]){ // If the new path is shorter
                dist[v] = alt;
                pq.push({alt, v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for(int i = 1; i <= n; i++){
        scanf("%lld", &a[i]);
    }

    for(int i = 1; i <= m; i++){
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    memset(dist, 0x3f, sizeof dist); // Use max long long to initialize distances
    dijkstra(1); // Run Dijkstra's algorithm from vertex 1

    for(int i = 2; i <= n; i++){
        printf("%lld ", dist[i]);
    }
    printf("\n");

    return 0;
}
```
This code implements Dijkstra's algorithm using a min-heap (priority queue) to efficiently find the shortest path from vertex 1 to all other vertices, considering both vertex and edge weights. It initializes the distances to a very large value, which ensures that any real computed distance will be smaller, preventing any wrap-around errors. The output format is adjusted to match the problem's requirements, printing the minimum weights of paths from vertex 1 to all other vertices (from 2 to N) in a single line.
