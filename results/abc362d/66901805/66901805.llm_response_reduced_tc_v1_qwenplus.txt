#include<bits/stdc++.h>
using namespace std;
#define int long long

vector<pair<int,int>> G[200005];
int dis[200005];
int a[200005];
int n, m;

void dij() {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    memset(dis, 0x3f, sizeof(dis));
    dis[1] = a[1];
    pq.push({dis[1], 1});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dis[u]) continue;

        for (auto [v, w] : G[u]) {
            int new_cost = dis[u] + w + a[v];
            if (new_cost < dis[v]) {
                dis[v] = new_cost;
                pq.push({dis[v], v});
            }
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    dij();

    for (int i = 2; i <= n; i++) {
        cout << dis[i] << " ";
    }
    cout << "\n";

    return 0;
}
