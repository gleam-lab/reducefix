```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],a[200005],n,m;
const int INF = LLONG_MAX;

void dij(int s) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, s});
    dis[s] = 0;
    while (!pq.empty()) {
        auto [dist, u] = pq.top(); pq.pop();
        if (dist > dis[u]) continue;
        for (auto [v, w] : G[u]) {
            long long new_dist = dist + a[v] + w;
            if (new_dist < dis[v]) {
                dis[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    memset(dis, 0x3f, sizeof dis);
    dij(1);
    for (int i = 2; i <= n; i++) {
        printf("%lld ", dis[i]);
    }
    cout << "\n";
    return 0;
}
```

This solution uses Dijkstra's algorithm to find the shortest paths from the source node (vertex 1) to all other nodes in the graph. It initializes the distance array with infinity except for the starting node, which is set to zero. The algorithm iteratively selects the node with the smallest tentative distance, updates the distances of its neighbors, and adds them to the priority queue if they have not been visited yet. This ensures that each node is processed only once, resulting in an efficient O(E log V) time complexity, where E is the number of edges and V is the number of vertices.
