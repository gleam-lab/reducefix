#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],a[200005],n,m;
bool vis[200005];

// Relaxation function to update distances
void relax(int u, int v, int w) {
    if(dis[u] + w + a[v] < dis[v]) {
        dis[v] = dis[u] + w + a[v];
    }
}

// Dijkstra's algorithm modified to handle negative weights correctly
void dij(int s) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    memset(dis, 0x3f, sizeof dis); // Initialize distances to infinity
    dis[s] = a[s]; // Distance to start node is its own value
    pq.push({dis[s], s}); // Push start node into the priority queue

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        if (vis[u]) continue; // Skip already visited nodes
        vis[u] = true;

        for (auto [v, w] : G[u]) {
            relax(u, v, w); // Relax all outgoing edges from u
            pq.push({dis[v], v}); // Push updated distance to v into the priority queue
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    dij(1); // Run Dijkstra's algorithm starting from vertex 1

    for (int i = 2; i <= n; i++) {
        cout << dis[i] << ' '; // Print minimum distance to each vertex except 1
    }
    cout << "\n";

    return 0;
}
