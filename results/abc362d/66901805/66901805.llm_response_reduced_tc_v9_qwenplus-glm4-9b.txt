#include <bits/stdc++.h>
using namespace std;
#define int long long
#define LLONG_MAX 9223372036854775807LL

vector<pair<int, int>> G[200005];
int dist[200005], a[200005], n, m;
bool visited[200005];

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});
    dist[start] = a[start];

    while (!pq.empty()) {
        int cur_weight = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (visited[u]) continue;
        visited[u] = true;

        for (auto &edge : G[u]) {
            int v = edge.first;
            int weight = edge.second;
            if (dist[u] != LLONG_MAX && dist[v] > dist[u] + weight + a[v]) {
                dist[v] = dist[u] + weight + a[v];
                pq.push({dist[v], v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    memset(dist, LLONG_MAX, sizeof(dist));
    dijkstra(1);

    for (int i = 2; i <= n; i++) {
        cout << dist[i] << ' ';
    }
    cout << '\n';

    return 0;
}
