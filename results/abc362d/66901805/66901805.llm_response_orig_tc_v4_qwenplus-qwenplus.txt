#include <bits/stdc++.h>
using namespace std;

#define int long long

const int INF = 1e18;

vector<vector<pair<int, int>>> graph(200005); // graph[u] = list of (v, B_uv)
int dist[200005];                           // Minimum distance from node 1 to each node
bool visited[200005];
int A[200005];

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    dist[start] = A[start];
    pq.push({dist[start], start});

    while (!pq.empty()) {
        int u = pq.top().second;
        if (visited[u]) {
            pq.pop();
            continue;
        }
        visited[u] = true;

        for (auto [v, cost] : graph[u]) {
            if (!visited[v] && dist[v] > dist[u] + A[v] + cost) {
                dist[v] = dist[u] + A[v] + cost;
                pq.push({dist[v], v});
            }
        }
        pq.pop();
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
        dist[i] = INF;
    }

    for (int i = 1; i <= M; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        graph[u].push_back({v, b});
        graph[v].push_back({u, b});
    }

    dijkstra(1);

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
