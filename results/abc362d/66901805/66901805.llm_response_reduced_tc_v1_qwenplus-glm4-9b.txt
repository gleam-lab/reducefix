#include <bits/stdc++.h>
using namespace std;
#define int long long
#define INF LLONG_MAX

vector<pair<int, int>> G[200005]; // Adjacency list for the graph
int dist[200005]; // Distance from vertex 1 to all other vertices
int weight[200005]; // Weight of each vertex

void Dijkstra(int src) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, src});
    dist[src] = weight[src]; // Initialize distance to vertex's own weight

    while (!pq.empty()) {
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (dist[u] < d) continue; // If already processed with shorter distance, skip

        for (auto &edge : G[u]) {
            int v = edge.first;
            int w = edge.second;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}

signed main() {
    int n, m;
    scanf("%d %d", &n, &m);

    for (int i = 1; i <= n; i++) {
        scanf("%lld", &weight[i]);
    }

    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    fill(dist, dist + n + 1, INF); // Initialize distances with infinity

    Dijkstra(1); // Run Dijkstra's algorithm starting from vertex 1

    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
