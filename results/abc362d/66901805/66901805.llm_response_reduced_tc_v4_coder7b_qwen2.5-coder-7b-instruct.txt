void dij(int start) {
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    pq.push({0, start});
    dis[start] = 0;

    while (!pq.empty()) {
        auto [dist, u] = pq.top();
        pq.pop();

        if (dist > dis[u]) continue;

            if (new_dist < dis[v]) {

                pq.push({new_dist, v});
            }
        }
    }
}
