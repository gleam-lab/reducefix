```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, int>;
const ll inf = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    vector<vector<pair<int, ll>>> G(N);

    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    for (int i = 0; i < M; i++) {
        int u, v;
        ll b;
        cin >> u >> v >> b;
        u--; v--;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }

    vector<ll> dist(N, inf);
    priority_queue<P, vector<P>, greater<P>> pq;

    // Start from vertex 0 (vertex 1 in 1-indexed)
    // Initial cost is just the vertex weight of start
    dist[0] = A[0];
    pq.push({dist[0], 0});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d != dist[u]) continue;

        for (auto [v, w] : G[u]) {
            ll new_cost = dist[u] + w + A[v];
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({new_cost, v});
            }
        }
    }

    for (int i = 1; i < N; i++) {
        cout << dist[i];
        if (i < N - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
```
