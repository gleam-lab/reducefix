#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll inf = 1000000000;

int main() {
  ll N, M;
  cin >> N >> M;
  
  vector<ll> A(N);
  for (ll i = 0; i < N; ++i) {
    cin >> A[i];
  }
  
  vector<vector<pair<ll, ll>>> G(N);
  for (ll i = 0; i < M; ++i) {
    ll u, v, b;
    cin >> u >> v >> b;
    --u;
    --v;
    G[u].push_back({v, b});
    G[v].push_back({u, b});
  }
  
  vector<ll> dist(N, inf);
  dist[0] = A[0];
  
  // Priority queue to perform Dijkstra's algorithm
  priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
  pq.push({A[0], 0});
  
  while (!pq.empty()) {
    ll current_dist = pq.top().first;
    ll current_vertex = pq.top().second;
    pq.pop();
    
    if (current_dist > dist[current_vertex]) continue;
    
    for (auto &edge : G[current_vertex]) {
      ll next_vertex = edge.first;
      ll weight = edge.second;
      ll new_dist = current_dist + weight + A[next_vertex];
      
      if (new_dist < dist[next_vertex]) {
        dist[next_vertex] = new_dist;
        pq.push({new_dist, next_vertex});
      }
    }
  }
  
  // Output the minimum weights for paths from vertex 1 to all other vertices
  for (ll i = 1; i < N; ++i) {
    cout << dist[i] << " ";
  }
  cout << endl;
  
  return 0;
}
