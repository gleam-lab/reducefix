void dijkstra(const vector<vector<P>>& g, int s, VL& dist) {

    dist[s] = 0;
    pq.emplace(0, s);

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;

            if (dist[v] > dist[u] + w) {

                pq.emplace(dist[v], v);

        }
    }
}
