The main issue with the original code is that it uses BFS (queue) for a shortest path problem with weighted edges, but this only works for unweighted graphs. Since edge weights are non-negative, we need to use Dijkstra's algorithm with a priority queue.

Here's the fixed code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N);
    vector<vector<P>> G(N);
    
    for(int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    for(int i = 0; i < M; i++) {
        int u, v;
        ll b;
        cin >> u >> v >> b;
        u--; v--;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }
    
    vector<ll> ans(N, LLONG_MAX);
    priority_queue<P, vector<P>, greater<P>> pq;
    
    // Start from vertex 0 (vertex 1 in 1-indexed)
    ans[0] = A[0];  // Include starting vertex weight
    pq.push({A[0], 0});
    
    while(!pq.empty()) {
        auto [cost, u] = pq.top();
        pq.pop();
        
        if(cost > ans[u]) continue;
        
        for(auto [v, w] : G[u]) {
            ll new_cost = cost + w + A[v];
            if(new_cost < ans[v]) {
                ans[v] = new_cost;
                pq.push({new_cost, v});
            }
        }
    }
    
    for(int i = 1; i < N; i++) {
        cout << ans[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
