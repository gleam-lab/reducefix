```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<int, int>;
const ll inf = 1000000000;

struct Edge {
    int to;
    ll cost;
};

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    vector<vector<Edge>> G(N);

    for(int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    for(int i = 0; i < M; ++i) {
        int u, v;
        ll b;
        cin >> u >> v >> b;
        --u; --v;
        G[u].emplace_back(v, b);
        G[v].emplace_back(u, b);
    }

    vector<ll> dist(N, LLONG_MAX);
    priority_queue<P, vector<P>, greater<>> pq;
    dist[0] = 0;
    pq.emplace(0, A[0]);

    while(!pq.empty()) {
        auto [cost, node] = pq.top(); pq.pop();

        if(cost > dist[node]) continue;

        for(const auto& e : G[node]) {
            ll next_cost = cost + e.cost + A[e.to];
            if(next_cost < dist[e.to]) {
                dist[e.to] = next_cost;
                pq.emplace(next_cost, A[e.to]);
            }
        }
    }

    for(int i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
```

This solution uses Dijkstra's algorithm to find the shortest path from vertex 1 to all other vertices in the graph. It handles the constraints efficiently using a priority queue to always expand the least costly path first. The time complexity is \(O((N + M) \log N)\), which is efficient enough for the given problem constraints.
