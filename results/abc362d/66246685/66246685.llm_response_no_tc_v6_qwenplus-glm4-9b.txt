#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<ll, ll>;  // Use long long for path weights
const ll inf = 1000000000;

struct Edge {
    int to, weight;
};

int main() {
    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N);
    vector<vector<Edge>> G(N);
    vector<ll> dist(N, inf);  // Initialize distances to a large number
    
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    for (int i = 0; i < M; ++i) {
        int u, v, weight;
        cin >> u >> v >> weight;
        --u; --v;  // Convert to zero-indexed
        G[u].emplace_back(v, weight);
        G[v].emplace_back(u, weight);
    }
    
    // Use Dijkstra's algorithm to find the shortest path from 0 to all other vertices
    priority_queue<P, vector<P>, greater<P>> pq;  // Min-heap for the priority queue
    pq.push({0, A[0]});  // Start from vertex 0 with initial weight A[0]
    dist[0] = A[0];  // Set the distance to the starting vertex
    
    while (!pq.empty()) {
        ll current = pq.top().first;
        ll d = pq.top().second;
        pq.pop();
        
        if (d > dist[current]) continue;  // Skip if we already found a better path
        
        for (auto &edge : G[current]) {
            ll next = edge.to;
            ll weight = edge.weight;
            ll nextDist = d + weight + A[next];  // Vertex weight + edge weight
            
            if (nextDist < dist[next]) {  // Found a better path
                dist[next] = nextDist;
                pq.push({next, nextDist});
            }
        }
    }
    
    // Output the distances from vertex 1 to all other vertices
    for (ll i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
