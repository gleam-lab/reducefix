```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<int, int>;
const ll inf = 1e18;

void dijkstra(const vector<vector<P>>& g, const VL& a, int s, VL& dist) {
    int n = g.size();
    priority_queue<P, vector<P>, greater<>> pq;
    pq.emplace(a[s], s);
    dist[s] = a[s];

    while (!pq.empty()) {
        auto [cost, u] = pq.top(); pq.pop();
        if (dist[u] < cost) continue;
        
        for (auto [v, d] : g[u]) {
            if (dist[v] > dist[u] + d + a[v]) {
                dist[v] = dist[u] + d + a[v];
                pq.emplace(dist[v], v);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N), B(M), ans(N - 1);
    vector<vector<pair<ll, ll>>> G(N);
    
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v >> B[i];
        --u; --v;
        G[u].emplace_back(v, B[i]);
        G[v].emplace_back(u, B[i]);
    }

    for (int i = 1; i < N; ++i) {
        VL dist(N, inf);
        dijkstra(G, A, i, dist);
        ans[i - 1] = dist[0];
    }

    for (int i = 0; i < N - 1; ++i) {
        cout << ans[i] << " ";
    }
    cout << endl;

    return 0;
}
```

### Explanation:
1. **Graph Representation**: The graph is represented using an adjacency list where each node stores pairs of adjacent nodes and their respective edge weights.
2. **Dijkstra's Algorithm**: For each vertex \(i\) starting from 1 to \(N-1\), we run Dijkstra's algorithm to find the shortest path from vertex 1 to vertex \(i\).
3. **Distance Calculation**: We maintain a distance array `dist` where `dist[j]` represents the minimum distance from vertex 1 to vertex \(j\). Initially, all distances are set to infinity except for the source vertex (vertex 1).
4. **Priority Queue**: We use a priority queue to efficiently get the next closest vertex to process.
5. **Edge Relaxation**: During the traversal, we update the distance array whenever we find a shorter path to a vertex.
6. **Output**: Finally, we print the results for each \(i\) from 2 to \(N\).

This approach ensures that we correctly compute the minimum weight path from vertex 1 to each other vertex efficiently within the constraints provided.
