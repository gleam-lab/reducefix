#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<ll, ll>;
const ll inf = 1000000000;

int main() {
  ll N, M;
  cin >> N >> M;
  
  VL A(N), B(M), ans(N, inf);
  VL sum(N, 0);
  vector<vector<pair<ll, ll>>> G(N);
  
  for (ll i = 0; i < N; i++) {
    cin >> A[i];
  }
  
  for (ll i = 0; i < M; i++) {
    ll u, v, b;
    cin >> u >> v >> b;
    u--; v--;  // vertex indices are 0-based
    G[u].push_back({v, b});
    G[v].push_back({u, b});
  }
  
  // Use a modified Dijkstra's algorithm to find the minimum path sum
  priority_queue<P, VL, greater<P>> pq;
  pq.push({A[0], 0});  // We start from vertex 0 (1 in 1-based index)
  ans[0] = A[0];
  
  while (!pq.empty()) {
    ll u = pq.top().second;
    ll dist = pq.top().first;
    pq.pop();
    
    if (dist > ans[u]) continue;
    
    for (auto &neigh : G[u]) {
      ll v = neigh.first;
      ll weight = neigh.second;
      ll newDist = dist + weight + A[v];
      
      if (newDist < ans[v]) {
        ans[v] = newDist;
        pq.push({newDist, v});
      }
    }
  }
  
  for (ll i = 1; i < N; i++) {
    cout << ans[i] << (i < N - 1 ? " " : "");
  }
  cout << endl;
  
  return 0;
}
