        G[u].emplace_back(v, b + A[v]);

    pq.emplace(A[0], 0);

    while (!pq.empty()) {
        auto [cost, u] = pq.top(); pq.pop();
        if (seen[u]) continue;
        seen[u] = true;

        for (auto [v, d] : G[u]) {
            if (seen[v]) continue;
            ans[v] = min(ans[v], cost + d);
            pq.emplace(cost + d, v);
        }
