#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<ll, ll>; // Use long long for pair to handle large weights correctly

const ll inf = 1e18; // Use a large constant to represent infinity

int main() {
  ll N, M;
  cin >> N >> M;

  vector<ll> A(N, 0);
  vector<vector<pair<ll, ll>>> G(N);
  vector<ll> dist(N, inf);

  for (ll i = 0; i < N; i++) {
    cin >> A[i];
  }

  for (ll i = 0; i < M; i++) {
    ll u, v, b;
    cin >> u >> v >> b;
    u--; v--;
    G[u].push_back({v, b});
    G[v].push_back({u, b});
  }

  // Dijkstra's algorithm to find the shortest path from vertex 1 to all other vertices
  priority_queue<P, vector<P>, greater<P>> pq;
  pq.push({A[0], 0}); // Start from vertex 1 with weight equal to its vertex weight
  dist[0] = A[0];

  while (!pq.empty()) {
    ll u = pq.top().second;
    ll d = pq.top().first;
    pq.pop();

    if (d > dist[u]) continue;

    for (auto &p : G[u]) {
      ll v = p.first;
      ll weight = p.second;
      ll new_dist = d + A[v] + weight;

      if (new_dist < dist[v]) {
        dist[v] = new_dist;
        pq.push({new_dist, v});
      }
    }
  }

  // Output the results for all vertices from 2 to N-1
  for (ll i = 1; i < N; i++) {
    cout << dist[i] << " ";
  }

  return 0;
}
