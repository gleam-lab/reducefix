#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

const ll INF = LLONG_MAX/2;

vector<ll> dijkstra(const vector<vector<pii>>& adj, int start, const vector<ll>& weights) {
    int n = adj.size();
    vector<ll> dist(n, INF);
    dist[start] = 0;
    priority_queue<pll, vector<pll>, greater<pll>> pq;
    pq.push({0, start});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        
        if (d > dist[u]) continue;
        
        for (auto [v, w] : adj[u]) {
            ll new_dist = d + w + weights[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
    
    return dist;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    int n, m; cin >> n >> m;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    vector<vector<pii>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, b; cin >> u >> v >> b;
        --u; --v;
        adj[u].emplace_back(v, b);
        adj[v].emplace_back(u, b);
    }
    
    vector<ll> res = dijkstra(adj, 0, a);
    
    for (int i = 1; i < n; ++i) cout << res[i] << " ";
    cout << "\n";
    
    return 0;
}
