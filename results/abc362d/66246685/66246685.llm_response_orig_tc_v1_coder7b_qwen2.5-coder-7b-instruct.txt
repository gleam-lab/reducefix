#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<int, int>;
const ll inf = 1e18;

struct Edge {
    int to;
    ll cost;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N), ans(N, LLONG_MAX);
    vector<vector<Edge>> G(N);

    for (int i = 0; i < N; ++i) {
        cin >> A.at(i);
    }

    for (int i = 0; i < M; ++i) {
        int u, v;
        ll b;
        cin >> u >> v >> b;
        u--;
        v--;
        G.at(u).emplace_back(v, b + A.at(v));
        G.at(v).emplace_back(u, b + A.at(u));
    }

    vector<bool> seen(N, false);
    priority_queue<P, vector<P>, greater<>> pq;
    pq.emplace(A.at(0), 0);

    while (!pq.empty()) {
        auto [dist, u] = pq.top();
        pq.pop();

        if (seen.at(u)) continue;
        seen.at(u) = true;

        ans.at(u) = min(ans.at(u), dist);

        for (auto [v, d] : G.at(u)) {
            if (seen.at(v)) continue;
            pq.emplace(dist + d, v);
        }
    }

    for (int i = 1; i < N; ++i) {
        cout << ans.at(i) << " ";
    }
    cout << endl;

    return 0;
}
