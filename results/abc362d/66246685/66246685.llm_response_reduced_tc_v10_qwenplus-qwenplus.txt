```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>; // (distance, node)

const ll INF = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<vector<P>> G(N); // graph: list of (neighbor, edge weight)
    for (ll i = 0; i < M; ++i) {
        ll u, v, b;
        cin >> u >> v >> b;
        --u; --v;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }

    priority_queue<P, vector<P>, greater<>> pq; // min-heap (distance, node)
    vector<ll> dist(N, INF);

    // Start from node 0 with its weight
    pq.push({A[0], 0});
    dist[0] = A[0];

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto [v, b] : G[u]) {
            ll new_dist = d + b + A[v]; // path cost to v via u
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    // Output results for nodes 1 to N-1 (2nd to Nth vertices)
    for (ll i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
