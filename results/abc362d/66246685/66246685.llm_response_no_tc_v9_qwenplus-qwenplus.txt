```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>; // (distance, node)

const ll INF = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<vector<P>> graph(N);
    for (ll i = 0; i < M; ++i) {
        ll u, v, b;
        cin >> u >> v >> b;
        --u; --v;
        graph[u].push_back({v, b});
        graph[v].push_back({u, b});
    }

    priority_queue<P, vector<P>, greater<P>> pq;
    vector<ll> dist(N, INF);
    
    // Start from node 0 with cost A[0] (starting node's weight)
    pq.emplace(A[0], 0);
    dist[0] = A[0];

    while (!pq.empty()) {
        auto [cost, u] = pq.top();
        pq.pop();

        if (cost > dist[u]) continue;

        for (auto [v, edge_cost] : graph[u]) {
            ll new_cost = cost + edge_cost + A[v]; // Add vertex weight only once when entering the node
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.emplace(new_cost, v);
            }
        }
    }

    for (ll i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << '\n';

    return 0;
}
```
