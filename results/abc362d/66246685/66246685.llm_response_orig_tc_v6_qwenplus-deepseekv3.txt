```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>;
const ll INF = LLONG_MAX;

int main() {
    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    vector<vector<P>> G(N);
    for (int i = 0; i < M; ++i) {
        ll u, v, b;
        cin >> u >> v >> b;
        u--; v--;
        G[u].emplace_back(v, b);
        G[v].emplace_back(u, b);
    }
    
    vector<ll> dist(N, INF);
    dist[0] = A[0];
    priority_queue<P, vector<P>, greater<P>> pq;
    pq.emplace(dist[0], 0);
    
    while (!pq.empty()) {
        ll current_dist = pq.top().first;
        ll u = pq.top().second;
        pq.pop();
        
        if (current_dist > dist[u]) continue;
        
        for (auto &edge : G[u]) {
            ll v = edge.first;
            ll b = edge.second;
            ll new_dist = current_dist + b + A[v];
            
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.emplace(new_dist, v);
            }
        }
    }
    
    for (int i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```

2. **Graph Representation**: The graph is represented as an adjacency list `G`, where each vertex points to a list of pairs (neighbor vertex, edge weight).
3. **Dijkstra's Algorithm**: 
   - A priority queue is used to always process the vertex with the current smallest distance.
   - The distance array `dist` is initialized with `INF` (a very large number), except for the starting vertex (vertex 0, 1-based vertex 1), which is set to its own weight `A[0]`.
   - For each vertex pulled from the priority queue, the algorithm checks all adjacent vertices. If a shorter path is found (current distance + edge weight + adjacent vertex weight), the distance is updated, and the adjacent vertex is added to the priority queue.
4. **Output**: The distances from vertex 1 to all other vertices are printed, starting from vertex 2 (1-based).
