#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>; // (current cost, node)

const ll INF = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<vector<P>> G(N); // Adjacency list: G[u] contains pairs (v, edge_weight)
    for (ll i = 0; i < M; ++i) {
        ll u, v, b;
        cin >> u >> v >> b;
        --u; --v;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }

    vector<ll> dist(N, INF);
    dist[0] = A[0]; // Start from node 0 with its vertex weight

    // Min-heap priority queue for Dijkstra's algorithm
    priority_queue<P, vector<P>, greater<P>> pq;
    pq.push({dist[0], 0}); // (cost to reach node, node)

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue; // Skip outdated entries

        for (auto [v, b] : G[u]) {
            ll new_cost = dist[u] + b + A[v]; // Cost to reach v via u
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({dist[v], v});
            }
        }
    }

    // Output results for nodes 2 to N
    for (ll i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
