The issue with the original code is that it uses BFS (queue) instead of Dijkstra's algorithm for finding shortest paths in a weighted graph. Since edge weights can be large and non-negative, we need to use Dijkstra's algorithm with a priority queue to ensure we always process the node with the minimum distance first.

Here's the corrected code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<ll, ll>;

int main() {
    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N);
    vector<vector<P>> G(N);
    
    for(int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    for(int i = 0; i < M; i++) {
        int u, v;
        ll b;
        cin >> u >> v >> b;
        u--; v--;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }
    
    vector<ll> dist(N, LLONG_MAX);
    priority_queue<P, vector<P>, greater<P>> pq;
    
    dist[0] = A[0];
    pq.push({A[0], 0});
    
    while(!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if(d > dist[u]) continue;
        
        for(auto [v, w] : G[u]) {
            ll new_dist = dist[u] + w + A[v];
            if(new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
    
    for(int i = 1; i < N; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
