#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<ll, ll>;
const ll inf = 1000000000;

int main() {
  ll N, M;
  cin >> N >> M;

  VL A(N), B(M), ans(N, LLONG_MAX);
  VL dist(N, LLONG_MAX);
  vector<VL> G(N);

  for (ll i = 0; i < N; ++i) {
    cin >> A[i];
  }

  for (ll i = 0; i < M; ++i) {
    ll u, v, b;
    cin >> u >> v >> b;
    u--; v--;
    G[u].push_back(v);
    G[v].push_back(u);
    B[i] = b;
  }

  // Start BFS from vertex 1 (index 0)
  queue<ll> q;
  q.push(0);
  dist[0] = A[0];

  while (!q.empty()) {
    ll v = q.front();
    q.pop();

    for (ll u : G[v]) {
      ll edgeWeight = B[v * N + u]; // Edge weight between v and u

      // Check if the new path through (v, u) is better than the known path to u
      if (dist[v] + edgeWeight < dist[u]) {
        dist[u] = dist[v] + edgeWeight;
        q.push(u);
      }
    }
  }

  // Collect the minimum path weights for vertices 2 to N
  for (ll i = 1; i < N; ++i) {
    ans[i] = dist[i];
  }

  // Output the results for vertices 2 to N
  for (ll i = 1; i < N; ++i) {
    cout << ans[i] << (i < N - 1 ? " " : "");
  }

  return 0;
}
