#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<ll, ll>;
const ll inf = 1000000000LL;

int main() {
  ll N, M;
  cin >> N >> M;
  
  vector<ll> A(N), B(M), ans(N, LLONG_MAX);
  vector<vector<pair<ll, ll>>> G(N);
  
  for (ll i = 0; i < N; i++) {
    cin >> A[i];
  }
  
  for (ll i = 0; i < M; i++) {
    ll u, v, b;
    cin >> u >> v >> b;
    u--; v--;  // Convert to zero-based index
    G[u].push_back({v, b});
    G[v].push_back({u, b});
  }
  
  // Use Dijkstra's algorithm with a min-heap to find the shortest path from vertex 0 to all other vertices
  auto comp = [&](const P& a, const P& b) {
    return a.first > b.first;
  };
  priority_queue<P, vector<P>, decltype(comp)> pq(comp);
  
  pq.push({A[0], 0});  // Push the initial vertex with its value and zero-based index
  
  while (!pq.empty()) {
    ll current_cost = pq.top().first;
    ll current_vertex = pq.top().second;
    pq.pop();
    
    if (current_vertex >= N) continue;  // Skip processed vertices
    
    for (auto& [neighbor, weight] : G[current_vertex]) {
      // Calculate new cost to reach the neighbor
      ll new_cost = current_cost + weight + A[neighbor];
      if (new_cost < ans[neighbor]) {
        ans[neighbor] = new_cost;
        pq.push({new_cost, neighbor});
      }
    }
  }
  
  // Output the minimum weights for paths to all vertices except the first one
  for (ll i = 1; i < N; i++) {
    cout << ans[i] << " ";
  }
  cout << endl;
  
  return 0;
}
