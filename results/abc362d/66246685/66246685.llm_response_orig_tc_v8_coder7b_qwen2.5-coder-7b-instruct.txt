#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VL = vector<ll>;
using P = pair<int, int>;
const ll inf = 1000000000;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  ll N, M;
  cin >> N >> M;

  vector<ll> A(N), B(M);
  vector<vector<pair<ll, ll>>> G(N);
  vector<bool> seen(N, false);
  
  for (ll i = 0; i < N; ++i) {
    cin >> A[i];
  }

  for (ll i = 0; i < M; ++i) {
    ll u, v, w;
    cin >> u >> v >> w;
    --u, --v;
    G[u].emplace_back(v, w);
    G[v].emplace_back(u, w);
  }

  vector<ll> dist(N, inf);
  priority_queue<P, vector<P>, greater<>> pq;
  
  dist[0] = A[0];
  pq.emplace(dist[0], 0);

  while (!pq.empty()) {
    auto [d, u] = pq.top(); pq.pop();

    if (seen[u]) continue;
    seen[u] = true;

    for (auto [v, w] : G[u]) {
      if (dist[v] > d + w + A[v]) {
        dist[v] = d + w + A[v];
        pq.emplace(dist[v], v);
      }
    }
  }

  for (ll i = 1; i < N; ++i) {
    cout << dist[i] << " ";
  }

  return 0;
}
