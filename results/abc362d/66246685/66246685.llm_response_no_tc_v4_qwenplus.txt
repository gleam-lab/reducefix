```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll LLONG_MAX = 1e18;

int main() {
    ll N, M;
    cin >> N >> M;
    
    vector<ll> A(N);
    vector<vector<pair<ll, ll>>> G(N);
    
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    for (int i = 0; i < M; i++) {
        int u, v, b;
        cin >> u >> v >> b;
        u--;
        v--;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }
    
    vector<ll> dist(N, LLONG_MAX);
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
    
    // Start from vertex 0 (vertex 1 in 1-indexing)
    // The path weight includes the starting vertex's weight
    dist[0] = A[0];
    pq.push({dist[0], 0});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (d > dist[u]) continue;
        
        for (auto [v, edge_weight] : G[u]) {
            ll new_cost = dist[u] + edge_weight + A[v];
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({new_cost, v});
            }
        }
    }
    
    for (int i = 1; i < N; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
