typedef long long ll;

    ll u, v, w;
};

bool comp(const Edge &a, const Edge &b) {
    return a.w < b.w;
}

ll parent[200005], rank[200005];

ll find(ll x) {
    if (parent[x] != x) parent[x] = find(parent[x]);
    return parent[x];
}

void merge(ll x, ll y) {
    x = find(x), y = find(y);
    if (rank[x] > rank[y]) parent[y] = x;
    else parent[x] = y;
    if (rank[x] == rank[y]) rank[y]++;
}
