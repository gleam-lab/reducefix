```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < n; ++i)
#define repr(i, n) for(int i = n-1; i >= 0; --i)
#define all(x) (x).begin(),(x).end()
#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define sz(x) ((int)(x).size())
#define endl '\n'

typedef long long ll;
typedef pair<int,int> P;

const ll INF = 1e18;

struct Edge{
    int to, cost;
    Edge(int to, int cost) : to(to), cost(cost) {}
};

vector<Edge> G[200005];
vector<ll> dis(200005, INF);

void dijkstra(int s){
    priority_queue<P,vector<P>,greater<P>> que;
    que.push(P(0,s));
    dis[s]=0;
    while(!que.empty()){
        P p = que.top(); que.pop();
        int v=p.se, d=p.fi;
        if(d > dis[v]) continue;
        for(auto e:G[v]){
            if(dis[e.to] > dis[v]+e.cost){
                dis[e.to] = dis[v]+e.cost;
                que.push(P(dis[e.to],e.to));
            }
        }
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n,m;
    cin >> n >> m;
    vector<int> a(n);
    rep(i,n) cin >> a[i];
    
    rep(i,m){
        int u,v,b;
        cin >> u >> v >> b;
        --u; --v;
        G[u].pb(Edge(v,b));
        G[v].pb(Edge(u,b));
    }
    
    vector<ll> ans(n-1);
    rep(i,n-1){
        ans[i]=INF;
    }
    
    rep(i,n){
        dijkstra(i);
        rep(j,n){
            if(j==i) continue;
            ans[min(i,j)-1]=min(ans[min(i,j)-1],dis[j]+a[j]);
        }
    }
    
    rep(i,n-1){
        cout << ans[i] << ' ';
    }
    cout << endl;
    
    return 0;
}
```
This solution uses Dijkstra's algorithm to find the shortest path from the starting vertex to all other vertices in the graph. It then calculates the minimum weight of a path from vertex 1 to every other vertex and prints them out.
