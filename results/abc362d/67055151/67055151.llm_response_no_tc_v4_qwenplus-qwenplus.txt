#include <bits/stdc++.h>
using namespace std;

#define vec vector
#define ll long long
#define INF 5LL << 60

template <typename T>
void chmin(T& x, const T& y) { x = min(x, y); }

const ll NMAX = 2e5+10;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    for (ll i = 0; i < N; i++) {
        cin >> A[i];
    }

    vec<vec<pair<ll, ll>>> graph(N);
    for (ll i = 0; i < M; i++) {
        ll u, v, b;
        cin >> u >> v >> b;
        u--; v--;
        graph[u].push_back({v, b});
        graph[v].push_back({u, b});
    }

    // Dijkstra's algorithm
    vec<ll> dist(N, INF);
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;

    dist[0] = A[0];  // Start node's path starts with its own weight
    pq.push({dist[0], 0});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();

        if (d > dist[u]) continue;

        for (auto [v, b] : graph[u]) {
            ll new_cost = d + b + A[v];  // Path cost: current cost + edge weight + vertex weight
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({dist[v], v});
            }
        }
    }

    // Output results for nodes 2 to N (1-based index)
    for (ll i = 1; i < N; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
