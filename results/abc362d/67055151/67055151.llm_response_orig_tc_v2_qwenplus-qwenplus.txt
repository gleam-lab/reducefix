```cpp
#include <bits/stdc++.h>

using namespace std;

#define ll long long

const ll INF = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    vector<ll> A(N);
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<vector<pair<ll, ll>>> graph(N);
    for (ll i = 0; i < M; ++i) {
        ll u, v, b;
        cin >> u >> v >> b;
        u--;
        v--;
        graph[u].push_back({v, b});
        graph[v].push_back({u, b});
    }

    // Dijkstra's algorithm
    vector<ll> dist(N, INF);
    dist[0] = A[0]; // Start at node 0, initial cost is A[0]

    using pll = pair<ll, ll>;
    priority_queue<pll, vector<pll>, greater<pll>> heap;
    heap.push({A[0], 0});

    while (!heap.empty()) {
        ll d = heap.top().first;
        ll u = heap.top().second;
        heap.pop();

        if (d > dist[u]) continue;

        for (auto& [v, b] : graph[u]) {
            ll new_cost = dist[u] + b + A[v];
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                heap.push({dist[v], v});
            }
        }
    }

    for (ll i = 1; i < N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
