#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>
#include <climits>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, int> pli;

struct Edge {
    int to;
    ll weight;
};

struct State {
    int u;
    ll cost;
    bool operator<(const State &other) const {
        return cost > other.cost;
    }
};

vector<ll> dijkstra(int start, vector<vector<Edge>>& graph, vector<ll>& weights) {
    int n = graph.size();
    vector<ll> dist(n, LLONG_MAX);
    set<pli> pq;

    dist[start] = weights[start];
    pq.emplace(dist[start], start);

    while (!pq.empty()) {
        int u = pq.begin()->second;
        ll d = pq.begin()->first;
        pq.erase(pq.begin());

        if (d > dist[u])
            continue;

        for (const Edge &e : graph[u]) {
            int v = e.to;
            ll w = e.weight;
            ll alt = dist[u] + weights[v] + w;

            if (alt < dist[v]) {
                dist[v] = alt;
                pq.emplace(dist[v], v);
            }
        }
    }

    return dist;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<ll> weights(n);
    for (int i = 0; i < n; ++i) {
        cin >> weights[i];
    }

    vector<vector<Edge>> graph(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        --u, --v;
        graph[u].emplace_back(v, w);
        graph[v].emplace_back(u, w);
    }

    vector<ll> results(n - 1);
    for (int i = 1; i < n; ++i) {
        results[i - 1] = dijkstra(0, graph, weights)[i];
    }

    for (ll result : results) {
        cout << result << " ";
    }
    cout << endl;

    return 0;
}

This solution uses Dijkstra's algorithm to find the shortest path from vertex 1 to all other vertices in the graph. It handles the input and output formats correctly and ensures that the algorithm runs efficiently within the given constraints.
