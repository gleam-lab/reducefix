#include <bits/stdc++.h>

struct Edge {
    int to, cost;
    Edge(int to, int cost) : to(to), cost(cost) {}
};

class Dijkstra {
public:
    vector<int> dijkstra(vector<vector<Edge>>& graph, int start) {
        int n = graph.size();
        vector<int> dist(n, INF);
        dist[start] = 0;
        priority_queue<PII, vector<PII>, greater<PII>> pq;
        pq.push({start, 0});

            auto [u, d] = pq.top(); pq.pop();

            for (auto& e : graph[u]) {
                int v = e.to, w = e.cost;
                if (dist[v] > dist[u] + w) {
                    dist[v] = dist[u] + w;
                    pq.push({v, dist[v]});
                }
            }
        }
        return dist;
    }
};
