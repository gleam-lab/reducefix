#include <bits/stdc++.h>

using namespace std;

const long long INF = 1e18;

struct Edge {
    long long u, v, w;
};

struct DisjointSet {
    vector<long long> parent, rank;

    DisjointSet(long long n) : parent(n), rank(n, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    long long find(long long u) {
        if (u != parent[u]) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }

    void merge(long long u, long long v) {
        u = find(u);
        v = find(v);
        if (u != v) {
            if (rank[u] > rank[v]) {
                parent[v] = u;
            } else if (rank[u] < rank[v]) {
                parent[u] = v;
            } else {
                parent[v] = u;
                rank[u]++;
            }
        }
    }
};

struct Graph {
    long long n, m;
    vector<Edge> edges;

    Graph(long long n, long long m) : n(n), m(m) {}

    void addEdge(long long u, long long v, long long w) {
        edges.emplace_back(u, v, w);
    }

    vector<long long> shortestPath(long long source) {
        vector<long long> dist(n, INF);
        dist[source] = 0;

        for (long long i = 0; i < n - 1; ++i) {
            for (const auto& edge : edges) {
                long long u = edge.u, v = edge.v, w = edge.w;
                if (dist[u] != INF && dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                }
            }
        }

        return dist;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    long long n, m;
    cin >> n >> m;

    vector<long long> a(n);
    for (long long i = 0; i < n; ++i) {
        cin >> a[i];
    }

    Graph g(n, m);
    for (long long i = 0; i < m; ++i) {
        long long u, v, w;
        cin >> u >> v >> w;
        u--, v--;
        g.addEdge(u, v, w);
    }

    vector<long long> ans(n - 1, INF);
    DisjointSet ds(n);

    for (long long i = 0; i < n; ++i) {
        auto dist = g.shortestPath(i);
        for (long long j = 0; j < n; ++j) {
            if (ds.find(i) != ds.find(j)) {
                ds.merge(i, j);
                ans[abs(i - j) - 1] = min(ans[abs(i - j) - 1], dist[j] + a[j]);
            }
        }
    }

    for (long long i = 1; i < n; ++i) {
        cout << ans[i - 1] << " ";
    }
    cout << endl;

    return 0;
}
