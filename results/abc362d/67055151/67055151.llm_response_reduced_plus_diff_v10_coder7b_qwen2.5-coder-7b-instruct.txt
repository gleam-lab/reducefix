vvpl buildGraph(int n, int m, vector<int>& a, vector<vpl>& edges) {
    vvpl graph(n);
    for (const auto& edge : edges) {

vl dijkstra(int start, vvpl& graph, const vl& a) {

            if (nd < dist[v]) {

                pq.emplace(nd, v);
            }
        }
    }

}
