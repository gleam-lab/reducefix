```cpp
#include <bits/stdc++.h>
using namespace std;

#define forn(i, n) for(int i = 0; i < n; ++i)
#define all(x) begin(x), end(x)
#define sz(x) (int)(x).size()

typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;

struct DSU {
    vector<int> par, rank;

    DSU(int n) : par(n), rank(n, 1) {
        iota(all(par), 0);
    }

    int find(int u) {
        return par[u] == u ? u : par[u] = find(par[u]);
    }

    void merge(int u, int v) {
        int pu = find(u), pv = find(v);
        if(pu == pv) return;
        if(rank[pu] > rank[pv]) swap(pu, pv);
        par[pu] = pv;
        rank[pv] += rank[pu];
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<ll> a(n);
    forn(i, n) cin >> a[i];

    vector<vector<pii>> g(n);
    forn(i, m) {
        int u, v, w;
        cin >> u >> v >> w;
        --u, --v;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    vector<ll> ans(n - 1, LLONG_MAX);
    DSU dsu(n);

    priority_queue<pll, vector<pll>, greater<pll>> pq;
    pq.emplace(a[0], 0);

    while(!pq.empty()) {
        auto [cost, u] = pq.top();
        pq.pop();

        if(ans[u] != LLONG_MAX) continue;
        ans[u] = cost;

        for(auto [v, w] : g[u]) {
            if(dsu.find(u) != dsu.find(v)) {
                dsu.merge(u, v);
                pq.emplace(cost + w + a[v], v);
            }
        }
    }

    forn(i, n - 1) cout << ans[i] << ' ';
    cout << '\n';

    return 0;
}
```

This solution uses Kruskal's algorithm with a Disjoint Set Union (DSU) data structure to efficiently find the minimum spanning tree (MST) of the graph. The MST is then used to determine the minimum path weights from vertex 1 to other vertices. The use of a priority queue ensures that the algorithm runs efficiently even for large inputs within the constraints.
