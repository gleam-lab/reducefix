```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll INF = 1LL << 60;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n, m; cin >> n >> m;
    vector<ll> a(n); 
    for (ll i = 0; i < n; i++) cin >> a[i];

    vector<vector<pair<ll, ll>>> graph(n);
    for (ll i = 0; i < m; i++) {
        ll u, v, b; cin >> u >> v >> b;
        u--; v--;
        graph[u].emplace_back(v, b);
        graph[v].emplace_back(u, b);
    }

    vector<ll> dist(n, INF);
    dist[0] = a[0]; // Start with vertex 1's weight

    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
    pq.emplace(dist[0], 0);

    while (!pq.empty()) {
        auto [current_dist, u] = pq.top();
        pq.pop();

        if (current_dist > dist[u]) continue;

        for (auto [v, b] : graph[u]) {
            ll new_dist = current_dist + b + a[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.emplace(new_dist, v);
            }
        }
    }

    for (ll i = 1; i < n; i++) {
        cout << dist[i] << (i == n-1 ? "\n" : " ");
    }

    return 0;
}
```

1. Proper Dijkstra's algorithm implementation using min-heap
