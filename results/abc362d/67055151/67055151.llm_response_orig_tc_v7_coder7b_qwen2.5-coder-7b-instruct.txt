#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

struct Edge {
    ll to, cost;
    Edge(ll to, ll cost) : to(to), cost(cost) {}
};

const ll INF = numeric_limits<ll>::max() / 2;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    ll n, m;
    cin >> n >> m;
    
    vector<ll> a(n);
    for (ll i = 0; i < n; ++i) cin >> a[i];
    
    vector<vector<Edge>> adj(n);
    for (ll i = 0; i < m; ++i) {
        ll u, v, w;
        cin >> u >> v >> w;
        --u, --v;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
    
    vector<ll> dist(n, INF);
    dist[0] = 0;
    
    set<pair<ll, ll>> pq;
    pq.emplace(dist[0], 0);
    
    while (!pq.empty()) {
        ll u = pq.begin()->second;
        pq.erase(pq.begin());
        
        for (auto &[v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                pq.erase(pq.find({dist[v], v}));
                dist[v] = dist[u] + w;
                pq.emplace(dist[v], v);
            }
        }
    }
    
    for (ll i = 1; i < n; ++i) cout << dist[i] << " ";
    cout << "\n";
    
    return 0;
}
