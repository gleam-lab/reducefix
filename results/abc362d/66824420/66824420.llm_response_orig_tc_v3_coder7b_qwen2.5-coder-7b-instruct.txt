#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

struct Edge {
    int to;
    ll w;
};

void dijkstra(const vector<vector<Edge>>& adj, const vector<ll>& weights, int start, vector<ll>& dist) {
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    dist[start] = weights[start];
    pq.push({weights[start], start});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();

        if (d > dist[u]) continue;

        for (const auto& e : adj[u]) {
            int v = e.to;
            ll cost = d + e.w + weights[v];

            if (cost < dist[v]) {
                dist[v] = cost;
                pq.push({cost, v});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<ll> weights(n);
    for (int i = 0; i < n; ++i) {
        cin >> weights[i];
    }

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        --u; --v; // Adjust for zero-based indexing
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w); // For undirected graph
    }

    vector<ll> dist(n, LLONG_MAX);
    dijkstra(adj, weights, 0, dist);

    for (int i = 1; i < n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
