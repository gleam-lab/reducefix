```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    vector<vector<pair<int, int>>> adj(n + 1);
    for (int j = 0; j < m; j++) {
        int x, y, w;
        cin >> x >> y >> w;
        adj[x].push_back({y, w});
        adj[y].push_back({x, w});
    }

    vector<long long> dist(n + 1, LLONG_MAX);
    dist[1] = a[0];
    priority_queue<pair<long long, long long>, vector<pair<long long, long long>>, greater<pair<long long, long long>>> q;
    q.push({a[0], 1});

    while (!q.empty()) {
        pair<long long, long long> p = q.top();
        q.pop();
        long long val = p.first, node = p.second;
        if (val > dist[node]) continue; // Skip outdated nodes
        for (auto it : adj[node]) {
            long long total_dist = val + it.second + a[it.first - 1];
            if (total_dist < dist[it.first]) {
                dist[it.first] = total_dist;
                q.push({total_dist, it.first});
            }
        }
    }

    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }

    return 0;
}
```

### Explanation of Fixes:
1. **Priority Queue Comparator**: Changed the comparator to `greater<pair<long long, long long>>` to ensure the queue pops the smallest element first.
2. **Distance Array Initialization**: Initialized `dist` array with `LLONG_MAX` instead of `1E10+0` to handle larger values correctly.
3. **Outdated Node Check**: Added a check inside the loop to skip processing nodes whose current distance value is already outdated due to a previous update. This optimizes the Dijkstra's algorithm.
4. **Correct Indexing**: Ensured correct indexing when accessing the adjacency list and updating distances.
