#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<long long> dist(n + 1, 1e18);
    dist[1] = a[0];
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({a[0], 1});

    while (!pq.empty()) {
        auto [cost, vertex] = pq.top();
        pq.pop();
        if (cost > dist[vertex]) continue;
        for (auto &edge : adj[vertex]) {
            int next = edge.first;
            long long weight = edge.second;
            long long total_cost = cost + weight + a[next - 1];
            if (total_cost < dist[next]) {
                dist[next] = total_cost;
                pq.push({total_cost, next});
            }
        }
    }

    vector<long long> result(n);
    for (int i = 2; i <= n; i++) {
        result[i - 2] = dist[i];
    }

    for (int i = 0; i < (int)result.size(); i++) {
        cout << result[i];
        if (i != result.size() - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
