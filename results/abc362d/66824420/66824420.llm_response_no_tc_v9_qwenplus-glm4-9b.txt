#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> A(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> A[i];
    }
    
    // Adjacency list for the graph
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v, b;
        cin >> u >> v >> b;
        adj[u].push_back({v, b});
        adj[v].push_back({u, b});
    }
    
    // Distance array, initialized to a large value
    vector<long long> dist(n + 1, 1e18);
    dist[1] = A[1];
    
    // Priority queue to perform Dijkstra's algorithm
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({A[1], 1});
    
    while (!pq.empty()) {
        auto [curr_dist, u] = pq.top();
        pq.pop();
        
        // If the current distance is greater than the recorded, skip processing
        if (curr_dist > dist[u]) continue;
        
        // Process all adjacent vertices
        for (auto &[v, weight] : adj[u]) {
            long long new_dist = curr_dist + weight + A[v];
            // Only consider this path if it's better
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
    
    // Output the minimum distances to all vertices (except vertex 1)
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
