```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    
    vector<long long> a(n + 1); // 1-indexed
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    vector<vector<pair<int, long long>>> adj(n + 1); // 1-indexed
    for (int j = 0; j < m; ++j) {
        int u, v;
        long long b;
        cin >> u >> v >> b;
        adj[u].push_back({v, b});
        adj[v].push_back({u, b});
    }

    // Initialize distances with large value, we use 1-indexed
    vector<long long> dist(n + 1, numeric_limits<long long>::max());
    dist[1] = a[1]; // Start from vertex 1

    // Min-heap priority queue: (distance, node)
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [curr_dist, curr_node] = pq.top();
        pq.pop();

        // If current distance is not the shortest one, skip
        if (curr_dist > dist[curr_node]) continue;

        // Explore neighbors
        for (auto &[next_node, edge_weight] : adj[curr_node]) {
            long long new_dist = curr_dist + edge_weight + a[next_node];

            if (new_dist < dist[next_node]) {
                dist[next_node] = new_dist;
                pq.push({new_dist, next_node});
            }
        }
    }

    // Output results for nodes 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
