```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

typedef pair<int, int> pii;

void dijkstra(const vector<vector<pii>>& adj, const vector<int>& a, vector<long long>& dist) {
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    pq.push({a[0], 1}); // Start from node 1 with distance equal to its own weight
    dist[1] = a[0];

    while (!pq.empty()) {
        auto [currentDist, u] = pq.top();
        pq.pop();

        if (currentDist > dist[u]) continue;

        for (const auto& [v, w] : adj[u]) {
            long long newDist = currentDist + w + a[v - 1]; // Weight of the edge plus the vertex weight
            if (newDist < dist[v]) {
                dist[v] = newDist;
                pq.push({newDist, v});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    vector<vector<pii>> adj(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w); // Since the graph is undirected
    }

    vector<long long> dist(n + 1, LLONG_MAX);
    dijkstra(adj, a, dist);

    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    return 0;
}
```

This solution uses a min-heap (priority queue) to always expand the least distant unvisited vertex, ensuring efficient computation of the shortest paths. It handles all edge cases and should work correctly for large inputs up to the specified constraints.
