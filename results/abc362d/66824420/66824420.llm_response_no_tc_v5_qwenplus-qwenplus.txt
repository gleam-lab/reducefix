```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> a(n + 1); // Vertex weights (1-indexed)
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    // Build adjacency list: each edge stores connected node and its weight
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; ++j) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    // Initialize distance array with large values
    vector<long long> dist(n + 1, 1e18);
    dist[1] = a[1]; // Start from vertex 1 with its weight

    // Dijkstra's algorithm using min-heap priority queue
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [curr_dist, u] = pq.top();
        pq.pop();

        // Skip if we've already found a better path to u
        if (curr_dist > dist[u]) continue;

        for (auto [v, edge_weight] : adj[u]) {
            // Calculate new path cost: current path cost - u's vertex weight + edge weight + v's vertex weight
            long long new_dist = curr_dist + edge_weight + a[v] - a[u];
            
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    // Output results for nodes 2 through N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
