```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; j++) {
        int x, y;
        long long w;
        cin >> x >> y >> w;
        adj[x].push_back({y, w + a[y - 1]});
        adj[y].push_back({x, w + a[x - 1]});
    }

    vector<long long> dist(n + 1, LLONG_MAX);
    dist[1] = a[0];
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> q;
    q.push({a[0], 1});
    while (!q.empty()) {
        auto p = q.top();
        q.pop();
        long long val = p.first;
        int node = p.second;
        if (val > dist[node]) continue;
        for (auto it : adj[node]) {
            long long total_dist = val + it.second;
            if (total_dist < dist[it.first]) {
                dist[it.first] = total_dist;
                q.push({total_dist, it.first});
            }
        }
    }

    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
}
```

3. **Dijkstra's Algorithm**: The algorithm is initialized with the starting vertex 1, whose distance is set to its vertex weight `a[0]`. The priority queue processes vertices in order of increasing path weight.
4. **Distance Calculation**: For each vertex, the algorithm relaxes edges by updating the distance to neighboring vertices if a shorter path is found, incorporating both the edge weight and the vertex weight of the destination.
5. **Output**: The minimum path weights from vertex 1 to all other vertices are printed for `i = 2` to `n`.

This approach ensures that both vertex and edge weights are correctly accounted for in the path weight calculations, providing the correct minimum path weights from vertex 1 to all other vertices.
