#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; ++j) {
        int x, y, w;
        cin >> x >> y >> w;
        adj[x].push_back({y, w});
        adj[y].push_back({x, w});
    }
    
    // Use priority queue to perform Dijkstra's algorithm
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    vector<long long> dist(n + 1, LLONG_MAX);
    dist[1] = a[1];
    pq.push({a[1], 1});
    
    while (!pq.empty()) {
        auto [d, node] = pq.top();
        pq.pop();
        
        if (d > dist[node]) continue;
        
        for (auto &edge : adj[node]) {
            int nextNode = edge.first;
            long long nextDist = d + edge.second;
            
            if (nextDist < dist[nextNode]) {
                dist[nextNode] = nextDist;
                pq.push({nextDist, nextNode});
            }
        }
    }
    
    // Output the minimum weights for paths from 1 to all other nodes
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
