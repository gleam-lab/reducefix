```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> A(n);
    for (int i = 0; i < n; ++i) cin >> A[i];

    // Adjacency list: each entry is a list of (neighbor, edge_weight)
    vector<vector<pair<int, long long>>> adj(n + 1); // 1-indexed vertices
    for (int j = 0; j < m; ++j) {
        int u, v;
        long long b;
        cin >> u >> v >> b;
        adj[u].push_back({v, b});
        adj[v].push_back({u, b});
    }

    // Dijkstra's algorithm
    vector<long long> dist(n + 1, 1e18);
    dist[1] = A[0]; // Starting at node 1 with its vertex weight

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto &[v, b] : adj[u]) {
            long long new_dist = d + b + A[v - 1]; // current path cost + edge weight + destination vertex weight
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    // Output the results for nodes 2 through N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
```
