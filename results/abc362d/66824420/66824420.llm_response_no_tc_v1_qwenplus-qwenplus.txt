```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> A(n + 1);  // Using 1-based indexing
    for (int i = 1; i <= n; ++i) {
        cin >> A[i];
    }

    // Build adjacency list: each edge stores (neighbor, weight)
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; ++j) {
        int u, v, b;
        cin >> u >> v >> b;
        adj[u].push_back({v, b});
        adj[v].push_back({u, b});  // since the graph is undirected
    }

    // Dijkstra's algorithm
    vector<long long> dist(n + 1, 1e18);
    dist[1] = A[1];  // Start from vertex 1, include its vertex weight

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [curr_dist, u] = pq.top();
        pq.pop();

        // Early exit: if we've already found a shorter path to u
        if (curr_dist > dist[u]) continue;

        for (auto [v, b] : adj[u]) {
            long long new_dist = curr_dist + b + A[v];  // current path weight + edge weight + destination vertex weight
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    // Output results for vertices 2 through N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;

    return 0;
}
```
