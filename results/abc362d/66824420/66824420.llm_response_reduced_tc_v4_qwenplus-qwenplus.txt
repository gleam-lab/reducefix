```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<long long> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<vector<pair<int, long long>>> adj(N + 1); // adj[0] unused
    for (int i = 0; i < M; ++i) {
        int U, V;
        long long B;
        cin >> U >> V >> B;
        adj[U].push_back({V, B});
        adj[V].push_back({U, B});
    }

    const long long INF = numeric_limits<long long>::max();
    vector<long long> dist(N + 1, INF);
    dist[1] = A[0]; // Start from vertex 1

    using pii = pair<long long, int>;
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto& [v, b] : adj[u]) {
            long long new_dist = d + b + A[v - 1];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
