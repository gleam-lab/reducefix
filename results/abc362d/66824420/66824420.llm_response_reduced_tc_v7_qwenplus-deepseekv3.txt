#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; ++j) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].emplace_back(v, w + a[v - 1]);
        adj[v].emplace_back(u, w + a[u - 1]);
    }
    
    vector<long long> dist(n + 1, LLONG_MAX);
    dist[1] = a[0];
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.emplace(dist[1], 1);
    
    while (!pq.empty()) {
        auto [current_dist, u] = pq.top();
        pq.pop();
        if (current_dist > dist[u]) continue;
        for (auto [v, weight] : adj[u]) {
            if (dist[v] > dist[u] + weight) {
                dist[v] = dist[u] + weight;
                pq.emplace(dist[v], v);
            }
        }
    }
    
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}

3. **Dijkstra's Algorithm**: The algorithm initializes the distance from vertex 1 to itself as `a[0]` (the weight of vertex 1). For all other vertices, distances are initialized to infinity. The priority queue is used to process vertices in order of increasing distance.
4. **Relaxation Step**: For each vertex `u`, the algorithm checks all adjacent vertices `v`. If the path from vertex 1 to `v` via `u` offers a shorter path, the distance to `v` is updated, and `v` is added to the priority queue.
5. **Output**: The distances from vertex 1 to vertices 2 through `n` are printed, representing the minimum path weights for these vertices.

This approach ensures that the path weight is correctly calculated by summing the weights of the vertices and edges traversed, and efficiently computes the shortest paths using Dijkstra's algorithm.
