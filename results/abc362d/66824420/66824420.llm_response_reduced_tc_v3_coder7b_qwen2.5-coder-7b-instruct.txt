#include <bits/stdc++.h>

    vector<vector<pair<unsigned long long, unsigned long long>>> adj(n + 1);
    for(unsigned long long j = 0; j < m; j++) {
        unsigned long long x, y, w;
        cin >> x >> y >> w;
        adj[x].push_back({y, w});
        adj[y].push_back({x, w});
    }

    vector<unsigned long long> dist(n + 1, numeric_limits<unsigned long long>::max());
    dist[1] = a[0];
    priority_queue<pair<unsigned long long, unsigned long long>, vector<pair<unsigned long long, unsigned long long>>, greater<pair<unsigned long long, unsigned long long>>> pq;
    pq.push({a[0], 1});

    while(!pq.empty()) {
        auto p = pq.top();
        pq.pop();
        unsigned long long val = p.first, node = p.second;

        for(const auto& it : adj[node]) {
            unsigned long long total_dist = val + it.second + a[it.first - 1];
            if(total_dist < dist[it.first]) {
                dist[it.first] = total_dist;
                pq.push({total_dist, it.first});
            }
        }
    }

    for(unsigned long long i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }

    return 0;
}
