```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    vector<vector<pair<int, long long>>> adj(n + 1);
    for (int j = 0; j < m; j++) {
        int x, y;
        long long w;
        cin >> x >> y >> w;
        adj[x].push_back({y, w});
        adj[y].push_back({x, w});
    }
    
    vector<long long> dist(n + 1, LLONG_MAX);
    dist[1] = a[0]; // Include vertex 1's weight at the start
    
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> q;
    q.push({a[0], 1});
    
    while (!q.empty()) {
        auto [val, node] = q.top();
        q.pop();
        
        // If we've already found a better distance, skip
        if (val > dist[node]) continue;
        
        for (auto [neighbor, edge_weight] : adj[node]) {
            long long new_dist = val + edge_weight + a[neighbor - 1];
            if (new_dist < dist[neighbor]) {
                dist[neighbor] = new_dist;
                q.push({new_dist, neighbor});
            }
        }
    }
    
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << (i == n ? "" : " ");
    }
    cout << endl;
    
    return 0;
}
```
