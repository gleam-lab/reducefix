```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    
    vector<vector<pair<int, int>>> adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    
    vector<long long> dist(n + 1, LONG_LONG_MAX);
    dist[1] = 0; // 猫碌路莽聜鹿氓聢掳猫碌路莽聜鹿莽職聞忙聺聝茅聡聧盲赂潞 0
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> q;
    q.push({0, 1}); // 盲禄楼 (猫路聺莽娄禄, 猫聤聜莽聜鹿) 氓陆垄氓录聫茅聵聼
    
    while (!q.empty()) {
        auto [d, u] = q.top();
        q.pop();
        if (d > dist[u]) continue; // 茅聛驴茅聡聧氓陇聧氓陇聞莽聬聠
        for (auto &[v, w] : adj[u]) {
            long long nextDist = d + w + a[v];
            if (nextDist < dist[v]) {
                dist[v] = nextDist;
                q.push({nextDist, v});
            }
        }
    }
    
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
    return 0;
}
```
