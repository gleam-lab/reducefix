The issue in your code is not algorithmic but numeric. You're initializing the distance array with `1E10+0`, which is insufficient for large values in the problem. The maximum path cost could easily exceed `1e10` due to the constraints (vertex and edge weights up to `1e9` and paths possibly including many nodes and edges).

This causes incorrect comparisons and results in some shortest paths not being updated properly because the default value isn't large enough.

---

### 🔧 Fix:

Replace:
```cpp
vector<long long> dist(n + 1, 1E10 + 0);
```
with:
```cpp
vector<long long> dist(n + 1, numeric_limits<long long>::max());
```

Additionally, ensure that you don’t overflow when doing arithmetic operations by using appropriate types (`long long`) consistently.

Also, make sure the priority queue is initialized correctly with `dist[1] = A_1`.

### ✅ Corrected Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> a(n + 1); // Vertex weights
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    vector<vector<pair<int, long long>>> adj(n + 1); // Adjacency list

    for (int j = 0; j < m; ++j) {
        int u, v;
        long long b;
        cin >> u >> v >> b;
        adj[u].push_back({v, b});
        adj[v].push_back({u, b});
    }

    vector<long long> dist(n + 1, numeric_limits<long long>::max());
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> q;

    dist[1] = a[1]; // Start node includes its own vertex weight
    q.push({dist[1], 1});

    while (!q.empty()) {
        auto [d, u] = q.top();
        q.pop();

        if (d > dist[u]) continue;

        for (auto [v, edge_weight] : adj[u]) {
            long long new_cost = d + edge_weight + a[v];

            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                q.push({new_cost, v});
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
