#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using pii = pair<int, int>;

constexpr int MAXN = 2e5 + 5;

vector<pii> adj[MAXN];
i64 dist[MAXN];
bool visited[MAXN];

void solve() {
    int N, M;
    cin >> N >> M;

    vector<i64> A(N + 1);
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }

    for (int i = 0; i < M; ++i) {
        int U, V;
        i64 B;
        cin >> U >> V >> B;
        adj[U].emplace_back(V, B);
        adj[V].emplace_back(U, B);
    }

    // Initialize distances with infinity
    fill(dist + 1, dist + N + 1, numeric_limits<i64>::max());
    fill(visited + 1, visited + N + 1, false);

    // Start from node 1 with initial cost of A[1]
    priority_queue<pair<i64, int>, vector<pair<i64, int>>, greater<>> pq;
    dist[1] = A[1];
    pq.emplace(dist[1], 1);

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (visited[u]) continue;
        visited[u] = true;

        for (const auto& [v, b] : adj[u]) {
            i64 new_cost = d + b + A[v];
            if (!visited[v] && dist[v] > new_cost) {
                dist[v] = new_cost;
                pq.emplace(dist[v], v);
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
