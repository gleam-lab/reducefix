#include <bits/stdc++.h>

template <class T>

template <class T>

struct Node {
    int x;
    int64_t w;
    bool operator < (const Node& other) const {
        return w > other.w;
    }
};

constexpr int64_t INF = 1LL << 60;

void solve() {
    int n, m;
    cin >> n >> m;
    V<int> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    VV<pair<int, int>> g(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        g[u].emplace_back(v, b);
        g[v].emplace_back(u, b);
    }
    V<int64_t> d(n + 1, INF);
    d[1] = a[1];
    priority_queue<Node> pq;
    pq.push({1, d[1]});
    V<bool> vis(n + 1, false);
    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        for (auto [y, b] : g[x]) {
            if (d[y] > d[x] + b + a[y]) {
                d[y] = d[x] + b + a[y];
                pq.push({y, d[y]});
            }
        }
    }
    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " \n"[i == n];
    }
}

INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    while (t--) {

    }
    return 0;
}

### Explanation
1. **Graph Representation**: The graph is stored in an adjacency list `g`, where each vertex `u` has a list of pairs `(v, b)` representing edges to vertex `v` with weight `b`.
2. **Initialization**: The distance array `d` is initialized with `INF` (a very large value), except for the starting vertex 1, which is initialized to its own weight `a[1]`.
3. **Priority Queue**: The priority queue is used to always expand the vertex with the current minimum path weight. This ensures that the first time a vertex is visited, the distance stored is indeed the shortest.
4. **Relaxation**: For each vertex extracted from the queue, the algorithm checks all neighbors. If a shorter path is found (sum of current path weight, edge weight, and neighbor's vertex weight), the neighbor's distance is updated, and it is added to the queue.
5. **Output**: After processing all vertices, the distances from vertex 1 to all other vertices are printed. The output format ensures that spaces separate the values, and a newline follows the last value.
