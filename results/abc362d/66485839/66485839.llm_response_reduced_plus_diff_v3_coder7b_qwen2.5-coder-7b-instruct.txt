    int fa[MAXN], sz[MAXN];

    DSU(int n) {
        iota(fa, fa + n, 0);
        fill(sz, sz + n, 1);
    }

    int find(int x) {
        if (fa[x] != x) fa[x] = find(fa[x]);
        return fa[x];
    }

        x = find(x), y = find(y);
        if (x == y) return;
        if (sz[x] < sz[y]) swap(x, y);
        fa[y] = x;
        sz[x] += sz[y];
    }
};
