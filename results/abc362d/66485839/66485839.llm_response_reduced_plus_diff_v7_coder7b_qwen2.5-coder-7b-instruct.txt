const int INF = 1e9+7;

void dijkstra(const vector<vpii>& adj, int src, vll& dist) {
    priority_queue<pll, vpll, greater<pll>> pq;
    dist[src] = 0;
    pq.push({0, src});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        
        if (d > dist[u]) continue;
        
        trav(v, adj[u]) {
            int w = v.second;
            int v1 = v.first;
            
            if (dist[u] + w + dist[v1] < dist[v1]) {
                dist[v1] = dist[u] + w + dist[v1];
                pq.push({dist[v1], v1});
            }
        }
    }
}
