#include <bits/stdc++.h>
using namespace std;
#define int long long

void solve() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n + 1);
    vector<pair<int, int>> adj[n + 1]; // adjacency list
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    // Dijkstra's algorithm to find the shortest path from vertex 1 to all other vertices
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq; // min-heap based on the sum of vertex and edge weights
    vector<int> dist(n + 1, LLONG_MAX); // distance vector
    dist[1] = 0;
    pq.push({0, 1}); // start from vertex 1 with initial distance 0

    while (!pq.empty()) {
        int u = pq.top().second;
        int currDist = pq.top().first;
        pq.pop();

        if (currDist > dist[u]) continue;

        for (auto& edge : adj[u]) {
            int v = edge.first;
            int edgeWeight = edge.second;
            int newDist = currDist + edgeWeight + a[v]; // total weight including vertex weight
            if (newDist < dist[v]) {
                dist[v] = newDist;
                pq.push({newDist, v});
            }
        }
    }

    // Output the minimum distances for vertices 2 through n
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << (i == n ? "" : " ");
    }
    cout << "\n";
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
