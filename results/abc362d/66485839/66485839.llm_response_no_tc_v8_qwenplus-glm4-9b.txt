#include <bits/stdc++.h>
using namespace std;
#define INF 1e18

using i64 = long long;

void solve() {
    int n, m;
    cin >> n >> m;
    vector<i64> a(n + 1);
    vector<pair<i64, pair<int, int>>> edges; // pair<weight, pair(u, v)>
    
    // Read vertex weights
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    // Initialize graph edges
    for (int i = 1; i <= m; ++i) {
        int u, v, c;
        cin >> u >> v >> c;
        edges.emplace_back(c, {u, v});
    }
    
    // Kruskal's algorithm to find the Minimum Spanning Tree (MST)
    vector<int> parent(n + 1);
    i64 mst_weight = 0;
    
    // Union-Find data structure
    for (int i = 1; i <= n; ++i) {
        parent[i] = i;
    }
    
    sort(edges.begin(), edges.end());
    
    for (auto& [weight, uv] : edges) {
        int u = uv.first, v = uv.second;
        int root_u = find(parent, u), root_v = find(parent, v);
        if (root_u != root_v) {
            // Connect the components
            parent[root_u] = root_v;
            mst_weight += weight;
        }
    }
    
    // After MST, calculate minimum path from 1 to all other nodes
    vector<i64> dist(n + 1, INF);
    dist[1] = a[1];
    priority_queue<pair<i64, int>, vector<pair<i64, int>>, greater<pair<i64, int>>> pq;
    pq.push({a[1], 1});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dist[u]) continue;
        for (int v = 1; v <= n; ++v) {
            if (find(parent, u) != find(parent, v)) continue; // Ensure u and v are in the same connected component
            i64 new_dist = d + a[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
    
    // Output minimum distances from 1 to all other nodes except 1
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << (i < n ? " " : "\n");
    }
}

int find(vector<int>& parent, int u) {
    if (parent[u] != u) parent[u] = find(parent, parent[u]);
    return parent[u];
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
