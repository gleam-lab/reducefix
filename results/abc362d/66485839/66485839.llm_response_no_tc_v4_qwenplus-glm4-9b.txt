#include <iostream>
#include <vector>
#include <queue>
#include <tuple>

using namespace std;

const int MAX_N = 200010;

int n, m;
vector<pair<int, int>> adj[MAX_N]; // Adjacency list
vector<int> vertex_weights(MAX_N); // Vertex weights
vector<pair<int, int>> edges; // All edges
vector<long long> dist(MAX_N, LLONG_MAX); // Distance Array
priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq; // Min-heap for Dijkstra's algorithm

void dijkstra(int start) {
    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        auto [d, node] = pq.top();
        pq.pop();

        if (dist[node] < d) continue;

        for (auto [neighbour, weight] : adj[node]) {
            long long new_dist = d + weight + vertex_weights[neighbour];
            if (new_dist < dist[neighbour]) {
                dist[neighbour] = new_dist;
                pq.push({new_dist, neighbour});
            }
        }
    }
}

int main() {
    cin >> n >> m;
    
    vertex_weights[1] = 0; // Assuming vertex 1 has zero weight for simplicity
    
    for (int i = 0; i < n; i++) {
        cin >> vertex_weights[i + 1];
    }
    
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        edges.emplace_back(u, v);
    }

    // Run Dijkstra's algorithm from vertex 1
    dijkstra(1);
    
    // Output the minimum weight for each vertex 2 through N
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
    
    return 0;
}
