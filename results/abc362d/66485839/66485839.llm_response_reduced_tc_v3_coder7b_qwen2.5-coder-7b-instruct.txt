```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define LL long long
#define INF 0x3f3f3f3f3f3f3f3f
#define pb push_back
#define mp make_pair
#define all(x) begin(x), end(x)

const int MAXN = 2e5 + 10;

int n, m;
LL a[MAXN], dis[MAXN];
vector<pair<int, int>> adj[MAXN];

void dijkstra(int src) {
    priority_queue<pair<LL, int>, vector<pair<LL, int>>, greater<pair<LL, int>>> pq;
    fill(dis, dis + n + 1, INF);
    pq.push({a[src], src});
    dis[src] = a[src];

    while (!pq.empty()) {
        auto [dist, u] = pq.top();
        pq.pop();

        if (dis[u] < dist) continue;

        for (auto &[v, w] : adj[u]) {
            if (dis[v] > dis[u] + w + a[v]) {
                dis[v] = dis[u] + w + a[v];
                pq.push({dis[v], v});
            }
        }
    }
}

signed main() {
    ios_base::sync_with_stdio(0); cin.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) cin >> a[i];

    for (int i = 1; i <= m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].pb({v, w});
        adj[v].pb({u, w});
    }

    dijkstra(1);

    for (int i = 2; i <= n; ++i) {
        cout << dis[i] << " ";
    }
    cout << "\n";

    return 0;
}
```

This solution uses Dijkstra's algorithm to find the shortest path from vertex 1 to every other vertex in the graph. It handles large numbers using `long long` and ensures that the graph is represented correctly with adjacency lists. The algorithm efficiently computes the minimum weight paths using a priority queue, ensuring optimal performance even for large graphs up to the specified constraints.
