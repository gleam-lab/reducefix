2. **Insight**: The standard Dijkstra's algorithm can be adapted where the distance to a vertex includes the sum of the edge weights and the vertex's weight. However, the vertex weight should be added only once when the vertex is first encountered.
3. **Algorithm**:

   - The distance to a vertex `v` from the source vertex `1` is the sum of the edge weights along the path plus the weights of the vertices along the path (each vertex's weight included only once).
   - The initial distance to vertex `1` is its own weight `A[1]`.
   - For each neighbor `u` of vertex `v`, the distance to `u` is the distance to `v` plus the edge weight between `v` and `u` plus the weight of `u` (if `u` hasn't been visited yet).
4. **Complexity Analysis**: The algorithm runs in O(M log N) time, where M is the number of edges and N is the number of vertices, due to the use of a priority queue.

### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

const i64 INF = 1e18;

void solve() {
    int N, M;
    cin >> N >> M;
    vector<i64> A(N + 1);
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }
    vector<vector<pair<int, i64>>> g(N + 1);
    for (int i = 0; i < M; ++i) {
        int u, v;
        i64 b;
        cin >> u >> v >> b;
        g[u].emplace_back(v, b);
        g[v].emplace_back(u, b);
    }

    vector<i64> dist(N + 1, INF);
    dist[1] = A[1];
    priority_queue<pair<i64, int>, vector<pair<i64, int>>, greater<pair<i64, int>>> pq;
    pq.emplace(dist[1], 1);

    while (!pq.empty()) {
        auto [current_dist, u] = pq.top();
        pq.pop();
        if (current_dist > dist[u]) continue;
        for (auto [v, b] : g[u]) {
            i64 new_dist = current_dist + b + A[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.emplace(new_dist, v);
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " \n"[i == N];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
