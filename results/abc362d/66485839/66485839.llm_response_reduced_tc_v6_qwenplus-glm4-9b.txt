#include <bits/stdc++.h>
using namespace std;
#define INT signed long long 

template <class T>
using V = vector<T>;
template <class T>
using VV = vector<vector<T>>;

struct Edge {
    int x, y, c;
    bool operator < (const Edge& other) const {
        return c < other.c;
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    V<int> a(n + 1);
    VV<int> g(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        g[u].push_back({v, b});
        g[v].push_back({u, b});
    }

    // Kruskal's algorithm to find the Minimum Spanning Tree (MST)
    sort(g[1].begin(), g[1].end());
    V<int> parent(n + 1);
    fill(parent.begin(), parent.end(), -1);
    V<int> size(n + 1, 1);
    int total_weight = 0;
    V<int> mst_weight(n + 1);
    for (auto& [v, c] : g[1]) {
        if (find_set(v, parent, size) != find_set(1, parent, size)) {
            union_set(1, v, parent, size);
            mst_weight[v] = a[1] + c;
            total_weight += a[v] + c;
        }
    }

    // Calculate the minimum weight from vertex 1 to all other vertices
    V<int> min_weight(n + 1, INT_MAX);
    min_weight[1] = 0;
    for (int i = 1; i <= n; ++i) {
        if (parent[i] == -1) {
            min_weight[i] = min(min_weight[i], a[i]);
        } else {
            min_weight[i] = min(min_weight[i], mst_weight[i] + min_weight[parent[i]]);
        }
    }

    // Output the results for each vertex
    for (int i = 2; i <= n; ++i) {
        cout << min_weight[i] << (i == n ? "\n" : " ");
    }
}

int find_set(int x, V<int>& parent, V<int>& size) {
    if (parent[x] != x) {
        parent[x] = find_set(parent[x], parent, size);
    }
    return parent[x];
}

void union_set(int x, int y, V<int>& parent, V<int>& size) {
    int rootX = find_set(x, parent, size);
    int rootY = find_set(y, parent, size);
    if (rootX != rootY) {
        if (size[rootX] < size[rootY]) {
            parent[rootX] = rootY;
        } else if (size[rootX] > size[rootY]) {
            parent[rootY] = rootX;
        } else {
            parent[rootY] = rootX;
            size[rootX]++;
        }
    }
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
