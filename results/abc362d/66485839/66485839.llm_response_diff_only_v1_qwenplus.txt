#include <bits/stdc++.h>
using namespace std;

using i64 = long long;

struct Node {
    int x;
    i64 w;
    bool operator < (const Node& u) const {
        return w > u.w; // Min-heap: smaller weight has higher priority
    }
};

constexpr int N = 200010;
vector<pair<int, int>> g[N];

void solve() {
    int n, m;
    cin >> n >> m;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }

    vector<i64> dist(n + 1, numeric_limits<i64>::max());
    vector<bool> visited(n + 1, false);
    priority_queue<Node> pq;

    // Start from vertex 1: initial cost is just the vertex weight A[1]
    dist[1] = a[1];
    pq.push({1, dist[1]});

    while (!pq.empty()) {
        auto [u, d] = pq.top();
        pq.pop();

        if (visited[u]) continue;
        visited[u] = true;

        for (auto [v, w] : g[u]) {
            i64 new_cost = dist[u] + w + a[v]; // current path cost + edge weight + destination vertex weight
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({v, dist[v]});
            }
        }
    }

    // Output distances for vertices 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i];
        if (i < n) cout << ' ';
        else cout << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();

    return 0;
}
