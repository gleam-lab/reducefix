#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
constexpr int MAXN = 2e5 + 5;
constexpr i64 INF = 1e18;

vector<pair<int, i64>> g[MAXN];
i64 d[MAXN], A[MAXN];

void solve() {
    int N, M;
    cin >> N >> M;
    
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
        d[i] = INF;
    }

    for (int i = 0; i < M; ++i) {
        int u, v;
        i64 w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    priority_queue<pair<i64, int>, vector<pair<i64, int>>, greater<>> pq;
    d[1] = A[1];
    pq.emplace(d[1], 1);

    while (!pq.empty()) {
        auto [dist_u, u] = pq.top();
        pq.pop();

        if (d[u] < dist_u) continue;

        for (auto& [v, cost] : g[u]) {
            i64 new_dist = d[u] + cost + A[v];
            if (new_dist < d[v]) {
                d[v] = new_dist;
                pq.emplace(d[v], v);
            }
        }
    }

    for (int i = 2; i <= N; ++i)
        cout << d[i] << " ";
    cout << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
