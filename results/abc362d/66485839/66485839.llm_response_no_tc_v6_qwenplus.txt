#include <bits/stdc++.h>
using namespace std;

#define INT signed
#define pbk push_back
template <class T>
using V = vector<T>;

struct Node {
    int x;
    long long w;
    bool operator < (const Node& u) const {
        return w > u.w; // Min-heap: smaller weight has higher priority
    }
};

constexpr int MAX_N = 200000;
vector<pair<int, long long>> g[MAX_N + 1];

void solve() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n + 1);
    vector<long long> d(n + 1, LLONG_MAX);
    vector<bool> vis(n + 1, false);

    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    for (int i = 0; i < m; ++i) {
        int x, y;
        long long c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }

    priority_queue<Node> pq;
    d[1] = a[1]; // Start with vertex weight at vertex 1
    pq.push({1, d[1]});

    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();

        if (vis[x]) continue;
        vis[x] = true;

        for (const auto& [y, edge_weight] : g[x]) {
            long long new_cost = d[x] + edge_weight + a[y];
            if (new_cost < d[y]) {
                d[y] = new_cost;
                pq.push({y, d[y]});
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << d[i];
        if (i < n) cout << ' ';
        else cout << '\n';
    }
}

INT main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();

    return 0;
}
