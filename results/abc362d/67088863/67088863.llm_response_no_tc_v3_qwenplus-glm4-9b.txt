#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

const int N = 2 * 100000 + 5;
const long long INF = 1e18;

int n, m;
long long vertex_weights[N];
vector<pair<int, long long>> adj[N]; // adj[i] will store pairs of (neighbor, edge_weight + vertex_weight)

void solve() {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq; // priority queue to store (current_distance, vertex)
    vector<long long> min_distance(n, INF); // to store the minimum distance from vertex 1 to all other vertices

    // Read number of vertices and edges
    cin >> n >> m;

    // Read vertex weights
    for (int i = 0; i < n; ++i) {
        cin >> vertex_weights[i];
    }

    // Read edges
    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        u--; v--; // Convert to 0-based index
        adj[u].push_back({v, b});
        adj[v].push_back({u, b}); // Graph is undirected
    }

    // Initialize the priority queue with the first vertex
    pq.push({0, 0});
    min_distance[0] = 0; // Distance from vertex 1 to itself is 0

    while (!pq.empty()) {
        auto [current_distance, current_vertex] = pq.top();
        pq.pop();

        // If we have already visited this vertex, skip it
        if (current_distance > min_distance[current_vertex]) continue;

        // Explore the neighbors of the current vertex
        for (auto &neighbor : adj[current_vertex]) {
            int next_vertex = neighbor.first;
            long long edge_weight = neighbor.second;

            // Calculate the next distance to the neighbor
            long long next_distance = current_distance + vertex_weights[next_vertex] + edge_weight;

            // If we find a shorter path to the neighbor, update our distance and push it to the priority queue
            if (next_distance < min_distance[next_vertex]) {
                min_distance[next_vertex] = next_distance;
                pq.push({min_distance[next_vertex], next_vertex});
            }
        }
    }

    // Output the minimum distances from vertex 1 to all other vertices
    for (int i = 1; i < n; ++i) {
        cout << min_distance[i] << " ";
    }
    cout << endl;
}

int main() {
    // Read the number of test cases
    int t;
    cin >> t;

    while (t--) {
        solve();
    }

    return 0;
}
