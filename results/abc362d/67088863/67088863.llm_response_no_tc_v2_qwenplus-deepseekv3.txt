const ll INF = 1e18;

void solve() {

    vector<ll> A(n);
    for (ll &a : A) cin >> a;

    vector<vector<pair<int, ll>>> adj(n);
    for (int i = 0; i < m; i++) {

        u--, v--;
        adj[u].emplace_back(v, b + A[v]);
        adj[v].emplace_back(u, b + A[u]);
    }

    vector<ll> dist(n, INF);
    dist[0] = A[0];
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    pq.emplace(dist[0], 0);

        pq.pop();
        if (current_dist > dist[u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.emplace(dist[v], v);
            }
        }
    }

    for (int i = 1; i < n; i++) {
        cout << dist[i] << (i == n - 1 ? "\n" : " ");
    }
