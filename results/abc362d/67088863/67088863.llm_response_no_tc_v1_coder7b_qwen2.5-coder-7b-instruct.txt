// Function to perform Dijkstra's algorithm
void dijkstra(const vector<vector<pair<int, ll>>>& adj, vector<ll>& dist, int start) {
    dist[start] = 0;
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        auto [w, v] = pq.top();
        pq.pop();
        if (w > dist[v]) continue; // Skip outdated entries

        for (auto [u, ww] : adj[v]) {
            if (ww + dist[v] < dist[u]) {
                dist[u] = ww + dist[v];
                pq.push({dist[u], u});
            }
        }
    }
}
