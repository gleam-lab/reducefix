vector<vector<pair<int, ll>>> adj(N);

void dijkstra(int start) {
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});
    dist[start] = 0;

    while (!pq.empty()) {
        auto [w, v] = pq.top();
        pq.pop();

        if (dist[v] < w) continue;

        for (auto [u, weight] : adj[v]) {
            ll new_dist = dist[v] + weight;
            if (new_dist < dist[u]) {
                dist[u] = new_dist;
                pq.push({new_dist, u});
            }
        }
    }
}
