#include <iostream>

void solve() {

    vector<ll> val(n);
    for (ll &i : val) cin >> i;
    
    vector<vector<pair<int, ll>>> adj(n);

        cin >> a >> b >> w;

        a--, b--;
        adj[a].push_back({b, w});
        adj[b].push_back({a, w});
    }
    
    vector<ll> dist(n, INF);
    dist[0] = 0;
    vector<bool> vis(n, 0);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, less<pair<ll, int>>> pq;
    pq.push({0, 0});

    while (!pq.empty()) {
        auto [w, v] = pq.top();
        pq.pop();
        if (vis[v]) continue;
        vis[v] = 1;

        for (auto [u, w] : adj[v]) {
            if (w + dist[v] < dist[u]) {
                dist[u] = w + dist[v];
                pq.push({dist[u], u});
            }
        }
