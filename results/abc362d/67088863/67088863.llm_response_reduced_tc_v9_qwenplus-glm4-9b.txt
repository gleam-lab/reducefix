#include <iostream>
#include <vector>
#include <queue>
#include <limits>

using namespace std;

const int N = 200005;
int n, m;
vector<pair<int, int>> adj[N]; // pair of (neighbor, edge weight + vertex value)

void solve() {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    long long dist[N]; // minimum distance from vertex 1 to i
    fill(dist, dist + N, numeric_limits<long long>::max());
    
    // Initialize the priority queue with vertex 1, with distance 0
    pq.push({0, 0});
    dist[0] = 0;

    while (!pq.empty()) {
        long long d = pq.top().first;
        int v = pq.top().second;
        pq.pop();

        if (d > dist[v]) {
            continue; // skip if we have already found a shorter path
        }

        // Process all adjacent vertices
        for (auto &edge : adj[v]) {
            int to = edge.first;
            long long edge_weight = edge.second;

            // If the new distance is smaller, update it and push to the queue
            if (d + edge_weight < dist[to]) {
                dist[to] = d + edge_weight;
                pq.push({dist[to], to});
            }
        }
    }

    // Output the minimum distances from vertex 1 to all other vertices
    for (int i = 1; i < n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
}

int main() {
    cin >> n >> m;
    
    int u, v, w;
    for (int i = 0; i < n; ++i) {

        for (int j = 0; j < adj[i].size(); ++j) {
            adj[i][j].first--; // Convert to 0-based index for internal processing

        adj[u].push_back({v, w});

    }

    solve();
    return 0;
