#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],dis2[200005],a[200005],n,m,vis[200005];

// Function to perform Dijkstra's algorithm from source node s
void dij(int s) {
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> q;
    q.push({a[s],s});
    while (!q.empty()) {
        auto [dist, u] = q.top(); q.pop();
        if (vis[u]) continue;
        vis[u] = 1;
        for (auto [v, w] : G[u]) {
            if (dis[v] > dist + a[v] + w) {
                dis[v] = dist + a[v] + w;
                q.push({dis[v], v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; ++i) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    
    // Initialize distances with maximum possible value
    memset(dis, 0x3f, sizeof dis);
    dis[1] = a[1]; // Distance from source to itself is the node's own weight
    
    // Perform Dijkstra's algorithm starting from node 1
    dij(1);

    // Print the results for nodes from 2 to n
    for (int i = 2; i <= n; ++i) {
        printf("%lld ", dis[i]);
    }
    return 0;
}
