#include<bits/stdc++.h>
using namespace std;
#define int long long
const int INF = 1e18;
vector<pair<int, int>> G[200005];
int dist[200005], dist2[200005], a[200005], n, m;

void dijkstra(int s) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, s});
    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();
        if (dist[u] < d) continue;
        for (auto edge : G[u]) {
            int v = edge.first;
            int weight = edge.second;
            int newDist = d + a[v] + weight;
            if (dist[v] > newDist) {
                dist[v] = newDist;
                pq.push({newDist, v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; ++i) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    memset(dist, INF, sizeof(dist));
    dist[1] = 0;
    dijkstra(1);
    for (int i = 2; i <= n; ++i) {
        printf("%lld ", dist[i]);
    }
    printf("\n");
    return 0;
}
