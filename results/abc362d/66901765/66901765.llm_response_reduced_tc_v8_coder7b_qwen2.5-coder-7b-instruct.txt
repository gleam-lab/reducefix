#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,pair<int,int>>>G;
int dis[200005],a[200005],n,m;
void dij(){
	priority_queue<pair<int,pair<int,int>>,vector<pair<int,pair<int,int>>>,greater<pair<int,pair<int,int>>>>q;
	q.push({a[1],{1,-1}});
	while(!q.empty()){
		auto [d,{u,p}]=q.top(); q.pop();
		if(d>dis[u]) continue;
		if(u==n) break;
		for(auto [v,w]:G){
			if(v==p||u==v) continue;
			int nd=d+a[v]+w;
			if(nd<dis[v]){
				dis[v]=nd;
				q.push({nd,{v,u}});
			}
		}
	}
}

signed main() {
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=m;i++){
		int u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		G.push_back({u,{v,w}});
		G.push_back({v,{u,w}});
	}
	memset(dis,0x3f,sizeof dis);
	dis[1]=a[1];
	dij();
	for(int i=2;i<=n;i++){
		cout<<dis[i]<<' ';
	}
	cout<<"\n";
	return 0;
}
