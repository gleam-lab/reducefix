```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
const ll INF = LLONG_MAX;

struct Edge {
    int to, weight;
};

vector<Edge> G[200005];
ll dis[200005], a[200005], n, m;

void dijkstra(int start) {
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    fill(dis, dis + n + 1, INF);
    dis[start] = a[start];
    pq.push({dis[start], start});

    while (!pq.empty()) {
        auto [dist, u] = pq.top(); pq.pop();
        if (dist > dis[u]) continue;

        for (auto &e : G[u]) {
            int v = e.to, w = e.weight;
            if (dis[u] + w + a[v] < dis[v]) {
                dis[v] = dis[u] + w + a[v];
                pq.push({dis[v], v});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    scanf("%lld %lld", &n, &m);
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
    }

    for (int i = 0; i < m; ++i) {
        int u, v, w;
        scanf("%lld %lld %lld", &u, &v, &w);
        G[u].emplace_back(v, w);
        G[v].emplace_back(u, w);
    }

    dijkstra(1);

    for (int i = 2; i <= n; ++i) {
        printf("%lld ", dis[i]);
    }
    puts("");

    return 0;
}
```

This solution uses Dijkstra's algorithm to find the shortest paths from vertex 1 to all other vertices in the graph. It handles the constraints and edge cases appropriately, ensuring the solution is both correct and efficient.
