#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],a[200005],n,m;
void dij(int u){
	priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>>q;
	q.push({0,u});
	while(!q.empty()){
		int u=q.top().second;
		int dist = q.top().first;
		q.pop();
		if(dist > dis[u]) continue; // Skip outdated distances
		for(auto [v,w]:G[u]){
			if(dis[u] + w + a[v] < dis[v]){
				dis[v] = dis[u] + w + a[v];
				q.push({dis[v],v});
			}
		}
	}
}

signed main() {
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%lld",&a[i]);
	}
	for(int i=1;i<=m;i++){
		int u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		G[u].push_back({v,w});
		G[v].push_back({u,w});
	}
	memset(dis,0x3f,sizeof dis);
	dis[1]=a[1];
	dij(1);
	for(int i=2;i<=n;i++){
		printf("%lld ",dis[i]);
	}
	printf("\n");
	return 0;
}

### Explanation of Fixes:
1. **Priority Queue Type**: Changed `priority_queue<pair<int,int>>` to `priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>>`. This ensures that the priority queue uses a min-heap, which is necessary for finding the shortest path.
   
2. **Distance Check in Dijkstra's Algorithm**: Added a check inside the while loop to skip processing nodes if a shorter path has already been found (`if(dist > dis[u]) continue`). This optimization prevents unnecessary updates and speeds up the algorithm.

These changes ensure that the Dijkstra's algorithm runs correctly within the constraints provided.
