#include <bits/stdc++.h>
using namespace std;
#define int long long
#define INF 9e18
vector<pair<int, int>> G[200005];
int dis[200005], vis[200005];
int a[200005], n, m;

void dij(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});
    dis[start] = a[start]; // Initialize the distance to the start node as its own vertex weight
    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();
        if (vis[u]) continue;
        vis[u] = 1;
        for (auto &[v, w] : G[u]) {
            if (!vis[v]) {
                int nextDis = dis[u] + w + a[v]; // Calculate the distance passing through vertex v
                if (nextDis < dis[v]) { // If this new distance is smaller, update it
                    dis[v] = nextDis;
                    pq.push({dis[v], v});
                }
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    memset(dis, INF, sizeof(dis));
    memset(vis, 0, sizeof(vis));
    dij(1); // Start Dijkstra's from vertex 1
    for (int i = 2; i <= n; i++) {
        cout << dis[i] << " "; // Output the minimum distance for each i from 2 to n
    }
    cout << "\n";
    return 0;
}
