#include<bits/stdc++.h>
using namespace std;

#define int long long 

vector<pair<int, int>> G[200005]; // Adjacency list for the graph
int n, m, a[200005]; // a[i] is the weight of vertex i
int dis[200005]; // dis[i] will store the minimum weight of path from 1 to i

void dijkstra() {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq; // Min-heap for Dijkstra's algorithm
    pq.push({0, 1}); // Start from vertex 1
    dis[1] = a[1]; // Initialize the distance to vertex 1

    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();

        if (d > dis[u]) continue; // Skip if we already found a better path

        for (auto& [v, w] : G[u]) { // Explore neighbors
            int ed = d + a[v] + w; // Calculate edge weight
            if (ed < dis[v]) { // If a shorter path is found
                dis[v] = ed;
                pq.push({ed, v}); // Push this path to the priority queue
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m); // Read number of vertices and edges
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]); // Read vertex weights
    }
    for (int i = 1; i <= m; ++i) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w); // Read edges
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    memset(dis, 0x3f, sizeof(dis)); // Initialize distances with maximum int value
    dijkstra(); // Run Dijkstra's algorithm from vertex 1

    // Output the minimal weight of paths for vertices 2 through N
    for (int i = 2; i <= n; i++) {
        cout << dis[i] << ' ';
    }
    cout << '\n';
    return 0;
}
