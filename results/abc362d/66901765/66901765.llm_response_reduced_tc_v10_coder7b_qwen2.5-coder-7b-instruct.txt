#include<bits/stdc++.h>
using namespace std;
#define ll long long 
vector<pair<ll,ll>>G[200005];
ll dis[200005],a[200005],n,m;
void dij(){
    priority_queue<pair<ll,ll>, vector<pair<ll,ll>>, greater<pair<ll,ll>>> pq;
    pq.push({a[1],1});
    while(!pq.empty()){
        auto [d,u] = pq.top(); pq.pop();
        if(d > dis[u]) continue;
        for(auto [v,w]: G[u]){
            ll nd = d + w + a[v];
            if(nd < dis[v]){
                dis[v] = nd;
                pq.push({nd,v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=n;i++){
        scanf("%lld",&a[i]);
        dis[i] = LLONG_MAX;
    }
    for(int i=1;i<=m;i++){
        int u,v,w;
        scanf("%lld%lld%lld",&u,&v,&w);
        G[u].push_back({v,w});
        G[v].push_back({u,w});
    }
    dis[1] = a[1];
    dij();
    for(int i=2;i<=n;i++){
        printf("%lld ",dis[i]);
    }
    return 0;
}
