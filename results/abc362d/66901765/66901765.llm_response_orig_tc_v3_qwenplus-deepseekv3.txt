3. **Visited Array**: Mark a vertex as visited only after it has been processed to ensure we do not process it multiple times.

const int INF = 1e18;

vector<pair<int,int>> G[200005];
int dis[200005];
int a[200005];
int n, m;

void dij(int u) {
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> q;
    q.push({a[u], u});
    dis[u] = a[u];
    while (!q.empty()) {
        int u = q.top().second;
        int current_dis = q.top().first;
        q.pop();
        if (current_dis > dis[u]) continue;
        for (auto [v, w] : G[u]) {
            if (dis[v] > dis[u] + a[v] + w) {
                dis[v] = dis[u] + a[v] + w;
                q.push({dis[v], v});
            }
        }
    }
}
