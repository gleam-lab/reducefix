#include<bits/stdc++.h>
using namespace std;
#define int long long

// Each adjacency list entry will store {target node, edge weight}
vector<vector<pair<int, int>>> graph(200005);

// Distance array for Dijkstra's algorithm
int dist[200005];
int A[200005]; // Vertex weights

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    
    // Start from node 1 with distance equal to its vertex weight
    dist[start] = A[start];
    pq.push({dist[start], start});
    
    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();
        
        // If we've already found a better path, skip
        if (d > dist[u]) continue;
        
        for (auto [v, edge_weight] : graph[u]) {
            // Path cost includes:
            // - current distance to u
            // - edge weight
            // - vertex weight of v
            int new_dist = dist[u] + edge_weight + A[v];
            
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({dist[v], v});
            }
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    for (int i = 1; i <= N; i++) {
        cin >> A[i];
        dist[i] = numeric_limits<int>::max(); // Initialize distances to infinity
    }
    
    for (int i = 0; i < M; i++) {
        int u, v, b;
        cin >> u >> v >> b;
        graph[u].push_back({v, b});
        graph[v].push_back({u, b});
    }
    
    dijkstra(1);
    
    for (int i = 2; i <= N; i++) {
        cout << dist[i] << " ";
    }
    cout << "\n";
    
    return 0;
}
