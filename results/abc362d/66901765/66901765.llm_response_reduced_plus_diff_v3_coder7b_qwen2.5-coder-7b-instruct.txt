struct Edge {

};

ll dis[200005];

void dijkstra(int n, int s) {
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
    fill(dis, dis + n + 1, INF);
    dis[s] = 0;
    pq.push({0, s});

    while (!pq.empty()) {
        auto [dist_u, u] = pq.top(); pq.pop();

        if (dist_u > dis[u]) continue;

        for (auto& [v, w] : G[u]) {
            if (dis[u] + w < dis[v]) {
                dis[v] = dis[u] + w;
                pq.push({dis[v], v});
            }
        }
    }
}
