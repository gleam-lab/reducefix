```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,int>>G[200005];
int dis[200005],dis2[200005],a[200005],n,m,vis[200005];
void dij(int u){
    priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> q;
    q.push({0,u});
    while(!q.empty()){
        auto [d,node] = q.top(); q.pop();
        if(d > dis[node]) continue;
        for(auto [v,w]: G[node]){
            if(dis[v] > d + a[v] + w){
                dis[v] = d + a[v] + w;
                q.push({dis[v], v});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=n;i++){
        scanf("%lld",&a[i]);
    }
    for(int i=1;i<=m;i++){
        int u,v,w;
        scanf("%lld%lld%lld",&u,&v,&w);
        G[u].push_back({v,w});
        G[v].push_back({u,w});
    }
    memset(dis,0x3f, sizeof dis);
    dis[1] = a[1];
    dij(1);
    for(int i=2;i<=n;i++){
        cout<<dis[i]<<' ';
    }

    return 0;
}
```

This solution uses Dijkstra's algorithm to find the shortest path from node 1 to all other nodes in the graph. It initializes a distance array `dis` with a large value (`0x3f...`) to represent infinity and sets the distance to the starting node (node 1) to its own weight. Then it iteratively selects the unvisited node with the smallest distance, updates the distances of its neighbors if a shorter path is found, and marks it as visited. This process continues until all nodes have been visited or all reachable nodes have had their shortest paths calculated. The output is the minimum weight of a path from vertex 1 to each vertex from 2 to N.
