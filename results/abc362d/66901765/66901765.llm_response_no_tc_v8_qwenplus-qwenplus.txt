#include <bits/stdc++.h>
using namespace std;
#define int long long

const int INF = 1e18;

vector<pair<int, int>> G[200005]; // {to, weight}
int dist[200005];
int A[200005];

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    dist[start] = A[start];
    pq.push({dist[start], start});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;

        for (auto [v, edge_weight] : G[u]) {
            int new_cost = d + edge_weight + A[v];
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({dist[v], v});
            }
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);

    int N, M;
    cin >> N >> M;
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
        dist[i] = INF;
    }

    for (int i = 0; i < M; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        G[u].push_back({v, b});
        G[v].push_back({u, b});
    }

    dijkstra(1);

    for (int i = 2; i <= N; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";

    return 0;
}
