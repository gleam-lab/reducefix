#include<bits/stdc++.h>
using namespace std;
#define int long long

vector<pair<int,int>> G[200005];
int dis[200005], a[200005], n, m;
const int INF = 9223372036854775807LL;

void dij(int start) {
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> q;
    for(int i = 1; i <= n; i++) {
        dis[i] = INF;
    }
    dis[start] = a[start];
    q.push({dis[start], start});
    
    while(!q.empty()) {
        auto [d, u] = q.top();
        q.pop();
        
        if(d > dis[u]) continue;
        
        for(auto [v, w] : G[u]) {
            int new_dist = dis[u] + w + a[v];
            if(new_dist < dis[v]) {
                dis[v] = new_dist;
                q.push({dis[v], v});
            }
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n >> m;
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    for(int i = 1; i <= m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    
    dij(1);
    
    for(int i = 2; i <= n; i++) {
        cout << dis[i] << ' ';
    }
    cout << '\n';
    
    return 0;
}
