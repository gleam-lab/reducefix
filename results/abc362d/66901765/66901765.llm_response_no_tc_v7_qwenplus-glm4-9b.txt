#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int, int>> G[200005];
int distA[200005], distB[200005], A[200005], N, M;
vector<pair<int, int>> adj[200005];

void dijkstra(int src) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({A[src], src});
    distA[src] = 0;

    while (!pq.empty()) {
        int node = pq.top().second;
        int dist = pq.top().first;
        pq.pop();

        if (dist > distA[node]) continue;

        for (auto& edge : adj[node]) {
            int nextNode = edge.first;
            int weight = edge.second + A[nextNode];
            if (weight < distB[nextNode]) {
                distB[nextNode] = weight;
                pq.push({weight, nextNode});
            }
        }
    }
}

signed main() {
    cin >> N >> M;
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }
    for (int i = 1; i <= M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].push_back({v, w});
        G[v].push_back({u, w});
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    memset(distB, 0x3f, sizeof(distB));
    dijkstra(1);

    vector<int> results(N);
    for (int i = 2; i <= N; ++i) {
        results[i - 2] = distB[i];
    }

    for (int i = 0; i < N - 2; ++i) {
        cout << results[i] << ' ';
    }
    cout << '\n';

    return 0;
}
