#include<bits/stdc++.h>
using namespace std;

const int INF = 1e18;
const int MAXN = 2e5 + 10;

struct Edge {
    int v, w;
};

vector<Edge> G[MAXN];
long long dist[MAXN];

void dijkstra(int s) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    fill(dist, dist + MAXN, INF);
    dist[s] = 0;
    pq.push({0, s});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();

        if (dist[u] != d) continue;

        for (auto &e : G[u]) {
            int v = e.v, w = e.w;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}
