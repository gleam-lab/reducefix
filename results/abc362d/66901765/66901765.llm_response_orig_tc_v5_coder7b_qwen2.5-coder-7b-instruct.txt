void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    fill(dist, dist + n + 1, INF);
    dist[start] = a[start];
    pq.push({dist[start], start});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (vis[u]) continue;
        vis[u] = 1;

        for (auto& e : G[u]) {
            int v = e.v, w = e.w;
            if (dist[u] + a[v] + w < dist[v]) {
                dist[v] = dist[u] + a[v] + w;
                pq.push({dist[v], v});
            }
        }
    }
}
