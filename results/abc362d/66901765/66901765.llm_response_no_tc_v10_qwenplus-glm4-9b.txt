#include<bits/stdc++.h>
using namespace std;
#define int long long

vector<pair<int, int>> G[200005];
int dist[200005], a[200005], n, m;

void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});
    dist[start] = a[start]; // Add the vertex weight to the initial distance

    while (!pq.empty()) {
        int current_dist = pq.top().first;
        int current_vertex = pq.top().second;
        pq.pop();

        if (current_dist > dist[current_vertex]) continue;

        for (auto &edge : G[current_vertex]) {
            int next_vertex = edge.first;
            int weight = edge.second;
            int new_dist = dist[current_vertex] + weight;

            if (new_dist < dist[next_vertex]) {
                dist[next_vertex] = new_dist;
                pq.push({new_dist, next_vertex});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }

    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    memset(dist, 0x3f, sizeof(dist)); // Initialize distances to infinity

    dijkstra(1); // Run Dijkstra's from vertex 1

    for (int i = 2; i <= n; i++) {
        cout << dist[i] << ' ';
    }
    cout << '\n';

    return 0;
}

### Explanation:
- The corrected code uses Dijkstra's algorithm to find the shortest paths from vertex 1 to all other vertices.
- The priority queue is used to always expand the least distant vertex first.
- The distance to each vertex is initialized with the weight of the vertex itself.
- The algorithm updates the distances to all neighboring vertices if a shorter path is found.
- The output format is adjusted to meet the problem's requirements.
