```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long 
vector<pair<int,pair<int, int>>>G[200005]; // Pair of (neighbor, weight)
int dis[200005], a[200005], n, m;

// Dijkstra's algorithm with a priority queue to find the shortest path from the starting vertex to all other vertices
void dijkstra(int start) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq; // Min-heap
    pq.push({0, start});
    for (int i = 1; i <= n; i++) dis[i] = a[i] + 1e18; // Initialize distances to a large number
    dis[start] = a[start]; // Distance to itself is just its own weight

    while (!pq.empty()) {
        int current_weight = pq.top().first;
        int current_vertex = pq.top().second;
        pq.pop();

        if (dis[current_vertex] < current_weight) continue;

        for (auto edge : G[current_vertex]) {
            int next_vertex = edge.first;
            int edge_weight = edge.second;

            if (dis[current_vertex] + edge_weight < dis[next_vertex]) {
                dis[next_vertex] = dis[current_vertex] + edge_weight;
                pq.push({dis[next_vertex], next_vertex});
            }
        }
    }
}

signed main() {
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%lld%lld%lld", &u, &v, &w);
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }
    
    dijkstra(1); // Run Dijkstra's from vertex 1

    // Output the minimum weight path distances to all other vertices except the first one
    for (int i = 2; i <= n; i++) {
        cout << dis[i] << " ";
    }
    cout << "\n";

    return 0;
}
```

This code uses Dijkstra's algorithm to find the shortest paths from the starting vertex (vertex 1) to all other vertices. It initializes the distances to all vertices as a very large number except the starting vertex, then iteratively updates the shortest distances based on the edges and their weights. After computing all shortest paths, it outputs the distances to all vertices except the starting vertex (vertex 1). The priority queue (min-heap) is used to efficiently retrieve the next vertex to process based on the current shortest distance.
