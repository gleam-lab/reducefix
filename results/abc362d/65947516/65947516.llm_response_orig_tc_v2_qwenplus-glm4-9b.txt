The problem statement is asking you to output the minimum weight of a path from vertex 1 to all other vertices except vertex 1. Here are the issues in your code:

1. You incorrectly added the vertex weight `a[i]` to the total distance in the output loop. Since you already added the edge weight `B_j` in the `heap_dijkstra` function, you should not add `a[i]` again.

2. You are not specifying edge weights correctly in the priority queue. The distance should be the sum of the edge weight and the vertex weight of the destination vertex.

const int N=2e5+5;

ll dis[N];
bool vis[N];
vector<pii> g[N]; // Change to pair<int, ll> for edge weight

struct node {
    int id;
    ll dis;
    bool operator<(const node& x)const {
        return dis > x.dis; // Invert comparison since priority queue is a min-heap
    }
};

void heap_dijkstra() {
    memset(vis, false, sizeof(vis));
    for (int i = 1; i <= n; i++)
        dis[i] = LLONG_MAX;
    dis[1] = 0; // Start from vertex 1 with 0 distance
    priority_queue<node> q;
    q.push({1, 0});
    while (!q.empty()) {
        node now = q.top();
        q.pop();
        if (vis[now.id]) continue;
        vis[now.id] = true;
        for (pii i : g[now.id]) { // Use edge weight from the vector
            if (dis[i.first] > dis[now.id] + i.second + a[i.first]) {
                dis[i.first] = dis[now.id] + i.second + a[i.first];
                if (!vis[i.first])
                    q.push({i.first, dis[i.first]});
            }
        }
    }
}
