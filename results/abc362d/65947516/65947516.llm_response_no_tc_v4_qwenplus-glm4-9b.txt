void dijkstra() {
    // Initialize all distances to a large number
    memset(dis, 0x3f, sizeof(dis)); // Use 0x3f for initializing to INT_MAX
    priority_queue<pii, vector<pii>, greater<pii>> pq; // Max-heap based on distance
    dis[1] = a[1]; // Start from vertex 1
    pq.push({-dis[1], 1}); // Insert with negative distance to simulate max-heap

    while (!pq.empty()) {
        int cur = pq.top().second;
        ll cur_dis = -pq.top().first;
        pq.pop();

        if (vis[cur]) continue;
        vis[cur] = true;

        for (auto &edge : g[cur]) {
            int next = edge.first;
            ll weight = edge.second + a[next];
            if (dis[next] > dis[cur] + weight) {
                dis[next] = dis[cur] + weight;
                pq.push({-dis[next], next});
            }
        }
    }
}
