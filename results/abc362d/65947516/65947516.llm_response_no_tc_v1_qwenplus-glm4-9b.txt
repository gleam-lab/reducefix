```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
using namespace std;

const int N = 200005;
int n, m;
long long a[N];
vector<pair<int, long long>> adj[N]; // adj[u] will store pairs (v, weight), where v is the vertex and weight is the edge weight

// Custom comparator for priority queue
struct compare {
    bool operator()(pair<long long, int> p1, pair<long long, int> p2) {
        return p1.first > p2.first;
    }
};

void dijkstra(int start) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, compare> pq;
    long long dist[N + 1];
    fill(dist, dist + N + 1, numeric_limits<long long>::max());

    pq.push({0, start});
    dist[start] = a[start];

    while (!pq.empty()) {
        auto [currentDist, u] = pq.top();
        pq.pop();

        if (currentDist > dist[u]) continue;

        for (auto [v, weight] : adj[u]) {
            long long newDist = currentDist + weight + a[v];
            if (dist[v] > newDist) {
                dist[v] = newDist;
                pq.push({dist[v], v});
            }
        }
    }

    // Print the minimum distance from vertex 1 to all other vertices
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    cout << endl;
}

int main() {
    cin >> n >> m;
    long long start, end, weight;

    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    for (int i = 0; i < m; i++) {
        cin >> start >> end >> weight;
        adj[start].push_back({end, weight});
        adj[end].push_back({start, weight});
    }

    dijkstra(1);
    return 0;
}
```

This code uses Dijkstra's algorithm to find the shortest paths from vertex 1 to all other vertices. It uses a priority queue to efficiently retrieve the next vertex with the smallest distance. The graph is represented using an adjacency list, where each vertex has a vector of pairs representing its adjacent vertices and the weights of the edges connecting them. The main function reads the input, constructs the graph, and then calls the `dijkstra` function to compute and print the minimum weight of paths from vertex 1 to all other vertices.
