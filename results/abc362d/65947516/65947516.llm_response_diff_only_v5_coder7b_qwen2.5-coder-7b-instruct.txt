#include<queue>
#include<vector>
#include<cstdio>
#include<cstring>
#include<climits> 
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll,int> pii; // Change the type of dis to ll
const int N=2e5+5;
int n,m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pii>g[N];

struct cmp {
    bool operator()(const pii &x, const pii &y) {
        return x.first > y.first; // Corrected comparison
    }
};

void dijkstra() {
    memset(vis, false, sizeof(vis));
    for(int i=1;i<=n;i++) dis[i] = LONG_LONG_MAX;
    dis[1] = 0;
    priority_queue<pii, vector<pii>, cmp> pq;
    pq.push({0, 1});

    while(!pq.empty()) {
        auto [dist, u] = pq.top(); pq.pop();
        if(vis[u]) continue;
        vis[u] = true;

        for(auto [v, w] : g[u]) {
            if(dis[v] > dist + w + a[u]) {
                dis[v] = dist + w + a[u];
                pq.push({dis[v], v});
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for(int i=1;i<=n;i++) scanf("%lld", &a[i]);
    for(int i=1;i<=m;i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }

    dijkstra();

    for(int i=2;i<=n;i++) printf("%lld ", dis[i]);

    return 0;
}
