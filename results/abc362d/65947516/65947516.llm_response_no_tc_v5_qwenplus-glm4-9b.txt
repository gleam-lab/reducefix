
```cpp
#include<queue>
#include<vector>
#include<cstdio>
#include<cstring>
#include<climits> 
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
const int N = 2e5 + 5;
int n, m;
ll a[N];
ll dis[N];
vector<pii> g[N]; // Using vector instead of list for potentially better cache performance

void dijkstra(int src) {
    priority_queue<pii, vector<pii>, greater<pii>> pq; // Max-heap for storing vertices and their current distance
    memset(dis, LLONG_MAX, sizeof(dis)); // Initialize distances to a large number
    dis[src] = 0;
    pq.push({0, src}); // Push the source with 0 distance

    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();

        if (d > dis[u]) continue; // Skip if the popped distance is not the shortest

        for (auto &edge : g[u]) {
            int v = edge.first;
            int weight = edge.second;
            if (dis[v] > d + weight + a[v]) { // Calculate cost to v through u
                dis[v] = d + weight + a[v];
                pq.push({dis[v], v}); // Push v with the new distance
            }
        }
    }
}

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }

    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }

    dijkstra(1); // Compute shortest paths from vertex 1 to all other vertices

    for (int i = 2; i <= n; i++) {
        printf("%lld ", dis[i]);
    }
    return 0;
}
```

### Explanation of Changes:

1. Changed `list<pii>` to `vector<pii>` in the adjacency list for potentially better cache performance.
2. Used a max-heap (priority queue with `<` comparator) to store vertices and their current distance, allowing us to always process the vertex with the minimum distance first, which is a standard approach for Dijkstra's algorithm.
3. Initialized `dis` array with `LLONG_MAX` to guarantee that even very large initial distances are shorter than the actual distances found by Dijkstra's algorithm.
4. Added checks to ensure that we do not reprocess a vertex with a longer distance than already found.
5. Removed unnecessary comments and made variable names more descriptive to improve readability.
