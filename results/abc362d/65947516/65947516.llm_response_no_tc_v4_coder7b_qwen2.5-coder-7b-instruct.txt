#include<queue>
#include<vector>
#include<cstdio>
#include<cstring>
#include<climits> 
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll,int> pli; // Change to pair<ll, int> for distance and vertex
const int N=2e5+5;
int n,m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pli>g[N]; // Update vector type to store pairs of distances and vertices
struct node
{
	ll dis;
	int id;
	bool operator<(const node& x)const
	{
		return dis>x.dis;
	} 
};
void heap_dijkstra()
{
	memset(vis,false,sizeof(vis));
	for(int i=1;i<=n;i++)
		dis[i]=LONG_LONG_MAX;
	dis[1]=0;
	priority_queue<node>q;
	q.push({0,1}); // Start from vertex 1 with distance 0
	while(!q.empty())
	{
		node now=q.top();
		q.pop();
		if(vis[now.id])continue;
		vis[now.id]=true;
		for(pli i:g[now.id]) // Iterate over neighbors using updated pair type
			if(dis[i.second]>dis[now.id]+i.first+a[now.id])
			{
				dis[i.second]=dis[now.id]+i.first+a[now.id];
				if(!vis[i.second])
					q.push({dis[i.second],i.second});
			}
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%lld",&a[i]);
	for(int i=1;i<=m;i++)
	{
		int u,v,w;
		scanf("%d%d%d",&u,&v,&w);
		g[u].push_back({w,v}); // Store edge weight first, then vertex
		g[v].push_back({w,u});
	}
	heap_dijkstra();
	for(int i=2;i<=n;i++)

}
