#include <iostream>
#include <vector>
#include <queue>
#include <limits>

using namespace std;

const int MAXN = 2 * 100005;
vector<pair<int, int>> adj[MAXN];

// Function to perform Dijkstra's algorithm from source vertex
void dijkstra(int source, int n, vector<long long>& dist) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[source] = 0;
    pq.push({0, source});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();

        // Skip if already visited or distance is more than calculated
        if (dist[u] != d) continue;

        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<long long> a(n), dist(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        --u, --v; // Adjusting for 0-based indexing
        adj[u].emplace_back(v, b);
        adj[v].emplace_back(u, b);
    }

    // Perform Dijkstra's algorithm from vertex 0
    dijkstra(0, n, dist);

    // Print results for vertices 1 to n-1
    for (int i = 1; i < n; ++i) {
        cout << dist[i] + a[i] << " ";
    }

    return 0;
}

This solution uses Dijkstra's algorithm to find the shortest paths from vertex 1 to all other vertices in the graph. It then prints the result for each vertex from 2 to N. This approach ensures that we handle all edge cases correctly and efficiently, meeting the constraints provided.
