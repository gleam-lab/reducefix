// Corrected comparator function for priority queue

    bool operator() (const pii &p1, const pii &p2){
        return p1.first > p2.first;

void dijkstra()
{
    memset(vis,false,sizeof(vis));

        dis[i]=LLONG_MAX;

    priority_queue<pii,vector<pii>,cmp>q; // Using custom comparator

    while(!q.empty())

        }
    }
}
