#include <queue>
#include <vector>
#include <cstdio>
#include <cstring>
#include <climits>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll, int> pil; // Changed to use ll for distance
const int N = 2e5 + 5;

int n, m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pil> g[N]; // Changed second type to ll for edge weights

struct node {
    int id;
    ll dist;
    bool operator<(const node& x) const {
        return dist > x.dist;
    }
};

void heap_dijkstra() {
    memset(vis, false, sizeof(vis));
    for (int i = 1; i <= n; i++)
        dis[i] = LLONG_MAX; // Use LLONG_MAX for long long
    dis[1] = a[1]; // Include A_1 since we visit vertex 1
    priority_queue<node> q;
    q.push({1, dis[1]});
    
    while (!q.empty()) {
        node now = q.top();
        q.pop();
        if (vis[now.id]) continue;
        vis[now.id] = true;
        
        for (auto &edge : g[now.id]) {
            int neighbor = edge.first;
            ll edge_weight = edge.second;
            
            // When moving to neighbor, include the vertex weight of neighbor in its path
            if (!vis[neighbor] && dis[neighbor] > dis[now.id] + edge_weight + a[neighbor]) {
                dis[neighbor] = dis[now.id] + edge_weight + a[neighbor];
                q.push({neighbor, dis[neighbor]});
            }
        }
    }
}

int main() {
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%lld",&a[i]);
    for(int i=1;i<=m;i++) {
        int u, v;
        ll w;
        scanf("%d%d%lld",&u,&v,&w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }
    
    heap_dijkstra();
    
    for(int i=2;i<=n;i++)
        printf("%lld ", dis[i]); // The final result is already computed correctly
    
    return 0;
}
