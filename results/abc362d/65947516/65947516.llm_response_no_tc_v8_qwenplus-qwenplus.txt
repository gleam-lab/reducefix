#include <queue>
#include <vector>
#include <cstdio>
#include <cstring>
#include <climits>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll, int> pli;  // Changed to use ll for distance
const int N = 2e5 + 5;

int n, m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pli> g[N];  // Changed to store edge weights properly

void heap_dijkstra() {
    memset(vis, false, sizeof(vis));
    for (int i = 1; i <= n; i++)
        dis[i] = LLONG_MAX;  // Use LLONG_MAX for long long
    
    dis[1] = a[1];  // Start with vertex 1's weight
    priority_queue<pli, vector<pli>, greater<pli>> q;  // Min-heap based on distance
    q.push({dis[1], 1});
    
    while (!q.empty()) {
        pli now = q.top();
        q.pop();
        int u = now.second;
        
        if (vis[u]) continue;
        vis[u] = true;
        
        for (auto &edge : g[u]) {
            int v = edge.first;
            ll edge_weight = edge.second;
            
            // Calculate new path cost: current path cost + edge weight
            ll new_dist = dis[u] + edge_weight;
            
            if (!vis[v] && dis[v] > new_dist) {
                dis[v] = new_dist;
                q.push({dis[v], v});
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++)
        scanf("%lld", &a[i]);
    
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }
    
    heap_dijkstra();
    
    for (int i = 2; i <= n; i++) {
        printf("%lld ", dis[i]);  // No need to add A[i], already included in path cost
    }
    
    return 0;
}
