#include<queue>
#include<vector>
#include<cstdio>
#include<cstring>
#include<climits> 
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<int,pair<int, int>> pipii; // pair<to, pair<weight, vertex>>
const int N=2e5+5;
int n,m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pipii>g[N];

void dijkstra() {
    memset(vis, false, sizeof(vis));
    for(int i=1; i<=n; ++i) {
        dis[i] = LLONG_MAX;
    }
    dis[1] = 0;
    priority_queue<pipii, vector<pipii>, greater<pipii>> q;
    q.push({0, {0, 1}}); // Initially, we start from vertex 1 with 0 additional cost and 0 total cost

    while(!q.empty()) {
        auto cur = q.top();
        q.pop();

        int to = cur.second.first;
        int dist_from_prev = cur.second.second;
        int cost_from_prev = cur.first;

        if(vis[to]) continue;
        if(cost_from_prev > dis[to]) continue;

        vis[to] = true;
        for(auto& edge : g[to]) {
            int next = edge.first;
            int weight = edge.second;
            ll new_cost = cost_from_prev + weight + a[next];
            if(dis[next] > new_cost) {
                dis[next] = new_cost;
                q.push({new_cost, {next, dist_from_prev + 1}});
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for(int i=1; i<=n; ++i) {
        scanf("%lld", &a[i]);
    }
    for(int i=1; i<=m; ++i) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }
    dijkstra();
    for(int i=2; i<=n; ++i) {
        printf("%lld ", dis[i] + a[i]);
    }
    return 0;
}
