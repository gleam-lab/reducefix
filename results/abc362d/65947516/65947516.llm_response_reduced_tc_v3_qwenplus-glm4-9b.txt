#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

const int N = 2e5 + 5;

vector<pair<int, int>> graph[N]; // Graph as adjacency list
int n, m;
long long a[N];
long long min_dists[N]; // Minimum distance to each vertex from vertex 1

void dijkstra() {
    // Priority queue to store the minimum distance vertex
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;

    // Initialize the priority queue and min_dists array
    pq.push({0, 1});
    min_dists[1] = a[1];

    while (!pq.empty()) {
        auto [dist, u] = pq.top();
        pq.pop();

        if (min_dists[u] < dist) continue;

        for (auto &[v, weight] : graph[u]) {
            long long new_dist = dist + weight + a[v];

            // Only consider new distance if it's smaller
            if (min_dists[v] > new_dist) {
                min_dists[v] = new_dist;
                pq.push({new_dist, v});
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }
    dijkstra();

    // Output the minimum distances for all vertices 2 through n
    for (int i = 2; i <= n; i++) {
        cout << min_dists[i] << (i < n ? " " : "");
    }
    return 0;
}
