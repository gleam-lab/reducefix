#include <queue>
#include <vector>
#include <cstdio>
#include <cstring>
#include <climits>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll, int> pli;  // Changed to use ll for distance
typedef pair<int, ll> pil;
const int N = 2e5 + 5;

int n, m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pil> g[N];  // Pair of (node, weight)

struct node {
    int id;
    ll dist;
    bool operator<(const node& x) const {
        return dist > x.dist;
    }
};

void heap_dijkstra() {
    memset(vis, false, sizeof(vis));
    for (int i = 1; i <= n; i++)
        dis[i] = LLONG_MAX;  // Use LLONG_MAX instead of LONG_LONG_MAX for consistency
    
    dis[1] = a[1];  // Start with vertex 1's weight
    
    priority_queue<node> q;
    q.push({1, dis[1]});
    
    while (!q.empty()) {
        node now = q.top();
        q.pop();
        
        if (vis[now.id]) continue;
        vis[now.id] = true;
        
        for (auto edge : g[now.id]) {
            int next_node = edge.first;
            ll edge_weight = edge.second;
            
            ll new_dist = now.dist + edge_weight + a[next_node];
            
            if (new_dist < dis[next_node]) {
                dis[next_node] = new_dist;
                if (!vis[next_node])
                    q.push({next_node, dis[next_node]});
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++)
        scanf("%lld", &a[i]);
    
    for (int i = 1; i <= m; i++) {
        int u, v;
        ll w;
        scanf("%d%d%lld", &u, &v, &w);
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }
    
    heap_dijkstra();
    
    for (int i = 2; i <= n; i++) {
        printf("%lld ", dis[i]);
    }
    
    return 0;
}
