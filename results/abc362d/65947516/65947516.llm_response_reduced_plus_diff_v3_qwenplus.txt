#include<queue>
#include<vector>
#include<cstdio>
#include<cstring>
#include<climits> 
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll,int> pii;
const int N=2e5+5;
int n,m;
ll a[N];
ll dis[N];
bool vis[N];
vector<pii>g[N];

void dijkstra()
{
    memset(vis,false,sizeof(vis));
    for(int i=1;i<=n;i++)
        dis[i]=LLONG_MAX;
    dis[1]=a[1];
    priority_queue<pii,vector<pii>,greater<pii>>q;
    q.push({dis[1],1});
    while(!q.empty())
    {
        int now=q.top().second;
        q.pop();
        if(vis[now])continue;
        vis[now]=true;
        for(pii edge:g[now]) 
        {
            int v=edge.first;
            ll w=edge.second;
            if(dis[v]>dis[now]+w+a[v])
            {
                dis[v]=dis[now]+w+a[v];
                q.push({dis[v],v});
            }
        }
    }
}

int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%lld",&a[i]);
    for(int i=1;i<=m;i++)
    {
        int u,v,w;
        scanf("%d%d%d",&u,&v,&w);
        g[u].push_back({v,w});
        g[v].push_back({u,w});
    }
    dijkstra();
    for(int i=2;i<=n;i++)
        printf("%lld ",dis[i]); 
    return 0;
}
