#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

const int N = 3e5 + 10;

vector<pair<int, int>> graph[N]; // Graph to store edges with weights
long long vertex_weights[N];     // Weights of the vertices
long long min_path_weights[N];  // Minimum path weights to each vertex

void dijkstra(int start) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({vertex_weights[start], start});
    min_path_weights[start] = vertex_weights[start]; // Start by setting the starting vertex's weight to its vertex weight

    while (!pq.empty()) {
        auto [current_weight, u] = pq.top();
        pq.pop();

        if (current_weight > min_path_weights[u])
            continue;

        for (auto& [v, weight] : graph[u]) {
            long long next_weight = current_weight + weight + vertex_weights[v];
            if (next_weight < min_path_weights[v]) {
                min_path_weights[v] = next_weight;
                pq.push({min_path_weights[v], v});
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    cin >> vertex_weights[1]; // Special case for vertex 1's weight
    for (int i = 2; i <= n; ++i) { // Read vertex weights for other vertices
        cin >> vertex_weights[i];
    }

    for (int i = 0; i < m; ++i) { // Read edges
        int u, v, weight;
        cin >> u >> v >> weight;
        graph[u].push_back({v, weight});
        graph[v].push_back({u, weight});
    }

    dijkstra(1); // Run Dijkstra's algorithm from vertex 1

    // Output the minimum weights for paths from vertex 1 to every other vertex
    for (int i = 2; i <= n; ++i) {
        cout << min_path_weights[i] << (i == n ? '\n' : ' ');
    }

    return 0;
}

This code fixes the original code by using Dijkstra's algorithm to find the shortest paths from vertex 1 to all other vertices efficiently. The priority queue is used to ensure that the vertex with the smallest current path weight is processed next, guaranteeing the correctness of the Dijkstra's algorithm. The special case for vertex 1's weight is handled separately since the input format may not include vertex 1's weight if the graph is undirected and starts from vertex 2.
