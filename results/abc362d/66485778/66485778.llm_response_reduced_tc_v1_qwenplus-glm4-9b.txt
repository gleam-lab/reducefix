#include <bits/stdc++.h>
using namespace std;
#define INT long long 
#define pb push_back

template <class T>
using V = vector<T>;
template <class T>
using VV = vector<V<T>>;

struct DSU {
private:
    V<int> par, siz;
public:
    DSU(int n) : par(n), siz(n, 1) {
        iota(par.begin(), par.end(), 0);
    }
    int find(int x) {
        if (par[x] != x) {
            par[x] = find(par[x]);
        }
        return par[x];
    }
    void unite(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX != rootY) {
            par[rootX] = rootY;
            siz[rootY] += siz[rootX];
        }
    }
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    int size(int x) {
        return siz[find(x)];
    }
};

int N, M;
V<int> A;
V<V<pair<int, int>>> G;

void solve() {
    DSU dsu(N);
    for (auto& edge : G[1]) {
        dsu.unite(edge.first, edge.second);
    }

    priority_queue<pair<LONG_LONG, int>, V<pair<LONG_LONG, int>>, greater<pair<LONG_LONG, int>>> pq;
    pq.push({A[1] + 0, 1});
    V<LONG_LONG> dist(N + 1, LONG_LONG_MAX);
    dist[1] = A[1];

    while (!pq.empty()) {
        auto [d, node] = pq.top();
        pq.pop();

        if (d > dist[node]) continue;

        for (auto& edge : G[node]) {
            LONG_LONG weight = d + edge.second + A[edge.first];
            if (weight < dist[edge.first]) {
                dist[edge.first] = weight;
                pq.push({dist[edge.first], edge.first});
            }
        }
    }

    V<LONG_LONG> result(N);
    for (int i = 2; i <= N; ++i) {
        result[i - 2] = dist[i] - A[i];
    }
    for (int i = 0; i < N - 2; ++i) {
        cout << result[i] << (i < N - 3 ? " " : "");
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M;
    A.resize(N + 1);
    G.resize(N + 1);

    cin >> A[1];
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].push_back({v, w});
        G[v].push_back({u, w});
    }

    solve();

    return 0;
}
