
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <limits>

using namespace std;

struct Edge {
    int to;
    long long weight;
};

const int MAX_N = 200005;

int N, M;
vector<vector<Edge>> graph(MAX_N);
long long A[MAX_N];
priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;

void solve() {
    // First, add all vertices as edges to themselves with their own weights
    for (int i = 1; i <= N; ++i) {
        graph[i].push_back({i, A[i]});
    }

    // Start with vertex 1
    pq.push({A[1], 1});

    vector<long long> min_weight_to_vertex(N + 1, numeric_limits<long long>::max());
    min_weight_to_vertex[1] = A[1];

    while (!pq.empty()) {
        auto [weight, vertex] = pq.top();
        pq.pop();

        // If we've already found a better weight, skip processing
        if (weight > min_weight_to_vertex[vertex]) continue;

        for (auto& edge : graph[vertex]) {
            long long next_weight = weight + edge.weight;
            if (next_weight < min_weight_to_vertex[edge.to]) {
                min_weight_to_vertex[edge.to] = next_weight;
                pq.push({next_weight, edge.to});
            }
        }
    }

    // Output the minimum weights from vertex 1 to all other vertices
    for (int i = 2; i <= N; ++i) {
        cout << min_weight_to_vertex[i] << (i == N ? "" : " ");
    }
}

int main() {
    cin >> N >> M;
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }

    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }

    solve();
    return 0;
}
```
