#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
using pii = pair<int, int>;
using Edge = tuple<int, int, long long>;

constexpr int N = 200005;
vector<vector<pii>> g(N);
vector<long long> dist(N, numeric_limits<long long>::max());
vector<bool> vis(N, false);

void solve() {
    int n, m;
    cin >> n >> m;
    vector<long long> A(n + 1);
    for (int i = 1; i <= n; ++i) cin >> A[i];
    for (int i = 1; i <= m; ++i) {
        int u, v;
        long long c;
        cin >> u >> v >> c;
        g[u].emplace_back(v, c);
        g[v].emplace_back(u, c);
    }

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[1] = A[1];
    pq.push({dist[1], 1});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (vis[u]) continue;
        vis[u] = true;
        for (auto &[v, cost] : g[u]) {
            if (!vis[v] && dist[v] > d + cost + A[v]) {
                dist[v] = d + cost + A[v];
                pq.push({dist[v], v});
            }
        }
    }

    for (int i = 2; i <= n; ++i) cout << dist[i] << " ";
    cout << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
