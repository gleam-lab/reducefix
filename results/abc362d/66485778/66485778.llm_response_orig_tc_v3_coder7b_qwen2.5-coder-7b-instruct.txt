#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define INF LLONG_MAX

struct Edge {
    int v, w;
};

vector<Edge> adj[100001];

void dijkstra(int n, int src, vector<LL>& dist) {
    priority_queue<pair<LL, int>, vector<pair<LL, int>>, greater<pair<LL, int>>> pq;
    
    dist[src] = 0;
    pq.push({0, src});
    
    while (!pq.empty()) {
        int u = pq.top().second;
        LL d = pq.top().first;
        pq.pop();
        
        if (d > dist[u]) continue;
        
        for (auto& edge : adj[u]) {
            int v = edge.v;
            LL w = edge.w;
            
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}
