#include <bits/stdc++.h>
using namespace std;
#define int long long 

template <class T>
using V = vector<T>;

using i128 = __int128;

const int N = 3e5 + 10;

V<int> a(N + 1);
V<V<pair<int, int>>> g(N + 1);

int n, m;

// Union-Find with path compression and size tracking
struct DSU {
    V<int> parent;
    V<int> size;

    DSU(int n) : parent(n + 1), size(n + 1, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    bool unite(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return false;
        if (size[x] > size[y]) swap(x, y);
        parent[x] = y;
        size[y] += size[x];
        return true;
    }
};

void solve() {
    priority_queue<pair<i128, int>, V<pair<i128, int>>, greater<pair<i128, int>>> pq;
    V<i128> dist(N + 1, 1e18);
    V<bool> visited(N + 1, false);

    // Initialize the priority queue with the starting point
    pq.push({a[1], 1});
    dist[1] = a[1];

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (visited[u]) continue;
        visited[u] = true;

        for (auto& [v, w] : g[u]) {
            i128 new_dist = d + w + a[v];
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({dist[v], v});
            }
        }
    }

    // Output the minimum distances for each vertex 2 to N
    for (int i = 2; i <= n; i++) {
        cout << dist[i] << (i < n ? " " : "");
    }
    cout << endl;
}

int main() {
    cin >> n >> m;
    a.assign(n + 1);
    g.assign(n + 1);

    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }

    solve();
    return 0;
}
