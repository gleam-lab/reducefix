```cpp
#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
using pii = pair<int, int>;

struct Node {
    int node;
    i64 dist;
    bool operator<(const Node& other) const {
        return dist > other.dist;
    }
};

constexpr int MAXN = 2 * 1e5 + 5;
constexpr i64 INF = 1e18;

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<i64> A(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> A[i];
    }
    
    vector<vector<pair<int, i64>>> graph(n + 1);
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        i64 b;
        cin >> u >> v >> b;
        graph[u].emplace_back(v, b);
        graph[v].emplace_back(u, b);
    }
    
    priority_queue<Node> pq;
    vector<i64> minPath(n + 1, INF);
    vector<bool> visited(n + 1, false);
    
    // Start from node 1 with initial cost of A[1]
    minPath[1] = A[1];
    pq.push({1, A[1]});
    
    while (!pq.empty()) {
        Node current = pq.top();
        pq.pop();
        
        int u = current.node;
        if (visited[u]) continue;
        visited[u] = true;
        
        for (const auto& edge : graph[u]) {
            int v = edge.first;
            i64 edgeCost = edge.second;
            
            // Path cost to v through u: current path cost to u + edge weight + vertex weight of v
            i64 newCost = minPath[u] + edgeCost + A[v];
            
            if (newCost < minPath[v]) {
                minPath[v] = newCost;
                pq.push({v, newCost});
            }
        }
    }
    
    for (int i = 2; i <= n; ++i) {
        cout << minPath[i] << " ";
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    
    return 0;
}
```
