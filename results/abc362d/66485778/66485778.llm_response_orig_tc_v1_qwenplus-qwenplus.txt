#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
using pii = pair<int, int>;
using Edge = pair<int, int>;

const i64 INF = 1e18;

void solve2() {
    int N, M;
    cin >> N >> M;
    vector<i64> A(N + 1);
    for (int i = 1; i <= N; ++i) cin >> A[i];

    vector<vector<Edge>> G(N + 1);
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        G[u].emplace_back(v, w);
        G[v].emplace_back(u, w);
    }

    priority_queue<pair<i64, int>, vector<pair<i64, int>>, greater<>> pq;
    vector<i64> dist(N + 1, INF);
    dist[1] = A[1];
    pq.emplace(dist[1], 1);

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dist[u]) continue;
        for (auto& [v, w] : G[u]) {
            if (dist[v] > dist[u] + w + A[v]) {
                dist[v] = dist[u] + w + A[v];
                pq.emplace(dist[v], v);
            }
        }
    }

    for (int i = 2; i <= N; ++i)
        cout << dist[i] << " \n"[i == N];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve2();
    return 0;
}
