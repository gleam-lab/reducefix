#include <bits/stdc++.h>
using namespace std;

template <class T>
struct Edge {
    int to, cost;
    Edge(int to, T cost) : to(to), cost(cost) {}
};

template <class T>
struct Node {
    int node;
    T key;
    Node(int node, T key) : node(node), key(key) {}
};

struct compare {
   template<typename T1, typename T2>
    bool operator()(pair<T1, Node<T2>>& a, pair<T1, Node<T2>>& b) const {
        return a.first > b.first;
    }
};

constexpr int MAX_N = 3e5 + 10;

int N, M;
vector<Edge<int>> graph[MAX_N];
int vertex_weights[MAX_N];

void dijkstra(int source) {
    priority_queue<pair<int, Node<int>>, vector<pair<int, Node<int>>>, compare> pq;
    vector<long long> distances(N + 1, LLONG_MAX);

    pq.emplace(0, Node<int>(source, 0));
    distances[source] = 0;

    while (!pq.empty()) {
        auto [_, current_node] = pq.top();
        pq.pop();

        int node = current_node.node;
        long long dist = current_node.key;

        if (dist != distances[node]) continue;

        for (auto& edge : graph[node]) {
            int to = edge.to;
            long long edge_cost = edge.cost;

            if (dist + edge_cost < distances[to]) {
                distances[to] = dist + edge_cost;
                pq.emplace(distances[to], {to, distances[to]});
            }
        }
    }

    for (int i = 2; i <= N; ++i) {
        cout << distances[i] << " ";
    }
    cout << endl;
}

int main() {
    cin >> N >> M;

    for (int i = 1; i <= N; ++i) {
        cin >> vertex_weights[i];
    }

    int u, v, w;
    for (int i = 1; i <= M; ++i) {
        cin >> u >> v >> w;
        graph[u].emplace_back(v, w);
        graph[v].emplace_back(u, w);
    }

    dijkstra(1);

    return 0;
}
