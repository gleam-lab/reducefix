```cpp
#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
using pii = pair<int, int>;

struct Node {
    int node;
    i64 dist;
    bool operator<(const Node& other) const {
        return dist > other.dist;
    }
};

constexpr int MAXN = 2e5 + 10;
constexpr i64 INF = 9223372036854775807; // Use maximum representable i64 value

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<i64> A(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> A[i];
    }
    
    vector<vector<pii>> graph(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v, b;
        cin >> u >> v >> b;
        graph[u].emplace_back(v, b);
        graph[v].emplace_back(u, b);
    }
    
    vector<i64> dist(n + 1, INF);
    priority_queue<Node> pq;
    
    dist[1] = A[1]; // Start from vertex 1 with its weight
    pq.push({1, A[1]});
    
    while (!pq.empty()) {
        auto current = pq.top();
        pq.pop();
        
        int curr_node = current.node;
        i64 curr_dist = current.dist;
        
        if (curr_dist > dist[curr_node]) continue;
        
        for (auto [next_node, edge_weight] : graph[curr_node]) {
            i64 new_dist = curr_dist + edge_weight + A[next_node];
            
            if (new_dist < dist[next_node]) {
                dist[next_node] = new_dist;
                pq.push({next_node, new_dist});
            }
        }
    }
    
    for (int i = 2; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << endl;
}

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    solve();
    return 0;
}
```
