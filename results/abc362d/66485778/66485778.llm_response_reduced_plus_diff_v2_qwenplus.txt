#include <bits/stdc++.h>
using namespace std;

#define int long long
#define pbk push_back

template <class T>
using V = vector<T>;
template <class T>
using VV = vector<vector<T>>;

struct Node {
    int x, w;
    bool operator < (const Node& u) const {
        return w > u.w; // Min-heap: smaller weight has higher priority
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    V<int> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    vector<vector<pair<int, int>>> g(n + 1);
    for (int i = 0; i < m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }

    const int inf = 1e18;
    V<int> d(n + 1, inf);
    V<bool> vis(n + 1, false);
    priority_queue<Node> pq;

    d[1] = a[1];
    pq.push({1, d[1]});

    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;

        for (const auto& [y, edge_weight] : g[x]) {
            int new_cost = d[x] + edge_weight + a[y];
            if (new_cost < d[y]) {
                d[y] = new_cost;
                pq.push({y, d[y]});
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " \n"[i == n];
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();
    return 0;
}
