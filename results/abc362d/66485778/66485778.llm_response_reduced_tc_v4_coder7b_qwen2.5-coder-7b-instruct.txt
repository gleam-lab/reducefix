#include <bits/stdc++.h>
using namespace std;

#define ll long long

const int MAXN = 2e5 + 5;
const ll INF = LLONG_MAX / 2;

struct Edge {
    int v, w;
};

struct DSU {
    vector<int> parent, rank;

    DSU(int n) {
        parent.resize(n + 1);
        rank.resize(n + 1, 0);
        for (int i = 1; i <= n; ++i) parent[i] = i;
    }

    int find(int u) {
        if (parent[u] != u) parent[u] = find(parent[u]);
        return parent[u];
    }

    void unite(int u, int v) {
        u = find(u), v = find(v);
        if (u == v) return;
        if (rank[u] > rank[v]) parent[v] = u;
        else {
            parent[u] = v;
            if (rank[u] == rank[v]) ++rank[v];
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);

    int n, m;
    cin >> n >> m;

    vector<ll> a(n + 1);
    for (int i = 1; i <= n; ++i) cin >> a[i];

    vector<vector<Edge>> adj(n + 1);
    for (int i = 1; i <= m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    priority_queue<pair<ll, pair<int, int>>, vector<pair<ll, pair<int, int>>>, greater<>> pq;
    pq.push({a[1], {1, -1}}); // distance, node, parent

    vector<ll> dist(n + 1, INF);
    dist[1] = a[1];

    while (!pq.empty()) {
        auto [dist_u, p] = pq.top();
        pq.pop();

        int u = p.first, parent = p.second;
        if (dist[u] < dist_u) continue;

        for (auto &e : adj[u]) {
            int v = e.v, w = e.w;
            if (v == parent) continue;
            if (dist[u] + a[v] + w < dist[v]) {
                dist[v] = dist[u] + a[v] + w;
                pq.push({dist[v], {v, u}});
            }
        }
    }

    for (int i = 2; i <= n; ++i) cout << dist[i] << " ";
    cout << "\n";

    return 0;
}
