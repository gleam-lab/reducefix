#include <bits/stdc++.h>
using namespace std;

#define INT signed
#define pbk push_back
using i64 = long long;

struct Node {
    int x;
    i64 w;
    bool operator < (const Node& u) const {
        return w > u.w; // Min-heap: smaller weight has higher priority
    }
};

constexpr int MAXN = 200010;
constexpr i64 INF = 1e18;

vector<pair<int, int>> g[MAXN];
i64 dist[MAXN];
bool visited[MAXN];

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    // Clear graph
    for (int i = 1; i <= n; ++i) {
        g[i].clear();
    }
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }
    
    // Dijkstra's algorithm
    priority_queue<Node> pq;
    
    // Initialize distances
    for (int i = 1; i <= n; ++i) {
        dist[i] = INF;
        visited[i] = false;
    }
    
    // Start from vertex 1
    dist[1] = a[1]; // Include the starting vertex's weight
    pq.push({1, dist[1]});
    
    while (!pq.empty()) {
        auto [u, d] = pq.top();
        pq.pop();
        
        if (visited[u]) continue;
        visited[u] = true;
        
        // Explore neighbors
        for (auto& [v, edge_weight] : g[u]) {
            if (visited[v]) continue;
            
            // Calculate new distance: current path cost + edge weight + destination vertex weight
            i64 new_dist = dist[u] + edge_weight + a[v];
            
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                pq.push({v, new_dist});
            }
        }
    }
    
    // Output results for vertices 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i];
        if (i < n) cout << ' ';
        else cout << '\n';
    }
}

INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
