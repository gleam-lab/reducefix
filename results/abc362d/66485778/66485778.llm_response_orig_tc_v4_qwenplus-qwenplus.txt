#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
using pii = pair<int, int>;
constexpr int N = 2 * 1e5 + 10;
const i64 inf = 1e18;

vector<pii> g[N];
i64 d[N];
bool vis[N];

void solve() {
    int n, m;
    cin >> n >> m;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }

    priority_queue<pair<i64, int>> pq;
    fill(d, d + n + 1, inf);
    fill(vis, vis + n + 1, false);

    d[1] = a[1];
    pq.emplace(-d[1], 1); // Using min-heap logic with negative values

    while (!pq.empty()) {
        auto [dw, x] = pq.top();
        pq.pop();
        if (vis[x]) continue;
        vis[x] = true;
        for (auto& [y, cost] : g[x]) {
            if (!vis[y] && d[y] > d[x] + cost + a[y]) {
                d[y] = d[x] + cost + a[y];
                pq.emplace(-d[y], y);
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        cout << d[i] << " ";
    }
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
