struct Edge {

struct Node {

    bool operator<(const Node& other) const {

        g[u].push_back({v, b});
        g[v].push_back({u, b});
    }

    vector<i64> dist(N + 1, INF);
    priority_queue<Node> pq;
    dist[1] = A[1];
    pq.push({1, dist[1]});

    while (!pq.empty()) {
        Node node = pq.top();
        pq.pop();
        int u = node.id;
        i64 d = node.dist;
        if (d > dist[u]) continue;
        for (const Edge& e : g[u]) {
            int v = e.to;

            if (dist[v] > dist[u] + cost + A[v]) {
