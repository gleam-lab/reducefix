#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>
#define vvi vector<vi>
#define pi pair<ll, ll>
#define all(v) begin(v), end(v)

struct DSU {
    vi parent, rank;
    DSU(ll n) : parent(n+1), rank(n+1, 1) {
        for (ll i = 1; i <= n; ++i) parent[i] = i;
    }
    
    ll find(ll node) {
        if (parent[node] != node) parent[node] = find(parent[node]);
        return parent[node];
    }
    
    bool unite(ll u, ll v) {
        u = find(u), v = find(v);
        if (u == v) return false;
        if (rank[u] > rank[v]) swap(u, v);
        parent[u] = v;
        if (rank[u] == rank[v]) rank[v]++;
        return true;
    }
};

struct Edge {
    ll u, v, w;
    bool operator<(const Edge& e) const { return w < e.w; }
};

void dijkstra(const vvi& adj, const vi& weights, ll src, vi& dist) {
    priority_queue<pi, vector<pi>, greater<pi>> pq;
    dist[src] = weights[src];
    pq.push({dist[src], src});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w + weights[v] < dist[v]) {
                dist[v] = dist[u] + w + weights[v];
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    ll n, m;
    cin >> n >> m;
    vi weights(n + 1);
    for (ll i = 1; i <= n; ++i) cin >> weights[i];

    vvi adj(n + 1);
    for (ll i = 0; i < m; ++i) {
        ll u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vi ans(n);
    dijkstra(adj, weights, 1, ans);

    for (ll i = 1; i < n; ++i) cout << ans[i] << " ";
    cout << "\n";

    return 0;
}
