#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define mod 1000000007

struct dsu {
    vector<ll> parent, rank, size;
    dsu(ll n) {
        parent.resize(n+1);
        rank.resize(n+1, 0);
        size.resize(n+1, 1);
        for (ll i = 0; i <= n; i++) parent[i] = i;
    }
    
    ll find(ll node) {
        if (parent[node] == node) return node;
        return parent[node] = find(parent[node]);
    }
    
    void unite(ll a, ll b) {
        a = find(a);
        b = find(b);
        if (a != b) {
            if (rank[a] < rank[b]) swap(a, b);
            parent[b] = a;
            if (rank[a] == rank[b]) rank[a]++;
            size[a] += size[b];
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    ll n, m;
    cin >> n >> m;
    vector<ll> dist(n+1, LLONG_MAX);
    vector<vector<pair<ll, ll>>> adj(n+1);
    for (ll i = 1; i <= n; i++) {
        cin >> dist[i];
    }
    for (ll i = 0; i < m; i++) {
        ll u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    
    priority_queue<pair<ll, pair<ll, ll>>, vector<pair<ll, pair<ll, ll>>>, greater<pair<ll, pair<ll, ll>>>> pq;
    pq.push({dist[1], {1, 0}});
    
    while (!pq.empty()) {
        auto [w, p] = pq.top();
        pq.pop();
        
        ll node = p.first, prev_cost = p.second;
        if (prev_cost != dist[node]) continue;
        
        for (auto &child : adj[node]) {
            ll child_node = child.first, child_w = child.second;
            if (dist[child_node] > dist[node] + child_w) {
                dist[child_node] = dist[node] + child_w;
                pq.push({dist[child_node], {child_node, dist[node] + child_w}});
            }
        }
    }
    
    for (ll i = 2; i <= n; i++) {
        cout << dist[i] << " ";
    }
    
    return 0;
}
