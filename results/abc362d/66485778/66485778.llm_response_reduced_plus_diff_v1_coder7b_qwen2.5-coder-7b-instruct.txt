#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll INF = 1e18;

struct dsu {
    vector<ll> p, r;
    dsu(ll n) : p(n+1), r(n+1, 1) { iota(p.begin(), p.end(), 0); }
    ll find(ll x) { return p[x] = p[x] == x ? x : find(p[x]); }
    void unite(ll x, ll y) {
        x = find(x), y = find(y);
        if (x == y) return;
        if (r[x] < r[y]) swap(x, y);
        r[x] += r[y], p[y] = x;
    }
};

struct edge {
    ll u, v, w;
    bool operator<(const edge &e) const { return w < e.w; }
};

vector<ll> dijkstra(vector<vector<pair<ll, ll>>> &g, ll src) {
    ll n = g.size();
    vector<ll> dist(n, INF);
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
    dist[src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        ll u = pq.top().second, d = pq.top().first;
        pq.pop();
        if (d > dist[u]) continue;
        for (auto &[v, w] : g[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

int main() {
    ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

    ll n, m;
    cin >> n >> m;
    vector<ll> a(n);
    for (ll i = 0; i < n; i++) cin >> a[i];

    vector<vector<pair<ll, ll>>> g(n);
    for (ll i = 0; i < m; i++) {
        ll u, v, w;
        cin >> u >> v >> w;
        u--, v--;
        g[u].push_back({v, w});
        g[v].push_back({u, w});
    }

    vector<ll> ans(n-1);
    for (ll i = 1; i < n; i++) {
        ans[i-1] = min(a[i], dijkstra(g, i)[0] + a[0]);
    }

    for (ll i = 0; i < n-1; i++) cout << ans[i] << " ";
    cout << "\n";

    return 0;
}
