```cpp
#include <bits/stdc++.h>
using namespace std;

constexpr int N = 200010;
constexpr long long inf = 1e18;

struct Node {
    int x;
    long long w;
    bool operator < (const Node& u) const {
        return w > u.w;
    }
};

vector<pair<int, int>> g[N];
long long d[N];
bool vis[N];

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<long long> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    for (int i = 1; i <= m; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[x].emplace_back(y, c);
        g[y].emplace_back(x, c);
    }
    
    priority_queue<Node> pq;
    
    // Initialize distances
    for (int i = 1; i <= n; ++i) {
        d[i] = inf;
        vis[i] = false;
    }
    
    // Start from vertex 1 with initial cost of A[1]
    d[1] = a[1];
    pq.push({1, d[1]});
    
    while (!pq.empty()) {
        auto [x, w] = pq.top();
        pq.pop();
        
        if (vis[x]) continue;
        vis[x] = true;
        
        // Relax all edges from vertex x
        for (const auto& [y, edge_weight] : g[x]) {
            long long new_cost = d[x] + edge_weight + a[y];
            if (new_cost < d[y]) {
                d[y] = new_cost;
                pq.push({y, d[y]});
            }
        }
    }
    
    // Output results for vertices 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << d[i];
        if (i < n) cout << " ";
        else cout << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    
    return 0;
}
```
