#include <bits/stdc++.h>
using namespace std;

#define INT signed
#define pbk push_back
template <class T>
using V = vector<T>;

struct Node {
    int x;
    long long w;
    bool operator < (const Node& u) const {
        return w > u.w; // Min-heap: smaller weight has higher priority
    }
};

using i64 = long long;

constexpr int MAX_N = 200010;
vector<pair<int, int>> g[MAX_N];
long long dist[MAX_N];
bool visited[MAX_N];

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<long long> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    // Initialize graph and distances
    for (int i = 1; i <= n; ++i) {
        g[i].clear();
    }
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }
    
    // Dijkstra's algorithm
    priority_queue<Node> pq;
    
    // Initialize distances
    for (int i = 1; i <= n; ++i) {
        dist[i] = LLONG_MAX;
        visited[i] = false;
    }
    
    // Start from vertex 1 with initial cost = A[1]
    dist[1] = a[1];
    pq.push({1, a[1]});
    
    while (!pq.empty()) {
        auto [u, d] = pq.top();
        pq.pop();
        
        if (visited[u]) continue;
        visited[u] = true;
        
        // Explore neighbors
        for (auto& [v, edge_weight] : g[u]) {
            if (visited[v]) continue;
            
            long long new_cost = dist[u] + edge_weight + a[v];
            if (new_cost < dist[v]) {
                dist[v] = new_cost;
                pq.push({v, new_cost});
            }
        }
    }
    
    // Output results for vertices 2 to N
    for (int i = 2; i <= n; ++i) {
        cout << dist[i];
        if (i < n) cout << ' ';
    }
    cout << endl;
}

INT main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    solve();
    return 0;
}
