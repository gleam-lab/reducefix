#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,k,sum,cnt,a[N],b[2*N],d[N];
long long ans;

int main()
{
	ios::sync_with_stdio(0),cin.tie(0),cout.tie(0); 
	int i,j,x,y,z,t;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		cin>>a[i];
	
	for(i=1;i<=n;i++)
		a[i+n]=a[i];
	
	map<int,int> mp;
	mp[0]=1;
	long long prefix=0;
	
	for(i=1;i<2*n;i++)
	{
		prefix=(prefix+a[i])%m;
		if(prefix<0) prefix+=m;
		
		ans+=mp[prefix];
		mp[prefix]++;
		
		if(i>=n)
		{
			int prev_prefix=(prefix-a[i-n]+m)%m;
			if(prev_prefix<0) prev_prefix+=m;
			mp[prev_prefix]--;
			if(mp[prev_prefix]==0)
				mp.erase(prev_prefix);
		}
	}
	cout<<ans;
	return 0;
}
