#include<bits/stdc++.h>
using namespace std;
const int N = 1e6+5;
int n, m, k, sum, cnt, a[N], b[N], d[N];

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i + n] = a[i];  // Extend array a to handle wrap-around
    }

    // Initialize the frequency array b
    for (int i = 0; i < m; i++) {
        b[i] = 0;
    }
    b[0] = n;  // Initially, all elements in the array are in the first cycle

    for (int i = 1; i < n * 2; i++) {
        // Increment frequency of the current index % m in d
        if (i >= n) {
            b[d[i - n] % m]--;
        }

        // Calculate new value of d[i]
        d[i] = d[i - 1] + a[i];

        // Update the sum using the frequency of the current index % m in d
        sum += b[d[i] % m];

        // Increment frequency of the current index % m in d
        if (i < n) {
            b[d[i] % m]++;
        }
    }

    cout << sum;
    return 0;
}
