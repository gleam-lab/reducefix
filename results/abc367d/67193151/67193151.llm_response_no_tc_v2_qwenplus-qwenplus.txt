```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int n, m;
    cin >> n >> m;

    long long a[2*N+1] = {0};
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    long long prefix_sum = 0;
    long long count = 0;
    map<int, int> freq;
    freq[0] = 1; // Handle subarrays that start from index 0

    for(int i = 1; i < 2*n; ++i) {
        prefix_sum += a[i];
        int rem = prefix_sum % m;
        
        // Adjust for negative remainders
        if (rem < 0) rem += m;

        count += freq[rem];

        // Only update frequency for the first n elements
        if(i < n)
            freq[rem]++;
    }

    cout << count;
    return 0;
}
```
