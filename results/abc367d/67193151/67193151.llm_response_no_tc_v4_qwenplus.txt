```cpp
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,a[N],d[N];
long long sum;
map<int,int> cnt;

int main()
{
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    cin>>n>>m;
    
    for(int i=1;i<=n;i++)
        cin>>a[i];
    
    // Handle the case when m==1 separately to avoid modulo issues
    if(m == 1) {
        cout << 1LL * n * (n + 1) / 2 << endl;
        return 0;
    }
    
    // Use prefix sum modulo m
    cnt[0] = 1; // Empty prefix has sum 0
    int prefix = 0;
    
    for(int i=1; i<=n; i++) {
        prefix = (prefix + a[i]) % m;
        if(prefix < 0) prefix += m; // Ensure non-negative modulo
        
        // Add count of previous prefixes with same modulo value
        sum += cnt[prefix];
        cnt[prefix]++;
    }
    
    // Now consider circular case: extend array and process positions n+1 to 2n-1
    // But we need to be careful about overcounting and sliding window
    prefix = 0;
    for(int i=1; i<n; i++) {
        prefix = (prefix + a[i]) % m;
        if(prefix < 0) prefix += m;
        
        // Remove contribution of position i from cnt
        cnt[prefix]--;
        
        // Add element a[n+i] which is actually a[i]
        int next_val = a[i];
        int new_prefix = (d[i-1] + next_val) % m;
        if(new_prefix < 0) new_prefix += m;
        
        d[i] = new_prefix;
        sum += cnt[new_prefix];
        cnt[new_prefix]++;
    }
    
    cout<<sum;
    return 0;
}
```
