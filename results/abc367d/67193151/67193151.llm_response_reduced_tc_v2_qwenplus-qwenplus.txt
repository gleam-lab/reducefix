```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long sum;
int a[N], b[N];
long long d[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int i;
    cin >> n >> m;
    for (i = 1; i <= n; i++) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    // Initialize prefix sum array
    d[0] = 0;

    // We use a map-like array 'b' to count occurrences of prefix_sum % m
    // Since m can be up to 1e6, we initialize it with zeros
    fill(b, b + m, 0);
    b[0] = 1;  // Initial value for modulo 0

    for (i = 1; i < 2 * n; i++) {
        d[i] = d[i - 1] + a[i];
        int mod = ((d[i] % m) + m) % m;  // Ensure non-negative
        sum += b[mod];
        if (i < n) {
            b[mod]++;
        }
    }

    cout << sum << "\n";
    return 0;
}
```
