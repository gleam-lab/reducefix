#include<bits/stdc++.h>

const int N=1e6+5;

        if(d[i] < 0) d[i] += m; // Ensure positive modulo
        
        if(i >= n) {
            cnt[d[i-n]]--;
        }
        
        sum += cnt[d[i]];
        
        if(i < n) {

        }
    }
