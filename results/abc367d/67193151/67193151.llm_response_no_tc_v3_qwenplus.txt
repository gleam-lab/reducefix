#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,k,sum,cnt,a[N],b[N],d[N];
int main()
{
	ios::sync_with_stdio(0),cin.tie(0),cout.tie(0); 
	int i,j,x,y,z,t;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		cin>>a[i];
	
	sum = 0;
	memset(b, 0, sizeof(b));
	d[0] = 0;
	b[0] = 1; // Important: prefix sum mod m = 0 has occurred once at start
	
	for(i=1; i<=n; i++) {
		d[i] = d[i-1] + a[i];
		int mod = (d[i] % m + m) % m; // Handle negative modulo
		sum += b[mod];
		b[mod]++;
	}
	
	cout<<sum;
	return 0;
}
