```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m, a[N], b[N];
long long sum;
int d[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    b[0] = 1;
    for (int i = 1; i < n * 2; ++i) {
        if (i >= n)
            b[d[i - n] % m]--;
        d[i] = d[i - 1] + a[i];
        sum += b[d[i] % m];
        if (i < n)
            b[d[i] % m]++;
    }

    cout << sum << '\n';
    return 0;
}
```
