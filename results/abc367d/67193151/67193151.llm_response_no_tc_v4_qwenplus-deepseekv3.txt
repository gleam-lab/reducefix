    cnt[0] = 1; // For empty prefix

        int mod = (prefix[i] % m + m) % m;
        res += cnt[mod];
        cnt[mod]++;
    }
    
    // Reset for circular counting
    memset(cnt, 0, sizeof(cnt));
    cnt[0] = 1;
    
    for(int i=1; i<=2*n; i++) {

        if(i > n) {
            int prev_mod = (prefix[i-n] % m + m) % m;
            cnt[prev_mod]--;
        }
        res += cnt[mod];
        if(i < n) {

        }
    }

    return 0;
}
