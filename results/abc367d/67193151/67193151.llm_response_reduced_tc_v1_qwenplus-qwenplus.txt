```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long a[N * 2], prefix_mod_count[N], total = 0;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    prefix_mod_count[0] = 1; // initial count for mod 0

    for (int i = 1; i < 2 * n; ++i) {
        a[i] += a[i - 1]; // cumulative sum
        int mod = ((a[i] % m) + m) % m; // handle negative mods correctly

        if (i <= n) {
            total += prefix_mod_count[mod];
            prefix_mod_count[mod]++;
        } else {
            prefix_mod_count[mod]--;
        }
    }

    cout << total << endl;
    return 0;
}
```
