#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,k,sum,cnt,a[N],b[N],d[N];

int main()
{
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    cin >> n >> m;
    for(int i = 1; i <= n; i++)
        cin >> a[i];
    
    // Extend array a to handle circular range d[i] = d[i-1] + a[i]
    for(int i = 0; i < n; i++)
        a[i + n] = a[i];

    int current_d = 0;
    int current_mod = 0;
    b[current_mod]++; // Increment count for the first element's mod value
    for(int i = 1; i <= 2 * n; i++) {
        current_d += a[i]; // Increment by a[i]
        current_mod = current_d % m; // Update mod value

        // Sum the counts of the current mod value
        sum += b[current_mod];

        // Decrement and increment count for the current mod value
        if (i >= 2 * n && i < n * 2)
            b[current_mod]--;
        if (i < 2 * n)
            b[current_mod]++;
    }

    cout << sum;
    return 0;
}
