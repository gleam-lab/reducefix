```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 2e6 + 5;
int n, m;
long long a[N];
map<long long, long long> cnt;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i]; // Duplicate array for circular prefix sum handling
    }

    long long total = 0;
    long long prefix = 0;
    cnt[0] = 1; // Initialize with 0 mod count

    for (int i = 1; i < 2 * n; ++i) {
        prefix = (prefix + a[i]) % m;
        // For the first n elements, we increment count after checking
        // After n elements, we remove the one that's out of window
        if (i > n) {
            cnt[(prefix - a[i - n] + m) % m]--;
        }
        total += cnt[prefix];
        if (i < n) {
            cnt[prefix]++;
        }
    }

    cout << total << endl;
    return 0;
}
```
