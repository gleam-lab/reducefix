```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 2e6 + 5;

int n, m;
long long a[N];
map<int, int> cnt;
long long sum = 0;
long long prefix = 0;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];  // duplicate the array for circular handling
    }

    cnt[0] = 1;  // prefix sum mod m == 0 occurs once initially

    for (int i = 1; i < 2 * n; ++i) {
        prefix = (prefix + a[i]) % m;
        sum += cnt[prefix];
        cnt[prefix]++;
        
        // After processing the first n elements, start removing prefixes that are out of window
        if (i >= n) {
            int remove_prefix = (a[i - n + 1] + (i - n > 0 ? a[i - n] : 0)) % m;
            cnt[remove_prefix]--;
        }
    }

    cout << sum << endl;
    return 0;
}
```
