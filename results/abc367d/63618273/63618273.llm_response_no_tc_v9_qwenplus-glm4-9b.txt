#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int> a(n+1);
    for(int i = 1; i <= n; i++){
        cin >> a[i];
    }
    vector<int> s(n+1);
    for(int i = 1; i <= n; i++){
        s[i] = s[i-1] + a[i]; // Initialize with s[0] = 0
    }
    map<int, int> mp;
    ll ans = 0;
    ll L = s[n]; // Use s[n] instead of s[n+1]

    for(int i = 1; i <= n; i++){
        int remainder = (s[i] - L + mod) % mod; // Correct the formula for remainder
        ans += mp[remainder];
        mp[s[i] % mod]++; // Increment the count of s[i] % M in the map
    }
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t = 1;
    while(t--){
        solve();
    }
    return 0;
}
