```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n+1);
    for(int i=1; i<=n; i++){
        cin>>a[i];
    }
    vector<ll>s(n+2,0);
    for(int i=1; i<=n; i++){
        s[i]=(s[i-1]+a[i])%M;
    }
    map<int, ll>mp;
    mp[0]=1;  // empty prefix sum
    ll ans=0;
    for(int i=1; i<=n; i++){
        ans += mp[s[i]];
        mp[s[i]]++;
    }
    cout<<ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t=1;
    while(t--){
        solve();
    }
    return 0;
}
```

Key fixes:
1. Changed the prefix sum calculation to store modulo M results directly
2. Fixed the indexing of the prefix sum array
3. Properly initialized the map with mp[0] = 1 to account for subarrays starting from index 1
4. Simplified the counting logic - we just need to find how many previous prefix sums match the current one (s[i])
5. Changed the sum type to long long to prevent overflow
6. Removed the incorrect L calculation which wasn't needed

The algorithm now correctly counts all subarrays whose sum is divisible by M by tracking prefix sum remainders and using the property that if s[j] ≡ s[i] mod M, then the subarray from i+1 to j sums to a multiple of M.
