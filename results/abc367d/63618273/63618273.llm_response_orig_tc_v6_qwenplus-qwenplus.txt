#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 2); // Use ll to prevent overflow in prefix sums
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    // Compute prefix sums
    vector<ll> s(n + 2);
    for(int i = 1; i <= n; i++) {
        s[i] = s[i - 1] + a[i];
    }

    map<ll, ll> mp;
    ll ans = 0;

    // Traverse prefix sums modulo M
    for(int i = 0; i <= n; i++) {
        ll rem = ((s[i] % M) + M) % M; // Ensure positive remainder
        ans += mp[rem];
        mp[rem]++;
    }

    cout << ans << "\n";
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
