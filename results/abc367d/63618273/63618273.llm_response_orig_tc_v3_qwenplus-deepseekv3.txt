#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n+1);
    for(int i=1;i<=n;i++){
        cin>>a[i];
    }
    
    vector<ll>s(n+1,0);
    for(int i=1;i<=n;i++){
        s[i]=s[i-1]+a[i];
    }
    
    map<int, int>mp;
    mp[0]=1; // empty prefix sum
    ll ans=0;
    
    for(int i=1;i<=n;i++){
        int mod_val=(s[i]%M + M)%M;
        ans+=mp[mod_val];
        mp[mod_val]++;
    }
    
    cout<<ans;
}

int main()
{

    ll t=1;

    }

}
