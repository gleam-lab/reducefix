#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    // Prefix sums
    vector<ll> prefix(n + 2);
    for (int i = 1; i <= n; ++i) {
        prefix[i] = prefix[i - 1] + a[i];
    }

    map<ll, ll> cnt;
    ll ans = 0;

    // We need to count the number of subarrays whose sum mod M == 0
    for (int i = 0; i <= n; ++i) {
        ll rem = ((prefix[i] % M) + M) % M;
        ans += cnt[rem];
        cnt[rem]++;
    }

    cout << ans << endl;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
