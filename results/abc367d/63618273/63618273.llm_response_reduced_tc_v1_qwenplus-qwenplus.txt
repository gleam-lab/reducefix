```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10;
const int mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 2);  // Use ll for safety
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    vector<ll> prefix(n + 2);
    for(int i = 1; i <= n; ++i) {
        prefix[i] = prefix[i - 1] + a[i];
    }

    map<ll, ll> freq;
    ll ans = 0;

    // We want prefix[j] ≡ prefix[i-1] (mod M)
    // So we look for previous prefix[i-1] values with same mod
    for(int i = 0; i <= n; ++i) {
        ll rem = ((prefix[i] % M) + M) % M; // Ensure non-negative
        ans += freq[rem]++;
    }

    cout << ans << "\n";
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while(t--) {
        solve();
    }
    return 0;
}
```
