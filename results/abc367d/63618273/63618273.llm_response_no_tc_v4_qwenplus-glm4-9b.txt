#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int>a(n+10);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<int>prefix_sum(n+2);
    for (int i = 1; i <= n; i++) {
        prefix_sum[i+1] = (prefix_sum[i] + a[i]) % mod;
    }
    map<int, int>mp;
    ll ans = 0;
    ll L = prefix_sum[n+1];
    for (int i = 1; i <= n; i++) {
        ans += mp[(prefix_sum[i] - L + mod) % mod];
        mp[prefix_sum[i] % mod]++;
    }
    cout << ans << endl;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
