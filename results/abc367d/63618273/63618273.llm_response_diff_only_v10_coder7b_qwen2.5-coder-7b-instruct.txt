```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve() {
    int n, M;
    cin >> n >> M;
    vector<int> a(n + 10);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<int> s(n + 10);
    for (int i = 1; i <= n; i++) {
        s[i] = s[i - 1] + a[i]; // Corrected to accumulate from previous sum
    }
    map<int, int> mp;
    ll ans = 0, L = s[n];
    for (int i = 1; i <= n; i++) {
        ans += mp[(s[i] - L * (i / n)) % M]; // Adjusted to consider full cycles
        ans += mp[(s[i] - L * (i / n) + M) % M]; // Ensures positive result
        mp[(s[i] - L * (i / n)) % M]++;
    }
    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
```

### Explanation of Changes:
1. **Array Indexing Correction**: Changed `s[i+1]=s[i]+a[i];` to `s[i]=s[i-1]+a[i];` to correctly accumulate sums up to index `i`.
2. **Cycle Adjustment**: Added `(i / n)` to adjust for the number of complete cycles within the prefix sum array, ensuring that we account for all possible remainders.
3. **Edge Case Handling**: Ensured that the remainder is always positive by using `(s[i] - L * (i / n) + M) % M`.
4. **Efficiency Improvement**: Removed unnecessary initialization of `L` as `s[n+1]`, which was incorrect. Instead, directly used `s[n]` to calculate the total sum `L`.
