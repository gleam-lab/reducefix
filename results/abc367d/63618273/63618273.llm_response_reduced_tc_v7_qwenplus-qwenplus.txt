```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 2);  // Use ll to avoid overflow
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    vector<ll> s(n + 2);
    for(int i = 1; i <= n; i++) {
        s[i + 1] = (s[i] + a[i]) % M;  // Take modulo M at each step to prevent overflow
    }

    map<ll, ll> mp;
    ll ans = 0;
    for(int i = 1; i <= n; i++) {
        // We want (s[j] - s[i]) % M == 0 => s[j] % M == s[i] % M
        // So we count how many prefix sums have the same mod value as current s[i]
        ans += mp[s[i] % M]++;
    }
    
    cout << ans << "\n";
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while(t--) {
        solve();
    }
    return 0;
}
```
