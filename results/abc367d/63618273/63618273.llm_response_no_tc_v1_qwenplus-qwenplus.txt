```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 2);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    // Prefix sum array
    vector<ll> s(n + 2);
    for (int i = 1; i <= n; i++) {
        s[i] = (s[i - 1] + a[i]) % M;
    }

    map<ll, ll> cnt;
    cnt[0] = 1; // Initialize with prefix sum 0 appearing once

    ll ans = 0;
    for (int i = 1; i <= n; i++) {
        // We want (s[i] - s[j]) % M == 0 => s[i] ≡ s[j] mod M
        ans += cnt[s[i]];
        cnt[s[i]]++;
    }

    cout << ans << endl;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
```
