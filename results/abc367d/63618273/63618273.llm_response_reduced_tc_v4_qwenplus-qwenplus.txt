#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 2);  // Use ll to prevent overflow
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    vector<ll> s(n + 2);
    for(int i = 1; i <= n; i++) {
        s[i + 1] = s[i] + a[i];
    }

    map<ll, ll> mp;
    ll ans = 0;
    ll L = s[n + 1];  // Total sum of array

    // We want to find pairs (i, j) such that (s[i] - s[j]) % M == 0
    // That is equivalent to s[i] % M == s[j] % M
    // So we count how many times each prefix sum modulo M has occurred

    for(int i = 1; i <= n; i++) {
        ll target = ((s[i] - L % M) % M + M) % M;
        ans += mp[target];

        // Now add current prefix mod M to the map
        ll currMod = s[i] % M;
        mp[currMod]++;
    }

    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while(t--) {
        solve();
    }
    return 0;
}
