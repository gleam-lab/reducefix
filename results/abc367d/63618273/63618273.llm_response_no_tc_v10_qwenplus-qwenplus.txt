```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10;

void solve() {
    int n;
    ll M;
    cin >> n >> M;
    vector<ll> a(n + 2);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    // Prefix sum array
    vector<ll> s(n + 2, 0);
    for (int i = 1; i <= n; ++i) {
        s[i] = (s[i - 1] + a[i]) % M;
    }

    // Map to count occurrences of each prefix sum mod M
    map<ll, ll> mp;
    mp[0] = 1; // To count subarrays starting from index 0
    ll ans = 0;

    for (int i = 1; i <= n; ++i) {
        // Count how many times this mod value has occurred before
        ans += mp[s[i]];
        // Increment the count for this mod value
        mp[s[i]]++;
    }

    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
