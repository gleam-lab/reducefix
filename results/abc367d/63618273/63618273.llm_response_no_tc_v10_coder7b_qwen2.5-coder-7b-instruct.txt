```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
	int n,M;
	cin>>n>>M;
	vector<int>a(n+10);
	for(int i=1; i<=n; i++){
		cin>>a[i];
	}
	vector<int>s(n+10);
	for(int i=1; i<=n; i++){
		s[i+1]=(s[i]+a[i])%M;
	}
	map<int, int>mp;
	ll ans=0,L=s[n+1];
	for(int i=1; i<=n; i++){
		ans+=mp[(s[i] - (L*M-1)/M + M)%M];
		mp[s[i]]++;
	}
	cout<<ans<<'\n';
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
		solve();
	}
	return 0;
}
```

Explanation of changes:
1. Corrected the modulo operation to handle negative values properly.
2. Used `(s[i] - (L*M-1)/M + M)%M` instead of `(s[i]- L%M + M) % M` to ensure the calculation is correct even when `L` is not divisible by `M`.
3. Added a newline character at the end of the output for better readability.
