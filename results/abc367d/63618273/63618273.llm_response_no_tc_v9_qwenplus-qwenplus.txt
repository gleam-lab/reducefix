#include <bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve() {
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    // Compute prefix sums modulo M
    vector<ll> prefix_mod(n + 1);
    for (int i = 1; i <= n; i++) {
        prefix_mod[i] = (prefix_mod[i - 1] + a[i]) % M;
    }

    // Use map to count occurrences of each mod value
    map<ll, ll> freq;
    ll ans = 0;

    // We want to find number of subarrays whose sum is divisible by M
    // That corresponds to: (prefix[j] - prefix[i]) % M == 0 => prefix[j] % M == prefix[i] % M

    for (int i = 0; i <= n; i++) {
        ans += freq[prefix_mod[i]];
        freq[prefix_mod[i]]++;
    }

    cout << ans << endl;
}

int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
