3. The logic for counting subarrays is flawed - we should initialize the map with {0,1} since a sum of 0 exists before we start

5. The count increment logic was backwards

Here's the corrected version:

```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n+10);
    for(int i=1; i<=n; i++){
        cin>>a[i];
    }
    vector<ll>s(n+10);
    for(int i=1; i<=n; i++){
        s[i]=s[i-1]+a[i];
    }
    map<ll, int>mp;
    mp[0] = 1;  // initialize with sum 0 occurring once
    ll ans=0;
    for(int i=1; i<=n; i++){
        ll target = s[i]%M;
        ans += mp[target];
        mp[target]++;
    }
    cout<<ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t=1;
    while(t--){
        solve();
    }
    return 0;
}
```

1. Correct prefix sum calculation starting from s[1]
2. Proper initialization of the map with {0,1} to account for subarrays starting from index 1
3. Simplified the counting logic - just look for current sum mod M in the map
4. Removed unnecessary variable L
