{

	}
	vector<int>s(n+10);
	for(int i=1; i<=n; i++){
		s[i]=s[i-1]+a[i];
	}
	map<int, int>mp;
	ll ans=0,L=s[n];
	for(int i=0; i<n; i++){
		ans+=mp[(s[i] - L * (i/M) + M) % M];
		mp[s[i] % M]++;
	}
	cout<<ans;
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
