#include <bits/stdc++.h>
using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    int sum = 0;
    for(int &x: a) {
        std::cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = 0;
    std::map<int, int> prefix_count;
    prefix_count[0]++;

    for(int i = 0, pre = 0; i < n; ++i) {
        pre = (pre + a[i]) % m;
        res += prefix_count[(pre - sum + m) % m];
        prefix_count[pre]++;
    }

    std::cout << res << '\n';
    return;
}
