#include <iostream>

void solve() {

    if (sum == 0) res++;
    map<int, int> map[2];
    map[0][0] = 1;
    int pre = 0;
    for (int x : a) {
        pre = (pre + x) % m;
        res += map[0].count(pre);
        map[0][pre]++;
    }
    for (int x : a) {
        pre = (pre - x + m) % m;
        res += map[0].count(pre);
        map[0][pre]++;
    }

    cout << res << '\n';
}

    ios::sync_with_stdio(false);
