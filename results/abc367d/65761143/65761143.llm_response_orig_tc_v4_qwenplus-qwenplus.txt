#include <bits/stdc++.h>
using namespace std;
using LL = long long;

void solve() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n);

    // Precompute prefix sums modulo m
    vector<int> prefix_mod(n + 1);
    LL total_mod = 0;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i] %= m;
        total_mod = (total_mod + a[i]) % m;
        prefix_mod[i + 1] = (prefix_mod[i] + a[i]) % m;
    }

    if (total_mod == 0) {
        cout << n * (n + 1LL) / 2 - 1 << '\n'; // Subtract 1 to exclude the full array case if it's zero mod m
        return;
    }

    // Count prefix mod occurrences for each residue class
    map<int, LL> cnt[2];
    cnt[0][0] = 1; // For prefix_mod[0] = 0 at index 0

    LL res = 0;
    for (int i = 1; i <= n; ++i) {
        int curr = prefix_mod[i];
        int target1 = curr;
        int target2 = (curr - total_mod + m) % m;

        res += cnt[0][target1];
        res += cnt[1][target2];

        cnt[0][curr]++;
        cnt[1][curr]++;
    }

    cout << res << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

#ifndef ONLINE_JUDGE
    static ifstream in("in.txt");
    static ofstream out("out.txt");
    cin.rdbuf(in.rdbuf());
    cout.rdbuf(out.rdbuf());
#endif

    int T = 1;
    // cin >> T;
    while (T--) {
        solve();
    }

    return 0;
}
