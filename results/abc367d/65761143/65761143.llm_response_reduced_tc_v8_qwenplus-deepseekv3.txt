#include <bits/stdc++.h>
using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    for(int &x: a) {
        std::cin >> x;
        x %= m;
    }

    std::map<int, LL> freq;
    freq[0] = 1;
    LL res = 0;
    int prefix = 0;

    for(int x : a) {
        prefix = (prefix + x) % m;
        res += freq[prefix];
        freq[prefix]++;
    }

    std::cout << res << '\n';
    return;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0); std::cout.tie(0);

#endif
    int T = 1;
    // std::cin >> T;
    while(T--) solve();
#ifndef ONLINE_JUDGE
    std::chrono::system_clock::time_point stop = std::chrono::system_clock::now();

}

1. Simplified the approach to use a single frequency map tracking prefix sums modulo m
2. Removed unnecessary calculations involving sum
