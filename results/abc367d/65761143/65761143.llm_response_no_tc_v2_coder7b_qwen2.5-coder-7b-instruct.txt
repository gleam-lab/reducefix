```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    int n, m; cin >> n >> m;
    vector<ll> a(n);
    for(int i=0; i<n; i++) cin >> a[i];

    ll prefix_sum = 0;
    unordered_map<ll, int> mp;
    mp[0] = 1;
    ll count = 0;
    
    for(int i=0; i<n; i++){
        prefix_sum = (prefix_sum + a[i]) % m;
        count += mp[(prefix_sum * 2) % m];
        mp[prefix_sum]++;
    }
    
    cout << count << endl;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int tc=1;
    // cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
```

This solution uses a hash-map to store the frequency of prefix sums which helps in finding pairs that satisfy the condition efficiently. The time complexity is O(N) which is suitable for large inputs.
