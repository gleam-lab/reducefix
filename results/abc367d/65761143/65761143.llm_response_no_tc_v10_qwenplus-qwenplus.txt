```cpp
#include <bits/stdc++.h>
using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    int sum = 0;
    for (int &x : a) {
        std::cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = 0;
    std::map<int, int> cnt[2]; // cnt[0] for prefix sums, cnt[1] for (prefix - totalSum) mod m
    int prefix = 0;

    // Initially we have prefix sum 0 once
    cnt[0][0] = 1;
    if (sum != 0) cnt[1][sum % m] = 1;

    for (int x : a) {
        prefix = (prefix + x) % m;

        // Count subarrays ending at current position with sum % m == 0
        res += cnt[0][prefix];

        // For circular case: check if (prefix - totalSum) % m matches
        int target = (prefix - sum + m) % m;
        res += cnt[1][target];

        // Update counters for next iteration
        cnt[0][prefix]++;
        cnt[1][(prefix - sum + m) % m]++;
    }

    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0); std::cout.tie(0);
    int T; std::cin >> T;
    while (T--) solve();
    return 0;
}
```
