#include <iostream>
#include <vector>
#include <map>

using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    int sum = 0;
    for (int &x : a) {
        std::cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = 0;
    std::map<int, LL> countMap[2];
    countMap[0][0] = 1; // Base case: one way to have prefix sum 0 before any numbers are processed
    countMap[1][0] = 1; // Base case: one way to have prefix sum 0 after all numbers are processed

    int prefixSum = 0;
    for (int x : a) {
        prefixSum = (prefixSum + x) % m;
        res += countMap[0][prefixSum]; // Count ways to achieve this prefix sum before processing current number
        res += countMap[1][((prefixSum - sum + m) % m)]; // Count ways to achieve this prefix sum after processing current number
        countMap[0][prefixSum]++; // Increment the count of this prefix sum before processing current number
        countMap[1][prefixSum]++; // Increment the count of this prefix sum after processing current number
    }

    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0);
    int T = 1;
    // std::cin >> T;
    while (T--) solve();
    return 0;
}
