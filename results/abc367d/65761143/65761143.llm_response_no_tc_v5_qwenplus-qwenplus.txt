#include <bits/stdc++.h>
using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    int sum = 0;
    for (int &x : a) {
        std::cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = 0;
    std::map<int, int> cnt;
    cnt[0] = 1;  // Initialize with prefix sum 0 having count 1

    int curr = 0;
    for (int x : a) {
        curr = (curr + x) % m;
        int target = (curr - sum + m) % m;
        res += cnt[target];
        cnt[curr]++;
    }

    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0); std::cout.tie(0);

    int T = 1;
    std::cin >> T;
    while (T--) solve();

    return 0;
}
