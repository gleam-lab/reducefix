#include <iostream>
#include <vector>
#include <map>

using LL = long long;

void solve() {
    int n, m;
    std::cin >> n >> m;
    std::vector<int> a(n);

    int sum = 0;
    for(int &x: a) {
        std::cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = 0;
    std::map<int, int> map[2];
    map[0][0] = 1;
    int pre = 0;
    for(int x: a) {
        pre = (pre + x) % m;
        res += map[0].count(pre);
        res += map[1].count(((pre - sum) % m + m) % m);
        map[0][pre]++;
        map[1][pre]++;
    }

    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0); std::cout.tie(0);
    int T = 1;
    while(T--) solve();
    return 0;
}
