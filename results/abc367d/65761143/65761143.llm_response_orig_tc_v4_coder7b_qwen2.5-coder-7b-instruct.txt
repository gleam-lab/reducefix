#include <bits/stdc++.h>
using namespace std;
using LL = long long;

void solve() {
    int n, m; cin >> n >> m;
    vector<int> a(n);

    int sum = 0;
    for(int &x: a) {
        cin >> x;
        x %= m;
        sum = (sum + x) % m;
    }

    LL res = -1;
    if(sum == 0) res++;
    unordered_map<int, int> mp;
    mp[0]++;

    int pre = 0;
    for(int x: a) {
        pre = (pre + x) % m;
        res += mp[(pre - sum + m) % m];
        mp[pre]++;
    }

    cout << res << '\n';
    return;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
#ifndef ONLINE_JUDGE
    static ifstream in("in.txt");
    static ofstream out("out.txt");
    cin.rdbuf(in.rdbuf());
    cout.rdbuf(out.rdbuf());
    chrono::system_clock::time_point start = chrono::system_clock::now();
#endif
    int T = 1;
    // cin >> T;
    while(T--) solve();
#ifndef ONLINE_JUDGE
    chrono::system_clock::time_point stop = chrono::system_clock::now();
    auto duration = chrono::duration_cast<chrono::milliseconds>(stop - start).count();
    cout << "Run Time: " << duration << "ms\n";
#endif
    return 0;
}
