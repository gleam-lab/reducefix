using LL = long long;

void solve() {
    int n, m; std::cin >> n >> m;
    std::vector<int> a(n);

    for(int &x: a) {
        std::cin >> x;
        x %= m;
    }

    LL res = 0;
    std::unordered_map<int, int> prefix_counts;
    prefix_counts[0] = 1; // empty prefix
    int current_prefix = 0;

    for(int x: a) {
        current_prefix = (current_prefix + x) % m;
        if(current_prefix < 0) current_prefix += m; // ensure positive
        res += prefix_counts[current_prefix];
        prefix_counts[current_prefix]++;
    }

    std::cout << res << '\n';
    return;
}
