```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int maxn = 2e5 + 7;

int n, m;
ll A[maxn], dp[maxn];

void solve() {
    cin >> n >> m;
    for(int i = 1; i <= n; ++i)
        cin >> A[i];
    
    memset(dp, 0, sizeof(dp));
    dp[0] = 1;
    ll ans = 0, sum = 0;
    
    for(int i = 1; i <= n; ++i) {
        sum += A[i];
        sum %= m;
        ans += dp[(m - sum) % m];
        dp[sum]++;
    }
    
    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    
    int T = 1;
    cin >> T;
    while(T--) solve();
    
    return 0;
}
```
