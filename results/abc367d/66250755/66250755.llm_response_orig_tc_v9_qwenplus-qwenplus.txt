```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int MAXN = 400005;

int n, m;
ll a[MAXN * 2], prefix_mod[2 * MAXN];
map<ll, ll> cnt;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    // Compute prefix sums modulo m
    for (int i = 1; i <= 2 * n; ++i) {
        prefix_mod[i] = (prefix_mod[i - 1] + a[i]) % m;
    }

    ll result = 0;
    cnt[0] = 1; // For subarrays starting at index 0

    for (int i = 1; i <= 2 * n; ++i) {
        result += cnt[prefix_mod[i]];
        cnt[prefix_mod[i]]++;
        
        if (i >= n) {
            cnt[prefix_mod[i - n]]--;
        }
    }

    cout << result << endl;
    return 0;
}
```
