```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 400005;

int n, m, a[MAXN], t[MAXN << 2], an, w[MAXN];

int lowbit(int x) {
    return x & (-x);
}

void update(int p, int val) {
    for (; p <= 2 * n; p += lowbit(p)) {
        t[p] += val;
    }
}

int query(int p) {
    int sum = 0;
    for (; p > 0; p -= lowbit(p)) {
        sum += t[p];
    }
    return sum;
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    for (int i = 1; i < 2 * n; ++i) {
        w[i + 1] = (w[i] + a[i]) % m;
        if (i > n) update(w[i - n] + 1, -1);
        if (i >= n) {
            an += query(w[i]);
        }
        update(w[i] + 1, 1);
    }

    cout << an << endl;
    return 0;
}
```
