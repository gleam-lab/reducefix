```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, m;
    cin >> n >> m;
    vector<int> a(2 * n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    // Prefix sums modulo m
    vector<int> prefix(2 * n + 1);
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
    }

    // We will use a map to count frequencies of prefix sums
    map<int, int> freq;
    int result = 0;

    // Only consider valid subarrays of length at most n
    for (int i = 1; i <= 2 * n; ++i) {
        if (i > n) {
            // Remove the contribution of prefix[i - n]
            freq[prefix[i - n]]--;
        }
        result += freq[prefix[i]];
        freq[prefix[i]]++;
    }

    cout << result << endl;
    return 0;
}
```
