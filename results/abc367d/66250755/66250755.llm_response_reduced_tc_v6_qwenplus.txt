#include<bits/stdc++.h>
using namespace std;
int n,m,a[400005],t[1000005],an,w[400005];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		a[i]%=m;
		a[i+n]=a[i];
	}
	memset(t,0,sizeof(t));
	for(int i=1;i<2*n;i++)
	{
		w[i+1]=(w[i]+a[i])%m;
		if(i>=n)
		{
			an+=t[w[i]];
		}
		if(i>n)t[w[i-n]]--;
		t[w[i]]++;
	}
	cout<<an;
	return 0;
}
