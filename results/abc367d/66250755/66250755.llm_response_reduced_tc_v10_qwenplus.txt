```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<int> a(2 * n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }
    
    long long an = 0;
    map<int, int> t;
    vector<int> w(2 * n + 1, 0);
    
    // w[0] = 0 is already set
    t[0] = 1; // Important: empty prefix has sum 0
    
    for (int i = 0; i < 2 * n - 1; i++) {
        w[i + 1] = (w[i] + a[i]) % m;
        
        // When we have more than n elements, remove the contribution of the prefix that is now out of range
        if (i >= n - 1) {
            // Remove the prefix that was starting at position 0 and ending at i-n
            t[w[i - n + 1]]--;
        }
        
        // Count how many previous prefixes have the same sum (which means the subarray between them has sum divisible by m)
        an += t[w[i + 1]];
        
        // Add current prefix to the map
        t[w[i + 1]]++;
    }
    
    cout << an;
    return 0;
}
```
