#include<bits/stdc++.h>
using namespace std;

int n, m, a[400005], t[1000005], an;
int w[2 * 400005];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];  // Extend array to simulate circular behavior
        w[i + 1] = a[i];  // Initialize the prefix sums array
    }
    
    // Compute prefix sums
    for (int i = 2; i <= 2 * n; i++) {
        w[i] = (w[i - 1] + a[i - 1]) % m;
    }
    
    // Count frequency of each remainder in the extended array
    for (int i = 1; i <= 2 * n; i++) {
        t[w[i]]++;
    }
    
    // Calculate the answer using the frequency array
    for (int i = 1; i <= 2 * n; i++) {
        if (i > n) {
            an += t[w[i - n]];  // Sum the frequency of the current position minus n
        }
    }
    
    cout << an;
    return 0;
}
