```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, m;
    cin >> n >> m;
    
    int a[400005] = {}, prefix[400005] = {};
    long long ans = 0;
    int cnt[1000005] = {};

    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
    }

    // Double the array to handle circular property
    for (int i = n + 1; i <= 2 * n; ++i) {
        a[i] = a[i - n];
    }

    // Compute prefix sums modulo m
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
    }

    // Use sliding window of size n
    cnt[0] = 1;  // To count subarrays starting from index 0
    for (int i = 1; i <= 2 * n; ++i) {
        if (i > n) {
            cnt[prefix[i - n]]--;
        }
        ans += cnt[prefix[i]];
        cnt[prefix[i]]++;
    }

    cout << ans;
    return 0;
}
```
