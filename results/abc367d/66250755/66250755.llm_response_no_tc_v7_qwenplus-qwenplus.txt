```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    
    vector<int> a(2 * n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    // Prefix sum mod m
    vector<int> prefix(2 * n + 1);
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
    }

    // Use a map to count occurrences of each mod value
    map<int, int> count;
    count[0] = n;  // Important: There are n positions where prefix sum is 0 initially

    long long result = 0;
    for (int i = n; i >= 1; --i) {
        result += count[prefix[i]];
        // Move window: remove the one going out (i + n)
        count[prefix[i + n]]--;
        // Add the one coming in (i)
        count[prefix[i]]++;
    }

    cout << result << '\n';
    return 0;
}
```
