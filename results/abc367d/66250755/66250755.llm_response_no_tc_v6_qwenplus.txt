```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(2 * n + 1);
    vector<int> t(m, 0);
    
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }
    
    long long an = 0;
    vector<int> w(2 * n + 1, 0);
    
    for (int i = 1; i < 2 * n; i++) {
        w[i] = (w[i-1] + a[i]) % m;
        
        // When we have passed the first window of size n, start removing old prefix sums
        if (i >= n) {
            t[w[i - n]]--;
        }
        
        // Count how many previous prefix sums are equal to current one
        an += t[w[i]];
        
        // Add current prefix sum to frequency table
        t[w[i]]++;
    }
    
    cout << an;
    return 0;
}
```
