#include<bits/stdc++.h>
using namespace std;
int n, m, a[400005], w[400005];
int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
    }
    for (int i = 1; i < 2 * n; i++) {
        w[i] = (w[i - 1] + a[i % n]) % m;
    }
    memset(w, 0, sizeof(w));
    for (int i = 1; i < 2 * n; i++) {
        w[i] = (w[i - 1] + a[i % n]) % m;
        t[w[i]]++;
    }
    int an = 0;
    memset(t, 0, sizeof(t));
    for (int i = 1; i < 2 * n; i++) {
        if (i >= n) t[w[i]]++;
        if (i >= n) an += t[w[i] - 1];
        t[w[i]]--;
    }
    cout << an;
    return 0;
}
