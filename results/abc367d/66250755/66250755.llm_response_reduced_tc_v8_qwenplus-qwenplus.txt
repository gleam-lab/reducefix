```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> a(2 * n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    vector<long long> prefix(2 * n + 1);
    map<long long, int> count;
    long long ans = 0;

    count[0] = 1; // To count subarrays with sum divisible by m
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
        if (i >= n) {
            ans += count[prefix[i]];
        }
        if (i < n) {
            count[prefix[i]]++;
        } else {
            count[prefix[i - n]]--;
            count[prefix[i]]++;
        }
    }

    cout << ans;
    return 0;
}
```
