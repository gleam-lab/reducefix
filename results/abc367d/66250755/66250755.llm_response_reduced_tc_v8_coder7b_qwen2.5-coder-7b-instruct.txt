```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(2 * n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[n + i] = a[i];
    }
    
    vector<int> prefix_sum(2 * n + 1, 0);
    vector<int> count(m, 0);
    int answer = 0;
    
    for (int i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i - 1]) % m;
        if (i > n) {
            --count[prefix_sum[i - n]];
        }
        if (i >= n) {
            answer += count[prefix_sum[i]];
        }
        ++count[prefix_sum[i]];
    }
    
    cout << answer << endl;
    return 0;
}
```
