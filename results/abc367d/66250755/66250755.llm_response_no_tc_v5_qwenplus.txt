#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n, m;
    cin >> n >> m;
    vector<int> a(n);
    
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        a[i] %= m;
    }
    
    long long result = 0;
    
    // For each possible starting position
    for (int start = 0; start < n; start++) {
        int current_sum = 0;
        // Extend the subarray up to length n
        for (int len = 1; len <= n; len++) {
            int pos = (start + len - 1) % n;
            current_sum = (current_sum + a[pos]) % m;
            if (current_sum == 0) {
                result++;
            }
        }
    }
    
    cout << result;
    return 0;
}
