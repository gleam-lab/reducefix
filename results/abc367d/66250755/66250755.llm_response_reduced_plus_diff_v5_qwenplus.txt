```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, m;
ll a[400005], w[800005];
map<ll, ll> t;

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }
    
    ll an = 0;
    w[0] = 0;
    t[0] = 1;
    
    for (int i = 1; i < 2 * n; i++) {
        w[i] = (w[i-1] + a[i]) % m;
        
        if (i >= n) {
            an += t[w[i]];
        }
        
        if (i >= n) {
            t[w[i - n]]--;
            if (t[w[i - n]] == 0) {
                t.erase(w[i - n]);
            }
        }
        
        t[w[i]]++;
    }
    
    cout << an;
    return 0;
}
```
