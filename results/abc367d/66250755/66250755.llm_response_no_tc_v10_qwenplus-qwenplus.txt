```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    
    vector<int> a(2 * n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    // Prefix sum mod m
    vector<int> prefix(2 * n + 1);
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
    }

    // Use a map to count occurrences of each mod value
    map<int, int> count;
    int result = 0;

    // We want to find the number of subarrays of length ≤ n with max prefix difference
    for (int i = 0; i <= 2 * n; ++i) {
        // Remove values that are more than n steps behind
        if (i > n) {
            count[prefix[i - n - 1]]--;
        }

        // Count how many previous prefix sums have <= current value
        // This is equivalent to counting all valid j where (prefix[i] - prefix[j]) % m >= 0
        // and j ∈ [i-n-1, i-1]
        auto low = count.lower_bound(prefix[i]);
        int total = 0;
        for (auto it = count.begin(); it != low; ++it) {
            total += it->second;
        }

        result += total;
        count[prefix[i]]++;
    }

    cout << result << "\n";
    return 0;
}
```
