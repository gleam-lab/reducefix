```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 400005;
const int MAXM = 1000005;

int n, m;
long long a[MAXN], prefix[MAXN];
long long t[MAXM]; // count of each mod value
long long result = 0;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i]; // duplicate array to simulate circular property
    }

    // Compute prefix sums modulo m
    for (int i = 1; i <= 2 * n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % m;
    }

    // Initialize with the first window of size n
    for (int i = 0; i <= n; ++i) {
        t[prefix[i]]++;
    }

    // Sliding window approach
    for (int i = n + 1; i <= 2 * n; ++i) {
        // Remove the element going out of window
        int out_index = i - n - 1;
        t[prefix[out_index]]--;

        // Add the new element coming into window
        result += t[prefix[i]];
        t[prefix[i]]++;

        // Uncomment below line to debug
        // cout << "At step " << i-n << ", current result: " << result << endl;
    }

    cout << result << endl;
    return 0;
}
```
