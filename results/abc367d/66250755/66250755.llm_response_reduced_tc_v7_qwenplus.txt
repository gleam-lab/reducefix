```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<int> a(2 * n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i];
    }

    vector<int> prefix(2 * n + 1, 0);
    map<int, int> count;
    long long ans = 0;

    // Initialize: prefix sum of empty subarray is 0
    count[0] = 1;

    for (int i = 1; i <= 2 * n; i++) {
        prefix[i] = (prefix[i-1] + a[i]) % m;

        // For positions beyond the first n, we start removing old contributions
        if (i > n) {
            // Remove the prefix that is now out of the sliding window of length n
            count[prefix[i - n]]--;
        }

        // Count how many previous prefix sums are equal to current one
        // This means the subarray between them has sum divisible by m
        ans += count[prefix[i]];

        // Add current prefix to the map
        count[prefix[i]]++;
    }

    cout << ans << endl;

    return 0;
}
```
