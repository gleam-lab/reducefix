#include<bits/stdc++.h>
using namespace std;
int n, m, a[400005], t[1000005], an, w[400005];
int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i] %= m;
        a[i + n] = a[i]; // Extend the array by repeating the elements
    }
    for (int i = 1; i < 2 * n; i++) {
        w[i + 1] = (w[i] + a[i]) % m; // Update the prefix sum array with modulo m
        if (i >= n) { // Only consider indices n to 2n-1
            t[w[i]]++; // Increment the count of the current prefix sum
        }
        if (i > n) { // Only consider indices n+1 to 2n-1
            t[w[i - n]]--; // Decrement the count of the previous prefix sum
        }
        if (i >= n) {
            an += t[w[i]]; // Accumulate the count of the current prefix sum
        }
    }
    cout << an;
    return 0;
}
