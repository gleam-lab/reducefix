#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
	int n,M;
	cin>>n>>M;
	vector<int>a(n+10);
	for(int i=0; i<n; i++){
		cin>>a[i];
	}
	vector<int>s(n+10);
	for(int i=0; i<n; i++){
		s[i+1]=(s[i]+a[i])%M;
	}
	map<int, int>mp;
	ll ans=0,L=s[n];
	for(int i=0; i<=n; i++){
		ans+=mp[(L-s[i] + M) % M];
		mp[s[i]]++;
	}
	cout<<ans<<"\n";
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
		solve();
	}
	return 0;
}

This solution corrects the algorithm by ensuring that all calculations are done modulo `M` to handle large numbers properly and to ensure that the map keys are always within the range of valid remainders when divided by `M`. It also fixes the loop condition and ensures that the last element is considered in the sum array `s`.
