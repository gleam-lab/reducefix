#include<bits/stdc++.h>
using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n), s(n+1);
    for(int i=0; i<n; i++) {
        cin>>a[i];
        s[i+1] = (s[i] + a[i]) % M;
    }
    map<int, int>mp;
    mp[0]++;
    ll ans=0;
    for(int i=0; i<=n; i++){
        ans += mp[s[i]];
        mp[s[i]]++;
    }
    cout<<ans<<endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    ll t=1;
    while(t--){
        solve();
    }
    return 0;
}
