Here's the fixed version that correctly counts the number of subarrays whose sum is divisible by M:

```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{

    }

    for(int i=0; i<n; i++){
        s[i+1]=s[i]+a[i];
    }
    map<int, int>mp;
    mp[0] = 1; // empty prefix sum
    ll ans=0;
    for(int i=1; i<=n; i++){
        int rem = (s[i] % M + M) % M; // handle negative remainders
        ans += mp[rem];
        mp[rem]++;
    }
    cout<<ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t=1;
    while(t--){
        solve();
    }
    return 0;

```

Key fixes:
1. Changed the prefix sum to be long long to prevent overflow
2. Simplified the counting logic - we only need to track remainders of prefix sums
3. Initialize the map with {0:1} to account for empty prefix
4. For each position, we calculate the remainder and add the count of previous prefixes with the same remainder
5. Removed unnecessary variables and simplified the modulo operations

This solution correctly counts the number of subarrays whose sum is divisible by M by using the property that if two prefix sums have the same remainder when divided by M, the subarray between them has a sum divisible by M.

The time complexity is O(n) for the loop and O(log n) for map operations, resulting in O(n log n) overall, which is efficient enough for the constraints (n ≤ 2e5).
