#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10, mod=1e9+7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int> a(n+10);
    for(int i = 0; i < n; i++)
        cin >> a[i];
    
    vector<int> s(n+10);
    for(int i = 0; i < n; i++)
        s[i+1] = s[i] + a[i];
    
    map<int, int> mp;
    ll ans = 0, L = s[n];
    
    // Count occurrences of remainders when divided by M
    for(int i = 0; i <= n; i++)
        mp[s[i] % M]++;
    
    // Calculate the answer
    for(int i = 0; i <= n; i++)
    {
        int r = (s[i] - L + M) % M;
        ans += mp[r];
    }
    
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t = 1;
    while(t--)
        solve();
    return 0;
}
