#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, m;
    cin >> n >> m;
    vector<long long> a(n), s(n + 1);
    for (int i = 0; i < n; ++i) cin >> a[i];
    partial_sum(a.begin(), a.end(), s.begin() + 1);
    map<long long, int> mp;
    long long ans = 0;
    for (int i = 0; i < n; ++i) {
        ans += mp[(s[i] - s[n]) % m];
        ans += mp[s[i] % m]++;
    }
    cout << ans << '\n';
    return 0;
}
