#include <bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10;
const int mod = 1e9 + 7;

void solve() {
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 10); // Use ll to avoid overflow in prefix sums
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    vector<ll> s(n + 1);
    for (int i = 0; i < n; ++i) {
        s[i + 1] = s[i] + a[i]; // Prefix sum array
    }

    map<ll, ll> mp; // Count occurrences of each modulo value
    ll ans = 0;
    ll total_mod = s[n] % M;

    for (int i = 0; i <= n; ++i) {
        ll current_mod = s[i] % M;
        // Compute adjusted mod to handle negative values correctly
        if (current_mod < 0) current_mod += M;

        // For the condition: (s[j] - s[i]) % M == 0 => count += mp[s[i]]
        ans += mp[current_mod];

        // Increment count for current prefix sum mod
        mp[current_mod]++;
    }

    cout << ans << endl;
}

int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
