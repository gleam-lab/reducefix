#include<bits/stdc++.h>
using namespace std;
using ll = long long;

const int N = 2e5 + 10, mod = 1e9 + 7;

void solve() {

        cin >> a[i];
        s[i + 1] = s[i] + a[i];
    }
    map<ll, int> mp;
    ll ans = 0, L = s[n];
    for (int i = 0; i <= n; i++) {
        ans += mp[(s[i] - L * (i / n)) % M];
        if (i > 0)
            mp[s[i - 1] % M]++;
    }
    cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    ll t = 1;
    while (t--) {
        solve();
