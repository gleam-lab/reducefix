```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n+10);
    for(int i=0; i<n; i++){
        cin>>a[i];
    }
    vector<int}s(n+10);
    for(int i=0; i<n; i++){
        s[i+1]=s[i]+a[i];
    }
    map<int, int>mp;
    ll ans=0;
    ll L=s[n];
    for(int i=0; i<n; i++){
        // 盲陆驴莽聰篓LLONG_MAX忙颅拢猫麓聼忙聲掳忙篓隆猫驴聬莽庐聴茅聴庐茅垄聵
        ans += mp[LLONG_MAX - (L - s[i] + M) % M];
        // 忙鲁篓忙聞聫忙篓隆猫驴聬莽庐聴莽職聞莽禄聯忙聻聹氓聫炉猫聝陆盲赂潞0茂录聦忙聢聭盲禄卢茅聹聙猫娄聛氓陇聞莽聬聠
        ans += mp[s[i] % M]++;

        // 忙聸麓忙聳掳map茂录聦猫庐掳氓戮聴氓炉鹿忙篓隆猫驴聬莽庐聴莽禄聯忙聻聹氓聠聧忙卢隆氓聫聳忙篓隆
        mp[s[i] % M] = (mp[s[i] % M] + 1) % mod;
    }
    cout<<ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t=1;
    while(t--){
        solve();
    }
    return 0;
}
```
