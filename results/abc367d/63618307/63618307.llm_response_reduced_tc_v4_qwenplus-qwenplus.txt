#include <bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve() {
    int n;
    ll M;
    cin >> n >> M;
    vector<ll> a(n + 10);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    vector<ll> s(n + 1);
    for (int i = 0; i < n; i++) {
        s[i + 1] = (s[i] + a[i]) % M;
    }

    map<ll, ll> mp;
    ll ans = 0;
    for (int i = 0; i <= n; i++) {
        // Compute (s[i] - L) % M where L = s[n]
        ll rem = (s[i] - s[n] + M) % M;
        ans += mp[rem];
        mp[s[i]]++;
    }

    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    return 0;
}
