#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int> a(n+10);
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    vector<int> s(n+10);
    for(int i = 0; i < n; i++){
        s[i+1] = s[i] + a[i];
    }
    map<int, int> mp;
    ll ans = 0;
    ll L = s[n];
    for(int i = 0; i < n; i++){
        // Adjust s[i] and L to be within the range [0, M-1]
        ll adjusted_s = (s[i] - L + mod) % mod;
        ans += mp[adjusted_s];
        // Increase count of current mod value of s[i]
        mp[s[i] % mod]++;
    }
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while(t--){
        solve();
    }
    return 0;
}
