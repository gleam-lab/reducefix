#include<bits/stdc++.h>

using namespace std;
using ll=long long;

        cin>>a[i];
    }
    vector<ll>s(n+10);
    for(int i=0; i<n; i++){
        s[i+1]=s[i]+a[i];
    }
    
    map<int, int>mp;
    mp[0] = 1; // empty prefix sum
    ll ans=0;
    for(int i=1; i<=n; i++){
        int rem = (s[i] % M + M) % M; // handle negative remainders
        ans += mp[rem];
        mp[rem]++;
    }
    cout<<ans;
}

Key fixes:
1. Properly initializes the map with mp[0] = 1 to account for the empty prefix sum
2. Correctly calculates remainders by using (s[i] % M + M) % M to ensure positive remainders
3. Iterates from 1 to n (inclusive) to properly handle all prefix sums
4. Simplified the counting logic by directly adding mp[rem] to ans before incrementing mp[rem]

This solution efficiently counts the number of subarrays with sum divisible by M using prefix sums and remainder tracking, achieving O(N) time complexity with O(M) space complexity (since there are at most M distinct remainders).
