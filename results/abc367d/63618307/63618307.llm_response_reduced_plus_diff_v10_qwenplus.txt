#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
	int n,M;
	cin>>n>>M;
	vector<int>a(n);
	for(int i=0; i<n; i++){
		cin>>a[i];
	}
	vector<ll>s(n+1);
	for(int i=0; i<n; i++){
		s[i+1]=s[i]+a[i];
	}
	map<ll, ll>mp;
	ll ans=0,L=s[n];
	mp[0]=1;
	for(int i=1; i<=n; i++){
		ll target = (s[i] - L) % M;
		if(target < 0) target += M;
		ans += mp[target];
		mp[s[i] % M]++;
	}
	cout<<ans;
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
		solve();
	}
	return 0;
}
