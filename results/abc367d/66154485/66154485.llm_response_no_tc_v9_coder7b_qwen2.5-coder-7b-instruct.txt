```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    for (int start = 0; start < n; start++) {
        int sum = 0;
        for (int len = 1; len <= n - start; len++) {
            sum += arr[(start + len - 1) % n];
            if (sum % m == 0)
                ans++;
        }
    }
    cout << ans << endl;
    return 0;
}
```
