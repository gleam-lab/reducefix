```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    for (int i = 0; i < n; i++) {
        int sum = 0;
        for (int j = 0; j < n; j++) { // Start from 0 to include all possible subarrays
            sum += arr[(i + j) % n];
            if (sum % m == 0)
                ans++;
        }
    }
    cout << ans;
    return 0;
}
```
Explanation:
- The original code had an off-by-one error in the inner loop where it started from 1 instead of 0, which caused it to miss some valid subarrays.
- I replaced the array with a `vector<int>` for better flexibility and readability.
- The logic remains the same but now correctly calculates all subarray sums and counts those divisible by `m`.
