```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Prefix sum mod array
    vector<int> prefix_mod(n + 1, 0);
    for (int i = 0; i < n; i++) {
        prefix_mod[i + 1] = (prefix_mod[i] + arr[i]) % m;
    }

    // Use a frequency map to count occurrences of each mod value
    vector<int> freq(m, 0);
    int ans = 0;

    for (int i = 0; i < n; i++) {
        // Count how many times this mod value has occurred before
        ans += freq[prefix_mod[i]];
        // Increment the frequency of current prefix mod value
        freq[prefix_mod[i]]++;
    }

    cout << ans << endl;
    return 0;
}
```
