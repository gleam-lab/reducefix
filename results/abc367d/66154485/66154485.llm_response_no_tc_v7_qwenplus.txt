```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    
    // Use prefix sums and modular arithmetic
    for (int start = 0; start < n; start++) {
        int prefix_sum = 0;
        // Check all subarrays starting at 'start' with length 1 to n-1
        for (int len = 1; len < n; len++) {
            int idx = (start + len - 1) % n;
            prefix_sum = (prefix_sum + arr[idx]) % m;
            if (prefix_sum == 0) {
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
