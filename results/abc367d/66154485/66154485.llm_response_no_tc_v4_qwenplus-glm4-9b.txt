#include <iostream>
using namespace std;

int main() {
    int n, m;
    long long ans = 0;
    cin >> n >> m;
    
    int arr[200005];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    // Calculate the prefix sums of the array
    long long prefixSum[200005];
    prefixSum[0] = 0;
    for (int i = 1; i <= n; i++) {
        prefixSum[i] = prefixSum[i - 1] + arr[i - 1];
    }
    
    // Iterate through the array to check for sums that are multiples of m using prefix sums
    for (int i = 1; i <= n; i++) {
        for (int j = i; j <= n; j++) {
            long long rangeSum = prefixSum[j] - prefixSum[i - 1];
            
            if (rangeSum % m == 0) {
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
