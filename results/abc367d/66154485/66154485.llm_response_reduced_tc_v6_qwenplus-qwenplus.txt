```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<int> arr(n);
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }

    // Prefix sum modulo m
    long long ans = 0;
    vector<int> count(m, 0);
    int prefix_sum = 0;

    // To handle subarrays wrapping around the array
    // We duplicate the array to simulate circular behavior
    for (int k = 0; k < 2; ++k) {
        for (int i = 0; i < n; ++i) {
            prefix_sum = (prefix_sum + arr[i]) % m;
            ans += count[prefix_sum]++;
        }
    }

    // Subtract overcounted cases: full array sums counted twice
    // The number of times the full array sum is zero mod m
    int full_sum = 0;
    for (int x : arr) {
        full_sum = (full_sum + x) % m;
    }

    if (full_sum == 0) {
        // Each valid starting point contributes one extra count due to wrap-around
        ans -= n;
    }

    cout << ans << endl;
    return 0;
}
```
