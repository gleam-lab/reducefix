```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for(int i = 0; i < n; ++i) {
        cin >> arr[i];
    }

    long long count = 0;
    for(int i = 0; i < n; ++i) {
        long long sum = 0;
        for(int j = 1; j <= n; ++j) {
            sum += arr[(i + j) % n];
            if(sum % m == 0) {
                count++;
            }
        }
    }

    cout << count << endl;

    return 0;
}
```
