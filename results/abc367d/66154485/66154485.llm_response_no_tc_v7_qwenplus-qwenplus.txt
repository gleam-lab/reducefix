```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Prefix sum mod array
    vector<int> prefix_mod(n + 1, 0);
    for (int i = 0; i < n; i++) {
        prefix_mod[i + 1] = (prefix_mod[i] + arr[i]) % m;
    }

    // Use a frequency map to count occurrences of each mod value
    map<int, int> freq;
    int ans = 0;

    for (int i = 0; i <= n; i++) {
        // For each prefix_mod[i], count how many previous prefixes have the same mod
        // Subarrays between those indices will have sum divisible by m
        ans += freq[prefix_mod[i]];
        freq[prefix_mod[i]]++;
    }

    cout << ans;
    return 0;
}
```
