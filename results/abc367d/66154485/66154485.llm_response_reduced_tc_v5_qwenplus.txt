```cpp
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    vector<int> prefix_mod(m, 0);
    prefix_mod[0] = 1; // empty prefix has mod 0
    
    int prefix_sum = 0;
    for (int i = 0; i < n; i++) {
        prefix_sum = (prefix_sum + arr[i]) % m;
        if (prefix_sum < 0) prefix_sum += m; // handle negative mods
        ans += prefix_mod[prefix_sum];
        prefix_mod[prefix_sum]++;
    }
    
    cout << ans;
    return 0;
}
```
