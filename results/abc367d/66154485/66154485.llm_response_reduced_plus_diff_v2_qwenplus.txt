```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    vector<long long> prefix(n + 1, 0);
    
    // Calculate prefix sums
    for (int i = 0; i < n; i++) {
        prefix[i + 1] = prefix[i] + arr[i];
    }
    
    // Total sum of the array
    long long total_sum = prefix[n];
    
    // For each starting position
    for (int start = 0; start < n; start++) {
        // Map to store frequency of (sum % m) values
        map<long long, int> mod_count;
        mod_count[0] = 1; // Empty subarray has sum 0
        
        long long current_sum = 0;
        
        // Consider all subarrays starting at 'start' and going clockwise
        for (int len = 1; len < n; len++) {
            int idx = (start + len - 1) % n;
            current_sum += arr[idx];
            
            // We want current_sum % m == 0
            // This is equivalent to finding how many previous prefix sums
            // had the same modulo value
            long long target_mod = ((current_sum % m) + m) % m;
            
            if (target_mod == 0) {
                ans++;
            }
            
            // Add current state to the map for future reference
            mod_count[target_mod]++;
        }
    }
    
    cout << ans;
    return 0;
}
```
