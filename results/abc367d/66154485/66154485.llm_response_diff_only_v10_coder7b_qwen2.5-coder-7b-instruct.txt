```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    for (int start = 0; start < n; start++) {
        int sum = 0;
        for (int len = 1; len <= n - start; len++) {
            sum += arr[(start + len - 1) % n];
            if (sum % m == 0)
                ans++;
        }
    }
    cout << ans << endl;
    return 0;
}
```

1. **Vector Usage**: Replaced the array with a `std::vector<int>` for better flexibility and easier management.
2. **Loop Structure**: Adjusted the inner loop to correctly calculate the sum of subarrays starting from each index without using modulo arithmetic incorrectly.
3. **Output Formatting**: Ensured that the output is formatted correctly as per the guidelines.
