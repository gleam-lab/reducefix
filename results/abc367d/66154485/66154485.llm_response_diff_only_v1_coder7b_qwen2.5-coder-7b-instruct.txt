#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    // Fixing the inner loop to correctly calculate subarray sums
    for (int i = 0; i < n; i++) {
        int sum = 0;
        for (int j = i; j < n + i; j++) {
            sum += arr[j % n];
            if (sum % m == 0)
                ans++;
        }
    }
    
    cout << ans;
    return 0;
}
