#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < n; i++) {
        long long sum = 0;
        unordered_map<int, int> prefix_count;
        prefix_count[0] = 1;  // To handle the case where sum % m == 0 from the start

        for (int j = 0; j < n; j++) {
            sum = (sum + arr[(i + j) % n]) % m;
            if (prefix_count.count(sum)) {
                ans += prefix_count[sum];
            }
            prefix_count[sum]++;
        }
    }
    
    cout << ans;
    return 0;
}
