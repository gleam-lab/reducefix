#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    long long total = 0;
    vector<long long> prefix(n + 1, 0);
    for (int i = 0; i < n; i++) {
        prefix[i + 1] = prefix[i] + arr[i];
    }

    unordered_map<int, int> mod_counts;
    mod_counts[0] = 1;  // empty prefix
    for (int i = 1; i <= n; i++) {
        int mod = (prefix[i] % m + m) % m;
        total += mod_counts[mod];
        mod_counts[mod]++;
    }

    cout << total << endl;
    return 0;
}

1. Computes prefix sums of the array
2. Uses modular arithmetic to find subarrays whose sum is divisible by m
3. Counts these subarrays efficiently using a hash map to track remainders
4. Has O(N) time complexity, which is optimal for the problem constraints
