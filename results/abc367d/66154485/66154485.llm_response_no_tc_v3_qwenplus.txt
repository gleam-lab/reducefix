```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    
    // Check all possible contiguous subarrays
    for (int i = 0; i < n; i++) {
        long long sum = 0;
        for (int j = 0; j < n; j++) {
            // Add the current element
            sum += arr[(i + j) % n];
            // Check if this contiguous subarray sum is divisible by m
            if (sum % m == 0) {
                ans++;
            }
            // If we've gone through a complete cycle, break
            if (j == n - 1) break;
        }
    }
    
    cout << ans;
    return 0;
}
```
