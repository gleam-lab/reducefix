```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, ans = 0;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Initialize prefix sums modulo m
    vector<int> prefixSumModM(m, 0);
    int currentSum = 0;
    for (int i = 0; i < n; i++) {
        currentSum = (currentSum + arr[i]) % m;
        prefixSumModM[currentSum]++;
    }

    // Calculate the number of valid subarrays
    for (int i = 0; i < m; i++) {
        if (prefixSumModM[i] > 0) {
            // For each non-zero count, there are prefixSumModM[i]*(prefixSumModM[i]-1)/2 valid subarrays ending at any position
            ans += prefixSumModM[i] * (prefixSumModM[i] - 1) / 2;
        }
    }

    // Add subarrays that wrap around the array
    for (int i = 0; i < n; i++) {
        currentSum = (currentSum + arr[i]) % m;
        if (currentSum == 0) {
            ans++;
        }
    }

    cout << ans;
    return 0;
}
```
