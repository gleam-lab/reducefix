```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<int> arr(n);
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }

    // Prefix sum mod m
    long long total = 0;
    vector<int> prefix_mod(n);
    prefix_mod[0] = arr[0] % m;
    total += (prefix_mod[0] == 0);
    for (int i = 1; i < n; ++i) {
        prefix_mod[i] = (prefix_mod[i - 1] + arr[i]) % m;
        total += (prefix_mod[i] == 0);
    }

    // If all elements are considered
    if (m == 1) {
        cout << 1LL * n * (n + 1) / 2;
        return 0;
    }

    // Count frequency of each mod value in prefix sums
    map<int, int> freq;
    freq[0] = 1; // Include empty subarray

    long long count = 0;
    for (int i = 0; i < n; ++i) {
        // For circular array: extend prefix sums by one more element
        int val = prefix_mod[i];
        count += freq[val];
        freq[val]++;
    }

    cout << count;
    return 0;
}
```
