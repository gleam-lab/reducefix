```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Prefix sum modulo m
    long long ans = 0;
    vector<int> count(m, 0);
    vector<long long> prefix_sums(n + 1, 0);

    for (int i = 0; i < n; ++i) {
        prefix_sums[i + 1] = (prefix_sums[i] + arr[i]) % m;
        if (prefix_sums[i + 1] == 0) ans++;
        count[prefix_sums[i + 1]]++;
    }

    // For each remainder, add combinations of same remainders
    for (int r = 0; r < m; ++r) {
        long long c = count[r];
        ans += c * (c - 1) / 2;
    }

    cout << ans << endl;
    return 0;
}
```
