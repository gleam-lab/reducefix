```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    
    // Use prefix sums and modular arithmetic
    for (int start = 0; start < n; start++) {
        int sum = 0;
        for (int len = 1; len <= n - 1; len++) {  // Length from 1 to n-1 (contiguous subsequence)
            int idx = (start + len - 1) % n;
            sum = (sum + arr[idx]) % m;
            if (sum % m == 0) {
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
