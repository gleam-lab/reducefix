    vector<int> prefix_sum_mod_m(m, 0);
    for (int i = 0; i < n; i++) {
        prefix_sum_mod_m[arr[i] % m]++;
    }
    
    // Count pairs with sum divisible by m
    for (int i = 0; i < m; i++) {
        if ((i * 2) % m == 0) {
            ans += prefix_sum_mod_m[i] * (prefix_sum_mod_m[i] - 1) / 2;
        } else {
            ans += prefix_sum_mod_m[i] * prefix_sum_mod_m[m - i];
        }
    }
