```cpp
#include <iostream>
#include <vector>
#include <numeric>
#include <unordered_set>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<int> arr(n);
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }

    long long ans = 0;
    unordered_set<long long> sums;

    for (int start = 0; start < n; ++start) {
        long long sum = 0;
        for (int len = 1; len <= n; ++len) {
            int idx = (start + len - 1) % n;
            sum += arr[idx];
            if (sum % m == 0) {
                ++ans;
            }
            if (sums.find(sum) != sums.end()) {
                ans += sums[sum];
            }
            sums[sum]++;
        }
    }

    cout << ans << '\n';

    return 0;
}
```
