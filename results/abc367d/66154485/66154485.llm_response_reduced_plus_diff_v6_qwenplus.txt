```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    
    // For each starting position, we want to count subarrays whose sum % m == 0
    // We'll use prefix sums and modular arithmetic
    for (int start = 0; start < n; start++) {
        unordered_map<int, int> mod_count;
        mod_count[0] = 1; // Empty prefix has sum 0
        
        long long prefix_sum = 0;
        
        // Consider all subarrays starting at 'start' with various lengths
        for (int len = 1; len <= n; len++) {
            int idx = (start + len - 1) % n;
            prefix_sum += arr[idx];
            
            int mod = prefix_sum % m;
            if (mod < 0) mod += m; // Handle negative modulo
            
            // If we've seen this modulo before, it means there's a subarray
            // ending at current position with sum divisible by m
            ans += mod_count[mod];
            
            mod_count[mod]++;
        }
    }
    
    cout << ans;
    return 0;
}
```
