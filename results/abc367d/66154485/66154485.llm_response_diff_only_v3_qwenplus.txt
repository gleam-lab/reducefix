```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long ans = 0;
    for (int i = 0; i < n; i++) {
        long long sum = 0;
        for (int j = i; j < i + n; j++) {
            sum += arr[j % n];
            if (sum % m == 0 && j > i) {
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
