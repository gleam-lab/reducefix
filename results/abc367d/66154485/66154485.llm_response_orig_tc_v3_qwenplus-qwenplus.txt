```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // To store the count of prefix sums modulo m
    vector<long long> mod_count(m, 0);
    long long ans = 0;
    long long prefix_sum = 0;

    // We start with a prefix sum of 0 to handle subarrays that are themselves divisible by m
    mod_count[0] = 1;

    for (int i = 0; i < n; i++) {
        prefix_sum += arr[i];
        int mod = ((prefix_sum % m) + m) % m;  // Ensure non-negative mod

        ans += mod_count[mod];
        mod_count[mod]++;
    }

    cout << ans << endl;
    return 0;
}
```
