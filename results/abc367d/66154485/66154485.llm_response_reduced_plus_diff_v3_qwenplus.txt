```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    long long ans = 0;
    // Use prefix sums and map to count subarrays with sum divisible by m
    vector<int> prefix(n + 1);
    prefix[0] = 0;
    
    for (int i = 0; i < n; i++) {
        prefix[i + 1] = (prefix[i] + arr[i]) % m;
        // Ensure non-negative modulo
        if (prefix[i + 1] < 0) prefix[i + 1] += m;
    }
    
    // Count frequency of each remainder
    map<int, int> count;
    for (int i = 0; i <= n; i++) {
        count[prefix[i]]++;
    }
    
    // For each remainder, if it appears k times, we can form k*(k-1)/2 pairs
    for (auto& [rem, freq] : count) {
        ans += 1LL * freq * (freq - 1) / 2;
    }
    
    cout << ans;
    return 0;
}
```
