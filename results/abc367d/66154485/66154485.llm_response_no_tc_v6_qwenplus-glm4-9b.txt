
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> arr(n);
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    vector<int> prefix_sum(n + 1, 0);
    for (int i = 0; i < n; i++) {
        prefix_sum[i + 1] = prefix_sum[i] + arr[i];
    }
    
    long long ans = 0;
    for (int i = 1; i <= n; i++) {
        if (prefix_sum[i] % m == 0) {
            ans++;
        }
        
        if (prefix_sum[i] - prefix_sum[i - 1] % m == 0) {
            ans++;
        }
    }
    
    cout << ans;
    return 0;
}
```
