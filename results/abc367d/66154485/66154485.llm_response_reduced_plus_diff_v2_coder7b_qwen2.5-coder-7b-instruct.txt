#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    vector<int> arr(n);
    for(int i = 0; i < n; ++i) {
        cin >> arr[i];
    }

    long long ans = 0;
    for(int i = 0; i < n; ++i) {
        int sum = 0;
        for(int j = 1; j <= n; ++j) {
            sum += arr[(i + j) % n];
            if(sum % m == 0) {
                ++ans;
            }
        }
    }

    cout << ans << endl;
    return 0;
}
