#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN],sum[MAXN];
map<ll,ll> cnt;

int main(){
    ll n,m,ans=0;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
        a[i+n]=a[i];
    }
    sum[0] = 0;
    for(ll i=1;i<=2*n;i++){
        sum[i]=(sum[i-1]+a[i])%m;
    }
    
    // Initialize frequency of prefix sums for first n elements
    for(ll i=0;i<n;i++){
        cnt[sum[i]]++;
    }
    
    for(ll i=n;i<2*n;i++){
        // Remove the prefix that is now out of window
        cnt[sum[i-n]]--;
        // Add pairs where (sum[i] - sum[j]) % m == 0, i.e., sum[i] == sum[j]
        ans += cnt[sum[i]];
        // Add current prefix to window
        cnt[sum[i]]++;
    }
    
    printf("%lld",ans);
    return 0;
}
