typedef long long ll;
const int MAXN = 4e5 + 10;

ll a[MAXN], sum[MAXN], cnt[MAXN];

    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);

    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }

    // Compute prefix sums modulo m
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m;
    }

    // Count occurrences of each prefix sum
    unordered_map<ll, int> count;
    count[sum[0]]++;

    // Iterate through the second half of the array to find valid pairs
    for (ll i = n + 1; i <= 2 * n; i++) {
        ans += count[(sum[i] - sum[i - n] + m) % m]; // Ensure non-negative index
        count[sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}

### Explanation
The original code had several issues:
1. The `cnt` array was not used correctly to store the frequency of prefix sums.
2. The logic to update and query the count of prefix sums was incorrect.
3. The use of an unordered map is more efficient for counting occurrences of prefix sums compared to a static array.

This corrected version uses an unordered map to efficiently count the occurrences of each prefix sum and calculates the number of valid pairs using this count.
