#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN = 4e5 + 10;
ll a[MAXN], sum[MAXN], cnt[MAXN];

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i]; // Extend the array by mirroring values
    }
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m; // Compute cumulative sum with modulo
    }
    for (ll i = 1; i <= n; i++) {
        cnt[sum[i]]++; // Count occurrences in the first half
    }
    for (ll i = n + 1; i <= 2 * n; i++) {
        ans += cnt[sum[i]]; // Add counts from the mirrored half
        cnt[sum[i]]++; // Increment count for the current sum
    }
    printf("%lld", ans);
    return 0;
}
