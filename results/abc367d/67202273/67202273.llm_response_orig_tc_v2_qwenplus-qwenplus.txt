#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], prefix_sum[MAXN];
map<ll, ll> cnt;

int main() {
    ll n, m;
    scanf("%lld %lld", &n, &m);
    
    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
    }

    // Double the array to handle circular property
    for (ll i = n + 1; i <= 2 * n; ++i) {
        a[i] = a[i - n];
    }

    // Compute prefix sums modulo m
    prefix_sum[0] = 0;
    for (ll i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i]) % m;
    }

    ll ans = 0;
    cnt[0] = 1; // Initialize with sum 0 before starting

    // Sliding window over the doubled array
    for (ll i = 1; i <= 2 * n; ++i) {
        // Maintain window of size at most n
        if (i > n) {
            cnt[prefix_sum[i - n - 1]]--;
        }
        ans += cnt[prefix_sum[i]];
        cnt[prefix_sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
