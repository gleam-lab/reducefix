```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 4e5 + 10;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll n, m;
    cin >> n >> m;

    vector<ll> a(n);
    for (ll i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // Since all elements are same and large, prefix sum mod m will repeat
    // If all elements are equal to x, then the prefix sums mod m are:
    // 0, x%m, 2x%m, 3x%m, ..., (n-1)x%m

    ll x = a[0] % m;
    ll ans = 0;

    if (x == 0) {
        // Every subarray has sum divisible by m
        ans = n * (n + 1LL) / 2;
    } else {
        unordered_map<ll, ll> cnt;
        ll sum = 0;
        cnt[0] = 1;

        for (ll i = 0; i < n; ++i) {
            sum = (sum + x) % m;
            ans += cnt[sum];
            cnt[sum]++;
        }
    }

    cout << ans << "\n";
    return 0;
}
```
