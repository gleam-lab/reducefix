#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], prefix_sum[MAXN];
map<ll, ll> cnt;

int main() {
    ll n, m;
    scanf("%lld %lld", &n, &m);
    
    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];  // Duplicate array for circular handling
    }

    // Compute prefix sums modulo m
    for (ll i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i]) % m;
    }

    ll ans = 0;
    cnt[prefix_sum[0]] = 1;  // Initialize count of prefix sum 0

    for (ll i = 1; i <= n; ++i) {
        ans += cnt[prefix_sum[i]];
        cnt[prefix_sum[i]]++;
    }

    printf("%lld\n", ans);

    return 0;
}
