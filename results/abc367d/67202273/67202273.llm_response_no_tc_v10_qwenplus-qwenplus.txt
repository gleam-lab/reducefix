#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

int main() {
    ll n, m;
    scanf("%lld %lld", &n, &m);
    
    vector<ll> a(2 * n + 2, 0);
    vector<ll> prefix_sum(2 * n + 2, 0);
    
    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i]; // Duplicate the array to handle circular property
    }

    // Compute prefix sums modulo m
    for (ll i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i]) % m;
    }

    // Use a map to count occurrences of each prefix sum mod value
    map<ll, ll> cnt;
    ll ans = 0;

    // First pass: count valid subarrays ending in the first n elements
    for (ll i = 0; i <= n; ++i) {
        cnt[prefix_sum[i]]++;
    }

    // Second pass: slide the window and update counts
    for (ll i = n + 1; i <= 2 * n; ++i) {
        // Remove the prefix sum that is now out of the window
        cnt[prefix_sum[i - n]]--;

        // Count how many times this prefix sum has been seen
        ans += cnt[prefix_sum[i]];

        // Add current prefix sum into the count
        cnt[prefix_sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
