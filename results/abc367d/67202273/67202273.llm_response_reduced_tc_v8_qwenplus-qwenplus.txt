#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], sum[MAXN];
map<ll, ll> cnt;

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);

    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i]; // Duplicate the array for circular handling
    }

    // Compute prefix sums modulo m
    sum[0] = 0;
    for (ll i = 1; i <= 2 * n; ++i) {
        sum[i] = (sum[i - 1] + a[i]) % m;
    }

    // Reset count map and result
    cnt.clear();
    ans = 0;

    // Initialize the count with the first n prefix sums (excluding sum[0])
    for (ll i = 0; i <= n; ++i) {
        cnt[sum[i]]++;
    }

    // Sliding window: remove sum[i-n] and add sum[i] while counting matches
    for (ll i = n + 1; i <= 2 * n; ++i) {
        // Remove the element that's sliding out of the window
        cnt[sum[i - n]]--;

        // Count how many prefix sums equal to current sum[i]
        ans += cnt[sum[i]];

        // Add the current prefix sum to the map
        cnt[sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
