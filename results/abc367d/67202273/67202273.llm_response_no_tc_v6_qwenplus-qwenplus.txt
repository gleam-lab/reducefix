#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], prefix_sum[MAXN];
map<ll, ll> count_map;

int main() {
    ll n, m;
    scanf("%lld %lld", &n, &m);

    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];  // Duplicate array for circular handling
    }

    // Compute prefix sums modulo m
    for (ll i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i]) % m;
    }

    ll ans = 0;
    count_map[prefix_sum[0]] = 1;  // Initialize with prefix_sum[0] = 0

    for (ll i = 1; i <= n; ++i) {
        // Only track prefix sums in the current valid window [i-n+1, i]
        if (i > n) {
            count_map[prefix_sum[i - n]]--;
        }
        ans += count_map[prefix_sum[i]];
        count_map[prefix_sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
