#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN], sum[MAXN], cnt[MAXN];

int main() {
    ll n, m;
    scanf("%lld %lld", &n, &m);
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];  // Extend array by mirroring elements
    }
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m;  // Update sum array
        if (sum[i] < 0) sum[i] += m;  // Ensure non-negative sum
    }
    ll ans = 0;
    for (ll i = n + 1; i <= 2 * n; i++) {
        cnt[sum[i]]++;  // Increment count of current sum
        ans += cnt[sum[i]];  // Add current count to answer
        cnt[sum[i - n]]--;  // Decrement count of previous sum
    }
    printf("%lld", ans);
    return 0;
}
