#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN], sum[MAXN], cnt[MAXN];
int main(){
    ll n, m;
    scanf("%lld %lld", &n, &m);
    for(ll i = 1; i <= n; i++){
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }
    for(ll i = 1; i <= 2 * n; i++){
        sum[i + 1] = (sum[i] + a[i]) % m;
        // Fix: Increment cnt at i after updating sum[i+1]
        if (i <= n) cnt[sum[i + 1]]++;
    }
    for(ll i = n + 1; i <= 2 * n; i++){
        cnt[sum[i - n]]--;
        // Fix: Update ans with cnt[sum[i]] since cnt[sum[i - n]] has already been decreased
        ans = ans + cnt[sum[i]];
        cnt[sum[i]]++;
    }
    printf("%lld", ans);
    return 0;
}
