```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN], sum[MAXN], cnt[MAXN];
int main(){
    ll n, m;
    scanf("%lld %lld", &n, &m);
    for(ll i = 1; i <= n; i++){
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }
    for(ll i = 1; i <= 2 * n; i++){
        sum[i] = (sum[i - 1] + a[i]) % m;
        if(i <= n) cnt[sum[i]]++;
    }
    for(ll i = n + 1; i <= 2 * n; i++){
        cnt[sum[i - n]]--;
        ans = ans + cnt[sum[i]];
        cnt[sum[i]]++;
    }
    printf("%lld", ans);
    return 0;
}
```

### Explanation
1. **Initialization of sum array**: The `sum[i + 1]` in the original code was incorrect. It should be `sum[i]` because `sum[i]` holds the cumulative sum up to index `i`.
2. **Loop for updating sum and count**: The loop should start from `1` to `2 * n` to include all elements. The original code already did this, but the indexing in the sum array needed correction.
3. **Result Calculation**: The loop for calculating the answer is correct. It correctly subtracts the count of `sum[i - n]` and adds the count of `sum[i]` to the answer.
4. **Output**: The result is correctly printed using `printf`.
