#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], prefix_sum[MAXN];
map<ll, ll> count_map;

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);

    for (ll i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];  // Duplicate array for circular handling
    }

    // Precompute prefix sums modulo m
    for (ll i = 1; i <= 2 * n; ++i) {
        prefix_sum[i] = (prefix_sum[i - 1] + a[i]) % m;
    }

    // Initialize counts for the first n prefix sums
    for (ll i = 0; i <= n; ++i) {
        count_map[prefix_sum[i]]++;
    }

    // Sliding window over the circular array
    for (ll i = 1; i <= n; ++i) {
        // Remove the prefix sum that is sliding out of the window
        count_map[prefix_sum[i - 1]]--;

        // Count valid subarrays ending at current position
        ans += count_map[prefix_sum[i + n]];

        // Add the current prefix sum to the map for future use
        count_map[prefix_sum[i + n]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
