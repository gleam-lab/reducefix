#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN],sum[MAXN];
map<ll,ll> cnt;

int main(){
    ll n,m,ans=0;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
        a[i+n]=a[i];
    }
    for(ll i=1;i<=2*n;i++){
        sum[i]=(sum[i-1]+a[i])%m;
    }
    
    cnt[0]=n;  // Initialize with prefix sum 0 appearing n times (for each starting position)
    
    for(ll i=n;i>=1;i--){
        cnt[sum[i]]--;
        ans += cnt[sum[i+n]];
        cnt[sum[i+n]]++;
    }
    
    printf("%lld",ans);
    return 0;
}
