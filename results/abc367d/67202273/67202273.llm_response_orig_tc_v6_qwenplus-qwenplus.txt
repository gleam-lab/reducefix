#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int MAXN = 4e5 + 10;

ll a[MAXN], sum[MAXN], cnt[MAXN];

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);
    
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }
    
    // Compute prefix sums modulo m
    sum[0] = 0;
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m;
    }

    // Initialize count with the first n prefix sums
    memset(cnt, 0, sizeof(cnt));
    cnt[sum[0]] = 1;  // Include empty prefix

    // Sliding window technique
    for (ll i = 1; i <= 2 * n; i++) {
        // Maintain a window of size n
        if (i > n) {
            // Remove the prefix sum that's sliding out of the window
            cnt[sum[i - n - 1]]--;
        }
        
        // Count valid subarrays ending at i with sum divisible by m
        ans += cnt[sum[i]];
        
        // Add current prefix sum to the map if we're still within initial window
        if (i <= n) {
            cnt[sum[i]]++;
        }
    }

    printf("%lld\n", ans);
    return 0;
}
