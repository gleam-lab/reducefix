#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=4e5+10;
ll a[MAXN],sum[MAXN];
map<ll,ll> cnt;

int main(){
    ll n,m,ans=0;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
        a[i+n]=a[i];
    }
    sum[0] = 0;
    for(ll i=1;i<=2*n;i++){
        sum[i]=(sum[i-1]+a[i])%m;
    }
    
    // Initialize frequency map with prefix sums from first segment
    for(ll i=0;i<=n;i++){
        cnt[sum[i]]++;
    }
    
    // Sliding window: remove leftmost prefix, add new prefix
    for(ll i=n+1;i<=2*n;i++){
        cnt[sum[i-n]]--;
        ans += cnt[sum[i]];
        cnt[sum[i]]++;
    }
    
    printf("%lld",ans);
    return 0;
}
