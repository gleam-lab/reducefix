```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], sum[MAXN];
map<ll, ll> cnt;

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);

    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i]; // Duplicate array for circular property
    }

    // Compute prefix sums modulo m
    sum[0] = 0;
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m;
        if (i <= n) {
            cnt[sum[i]]++;
        }
    }

    // Sliding window over circular subarrays of length <= n
    for (ll i = n + 1; i <= 2 * n; i++) {
        // Remove the element exiting the window
        cnt[sum[i - n]]--;

        // Add number of previous prefix sums equal to current
        ans += cnt[sum[i]];

        // Add the current prefix sum to the map
        cnt[sum[i]]++;
    }

    printf("%lld\n", ans);
    return 0;
}
```
