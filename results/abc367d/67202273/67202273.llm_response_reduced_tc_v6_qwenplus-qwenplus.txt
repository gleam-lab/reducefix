#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 4e5 + 10;

ll a[MAXN], sum[MAXN], cnt[MAXN];

int main() {
    ll n, m, ans = 0;
    scanf("%lld %lld", &n, &m);

    // Read input and duplicate the array
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }

    // Compute prefix sums modulo m
    sum[0] = 0;
    for (ll i = 1; i <= 2 * n; i++) {
        sum[i] = (sum[i - 1] + a[i]) % m;
        if (i <= n) cnt[sum[i]]++;
    }

    // Sliding window to count valid subarrays of length ≤n with sum mod m == 0
    for (ll i = 1; i <= 2 * n; i++) {
        // Remove the element going out of the window
        if (i - n > 0) {
            cnt[sum[i - n - 1]]--;
        }

        // Count how many times this prefix sum has been seen before
        ans += cnt[sum[i]];

        // Only add to map if we're still building initial window
        if (i <= 2 * n - 1 && i < n) {
            cnt[sum[i]]++;
        }
    }

    printf("%lld\n", ans);
    return 0;
}
