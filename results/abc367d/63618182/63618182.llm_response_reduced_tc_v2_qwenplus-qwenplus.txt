#include <bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<ll> s(n + 1);
    for (int i = 1; i <= n; i++) {
        s[i] = s[i - 1] + a[i];
    }

    map<ll, ll> mp;
    ll ans = 0;
    ll L = s[n];

    // We want s[j] ≡ s[i] - L mod M
    // Which is equivalent to s[j] ≡ (s[i] - L) mod M
    // So for each prefix sum s[i], we look for:
    // (s[i] - L) mod M in the map

    for (int i = 0; i <= n; i++) {
        ll current = s[i] % M;
        ll target = (current - L % M + M) % M;
        ans += mp[target];
        mp[current]++;
    }

    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
