```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    vector<ll> prefix(n + 1);
    for (int i = 1; i <= n; i++) {
        prefix[i] = (prefix[i - 1] + a[i]) % M;
    }

    map<ll, ll> count;
    count[0] = 1; // To handle subarrays starting from index 0
    ll result = 0;
    ll curr_mod = 0;

    for (int i = 1; i <= n; ++i) {
        curr_mod = (curr_mod + a[i]) % M;
        // We want the number of previous prefix sums with same modulo value
        // because (prefix[j] - prefix[i]) % M == 0 means sum from i+1 to j is divisible by M
        result += count[curr_mod]++;
    }

    cout << result << endl;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while(t--) {
        solve();
    }
    return 0;
}
```
