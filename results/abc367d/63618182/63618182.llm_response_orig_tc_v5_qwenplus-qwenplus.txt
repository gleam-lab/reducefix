```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10;
const int mod = 1e9 + 7;

void solve() {
    int n;
    ll M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    vector<ll> s(n + 1);
    for (int i = 1; i <= n; ++i) {
        s[i] = (s[i - 1] + a[i]) % M;
    }

    map<ll, ll> cnt;
    ll ans = 0;
    cnt[0] = 1; // To count subarrays starting from index 0

    for (int i = 1; i <= n; ++i) {
        // For current prefix sum s[i], we want to find number of j < i such that
        // (s[i] - s[j]) % M == 0 => s[i] == s[j] (mod M)
        ans += cnt[s[i]]++;
    }

    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
