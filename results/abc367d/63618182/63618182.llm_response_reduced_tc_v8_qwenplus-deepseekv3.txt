using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

    }

    for(int i=1; i<=n; i++){
        s[i]=s[i-1]+a[i];

    mp[0] = 1; // Initialize with sum 0 at index 0
    ll ans=0;
    for(int i=1; i<=n; i++){
        ll mod_val = (s[i] % M + M) % M; // Ensure positive modulo
        ans += mp[mod_val];
        mp[mod_val]++;
    }

}

int main()
