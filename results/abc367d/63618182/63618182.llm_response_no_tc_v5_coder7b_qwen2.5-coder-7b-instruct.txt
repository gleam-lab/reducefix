```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
	int n,M;
	cin>>n>>M;
	vector<int>a(n+10);
	for(int i=1; i<=n; i++){
		cin>>a[i];
	}
	vector<int>s(n+10);
	for(int i=1; i<=n; i++){
		s[i]=s[i-1]+a[i];
	}
	map<ll, int>mp;
	ll ans=0,L=s[n];
	for(int i=1; i<=n; i++){
		ans+=mp[(L - s[i] + M) % M];
		mp[s[i]%M]++;
	}
	cout<<ans<<'\n';
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
		solve();
	}
	return 0;
}
```
In the original code, there were two issues:
1. The map was used incorrectly for counting occurrences of remainders. It should be updated after calculating the answer to avoid double-counting.
2. The modulo operation needed to ensure it handles negative numbers correctly. This is achieved by adding `M` before taking the modulo, ensuring that the result is always non-negative.
