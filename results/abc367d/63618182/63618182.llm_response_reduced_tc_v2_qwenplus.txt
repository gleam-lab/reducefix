#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int> a(n + 10);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<int> s(n + 10);
    for (int i = 1; i <= n; i++) {
        s[i] = s[i-1] + a[i];
    }
    map<int, int> mp;
    ll ans = 0;
    ll L = s[n];
    
    // Initialize with prefix sum 0 to account for subarrays starting at index 0
    mp[0]++;
    
    for (int i = 1; i <= n; i++) {
        // We want s[j] ≡ s[i] (mod M) for j < i, which gives sum from j+1 to i divisible by M
        int target = s[i] % M;
        ans += mp[target];
        
        mp[s[i] % M]++;
    }
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
