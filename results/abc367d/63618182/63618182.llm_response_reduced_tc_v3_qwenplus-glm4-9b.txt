#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10, mod=1e9+7;

void solve() {
    int n, M;
    ll L = 0;
    cin >> n >> M;
    vector<ll> a(n+10);
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
        L = (L + a[i]) % mod;
    }
    vector<int> s(n+10);
    for(int i = 1; i <= n; i++) {
        s[i] = (s[i-1] + a[i]) % mod;
    }
    map<int, int> mp;
    ll ans = 0;
    for(int i = 1; i <= n; i++) {
        ans = (ans + mp[(s[i] - L + mod) % mod]) % mod;
    }
    cout << ans << endl;
}

int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while(t--) {
        solve();
    }
    return 0;
}
