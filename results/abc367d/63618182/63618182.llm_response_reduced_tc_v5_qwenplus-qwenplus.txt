```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;

const int N = 2e5 + 10;
const int mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<ll> s(n + 1);
    for(int i = 1; i <= n; i++) {
        s[i] = (s[i - 1] + a[i]) % M;
    }

    map<ll, ll> mp;
    ll ans = 0;
    mp[0] = 1; // Handle prefix sums that are directly divisible by M
    
    for(int i = 1; i <= n; i++) {
        // (s[i] - L) % M == 0 -> s[i] == L mod M
        // We want to count number of previous prefix sums with same mod value
        ans += mp[s[i]];
        mp[s[i]]++;
    }
    
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while(t--) {
        solve();
    }
    return 0;
}
```
