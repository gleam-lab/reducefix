```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
typedef pair<int, int> PII;

void solve() {
    int n;
    ll M;
    cin >> n >> M;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    // Prefix sums
    vector<ll> prefix(n + 1);
    for (int i = 1; i <= n; ++i) {
        prefix[i] = (prefix[i - 1] + a[i]) % M;
    }

    map<ll, ll> count;
    ll result = 0;

    // We want to count the number of pairs (i, j) such that (prefix[j] - prefix[i]) % M == 0
    // This is equivalent to prefix[j] % M == prefix[i] % M
    // So we just count frequencies of each remainder and accumulate combinations
    for (int i = 0; i <= n; ++i) {
        result += count[prefix[i]];
        count[prefix[i]]++;
    }

    cout << result << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
```
