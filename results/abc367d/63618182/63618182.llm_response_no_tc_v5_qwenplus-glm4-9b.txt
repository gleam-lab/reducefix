#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
    int n,M;
    cin>>n>>M;
    vector<int>a(n+10);
    for(int i=1; i<=n; i++){
        cin>>a[i];
    }
    vector<int>s(n+10);
    for(int i=1; i<=n; i++){
        s[i] = (i == 1) ? a[i] : (s[i-1] + a[i]);
    }
    map<int, int>mp;
    ll ans=0;
    ll L = s[n];
    for(int i=1; i<=n; i++){
        int modified_index = (s[i] - L + mod) % mod;
        ans += mp[modified_index];
        mp[s[i] % mod]++;
    }
    cout<<ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    int t=1;
    while(t--){
        solve();
    }
    return 0;
}

3. The variable `L` should be initialized as a long long to avoid overflow when subtracting from `s[i]`.
