```cpp
#include<bits/stdc++.h>

using namespace std;
using ll=long long;
typedef pair<int, int>PII;
const int N=2e5+10,mod=1e9+7;

void solve()
{
	int n,M;
	cin>>n>>M;
	vector<int>a(n+10);
	for(int i=1; i<=n; i++){
		cin>>a[i];
	}
	vector<int>s(n+10);
	for(int i=1; i<=n; i++){
		s[i]=(s[i-1]+a[i])%M;
	}
	map<int, int>mp;
	mp[0] = 1; // important initialization
	ll ans=0;
	for(int i=1; i<=n; i++){
		ans += mp[s[i]];
		mp[s[i]]++;
	}
	cout<<ans;
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(0);
	ll t=1;
	while(t--){
		solve();
	}
	return 0;
}
```

Key corrections made:
1. Removed the unnecessary total sum calculation (L)
2. Fixed the prefix sum calculation to store mod M values directly
3. Properly initialized the map with mp[0] = 1 to count subarrays starting from index 0
4. Simplified the counting logic to just track how many times each mod value has appeared
5. The answer is now correctly accumulated by looking up how many previous prefix sums had the same mod value

This solution correctly counts the number of subarrays whose sum is divisible by M in O(n) time with O(M) space complexity.
