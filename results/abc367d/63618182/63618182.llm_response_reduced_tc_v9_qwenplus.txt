#include<bits/stdc++.h>

using namespace std;
using ll = long long;
typedef pair<int, int> PII;
const int N = 2e5 + 10, mod = 1e9 + 7;

void solve()
{
    int n, M;
    cin >> n >> M;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    vector<int> s(n + 1);
    for (int i = 1; i <= n; i++) {
        s[i] = s[i - 1] + a[i];
    }
    map<int, int> mp;
    ll ans = 0;
    ll L = s[n];
    
    // We are looking for subarrays where sum % M == 0
    // For subarray [i+1, j], sum = s[j] - s[i], we want (s[j] - s[i]) % M == 0
    // => s[j] % M == s[i] % M
    // So count frequency of prefix sums modulo M
    
    mp[0] = 1; // empty prefix at index 0
    for (int i = 1; i <= n; i++) {
        int r = s[i] % M;
        if (r < 0) r += M;
        ans += mp[r];
        mp[r]++;
    }
    cout << ans;
}

int main()
{
    cin.tie(0);
    ios::sync_with_stdio(0);
    ll t = 1;
    while (t--) {
        solve();
    }
    return 0;
}
