N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和を計算（時計回りの距離）
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# 全体の周長
total = cum[N]

# mod M の値の出現回数をカウント
count_mod = defaultdict(int)
answer = 0

    mod = val % M
    answer += count_mod[mod]
    count_mod[mod] += 1
