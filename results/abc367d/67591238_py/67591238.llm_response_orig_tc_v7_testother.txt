```python
    count_mod[val % M] += 1

answer = 0

# 時計回りの距離がMの倍数になるペア
for count in count_mod.values():
    answer += count * (count - 1) // 2

# 反時計回りの距離がMの倍数になるペア (時計回りがMの倍数でない)
# total - (cum[j] - cum[i]) ≡ 0 (mod M)
# => cum[j] - cum[i] ≡ total (mod M)
# => cum[j] ≡ cum[i] + total (mod M)

count_target = defaultdict(int)
target = total % M
for val in cum:
    needed = (val + target) % M
    count_target[needed] += 1

for val in cum:
    current_mod = val % M
    answer += count_target[current_mod]

# 同一インデックスのペアを引く (i == j)
answer -= (N + 1)

# 実際には、(i,j) と (j,i) を両方カウントしているが、
# ここでは順序付きペアを数えているため、上記の計算は正しい。

count_rev = defaultdict(int)
```

```python
from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# mod M の値の出現回数
count_mod = defaultdict(int)
for val in cum:
    count_mod[val % M] += 1

answer = 0

# 時計回りの距離がMの倍数になるペア
for count in count_mod.values():
    answer += count * (count - 1) // 2

# 反時計回りの距離がMの倍数になるペア（時計回りはMの倍数でない）
# 反時計回りの距離がMの倍数 <=> 時計回りの距離 ≡ total (mod M)
count_target = defaultdict(int)
total = cum[N]
target = total % M

# cum[j] - cum[i] ≡ total (mod M)  <=> cum[j] ≡ cum[i] + total (mod M)
for val in cum:
    count_target[(val + target) % M] += 1

for val in cum:
    answer += count_target[val % M]

# 各iについて、j=iのケース（距離0）を引く
answer -= (N + 1)

print(answer)
```
