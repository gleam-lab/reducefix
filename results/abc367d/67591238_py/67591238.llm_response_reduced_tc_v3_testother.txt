from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# mod M の値の出現回数
count_mod = defaultdict(int)
answer = 0

for val in cum:
    mod = val % M
    answer += count_mod[mod]  # すでに同じmodのものとペアになれる
    count_mod[mod] += 1

# 同じ地点からの移動（s == t）を除外するため、
# 累積和の最初の0が余分にカウントされないように調整する必要はないが、
# 実際には s != t かつ clockwise 移動なので、円形での距離を考える。

# 上記コードは累積和の同じmodをもつペアを数えているが、
# これは「時計回りにsからtへ行く距離がMの倍数」という条件を満たす(s,t)（s <= t）を数えている。
# ただし、この問題ではs != tだが、循環しているため、逆方向ではなく時計回りのみ。

# しかし、問題は「時計回りにsからtへ行く最小ステップ数がMの倍数」
# つまり、cum[t] - cum[s] ≡ 0 (mod M) → cum[t] ≡ cum[s] (mod M)
# よって、同じmodの累積和の値を持つインデックス同士が条件を満たす。

# ただし、s != t だが、上記コードではインデックスの順序でs < tとして扱っているので問題ない。
# しかし、入力例: N=1, M=1, A=[136872]
# cum = [0, 136872]
# mod 1 では 0%1=0, 136872%1=0
# count_mod[0] が最初に0 → answer += 0, その後1
# 次にval=136872, mod=0 → answer += 1 → answer=1
# つまり、(s=1, t=1) 相当？いや、cum[0]とcum[1] → s=1, t=2? だがN=1なのでt=1

# 誤解していた点: rest area 1からrest area 1への移動（一周）は許されるか？
# 問題文: "from rest area s to rest area t (s ≠ t)"
# よって、s ≠ t である必要がある。

# 上記コードでは、cum[0] と cum[1] は、s=1, t=1（一周）を意味する？→ いや、s=1, t=1+1=2だがN=1なのでt=1
# つまり、s=1, t=1 となり、s==t なので不正。

# 累積和のインデックスiは、rest area i+1の直前まで来たことを表す？
# 定義: cum[0]=0 → 開始位置（rest area 1の手前）
# cum[1] = A0 → rest area 1から2へ（N=1なら2→1）→ rest area 1に戻る？

# 正しい解釈:
# cum[i] は、rest area 1 から rest area i+1 へ時計回りに行くまでの距離（i=0..N）
# つまり、cum[0] = 0 → rest area 1
# cum[1] → rest area 2（N=1ならrest area 1に戻る？）

# しかし、N=1の場合、A_1はrest area 1から2へ、つまり1から1へ（一周）
# よって、cum[1] = A0 → rest area 1から1への距離（一周）

# このとき、s=1, t=1 だが s≠tではない → 不可
# したがって、(s,t) ペアで s≠t が必要。

# 上記コードでは、cum[0]とcum[1]を使って、s=1, t=1 を意味してしまう（距離=cum[1]-cum[0]=A0）
# しかし、s=1, t=1 → s==t → 無効。

# どうすればよいか？
# 実は、累積和のインデックスiとj（i < j）について、
# rest area i+1 から rest area j+1 への距離は cum[j] - cum[i]
# ここで、s = i+1, t = j+1
# s ≠ t とは、i+1 ≠ j+1 → i ≠ j
# しかし、i < j なら自動的に i ≠ j なので大丈夫？

# いや、iとjが異なればsとtは異なる。
# 例えば、i=0,j=1 → s=1, t=2 → N=1ならt=2は存在しない → rest areas are 1-indexed and modulo N

# 正しいラベリング:
# cum[0]: rest area 1 の位置（開始）
# cum[1]: rest area 2 の位置（N=1ならこれはrest area 1に戻る）
# つまり、rest area k の位置は cum[k-1] （k=1..N）

# したがって、s = i+1, t = j+1 とするとき、
# sとtは1..Nの範囲。j+1 > N の場合？ → jは0..Nまでで、j=Nのときt=N+1 → これは1
# つまり、累積和のインデックスは rest area 1 を指す。

# 重要な点：累積和の長さはN+1で、cum[0], cum[1], ..., cum[N]
# ここで、cum[N] は一周の合計距離。
# そして、任意のs,t（s≠t）について、時計回りの距離は (cum[t-1] - cum[s-1]) % total ではなく、
# 単純に cum[t-1] - cum[s-1] （s-1 < t-1 のとき）
# または、もしs-1 >= t-1 なら、時計回りに一周以上？ → いや、問題は「minimum number of steps required to walk clockwise」
# だが、clockwiseにsからtへ、というのは、sからtまで時計回りに進む最短経路（一周以上は考慮しない）。

# しかし、問題文には「minimum number of steps required to walk clockwise from rest area s to rest area t (s≠t) is a multiple of M」とある。
# ここで、「clockwise」に限定されているので、反時計回りは考慮しない。
# また、「minimum」なので、一周以上は考慮しない（常に直接時計回りに進む）。

# よって、s < t なら cum[t-1] - cum[s-1]
# s > t なら、時計回りに s -> s+1 -> ... -> N -> 1 -> ... -> t なので、
# cum[N] - cum[s-1] + cum[t-1] - cum[0] = cum[N] - cum[s-1] + cum[t-1]
# これは、全体の合計から反時計回りの部分を引くのと同じ。

# しかし、実は、cum配列を拡張せずに処理するのは難しい。

# 別のアプローチ：
# 全てのs,t（s≠t）について、時計回りの距離をD(s,t)とする。
# D(s,t) = (cum[t-1] - cum[s-1]) if s < t
#        = (cum[N] + cum[t-1] - cum[s-1]) if s > t

# これは、D(s,t) = (cum[t-1] - cum[s-1]) mod cum[N] ではない。
# なぜなら、mod M の話であって、mod cum[N] ではない。

# 条件は D(s,t) ≡ 0 (mod M)

# ここで、D(s,t) = (cum[t-1] - cum[s-1]) if s < t
#               = (cum[N] + cum[t-1] - cum[s-1]) if s > t

# これを一つの式にするのは難しい。

# 別解法：全ての時計回りのパスは、cum[j] - cum[i] （i < j）で表現できるが、
# ただし、j-i <= N-1 という制約がある？ → いや、一周以上は最小ではないので、最大でもN-1個の区間。

# しかし、問題は「clockwise」に限定されていて、かつ「minimum」なので、sからtへ直接時計回りに行く（一周未満）。
# したがって、s < t なら cum[t-1] - cum[s-1]
# s > t なら、cum[N] - cum[s-1] + cum[t-1] → これは cum[t-1] - cum[s-1] + cum[N]

# つまり、D(s,t) = (cum[t-1] - cum[s-1]) if s < t
#                = (cum[t-1] - cum[s-1] + cum[N]) if s > t

# そして、これが M の倍数。

# 今、累積和の配列を cum[0..N] とする。
# ここで、s = i+1, t = j+1 とすると、
# i, j ∈ {0,1,...,N-1} となる。

# s < t ⇔ i < j
# s > t ⇔ i > j

# D(s,t) = 
#   if i < j: cum[j] - cum[i]
#   if i > j: cum[j] - cum[i] + cum[N]

# これが ≡ 0 (mod M)

# つまり、
# case 1: i < j → cum[j] - cum[i] ≡ 0 → cum[j] ≡ cum[i] (mod M)
# case 2: i > j → cum[j] - cum[i] + cum[N] ≡ 0 → cum[j] - cum[i] ≡ -cum[N] (mod M) → cum[j] ≡ cum[i] - cum[N] (mod M)

# したがって、全ペア(i,j)（i≠j）について、
# - i < j かつ cum[i] ≡ cum[j] (mod M)
# - i > j かつ cum[j] ≡ cum[i] - cum[N] (mod M)

# これを効率的に数える。

# 方法：
# まず、cum[0] から cum[N-1] を使う（cum[N]は全体の合計）
# cum[N] = total

# Step 1: i < j で cum[i] % M == cum[j] % M を満たすペアを数える
# Step 2: i > j で (cum[i] - total) % M == cum[j] % M を満たすペアを数える

# ただし、i,j は 0 から N-1 まで。

# なぜ cum[N] は使わないのか？
# cum[i] (i=0..N-1) が各 rest area i+1 の位置を表す。
# cum[N] は一周の合計で、位置としては cum[0] と同じ（0）だが、値はtotal。

# したがって、累積和の配列として使用するのは cum[0] から cum[N-1] まで。

# 修正：
# cum_rest = cum[0:N]  # rest area 1 から N の位置

# しかし、元のコードでは cum[0..N] を全部使っており、
# cum[N] も含めて mod を取っていた。

# なぜ cum[N] を含めるとまずいのか？
# 例: N=1, M=1, A=[136872]
# cum = [0, 136872]
# cum_rest = [0] （index 0 だけ）
# すると、ペアは存在しない → answer=0 → 正解。

# つまり、累積和の最後の要素 cum[N] は、次の rest area 1 への到達を表すが、
# それはすでに cum[0] と同じ場所（一周後）なので、重複してカウントしてはならない。

# したがって、使用すべきは cum[0] から cum[N-1] まで。

# しかし、そうすると、s=1, t=2（N>=2）の場合、cum[1] が必要だが、N=2なら cum[0], cum[1] を使い、cum[2]は不要。

# 一般に、rest area k の位置は cum[k-1] （k=1..N）なので、必要なのは cum[0] から cum[N-1]。

# よって、修正：ループで見る cum の値は cum[0] から cum[N-1] まで。

# しかし、上のコードでは cum[0] から cum[N] まで見ている。

# したがって、以下のようになる：

# N=1のとき、cum = [0, total] → 使用するのは cum[0] だけ → ペアなし → answer=0

# 他のケース：例えば N=2, A=[2,3], M=5
# total = 5
# cum = [0,2,5] → 使用するのは cum[0]=0, cum[1]=2
# ペア (i,j) with i<j: 
#   i=0,j=1: cum[0]%5=0, cum[1]%5=2 → not equal
#   → 0ペア
# i>j: 
#   i=1,j=0: (cum[1]-total)%5 = (2-5)%5 = (-3)%5=2, cum[0]%5=0 → not equal
# → 0ペア
# しかし、実際には：
# s=1,t=2: D=2 → 2%5≠0
# s=2,t=1: D=3 → 3%5≠0
# どちらもMの倍数でないので、0で正しい。

# 別の例：N=2, A=[2,3], M=1 → 全ての整数は1の倍数
# s≠t なので、(1,2)と(2,1)の2ペア
# 上記方法：
# cum[0]=0, cum[1]=2
# i<j: i=0,j=1 → 0%1=0, 2%1=0 → 同じ → 1ペア
# i>j: i=1,j=0 → (cum[1]-5)%1 = (2-5)%1=0, cum[0]%1=0 → 同じ → 1ペア
# 合計2 → 正しい。

# よって、アルゴリズム：
# total = cum[N]
# cum_rest = cum[0:N]  # 長さN
# answer = 0
# 
# # Step 1: i < j で cum[i] % M == cum[j] % M
# count_mod = defaultdict(int)
# for j in range(N):
#     mod_val = cum[j] % M
#     answer += count_mod[mod_val]
#     count_mod[mod_val] += 1
# 
# # Step 2: i > j で (cum[i] - total) % M == cum[j] % M
# # これは、i > j なので、j < i を固定して、i を走査
# # 別のカウント配列で、cum[i] - total のmodを先に集計？
# # あるいは、jをループして、条件を満たすi>jを探すのはO(N^2) → 不可能
# 
# # 代わりに：
# # Step 2: i > j かつ cum[j] % M == (cum[i] - total) % M
# # → cum[j] ≡ cum[i] - total (mod M)
# # → cum[i] ≡ cum[j] + total (mod M)
# # しかし、i>jなので、jを小さい方から見て、i>jのcum[i]のうち (cum[i] % M) == (cum[j] + total) % M を数える
# 
# # 事前に、cum[i] % M (i=0..N-1) の頻度を取っておき、それから引いていく？
# 
# # 方法：Step 2のために、すべてのcum[i] % M (i=0..N-1) をカウントした辞書を作成
# # それをfreqとする
# # そして、j=0からN-1まで、
# #   target = (cum[j] + total) % M
# #   answer += freq[target]
# #   そして、cum[j] % M を freqから1減らす（i>j なので、jより小さいiは除外）
# 
# # つまり、右側にあるi（j<i）のうち、cum[i] % M == (cum[j] + total) % M を満たすものを数える。
# 
# # これでO(N)
