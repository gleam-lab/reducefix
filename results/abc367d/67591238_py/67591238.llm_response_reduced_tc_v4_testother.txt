from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和（時計回りの距離）
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# 全体の周長
total = cum[N]

# mod M の値の出現回数をカウント
count_mod = defaultdict(int)
answer = 0

for val in cum:
    mod = val % M
    answer += count_mod[mod]
    count_mod[mod] += 1

# 同じ地点からの移動（s == t）は除く必要があるが、
# cum配列には0から始まるN+1個の点があり、同じmodを持つ累積和の組み合わせがs→tの距離がMの倍数となる。
# ただし、問題では s ≠ t なので、s == t のケースはそもそも累積和の差が0（= Mの倍数）でも、
# 配列上で同じインデックス同士のペアは存在しない（異なるi,jであればs≠t）。
# よって、このままのロジックでよいが、注意点として total % M == 0 のときに
# 全周がMの倍数なら、すべてのsについてs→sのパス（全周）もMの倍数になるが、
# 問題は「sからtへの最短距離」であり、時計回りのみを考える場合、s→sは0ステップ（空パス）であり、
# 通常は考慮されない。また、入力制約よりN>=2だが、サンプルにN=1がある。

# 実際の問題: N=1のとき、rest areaは1つだけ。
# sとtは異なる必要がある(s ≠ t)が、rest areaが1つしかないので、そのようなペア(s,t)は存在しない。
# よって答えは0。

# 上記のコードはN=1のとき cum = [0, A0] → 0とA0の2つの値
# 両方mod Mが等しければanswer += 1（初期0のあと、0%1==0でcount_mod[0]が1になり、次にA0%M==0ならanswer+=1）
# つまり、A0 % M == 0 ならanswer=1、そうでなければ0。

# しかし、N=1のとき、sとtは異なっていなければならないが、rest areaは1つしかない。

# 問題文に「2 <= N」と明記されているが、サンプルにN=1があるので、おそらく制約が間違っているか、
# テストケースが緩く設定されている。

# 再確認：問題文のConstraintsに「2 <= N」とあるが、サンプルにN=1。
# これは誤りの可能性が高いが、テストケースにN=1がある以上、それを処理する必要がある。

# よって、N=1の場合は、s≠tを満たすペアが存在しないので、答えは0。

# しかし、上記のアルゴリズムは、N=1の場合でも cum[0]=0, cum[1]=A0 に対して、
# 両者の差（=A0）がMの倍数なら、(s=1, t=1)ではないか？いや、違う。

# 累積和の差 cum[j] - cum[i] は、rest area i+1 から j までの時計回りの距離。
# よって、i=0, j=1 なら、rest area 1 から rest area 1 への一周？
# いや、rest area 1 から 2 へだが、N=1なのでrest area 2は存在しない。

# 正しくは、累積和のインデックスiはrest area i+1の手前を意味する。
# cum[0]: rest area 1 の出発点
# cum[1]: rest area 1 → rest area 2 (but rest area 2 is rest area 1 again when N=1)

# つまり、i < j なる (i,j) に対して、rest area (i+1) から rest area (j+1) への距離。
# ただし、j+1 > N なら mod N で考える必要がある。

# しかし、この手法では、円形の道を線形に展開して累積和を使い、
# 「時計回りの距離がMの倍数」という条件を「cum[j] - cum[i] ≡ 0 (mod M)」と表現している。

# 注意：累積和のインデックスiは、rest area i+1の位置を示す（i=0,...,N）。ただし、i=Nは再びrest area 1の位置。

# ここで、i+1 と j+1 は 1..N の範囲だが、i=N なら i+1 = N+1 → 1 と解釈される。

# しかし、iとjは0..NのN+1個の点。
# 例えば、i=0, j=N → cum[N]-cum[0] = total → rest area 1 から rest area 1 への一周。
# これはs=tであり、問題で禁止されている。

# 正しい対応:
# - cum[0]: rest area 1 の出発点

# - ...

# よって、cum[k] に対応するrest area番号は:
#   if k < N: k+1

#   t = (j % N) + 1
# とすると、sとtが異なるかどうかをチェックできる。

# しかし、O(N^2)では不可能。累積和のmodで集計している。

# しかし、それは複雑。

# 別の視点：問題は「時計回りの距離がMの倍数」という条件だが、
# その距離は s から t への最短距離ではない。問題文に「minimum number of steps required to walk clockwise from rest area s to rest area t」とあるが、
# ここでの「minimum」は、時計回りと反時計回りの比較ではなく、与えられた経路が時計回りのみという意味か？

# 実際、問題文の最初に「It takes A_i steps to walk clockwise from rest area i to rest area i+1」とあり、
# そして「minimum number of steps required to walk clockwise from s to t」とある。
# 「clockwise only」なので、反時計回りは考慮しない。したがって、sからtへの時計回りの距離は一意に定まる（s==tなら0だが、s≠t）。

# しかし、sからtへの時計回りの距離は、sがtより後ろにある場合、一周以上もありうるか？
# 例えば、s=2, t=1 (N>=2) → 時計回りに 2→3→...→N→1 という経路。

# したがって、距離は cum[t-1] - cum[s-1] （ただし負なら total を足す）ではなく、
# 単に (cum[t-1] - cum[s-1]) mod total ではない。時計回りに進むので、sからtまで直接。

# 距離 d = cum[t-1] - cum[s-1]  if t > s
#         = cum[t-1] + (total - cum[s-1]) if t < s

# あるいは、d = (cum[t-1] - cum[s-1]) % total だが、負の場合を考えて、

# i < j なるインデックス i,j に対して、距離は cum[j] - cum[i] となる。
# ここで、iに対応するrest areaは i+1（i<N）、jは j+1（j<N）、
# しかし j=N なら j+1 は1（mod N）。

# 重要なのは、インデックスiとjが同じrest areaを指す場合（つまり (i % N) == (j % N)）は、s=tとなる。

# そして、s=t は問題で禁止されている。

# したがって、我々のカウント(answer)には、s=tとなるペアも含まれている可能性がある。

#   - j = i + k*N
#   0<=i<j<=N より、k=1 なら j=i+N
#   i>=0, j=i+N<=N → i<=0 → i=0, j=N

# 例えば、N=2のとき、i=0,j=2 → (0%2)=0, (2%2)=0 → s=(0%2)+1=1, t=(2%2)+1=1 → s=t

# よって j-i = N → j = i+N
# i>=0, j=i+N<=N → i<=0 → i=0, j=N
