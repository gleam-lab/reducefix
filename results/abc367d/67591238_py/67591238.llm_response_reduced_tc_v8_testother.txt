from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和（0から始まる）
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# mod M の値の出現回数をカウント
count_mod = defaultdict(int)
answer = 0

for val in cum:
    mod = val % M
    # 同じ余りを持つ累積和同士の組み合わせが、Mの倍数の距離を意味する
    answer += count_mod[mod]
    count_mod[mod] += 1
