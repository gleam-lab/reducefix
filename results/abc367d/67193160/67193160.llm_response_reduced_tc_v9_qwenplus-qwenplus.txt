```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 2e6 + 5; // Slightly increased buffer to avoid overflow

int n, m, a[N];
long long sum;
unordered_map<int, int> cnt;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i]; // Duplicate array for circular handling
    }

    long long prefix = 0;
    cnt[0] = 1; // Initialize count of mod 0 as 1

    for (int i = 1; i < 2 * n; ++i) {
        prefix += a[i];
        int mod = ((prefix % m) + m) % m; // Ensure non-negative mod

        // Subtract counts from the part that is out of window (i >= n)
        if (i >= n) {
            int old_mod = ((prefix - a[i - n + 1] * 1LL) % m + m) % m;
            cnt[old_mod]--;
            if (cnt[old_mod] == 0) cnt.erase(old_mod);
        }

        // Add current mod result to answer
        sum += cnt[mod];

        // Update count only if we are still in initial window (i < n)
        if (i < n) {
            cnt[mod]++;
        }
    }

    cout << sum << endl;
    return 0;
}
```
