#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,k,sum,cnt,a[N],b[N],d[N];
int main()
{
	ios::sync_with_stdio(0),cin.tie(0),cout.tie(0); 
	int i,j,x,y,z,t;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		cin>>a[i];
	
	sum = 0;
	memset(b, 0, sizeof(b));
	d[0] = 0;
	b[0] = 1; // Important: remainder 0 has count 1 initially
	
	// First pass: count subarrays in original array
	for(i=1; i<=n; i++) {
		d[i] = (d[i-1] + a[i]) % m;
		if(d[i] < 0) d[i] += m; // Handle negative remainders
		sum += b[d[i]];
		b[d[i]]++;
	}
	
	// Second pass: count subarrays that wrap around
	// For each starting position, we need to consider circular extensions
	memset(b, 0, sizeof(b));
	b[0] = 1;
	int prefix = 0;
	
	for(i=1; i<n; i++) { // Only up to n-1 to avoid double counting full circle
		prefix = (prefix + a[i]) % m;
		if(prefix < 0) prefix += m;
		
		// Add contribution from wrapping subarrays starting at i+1
		int suffix_sum = 0;
		for(j=i+1; j<=n; j++) {
			suffix_sum = (suffix_sum + a[j]) % m;
			if(suffix_sum < 0) suffix_sum += m;
			int target = ((-suffix_sum - prefix) % m + m) % m;
			sum += b[target];
		}
		
		b[prefix]++;
	}
	
	cout << sum;
	return 0;
}
