```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long a[2 * N];
map<long long, long long> cnt;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i + n] = a[i]; // Duplicate array for circular handling
    }

    long long total = 0;
    long long prefix = 0;

    cnt[0] = 1; // Initialize count of mod 0 as 1

    for (int i = 0; i < 2 * n; ++i) {
        prefix = (prefix + a[i]) % m;
        total += cnt[prefix];
        if (i < n) {
            cnt[prefix]++;
        }
    }

    cout << total << endl;
    return 0;
}
```
