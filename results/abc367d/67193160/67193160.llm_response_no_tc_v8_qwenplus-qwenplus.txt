```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 1e6 + 5;

int n, m;
ll a[N], prefix[2*N];
map<int, int> freq;

int main() {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    cin >> n >> m;
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];  // Duplicate array for circular handling
    }
    
    ll total = 0;
    prefix[0] = 0;
    freq[0] = 1;  // To count subarrays starting from index 0
    
    for(int i = 1; i < 2*n; ++i) {
        prefix[i] = prefix[i-1] + a[i];
        int rem = (prefix[i] % m + m) % m;  // Handle negative remainders
        
        if(i <= n) {
            total += freq[rem];  // Count valid subarrays ending at i
        }
        
        if(i < n) {
            freq[rem]++;  // Add to frequency only for first n positions
        } else {
            // Remove the contribution of the left half when sliding window
            int left_rem = (prefix[i - n] % m + m) % m;
            freq[left_rem]--;
        }
    }
    
    cout << total;
    return 0;
}
```
