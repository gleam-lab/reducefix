	map<int,int> cnt;

	for(i=1;i<=2*n;i++)
	{
		prefix += a[i];
		int mod = (prefix % m + m) % m;
		
		if(i > n)
		{
			int prev_mod = (prefix - a[i-n]*1LL*(i-n) % m - d[i-n-1] + m) % m;
			// Remove the contribution of the window that's sliding out
			cnt[(d[i-n-1] % m + m) % m]--;
		}

		ans += cnt[mod];
		cnt[mod]++;
	}

}
