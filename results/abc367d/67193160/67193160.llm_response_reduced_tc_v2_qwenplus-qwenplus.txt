```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 2e6 + 5;
int n, m, a[N];
long long sum;
int cnt[1 << 21]; // Using a large enough array for modulus m counts

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i]; // Duplicate array for circular handling
    }

    long long prefix = 0;
    cnt[0] = 1; // Count of prefix sums mod m seen so far

    for (int i = 1; i < 2 * n; ++i) {
        prefix += a[i];
        int mod = ((prefix % m) + m) % m; // Ensure non-negative mod
        sum += cnt[mod];
        cnt[mod]++;
    }

    cout << sum << "\n";
    return 0;
}
```
