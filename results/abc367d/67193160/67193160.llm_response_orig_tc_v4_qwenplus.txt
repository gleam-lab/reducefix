#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m,k,sum,cnt,a[N],b[N],d[N];
int main()
{
	ios::sync_with_stdio(0),cin.tie(0),cout.tie(0); 
	int i,j,x,y,z,t;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		cin>>a[i],a[i+n]=a[i];
	memset(b,0,sizeof(b));
	b[0]=1;
	long long sum = 0;
	long long prefix = 0;
	for(i=1;i<=2*n;i++)
	{
		prefix += a[i];
		int mod = prefix % m;
		sum += b[mod];
		if(i < n) {
			b[mod]++;
		} else if(i >= n) {
			// When we move past the first n elements, we need to remove the contribution
			// of the element that is now out of the sliding window of size n
			int prev_mod = (prefix - a[i-n]) % m;
			if(prev_mod < 0) prev_mod += m;
			b[prev_mod]--;
		}
	}
	cout<<sum;
	return 0;
}
