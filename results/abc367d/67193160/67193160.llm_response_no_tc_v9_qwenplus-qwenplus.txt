```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long sum;
int a[2*N], d[2*N], cnt_map[1000000];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];  // Duplicate array to handle circular property
    }

    // Initialize prefix sum count map with mod 0 appearing once
    cnt_map[0] = 1;

    for (int i = 1; i < 2 * n; ++i) {
        d[i] = d[i - 1] + a[i - 1];
        int mod = ((d[i] % m) + m) % m;  // Handle negative mod

        // If this mod has appeared before, add the count
        sum += cnt_map[mod];

        // Don't increment counts from the second half of the array
        if (i < n)
            cnt_map[mod]++;
    }

    cout << sum << "\n";
    return 0;
}
```
