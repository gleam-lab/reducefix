```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long a[N * 2];
unordered_map<int, int> cnt;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    long long res = 0;
    long long prefix_sum = 0;
    cnt[0] = 1;

    for (int i = 1; i < 2 * n; ++i) {
        prefix_sum += a[i];
        int mod = ((prefix_sum % m) + m) % m;
        res += cnt[mod];
        if (i < n)
            cnt[mod]++;
    }

    cout << res << endl;
    return 0;
}
```
