```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long a[N * 2];
map<long long, long long> cnt;
long long sum = 0;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    long long prefix = 0;
    cnt[0] = 1;

    for (int i = 1; i < 2 * n; ++i) {
        prefix = (prefix + a[i]) % m;

        // Adjust for circular nature: remove counts outside the first n elements' window
        if (i > n) {
            long long old_prefix = (a[i - n] + a[i - n + 1]) % m;
            for (int j = 0; j < n; ++j) {
                old_prefix = (old_prefix + a[i - n + 1 + j]) % m;
                cnt[old_prefix]--;
                if (cnt[old_prefix] == 0)
                    cnt.erase(old_prefix);
            }
        }

        sum += cnt[prefix];
        cnt[prefix]++;
    }

    cout << sum << endl;
    return 0;
}
```
