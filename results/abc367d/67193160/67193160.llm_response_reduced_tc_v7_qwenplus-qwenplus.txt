```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;
int n, m;
long long a[N * 2];
map<long long, long long> cnt;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i]; // duplicate the array for circular handling
    }

    // Prefix sums
    for (int i = 1; i < 2 * n; ++i) {
        a[i] += a[i - 1];
    }

    long long result = 0;
    cnt[0] = 1;

    for (int i = 0; i < 2 * n; ++i) {
        long long mod = a[i] % m;
        result += cnt[mod];
        if (i < n) {
            cnt[mod]++;
        } else {
            cnt[a[i - n] % m]--;
        }
    }

    cout << result << '\n';
    return 0;
}
```
