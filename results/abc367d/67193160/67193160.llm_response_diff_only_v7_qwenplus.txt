```cpp
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int n,m;
long long sum;
int a[N],d[N],b[2*N];

int main()
{
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    
    sum=0;
    memset(b,0,sizeof(b));
    b[0]=1;
    d[0]=0;
    
    for(int i=1;i<=n;i++)
    {
        d[i]=d[i-1]+a[i];
        int mod=(d[i]%m+m)%m;
        sum+=b[mod];
        b[mod]++;
    }
    
    cout<<sum;
    return 0;
}
```
