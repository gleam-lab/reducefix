#include<bits/stdc++.h>
using namespace std;
const int N = 1e6+5;
int n, m, k, sum, cnt, a[N], b[N], d[N];

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0); 
    cin >> n >> m;

    // Padding the array a to handle the wrap-around properly
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];
    }

    // Initialize the frequency array b
    memset(b, 0, sizeof(b));
    b[0] = 1;

    // Compute prefix sums and update frequencies
    for (int i = 1; i < n * 2; ++i) {
        if (i >= n) {
            b[d[i - n] % m]--;
        }
        d[i] = d[i - 1] + a[i];
        sum += b[d[i] % m];
        if (i < n) {
            b[d[i] % m]++;
        }
    }

    cout << sum;
    return 0;
}
