```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 1e6 + 5;

int n, m;
ll sum = 0;
int a[2*N];
map<int, int> cnt;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);

    cin >> n >> m;
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
        a[n+i] = a[i];  // Extend array for circular handling
    }

    // Prefix sums mod m
    ll curr = 0;
    cnt[0] = 1;  // Handle case where subarray % m == 0

    for(int i = 1; i < 2*n; i++) {
        curr = (curr + a[i]) % m;
        
        // Count previous occurrences of this mod value
        sum += cnt[curr];
        
        // If we've passed the first n elements, start removing old counts
        if(i >= n) {
            int out_of_window = (cnt[a[i-n+1]] > 0);
            cnt[a[i-n+1]]--;
        }
        
        cnt[curr]++;
    }

    cout << sum << "\n";
    return 0;
}
```
