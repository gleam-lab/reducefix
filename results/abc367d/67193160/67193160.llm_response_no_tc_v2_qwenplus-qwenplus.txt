```cpp
#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 1e6 + 5;

int n, m;
ll sum;
int a[2*N], cnt[N];
int mod;

// Custom modulo to handle negative numbers
int mod_op(int x, int m) {
    int result = x % m;
    if (result < 0) result += m;
    return result;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> m;
    mod = m;

    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i + n] = a[i];  // Duplicate array for circular property
    }

    sum = 0;
    cnt[0] = 1;  // Prefix sum of 0 occurs once

    ll prefix_sum = 0;
    for(int i = 1; i < 2 * n; ++i) {
        prefix_sum += a[i];

        int rem = mod_op(prefix_sum, m);

        // Count how many times this remainder has occurred before
        sum += cnt[rem];

        // If we're still within the original array length, update count
        if(i < n)
            cnt[rem]++;
    }

    cout << sum;
    return 0;
}
```
