```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // Count occurrences of each remainder when divided by m
    vector<int> remainder_counts(m, 0);
    
    // Start with prefix sum 0 (empty subarray)
    remainder_counts[0] = 1;
    
    int prefix_sum = 0;
    int result = 0;
    
    for(int i = 0; i < n; i++){
        prefix_sum += nums[i];
        // Keep prefix_sum modulo m to avoid large numbers
        int rem = ((prefix_sum % m) + m) % m;  // Ensure non-negative
        
        // If we've seen this remainder before, we found valid subarrays
        result += remainder_counts[rem];
        
        // Count current remainder
        remainder_counts[rem]++;
    }
    
    cout << result << endl;
    return 0;
}
```
