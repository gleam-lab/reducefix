```cpp
#include<bits/stdc++.h>
using namespace std;

int n, m;
vector<int> nums;
vector<int> prefixSum;

void precomputePrefixSum() {
    prefixSum.push_back(0); // Initialize with 0
    for (int i = 0; i < n; ++i) {
        prefixSum.push_back(prefixSum.back() + nums[i]);
    }
}

int findValidPairs() {
    int count = 0;
    for (int s = 0; s < n; ++s) {
        for (int t = s + 1; t < n; ++t) {
            int step;
            if (s <= t) {
                step = prefixSum[t + 1] - prefixSum[s];
            } else {
                step = prefixSum[n] - prefixSum[s] + prefixSum[t + 1];
            }
            if (step % m == 0) {
                count++;
            }
        }
    }
    return count;
}

int main() {
    cin >> n >> m;
    nums.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }

    precomputePrefixSum();
    int result = findValidPairs();

    cout << result << endl;
    return 0;
}
```

### Explanation of Fixes:
1. **Initialization**: The `prefixSum` vector is initialized with an extra zero at the beginning to simplify calculations.
2. **Loop Boundaries**: In the nested loops, the inner loop starts from `s + 1` instead of `1`, ensuring that each pair `(s, t)` is considered only once and avoiding redundant checks where `s == t`.
3. **Function Structure**: The logic has been broken down into functions (`precomputePrefixSum` and `findValidPairs`) for better readability and maintainability.
4. **Efficiency**: The original approach had a time complexity of \(O(n^3)\), which could be optimized further depending on specific constraints, but this change ensures clarity and correctness first.
