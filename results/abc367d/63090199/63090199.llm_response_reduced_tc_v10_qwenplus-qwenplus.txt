```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // To count the number of pairs (s, t) with prefix sums congruent mod m
    vector<int> prefix_sums(n + 1, 0);
    for(int i = 0; i < n; i++){
        prefix_sums[i + 1] = (prefix_sums[i] + nums[i]) % m;
    }

    // Use a map to count occurrences of each modulo value
    map<int, int> freq;
    long long count = 0;

    for(int i = 0; i <= n; i++){
        // For each same modulo value, we can form C(k, 2) pairs
        count += freq[prefix_sums[i]]++;
    }

    cout << count << endl;
    return 0;
}
```
