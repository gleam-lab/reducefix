```cpp
#include<bits/stdc++.h>
using namespace std;
int n,m;
int main(){
    cin>>n>>m;
    vector<int>nums(n);
    for(int i=0;i<n;i++){
        cin>>nums[i];
    }
    vector<int>prefixsum(n+1,0);
    for(int i=0;i<n;i++){
        prefixsum[i+1]=prefixsum[i]+nums[i];
    }
    long long count=0;
    // 使用map记录每个前缀和模m的余数出现的次数
    map<int,int> modCount;
    // 初始化：前缀和为0的情况（空前缀）
    modCount[0] = 1;

    // 遍历所有位置作为右端点t
    for(int t = 1; t <= n; t++) {
        int r = prefixsum[t] % m;
        // 如果之前有相同余数的前缀，说明存在子数组和能被m整除
        if(modCount.find(r) != modCount.end()) {
            count += modCount[r];
        }
        modCount[r]++;
    }

    // 减去单个点s=t的情况（题目中s != t）
    // 总共有n个这样的非法情况
    count -= n;

    cout << count << endl;
    return 0;
}
```
