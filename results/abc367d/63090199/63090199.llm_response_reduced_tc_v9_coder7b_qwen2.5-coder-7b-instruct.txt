```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++) {
        cin >> nums[i];
    }

    vector<int> prefixSum(n + 1, 0);
    for(int i = 0; i < n; i++) { // Calculate prefix sum
        prefixSum[i + 1] = prefixSum[i] + nums[i];
    }

    int count = 0;
    unordered_map<int, int> freq;
    for(int i = 0; i <= n; i++) {
        int remainder = prefixSum[i] % m;
        count += freq[remainder];
        freq[remainder]++;
    }

    cout << count << endl;
    return 0;
}
```
