```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }

    // Use a map to count the frequency of each prefix sum modulo m
    unordered_map<int, int> mod_count;
    mod_count[0] = 1;  // To handle subarrays starting from index 0

    long long count = 0;
    long long prefix_sum = 0;

    for (int i = 0; i < n; ++i) {
        prefix_sum += nums[i];
        int mod = ((prefix_sum % m) + m) % m;  // Ensure non-negative modulo

        // If this mod has been seen before, add its count to result
        count += mod_count[mod];

        // Increment the count of this mod value
        mod_count[mod]++;
    }

    cout << count << endl;
    return 0;
}
```
