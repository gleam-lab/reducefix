#include<bits/stdc++.h>
using namespace std;
int n, m;
int main() {
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) {
        cin >> nums[i];
    }
    vector<int> prefixsum(n+1, 0);
    for (int i = 0; i < n; i++) {
        prefixsum[i+1] = prefixsum[i] + nums[i];
    }
    
    // Use a map to store the frequency of remainders
    map<int, int> remainder_count;
    // Initialize the remainder count for 0 remainder
    remainder_count[0] = 1;
    
    int count = 0;
    // Calculate the remainder for each possible range and update the count
    for (int s = 1; s <= n; s++) {
        for (int t = 1; t <= n; t++) {
            if (s == t) continue;
            int step = (s < t) ? (prefixsum[t] - prefixsum[s]) : (prefixsum[n] - prefixsum[s] + prefixsum[t]);
            int remainder = step % m;
            count += remainder_count[remainder]; // Increment count by the number of times this remainder has been seen
            remainder_count[remainder]++; // Increment the count of this remainder
        }
    }
    
    cout << count << endl;
    return 0;
}
