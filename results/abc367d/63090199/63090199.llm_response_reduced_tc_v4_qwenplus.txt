```cpp
#include<bits/stdc++.h>
using namespace std;
int n,m;
int main(){
    cin>>n>>m;
    vector<int>nums(n);
    for(int i=0;i<n;i++){
        cin>>nums[i];
    }
    vector<int>prefixsum(n+1,0);
    for(int i=0;i<n;i++){
        prefixsum[i+1]=prefixsum[i]+nums[i];
    }
    long long count=0;
    map<int,int> remainderCount;
    for(int i=1;i<=n;i++){
        int rem = prefixsum[i] % m;
        if(rem == 0){
            count++;
        }
        count += remainderCount[rem];
        remainderCount[rem]++;
    }
    cout<<count<<endl;
    return 0;
}
```
