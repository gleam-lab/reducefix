```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // Prefix sum with modulo
    vector<int> prefixSumMod(n + 1, 0);
    for(int i = 0; i < n; i++){
        prefixSumMod[i+1] = (prefixSumMod[i] + nums[i]) % m;
    }
    
    // Count frequency of each remainder
    map<int, int> modCount;
    modCount[0] = 1;  // Initial value for prefixSumMod[0]
    
    long long count = 0;
    for(int i = 1; i <= n; i++){
        // We want to count how many previous indices have the same mod value
        count += modCount[prefixSumMod[i]];
        modCount[prefixSumMod[i]]++;
    }
    
    cout << count << endl;
    return 0;
}
```
