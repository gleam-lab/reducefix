#include<bits/stdc++.h>
using namespace std;
int n, m;
int main() {
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }
    vector<int> prefixsum(n + 1, 0);
    for (int i = 0; i < n; ++i) {
        prefixsum[i + 1] = prefixsum[i] + nums[i];
    }
    int count = 0;
    // Using a map to store frequency of remainders
    unordered_map<int, int> remainder_count;
    for (int i = 0; i <= n; ++i) {
        int remainder = (prefixsum[i] % m + m) % m; // Ensure non-negative remainder
        remainder_count[remainder]++;
    }
    // Calculate count of valid (s, t) pairs
    for (const auto& kv : remainder_count) {
        int remainder = kv.first;
        int frequency = kv.second;
        if (remainder == 0) {
            count += frequency * (frequency - 1) / 2;
        } else {
            int mod_count = remainder_count.count((m - remainder) % m);
            count += frequency * mod_count;
        }
    }
    // Subtract pairs (s, t) where s == t
    for (const auto& kv : remainder_count) {
        count -= kv.second;
    }
    cout << count << endl;
    return 0;
}
