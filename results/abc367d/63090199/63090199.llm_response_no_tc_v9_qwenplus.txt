```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // Compute prefix sum
    vector<int> prefixsum(n + 1, 0);
    for(int i = 0; i < n; i++){
        prefixsum[i + 1] = (prefixsum[i] + nums[i]) % m;
    }
    
    long long count = 0;
    // Use map to count frequency of each prefix sum mod m
    map<int, int> modCount;
    for(int i = 0; i <= n; i++){
        int mod = prefixsum[i] % m;
        count += modCount[mod];
        modCount[mod]++;
    }
    
    cout << count << endl;
    return 0;
}
```
