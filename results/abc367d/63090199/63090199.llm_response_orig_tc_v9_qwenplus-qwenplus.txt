#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }

    // Use a map to count remainders of prefix sums modulo m
    unordered_map<int, int> remainder_count;
    long long prefix_sum = 0;
    remainder_count[0] = 1;  // To handle subarrays starting from index 0

    long long total_pairs = 0;

    for (int i = 0; i < n; ++i) {
        prefix_sum += nums[i];
        int rem = ((prefix_sum % m) + m) % m;  // Ensure non-negative remainder

        // If this remainder has been seen before, there are previous indices to pair with
        total_pairs += remainder_count[rem];

        // Count the current remainder
        remainder_count[rem]++;
    }

    // Subtract cases where s == t (which is invalid as per problem statement)
    // There are n such invalid cases (each index with itself)
    cout << total_pairs - n << endl;

    return 0;
}
