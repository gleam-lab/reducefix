```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // We can avoid explicitly storing the entire prefix sum array
    // and instead use a map to count frequencies of remainders
    unordered_map<int, int> remainder_count;
    long long current_sum = 0;
    
    // Initialize with remainder 0 having one count (empty prefix)
    remainder_count[0] = 1;
    
    long long result = 0;
    
    for(int i = 0; i < n; i++){
        current_sum += nums[i];
        int rem = ((current_sum % m) + m) % m;  // Ensure non-negative remainder
        
        // If this remainder has been seen before, then there are subarrays
        // ending at current index with sum divisible by m
        result += remainder_count[rem];
        
        // Increment the count of this remainder
        remainder_count[rem]++;
    }
    
    cout << result << endl;
    return 0;
}
```
