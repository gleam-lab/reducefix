```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }

    // Prefix sum modulo m
    vector<int> prefix_mod(n + 1, 0);
    for (int i = 0; i < n; ++i) {
        prefix_mod[i + 1] = (prefix_mod[i] + nums[i]) % m;
    }

    // To count occurrences of each mod value
    map<int, int> mod_count;
    mod_count[0] = 1; // Initialize with 0 to handle subarrays starting from index 0

    long long result = 0;

    for (int i = 1; i <= n; ++i) {
        // For current prefix mod value, all previous prefixes with same mod contribute
        result += mod_count[prefix_mod[i]];
        mod_count[prefix_mod[i]]++;
    }

    cout << result << endl;
    return 0;
}
```
