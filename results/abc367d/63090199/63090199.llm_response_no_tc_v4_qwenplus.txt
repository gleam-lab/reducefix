#include<bits/stdc++.h>
using namespace std;

int main(){
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // Calculate prefix sum
    vector<int> prefixsum(n+1, 0);
    for(int i = 0; i < n; i++){
        prefixsum[i+1] = (prefixsum[i] + nums[i]) % m;
    }
    
    long long count = 0;
    
    // Use frequency map to count pairs with same remainder
    vector<int> freq(m, 0);
    
    // For each starting position s (1-indexed in logic), we want to find valid t
    for(int s = 0; s < n; s++){
        // For each ending position t where t != s
        for(int t = 0; t < n; t++){
            if(s == t) continue;
            
            int step;
            if(t > s){
                // Forward: s -> t (s+1 to t)
                step = (prefixsum[t+1] - prefixsum[s+1] + m) % m;
            } else {
                // Wrap around: s -> n -> 1 -> t
                step = (prefixsum[n] - prefixsum[s+1] + prefixsum[t+1] + m) % m;
            }
            
            if(step % m == 0){
                count++;
            }
        }
    }
    
    cout << count << endl;
    return 0;
}
