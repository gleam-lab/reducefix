```cpp
#include<bits/stdc++.h>
using namespace std;
int n,m;
int main(){
    cin>>n>>m;
    vector<int>nums(n);
    for(int i=0;i<n;i++){
        cin>>nums[i];
    }
    vector<int>prefixsum(n+1,0);
    for(int i=0;i<n;i++){
        prefixsum[i+1]=prefixsum[i]+nums[i];
    }
    long long count=0;
    map<int,int> remainder_count;
    
    // Count frequency of each remainder in prefix sums (mod m)
    for(int i=0;i<=n;i++){
        int r = prefixsum[i] % m;
        if(r < 0) r += m; // Handle negative remainders
        remainder_count[r]++;
    }
    
    // For each remainder, calculate number of pairs
    for(auto& [r, cnt] : remainder_count){
        if(cnt >= 2){
            count += 1LL * cnt * (cnt - 1);
        }
    }
    
    cout<<count<<endl;
    return 0;
}
```
