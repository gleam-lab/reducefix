#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }
    
    if (n == 1) {
        cout << 0 << endl;
        return 0;
    }
    
    vector<int> prefix_sum(n + 1, 0);
    for (int i = 0; i < n; ++i) {
        prefix_sum[i + 1] = (prefix_sum[i] + nums[i]) % m;
    }
    
    int total_sum = prefix_sum[n];
    unordered_map<int, int> mod_count;
    long long count = 0;
    
    for (int i = 1; i <= n; ++i) {
        int current_mod = prefix_sum[i];
        int target_mod = (current_mod - total_sum + m) % m;
        count += mod_count[target_mod];
        mod_count[current_mod]++;
    }
    
    cout << count << endl;
    return 0;
}

1. **Prefix Sum with Modulo**: We compute the prefix sums modulo `m` to simplify checking divisibility.
2. **Modular Arithmetic**: For each position `i`, the sum from `s` to `t` (wrapping around if needed) can be expressed in terms of the prefix sums and the total sum.
3. **Hash Map for Counting**: We use a hash map to count occurrences of each modulo value, which allows us to efficiently find pairs `(s, t)` where the sum is divisible by `m`.
4. **Edge Case Handling**: The case where `n = 1` is handled separately since no valid pairs exist.
