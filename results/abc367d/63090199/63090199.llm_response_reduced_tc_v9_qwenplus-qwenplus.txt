```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    vector<int> nums(n);
    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }
    
    // To count the number of pairs (i, j) such that prefixSum[i] ≡ prefixSum[j] mod m
    // This would imply sum(nums[j...i-1]) is divisible by m
    
    long long result = 0;
    vector<int> freq(m, 0);
    vector<long long> prefixSum(n + 1, 0);
    
    for(int i = 0; i < n; ++i){
        prefixSum[i + 1] = (prefixSum[i] + nums[i]) % m;
        if(prefixSum[i + 1] < 0) prefixSum[i + 1] += m; // Ensure non-negative
    }
    
    // Count frequency of each modulo value
    for(int i = 0; i <= n; ++i){
        result += freq[prefixSum[i]]++;
    }
    
    cout << result << endl;
    return 0;
}
```
