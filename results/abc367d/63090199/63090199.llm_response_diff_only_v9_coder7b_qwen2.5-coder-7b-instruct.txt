```cpp
#include<bits/stdc++.h>
using namespace std;

int n, m;
int main() {
    cin >> n >> m;
    vector<int> nums(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> nums[i];
    }

    vector<int> prefixSum(n + 1, 0);
    for (int i = 1; i <= n; ++i) { // 求前缀和
        prefixSum[i] = prefixSum[i - 1] + nums[i];
    }

    int count = 0;
    // 寻找有可能的(s, t)对
    for (int s = 1; s <= n; ++s) {
        for (int t = s + 1; t <= n; ++t) {
            int step = prefixSum[t] - prefixSum[s - 1];
            if (step % m == 0) {
                count++;
            }
        }
    }

    cout << count << endl;
    return 0;
}
```
