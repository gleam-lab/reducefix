
```cpp
#include<bits/stdc++.h>
using namespace std;
int n, m;
int main() {
    cin >> n >> m;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) {
        cin >> nums[i];
    }
    vector<int> prefixsum(n + 1, 0);
    for (int i = 0; i < n; i++) {
        prefixsum[i + 1] = prefixsum[i] + nums[i];
    }
    int count = 0;
    // Optimized approach to find pairs (s, t) such that (t - s) % m == 0
    for (int s = 1; s <= n; s++) {
        for (int t = s + 1; t <= n; t++) { // t starts from s+1 to avoid duplicates
            int step = prefixsum[t] - prefixsum[s];
            if (step % m == 0) {
                count++;
            }
        }
    }
    cout << count << endl;
    return 0;
}
```
