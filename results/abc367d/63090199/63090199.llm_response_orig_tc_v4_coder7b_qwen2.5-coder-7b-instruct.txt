```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int n, m;
    cin >> n >> m;

    vector<int> nums(n + 1);
    for(int i = 1; i <= n; ++i) {
        cin >> nums[i];
    }

    vector<long long> prefixSum(n + 1);
    for(int i = 1; i <= n; ++i) {
        prefixSum[i] = prefixSum[i - 1] + nums[i];
    }

    long long count = 0;
    for(int s = 1; s <= n; ++s) {
        for(int t = s + 1; t <= n; ++t) {
            long long step = prefixSum[t] - prefixSum[s];
            if((step % m == 0)) {
                count += n - t + 1;
            }
        }
    }

    cout << count << endl;
    return 0;
}
```
