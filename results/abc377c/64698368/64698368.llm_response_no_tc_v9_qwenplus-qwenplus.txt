#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = n-1; i >= (int)(0); i--)
#define rrepi(i, a, b) for (int i = (int)(a); i > (int)(b); i--)
#define ALL(a) (a).begin(),(a).end()
#define ALL_(a) (a),(a)+size(a)
#define IF_T if(Is_test)

#define ll long long
using namespace std;

// Custom hash for pair<int, int>
struct phash {
    inline size_t operator()(const pair<ll, ll> &p) const {
        // Use a simple combination of hash values
        return hash<ll>()(p.first * 1000000007 + p.second);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N;
    int M;
    cin >> N >> M;

    unordered_set<pair<ll, ll>, phash> attacked_cells;

    // The 8 possible positions that can be attacked by a knight-like move
    vector<pair<ll, ll>> attack_offsets = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // Read each piece's position and mark the attacked cells
    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        // Convert to 0-based index
        a--; b--;

        // Mark current cell as occupied
        attacked_cells.insert({a, b});

        // Mark all reachable cells from this piece
        for (auto [dx, dy] : attack_offsets) {
            ll x = a + dx;
            ll y = b + dy;
            if (x >= 0 && x < N && y >= 0 && y < N) {
                attacked_cells.insert({x, y});
            }
        }
    }

    // Total number of empty cells not under attack
    // Total cells: N*N, occupied/attacked: attacked_cells.size()
    cout << N * N - static_cast<ll>(attacked_cells.size()) << endl;

    return 0;
}
