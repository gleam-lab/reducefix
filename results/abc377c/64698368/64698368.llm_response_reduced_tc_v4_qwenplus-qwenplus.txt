#include <bits/stdc++.h>
using namespace std;

using ll = long long;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

// Hash struct for pair<int, int>
struct phash {
    inline size_t operator()(const pair<int, int>& p) const {
        const auto h1 = hash<int>()(p.first);
        const auto h2 = hash<int>()(p.second);
        return h1 ^ (h2 << 1);
    }
};

int main() {
    ll N;
    int M;
    cin >> N >> M;

    // Directions from which a piece can be attacked
    vector<pair<ll, ll>> attack_offsets = {
        {0, 0},
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    unordered_set<pair<ll, ll>, phash> attacked_cells;

    rep(i, M) {
        ll a, b;
        cin >> a >> b;
        // Convert to 0-based indexing
        a--;
        b--;

        for (auto offset : attack_offsets) {
            ll na = a + offset.first;
            ll nb = b + offset.second;
            if (na >= 0 && na < N && nb >= 0 && nb < N) {
                attacked_cells.insert({na, nb});
            }
        }
    }

    // Total number of cells is N * N
    ll total_cells = N * N;
    // Subtract the number of attacked cells
    ll result = total_cells - (ll)attacked_cells.size();

    cout << result << endl;

    return 0;
}
