using namespace std;

const int dir[8][2] = {{0, 2}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}};

int N, M;

void insert_capture_area(int x, int y) {
    if (0 <= x && x < N && 0 <= y && y < N) {
        for (auto& d : dir) {
            int nx = x + d[0], ny = y + d[1];
            if (0 <= nx && nx < N && 0 <= ny && ny < N) {
                st.insert({nx, ny});
            }
        }
    }
}

int main() {
    cin >> N >> M;
    N *= N;

    set<pair<int, int>> st;
    rep(i, M) {
        int x, y;
        cin >> x >> y;
        insert_capture_area(x - 1, y - 1);
    }

    // Add the coordinates of all cells to the set for simplicity
    for (int i = 0; i < N; ++i) {
        st.insert({i / N, i % N});
    }

    // Subtract the size of the capture area from the total number of cells
    cout << N - st.size();
    return 0;
}

It corrects the logic of checking the capture area by explicitly iterating through the possible directions and updating the set of captured cells. It also ensures that all cells are initially added to the set and then the size of the captured area is subtracted from the total number of cells to get the answer.
