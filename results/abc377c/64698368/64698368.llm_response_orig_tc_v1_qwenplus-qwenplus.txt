#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

#define ll long long
using namespace std;

// Use a hash-based container for fast lookups and insertions
struct phash {
    inline size_t operator()(const pair<int,int> &p) const {
        const auto h1 = hash<ll>()(p.first);
        const auto h2 = hash<ll>()((ll)p.second << 32);
        return h1 ^ h2;
    }
};

int main() {
    cin.tie(0);
    ios::sync_with_stdio(false);

    ll N;
    int M;
    cin >> N >> M;

    unordered_set<pair<int, int>, phash> attacked;
    vector<vector<int>> placed;

    // These are all the relative positions that can attack
    vector<pair<int, int>> offsets = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    rep(i, M) {
        int x, y;
        cin >> x >> y;
        x--; y--;

        // Add current piece to attacked set since it's not empty
        attacked.insert({x, y});

        // For each possible attacker position around this piece, add to attacked
        for (auto [dx, dy] : offsets) {
            int nx = x + dx;
            int ny = y + dy;
            if (nx >= 0 && nx < N && ny >= 0 && ny < N) {
                attacked.insert({nx, ny});
            }
        }
    }

    // Total number of cells minus the number of non-empty or attacked cells
    cout << (N * NLL - (ll)attacked.size()) << endl;
}
