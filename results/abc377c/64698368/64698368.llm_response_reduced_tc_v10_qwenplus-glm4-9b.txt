#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = (n) - 1; i >= 0; i--)
#define rrepi(i, a, b) for (int i = (int)(b) - 1; i >= (int)(a); i--)

int N, M;
vector<pair<int, int>> pieces;
set<pair<int, int>> captured;

void capture(int x, int y) {
    int dx[] = {2, 2, 1, -1, -2, -2, -1, 1};
    int dy[] = {1, -1, 2, 2, 1, -1, -2, -2};
    for (int i = 0; i < 8; i++) {
        int nx = x + dx[i], ny = y + dy[i];
        if (nx >= 0 && nx < N && ny >= 0 && ny < N) {
            captured.insert({nx, ny});
        }
    }
}

int main() {
    cin >> N >> M;
    rep(i, M) {
        int a, b;
        cin >> a >> b;
        pieces.emplace_back(a - 1, b - 1);
    }

    rep(i, M) {
        capture(pieces[i].first, pieces[i].second);
    }

    long long all_squares = N * N;
    long long free_squares = all_squares;
    rep(i, M) {
        auto [a, b] = pieces[i];
        // Remove captured squares
        int x[] = {a + 1, a + 2, a - 1, a - 2, a, a};
        int y[] = {b, b + 1, b, b - 1, b + 2, b - 2};
        for (int j = 0; j < 6; j++) {
            for (int k = 0; k < 6; k++) {
                if (x[j] >= 0 && x[j] < N && y[k] >= 0 && y[k] < N) {
                    free_squares--;
                }
            }
        }
    }
    
    // Remove squares captured by pieces
    for (auto [a, b] : captured) {
        if (a >= 0 && a < N && b >= 0 && b < N) {
            free_squares--;
        }
    }

    cout << free_squares << endl;

    return 0;
}
