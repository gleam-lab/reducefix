
```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

#define mod 1000000007

int main() {
    int N, M;
    cin >> N >> M;

    set<pair<int, int>> occupied;
    vector<pair<int, int>> moves = {
        {0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        occupied.insert({a, b});
        for (auto& move : moves) {
            int na = a + move.first;
            int nb = b + move.second;
            if (na >= 1 && na <= N && nb >= 1 && nb <= N) {
                occupied.insert({na, nb});
            }
        }
    }

    long long total_cells = (long long)N * N;
    long long safe_cells = total_cells;
    for (auto& cell : occupied) {
        safe_cells -= 1;
        for (auto& move : moves) {
            int na = cell.first + move.first;
            int nb = cell.second + move.second;
            if (na >= 1 && na <= N && nb >= 1 && nb <= N) {
                safe_cells -= 1;
            }
        }
    }

    cout << safe_cells << endl;

    return 0;
}
```
