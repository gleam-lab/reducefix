```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Set to store all the positions that are either occupied by existing pieces or can be attacked
    unordered_set<long long> attacked;

    // Direction vectors for all possible attack positions
    vector<pair<int, int>> directions{
        {0, 0},
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        // For each piece, mark all reachable positions (including itself)
        for (const auto& [dx, dy] : directions) {
            long long na = a + dx;
            long long nb = b + dy;

            // Check if the position is within bounds
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                // Hash the coordinate pair into a single integer
                long long hash = na * N + nb;
                attacked.insert(hash);
            }
        }
    }

    // Total number of cells is N*N
    // Subtract the number of attacked and occupied cells
    cout << N * N - static_cast<long long>(attacked.size()) << endl;

    return 0;
}
```
