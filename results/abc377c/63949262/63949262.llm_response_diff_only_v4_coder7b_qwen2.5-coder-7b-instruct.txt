```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long N, M;
    cin >> N >> M;

    set<pair<long long, long long>> positions;
    for(long long i = 0; i < M; ++i){
        long long x, y;
        cin >> x >> y;
        positions.insert({x, y});
    }

    long long count = 0;
    for(long long i = 1; i <= N; ++i){
        for(long long j = 1; j <= N; ++j){
            bool valid = true;
            vector<pair<long long, long long>> moves = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
            for(auto& move : moves){
                long long nx = i + move.first, ny = j + move.second;
                if(nx >= 1 && nx <= N && ny >= 1 && ny <= N){
                    if(positions.find({nx, ny}) != positions.end()){
                        valid = false;
                        break;
                    }
                }
            }
            if(valid){
                count++;
            }
        }
    }

    cout << count << endl;
    return 0;
}
```
