#include <iostream>
#include <vector>
#include <set>
#include <unordered_set>
#include <utility>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N;
    int M;
    cin >> N >> M;

    // Use a hash-based set for faster lookups
    unordered_set<int64_t> occupied; 
    const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;

        // Mark the piece's position
        occupied.insert(a * N + b); 

        // Mark all positions that this piece can attack
        for (int d = 0; d < 8; ++d) {
            int64_t na = a + dx[d];
            int64_t nb = b + dy[d];
            if (na >= 1 && na <= N && nb >= 1 && nb <= N) {
                occupied.insert(na * N + nb);
            }
        }
    }

    // Total number of cells minus the number of forbidden cells
    cout << (int64_t(N) * N - int64_t(occupied.size())) << "\n";

    return 0;
}
