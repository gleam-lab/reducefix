```cpp
#include <iostream>
#include <vector>
#include <set>
#include <utility>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N;
    int M;
    cin >> N >> M;

    set<pair<int64_t, int64_t>> forbidden;
    vector<pair<int64_t, int64_t>> pieces(M);

    // Directions where a piece can attack
    const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

    for (int i = 0; i < M; ++i) {
        int64_t x, y;
        cin >> x >> y;
        pieces[i] = {x, y};
        forbidden.insert({x, y});

        for (int d = 0; d < 8; ++d) {
            int64_t nx = x + dx[d];
            int64_t ny = y + dy[d];
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                forbidden.insert({nx, ny});
            }
        }
    }

    // Total number of cells that cannot be used
    int64_t total_forbidden = forbidden.size();
    // Total number of cells on the board
    int64_t total_cells = N * N;
    // Result is all cells minus occupied and attacked cells
    cout << (total_cells - total_forbidden) << "\n";

    return 0;
}
```
