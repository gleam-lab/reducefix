#include <iostream>
#include <unordered_set>
#include <vector>

using namespace std;

// A simple hash for pairs to use in unordered_set
struct pair_hash {
    template <class T1, class T2>
    size_t operator()(const pair<T1, T2>& p) const {
        auto h1 = hash<T1>{}(p.first);
        auto h2 = hash<T2>{}(p.second);
        return h1 ^ (h2 << 16);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // All positions where a piece is already placed or can be attacked
    unordered_set<pair<long long, long long>, pair_hash> occupied;

    vector<pair<long long, long long>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        long long x, y;
        cin >> x >> y;

        // Add the position itself
        occupied.insert({x, y});

        // Add all 8 positions this piece can attack
        for (auto [dx, dy] : knight_moves) {
            long long nx = x + dx;
            long long ny = y + dy;
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                occupied.insert({nx, ny});
            }
        }
    }

    // Total number of squares minus the number of occupied squares
    cout << (N * N - static_cast<long long>(occupied.size())) << "\n";

    return 0;
}
