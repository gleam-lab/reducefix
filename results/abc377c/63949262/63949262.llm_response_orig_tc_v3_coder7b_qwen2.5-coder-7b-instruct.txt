```cpp
#include <iostream>
#include <vector>
#include <unordered_set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int n, m;
    cin >> n >> m;

    unordered_set<int> forbidden;

    for(int i=0; i<m; ++i){
        int x, y;
        cin >> x >> y;
        --x; --y;
        forbidden.insert(x*n + y);

        for(int dx=-2; dx<=2; ++dx){
            for(int dy=-2; dy<=2; ++dy){
                if(dx==0 && dy==0 || abs(dx)!=abs(dy)) continue;
                int nx=x+dx, ny=y+dy;
                if(nx>=0 && nx<n && ny>=0 && ny<n){
                    forbidden.insert(nx*n + ny);
                }
            }
        }
    }

    cout << n*n - forbidden.size() << endl;

    return 0;
}
```
