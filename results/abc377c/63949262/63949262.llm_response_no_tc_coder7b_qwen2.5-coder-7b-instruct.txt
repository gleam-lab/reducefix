```cpp
#include <iostream>
#include <set>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    cin >> N;

    int M;
    cin >> M;

    set<pair<int, int>> forbidden;
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;

        forbidden.insert({a, b});

        vector<pair<int, int>> moves = {
            {a + 2, b + 1}, {a + 1, b + 2},
            {a - 1, b + 2}, {a - 2, b + 1},
            {a - 2, b - 1}, {a - 1, b - 2},
            {a + 1, b - 2}, {a + 2, b - 1}
        };

        for (const auto& mv : moves) {
            int x = mv.first, y = mv.second;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                forbidden.insert({x, y});
            }
        }
    }

    long long total = N * N;
    cout << (total - forbidden.size()) << endl;
}
```
