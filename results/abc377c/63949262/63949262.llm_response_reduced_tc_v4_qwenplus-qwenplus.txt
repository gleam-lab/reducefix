#include <iostream>
#include <vector>
#include <set>
#include <tuple>
#include <unordered_set>

using namespace std;

using i64 = long long;

struct PairHash {
    size_t operator()(const pair<i64, i64>& p) const {
        return hash<i64>()(p.first) ^ (hash<i64>()(p.second) << 32);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N;
    int M;
    cin >> N >> M;

    unordered_set<pair<i64, i64>, PairHash> attacked_cells;

    vector<pair<i64, i64>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int _ = 0; _ < M; ++_) {
        i64 x, y;
        cin >> x >> y;
        
        // Add current piece to attacked cells
        attacked_cells.insert({x, y});
        
        // Add all positions that this piece can attack
        for (auto [dx, dy] : knight_moves) {
            i64 nx = x + dx;
            i64 ny = y + dy;
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                attacked_cells.insert({nx, ny});
            }
        }
    }

    // Total number of cells minus the number of attacked cells
    cout << (N * N - static_cast<i64>(attacked_cells.size())) << endl;

    return 0;
}
