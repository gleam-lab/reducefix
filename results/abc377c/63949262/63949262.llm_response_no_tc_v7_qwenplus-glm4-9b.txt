#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

#define MAX_N 1000000000

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    // Create a grid with N+2xN+2 to handle edge cases easily
    vector<vector<bool>> can_capture(N + 2, vector<bool>(N + 2, false));
    for (int i = 0; i < M; ++i) {
        int x = pieces[i].first;
        int y = pieces[i].second;
        // Check all possible captures
        can_capture[x + 2][y + 1] = true;
        can_capture[x + 1][y + 2] = true;
        can_capture[x - 1][y + 2] = true;
        can_capture[x - 2][y + 1] = true;
        can_capture[x - 2][y - 1] = true;
        can_capture[x - 1][y - 2] = true;
        can_capture[x + 1][y - 2] = true;
        can_capture[x + 2][y - 1] = true;
    }

    // Calculate the number of valid positions to place the piece
    long long valid_positions = 0;
    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= N; ++j) {
            bool valid = true;
            // Check if the current position can be captured by any piece
            for (int dx = -2; dx <= 2; ++dx) {
                for (int dy = -2; dy <= 2; ++dy) {
                    int nx = i + dx;
                    int ny = j + dy;
                    if (nx >= 1 && nx <= N && ny >= 1 && ny <= N && can_capture[nx][ny]) {
                        valid = false;
                        break;
                    }
                }
                if (!valid) break;
            }
            if (valid) {
                ++valid_positions;
            }
        }
    }

    // Output the result
    cout << valid_positions << endl;

    return 0;
}

This code initializes a grid of size \(N+2 \times N+2\) to easily handle edge cases. It then checks each piece's capture range and marks the grid accordingly. Finally, it iterates over the grid to count the number of valid positions where a new piece can be placed without being captured by any existing piece.
