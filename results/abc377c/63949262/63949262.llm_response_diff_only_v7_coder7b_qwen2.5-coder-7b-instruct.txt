#include <bits/stdc++.h>

using namespace std;

#define ll long long

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    int n, m;
    cin >> n >> m;
    
    unordered_set<ll> pieces;
    
    auto encode = [&](int x, int y) -> ll { return ((ll)x << 32) | y; };
    auto decode_x = [&](ll p) -> int { return p >> 32; };
    auto decode_y = [&](ll p) -> int { return p & 0xFFFFFFFF; };
    
    for(int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        
        ll pos = encode(x, y);
        pieces.insert(pos);
        
        for(int dx = -2; dx <= 2; ++dx) {
            for(int dy = -2; dy <= 2; ++dy) {
                if(abs(dx) + abs(dy) == 3 && dx != 0 && dy != 0) {
                    int nx = x + dx, ny = y + dy;
                    if(nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                        pieces.insert(encode(nx, ny));
                    }
                }
            }
        }
    }
    
    ll total_positions = 1LL * n * n;
    ll occupied_positions = pieces.size();
    
    cout << total_positions - occupied_positions << endl;
    
    return 0;
}
