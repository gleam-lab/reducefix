#include <iostream>
#include <unordered_set>
#include <vector>
#include <tuple>

using namespace std;

// Hash function for pairs (used with unordered_set)
struct pair_hash {
    template <class T1, class T2>
    size_t operator()(const pair<T1, T2>& p) const {
        auto h1 = hash<T1>{}(p.first);
        auto h2 = hash<T2>{}(p.second);
        return h1 ^ (h2 << 16);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    unordered_set<pair<long long, long long>, pair_hash> attacked_positions;
    vector<pair<long long, long long>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // Store all occupied positions to avoid placing on them
    unordered_set<pair<long long, long long>, pair_hash> occupied;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        occupied.insert({a, b});

        // Add all positions that this piece can attack
        for (auto [dx, dy] : knight_moves) {
            long long x = a + dx;
            long long y = b + dy;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                attacked_positions.insert({x, y});
            }
        }
    }

    // Remove positions that are already occupied from the attacked set
    for (auto pos : occupied) {
        attacked_positions.erase(pos);
    }

    // Total valid positions: total grid - occupied - attacked
    long long total_positions = N * N;
    long long result = total_positions - occupied.size() - attacked_positions.size();

    cout << result << "\n";

    return 0;
}
