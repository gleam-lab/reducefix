#include <iostream>
#include <unordered_set>
#include <vector>

using namespace std;

// Custom hash to allow pair as key in unordered_set
struct pair_hash {
    template <class T1, class T2>
    size_t operator()(const pair<T1, T2>& p) const {
        auto h1 = hash<T1>{}(p.first);
        auto h2 = hash<T2>{}(p.second);
        return h1 ^ (h2 << 1); // or some other combination
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    long long M;
    cin >> N >> M;

    unordered_set<pair<long long, long long>, pair_hash> attacked;
    vector<pair<long long, long long>> pieces(M);
    
    // Knight move directions
    const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

    for (long long i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        // Add current piece position to attacked set
        attacked.insert({a, b});
    }

    // For each piece, mark all squares it can attack
    for (const auto& [x, y] : pieces) {
        for (int d = 0; d < 8; ++d) {
            long long nx = x + dx[d];
            long long ny = y + dy[d];
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                attacked.insert({nx, ny});
            }
        }
    }

    // Calculate the result
    unsigned __int128 total = (unsigned __int128)N * (unsigned __int128)N;
    total -= attacked.size();

    // Output the result
    cout << total << "\n";

    return 0;
}
