```python
n, m = map(int, input().split())
blocked = set()

# The 8 possible capture offsets (knight moves)
moves = [(2,1), (1,2), (-1,2), (-2,1), (-2,-1), (-1,-2), (1,-2), (2,-1)]

for _ in range(m):
    a, b = map(int, input().split())
    # Add the piece's position (cannot place on occupied squares)
    blocked.add((a, b))
    # For each knight move, mark the square that would be captured by this piece
    for dx, dy in moves:
        x, y = a + dx, b + dy
        if 1 <= x <= n and 1 <= y <= n:
            blocked.add((x, y))

# Total empty squares minus those that are under threat or occupied
ans = n * n - len(blocked)
print(ans)
```
