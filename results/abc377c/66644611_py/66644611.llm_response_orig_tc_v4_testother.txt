```python
n, m = map(int, input().split())
captured = set()

# Define the 8 possible knight moves
moves = [
    (2, 1), (1, 2), (-1, 2), (-2, 1),
    (-2, -1), (-1, -2), (1, -2), (2, -1)
]

for _ in range(m):
    a, b = map(int, input().split())
    # For each existing piece, mark all squares it can capture
    for dx, dy in moves:
        x, y = a + dx, b + dy
        if 1 <= x <= n and 1 <= y <= n:
            captured.add((x, y))

# Count empty squares that are not captured
ans = n * n - m  # Start with total squares minus occupied squares
ans -= len(captured)  # Subtract squares that are captured

print(ans)
```
