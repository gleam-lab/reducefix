```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); ++i)
using ll = long long;

// Directions that a placed piece can attack
const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

int main() {
    ll n, m;
    cin >> n >> m;

    set<pair<ll, ll>> occupied_or_attacked;

    vector<pair<ll, ll>> pieces(m);
    for (ll i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        occupied_or_attacked.insert({a, b});
    }

    // For each existing piece, mark all squares it can attack
    for (auto [x, y] : pieces) {
        for (int d = 0; d < 8; ++d) {
            ll nx = x + dx[d];
            ll ny = y + dy[d];
            if (nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                occupied_or_attacked.insert({nx, ny});
            }
        }
    }

    // Total number of cells is n*n, subtract the number of occupied or attacked cells
    cout << n * n - (ll)occupied_or_attacked.size() << endl;

    return 0;
}
```
