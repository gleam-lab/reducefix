#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    set<pair<long long, long long>> s;
    for(long long i = 0; i < m; i++){
        long long a, b;
        cin >> a >> b;
        s.insert({a, b});
    }

    long long ans = n * n;
    for(auto [x, y] : s){
        // Check all possible positions that can attack (x, y)
        for(long long dx = -2; dx <= 2; dx += 2){
            for(long long dy = -2; dy <= 2; dy += 2){
                if(abs(dx) + abs(dy) == 3 && x + dx >= 1 && x + dx <= n && y + dy >= 1 && y + dy <= n){
                    auto it = s.find({x + dx, y + dy});
                    if(it == s.end()){
                        ans--;
                        break;
                    }
                }
            }
        }
    }
    
    cout << ans << endl;
    return 0;
}
