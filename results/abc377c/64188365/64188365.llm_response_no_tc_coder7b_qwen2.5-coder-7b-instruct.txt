```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    ll n, m, a, b;
    cin >> n >> m;
    set<pair<ll, ll>> forbidden;
    rep(i, m) {
        cin >> a >> b;
        a--; b--;
        // Check all 8 directions
        if (a + 2 < n) {
            if (b + 1 < n) forbidden.emplace(a + 2, b + 1);
            if (b - 1 >= 0) forbidden.emplace(a + 2, b - 1);
        }
        if (a + 1 < n) {
            if (b + 2 < n) forbidden.emplace(a + 1, b + 2);
            if (b - 2 >= 0) forbidden.emplace(a + 1, b - 2);
        }
        if (a - 2 >= 0) {
            if (b + 1 < n) forbidden.emplace(a - 2, b + 1);
            if (b - 1 >= 0) forbidden.emplace(a - 2, b - 1);
        }
        if (a - 1 >= 0) {
            if (b + 2 < n) forbidden.emplace(a - 1, b + 2);
            if (b - 2 >= 0) forbidden.emplace(a - 1, b - 2);
        }
    }
    cout << (n * n - forbidden.size() - m) << endl;
}
```
