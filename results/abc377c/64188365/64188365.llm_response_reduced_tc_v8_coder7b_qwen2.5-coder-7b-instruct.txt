```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    long long n, m, a, b;
    cin >> n >> m;
    int ans = 0;
    set<pair<long long, long long>> c;
    rep(i, m) {
        cin >> a >> b;
        a--, b--;
        c.emplace(a, b);
        if (a + 2 < n) {
            if (b + 1 < n) {
                if (!c.count({a + 2, b + 1})) {
                    ans++;
                    c.emplace(a + 2, b + 1);
                }
            }
            if (b - 1 >= 0) {
                if (!c.count({a + 2, b - 1})) {
                    ans++;
                    c.emplace(a + 2, b - 1);
                }
            }
        }
        if (a + 1 < n) {
            if (b + 2 < n) {
                if (!c.count({a + 1, b + 2})) {
                    ans++;
                    c.emplace(a + 1, b + 2);
                }
            }
            if (b - 2 >= 0) {
                if (!c.count({a + 1, b - 2})) {
                    ans++;
                    c.emplace(a + 1, b - 2);
                }
            }
        }
        if (a - 2 >= 0) {
            if (b + 1 < n) {
                if (!c.count({a - 2, b + 1})) {
                    ans++;
                    c.emplace(a - 2, b + 1);
                }
            }
            if (b - 1 >= 0) {
                if (!c.count({a - 2, b - 1})) {
                    ans++;
                    c.emplace(a - 2, b - 1);
                }
            }
        }
        if (a - 1 >= 0) {
            if (b + 2 < n) {
                if (!c.count({a - 1, b + 2})) {
                    ans++;
                    c.emplace(a - 1, b + 2);
                }
            }
            if (b - 2 >= 0) {
                if (!c.count({a - 1, b - 2})) {
                    ans++;
                    c.emplace(a - 1, b - 2);
                }
            }
        }
    }
    cout << n * n - c.size() - ans << endl;
}
```
