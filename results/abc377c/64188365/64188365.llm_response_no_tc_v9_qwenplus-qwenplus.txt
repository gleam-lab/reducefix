```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    ll n, m;
    cin >> n >> m;

    set<pair<ll, ll>> occupied;
    vector<pair<ll, ll>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        a--; b--;
        occupied.insert({a, b});
        for (auto [dx, dy] : knight_moves) {
            ll nx = a + dx;
            ll ny = b + dy;
            if (nx >= 0 && nx < n && ny >= 0 && ny < n) {
                occupied.insert({nx, ny});
            }
        }
    }

    cout << n * n - (ll)occupied.size() << endl;
}
```
