#include <iostream>
#include <set>
#include <vector>

using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main() {
    long long n, m;
    cin >> n >> m;
    vector<pair<long long, long long>> pieces(m);
    set<pair<long long, long long>> occupied;
    
    for (int i = 0; i < m; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
        occupied.emplace(pieces[i].first - 1, pieces[i].second - 1);
    }

    // Create a grid to store if a position is occupied or not
    vector<vector<int>> grid(n, vector<int>(n, 0));
    for (auto& p : occupied) {
        int x = p.first;
        int y = p.second;
        grid[x][y] = 1;
        // Check and mark the pieces that can be captured from (x, y)
        if (x + 2 < n && y + 1 < n) grid[x + 2][y + 1] = 1;
        if (x + 1 < n && y + 2 < n) grid[x + 1][y + 2] = 1;
        if (x - 1 < n && y + 2 < n) grid[x - 1][y + 2] = 1;
        if (x - 2 < n && y + 1 < n) grid[x - 2][y + 1] = 1;
        if (x - 2 < n && y - 1 >= 0) grid[x - 2][y - 1] = 1;
        if (x - 1 < n && y - 2 >= 0) grid[x - 1][y - 2] = 1;
        if (x + 1 < n && y - 2 >= 0) grid[x + 1][y - 2] = 1;
        if (x + 2 < n && y - 1 >= 0) grid[x + 2][y - 1] = 1;
    }

    long long count = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            // If the position is not occupied and not captured by any piece, it's a valid position to place the new piece
            if (grid[i][j] == 0) {
                bool valid = true;
                for (auto& p : occupied) {
                    if (p.first == i && p.second == j) {
                        valid = false;
                        break;
                    }
                }
                if (valid) count++;
            }
        }
    }

    cout << count << endl;
}
