```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll N, M;
    cin >> N >> M;

    set<pair<ll, ll>> occupied;

    // All 8 possible attack offsets from (i,j)
    const array<array<ll, 2>, 8> deltas = {{
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    }};

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        // Convert to 0-based index
        a--;
        b--;

        occupied.insert({a, b});

        // Add all attacked squares by this piece
        for (const auto& d : deltas) {
            ll ni = a + d[0];
            ll nj = b + d[1];
            if (ni >= 0 && ni < N && nj >= 0 && nj < N) {
                occupied.insert({ni, nj});
            }
        }
    }

    // Total number of cells - number of non-empty or threatened cells
    cout << (N * N - (ll)occupied.size()) << endl;

    return 0;
}
```
