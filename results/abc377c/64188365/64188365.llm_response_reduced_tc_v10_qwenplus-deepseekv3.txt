#include <bits/stdc++.h>

    set<pair<ll, ll>> dangerous_positions;

                    if (ni >= 0 && ni < n && nj >= 0 && nj < n) {
                        dangerous_positions.emplace(ni, nj);
                    }
                }
            }

    ll total_safe = n * n - existing_pieces.size();
    for (auto pos : dangerous_positions) {

        }
    }
    
    cout << total_safe << endl;
}
