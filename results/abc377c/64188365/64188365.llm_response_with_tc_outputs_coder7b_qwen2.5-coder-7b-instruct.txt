#include <bits/stdc++.h>
using namespace std;

struct HashPair {
    size_t operator()(const pair<long long, long long>& p) const {
        return hash<long long>()(p.first) ^ hash<long long>()(p.second);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    long long n, m;
    cin >> n >> m;
    int ans = 0;
    unordered_set<pair<long long, long long>, HashPair> c;
    for (int i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        --a; --b;
        c.emplace(a, b);
        if (a + 2 < n) {
            if (b + 1 < n) {
                c.emplace(a + 2, b + 1);
            }
            if (b - 1 >= 0) {
                c.emplace(a + 2, b - 1);
            }
        }
        if (a + 1 < n) {
            if (b + 2 < n) {
                c.emplace(a + 1, b + 2);
            }
            if (b - 2 >= 0) {
                c.emplace(a + 1, b - 2);
            }
        }
        if (a - 2 >= 0) {
            if (b + 1 < n) {
                c.emplace(a - 2, b + 1);
            }
            if (b - 1 >= 0) {
                c.emplace(a - 2, b - 1);
            }
        }
        if (a - 1 >= 0) {
            if (b + 2 < n) {
                c.emplace(a - 1, b + 2);
            }
            if (b - 2 >= 0) {
                c.emplace(a - 1, b - 2);
            }
        }
    }
    cout << n * n - (long long)c.size() << "\n";
}
