#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    set<pair<long long, long long>> c;
    rep(i, m) {
        long long a, b;
        cin >> a >> b;
        --a, --b;
        c.insert({a, b});
    }

    long long ans = n * n;
    for (auto [x, y] : c) {
        vector<pair<long long, long long>> moves = {
            {x + 2, y + 1}, {x + 1, y + 2}, {x - 1, y + 2}, {x - 2, y + 1},
            {x - 2, y - 1}, {x - 1, y - 2}, {x + 1, y - 2}, {x + 2, y - 1}
        };
        for (auto [dx, dy] : moves) {
            if (dx >= 0 && dx < n && dy >= 0 && dy < n) {
                ans -= c.count({dx, dy});
            }
        }
    }
    cout << ans << endl;
}
