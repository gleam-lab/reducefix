#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    long long n, m;
    cin >> n >> m;
    set<pair<long long, long long>> pieces;
    
    // Collect all pieces on the board
    rep(i, m) {
        long long a, b;
        cin >> a >> b;
        --a, --b; // Convert to zero-based indexing
        pieces.insert({a, b});
    }

    // Directions in which a piece can move to capture another piece
    vector<pair<int, int>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // Count the number of empty squares that can be occupied
    long long count = 0;
    rep(i, n) {
        rep(j, n) {
            bool canPlace = true;
            rep(d, 8) {
                int ni = i + directions[d].first;
                int nj = j + directions[d].second;
                if (ni >= 0 && ni < n && nj >= 0 && nj < n && pieces.count({ni, nj})) {
                    canPlace = false;
                    break;
                }
            }
            if (canPlace) {
                count++;
            }
        }
    }

    cout << count << endl;
    return 0;
}
