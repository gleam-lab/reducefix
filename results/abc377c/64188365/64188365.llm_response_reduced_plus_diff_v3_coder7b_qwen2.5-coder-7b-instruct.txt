#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e9 + 5;
vector<int> adj[MAXN];

bool valid(int x, int y, int n) {
    return x >= 0 && x < n && y >= 0 && y < n;
}

void solve() {
    int n, m;
    cin >> n >> m;
    vector<vector<bool>> board(n, vector<bool>(n, false));
    set<pair<int, int>> pieces;

    for (int i = 0; i < m; i++) {
        int x, y;
        cin >> x >> y;
        x--, y--;
        pieces.insert({x, y});
        board[x][y] = true;
    }

    for (auto [x, y] : pieces) {
        int dx[] = {-2, -2, -1, -1, 1, 1, 2, 2};
        int dy[] = {1, -1, 2, -2, 2, -2, 1, -1};

        for (int d = 0; d < 8; d++) {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (valid(nx, ny, n)) {
                board[nx][ny] = true;
            }
        }
    }

    int count = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (!board[i][j]) {
                count++;
            }
        }
    }

    cout << count << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }

    return 0;
}
