```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (n); ++i)

using ll = long long;

int main() {
    ll n, m;
    cin >> n >> m;
    
    set<pair<ll, ll>> pieces;
    rep(i, m) {
        ll a, b;
        cin >> a >> b;
        --a, --b; // Convert to zero-based index
        pieces.insert({a, b});
        
        // Check all possible positions a piece can capture
        vector<vector<bool>> directions(8, vector<bool>(2));
        directions[0] = {2, 1}; directions[1] = {1, 2};
        directions[2] = {-1, 2}; directions[3] = {-2, 1};
        directions[4] = {-2, -1}; directions[5] = {-1, -2};
        directions[6] = {1, -2}; directions[7] = {2, -1};

        for (const auto& dir : directions) {
            ll x = a + dir[0], y = b + dir[1];
            if (x >= 0 && x < n && y >= 0 && y < n) {
                pieces.insert({x, y});
            }
        }
    }

    cout << n * n - pieces.size() << endl;
    return 0;
}
```
