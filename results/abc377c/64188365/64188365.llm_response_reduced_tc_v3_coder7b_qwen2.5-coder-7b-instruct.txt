const int MOD = 1000000007;

bool isValidMove(int x, int y, vector<vector<bool>>& board) {

}

        for (auto& dir : directions) {
            int nx = a + dir.first;
            int ny = b + dir.second;
            if (isValidMove(nx, ny, board)) {

            }
        }
    }

    return 0;
