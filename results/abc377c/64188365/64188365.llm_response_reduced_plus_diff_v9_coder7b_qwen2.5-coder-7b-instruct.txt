```cpp
#include <iostream>
#include <set>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    long long n, m;
    cin >> n >> m;
    set<pair<long long, long long>> occupied;
    vector<vector<bool>> board(n + 1, vector<bool>(n + 1, false));

    for (long long i = 0; i < m; ++i) {
        long long x, y;
        cin >> x >> y;
        occupied.insert({x, y});
        board[x][y] = true;
    }

    long long count = 0;
    for (long long i = 1; i <= n; ++i) {
        for (long long j = 1; j <= n; ++j) {
            if (!board[i][j]) {
                bool safe = true;
                vector<pair<int, int>> directions = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
                for (const auto& [dx, dy] : directions) {
                    long long nx = i + dx, ny = j + dy;
                    if (nx >= 1 && nx <= n && ny >= 1 && ny <= n && occupied.find({nx, ny}) != occupied.end()) {
                        safe = false;
                        break;
                    }
                }
                if (safe) {
                    ++count;
                }
            }
        }
    }

    cout << count << endl;
    return 0;
}
```
