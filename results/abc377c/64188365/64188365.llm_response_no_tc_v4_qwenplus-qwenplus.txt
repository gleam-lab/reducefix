```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll N, M;
    cin >> N >> M;

    set<pair<ll, ll>> occupied;
    vector<pair<ll, ll>> knight_offsets = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        // Use 0-based indexing internally
        a--; b--;

        // Mark the piece's position
        occupied.emplace(a, b);

        // Mark all positions that this piece can attack
        for (auto [dx, dy] : knight_offsets) {
            ll x = a + dx;
            ll y = b + dy;
            if (x >= 0 && x < N && y >= 0 && y < N) {
                occupied.emplace(x, y);
            }
        }
    }

    // Total squares - threatened or occupied squares
    cout << (N * N - (ll)occupied.size()) << endl;

    return 0;
}
```
