#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    long long n, m, a, b;
    cin >> n >> m;
    int ans = 0;
    set<pair<ll, ll>> c;
    rep(i, m) {
        cin >> a >> b;
        a--, b--;
        c.emplace(a, b);
    }
    rep(i, n) {
        rep(j, n) {
            if (!c.count({i, j})) {
                bool valid = true;
                if (i + 2 < n && j + 1 < n && !c.count({i + 2, j + 1})) valid = false;
                if (i + 1 < n && j + 2 < n && !c.count({i + 1, j + 2})) valid = false;
                if (i - 1 >= 0 && j + 2 < n && !c.count({i - 1, j + 2})) valid = false;
                if (i - 2 >= 0 && j + 1 < n && !c.count({i - 2, j + 1})) valid = false;
                if (i - 2 >= 0 && j - 1 >= 0 && !c.count({i - 2, j - 1})) valid = false;
                if (i - 1 >= 0 && j - 2 >= 0 && !c.count({i - 1, j - 2})) valid = false;
                if (i + 1 < n && j - 2 >= 0 && !c.count({i + 1, j - 2})) valid = false;
                if (i + 2 < n && j - 1 >= 0 && !c.count({i + 2, j - 1})) valid = false;
                if (valid) ans++;
            }
        }
    }
    cout << ans << endl;
}
