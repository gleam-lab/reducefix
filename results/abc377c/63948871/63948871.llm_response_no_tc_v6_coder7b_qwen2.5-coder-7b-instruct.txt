```cpp
#include <iostream>
#include <vector>
#include <string>
#include <set>

using namespace std;

#define ll long long

ll N, M;
set<pair<ll, ll>> positions;

void add_position(ll x, ll y) {
    if (x >= 1 && x <= N && y >= 1 && y <= N) {
        positions.insert({x, y});
    }
}

int main() {
    cin >> N >> M;
    
    for (int i = 0; i < M; ++i) {
        ll x, y;
        cin >> x >> y;
        positions.insert({x, y});

        add_position(x + 2, y + 1);
        add_position(x + 1, y + 2);
        add_position(x - 1, y + 2);
        add_position(x - 2, y + 1);
        add_position(x - 2, y - 1);
        add_position(x - 1, y - 2);
        add_position(x + 1, y - 2);
        add_position(x + 2, y - 1);
    }

    ll total_positions = N * N;
    ll valid_positions = total_positions - positions.size();
    
    cout << valid_positions << endl;
    
    return 0;
}
```
