```cpp
#include <iostream>
#include <set>
#include <utility>

using namespace std;

using i32 = int;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N;
    i32 M;
    cin >> N >> M;

    set<pair<i64, i64>> attacked_positions;

    // The 8 possible attack patterns from a position (like knight moves in chess)
    const pair<i64, i64> attack_offsets[8] = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (i32 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;

        // Add current piece's position to the set of blocked positions
        attacked_positions.insert({a, b});

        // Mark all squares that this piece can attack
        for (auto [dx, dy] : attack_offsets) {
            i64 x = a + dx;
            i64 y = b + dy;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                attacked_positions.insert({x, y});
            }
        }
    }

    // Total number of cells: N * N
    // Subtract number of attacked or occupied positions
    cout << (N * N - attacked_positions.size()) << '\n';

    return 0;
}
```
