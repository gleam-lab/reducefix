```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    set<pair<int, int>> positions;

    for(int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        positions.insert({x, y});

        // Check all possible knight moves
        for(int dx = -2; dx <= 2; dx += 4) {
            for(int dy = -1; dy <= 1; dy += 2) {
                int nx = x + dx;
                int ny = y + dy;
                if(nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                    positions.insert({nx, ny});
                }
                nx = x + dy;
                ny = y + dx;
                if(nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                    positions.insert({nx, ny});
                }
            }
        }
    }

    cout << (long long)n * n - positions.size() << endl;

    return 0;
}
```
