#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;

long long add(long long a, long long b) {
    return (a + b) % MOD;
}

long long mul(long long a, long long b) {
    return (a * b) % MOD;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int n, m;
    cin >> n >> m;

    set<pair<int, int>> s;
    for(int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        --x, --y;
        s.insert({x, y});
        for(int dx = -2; dx <= 2; ++dx) {
            for(int dy = -2; dy <= 2; ++dy) {
                if(dx == 0 || dy == 0 || abs(dx) == abs(dy)) continue;
                int nx = x + dx, ny = y + dy;
                if(nx >= 0 && nx < n && ny >= 0 && ny < n) s.insert({nx, ny});
            }
        }
    }

    long long total = 1LL * n * n;
    long long valid = total - s.size();
    cout << valid << '\n';

    return 0;
}
