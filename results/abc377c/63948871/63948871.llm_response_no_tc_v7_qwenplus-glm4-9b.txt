        long long a, b;

        pair<long long, long long> start = make_pair(a, b);

        reachable_squares.insert(make_pair(a - 1, b + 2));
        reachable_squares.insert(make_pair(a - 2, b + 1));
        reachable_squares.insert(make_pair(a - 2, b - 1));
        reachable_squares.insert(make_pair(a - 1, b - 2));
        reachable_squares.insert(make_pair(a + 1, b - 2));
        reachable_squares.insert(make_pair(a + 2, b - 1));
