#include <iostream>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

// Custom hash for pair<int, int>
struct pair_hash {
    size_t operator()(const pair<int, int>& p) const {
        // Use a simple 64-bit mixing function to avoid collisions
        uint64_t a = static_cast<uint64_t>(p.first);
        uint64_t b = static_cast<uint64_t>(p.second);
        return a * 1000000007 + b;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long long M;
    cin >> N >> M;

    unordered_set<pair<int, int>, pair_hash> attacked;

    vector<pair<int, int>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    vector<pair<int, int>> pieces(M);
    for (auto& p : pieces) {
        cin >> p.first >> p.second;
    }

    for (const auto& p : pieces) {
        // Add the piece itself to the set of attacked positions
        attacked.insert(p);

        // Check all knight moves
        for (const auto& move : knight_moves) {
            int ni = p.first + move.first;
            int nj = p.second + move.second;
            if (ni >= 1 && ni <= N && nj >= 1 && nj <= N) {
                attacked.insert({ni, nj});
            }
        }
    }

    // Calculate total number of squares and subtract attacked ones
    unsigned long long total_squares = static_cast<unsigned long long>(N) * N;
    cout << (total_squares - attacked.size()) << endl;

    return 0;
}
