#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

int main() {
    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    set<pair<int, int>> occupied;
    for (const auto& piece : pieces) {
        occupied.insert(piece);
    }

    long long answer = N * N;
    for (const auto& piece : pieces) {
        int x = piece.first;
        int y = piece.second;

        for (int dx = -2; dx <= 2; ++dx) {
            for (int dy = -2; dy <= 2; ++dy) {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                    // The piece at (nx, ny) is not directly occupied
                    if (occupied.find({nx, ny}) == occupied.end()) {
                        // Calculate the bounds of the capture zone
                        int min_x = min(x, nx);
                        int max_x = max(x, nx);
                        int min_y = min(y, ny);
                        int max_y = max(y, ny);
                        // Remove all cells within the capture zone
                        for (int i = min_x; i <= max_x; ++i) {
                            for (int j = min_y; j <= max_y; ++j) {
                                if (i != nx || j != ny) {
                                    if (occupied.find({i, j}) != occupied.end()) {
                                        occupied.erase({i, j});
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    // Subtract the occupied cells from the total number of cells
    answer -= occupied.size();
    cout << answer << endl;

    return 0;
}
