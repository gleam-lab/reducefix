```cpp
#include <iostream>
#include <vector>
#include <string>
#include <set>
#include <cmath>

using namespace std;

#define ll long long

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    set<pair<ll, ll>> anti_dup;

    for (ll i = 0; i < M; ++i) {
        ll x, y;
        cin >> x >> y;

        anti_dup.insert({x, y});

        auto check_and_insert = [&](ll dx, ll dy) {
            if (x + dx >= 1 && x + dx <= N && y + dy >= 1 && y + dy <= N) {
                anti_dup.insert({x + dx, y + dy});
            }
        };

        check_and_insert(2, 1); check_and_insert(1, 2);
        check_and_insert(-1, 2); check_and_insert(-2, 1);
        check_and_insert(-2, -1); check_and_insert(-1, -2);
        check_and_insert(1, -2); check_and_insert(2, -1);
    }

    ll total_squares = N * N;
    ll valid_squares = total_squares - anti_dup.size();
    cout << valid_squares << endl;

    return 0;
}
```
