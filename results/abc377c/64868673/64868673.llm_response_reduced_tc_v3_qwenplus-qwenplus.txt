```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Set to store all occupied squares and attacked squares
    unordered_set<string> attacked_or_occupied;
    
    // 8 possible attack positions from a knight move in chess
    vector<pair<int, int>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;

        // Mark current piece position
        string pos = to_string(a) + "," + to_string(b);
        attacked_or_occupied.insert(pos);

        // Check all 8 attack positions
        for (auto [dx, dy] : directions) {
            int x = a + dx;
            int y = b + dy;

            // If within bounds
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                string attack_pos = to_string(x) + "," + to_string(y);
                attacked_or_occupied.insert(attack_pos);
            }
        }
    }

    // Total attacked or occupied squares
    long long total_attacked_or_occupied = attacked_or_occupied.size();

    // Total valid positions = total grid size - attacked or occupied count
    cout << N * N - total_attacked_or_occupied << "\n";

    return 0;
}
```
