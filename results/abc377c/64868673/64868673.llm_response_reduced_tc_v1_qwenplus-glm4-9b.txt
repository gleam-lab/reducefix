            long long nx = x + dx[i];
            long long ny = y + dy[i];
            if (nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                infeasible[nx][ny] = true;
            }
        }
    }

    // Calculate the number of infeasible cells
    long long infeasibleCells = 0;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (infeasible[i][j]) {
                infeasibleCells++;
            }
        }
    }

    // Calculate the number of feasible cells
    long long feasibleCells = n * n - infeasibleCells;
    cout << feasibleCells << endl;
