#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    vector<pair<long long, long long>> pieces(m);
    vector<pair<long long, long long>> to_check;
    set<pair<long long, long long>> occupied;

    for (long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        occupied.insert({a, b});
        for (long long j = 0; j < 8; ++j) {
            long long x = a + j / 2 + 2 * (j % 2);
            long long y = b + j % 2 - 2 * (j / 2);
            to_check.push_back({x, y});
        }
    }

    // Sort to_check to make it easier to add elements in order
    sort(to_check.begin(), to_check.end());

    long long ans = n * n; // Start with total number of cells
    long long current_cell = 1;

    for (auto& p : to_check) {
        if (p.first <= 0 || p.first > n || p.second <= 0 || p.second > n) continue;
        if (occupied.find({p.first, p.second}) != occupied.end()) continue;

        // If current cell is not inside the allowed range, add the range to answer
        if (current_cell > p.first) {
            ans -= p.first - current_cell + 1;
        }
        current_cell = max(current_cell, p.first + 1); // Update current cell to the next allowed starting point

        // If the range is too large, it might have been already subtracted
        if (current_cell > n) break;
    }

    cout << ans << endl;
    return 0;
}
