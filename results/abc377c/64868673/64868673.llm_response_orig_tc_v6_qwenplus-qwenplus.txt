```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Set to store positions of all existing pieces
    set<pair<long long, long long>> pieces;

    // Directions for the knight-like moves (relative)
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // Set to track all forbidden positions (occupied or under attack)
    set<pair<long long, long long>> forbidden;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        pair<long long, long long> pos = {a, b};
        pieces.insert(pos);
        forbidden.insert(pos);

        // Add all 8 possible attack positions from this piece
        for (const auto& [dx, dy] : directions) {
            long long x = a + dx;
            long long y = b + dy;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                forbidden.insert({x, y});
            }
        }
    }

    // Total number of squares: N*N
    // Subtract the number of forbidden squares
    cout << (N * N - (long long)forbidden.size()) << endl;

    return 0;
}
```
