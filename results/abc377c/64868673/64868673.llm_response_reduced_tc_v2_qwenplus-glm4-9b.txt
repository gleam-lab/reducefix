
```cpp
#include <iostream>
#include <vector>
using namespace std;

const int MAX_N = 1e9;

int main() {
    long long n, m;
    cin >> n >> m;

    vector<vector<long long>> pieces(MAX_N + 1, vector<long long>(MAX_N + 1, 0));
    vector<vector<long long>> capture(MAX_N + 1, vector<long long>(MAX_N + 1, 0));
    
    for (int i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        pieces[a][b]++;
    }

    // For each piece, calculate the cells it can capture
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (pieces[i][j] > 0) {
                for (int k = 0; k < 8; ++k) {
                    long long ni = i + dx[k];
                    long long nj = j + dy[k];
                    if (ni >= 1 && ni <= n && nj >= 1 && nj <= n) {
                        capture[ni][nj]++;
                    }
                }
            }
        }
    }

    // Calculate the number of cells that cannot be captured by any piece
    long long free_cells = 0;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (capture[i][j] == 0) {
                free_cells++;
            }
        }
    }

    cout << free_cells << endl;

    return 0;
}

const int dx[8] = {1, 1, -1, -1, 2, 2, -2, -2};
const int dy[8] = {2, -2, 2, -2, 1, -1, 1, -1};
```
