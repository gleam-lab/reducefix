#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

template<typename T, typename U>
struct Delta {
    int dx, dy;
    Delta() : dx(0), dy(0) {}
    Delta(T d, U o) : dx(d), dy(o) {}
};

template<typename T, typename U>
vector<Delta<T, U>> deltas = {
    Delta<T, U>(0,0), // idle
    Delta<T, U>(1,1), Delta<T, U>(1,-1), Delta<T, U>(-1,1), Delta<T, U>(-1,-1),
    Delta<T, U>(2,1), Delta<T, U>(2,-1), Delta<T, U>(-2,1), Delta<T, U>(-2,-1)
};

bool canPlace(int r, int c, int n, const vector<pair<int, int>>& pieces) {
    for (const auto& p : pieces) {
        int pr = p.first;
        int pc = p.second;
        bool canCapture = false;
        for (const auto& d : deltas<int, int>) {
            int nr = pr + d.dx;
            int nc = pc + d.dy;
            if (nr >= 1 && nr <= n && nc >= 1 && nc <= n) {
                // Check if the square is empty or the same as the current square
                if ((nr != r || nc != c) && ((nr == r && nc == c) || (nr != r && nc != c))) {
                    canCapture = true;
                    break;
                }
            }
        }
        if (canCapture) return false;
    }
    return true;
}

int main() {
    long long n, m;
    cin >> n >> m;
    vector<pair<int, int>> pieces(m);

    for (long long i = 0; i < m; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    long long ans = 0;
    for (int r = 1; r <= n; ++r) {
        for (int c = 1; c <= n; ++c) {
            if (canPlace(r, c, n, pieces)) {
                ans++;
            }
        }
    }

    cout << ans << endl;

    return 0;
}
