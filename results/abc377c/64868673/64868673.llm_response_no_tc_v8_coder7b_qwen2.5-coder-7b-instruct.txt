```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    unordered_set<long long> ex;
    long long n, m;
    cin >> n >> m;
    long long ans = n * n - m;
    
    for (long long i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        long long idx = (a - 1) * n + b - 1;
        
        // Check all possible positions that can capture this piece
        vector<vector<long long>> moves = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
        for (auto& move : moves) {
            long long nx = a + move[0], ny = b + move[1];
            if (nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                long long nidx = (nx - 1) * n + ny - 1;
                ex.insert(nidx);
            }
        }
    }

    // Subtract the size of the set from the total number of cells
    ans -= ex.size();
    cout << ans << endl;
    return 0;
}
```

### Explanation:
1. **Use `unordered_set` instead of `map`**: This provides faster lookup times, which is crucial given the constraints.
2. **Convert coordinates to a unique index**: By converting the 2D coordinates `(a, b)` to a 1D index `(a-1)*n + b-1`, we can use the set to efficiently check if a cell is already covered.
3. **Calculate the initial number of available cells**: Start with `n * n - m` as the initial count, subtracting the number of pieces already placed.
4. **Check all possible capturing positions**: For each piece, check its 8 possible capturing positions and add them to the set if they are within bounds.
5. **Subtract the size of the set from the initial count**: The final result is the number of empty cells minus the number of cells covered by other pieces.

This approach ensures correctness and efficiency while handling the large constraints effectively.
