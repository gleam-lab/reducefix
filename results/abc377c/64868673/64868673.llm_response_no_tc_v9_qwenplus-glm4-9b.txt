#include <iostream>
using namespace std;

const int dx[] = {1, 1, 2, 2, -1, -1, -2, -2};
const int dy[] = {2, -2, 1, -1, 2, -2, 1, -1};

int main() {
    int n, m;
    cin >> n >> m;
    bool grid[n][n];

    // Initialize the grid as empty
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            grid[i][j] = false;
        }
    }

    // Read the positions of the pieces and mark them as occupied
    for (int i = 0; i < m; ++i) {
        int r, c;
        cin >> r >> c;
        --r; --c; // Convert 1-based index to 0-based
        grid[r][c] = true;
        for (int j = 0; j < 8; ++j) {
            int nr = r + dx[j];
            int nc = c + dy[j];
            if (nr >= 0 && nr < n && nc >= 0 && nc < n) {
                grid[nr][nc] = true;
            }
        }
    }

    // Count the number of empty cells
    long long ans = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (!grid[i][j]) {
                ans++;
            }
        }
    }

    cout << ans << endl;

    return 0;
}
