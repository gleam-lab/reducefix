```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long N;
    int M;
    cin >> N >> M;
    
    // Set to store all occupied squares and squares under attack
    unordered_set<long long> attacked_or_occupied;
    
    // Directions where a piece can attack
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };
    
    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        
        // Store current piece position
        long long key = a * 1000000000LL + b;
        attacked_or_occupied.insert(key);
        
        // Check all eight attacking positions
        for (auto [dx, dy] : directions) {
            long long x = a + dx;
            long long y = b + dy;
            
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                long long attack_key = x * 1000000000LL + y;
                attacked_or_occupied.insert(attack_key);
            }
        }
    }
    
    // Total number of squares that are either occupied or under attack
    long long total_attacked_or_occupied = attacked_or_occupied.size();
    
    // The result is total squares minus those that are attacked or occupied
    cout << N * N - total_attacked_or_occupied << "\n";
    
    return 0;
}
```
