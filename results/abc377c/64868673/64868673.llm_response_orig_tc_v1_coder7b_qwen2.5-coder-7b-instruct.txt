#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(0);

    long long n, m;
    cin >> n >> m;

    // Using unordered_set for faster lookup
    unordered_set<long long> positions;
    for(long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        // Convert 2D position to 1D using a simple hash function
        positions.insert((a - 1) * n + b);
    }

    long long ans = 0;
    for(long long i = 1; i <= n; ++i) {
        for(long long j = 1; j <= n; ++j) {
            // Check all 8 directions
            bool valid = true;
            for(int d = 0; d < 8; ++d) {
                int dx[] = {2, 1, -1, -2, -2, -1, 1, 2};
                int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};
                int ni = i + dx[d], nj = j + dy[d];
                // Convert 2D position to 1D
                long long pos = (ni - 1) * n + nj;
                if(ni >= 1 && ni <= n && nj >= 1 && nj <= n && positions.find(pos) != positions.end()) {
                    valid = false;
                    break;
                }
            }
            if(valid) ++ans;
        }
    }
    cout << ans << endl;
    return 0;
}
