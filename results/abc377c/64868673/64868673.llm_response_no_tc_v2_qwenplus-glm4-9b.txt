
```cpp
#include <iostream>
#include <vector>
#include <cstring>
using namespace std;

const int dx[8] = {2, 2, -2, -2, 1, 1, -1, -1};
const int dy[8] = {1, -1, 1, -1, 2, -2, 2, -2};

int main() {
    long long n, m;
    cin >> n >> m;
    vector<vector<bool>> grid(n, vector<bool>(n, false));
    long long occupied = 0;

    // Read and mark the occupied positions, including the range of capture
    for (long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        grid[a - 1][b - 1] = true; // Mark the originating position
        for (int j = 0; j < 8; ++j) {
            long long x = a - 1 + dx[j];
            long long y = b - 1 + dy[j];
            if (x >= 0 && x < n && y >= 0 && y < n) {
                grid[x][y] = true; // Mark the capturing range
            }
        }
        occupied++;
    }

    // Calculate the number of empty positions
    long long empty_positions = n * n - occupied;

    // Subtract the positions that are within the capture range of each piece
    for (long long i = 0; i < n; ++i) {
        for (long long j = 0; j < n; ++j) {
            if (grid[i][j]) { // If the position is captured
                for (int k = 0; k < 8; ++k) {
                    long long x = i + dx[k];
                    long long y = j + dy[k];
                    if (x >= 0 && x < n && y >= 0 && y < n && grid[x][y]) {
                        empty_positions--;
                    }
                }
            }
        }
    }

    cout << empty_positions << endl;

    return 0;
}
```

### Explanation:
1. **Grid Representation**: We use a 2D vector `grid` to represent the board. If `grid[i][j]` is `true`, it means that the position `(i, j)` is either occupied by a piece or within the capture range of a piece.
2. **Marking Occupied Positions**: For each piece, we mark the originating position and the capture range in the `grid`.
3. **Calculating Empty Positions**: Initially, we calculate the total number of empty positions as `n * n`.
4. **Subtracting Captured Positions**: For each occupied position, we subtract the positions within its capture range from the count of empty positions.
5. **Output**: Finally, we output the count of positions that are not occupied and not within the capture range of any existing pieces.
