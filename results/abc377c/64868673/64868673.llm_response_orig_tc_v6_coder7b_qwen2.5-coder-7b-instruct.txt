#include <bits/stdc++.h>
using namespace std;

bool isValid(int x, int y, int n) {
    return x >= 0 && x < n && y >= 0 && y < n;
}

int main() {
    ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    long long n, m;
    cin >> n >> m;
    vector<vector<bool>> board(n, vector<bool>(n));
    set<pair<long long, long long>> occupied;
    
    for(long long i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        a--; b--;
        board[a][b] = true;
        occupied.insert({a, b});
    }

    long long directions[8][2] = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
    long long count = 0;

    for(long long i = 0; i < n; i++) {
        for(long long j = 0; j < n; j++) {
            if(!board[i][j]) {
                bool safe = true;
                for(auto dir : directions) {
                    long long ni = i + dir[0], nj = j + dir[1];
                    if(isValid(ni, nj, n) && board[ni][nj]) {
                        safe = false;
                        break;
                    }
                }
                if(safe) {
                    count++;
                }
            }
        }
    }

    cout << count << endl;
    return 0;
}
