```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long N, M;
    cin >> N >> M;
    
    // Set to store all positions that are either occupied or under attack
    set<pair<long long, long long>> forbidden;
    
    // All 8 possible attack vectors from a position
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };
    
    for (long long i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        
        // Current piece position
        forbidden.insert({a, b});
        
        // Check all 8 attack positions
        for (const auto& [dx, dy] : directions) {
            long long x = a + dx;
            long long y = b + dy;
            
            // Only include positions that are within the grid
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                forbidden.insert({x, y});
            }
        }
    }
    
    // The number of usable cells is total cells minus forbidden cells
    cout << N * N - (long long)forbidden.size() << endl;
    
    return 0;
}
```
