#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int n, m;
vector<pair<int, int>> occupied;

int main() {
    cin >> n >> m;
    occupied.resize(m);
    for (int i = 0; i < m; i++) {
        cin >> occupied[i].first >> occupied[i].second;
    }

    // Vector to keep track of the range of occupied cells
    vector<pair<int, int>> occupied_ranges;

    // Adding occupied cells to the range vector
    for (auto &cell : occupied) {
        int x = cell.first;
        int y = cell.second;

        // Adding ranges to the vector
        occupied_ranges.emplace_back(x - 2, y + 1);
        occupied_ranges.emplace_back(x - 1, y + 2);
        occupied_ranges.emplace_back(x + 1, y + 2);
        occupied_ranges.emplace_back(x + 2, y + 1);
        occupied_ranges.emplace_back(x + 2, y - 1);
        occupied_ranges.emplace_back(x + 1, y - 2);
        occupied_ranges.emplace_back(x - 1, y - 2);
        occupied_ranges.emplace_back(x - 2, y - 1);
    }

    // Sorting the ranges to merge overlapping ranges
    sort(occupied_ranges.begin(), occupied_ranges.end());

    // Merging overlapping ranges
    vector<pair<int, int>> merged_ranges;
    for (auto &range : occupied_ranges) {
        if (merged_ranges.empty() || merged_ranges.back().second < range.first) {
            merged_ranges.emplace_back(range);
        } else {
            merged_ranges.back().second = max(merged_ranges.back().second, range.second);
        }
    }

    // Calculating the number of safe cells
    int safe_cells = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int x = i;
            int y = j;
            bool is_safe = true;

            // Checking if the cell is safe from the occupied ranges
            for (auto &range : merged_ranges) {
                if (range.first <= x + 2 && range.second >= y - 2 && range.first + 2 >= x && range.second - 2 <= y) {
                    is_safe = false;
                    break;
                }
            }

            // If the cell is safe, increment the counter
            if (is_safe) {
                safe_cells++;
            }
        }
    }

    cout << safe_cells << endl;

    return 0;
}
