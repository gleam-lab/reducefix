```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N, M;
    cin >> N >> M;

    // Set to store positions with pieces
    unordered_set<string> occupied;
    // Set to store positions that are either occupied or can be attacked
    unordered_set<string> blocked;

    // Directions where a piece at (i,j) can attack
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (long long i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        string key = to_string(a) + "," + to_string(b);
        occupied.insert(key);

        // This square is not empty -> it cannot be considered for placing our piece
        blocked.insert(key);

        for (auto [dx, dy] : directions) {
            long long x = a + dx;
            long long y = b + dy;

            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                string block_key = to_string(x) + "," + to_string(y);
                blocked.insert(block_key);
            }
        }
    }

    // Total number of squares: N * N
    // Subtract the number of blocked squares
    // And also subtract the already occupied squares since they are not empty
    long long total_squares = N * N;
    long long result = total_squares - blocked.size();

    cout << result << endl;

    return 0;
}
```
