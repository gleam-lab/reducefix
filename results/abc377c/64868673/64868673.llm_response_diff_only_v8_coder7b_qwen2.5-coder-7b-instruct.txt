            ex.insert(p);
            ans++;

                {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, 
                {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
            };
            
            for (auto &dir : directions) {
                long long x = a + dir.first - 1, y = b + dir.second - 1; // Adjusting indices to zero-based
                if (x >= 0 && x < n && y >= 0 && y < n) {
                    pair<long long, long long> neighbor = make_pair(x, y);
                    if (!ex.count(neighbor)) {
                        ex.insert(neighbor);
                        ans++;
                    }
                }
            }
        }
    }
    
    cout << n * n - ans << endl;
    return 0;
}
