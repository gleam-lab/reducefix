```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long N;
    int M;
    cin >> N >> M;
    
    // Set to store all occupied squares and squares under attack
    unordered_set<string> attacked_or_occupied;
    
    // Directions for attacking moves
    vector<pair<int, int>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };
    
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        
        // Store the piece itself
        attacked_or_occupied.emplace(to_string(a) + "," + to_string(b));
        
        // Mark all reachable attacked positions
        for (auto [dx, dy] : directions) {
            int na = a + dx;
            int nb = b + dy;
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                attacked_or_occupied.emplace(to_string(na) + "," + to_string(nb));
            }
        }
    }
    
    // Total number of squares that are either occupied or under attack
    long long total_attacked_or_occupied = attacked_or_occupied.size();
    
    // The answer is total grid squares minus attacked/occupied squares
    cout << N * N - total_attacked_or_occupied << "\n";
    
    return 0;
}
```
