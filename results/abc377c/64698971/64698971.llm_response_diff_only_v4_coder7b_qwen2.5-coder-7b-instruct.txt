struct phash{
	inline size_t operator()(const pair<int,int> & p) const{
		const auto h1 = hash<int>()(p.first);
		const auto h2 = hash<int>()(p.second);
		return h1 ^ (h2 << 1);
	}
};

int N,M; 
unordered_set<pair<int,int>,phash> st;

void set_insert(int a,int b){
    if(0<=a && a<N && 0<=b && b<N) st.insert({a,b});
}

int main() {
    cin>>N>>M;

    vector<pair<int,int>> vc={{0,0},{2,1},{1,2},{-1,2},{-2,1},{-2,-1},{-1,-2},{1,-2},{2,-1}};
    rep(i,M){
        int a,b; cin>>a>>b;
        a--; b--;
        for(auto x:vc) set_insert(a+x.first,b+x.second);
    }

    // for(auto x:st) cout<<x.first<<' '<<x.second<<endl;

    cout<<N_-st.size();
}

### Explanation of Fixes
1. **Data Structure Choice**: Changed `set` to `unordered_set` with a custom hash function (`phash`) because we don't need the elements to be sorted and unordered sets generally provide faster insertions and lookups compared to sets.
2. **Hash Function**: Defined a simple hash function for pairs of integers to ensure that the unordered_set works correctly.
3. **Main Logic**: The rest of the logic remains the same, ensuring that we correctly identify and count the positions that cannot be used due to the presence of other pieces based on their possible capturing moves. Finally, we calculate the total number of available positions by subtracting the size of the set from \(N^2\).
