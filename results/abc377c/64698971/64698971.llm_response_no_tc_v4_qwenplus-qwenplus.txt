#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = n-1; i >= (int)(0); i--)
#define rrepi(i, a, b) for (int i = (int)(a); i > (int)(b); i--)
#define ALL(a) (a).begin(),(a).end()
#define ALL_(a) (a),(a)+size(a)
#define IF_T if(Is_test)

#define ll long long
using namespace std;

// Custom hash function for pair<int, int>
struct phash {
    inline size_t operator()(const pair<int,int> &p) const {
        const auto h1 = hash<int>()(p.first);
        const auto h2 = hash<int>()(p.second);
        return h1 ^ (h2 << 16); // Use shift to avoid collision
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N;
    int M;
    cin >> N >> M;

    unordered_set<pair<ll, ll>, phash> attacked_cells;
    
    // Define all possible attack patterns
    vector<pair<ll, ll>> attack_patterns = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // For each piece, mark the cells it can attack
    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;

        // Convert to 0-based index
        a--; b--;

        // Mark positions that this piece can attack
        for (auto [dx, dy] : attack_patterns) {
            ll x = a + dx;
            ll y = b + dy;

            // Only include valid cells in the grid
            if (x >= 0 && x < N && y >= 0 && y < N) {
                attacked_cells.insert({x, y});
            }
        }

        // Also mark the position of the existing piece (cannot place our piece there)
        attacked_cells.insert({a, b});
    }

    // Total number of cells: N*N
    // Subtract number of attacked or occupied cells
    cout << N * N - (ll)attacked_cells.size();

    return 0;
}
