#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int N, M;
set<pair<int, int>> st;
vector<pair<int, int>> moves{{0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};

void set_insert(int a, int b) {
    if (0 <= a && a < N && 0 <= b && b < N) st.insert({a, b});
}

bool can_place(int a, int b) {
    for (auto x : moves) {
        if (st.count({a + x.first, b + x.second})) return false;
    }
    return true;
}

int main() {
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
        pieces[i].first--; // Adjust position to 0-based
        pieces[i].second--; // Adjust position to 0-based
        for (auto x : moves) {
            set_insert(pieces[i].first + x.first, pieces[i].second + x.second);
        }
    }

    int count = 0;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            if (!st.count({i, j}) && can_place(i, j)) {
                count++;
            }
        }
    }

    cout << count << endl;

    return 0;
}
