#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define all(x) begin(x), end(x)
#define sz(x) (int)(x).size()
using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;

    // Using a set to store the positions of the captured cells
    set<pair<int, int>> captured_cells;

    // Directions for capturing pieces
    vector<pair<int, int>> directions{{0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};

    // Read the positions of the existing pieces and mark the captured cells
    for (int i = 0; i < M; ++i) {
        int x, y;
        cin >> x >> y;
        for (auto &dir : directions) {
            int nx = x + dir.first;
            int ny = y + dir.second;

            // Check if the new cell is within the grid boundaries
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                captured_cells.insert({nx, ny});
            }
        }
    }

    // Calculate the number of cells that are not captured
    long long free_cells = N * N - sz(captured_cells);

    // Output the number of free cells
    cout << free_cells << '\n';

    return 0;
}
