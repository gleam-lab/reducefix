```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    long long N;
    int M;
    cin >> N >> M;

    unordered_set<long long> attacked;
    // Use a hash function to map (a, b) to a unique key: (a * N + b)
    // To avoid overflow, we use a large prime number as multiplier
    const long long P = 1000000007;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        a--; b--;

        // These are the relative positions of the L-shaped moves in knight's tour
        vector<pair<long long, long long>> dir = {
            {-2, -1}, {-2, 1},
            {-1, -2}, {-1, 2},
            {1, -2}, {1, 2},
            {2, -1}, {2, 1}
        };

        for (auto [dx, dy] : dir) {
            long long na = a + dx;
            long long nb = b + dy;
            if (na >= 0 && na < N && nb >= 0 && nb < N) {
                attacked.insert((na * P + nb));
            }
        }

        // Also include the position itself because you can't place on occupied squares
        attacked.insert((a * P + b));
    }

    // Total number of cells is N*N
    // From this we subtract the number of attacked and occupied cells
    cout << N * N - (long long)attacked.size() << endl;

    return 0;
}
```
