#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

const int MAX_N = 1000000000;

int main() {
    int N, M;
    cin >> N >> M;
    
    set<pair<int, int>> occupied;
    for (int i = 1; i <= M; i++) {
        int x, y;
        cin >> x >> y;
        occupied.insert({x, y});
    }

    int result = 0;
    for (int i = 1; i <= N; i++) {
        for (int j = 1; j <= N; j++) {
            bool is_occupied = (occupied.find({i, j}) != occupied.end());
            bool can_attack = false;

            for (int dx = -2; dx <= 2; dx++) {
                for (int dy = -2; dy <= 2; dy++) {
                    if (dx == 0 && dy == 0) continue;
                    int nx = i + dx;
                    int ny = j + dy;
                    if (nx >= 1 && nx <= N && ny >= 1 && ny <= N && occupied.find({nx, ny}) != occupied.end()) {
                        can_attack = true;
                        break;
                    }
                }
                if (can_attack) break;
            }

            if (!is_occupied && !can_attack) {
                result++;
            }
        }
    }

    cout << result << endl;

    return 0;
}
