
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAXN = 1000000000; // Maximum value for N
const int MAX_M = 200000; // Maximum value for M
const vector<pair<int, int>> directions = {
    {0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
};

long long calculateAvailableSpaces(int N, const vector<pair<int, int>>& pieces) {
    bitset<MAXN> taken;
    for (const auto& piece : pieces) {
        taken[piece.first * N + piece.second] = true;
    }

    long long availableSpaces = 0;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            if (!taken[i * N + j]) {
                bool isSafe = true;
                for (const auto& dir : directions) {
                    int x = i + dir.first;
                    int y = j + dir.second;
                    if (x >= 0 && x < N && y >= 0 && y < N) {
                        if (taken[x * N + y]) {
                            isSafe = false;
                            break;
                        }
                    }
                }
                if (isSafe) {
                    availableSpaces++;
                }
            }
        }
    }
    return availableSpaces;
}

int main() {
    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        pieces[i] = {a - 1, b - 1};
    }

    cout << calculateAvailableSpaces(N, pieces) << endl;
    return 0;
}
```

### Explanation:
1. **Efficiency**: The solution uses a bitset to mark the pieces' positions for O(1) access. It iterates over all cells in the grid to find available spaces and checks each space against the pieces' capturing capabilities. This approach avoids the need for an expensive set insert operation, making it more efficient.
2. **Correctness**: The solution checks each cell in the grid against all possible capturing moves. If a cell is not taken and is not in the capture range of any existing piece, it is counted as an available space.
3. **Edge Cases**: The solution should handle edge cases such as when all cells are taken or when there are no existing pieces. The bitset ensures that out-of-bounds cells are not considered during the check for available spaces.
