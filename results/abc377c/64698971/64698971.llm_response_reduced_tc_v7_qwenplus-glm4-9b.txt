    captureRange[x + 2][y + 1] = captureRange[x + 1][y + 2] = captureRange[x - 1][y + 2] =
    captureRange[x - 2][y + 1] = captureRange[x - 2][y - 1] = captureRange[x - 1][y - 2] =
    captureRange[x + 1][y - 2] = captureRange[x + 2][y - 1] = 1;
}

    cin >> N >> M;
    
    for (int i = 0; i < M; ++i) {
        int x, y;

        occupied.insert({x, y});
        setCaptureRange(x, y);
    }
    
    long long countUnoccupied = 0;
    for (int i = 0; i < N; ++i) {
