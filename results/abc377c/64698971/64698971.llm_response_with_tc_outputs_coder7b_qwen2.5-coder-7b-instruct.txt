#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = n-1; i >= (int)(0); i--)
#define rrepi(i, a, b) for (int i = (int)(a); i > (int)(b); i--)
#define ALL(a) (a).begin(),(a).end()
#define ALL_(a) (a),(a)+size(a)
#define IF_T if(Is_test)

#define ll long long
using namespace std;

struct phash {
    inline size_t operator()(const pair<ll, ll>& p) const {
        const auto h1 = hash<ll>()(p.first);
        const auto h2 = hash<ll>()(p.second);
        return h1 ^ (h2 << 1);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll N, M;
    cin >> N >> M;

    unordered_set<pair<ll, ll>, phash> st;

    vector<pair<ll, ll>> moves = {
        {0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    rep(i, M) {
        ll a, b;
        cin >> a >> b;
        --a; --b;
        for (auto &[dx, dy] : moves) {
            ll nx = a + dx, ny = b + dy;
            if (nx >= 0 && nx < N && ny >= 0 && ny < N) {
                st.emplace(nx, ny);
            }
        }
    }

    ll total = N * N;
    cout << (total - (ll)st.size()) << "\n";
}
