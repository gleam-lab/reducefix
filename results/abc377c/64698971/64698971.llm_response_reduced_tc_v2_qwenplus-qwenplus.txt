#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = n-1; i >= (int)(0); i--)
#define rrepi(i, a, b) for (int i = (int)(a); i > (int)(b); i--)
#define ALL(a) (a).begin(),(a).end()
#define ALL_(a) (a),(a)+size(a)
#define IF_T if(Is_test)

#define ll long long
using namespace std;

struct phash {
    inline size_t operator()(const pair<ll,ll> & p) const {
        // Use hash that supports larger values
        return hash<ll>()(p.first * 1'000'000'000 + p.second);
    }
};

int N,M; 
unordered_set<pair<ll,ll>, phash> attacked;

void mark_attack(ll x, ll y) {
    attacked.insert({x, y});
}

int main() {
    cin>>N>>M;
    
    vector<pair<ll,ll>> attack_offsets = {
        {2,1}, {1,2}, {-1,2}, {-2,1},
        {-2,-1}, {-1,-2}, {1,-2}, {2,-1}
    };

    rep(i, M) {
        ll a, b;
        cin >> a >> b;
        a--; b--;

        // Mark the piece's position as occupied
        attacked.insert({a, b});

        // Mark all positions this piece can attack
        for (auto [dx, dy] : attack_offsets) {
            ll na = a + dx;
            ll nb = b + dy;
            if (0 <= na && na < N && 0 <= nb && nb < N) {
                attacked.insert({na, nb});
            }
        }
    }

    // Total number of cells is N*N, subtract number of attacked or occupied cells
    cout << (1LL * N * N - (ll)attacked.size()) << endl;
}
