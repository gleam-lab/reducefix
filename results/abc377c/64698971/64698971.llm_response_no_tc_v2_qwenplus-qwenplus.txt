#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define repi(i, a, b) for (int i = (int)(a); i < (int)(b); i++)
#define rrep(i, n) for (int i = n-1; i >= (int)(0); i--)
#define rrepi(i, a, b) for (int i = (int)(a); i > (int)(b); i--)
#define ALL(a) (a).begin(),(a).end()
using namespace std;

typedef long long ll;
typedef pair<int, int> P;

int N, M;
set<pair<ll, ll>> attacked_positions;
set<pair<ll, ll>> occupied_positions;

// Knight-like attack patterns
const vector<pair<ll, ll>> directions = {
    {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
    {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
};

int main() {
    cin.tie(nullptr);
    ios::sync_with_stdio(false);

    cin >> N >> M;

    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        // Convert to 0-based index
        a--; b--;
        
        // Add the position of the existing piece
        occupied_positions.insert({a, b});
        
        // Mark all reachable positions from this piece
        for (auto [dx, dy] : directions) {
            ll nx = a + dx;
            ll ny = b + dy;
            if (0 <= nx && nx < N && 0 <= ny && ny < N) {
                attacked_positions.insert({nx, ny});
            }
        }
    }

    // Remove positions that are already occupied by other pieces
    for (auto pos : occupied_positions) {
        attacked_positions.erase(pos);
    }

    // Total number of cells is N * N
    ll total_cells = 1LL * N * N;
    
    // Subtract the number of occupied cells and attacked cells
    ll available_cells = total_cells - occupied_positions.size() - attacked_positions.size();
    
    cout << available_cells << endl;

    return 0;
}
