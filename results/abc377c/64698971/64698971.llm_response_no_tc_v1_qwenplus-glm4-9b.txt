#include <iostream>
#include <vector>
#include <set>

using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define rrep(i, n) for (int i = (n) - 1; i >= 0; i--)
#define all(x) begin(x), end(x)

const vector<pair<int, int>> captures = {
    {0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
};

int main() {
    long long N, M;
    cin >> N >> M;
    set<pair<int, int>> occupied;

    // Read the positions of the existing pieces and mark them as occupied
    rep(i, M) {
        int x, y;
        cin >> x >> y;
        x--; y--; // Convert to 0-based index
        occupied.insert({x, y});
    }

    // Calculate the number of cells that are not occupied by any piece
    long long free_cells = N * N;
    for (const auto& p : occupied) {
        for (const auto& capture : captures) {
            int nx = p.first + capture.first;
            int ny = p.second + capture.second;
            // Check if the captured cell is within the bounds of the grid
            if (0 <= nx && nx < N && 0 <= ny && ny < N) {
                free_cells--;
            }
        }
    }

    cout << free_cells << endl;
}
