#include <iostream>
#include <vector>
#include <set>
using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define all(x) (x).begin(), (x).end()
#define sz(x) ((int)(x).size())

const vector<pair<int, int>> capture_moves = {
    {0, 0}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1},
    {-1, -2}, {1, -2}, {2, -1}
};

int main() {
    int N, M;
    cin >> N >> M;

    set<pair<int, int>> occupied;

    rep(m, M) {
        int a, b;
        cin >> a >> b;
        occupied.insert({a - 1, b - 1});
    }

    long long empty_count = 0;

    // Calculate all empty cells outside the occupied ones
    rep(i, N) {
        rep(j, N) {
            bool can_place = true;
            for (auto& cap : capture_moves) {
                int x = i + cap.first, y = j + cap.second;
                if (x >= 0 && x < N && y >= 0 && y < N && occupied.count({x, y})) {
                    can_place = false;
                    break;
                }
            }
            if (can_place) {
                empty_count++;
            }
        }
    }

    cout << empty_count << endl;

    return 0;
}

This code iterates through each cell in the grid, checks whether it can be occupied by examining the capture range of existing pieces, and counts the cells that can be occupied. It's efficient and should handle the constraints well.
