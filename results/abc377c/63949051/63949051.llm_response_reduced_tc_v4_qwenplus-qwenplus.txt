```cpp
#include <iostream>
#include <vector>
#include <set>
#include <tuple>
#include <unordered_set>

using namespace std;

using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N;
    int M;
    cin >> N >> M;

    set<pair<i64, i64>> attacked_positions;
    vector<pair<i64, i64>> pieces(M);

    // Read all pieces
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    // For each piece, mark all positions it can attack
    const array<array<i64, 2>, 8> attack_offsets = {{
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    }};

    for (const auto& [x, y] : pieces) {
        // Add the position of the piece itself to avoid placing there
        attacked_positions.insert({x, y});

        // Add all positions that this piece can attack
        for (const auto& [dx, dy] : attack_offsets) {
            i64 nx = x + dx;
            i64 ny = y + dy;
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                attacked_positions.insert({nx, ny});
            }
        }
    }

    // Total number of cells is N*N minus the number of attacked positions
    cout << N * N - static_cast<i64>(attacked_positions.size()) << endl;

    return 0;
}
```
