```cpp
#include <iostream>
#include <vector>
#include <set>
#include <tuple>

using namespace std;

using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N;
    int M;
    cin >> N >> M;

    set<pair<i64, i64>> occupied;
    vector<pair<i64, i64>> pieces(M);

    for (int i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        occupied.insert({a, b});
    }

    // The 8 possible positions that can attack a piece
    const vector<pair<i64, i64>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (const auto& [a, b] : pieces) {
        for (const auto& [dx, dy] : directions) {
            i64 x = a + dx;
            i64 y = b + dy;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                occupied.insert({x, y});
            }
        }
    }

    // Total squares = N * N
    // Subtract number of occupied squares
    cout << (N * NLL - static_cast<i64>(occupied.size())) << "\n";

    return 0;
}
```
