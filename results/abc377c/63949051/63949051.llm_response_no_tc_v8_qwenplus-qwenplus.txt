#include <iostream>
#include <vector>
#include <set>
#include <utility>

using namespace std;

using i32 = int;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i32 N;
    i32 M;
    cin >> N >> M;

    vector<pair<i32, i32>> pieces(M);
    for (auto &p : pieces) {
        cin >> p.first >> p.second;
    }

    // Directions a piece can attack
    const vector<pair<i32, i32>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    set<pair<i32, i32>> attacked_or_occupied;

    // Add occupied positions and attacked positions
    for (const auto &[x, y] : pieces) {
        attacked_or_occupied.emplace(x, y);

        for (const auto &[dx, dy] : directions) {
            i32 nx = x + dx;
            i32 ny = y + dy;
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                attacked_or_occupied.emplace(nx, ny);
            }
        }
    }

    // Total number of safe cells: N*N - number of attacked or occupied cells
    i64 total_cells = static_cast<i64>(N) * static_cast<i64>(N);
    i64 result = total_cells - static_cast<i64>(attacked_or_occupied.size());

    cout << result << '\n';

    return 0;
}
