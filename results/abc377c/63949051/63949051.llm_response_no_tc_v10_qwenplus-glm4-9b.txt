    set<pair<i64, i64>> attackable;

    for (int i = 0; i < M; ++i) {
        int x = pieces[i].fi;
        int y = pieces[i].se;

        // 茫聛聯茫聜聦茫聜聣茫聛庐茫聝聻茫聜鹿茫聛聦氓聫聳茫聜聣茫聜聦茫聜聥氓聫炉猫聝陆忙聙搂茫聛聦茫聛聜茫聜聥

                ++count;
            }
        }
    }

    cout << count << endl;

    return 0;
