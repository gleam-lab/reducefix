#include <iostream>
#include <vector>
#include <set>
#include <tuple>

using namespace std;

using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    i64 N, M;
    cin >> N >> M;
    
    set<pair<i64, i64>> attacked;
    vector<pair<i64, i64>> pieces(M);
    
    for (auto &[a, b] : pieces) {
        cin >> a >> b;
        
        // Add current piece to attacked set to exclude it from being target
        attacked.insert({a, b});
        
        const int dx[] = {2, 1, -1, -2, -2, -1, 1, 2};
        const int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};
        
        for (int k = 0; k < 8; ++k) {
            i64 nx = a + dx[k];
            i64 ny = b + dy[k];
            
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                attacked.insert({nx, ny});
            }
        }
    }
    
    // Total squares = N * N
    // Subtract number of attacked squares and existing pieces
    cout << (N * NLL - static_cast<i64>(attacked.size())) << '\n';
    
    return 0;
}
