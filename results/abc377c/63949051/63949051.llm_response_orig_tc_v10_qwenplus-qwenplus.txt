```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using pll = pair<ll, ll>;
#define fi first
#define se second

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    set<pll> attacked;
    vector<pll> pieces(M);
    set<pll> piece_set;

    // Define the 8 possible knight move directions
    const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        piece_set.insert({a, b});
    }

    for (auto [x, y] : pieces) {
        for (int k = 0; k < 8; ++k) {
            ll nx = x + dx[k];
            ll ny = y + dy[k];
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                if (!piece_set.count({nx, ny})) {
                    attacked.insert({nx, ny});
                }
            }
        }
    }

    cout << (N * N - M - (ll)attacked.size()) << '\n';
}
```
