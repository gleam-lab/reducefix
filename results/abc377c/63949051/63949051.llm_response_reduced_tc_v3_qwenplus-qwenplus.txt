#include <iostream>
#include <vector>
#include <set>
#include <tuple>
#include <cmath>

using namespace std;

using i32 = int;
using i64 = long long;
using u64 = unsigned long long;

#define fi first
#define se second

template <typename T>
inline void out(T out_data, string end = "\n") {
    cout << out_data << end;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i32 N;
    i32 M;
    cin >> N >> M;

    vector<pair<i32, i32>> pieces(M);
    for (auto& p : pieces) {
        cin >> p.first >> p.second;
    }

    set<pair<i32, i32>> forbidden;

    // Define all 8 possible attack patterns
    const vector<pair<i32, i32>> attacks = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (const auto& p : pieces) {
        forbidden.insert(p); // The piece itself is not available
        
        for (const auto& offset : attacks) {
            i32 ni = p.first + offset.fi;
            i32 nj = p.second + offset.se;
            
            if (ni >= 1 && ni <= N && nj >= 1 && nj <= N) {
                forbidden.insert({ni, nj});
            }
        }
    }

    // Calculate total available cells: N*N - forbidden.size()
    cout << (1LL * N * N - forbidden.size()) << endl;

    return 0;
}
