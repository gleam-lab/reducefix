#include <iostream>
#include <vector>
#include <set>
#include <tuple>
#include <unordered_set>

using namespace std;

using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N;
    int M;
    cin >> N >> M;

    set<pair<i64, i64>> occupied;

    vector<pair<i64, i64>> pieces(M);
    for (auto &[a, b] : pieces) {
        cin >> a >> b;
        occupied.insert({a, b});
    }

    // The 8 possible attack positions relative to (i,j)
    const vector<pair<i64, i64>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    auto is_valid = [&](i64 x) { return 1 <= x && x <= N; };

    set<pair<i64, i64>> attacked;

    for (auto [a, b] : pieces) {
        for (auto [da, db] : directions) {
            i64 na = a + da;
            i64 nb = b + db;
            if (is_valid(na) && is_valid(nb)) {
                attacked.insert({na, nb});
            }
        }
    }

    // Remove positions that are already occupied by pieces
    for (auto pos : pieces) {
        attacked.erase(pos);
    }

    // Total number of safe cells = total cells - (occupied + attacked)
    i64 total_cells = N * N;
    i64 used = occupied.size() + attacked.size();

    cout << total_cells - used << endl;

    return 0;
}
