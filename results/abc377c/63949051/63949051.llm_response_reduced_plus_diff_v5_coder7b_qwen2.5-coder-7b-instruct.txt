```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e9 + 7;

bool valid(int n, int i, int j) {
    return i >= 1 && i <= n && j >= 1 && j <= n;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    unordered_set<long long> attacked;

    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        for (int dx = -2; dx <= 2; ++dx) {
            for (int dy = -2; dy <= 2; ++dy) {
                if (abs(dx) + abs(dy) == 3) {
                    int x = a + dx, y = b + dy;
                    if (valid(n, x, y)) {
                        attacked.insert((long long)x * n + y);
                    }
                }
            }
        }
    }

    long long total_squares = 1LL * n * n;
    long long safe_squares = total_squares - attacked.size();

    cout << safe_squares % MAXN << endl;

    return 0;
}
```
