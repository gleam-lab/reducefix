```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long n, m;
    cin >> n >> m;

    set<pair<long long, long long>> positions;
    for(long long i=0; i<m; ++i){
        long long x, y;
        cin >> x >> y;
        positions.insert({x, y});
        
        // Add all positions a knight could move to from this position
        if(x+2<=n && y+1<=n) positions.insert({x+2, y+1});
        if(x+1<=n && y+2<=n) positions.insert({x+1, y+2});
        if(x-1>=1   && y+2<=n) positions.insert({x-1, y+2});
        if(x-2>=1   && y+1<=n) positions.insert({x-2, y+1});
        if(x-2>=1   && y-1>=1) positions.insert({x-2, y-1});
        if(x-1>=1   && y-2>=1) positions.insert({x-1, y-2});
        if(x+1<=n   && y-2>=1) positions.insert({x+1, y-2});
        if(x+2<=n   && y-1>=1) positions.insert({x+2, y-1});
    }

    long long total_positions = n * n;
    long long result = total_positions - positions.size();

    cout << result << endl;

    return 0;
}
```
