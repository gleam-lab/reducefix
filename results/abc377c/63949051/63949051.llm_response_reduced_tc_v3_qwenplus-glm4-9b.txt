#include <iostream>
#include <vector>
#include <set>
using namespace std;

// Directions vectors for capturing pieces based on the current piece's position
const vector<pair<int, int>> moves = {
    {-2, 1}, {-1, 2}, {1, 2}, {2, 1},
    {2, -1}, {1, -2}, {-1, -2}, {-2, -1}
};

int main() {
    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    // Set to keep track of cells that cannot be placed
    set<int> unreachable;

    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= N; ++j) {
            for (const auto& move : moves) {
                int ni = i + move.first;
                int nj = j + move.second;
                if (ni >= 1 && ni <= N && nj >= 1 && nj <= N) {
                    if (find(pieces.begin(), pieces.end(), {ni, nj}) != pieces.end()) {
                        unreachable.insert(i * N + j);
                        break;
                    }
                }
            }
        }
    }

    // Total number of cells minus the unreachable cells
    cout << (N * N) - unreachable.size() << endl;

    return 0;
}
