#include <bits/stdc++.h>
using namespace std;

int main() {
    unordered_set<pair<long long, long long>> ex;
    long long n, m;
    cin >> n >> m;
    long long ans = n * n - m; // Start with all empty squares minus the existing ones
    vector<int> dx = {1, 1, -1, -1, 2, 2, -2, -2};
    vector<int> dy = {2, -2, 2, -2, 1, -1, 1, -1};

    for (int i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        pair<long long, long long> p = make_pair(a, b);

        if (!ex.count(p)) {
            ex.insert(p);

            for (int i = 0; i < 8; i++) {
                long long x, y;
                x = a + dx[i];
                y = b + dy[i];

                if (x > 0 && x <= n && y > 0 && y <= n) {
                    p = make_pair(x, y);
                    if (!ex.count(p)) {
                        ans--;
                    }
                }
            }
        }
    }

    cout << ans << endl;
}
