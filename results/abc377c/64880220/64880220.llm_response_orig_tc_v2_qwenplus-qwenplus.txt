```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Set to store all positions that are either occupied or threatened
    set<pair<long long, long long>> blocked;

    // All 8 possible L-shaped moves (knight moves)
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        long long x, y;
        cin >> x >> y;

        // Block the position itself
        blocked.emplace(x, y);

        // Block all threatened positions from this piece
        for (auto [dx, dy] : directions) {
            long long nx = x + dx;
            long long ny = y + dy;
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                blocked.emplace(nx, ny);
            }
        }
    }

    // Total number of cells: N * N
    // Subtract the number of blocked cells
    cout << N * N - (long long)blocked.size() << endl;

    return 0;
}
```
