#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

const int dx[8] = {1, 1, -1, -1, 2, 2, -2, -2};
const int dy[8] = {2, -2, 2, -2, 1, -1, 1, -1};

void calculateCaptureRange(int n, const pair<int, int> &p, set<pair<int, int>> &captureRange) {
    for (int i = 0; i < 8; ++i) {
        int x = p.first + dx[i];
        int y = p.second + dy[i];
        while (x >= 1 && x <= n && y >= 1 && y <= n) {
            captureRange.insert({x, y});
            x += dx[i];
            y += dy[i];
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;
    vector<pair<int, int>> pieces;
    pieces.reserve(m);
    set<pair<int, int>> captureRange;
    set<pair<int, int>> usedPositions;

    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        a--; b--;
        pieces.push_back({a, b});
        usedPositions.insert({a, b});

        // Calculate the capture range of each piece and store it
        calculateCaptureRange(n, {a, b}, captureRange);
    }

    // Calculate the number of valid empty positions
    set<pair<int, int>> validPositions;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (usedPositions.count({i - 1, j - 1}) == 0 && captureRange.count({i - 1, j - 1}) == 0) {
                validPositions.insert({i - 1, j - 1});
            }
        }
    }

    cout << validPositions.size() << endl;
    return 0;
}
