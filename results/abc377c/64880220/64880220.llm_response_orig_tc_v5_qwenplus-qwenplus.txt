#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Directions a knight can move in L-shapes (8 directions)
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    map<pair<long long, long long>, bool> occupiedOrAttacked;
    set<pair<long long, long long>> placedPieces;

    for(int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        pair<long long, long long> pos = {a, b};
        if(placedPieces.count(pos)) {
            continue; // Skip duplicates
        }
        placedPieces.insert(pos);
        occupiedOrAttacked[pos] = true;

        for(auto& dir : directions) {
            long long x = a + dir.first;
            long long y = b + dir.second;
            if(x >= 1 && x <= N && y >= 1 && y <= N) {
                occupiedOrAttacked[{x, y}] = true;
            }
        }
    }

    // Total number of squares
    __int128 total = (__int128)N * (__int128)N;
    
    // Subtract the number of attacked or occupied squares
    __int128 ans = total - (long long)occupiedOrAttacked.size();
    
    cout << ans << endl;

    return 0;
}
