#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    // Set to store all attacked positions and occupied positions
    set<pair<long long, long long>> occupied_or_attacked;

    // Directions a knight can move in L-shaped moves (8 possible)
    vector<pair<long long, long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        pair<long long, long long> pos = {a, b};
        // Mark this position as occupied
        occupied_or_attacked.insert(pos);

        // For each direction, mark the reachable square as attacked
        for (const auto& dir : directions) {
            long long x = a + dir.first;
            long long y = b + dir.second;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                occupied_or_attacked.insert({x, y});
            }
        }
    }

    // Total number of squares
    unsigned long long total_squares = static_cast<unsigned long long>(N) * static_cast<unsigned long long>(N);
    // Subtract number of occupied or attacked squares
    long long safe_squares = total_squares - static_cast<long long>(occupied_or_attacked.size());

    cout << safe_squares << endl;

    return 0;
}
