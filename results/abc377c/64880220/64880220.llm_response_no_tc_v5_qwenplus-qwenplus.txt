```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;

    // Set to store all positions that are either occupied or under threat
    unordered_set<string> threatened_or_occupied;

    // Function to hash a coordinate pair into a string for set storage
    auto hash_coord = [](long long x, long long y) {
        return to_string(x) + "," + to_string(y);
    };

    // List of relative move offsets corresponding to the 8 possible attack patterns
    vector<pair<long long, long long>> moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (long long i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        // Mark this position as occupied
        threatened_or_occupied.insert(hash_coord(a, b));

        // For each of the 8 L-shaped moves, mark the resulting square as threatened
        for (auto [dx, dy] : moves) {
            long long x = a + dx;
            long long y = b + dy;

            // Only consider coordinates within the grid
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                threatened_or_occupied.insert(hash_coord(x, y));
            }
        }
    }

    // Total number of squares: N*N
    // Subtract the number of occupied or threatened squares
    cout << (N * N - (long long)threatened_or_occupied.size()) << endl;

    return 0;
}
```
