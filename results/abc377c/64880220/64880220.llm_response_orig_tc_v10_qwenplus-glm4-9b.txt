#include <iostream>
#include <vector>
using namespace std;

// 忙拢聙忙聼楼莽聜鹿 (x, y) 忙聵炉氓聬娄氓聹篓忙拢聥莽聸聵
bool isInside(int x, int y, int n) {
    return x >= 1 && x <= n && y >= 1 && y <= n;
}

int main() {
    int n, m;
    cin >> n >> m;
    vector<pair<int, int>> pieces(n * n, {-1, -1}); // 氓颅聵氓聜篓 (氓聺聬忙聽聡, 忙聧聲猫聨路猫聦聝氓聸麓)
    vector<pair<int, int>> occupied; // 氓颅聵氓聜篓 occupied 盲陆聧莽陆庐

    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        --a; --b; // 猫陆卢忙聧垄盲赂潞0-based莽麓垄氓录聲
        occupied.push_back({a, b});

        // 忙聸麓忙聳掳忙聧聲猫聨路氓聦潞氓聼聼
        for (int dx = -2; dx <= 2; ++dx) {
            for (int dy = -2; dy <= 2; ++dy) {
                if (dx != 0 || dy != 0) {
                    int nx = a + dx;
                    int ny = b + dy;
                    if (isInside(nx, ny, n)) {
                        pieces[nx * n + ny] = {nx, ny};
                    }
                }
            }
        }
    }

    // 猫庐隆莽庐聴氓聫炉忙聰戮莽陆庐盲陆聧莽陆庐忙聲掳茅聡聫
    int count = 0;
    for (int i = 0; i < n * n; ++i) {
        int x = i / n + 1;
        int y = i % n + 1;
        if (find(occupied.begin(), occupied.end(), {x - 1, y - 1}) == occupied.end()) {
            bool canPlace = true;
            for (int dx = -2; dx <= 2; ++dx) {
                for (int dy = -2; dy <= 2; ++dy) {
                    if (dx != 0 || dy != 0) {
                        int nx = x + dx;
                        int ny = y + dy;
                        if (isInside(nx, ny, n) && pieces[nx * n + ny].first != -1) {
                            canPlace = false;
                            break;
                        }
                    }
                }
                if (!canPlace) break;
            }
            if (canPlace) count++;
        }
    }

    cout << count << endl;
    return 0;
}
