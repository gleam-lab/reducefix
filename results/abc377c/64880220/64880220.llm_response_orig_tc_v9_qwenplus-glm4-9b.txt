
The problem is asking to find the number of empty squares where you can place your piece such that it cannot be captured by any of the existing pieces on the grid. The grid is defined by `N` x `N` and there are `M` pieces already placed on the grid.

Here's a high-level approach to solving the problem:

1. **Determine Capture Ranges**: First, for each placed piece, determine the squares that can be captured based on the given rules. This involves checking all eight possible moves (2 steps and 1 step in a diagonal direction) from the current position and taking note of the squares that would be inside the grid if moved in each direction.

2. **Mark Capture Ranges**: Use a set or a 2D array to mark the squares that can be captured by each existing piece. This helps in identifying the squares that are dangerous or cannot be used to place a new piece.

3. **Count the Safe Squares**: For each square in the grid, check if it is in the set of captured squares. If not, increment the count of safe squares.

#include <vector>
#include <set>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    
    // Vector of pairs to store the positions of the existing pieces
    vector<pair<long long, long long>> pieces(m);
    
    // Set to store the positions of the squares that can be captured
    set<pair<long long, long long>> capturedSquares;
    
    // Indices to track the captured squares
    int captures[2 * n + 1][2 * n + 1] = {0};
    
    // Read the positions of the existing pieces
    for (long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        for (int j = 0; j < 8; ++j) {
            long long dx = 2 * (j % 2) - 1; // Moves: 2, 2, -2, -2, 1, 1, -1, -1
            long long dy = 2 * (j / 2) - 1; // Moves: 1, -1, 1, -1, 2, 2, -2, -2
            long long x = a + dx;
            long long y = b + dy;
            if (x > 0 && x <= n && y > 0 && y <= n) {
                captures[x][y]++;
                if (captures[x][y] == 1) { // First time capturing this square
                    capturedSquares.insert({x, y});
                }
            }
        }
    }
    
    // Count the number of safe squares
    long long safeSquares = 0;
    for (long long i = 1; i <= n; ++i) {
        for (long long j = 1; j <= n; ++j) {
            if (capturedSquares.find({i, j}) == capturedSquares.end()) {
                safeSquares++;
            }
        }
    }
    
    // Output the result
    cout << safeSquares << endl;
    
    return 0;
}
