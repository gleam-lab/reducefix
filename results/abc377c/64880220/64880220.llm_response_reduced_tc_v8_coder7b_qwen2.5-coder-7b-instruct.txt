#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    unordered_set<long long> ex;
    long long ans = n * n;
    vector<int> dx = {1, 1, -1, -1, 2, 2, -2, -2};
    vector<int> dy = {2, -2, 2, -2, 1, -1, 1, -1};

    for (long long i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        long long key = (a - 1) * n + b;
        if (!ex.count(key)) {
            ex.insert(key);
            for (int d = 0; d < 8; d++) {
                long long x = a + dx[d], y = b + dy[d];
                if (x > 0 && x <= n && y > 0 && y <= n) {
                    long long neighborKey = (x - 1) * n + y;
                    if (!ex.count(neighborKey)) {
                        ans--;
                    }
                }
            }
        }
    }

    cout << ans << endl;
}
