```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);

    int n, m;
    cin >> n >> m;
    
    set<pair<int, int>> ex;
    for(int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        ex.insert({a, b});
    }

    int ans = 0;
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= n; ++j) {
            if(ex.find({i, j}) != ex.end()) continue;
            bool canPlace = true;
            vector<vector<int>> dir = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
            for(auto& d : dir) {
                int x = i + d[0], y = j + d[1];
                if(x >= 1 && x <= n && y >= 1 && y <= n && ex.find({x, y}) != ex.end()) {
                    canPlace = false;
                    break;
                }
            }
            if(canPlace) ans++;
        }
    }
    
    cout << ans << endl;
}
```
