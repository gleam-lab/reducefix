```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    map<pair<int, int>, int> ex;
    long long n, m;
    cin >> n >> m;
    long long ans = 0;
    vector<int> dx = {2, 1, -1, -2, -2, -1, 1, 2};
    vector<int> dy = {1, 2, 2, 1, -1, -2, -2, -1};
    for (int i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        pair<int, int> p = make_pair(a, b);
        if (!ex.count(p)) {
            ex[p] = 1;
            ans++;
        }
        for (int j = 0; j < 8; ++j) {
            long long x = a + dx[j];
            long long y = b + dy[j];
            pair<int, int> np = make_pair(x, y);
            if (x >= 1 && x <= n && y >= 1 && y <= n) {
                if (!ex.count(np)) {
                    ex[np] = 1;
                    ans++;
                }
            }
        }
    }
    cout << n * n - ans << endl;
    return 0;
}
```

**Explanation of Changes:**

1. **Direction Vectors (dx and dy):** The original code had the displacement vectors swapped (dx represented column changes, dy row changes). This was corrected by swapping the vectors to match the problem's conditions.

2. **Grid Boundary Check:** The condition was incorrectly using `< n` instead of `<= n`, which excluded edge squares. It was fixed to check `x >= 1 && x <= n` and similarly for `y`.

3. **Data Types:** The coordinates are stored as `int` in the map, which is safe given the constraints (coordinates up to \(10^9\) fit in 32-bit integers).

These changes ensure that all capture squares are correctly identified and counted, leading to the correct calculation of allowed squares.
