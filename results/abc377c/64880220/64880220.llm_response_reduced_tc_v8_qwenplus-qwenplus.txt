```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N, M;
    cin >> N >> M;

    // Set to store all blocked positions (either because they contain a piece or can be captured)
    unordered_set<long long> blocked;

    // Directions from which this piece can attack
    vector<pair<long long, long long>> attack_offsets = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (long long i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        // Block the current position
        blocked.insert(a * N + b); // Unique key for position (a,b)

        // Block all positions that this piece can attack
        for (auto [dx, dy] : attack_offsets) {
            long long x = a + dx;
            long long y = b + dy;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                blocked.insert(x * N + y);
            }
        }
    }

    // Total number of cells is N*N
    // Subtract number of blocked cells to get answer
    cout << (N * N - (long long)blocked.size()) << '\n';

    return 0;
}
```
