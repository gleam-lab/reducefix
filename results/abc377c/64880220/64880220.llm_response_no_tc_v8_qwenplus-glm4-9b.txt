#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    long long ans = n * n; // Start with total number of cells

    vector<pair<long long, long long>> pieces;
    pieces.resize(m);

    // Read the positions of the pieces
    for (long long i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
    }

    // For each piece, determine the range of cells it can attack
    for (const auto& piece : pieces) {
        long long a = piece.first;
        long long b = piece.second;

        // Calculate the bounds for the cells this piece can attack
        long long left = max(1LL, b - 2);
        long long right = min(n, b + 2);
        long long top = max(1LL, a - 2);
        long long bottom = min(n, a + 2);

        // If the piece is attacking any cell that is not on the border, it can attack up to 4 cells
        if (left < right && top < bottom) {
            ans -= 4; // Subtract the 4 cells that this piece can attack
        }
    }

    cout << ans << endl; // Output the number of cells available to place a new piece
}
