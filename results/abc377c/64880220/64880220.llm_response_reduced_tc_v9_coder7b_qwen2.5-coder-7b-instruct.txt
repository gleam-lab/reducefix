```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    long long n, m;
    cin >> n >> m;
    
    set<pair<long long, long long>> ex;
    for(long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        ex.insert({a, b});
    }

    vector<int> dx = {1, 1, -1, -1, 2, 2, -2, -2};
    vector<int> dy = {2, -2, 2, -2, 1, -1, 1, -1};

    set<pair<long long, long long>> reachable;
    for(auto& [x, y] : ex) {
        for(int i = 0; i < 8; ++i) {
            long long nx = x + dx[i], ny = y + dy[i];
            if(nx >= 1 && nx <= n && ny >= 1 && ny <= n) {
                reachable.insert({nx, ny});
            }
        }
    }

    cout << n * n - ex.size() - reachable.size() << endl;
    return 0;
}
```
