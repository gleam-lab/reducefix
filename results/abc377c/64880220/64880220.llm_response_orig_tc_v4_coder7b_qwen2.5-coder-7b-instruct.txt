```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    
    // Initialize a set to store positions of occupied squares
    set<pair<long long, long long>> occupied;
    for(long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        occupied.insert({a, b});
    }

    // Directions for all possible moves
    vector<vector<long long>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    long long valid_squares = 0;

    // Iterate over each cell in the grid
    for(long long i = 1; i <= n; ++i) {
        for(long long j = 1; j <= n; ++j) {
            bool can_place = true;
            // Check all possible moves from the current cell
            for(auto &dir : directions) {
                long long ni = i + dir[0], nj = j + dir[1];
                if(ni >= 1 && ni <= n && nj >= 1 && nj <= n && occupied.find({ni, nj}) != occupied.end()) {
                    can_place = false;
                    break;
                }
            }
            // If no move threatens this cell, it's a valid position to place a piece
            if(can_place) {
                valid_squares++;
            }
        }
    }

    cout << valid_squares << endl;
    return 0;
}
```
