#include <bits/stdc++.h>
using namespace std;

int main() {
    map<pair<long long, long long>, int> ex;
    long long n, m;
    cin >> n >> m;
    long long ans = 0;
    vector<int> dx = {1, 1, -1, -1, 2, 2, -2, -2};
    vector<int> dy = {2, -2, 2, -2, 1, -1, 1, -1};

    // Initialize the count of valid positions
    ans = n * n;

    for (int i = 0; i < m; i++) {
        long long a, b;
        cin >> a >> b;
        pair<int, int> p;
        p = make_pair(a, b);

        // Decrease the count for each position that can be captured
        for (int i = 0; i < 8; i++) {
            long long x, y;
            x = a + dx[i];
            y = b + dy[i];
            p = make_pair(x, y);
            if (x > 0 && x <= n && y > 0 && y <= n) {
                if (!ex.count(p)) {
                    ex[p] = 1;
                    ans--;
                }
            }
        }
    }

    cout << ans << endl;
    return 0;
}
