            long long y = b + dy[j];
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                unsafe_squares.insert({x, y});
            }
        }
    }
    
    long long total_safe = N * N - existing_pieces.size() - unsafe_squares.size();
    for (const auto& piece : existing_pieces) {
        if (unsafe_squares.count(piece)) {
            total_safe++;
        }
    }
    
    cout << max(0LL, total_safe) << endl;
    
    return 0;
}
