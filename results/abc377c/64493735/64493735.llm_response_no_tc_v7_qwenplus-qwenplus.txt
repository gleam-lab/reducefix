#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int64_t N;
    int M;
    cin >> N >> M;

    unordered_set<int64_t> attacked_or_occupied;
    // We'll use a hash to represent (a, b) as a single integer: a * P + b
    // where P is a prime larger than N to avoid collisions
    const int64_t P = 1000000007;

    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;

        // Mark current piece position
        attacked_or_occupied.insert(a * P + b);

        // Generate all 8 possible L-shaped attack moves
        vector<pair<int64_t, int64_t>> directions = {
            {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
            {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
        };

        for (const auto& [dx, dy] : directions) {
            int64_t nx = a + dx;
            int64_t ny = b + dy;
            if (1 <= nx && nx <= N && 1 <= ny && ny <= N) {
                attacked_or_occupied.insert(nx * P + ny);
            }
        }
    }

    // Total number of cells is N*N
    // Subtract the number of unique attacked or occupied cells
    int64_t total_cells = N * N;
    int64_t forbidden_cells = attacked_or_occupied.size();

    cout << (total_cells - forbidden_cells) << '\n';

    return 0;
}
