#include <iostream>
#include <set>
#include <vector>

using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    set<pair<int, int>> occupied;
    for (int i = 0; i < M; i++) {
        int x, y;
        cin >> x >> y;
        occupied.insert({x, y});
    }

    long long possible_positions = (long long)N * N;
    for (const auto& pos : occupied) {
        int a = pos.first, b = pos.second;
        // Check the positions that can be captured in the pattern described.
        vector<pair<int, int>> capturing_positions = {
            {a + 2, b + 1}, {a + 1, b + 2}, {a - 1, b + 2}, {a - 2, b + 1},
            {a - 2, b - 1}, {a - 1, b - 2}, {a + 1, b - 2}, {a + 2, b - 1}
        };

        for (const auto& pos : capturing_positions) {
            int x = pos.first, y = pos.second;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                possible_positions--;
            }
        }
    }

    cout << possible_positions << endl;

    return 0;
}
