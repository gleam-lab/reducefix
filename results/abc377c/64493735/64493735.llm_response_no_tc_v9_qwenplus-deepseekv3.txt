```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    long long N;
    int M;
    cin >> N >> M;
    
    set<pair<int, int>> forbidden;
    
    for (int i = 0; i < M; i++) {
        int a, b;
        cin >> a >> b;
        forbidden.insert({a, b});
        
        // Check all 8 possible knight moves
        int dx[] = {2, 1, -1, -2, -2, -1, 1, 2};
        int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};
        
        for (int k = 0; k < 8; k++) {
            int ni = a + dx[k];
            int nj = b + dy[k];
            if (1 <= ni && ni <= N && 1 <= nj && nj <= N) {
                forbidden.insert({ni, nj});
            }
        }
    }
    
    cout << N * N - forbidden.size() << '\n';
    
    return 0;
}
```

Key improvements:
1. Simplified the code structure by removing unnecessary class and functions
2. Fixed the knight moves checking by using dx/dy arrays for cleaner code
3. Properly handles large N by using long long for the total squares calculation
4. Uses standard C++ input/output which is simpler and more reliable
5. The set now correctly tracks all forbidden positions (occupied squares and their attack positions)
6. The output is the total squares minus the forbidden squares count

This solution efficiently handles the constraints and provides the correct count of safe squares to place the piece.
