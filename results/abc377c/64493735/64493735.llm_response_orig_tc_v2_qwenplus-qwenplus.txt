```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N;
    int M;
    cin >> N >> M;

    // Set to store all forbidden squares (including original pieces and attackable squares)
    set<pair<int64_t, int64_t>> forbidden;

    // Directions representing the 8 possible L-shaped moves a knight can make
    const vector<pair<int64_t, int64_t>> directions = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    // Read each piece's position and mark all forbidden squares
    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;

        // Mark current piece
        forbidden.emplace(a, b);

        // Check all 8 directions
        for (const auto& [dx, dy] : directions) {
            int64_t na = a + dx;
            int64_t nb = b + dy;

            // Only add valid positions within bounds
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                forbidden.emplace(na, nb);
            }
        }
    }

    // Total number of cells is N*N
    // Subtract the number of forbidden cells
    cout << (N * N - static_cast<int64_t>(forbidden.size())) << "\n";

    return 0;
}
```
