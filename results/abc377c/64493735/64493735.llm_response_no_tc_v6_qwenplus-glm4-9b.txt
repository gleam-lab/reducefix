#include <iostream>
#include <vector>

using namespace std;

#define MAX_N 1000000000

vector<vector<int>> valid_moves = {
    {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
    {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
};

int main() {
    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for (int i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    long long total_empty_cells = N * N;
    long long captured_cells = 0;

    for (int i = 0; i < M; ++i) {
        int x = pieces[i].first;
        int y = pieces[i].second;

        for (int dx : valid_moves) {
            int nx = x + dx[0];
            int ny = y + dx[1];

            if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                // Check if the cell to be captured is not already captured by another piece
                bool already_captured = false;
                for (int j = 0; j < M; ++j) {
                    if (i != j && abs(pieces[j].first - nx) <= 2 && abs(pieces[j].second - ny) <= 2) {
                        already_captured = true;
                        break;
                    }
                }
                
                if (!already_captured) {
                    captured_cells++;
                    for (int dx : valid_moves) {
                        int ex = nx + dx[0];
                        int ey = ny + dx[1];
                        if (ex >= 1 && ex <= N && ey >= 1 && ey <= N && (ex == x || ey == y)) {
                            captured_cells++;
                        }
                    }
                }
            }
        }
    }

    cout << total_empty_cells - captured_cells << endl;

    return 0;
}
