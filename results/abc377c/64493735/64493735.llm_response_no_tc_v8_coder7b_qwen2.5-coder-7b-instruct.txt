#include <iostream>
#include <set>
#include <cmath>
using namespace std;

typedef long long ll;

class Solution {
public:
    ll solve(ll N, ll M, vector<ll> &a, vector<ll> &b) {
        // Calculate the total number of cells
        ll total_cells = N * N;

        // Set to store all cells where pieces can be placed
        set<pair<ll, ll>> safe_cells;

        // Add all empty cells to the set
        for (ll i = 1; i <= N; ++i) {
            for (ll j = 1; j <= N; ++j) {
                safe_cells.insert({i, j});
            }
        }

        // Remove cells that are under attack
        for (ll k = 0; k < M; ++k) {
            ll x = a[k], y = b[k];
            for (ll dx = -2; dx <= 2; ++dx) {
                for (ll dy = -2; dy <= 2; ++dy) {
                    if (abs(dx) == 2 && abs(dy) == 1 || abs(dx) == 1 && abs(dy) == 2) {
                        ll nx = x + dx, ny = y + dy;
                        if (nx >= 1 && nx <= N && ny >= 1 && ny <= N) {
                            safe_cells.erase({nx, ny});
                        }
                    }
                }
            }
        }

        // The result is the number of safe cells
        return safe_cells.size();
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll N, M;
    cin >> N >> M;
    vector<ll> a(M), b(M);
    for (ll i = 0; i < M; ++i) {
        cin >> a[i] >> b[i];
    }

    Solution sol;
    cout << sol.solve(N, M, a, b) << endl;

    return 0;
}
