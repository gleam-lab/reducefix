```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N;
    int M;
    cin >> N >> M;

    unordered_set<int64_t> attacked;
    unordered_set<int64_t> placed;
    
    // Directions for attack positions
    const array<array<int64_t, 2>, 8> directions = {{
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    }};

    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;
        placed.insert(a * N + b); // Store placed positions

        // Mark all positions this piece can attack
        for (const auto& dir : directions) {
            int64_t na = a + dir[0];
            int64_t nb = b + dir[1];
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                attacked.insert(na * N + nb);
            }
        }
    }

    // Remove already placed pieces from attacked to avoid double counting
    for (const auto& pos : placed) {
        attacked.erase(pos);
    }

    // Total available positions = total squares - placed - attacked
    int64_t total = N * N;
    int64_t result = total - placed.size() - attacked.size();
    cout << result << endl;

    return 0;
}
```
