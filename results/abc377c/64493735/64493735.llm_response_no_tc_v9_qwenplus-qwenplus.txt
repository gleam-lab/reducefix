#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N, M;
    cin >> N >> M;

    unordered_set<int64_t> attacked;
    unordered_set<int64_t> pieces;

    // Directions for the knight-like moves (8 directions)
    const vector<pair<int64_t, int64_t>> dirs = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int64_t i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;

        // Store current piece to avoid placing on it
        pieces.insert(a * 1000000000 + b);

        // Mark all reachable squares that can be attacked
        for (auto [dx, dy] : dirs) {
            int64_t na = a + dx;
            int64_t nb = b + dy;
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                int64_t key = na * 1000000000 + nb;
                if (pieces.find(key) == pieces.end()) {
                    attacked.insert(key);
                }
            }
        }
    }

    // Now calculate how many empty and non-attacked squares are available
    int64_t total_squares = N * N;
    int64_t blocked = pieces.size() + attacked.size();

    cout << (total_squares - blocked) << endl;

    return 0;
}
