#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;

    set<pair<int, int>> pieces;
    for (int i = 0; i < M; i++) {
        int x, y;
        cin >> x >> y;
        pieces.insert({x, y});
    }

    long long safe_spaces = N * N;
    for (auto &p : pieces) {
        int x = p.first, y = p.second;
        // Check if the piece at (x, y) can be captured by any of its neighbors
        if (x + 2 <= N && y + 1 <= N && pieces.count({x + 2, y + 1})) safe_spaces--;
        if (x + 1 <= N && y + 2 <= N && pieces.count({x + 1, y + 2})) safe_spaces--;
        if (x - 1 <= N && y + 2 <= N && pieces.count({x - 1, y + 2})) safe_spaces--;
        if (x - 2 <= N && y + 1 <= N && pieces.count({x - 2, y + 1})) safe_spaces--;
        if (x - 2 <= N && y - 1 <= N && pieces.count({x - 2, y - 1})) safe_spaces--;
        if (x - 1 <= N && y - 2 <= N && pieces.count({x - 1, y - 2})) safe_spaces--;
        if (x + 1 <= N && y - 2 <= N && pieces.count({x + 1, y - 2})) safe_spaces--;
        if (x + 2 <= N && y - 1 <= N && pieces.count({x + 2, y - 1})) safe_spaces--;
    }

    cout << safe_spaces << endl;
    return 0;
}
