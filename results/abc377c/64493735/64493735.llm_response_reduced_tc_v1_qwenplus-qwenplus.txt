```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    long long N, M;
    cin >> N >> M;
    
    unordered_set<long long> attacked;
    auto hash = [&](int x, int y) {
        return x * 1000000000LL + y;
    };
    
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        
        // Mark the current square as occupied
        attacked.insert(hash(a, b));
        
        // Define all 8 possible attack positions
        vector<pair<int, int>> moves = {
            {a + 2, b + 1},
            {a + 1, b + 2},
            {a - 1, b + 2},
            {a - 2, b + 1},
            {a - 2, b - 1},
            {a - 1, b - 2},
            {a + 1, b - 2},
            {a + 2, b - 1}
        };
        
        // Mark all valid attack positions
        for (auto [x, y] : moves) {
            if (1 <= x && x <= N && 1 <= y && y <= N) {
                attacked.insert(hash(x, y));
            }
        }
    }
    
    // Total number of cells is N*N
    // Subtract the number of attacked or occupied cells
    cout << (N * NLL) - static_cast<long long>(attacked.size()) << '\n';
    
    return 0;
}
```
