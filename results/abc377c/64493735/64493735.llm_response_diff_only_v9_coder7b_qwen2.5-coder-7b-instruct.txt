#include <bits/stdc++.h>
using namespace std;
#define ll long long

class Solution {
public:
    ll solve(ll N, ll M, vector<ll>& a, vector<ll>& b) {
        set<pair<ll, ll>> pieces;
        for (ll i = 0; i < M; ++i) {
            pieces.insert({a[i], b[i]});
        }

        ll count = 0;
        for (ll i = 1; i <= N; ++i) {
            for (ll j = 1; j <= N; ++j) {
                if (pieces.count({i, j})) continue;

                bool can_place = true;
                vector<pair<ll, ll>> moves = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
                for (const auto& move : moves) {
                    ll ni = i + move.first, nj = j + move.second;
                    if (ni >= 1 && ni <= N && nj >= 1 && nj <= N && pieces.count({ni, nj})) {
                        can_place = false;
                        break;
                    }
                }

                if (can_place) ++count;
            }
        }

        return count;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Solution sol;
    ll N, M;
    cin >> N >> M;
    vector<ll> a(M), b(M);
    for (ll i = 0; i < M; ++i) {
        cin >> a[i] >> b[i];
    }

    cout << sol.solve(N, M, a, b) << endl;

    return 0;
}
