```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int64_t N;
    int M;
    cin >> N >> M;

    // Set to store all positions that are either occupied or under attack
    unordered_set<int64_t> attacked;

    // Directions where a piece at (i,j) can attack
    const array<array<int, 2>, 8> directions = {{
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    }};

    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;

        // Mark current piece position
        attacked.insert(a * N + b);

        // For each direction, calculate the attacked square
        for (const auto& dir : directions) {
            int64_t ni = a + dir[0];
            int64_t nj = b + dir[1];

            if (1 <= ni && ni <= N && 1 <= nj && nj <= N) {
                attacked.insert(ni * N + nj);
            }
        }
    }

    // Total number of squares is N*N minus the number of attacked or occupied squares
    int64_t totalSafeSquares = N * N - static_cast<int64_t>(attacked.size());

    cout << totalSafeSquares << endl;

    return 0;
}
```
