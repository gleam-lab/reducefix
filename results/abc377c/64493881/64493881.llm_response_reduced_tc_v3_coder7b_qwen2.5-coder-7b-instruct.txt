class Solution {
public:
    int countSquares(int n, int m, vector<int>& a, vector<int>& b) {
        unordered_set<int> st;
        for (int i = 0; i < m; i++) {
            int u = a[i], v = b[i];
            for (int k = 0; k < 8; k++) {
                int x = u + dx[k], y = v + dy[k];
                if (x >= 1 && x <= n && y >= 1 && y <= n) st.insert(x * n + y);
            }
        }
        return n * n - st.size();
    }
};
