        int dx[] = {2, 1, -1, -2, -2, -1, 1, 2};

            int y = b + dy[k];
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                attacked.insert({x, y});
            }
        }
    }
    
    cout << N * N - attacked.size() << '\n';
    
    return 0;
}

5. **Output**: The result is printed, representing the number of safe squares where the new piece can be placed without being captured.
