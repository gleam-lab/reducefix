```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    unordered_set<long long> attacked;
    auto pack = [&](int x, int y) {
        return (long long)x * 1000000000LL + y;
    };

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;

        // Define all 8 attack positions relative to (a,b)
        const int dx[] = {2, 1, -1, -2, -2, -1, 1, 2};
        const int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};

        for (int j = 0; j < 8; ++j) {
            int na = a + dx[j];
            int nb = b + dy[j];
            if (1 <= na && na <= N && 1 <= nb && nb <= N) {
                attacked.insert(pack(na, nb));
            }
        }
    }

    // Total cells - already occupied cells - attacked cells
    cout << (N * N - M - (long long)attacked.size()) << "\n";

    return 0;
}
```
