```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64_t N;
    int M;
    cin >> N >> M;

    set<pair<int64_t, int64_t>> attacked_positions;
    vector<pair<int64_t, int64_t>> knight_moves = {
        {2, 1}, {1, 2}, {-1, 2}, {-2, 1},
        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}
    };

    for (int i = 0; i < M; ++i) {
        int64_t a, b;
        cin >> a >> b;
        attacked_positions.insert({a, b});

        for (const auto& move : knight_moves) {
            int64_t ni = a + move.first;
            int64_t nj = b + move.second;
            if (1 <= ni && ni <= N && 1 <= nj && nj <= N) {
                attacked_positions.insert({ni, nj});
            }
        }
    }

    // Total squares: N*N
    // Subtract the number of attacked positions
    cout << (static_cast<int64_t>(N) * N - static_cast<int64_t>(attacked_positions.size())) << endl;

    return 0;
}
```
