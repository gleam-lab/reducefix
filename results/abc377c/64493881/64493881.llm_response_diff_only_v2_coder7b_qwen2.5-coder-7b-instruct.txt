#include <bits/stdc++.h>
using namespace std;
#define v(type) vector<type>
#define p(ate,bte) pair<ate,bte>
#define m(ate,bte) map<ate,bte>
#define s(type) set<type>
class Sub {
public:
    
int solve() {
    long long N, M;
    cin >> N >> M;

    s(p(long long, long long)) pieces;
    for (long long i = 0; i < M; i++) {
        long long a, b;
        cin >> a >> b;
        pieces.emplace(a, b);

        // Check all possible positions a knight can move to
        long long moves[8][2] = {{2, 1}, {1, 2}, {-1, 2}, {-2, 1}, {-2, -1}, {-1, -2}, {1, -2}, {2, -1}};
        for (auto &move : moves) {
            long long x = a + move[0];
            long long y = b + move[1];
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                pieces.emplace(x, y);
            }
        }
    }

    // The total number of squares minus the number of squares occupied by pieces
    cout << N * N - pieces.size();

    return 0;
};

private:
// Additional helper functions can be added here if needed
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    Sub *sub = new Sub();
    int back = sub->solve();
    delete sub;
    return back;
}
