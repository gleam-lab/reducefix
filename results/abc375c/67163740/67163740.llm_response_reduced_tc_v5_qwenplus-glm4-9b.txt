
```cpp
#include <iostream>
#include <cmath>
#include <vector>
#include <iomanip>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<pair<int, int>> points(N + 1);
    points[0].first = 0;
    points[0].second = 0;

    for (int i = 1; i <= N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_distance = 0.0;
    for (int i = 1; i <= N; ++i) {
        total_distance += sqrt(pow(points[i].first - points[i - 1].first, 2) + pow(points[i].second - points[i - 1].second, 2));
    }

    // Calculate the distance from the last point back to the origin
    total_distance += sqrt(pow(points[N].first, 2) + pow(points[N].second, 2));

    // Print the total distance with a precision of 20 decimal places
    cout << fixed << setprecision(20) << total_distance << endl;

    return 0;
}
```
