
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
using namespace std;

using ll = long long;

pair<ll, ll> read_point() {
    ll x, y;
    cin >> x >> y;
    return {x, y};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    ll n;
    cin >> n;

    vector<pair<ll, ll>> points(n + 1);
    points[0] = {0, 0}; // Start at origin

    for (ll i = 1; i <= n; i++) {
        points[i] = read_point();
    }

    double total_cost = 0.0;
    for (ll i = 1; i <= n; i++) {
        total_cost += sqrt(pow(points[i].first - points[i - 1].first, 2) + pow(points[i].second - points[i - 1].second, 2));
    }

    // Return to the origin
    total_cost += sqrt(pow(points[n].first - points[0].first, 2) + pow(points[n].second - points[0].second, 2));

    cout << fixed << setprecision(10) << total_cost << endl;

    return 0;
}
```
