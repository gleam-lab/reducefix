#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    int n;
    cin >> n;
    double x = 0, y = 0, total_cost = 0.0;

    for (int i = 0; i < n; ++i) {
        int a, b;
        cin >> a >> b;
        total_cost += sqrt((x - a) * (x - a) + (y - b) * (y - b));
        x = a;
        y = b;
    }

    // Add the cost of the final return to the origin
    total_cost += sqrt(x * x + y * y);

    cout << fixed << setprecision(10) << total_cost << endl;

    return 0;
}
