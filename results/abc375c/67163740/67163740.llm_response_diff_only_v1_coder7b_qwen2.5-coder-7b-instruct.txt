```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long double; // Changed float to long double for higher precision

void solve() {
    int n;
    ll x = 0, y = 0, t = 0;
    cin >> n;
    while (n--) {
        ll a, b;
        cin >> a >> b;
        t += sqrtl((x - a) * (x - a) + (y - b) * (y - b)); // Used sqrtl for long double
        x = a;
        y = b;
    }
    t += sqrtl(x * x + y * y); // Added missing calculation for returning to origin
    cout << setprecision(20) << t << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    cout << endl;
    return 0;
}
```
