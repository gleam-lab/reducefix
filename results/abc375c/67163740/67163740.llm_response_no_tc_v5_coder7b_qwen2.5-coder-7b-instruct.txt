```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long double; // Change float to long double for higher precision

void solve() {
    int n;
    ll x = 0, y = 0, t = 0; // Initialize with long double for high precision
    cin >> n;
    while (n--) {
        ll a, b;
        cin >> a >> b;
        t += sqrtl((x - a) * (x - a) + (y - b) * (y - b)); // Use sqrtl for long double
        x = a;
        y = b;
    }
    t += sqrtl(x * x + y * y); // Calculate distance back to origin using sqrtl
    cout << fixed << setprecision(20) << t << endl; // Set precision for output
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    cout << endl;
    return 0;
}
```
