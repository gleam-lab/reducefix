
```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

struct Point {
    long long x, y;
};

struct Edge {
    Point from, to;
    double cost;

    Edge(Point f, Point t, double c) : from(f), to(t), cost(c) {}
};

bool compareEdge(const Edge& a, const Edge& b) {
    return a.cost < b.cost;
}

int main() {
    int n;
    cin >> n;
    vector<Point> points(n + 1);
    points[0] = {0, 0};
    
    for (int i = 1; i <= n; ++i) {
        cin >> points[i].x >> points[i].y;
    }

    vector<Edge> edges;
    for (int i = 1; i <= n; ++i) {
        double cost = sqrt((points[i].x - points[0].x) * (points[i].x - points[0].x) + (points[i].y - points[0].y) * (points[i].y - points[0].y));
        edges.emplace_back(points[0], points[i], cost);
        cost = sqrt((points[i].x - points[n + 1].x) * (points[i].x - points[n + 1].x) + (points[i].y - points[n + 1].y) * (points[i].y - points[n + 1].y));
        edges.emplace_back(points[i], points[n + 1], cost);
    }

    // Sort edges by cost
    sort(edges.begin(), edges.end(), compareEdge);

    double totalCost = 0;
    vector<bool> used(n + 2, false);
    for (auto& edge : edges) {
        if ((used[edge.from.x + edge.from.y] == false) && (used[edge.to.x + edge.to.y] == false) && (edge.from != edge.to)) {
            totalCost += edge.cost;
            used[edge.from.x + edge.from.y] = true;
            used[edge.to.x + edge.to.y] = true;
        }
    }

    cout << fixed << setprecision(10) << totalCost << endl;

    return 0;
}
```
