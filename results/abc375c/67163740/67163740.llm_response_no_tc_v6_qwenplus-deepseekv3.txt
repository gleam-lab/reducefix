This can cause the calculated distances to be inaccurate, especially when squared differences are involved.

To fix this, we should use `double` instead of `float` for higher precision (about 15-16 decimal digits), which is crucial for maintaining accuracy when dealing with large numbers and accumulating many distance calculations.

using namespace std;
using ll = long long;

void solve() {

    cin >> n;
    while (n--) {
        double a, b;
        cin >> a >> b;
        t += sqrt((x - a) * (x - a) + (y - b) * (y - b));
        x = a;
        y = b;
    }
    t += sqrt(x * x + y * y);
    cout << fixed << setprecision(10) << t << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    return 0;
}
