#include <bits/stdc++.h>
using namespace std;
using ll = long double; // Use long double for higher precision

void solve() {
    int n;
    ll x = 0, y = 0, t = 0;
    cin >> n;
    while (n--) {
        ll a, b;
        cin >> a >> b;
        t += sqrtl((x - a) * (x - a) + (y - b) * (y - b)); // Use sqrtl for long double
        x = a;
        y = b;
    }
    t += sqrtl(x * x + y * y); // Calculate distance back to origin
    cout << setprecision(20) << t << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
    cout << endl;
    return 0;
}

It switches to `long double` for better precision, ensuring that the distances calculated accurately reflect the Euclidean distance between points. The rest of the logic remains unchanged, focusing on iterating through each point, calculating the distance moved, and summing these distances to find the total cost of the journey.
