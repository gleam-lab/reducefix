```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    long long x = 0, y = 0; // Use long long to prevent overflow
    double total_cost = 0.0;
    
    for (int i = 0; i < n; ++i) {
        long long new_x, new_y;
        cin >> new_x >> new_y;
        
        long long dx = x - new_x;
        long long dy = y - new_y;
        total_cost += sqrt(static_cast<double>(dx * dx + dy * dy));
        
        x = new_x;
        y = new_y;
    }
    
    // Return to origin
    total_cost += sqrt(static_cast<double>(x * x + y * y));
    
    cout << fixed << setprecision(15) << total_cost << endl;
    
    return 0;
}
```
