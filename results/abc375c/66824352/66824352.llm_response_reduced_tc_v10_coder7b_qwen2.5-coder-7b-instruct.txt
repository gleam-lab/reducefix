```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<pair<int, int>> points(n);
    for(int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0;
    // Start from origin
    int current_x = 0, current_y = 0;
    
    for(const auto& point : points) {
        total_cost += sqrt(pow(current_x - point.first, 2) + pow(current_y - point.second, 2));
        current_x = point.first;
        current_y = point.second;
    }

    // Return to origin
    total_cost += sqrt(pow(current_x, 2) + pow(current_y, 2));

    cout << setprecision(10) << total_cost << endl;
    return 0;
}
```
