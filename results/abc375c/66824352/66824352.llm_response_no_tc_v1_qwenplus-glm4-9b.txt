#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;

    double current_x = 0, current_y = 0;
    double total_cost = 0.0;

    while (n--) {
        int x, y;
        cin >> x >> y;
        total_cost += sqrt((current_x - x) * (current_x - x) + (current_y - y) * (current_y - y));
        current_x = x;
        current_y = y;
    }

    // Add the cost of returning to the origin
    total_cost += sqrt(current_x * current_x + current_y * current_y);

    cout << fixed << setprecision(10) << total_cost << endl;

    return 0;
}
