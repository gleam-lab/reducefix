```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    long long x = 0, y = 0, cx = 0, cy = 0;
    double c = 0;
    for(int i = 0; i < n; ++i){
        cin >> cx >> cy;
        c += sqrt((cx - x) * (cx - x) + (cy - y) * (cy - y));
        x = cx;
        y = cy;
    }
    c += sqrt(x * x + y * y);
    cout << setprecision(10) << c << endl;
    return 0;
}
```
