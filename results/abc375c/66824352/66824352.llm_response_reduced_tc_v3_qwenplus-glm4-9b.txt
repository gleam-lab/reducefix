
```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    vector<pair<double, double>> points;
    for (int i = 0; i < n; ++i) {
        int x, y;
        cin >> x >> y;
        points.push_back({x, y});
    }
    
    double total_cost = 0.0;
    double x = 0.0, y = 0.0;
    
    for (const auto& p : points) {
        total_cost += sqrt((x - p.first) * (x - p.first) + (y - p.second) * (y - p.second));
        x = p.first;
        y = p.second;
    }
    
    total_cost += sqrt(x * x + y * y);
    cout << fixed << setprecision(10) << total_cost << endl;
    
    return 0;
}
```
