```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    double total_cost = 0.0;
    int prev_x = 0, prev_y = 0;

    for (int i = 0; i < n; ++i) {
        int curr_x, curr_y;
        cin >> curr_x >> curr_y;
        int dx = curr_x - prev_x;
        int dy = curr_y - prev_y;
        total_cost += sqrt(static_cast<double>(dx * dx + dy * dy));
        prev_x = curr_x;
        prev_y = curr_y;
    }

    // Return to origin
    int dx = 0 - prev_x;
    int dy = 0 - prev_y;
    total_cost += sqrt(static_cast<double>(dx * dx + dy * dy));

    cout << fixed << setprecision(10) << total_cost << endl;
    return 0;
}
```
