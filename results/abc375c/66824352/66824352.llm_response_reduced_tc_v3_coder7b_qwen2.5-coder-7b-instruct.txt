```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    double x = 0, y = 0, total_cost = 0;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        int xi, yi;
        cin >> xi >> yi;
        total_cost += sqrt(x * x + y * y);
        total_cost += sqrt(xi * xi + yi * yi);
        total_cost += sqrt((xi - x) * (xi - x) + (yi - y) * (yi - y));
        x = xi;
        y = yi;
    }
    total_cost += sqrt(x * x + y * y); // Return to origin
    cout << setprecision(10) << total_cost << endl;
    return 0;
}
```

This solution calculates the total cost by considering the distance from the current position to each new point, as well as the distance from the last visited point back to the next point. This ensures that all segments of the journey are accounted for in the final cost calculation. The use of `setprecision` with a high number of decimal places helps in achieving the required precision for the output.
