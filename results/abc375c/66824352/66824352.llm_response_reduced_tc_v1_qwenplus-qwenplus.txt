```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    long long x = 0, y = 0; // Use long long to prevent overflow
    double cost = 0.0;
    
    for (int i = 0; i < n; ++i) {
        long long xi, yi;
        cin >> xi >> yi;
        
        long long dx = x - xi;
        long long dy = y - yi;
        cost += sqrt(static_cast<double>(dx * dx + dy * dy));
        
        x = xi;
        y = yi;
    }
    
    // Return to origin
    cost += sqrt(static_cast<double>(x * x + y * y));
    
    cout << fixed << setprecision(15) << cost << endl;
    
    return 0;
}
```
