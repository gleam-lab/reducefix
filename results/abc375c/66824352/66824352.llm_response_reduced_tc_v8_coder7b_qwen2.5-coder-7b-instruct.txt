```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<pair<int, int>> points(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0.0;
    for (size_t i = 1; i < n; ++i) {
        total_cost += sqrt(pow(points[i-1].first - points[i].first, 2) + pow(points[i-1].second - points[i].second, 2));
    }
    // Add the distance from the last point back to the origin
    total_cost += sqrt(pow(points.back().first, 2) + pow(points.back().second, 2));

    cout << fixed << setprecision(10) << total_cost << endl;
    return 0;
}
```

This solution correctly calculates the total cost of Takahashi's journey using the Euclidean distance formula. It reads the number of points and their coordinates, then iterates through the list to sum up the distances between consecutive points and finally adds the distance from the last point back to the origin. The result is printed with high precision as required.
