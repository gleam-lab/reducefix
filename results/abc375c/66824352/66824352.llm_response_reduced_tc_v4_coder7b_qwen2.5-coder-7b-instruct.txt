Here's the corrected version of your code. The issue was that you were using integer division in your distance calculation, which could lead to incorrect results due to truncation. Additionally, I've added a check to handle the case where `n` is zero, although it's not necessary as per the problem constraints but good practice to include it.

```cpp
#include <iostream>
#include <algorithm>
#include <iomanip>
#include <cmath>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    
    // If there are no points, the total cost is 0
    if (n == 0) {
        cout << "0" << endl;
        return 0;
    }

    vector<pair<int, int>> points(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double totalCost = 0.0;
    int startX = 0, startY = 0;

    for (const auto& point : points) {
        totalCost += sqrt(pow(startX - point.first, 2) + pow(startY - point.second, 2));
        startX = point.first;
        startY = point.second;
    }

    // Add the cost to return to the origin
    totalCost += sqrt(pow(startX, 2) + pow(startY, 2));

    cout << setprecision(10) << totalCost << endl;

    return 0;
}
```

### Explanation of Changes
1. **Avoiding Integer Division**: Changed the distance calculation to use floating-point arithmetic (`pow` instead of `*` for squaring).
2. **Edge Case Handling**: Added a check for `n == 0`, although it's not required by the constraints.
3. **Performance Optimization**: Used `ios::sync_with_stdio(false); cin.tie(nullptr);` to speed up input/output operations.
4. **Standard C++20 Features**: Used `std::vector` for storing points instead of an array, which is more flexible and easier to manage.
