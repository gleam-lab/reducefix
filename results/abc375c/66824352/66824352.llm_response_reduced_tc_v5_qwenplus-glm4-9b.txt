#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n, x = 0, y = 0;
    double total_cost = 0.0;
    cin >> n;

    for (int i = 0; i < n; i++) {
        int xi, yi;
        cin >> xi >> yi;
        total_cost += sqrt((x - xi) * (x - xi) + (y - yi) * (y - yi));
        x = xi;
        y = yi;
    }

    total_cost += sqrt(x * x + y * y);
    cout << fixed << setprecision(10) << total_cost << endl;
    return 0;
}
