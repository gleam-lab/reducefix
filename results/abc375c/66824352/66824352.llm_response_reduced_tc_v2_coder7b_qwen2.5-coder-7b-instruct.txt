#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    double c = 0;
    cin >> n;
    int x = 0, y = 0;
    for (int i = 0; i < n; ++i) {
        int x1, y1;
        cin >> x1 >> y1;
        c += sqrt(x * x + y * y);
        c += sqrt((x - x1) * (x - x1) + (y - y1) * (y - y1));
        x = x1;
        y = y1;
    }
    c += sqrt(x * x + y * y);
    cout << setprecision(10) << c << endl;
    return 0;
}
