```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double total = 0.0;
    double prev_x = 0.0, prev_y = 0.0;

    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;
        double dx = x - prev_x;
        double dy = y - prev_y;
        total += sqrt(dx * dx + dy * dy);
        prev_x = x;
        prev_y = y;
    }

    // Return to origin
    total += sqrt(prev_x * prev_x + prev_y * prev_y);

    // Print with sufficient precision
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
