
```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<pair<double, double>> points(N + 1); // Include the origin at (0, 0) at index 0

    // Read points and store in the vector
    for (int i = 0; i <= N; i++) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0.0;
    // Calculate the cost of moving from the origin to each point and back to the origin
    for (int i = 1; i <= N; i++) {
        total_cost += sqrt(pow(points[i].first, 2) + pow(points[i].second, 2));
    }

    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
```
