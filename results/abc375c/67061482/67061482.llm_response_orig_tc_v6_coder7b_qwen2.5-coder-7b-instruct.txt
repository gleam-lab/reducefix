#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    long long N;
    cin >> N;
    vector<long long> x(N), y(N);
    for(long long i = 0; i < N; ++i) {
        cin >> x[i] >> y[i];
    }

    double sum = 0;
    for(long long i = 1; i < N; ++i) {
        sum += hypot(x[i] - x[i-1], y[i] - y[i-1]);
    }
    sum += hypot(x[0] - x[N-1], y[0] - y[N-1]);

    printf("%.10f\n", sum);

    return 0;
}
