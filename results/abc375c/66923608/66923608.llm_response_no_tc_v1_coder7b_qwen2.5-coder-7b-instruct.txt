```cpp
#include <bits/stdc++.h>
using namespace std;

// Function to calculate Euclidean distance between two points
double Euclid(int x1, int y1, int x2, int y2) {
    return sqrt(pow(x1 - x2, 2) + pow(y1 - y2, 2));
}

int main() {
    int n; 
    cin >> n;
    
    // Initialize coordinates for the origin
    int x_origin = 0, y_origin = 0;
    double ans = 0; 
    
    // Read coordinates of the points
    vector<int> x(n), y(n);
    for (int i = 0; i < n; ++i) {
        cin >> x[i] >> y[i];
        // Calculate the distance from the previous point to the current one
        if (i > 0) {
            ans += Euclid(x[i-1], y[i-1], x[i], y[i]);
        }
    }
    // Calculate the distance from the last point back to the origin
    ans += Euclid(x[n-1], y[n-1], x_origin, y_origin);

    // Output the result with high precision
    cout << fixed << setprecision(12) << ans << endl;
}
```
