
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    vector<pair<int, int>> points(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }
    
    double total_cost = 0.0;
    
    // Calculate the cost of moving from the origin to the first point
    total_cost += sqrt(pow(points[0].first, 2) + pow(points[0].second, 2));
    
    // Calculate the cost of moving between consecutive points
    for (int i = 0; i < n - 1; ++i) {
        total_cost += sqrt(pow(points[i].first - points[i + 1].first, 2) + pow(points[i].second - points[i + 1].second, 2));
    }
    
    // Calculate the cost of moving from the last point back to the origin
    total_cost += sqrt(pow(points[n - 1].first, 2) + pow(points[n - 1].second, 2));
    
    // Set precision and output the result
    cout << fixed << setprecision(12) << total_cost << endl;
    
    return 0;
}
```
