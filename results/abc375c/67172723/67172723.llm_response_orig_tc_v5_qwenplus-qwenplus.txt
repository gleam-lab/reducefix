```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    double total = 0.0;
    long long px = 0, py = 0;
    long long x, y;

    for (int i = 0; i < n; ++i) {
        cin >> x >> y;
        long long dx = x - px;
        long long dy = y - py;
        total += sqrt(static_cast<double>(dx * dx + dy * dy));
        px = x;
        py = y;
    }

    // Return to origin
    total += hypot(static_cast<double>(px), static_cast<double>(py));

    printf("%.15f\n", total);
    return 0;
}
```
