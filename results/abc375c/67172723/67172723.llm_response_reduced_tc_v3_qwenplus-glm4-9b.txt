#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    int n;
    cin >> n;

    vector<pair<int, int>> points(n + 1);
    points[0] = {0, 0}; // Starting at the origin

    int x, y;
    for (int i = 1; i <= n; ++i) {
        cin >> x >> y;
        points[i] = {x, y};
    }

    double total_distance = 0;
    for (int i = 1; i <= n; ++i) {
        total_distance += hypot(points[i].first - points[i - 1].first, points[i].second - points[i - 1].second);
    }

    // Calculate the distance back to the origin
    total_distance += hypot(points[n].first, points[n].second);

    printf("%.10f\n", total_distance);
    return 0;
}
