#include <cmath>
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    vector<pair<int, int>> points;
    int x, y;
    for (int i = 0; i < n; i++) {
        cin >> x >> y;
        points.push_back({x, y});
    }
    
    double ans = 0.0;
    for (int i = 0; i < n; i++) {
        ans += hypot(points[i].first, points[i].second);
    }
    
    // Calculate the distance from the last point to the origin and add to the total cost
    ans += hypot(points[n - 1].first, points[n - 1].second);
    
    printf("%.10f\n", ans);
    return 0;
}
