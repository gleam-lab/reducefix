```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    long long px = 0, py = 0;
    double total = 0;

    // Use vectors to store points for better cache locality if needed
    vector<pair<long long, long long>> points(n);

    for (int i = 0; i < n; ++i) {
        long long x, y;
        cin >> x >> y;
        points[i] = {x, y};
    }

    // Add return to origin after last point separately to avoid extra condition in loop
    for (const auto& [x, y] : points) {
        total += hypot(static_cast<double>(x - px), static_cast<double>(y - py));
        px = x;
        py = y;
    }

    total += hypot(static_cast<double>(px), static_cast<double>(py));

    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
```
