#include <bits/stdc++.h>

  for(int i = 0; i <= n; i++) {
    if(i == n) {
      x = 0, y = 0; // Return to the origin

    }
    ans += sqrt((x - px) * (x - px) + (y - py) * (y - py));
    px = x;
    py = y;
  }
  
  printf("%.10f\n", ans);
  return 0;
}
