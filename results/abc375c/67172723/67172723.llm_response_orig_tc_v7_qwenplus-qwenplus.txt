```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    double total = 0.0;
    int prev_x = 0, prev_y = 0;

    for (int i = 0; i < n; ++i) {
        int x, y;
        cin >> x >> y;
        long long dx = x - prev_x;
        long long dy = y - prev_y;
        total += sqrt(static_cast<double>(dx * dx + dy * dy));
        prev_x = x;
        prev_y = y;
    }

    // Return to origin
    long long dx = 0 - prev_x;
    long long dy = 0 - prev_y;
    total += sqrt(static_cast<double>(dx * dx + dy * dy));

    // Use fixed and scientific to ensure sufficient precision
    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
```
