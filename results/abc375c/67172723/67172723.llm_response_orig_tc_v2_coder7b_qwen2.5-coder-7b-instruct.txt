```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  
  long long px=0, py=0;
  double ans=0;
  int x, y;
  for(int i=0; i<n; i++){
    cin >> x >> y;
    ans += hypot(px-x, py-y);
    px = x;
    py = y;
  }
  
  ans += hypot(px, py);
  printf("%.10f\n", ans);
  return 0;
}
```
