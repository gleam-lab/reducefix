
```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    double px = 0.0;
    double py = 0.0;
    double ans = 0.0;
    int x, y;
    
    for (int i = 0; i < n; ++i) {
        cin >> x >> y;
        ans += sqrt((x - px) * (x - px) + (y - py) * (y - py));
        px = x;
        py = y;
    }
    
    // Don't forget to add the distance from the last point back to the origin
    ans += sqrt((px) * (px) + (py) * (py));
    
    printf("%.10f\n", ans);
    return 0;
}
```
