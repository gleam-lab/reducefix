```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  
  // Initialize starting position
  long long px = 0, py = 0;
  double ans = 0;
  long long x, y;
  
  // Loop through each point
  for(int i = 0; i < n; i++){
    cin >> x >> y;
    // Calculate Euclidean distance to the next point and add to total cost
    ans += sqrt((x - px) * (x - px) + (y - py) * (y - py));
    // Update current position
    px = x;
    py = y;
  }
  
  // Return to the origin
  ans += sqrt(px * px + py * py);
  // Print the total cost with high precision
  printf("%.10f\n", ans);
  return 0;
}
```
