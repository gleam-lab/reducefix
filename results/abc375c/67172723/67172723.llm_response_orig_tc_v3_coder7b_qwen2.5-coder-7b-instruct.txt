```cpp
#include <iostream>
#include <vector>
#include <cmath>

int main() {
    int n;
    std::cin >> n;
    
    std::vector<std::pair<int, int>> points(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> points[i].first >> points[i].second;
    }

    double ans = 0;
    int px = 0, py = 0;
    for (const auto& p : points) {
        ans += std::hypot(px - p.first, py - p.second);
        px = p.first;
        py = p.second;
    }
    ans += std::hypot(px, py);

    std::cout << std::fixed << std::setprecision(15) << ans << '\n';
    return 0;
}
```

This solution reads the number of points and their coordinates, then calculates the total distance traveled by Takahashi as he moves from the origin to each point in sequence and finally back to the origin. The `std::hypot` function is used to compute the Euclidean distance between two points efficiently. The result is printed with high precision to ensure accuracy.
