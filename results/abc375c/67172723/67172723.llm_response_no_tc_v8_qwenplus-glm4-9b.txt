#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    double ans = 0.0;
    int px = 0, py = 0;
    
    for (int i = 0; i < n; ++i) {
        int x, y;
        cin >> x >> y;
        ans += sqrt((x - px) * (x - px) + (y - py) * (y - py));
        px = x;
        py = y;
    }
    
    // Add the distance from the last point to the origin
    ans += sqrt(px * px + py * py);
    
    cout.precision(10);
    cout << fixed << ans << endl;
    return 0;
}
