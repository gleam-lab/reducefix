```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);

  int n;
  cin >> n;

  double ans = 0.0;
  int prev_x = 0, prev_y = 0;
  int curr_x, curr_y;

  for (int i = 0; i < n; ++i) {
    cin >> curr_x >> curr_y;
    ans += hypot(curr_x - prev_x, curr_y - prev_y);
    prev_x = curr_x;
    prev_y = curr_y;
  }

  ans += hypot(prev_x, prev_y); // Return to origin

  // Print with sufficient precision
  cout.precision(15);
  cout << fixed << ans << '\n';

  return 0;
}
```
