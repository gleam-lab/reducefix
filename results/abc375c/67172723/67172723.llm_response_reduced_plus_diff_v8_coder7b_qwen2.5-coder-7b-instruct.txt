```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    vector<pair<int, int>> points(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0.0;
    for (size_t i = 1; i < points.size(); ++i) {
        int dx = points[i].first - points[i - 1].first;
        int dy = points[i].second - points[i - 1].second;
        total_cost += sqrt(dx * dx + dy * dy);
    }

    // Return to origin
    int dx = points[0].first - points[n - 1].first;
    int dy = points[0].second - points[n - 1].second;
    total_cost += sqrt(dx * dx + dy * dy);

    cout << fixed << setprecision(10) << total_cost << endl;
    return 0;
}
```
