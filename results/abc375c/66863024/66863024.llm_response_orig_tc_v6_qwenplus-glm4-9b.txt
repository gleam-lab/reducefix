#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<pair<long long, long long>> points(N + 1);

    long long x, y;
    for (int i = 1; i <= N; ++i) {
        cin >> x >> y;
        points[i] = make_pair(x, y);
    }

    long long total_cost = 0;
    for (int i = 1; i <= N; ++i) {
        total_cost += sqrt((points[i].first - 0LL) * (points[i].first - 0LL) + 
                           (points[i].second - 0LL) * (points[i].second - 0LL));
        total_cost += sqrt((points[i].first - points[i - 1].first) * (points[i].first - points[i - 1].first) + 
                           (points[i].second - points[i - 1].second) * (points[i].second - points[i - 1].second));
    }
    total_cost += sqrt((0LL - points[N].first) * (0LL - points[N].first) + 
                       (0LL - points[N].second) * (0LL - points[N].second));

    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
