```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;

    long double total = 0.0;
    long double prevX = 0.0, prevY = 0.0;
    long double currX, currY;

    for (int i = 0; i < N; ++i) {
        cin >> currX >> currY;

        long double dx = currX - prevX;
        long double dy = currY - prevY;

        total += sqrtl(dx * dx + dy * dy);

        prevX = currX;
        prevY = currY;
    }

    // Return to origin
    long double dx = 0.0 - prevX;
    long double dy = 0.0 - prevY;
    total += sqrtl(dx * dx + dy * dy);

    cout << fixed << setprecision(10) << total << endl;

    return 0;
}
```
