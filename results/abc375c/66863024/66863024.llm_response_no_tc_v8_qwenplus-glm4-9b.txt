#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int N;
    cin >> N;

    long double ans = 0.0;
    long double x = 0.0, y = 0.0;

    for (int i = 0; i < N; ++i) {
        long double nx, ny;
        cin >> nx >> ny;
        ans += sqrt((nx - x) * (nx - x) + (ny - y) * (ny - y));
        x = nx;
        y = ny;
    }

    // After visiting all points, return to the origin
    ans += sqrt(x * x + y * y);

    cout.precision(10);
    cout << fixed << ans << endl;

    return 0;
}
