
```cpp
#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main() {
    int N;
    cin >> N;
    
    vector<pair<long long, long long>> points(N);
    points[0] = {0, 0};
    
    for (int i = 1; i < N; i++) {
        long long x, y;
        cin >> x >> y;
        points[i] = {x, y};
    }
    
    double total_cost = 0.0;
    for (int i = 0; i < N; i++) {
        double dx = points[i].first - points[(i + 1) % N].first;
        double dy = points[i].second - points[(i + 1) % N].second;
        total_cost += sqrt(dx * dx + dy * dy);
    }
    
    cout.precision(12);
    cout << fixed << total_cost << endl;
    
    return 0;
}
```
