```cpp
#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N);
    for (int i = 0; i < N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    // Calculate the total distance
    long long total_distance = 0;
    for (int i = 0; i < N - 1; ++i) {
        total_distance += abs(points[i].first - points[i + 1].first) +
                          abs(points[i].second - points[i + 1].second);
    }

    // Add the distance back to the origin
    total_distance += abs(points[0].first) + abs(points[0].second);

    cout << total_distance << endl;

    return 0;
}
```
