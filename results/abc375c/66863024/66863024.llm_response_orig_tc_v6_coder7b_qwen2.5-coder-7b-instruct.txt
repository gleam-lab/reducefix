        total_cost += sqrtl(dx * dx + dy * dy);

    // Return to the origin
    long double dx = points[n].first - points[0].first;
    long double dy = points[n].second - points[0].second;
    total_cost += sqrtl(dx * dx + dy * dy);

    cout << setprecision(15) << fixed << total_cost << endl;

    return 0;
