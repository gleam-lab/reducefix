
```cpp
#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<pair<long long, long long>> points(N);
    
    // Read points
    for (int i = 0; i < N; ++i) {
        cin >> points[i].first >> points[i].second;
    }
    
    // Starting point (0, 0)
    long long x = 0, y = 0;
    long double total_cost = 0;
    
    // Calculate the cost between consecutive points
    for (int i = 0; i < N; ++i) {
        long long nextX = points[i].first;
        long long nextY = points[i].second;
        total_cost += sqrt(pow(x - nextX, 2) + pow(y - nextY, 2));
        x = nextX;
        y = nextY;
    }
    
    // Calculate the cost from the last point back to the origin
    total_cost += sqrt(pow(x, 2) + pow(y, 2));
    
    // Output the total cost
    cout << fixed << setprecision(9) << total_cost << endl;
    
    return 0;
}
```
