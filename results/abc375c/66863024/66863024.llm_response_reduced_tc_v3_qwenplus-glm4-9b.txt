
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N + 1); // +1 for the extra origin point

    // Read the points including the origin (0, 0)
    for (int i = 0; i <= N; ++i) {
        long long x, y;
        cin >> x >> y;
        points[i] = {x, y};
    }

    // Calculate the total distance using the Euclidean distance formula
    double sum = 0.0;
    for (int i = 1; i <= N; ++i) {
        double dx = points[i].first - points[i - 1].first;
        double dy = points[i].second - points[i - 1].second;
        sum += sqrt(dx * dx + dy * dy);
    }

    // Print the total distance with high precision
    cout << fixed << setprecision(20) << sum << endl;

    return 0;
}
```
