```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    long double cost = 0;
    int N;
    long double nowX = 0, nowY = 0;
    long double nextX, nextY;

    cin >> N;

    for (int i = 0; i < N; i++) {
        cin >> nextX >> nextY;

        long double dx = nowX - nextX;
        long double dy = nowY - nextY;

        cost += sqrtl(dx * dx + dy * dy);

        nowX = nextX;
        nowY = nextY;
    }

    cost += sqrtl(nowX * nowX + nowY * nowY);

    cout << fixed << setprecision(20) << cost << endl;

    return 0;
}
```

3. **Final Distance Calculation**: Used `nowX` and `nowY` for the final distance back to the origin, as these variables hold the last visited point after the loop.
