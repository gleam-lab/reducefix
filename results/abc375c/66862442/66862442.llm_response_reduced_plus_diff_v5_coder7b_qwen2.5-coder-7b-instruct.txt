        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x; 
        curY = y;
    }

    // Distance from the last point back to the origin
    total += sqrt(curX * curX + curY * curY); 

    // Print the result with high precision
    printf("%.10f\n", total);

}
