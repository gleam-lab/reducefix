#include <iostream>
#include <cmath>
#include <iomanip>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;

    double curX = 0, curY = 0;
    double total = 0;          // Change to double for higher precision

    for (int i = 0; i < N; ++i) {
        long long x, y;
        std::cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += std::sqrt(dx * dx + dy * dy); // Use sqrt for distance calculation

        curX = x; curY = y;
    }
    total += std::sqrt(curX * curX + curY * curY); // Distance back to origin

    // Output with high precision
    std::cout << std::fixed << std::setprecision(15) << total << '\n';

    return 0;
}
