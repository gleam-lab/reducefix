```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double curX = 0, curY = 0;
    double total = 0; 

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x; curY = y;
    }
    total += sqrt(curX * curX + curY * curY); // 原点へ戻る

    cout << total << '\n';
}

After the loop, the total distance is calculated again to include the return trip to the origin. Finally, the result is printed directly without attempting to convert it to a string. This approach avoids issues with integer overflow and ensures that the distances are computed accurately.
