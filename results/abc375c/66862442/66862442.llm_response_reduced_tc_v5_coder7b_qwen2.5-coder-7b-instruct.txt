int main() {

    double total = 0;          // Use double for floating-point calculations

        double x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x; curY = y;
    }

    printf("%.15f\n", total); // Print with high precision
}
