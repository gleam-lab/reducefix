```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double curX = 0, curY = 0;
    double total = 0;

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x; 
        curY = y;
    }

    total += sqrt(curX * curX + curY * curY); // 原点へ戻る

    printf("%.10f\n", total); // 使用printf以指定小数位数输出

}
```

- Replaced manual string conversion with `printf` to format the output correctly. This ensures that the output has exactly 10 decimal places, which is what the problem requires.
