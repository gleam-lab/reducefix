
```cpp
#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;
    __int128 total = 0; // Use __int128 for potential large distances

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        total += (__int128)(curX - x) * (curX - x) + (__int128)(curY - y) * (curY - y);
        curX = x;
        curY = y;
    }
    total += (__int128)(curX - 0) * (curX - 0) + (__int128)(curY - 0) * (curY - 0); // Return to origin

    // Using sqrt to calculate the total distance and set precision
    double result = sqrt(total);
    cout << fixed << setprecision(12) << result << "\n";

    return 0;
}
```
