#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;
    double total = 0.0; // Change to double for floating-point arithmetic

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long long dx = curX - x;
        long long dy = curY - y;
        total += sqrt(dx * dx + dy * dy); // Calculate Euclidean distance correctly

        curX = x; 
        curY = y;
    }
    total += sqrt(curX * curX + curY * curY); // Distance back to origin

    cout << fixed << setprecision(15) << total << '\n'; // Set precision for output

    return 0;
}
