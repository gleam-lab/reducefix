
```cpp
#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<double, double>> points;
    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;
        points.push_back({x, y});
    }

    double total_cost = 0.0;
    for (int i = 0; i < N; ++i) {
        double dx = points[i].first - 0.0;
        double dy = points[i].second - 0.0;
        total_cost += sqrt(dx * dx + dy * dy);
    }

    cout.precision(30);
    cout << fixed << total_cost << endl;

    return 0;
}
```

This code calculates the total travel cost for Takahashi to visit all given points in sequence and return to the origin. The distance between two points is calculated using the Euclidean distance formula, and the sum of these distances gives the total travel cost. The `cout.precision(30)` and `cout << fixed` are used to ensure that the output has enough precision to meet the error requirements.
