3. Sum all these distances.
4. Finally, calculate the distance from the last point back to the origin and add it to the sum.

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double total = 0.0;
    long long curX = 0, curY = 0;

        long long dx = curX - x;
        long long dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x;
        curY = y;
    }

    // Distance from last point back to origin
    total += sqrt(curX * curX + curY * curY);

    cout << fixed << setprecision(15) << total << '\n';
    return 0;
}
