
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    if (N == 0) {
        cout << "0" << endl;
        return 0;
    }

    long double curX = 0, curY = 0;
    long double total = 0; // Use long double for higher precision

    for (int i = 0; i < N; ++i) {
        long double x, y;
        cin >> x >> y;

        long double dx = curX - x;
        long double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x; curY = y;
    }
    total += sqrt(curX * curX + curY * curY); // Add the distance back to the origin

    cout.precision(30); // Set precision for output
    cout << fixed << total << endl; // Output with fixed decimal places
}
```
