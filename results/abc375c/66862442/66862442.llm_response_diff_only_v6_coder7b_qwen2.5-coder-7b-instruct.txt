```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;
    double total = 0;          // Change to double for accurate floating-point arithmetic

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy); // Calculate Euclidean distance directly

        curX = x; curY = y;
    }
    total += sqrt(curX * curX + curY * curY); // Return to the origin

    printf("%.6f\n", total); // Print with 6 decimal places for precision
}
```

This solution corrects the algorithm by using `double` for the total cost calculation instead of an integer type, which allows for accurate representation of floating-point numbers. The Euclidean distance is calculated directly within the loop and added to the total cost. Finally, the result is printed with 6 decimal places to ensure the required precision.
