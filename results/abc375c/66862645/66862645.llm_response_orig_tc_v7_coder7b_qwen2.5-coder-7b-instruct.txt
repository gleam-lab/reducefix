```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    long double x = 0, y = 0, cost = 0, ans = 0;

    for(int i = 0; i < n; i++) {
        long double nx, ny;
        cin >> nx >> ny;
        cost = sqrtl(powl(x - nx, 2) + powl(y - ny, 2));
        ans += cost;
        x = nx;
        y = ny;
    }

    cost = sqrtl(x * x + y * y);
    ans += cost;

    cout.precision(15);
    cout << ans << '\n';

    return 0;
}
```
