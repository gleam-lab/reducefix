```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    vector<pair<int, int>> points(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0;
    for (size_t i = 1; i < points.size(); ++i) {
        auto &prev = points[i - 1];
        auto &current = points[i];
        double dx = current.first - prev.first;
        double dy = current.second - prev.second;
        total_cost += sqrt(dx * dx + dy * dy);
    }

    // Return to origin
    auto &last = points.back();
    double dx = last.first;
    double dy = last.second;
    total_cost += sqrt(dx * dx + dy * dy);

    printf("%.15f\n", total_cost);
    return 0;
}
```
