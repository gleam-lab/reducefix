The cost of movement between two points is the Euclidean distance, which can be found using the formula \(\sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}\). Takahashi starts at the origin (0,0) and visits N points, and then returns to the origin. The goal is to compute the total travel cost for this journey.

4. **Return to Origin**:

#include <iostream>
#include <cmath>
using namespace std;

int main() {
    long double ans = 0.0;
    int nowX = 0, nowY = 0;
    
    int N;
    cin >> N;

    for (int i = 0; i < N; ++i) {

        cin >> nextX >> nextY;
        ans += sqrt((long double)(nextX - nowX) * (nextX - nowX) + (nextY - nowY) * (nextY - nowY));
        nowX = nextX;
        nowY = nextY;
    }
    ans += sqrt((long double)(nowX - 0) * (nowX - 0) + (nowY - 0) * (nowY - 0));
    
    cout.precision(10);

}
