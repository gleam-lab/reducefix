```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N;
    cin >> N;

    vector<pair<int, int>> points(N);
    for (int i = 0; i < N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    long double total_cost = 0;
    for (size_t i = 1; i < N; ++i) {
        int dx = points[i].first - points[i-1].first;
        int dy = points[i].second - points[i-1].second;
        total_cost += sqrt(dx * dx + dy * dy);
    }

    int last_x = points[N-1].first;
    int last_y = points[N-1].second;
    total_cost += sqrt(last_x * last_x + last_y * last_y);

    cout.precision(15);
    cout << total_cost << '\n';

    return 0;
}
```
