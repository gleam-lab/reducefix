#include <bits/stdc++.h>
using namespace std;

int main(){
    long double cost = 0;
    long double ans = 0;
    int operation;
    int nowX = 0 , nowY = 0;
    int nextX, nextY;

    cin >> operation;

    for(int i = 0;  i < operation; i++) {
        cin >> nextX >> nextY;

        cost = sqrt(pow(nextX - nowX, 2) + pow(nextY - nowY, 2));

        nowX = nextX;
        nowY = nextY;

        ans += cost;
    }

    cost = sqrt(pow(nowX - 0, 2) + pow(nowY - 0, 2));
    ans += cost;

    cout << ans << endl;

    return 0;
}
