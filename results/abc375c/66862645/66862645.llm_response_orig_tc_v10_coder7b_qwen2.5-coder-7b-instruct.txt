    vector<long double> x(n), y(n);

        cin >> x[i] >> y[i];
    }

    long double ans = 0;
    for (int i = 0; i < n; ++i) {
        long double dx = x[i] - x[(i + 1) % n];
        long double dy = y[i] - y[(i + 1) % n];

    return 0;
}
