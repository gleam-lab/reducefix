```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total = 0.0;
    long double x = 0, y = 0; // current position

    for (int i = 0; i < N; ++i) {
        long double nx, ny;
        cin >> nx >> ny;

        long double dx = x - nx;
        long double dy = y - ny;
        total += sqrt(dx * dx + dy * dy);

        x = nx;
        y = ny;
    }

    // Return to origin
    total += sqrt(x * x + y * y);

    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
```
