```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    double totalCost = 0.0;
    int prevX = 0, prevY = 0;
    
    for (int i = 0; i < N; ++i) {
        int x, y;
        cin >> x >> y;
        
        // Calculate the Euclidean distance between current and previous point
        double distance = sqrt(pow(x - prevX, 2) + pow(y - prevY, 2));
        totalCost += distance;
        
        // Update previous point to current point
        prevX = x;
        prevY = y;
    }
    
    // Return to origin from last point
    double finalDistance = sqrt(pow(prevX, 2) + pow(prevY, 2));
    totalCost += finalDistance;
    
    cout << totalCost << endl;
    
    return 0;
}
```
