
```cpp
#include <iostream>
#include <vector>
#include <cmath>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;

    std::vector<std::pair<int, int>> points(n + 1);
    points[0] = {0, 0}; // Starting at the origin
    long double total_cost = 0.0;

    for (int i = 1; i <= n; ++i) {
        std::cin >> points[i].first >> points[i].second;
    }

    for (int i = 1; i <= n; ++i) {
        total_cost += std::sqrt(std::pow(points[i].first - points[i - 1].first, 2) +
                                std::pow(points[i].second - points[i - 1].second, 2));
    }

    // Add the cost from the last point to the origin
    total_cost += std::sqrt(std::pow(points[n].first - points[0].first, 2) +
                            std::pow(points[n].second - points[0].second, 2));

    std::cout.precision(12);
    std::cout << std::fixed << total_cost << std::endl;

    return 0;
}
```

This code correctly calculates the total travel cost for Takahashi by summing the Euclidean distances between consecutive points and the return trip from the last point to the origin. It uses `std::pow` from the `<cmath>` library to compute the squares of the differences in the x and y coordinates, and `std::sqrt` to compute the Euclidean distance. The output is formatted to 12 decimal places to meet the precision requirement.
