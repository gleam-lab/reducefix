To fix this, we need to compute the Euclidean distance between consecutive points (including the origin at the start and end) by taking the square root of the sum of the squared differences in coordinates. We should accumulate these distances as floating-point numbers to ensure precision.

#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;   // Current position (origin initially)
    double total = 0.0;             // Total cost as a double to accumulate square roots

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long long dx = curX - x;
        long long dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // Add the actual Euclidean distance

        curX = x; curY = y;
    }

    // Add the distance from the last point back to the origin
    total += sqrt(curX * curX + curY * curY);

    // Output with sufficient precision
    cout << fixed << setprecision(20) << total << '\n';
    return 0;
}
