```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;
    double total = 0.0;

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long long dx = curX - x;
        long long dy = curY - y;
        total += sqrt(static_cast<double>(dx * dx + dy * dy));

        curX = x;
        curY = y;
    }

    // Return to origin
    long long dx = curX;
    long long dy = curY;
    total += sqrt(static_cast<double>(dx * dx + dy * dy));

    // Print with sufficient precision
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
