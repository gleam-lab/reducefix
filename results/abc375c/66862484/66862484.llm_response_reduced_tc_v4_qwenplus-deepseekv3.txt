    cin.tie(nullptr);

    int N;

    double curX = 0.0, curY = 0.0;   // Current position (origin)
    double total = 0.0;               // Total cost

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // Actual Euclidean distance

        curX = x; curY = y;

    // Distance from last point back to origin
    total += sqrt(curX * curX + curY * curY);

    // Print with sufficient precision

    return 0;
}
