
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <limits>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N + 1); // +1 for the return to the origin
    long long curX = 0, curY = 0;

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;
        points[i] = {x, y};
    }

    // Add the origin as the starting point
    points[N] = {0, 0};

    double totalCost = 0.0; // Use double to prevent overflow for the square root calculation

    for (int i = 0; i < N; ++i) {
        long long dx = curX - points[i].first;
        long long dy = curY - points[i].second;
        totalCost += sqrt(dx * dx + dy * dy); // Add the distance to the total cost
        curX = points[i].first;
        curY = points[i].second;
    }

    // Output the total cost
    cout.precision(numeric_limits<streamsize>::digits10);
    cout << fixed << totalCost << '\n';

    return 0;
}
```
