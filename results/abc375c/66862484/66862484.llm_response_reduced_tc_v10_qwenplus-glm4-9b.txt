
```cpp
#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<pair<int, int>> points(N + 1);
    points[0] = {0, 0};  // Start at the origin

    long double x, y;
    for (int i = 1; i <= N; ++i) {
        cin >> x >> y;
        points[i] = {x, y};
    }

    long double total_cost = 0.0;
    for (int i = 1; i <= N; ++i) {
        long double dx = points[i].first - points[i - 1].first;
        long double dy = points[i].second - points[i - 1].second;
        total_cost += sqrt(dx * dx + dy * dy);
    }

    cout << fixed << setprecision(15) << total_cost << '\n';
    return 0;
}
```
