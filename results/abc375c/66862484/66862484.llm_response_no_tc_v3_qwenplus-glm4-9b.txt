#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;   // 莽聫戮氓聹篓盲陆聧莽陆庐茂录聢氓聨聼莽聜鹿茂录聣
    long long total = 0;            // 莽路聫茫聜鲁茫聜鹿茫聝聢

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        total += (curX > x ? curX - x : x - curX) * (curY > y ? curY - y : y - curY); // 莽碌聦猫路炉茫聛庐氓路庐氓聢聠茫聛庐莽漏聧茫聜聮猫露鲁茫聛聶
        curX = x; curY = y;
    }

    // 忙聹聙氓戮聦茫聛庐莽聜鹿茫聛聥茫聜聣氓聨聼莽聜鹿茫聛赂忙聢禄茫聜聥
    total += (curX > 0 ? curX : -curX) * (curY > 0 ? curY : -curY);

    cout << fixed << setprecision(6) << sqrt(total) << '\n';
    return 0;
}
