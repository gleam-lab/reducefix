#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;   // 莽聫戮氓聹篓盲陆聧莽陆庐茂录聢氓聨聼莽聜鹿茂录聣
    double total = 0.0;             // 莽路聫茫聜鲁茫聜鹿茫聝聢

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        total += hypot((double)(curX - x), (double)(curY - y));  // 盲陆驴莽聰篓 hypot 氓聡陆忙聲掳

        curX = x; curY = y;
    }

    // 忙聹聙氓戮聦茫聛庐莽聜鹿茫聛聥茫聜聣氓聨聼莽聜鹿茫聛赂忙聢禄茫聜聥
    total += hypot((double)curX, (double)curY);

    /* --- 氓聡潞氓聤聸 ---
       double 莽卤禄氓聻聥茫聜聮忙聳聡氓颅聴氓聢聴氓聦聳茫聛聴茫聛娄氓聡潞氓聤聸茫聛聶茫聜聥 */
    cout.precision(15);
    cout << fixed << total << '\n';
    return 0;
}
