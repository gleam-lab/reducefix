#include <iostream>
#include <cmath>
#include <vector>

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;

    double curX = 0, curY = 0;   // Current position (origin)
    double total = 0;             // Total cost

    for (int i = 0; i < N; ++i) {
        int x, y;
        std::cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += std::sqrt(dx * dx + dy * dy);  // Include sqrt for actual cost calculation

        curX = x; curY = y;
    }

    // Return to origin from last point
    total += std::sqrt(curX * curX + curY * curY);

    // Output the result with appropriate precision
    std::cout.precision(15);
    std::cout << std::fixed << total << '\n';

    return 0;
}
