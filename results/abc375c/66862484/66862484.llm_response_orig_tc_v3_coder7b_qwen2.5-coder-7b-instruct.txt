        long double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // 正しく平方根を取る

        curX = x; curY = y;
    }

    // 最後の点から原点へ戻る
    total += sqrt(curX * curX + curY * curY);

    // 結果を出力
    printf("%.15Lf\n", total);  // 浮動小数点で出力

    return 0;
}
