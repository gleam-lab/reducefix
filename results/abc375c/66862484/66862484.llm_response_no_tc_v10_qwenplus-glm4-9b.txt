
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N + 1);
    long long total_cost = 0;
    long long current_x = 0, current_y = 0;

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;
        points[i + 1] = {x, y};
    }

    for (int i = 0; i <= N; ++i) {
        long long x = points[i].first;
        long long y = points[i].second;
        total_cost += static_cast<long long>(sqrt(pow(x - current_x, 2) + pow(y - current_y, 2)));
        current_x = x;
        current_y = y;
    }

    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
```
