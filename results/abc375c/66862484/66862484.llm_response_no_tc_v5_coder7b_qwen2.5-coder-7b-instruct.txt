```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double curX = 0, curY = 0;   // 現在位置（原点）
    long double total = 0;             // 総コスト (long double)

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long double dx = curX - x;
        long double dy = curY - y;
        total += sqrtl(dx * dx + dy * dy);  // 正しい距離計算

        curX = x; curY = y;
    }

    // 最後の点から原点へ戻る
    total += sqrtl(curX * curX + curY * curY);

    /* --- 出力 ---
       小数部分を考慮した出力 */
    printf("%.10Lf\n", total);
    return 0;
}
```
