```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double total = 0.0;  // Use double for floating-point precision

    long long prev_x = 0, prev_y = 0;  // Starting at origin

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long long dx = x - prev_x;
        long long dy = y - prev_y;
        total += sqrt(static_cast<double>(dx * dx + dy * dy));

        prev_x = x;
        prev_y = y;
    }

    // Return to origin
    long long dx = -prev_x;
    long long dy = -prev_y;
    total += sqrt(static_cast<double>(dx * dx + dy * dy));

    // Print with sufficient precision
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
