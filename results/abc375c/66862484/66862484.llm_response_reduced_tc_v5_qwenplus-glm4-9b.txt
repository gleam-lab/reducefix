#include <cmath>
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N + 1);

    // Read the points including the origin
    points[0] = {0, 0};
    for (int i = 1; i <= N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0.0;

    // Calculate the total cost of the path
    for (int i = 1; i <= N; ++i) {
        total_cost += sqrt(pow(points[i].first - points[i - 1].first, 2) + pow(points[i].second - points[i - 1].second, 2));
    }

    // Output the result with proper precision
    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
