
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long curX = 0, curY = 0;   // 莽聫戮氓聹篓盲陆聧莽陆庐茂录聢氓聨聼莽聜鹿茂录聣
    double total = 0.0;             // 莽路聫茫聜鲁茫聜鹿茫聝聢

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long long dx = curX - x;
        long long dy = curY - y;
        total += sqrt((double)(dx * dx + dy * dy));  // 茫聛聯茫聛聯茫聛搂 sqrt() 茫聜聮氓聭录茫聛鲁氓聡潞茫聛聶

        curX = x; curY = y;
    }

    // 忙聹聙氓戮聦茫聛庐莽聜鹿茫聛聥茫聜聣氓聨聼莽聜鹿茫聛赂忙聢禄茫聜聥
    total += sqrt((double)(curX * curX + curY * curY));

    cout.precision(15);
    cout << fixed << total << '\n';
    return 0;
}
```
