```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double curX = 0, curY = 0;   // 現在位置（原点）
    double total = 0;             // 総コスト (double)

    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // √あり！

        curX = x; curY = y;
    }

    // 最後の点から原点へ戻る
    total += sqrt(curX * curX + curY * curY);

    /* --- 出力 ---
       doubleを文字列化して出力すれば誤差 1e-6程度。 */
    printf("%.6f\n", total);
    return 0;
}
```

In the original code, the total cost was being calculated without taking the square root of the sum of squares, which is incorrect as per the problem statement. Additionally, the use of `__int128` for storing the total cost was unnecessary and caused issues with precision. The corrected version uses `double` to store the total cost and calculates the Euclidean distance correctly by including the square root operation.
