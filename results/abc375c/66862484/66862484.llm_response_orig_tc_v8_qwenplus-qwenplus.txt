```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total = 0.0;
    long double curX = 0.0, curY = 0.0; // 現在位置（原点）

    for (int i = 0; i < N; ++i) {
        long double x, y;
        cin >> x >> y;

        long double dx = curX - x;
        long double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x;
        curY = y;
    }

    // 最後の点から原点へ戻る
    total += sqrt(curX * curX + curY * curY);

    // 出力の精度を確保
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
