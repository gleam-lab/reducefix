```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double curX = 0, curY = 0;   // Current position (origin)
    double total = 0;             // Total cost

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // Correctly calculate the Euclidean distance

        curX = x; curY = y;
    }

    // Return from the last point back to the origin
    total += sqrt(curX * curX + curY * curY);

    /* --- Output ---
       Print the total cost with precision up to 10^-6 */
    printf("%.6f\n", total);

    return 0;
}
```
