```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    double curX = 0.0, curY = 0.0;
    double total = 0.0;

    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;

        double dx = curX - x;
        double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);

        curX = x;
        curY = y;
    }

    // Return to origin
    total += sqrt(curX * curX + curY * curY);

    // Set precision high enough to meet error requirements
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
