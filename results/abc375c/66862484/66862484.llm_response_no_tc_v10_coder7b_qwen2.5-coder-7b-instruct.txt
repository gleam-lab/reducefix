#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double curX = 0, curY = 0;   // Current position (origin)
    long double total = 0;             // Total cost

    for (int i = 0; i < N; ++i) {
        long long x, y;
        cin >> x >> y;

        long double dx = curX - x;
        long double dy = curY - y;
        total += sqrt(dx * dx + dy * dy);  // Correctly calculate Euclidean distance

        curX = x; curY = y;
    }

    // Return to the origin from the last point
    total += sqrt(curX * curX + curY * curY);

    /* --- Output ---
       Print the result directly without converting it to a string. */
    printf("%.6Lf\n", total);  // Ensure output matches required precision
    return 0;
}
