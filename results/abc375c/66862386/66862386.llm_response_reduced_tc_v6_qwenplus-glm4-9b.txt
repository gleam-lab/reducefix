#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<long long, long long>> points(N + 1);

    for (int i = 0; i < N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    long double total_cost = 0.0;
    long long Ox = 0, Oy = 0;

    for (int i = 0; i <= N; ++i) {
        if (i == N) {
            total_cost += sqrt(pow(Ox - points[i].first, 2) + pow(Oy - points[i].second, 2));
        } else {
            total_cost += sqrt(pow(points[i].first - points[i + 1].first, 2) + pow(points[i].second - points[i + 1].second, 2));
            Ox = points[i].first;
            Oy = points[i].second;
        }
    }

    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
