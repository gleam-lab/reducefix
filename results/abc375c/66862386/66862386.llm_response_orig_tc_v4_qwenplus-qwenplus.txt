```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total = 0.0;
    long double x = 0, y = 0, prev_x = 0, prev_y = 0;

    for (int i = 0; i < N; ++i) {
        cin >> x >> y;
        long double dx = x - prev_x;
        long double dy = y - prev_y;
        total += sqrt(dx * dx + dy * dy);
        prev_x = x;
        prev_y = y;
    }

    // Return to origin
    long double dx = 0 - prev_x;
    long double dy = 0 - prev_y;
    total += sqrt(dx * dx + dy * dy);

    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
```
