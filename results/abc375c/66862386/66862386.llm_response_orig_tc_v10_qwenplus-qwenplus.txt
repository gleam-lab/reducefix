```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total = 0.0;
    int currX = 0, currY = 0;

    for (int i = 0; i < N; ++i) {
        int x, y;
        cin >> x >> y;
        long double dx = static_cast<long double>(x - currX);
        long double dy = static_cast<long double>(y - currY);
        total += sqrt(dx * dx + dy * dy);
        currX = x;
        currY = y;
    }

    // Return to origin
    long double dx = static_cast<long double>(currX);
    long double dy = static_cast<long double>(currY);
    total += sqrt(dx * dx + dy * dy);

    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
```
