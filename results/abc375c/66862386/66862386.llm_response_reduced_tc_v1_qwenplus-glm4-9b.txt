#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<int, int>> points(N + 1);
    points[0] = {0, 0}; // Start at the origin

    for (int i = 1; i <= N; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    long double total_cost = 0.0;
    for (int i = 1; i <= N; ++i) {
        total_cost += sqrt(powl(points[i].first - points[i - 1].first, 2) + powl(points[i].second - points[i - 1].second, 2));
    }

    total_cost += sqrt(powl(points[N].first - points[0].first, 2) + powl(points[N].second - points[0].second, 2)); // Return to origin

    cout.precision(10);
    cout << fixed << total_cost << endl;

    return 0;
}
