        cin >> currX >> currY;
        long double dx = currX - prevX;
        long double dy = currY - prevY;
        total += sqrt(dx * dx + dy * dy);
        prevX = currX;
        prevY = currY;
    }

    // Return to origin
    long double dx = 0.0 - prevX;
    long double dy = 0.0 - prevY;
    total += sqrt(dx * dx + dy * dy);

    cout.precision(15);
    cout << fixed << total << endl;

    return 0;
}
