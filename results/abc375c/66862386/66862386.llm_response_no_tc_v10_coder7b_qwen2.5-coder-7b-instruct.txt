```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    long double cost = 0;
    long double ans = 0;
    int n;
    int x, y;
    int prev_x = 0, prev_y = 0;

    cin >> n;

    for(int i = 0; i < n; ++i){
        cin >> x >> y;
        cost = sqrt(pow(x - prev_x, 2) + pow(y - prev_y, 2));
        ans += cost;
        prev_x = x;
        prev_y = y;
    }

    cost = sqrt(prev_x * prev_x + prev_y * prev_y);
    ans += cost;

    cout << ans << endl;

    return 0;
}
```
