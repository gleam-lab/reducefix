using namespace std;

int main() {

        long double dy = nowY - nextY;
        ans += sqrtl(dx * dx + dy * dy);
        nowX = nextX;
        nowY = nextY;
    }
    
    long double dx = nowX;
    long double dy = nowY;

    return 0;
}
