
```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<pair<int, int>> points;
    points.emplace_back(0, 0); // Start at the origin

    for (int i = 0; i < N; ++i) {
        int x, y;
        cin >> x >> y;
        points.emplace_back(x, y);
    }

    double total_cost = 0.0;
    for (int i = 1; i <= N; ++i) {
        total_cost += sqrt(pow(points[i].first - points[i - 1].first, 2) + pow(points[i].second - points[i - 1].second, 2));
    }

    // The last move back to the origin
    total_cost += sqrt(pow(points[N].first - points[0].first, 2) + pow(points[N].second - points[0].second, 2));

    // Set the precision for the output
    cout << fixed << setprecision(9) << total_cost << endl;

    return 0;
}
```
