```cpp
#include <iostream>
#include <vector>
#include <cmath>

int main() {
    int n;
    std::cin >> n;
    std::vector<std::pair<int, int>> points(n);
    
    // Read all points
    for (int i = 0; i < n; ++i) {
        std::cin >> points[i].first >> points[i].second;
    }

    // Calculate the total cost
    double total_cost = 0;
    for (size_t i = 0; i < points.size(); ++i) {
        if (i > 0) {
            total_cost += std::sqrt(std::pow(points[i].first - points[i - 1].first, 2) +
                                   std::pow(points[i].second - points[i - 1].second, 2));
        }
    }
    // Add the cost of returning to the origin
    total_cost += std::sqrt(std::pow(points[0].first, 2) + std::pow(points[0].second, 2));

    // Print the result with high precision
    std::cout << std::fixed << std::setprecision(15) << total_cost << std::endl;

    return 0;
}
```

This solution reads the number of points and their coordinates, calculates the total distance traveled by summing up the Euclidean distances between consecutive points and back to the origin, and prints the result with high precision to meet the required tolerance.
