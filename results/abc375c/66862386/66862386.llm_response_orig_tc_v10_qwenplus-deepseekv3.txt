#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total_distance = 0.0;
    int prev_x = 0, prev_y = 0;

    for (int i = 0; i < N; ++i) {
        int x, y;
        cin >> x >> y;
        long double dx = x - prev_x;
        long double dy = y - prev_y;
        total_distance += sqrtl(dx * dx + dy * dy);
        prev_x = x;
        prev_y = y;
    }

    // Add the distance back to the origin
    long double dx = 0 - prev_x;
    long double dy = 0 - prev_y;
    total_distance += sqrtl(dx * dx + dy * dy);

    // Output with sufficient precision (20 decimal places)
    cout << fixed << setprecision(20) << total_distance << endl;

    return 0;
}

2. **Distance Calculation**: For each point, it calculates the Euclidean distance from the previous point (starting from the origin) and accumulates the total distance.
3. **Return to Origin**: After visiting all points, it calculates the distance from the last point back to the origin and adds it to the total distance.
4. **Output**: The total distance is printed with a fixed precision of 20 decimal places to ensure the output matches the required format and avoids scientific notation.
