
```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    int x, y;
    int startX = 0, startY = 0;
    double totalCost = 0.0;
    
    for (int i = 0; i < N; ++i) {
        cin >> x >> y;
        double distance = sqrt(pow(x - startX, 2) + pow(y - startY, 2));
        totalCost += distance;
        startX = x;
        startY = y;
    }
    
    // Return to the origin
    double returnDistance = sqrt(pow(startX, 2) + pow(startY, 2));
    totalCost += returnDistance;
    
    cout.precision(10);
    cout << fixed << totalCost;
    
    return 0;
}
```
