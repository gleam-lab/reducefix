#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main() {
    long long n;
    cin >> n;
    
    vector<pair<long long, long long>> points(n);
    for(long long i = 0; i < n; ++i){
        cin >> points[i].first >> points[i].second;
    }

    double total_cost = 0;
    for(long long i = 1; i <= n; ++i){
        total_cost += sqrt(pow(points[i % n].first - points[(i - 1) % n].first, 2) + 
                           pow(points[i % n].second - points[(i - 1) % n].second, 2));
    }

    printf("%.10f\n", total_cost);
    return 0;
}
