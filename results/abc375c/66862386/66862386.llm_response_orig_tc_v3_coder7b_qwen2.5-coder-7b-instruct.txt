```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    vector<pair<long long, long long>> points(n);
    for(int i = 0; i < n; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    // Starting from the origin
    double cost = 0;
    for(int i = 0; i < n; ++i) {
        long long dx = points[i].first;
        long long dy = points[i].second;
        cost += sqrt(dx * dx + dy * dy);
    }

    // Return to the origin
    cost += sqrt(points[n-1].first * points[n-1].first + points[n-1].second * points[n-1].second);

    cout << setprecision(15) << cost << '\n';

    return 0;
}
```
