```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    long double x[n], y[n];
    for(int i = 0; i < n; ++i) {
        cin >> x[i] >> y[i];
    }

    long double cost = 0;
    for(int i = 1; i < n; ++i) {
        cost += sqrtl(powl(x[i] - x[i-1], 2) + powl(y[i] - y[i-1], 2));
    }

    // Return to origin
    cost += sqrtl(powl(x[0], 2) + powl(y[0], 2));

    cout << fixed << setprecision(10) << cost << endl;

    return 0;
}
```
