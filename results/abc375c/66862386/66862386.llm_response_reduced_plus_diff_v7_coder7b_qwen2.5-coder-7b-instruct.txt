        // Update current position
        nowX = nextX;
        nowY = nextY;

        // Accumulate total cost
        ans += cost;
    }

    // Return to the origin
    cost = std::sqrtl(std::powl(nowX, 2.0L) + std::powl(nowY, 2.0L));
    ans += cost;

    // Print the result with sufficient precision

    return 0;
}
