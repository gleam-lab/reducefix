
```cpp
#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<pair<int, int>> points(N);
    long double totalCost = 0.0;

    for (int i = 0; i < N; i++) {
        cin >> points[i].first >> points[i].second;
    }

    // Starting from the origin (0,0)
    pair<int, int> origin = {0, 0};

    // Calculate cost from the origin to the first point
    totalCost += sqrt(pow(double(points[0].first - origin.first), 2) + pow(double(points[0].second - origin.second), 2));

    // Calculate the cost of traveling between all points
    for (int i = 0; i < N - 1; i++) {
        totalCost += sqrt(pow(double(points[i + 1].first - points[i].first), 2) + pow(double(points[i + 1].second - points[i].second), 2));
    }

    // Calculate the cost from the last point back to the origin
    totalCost += sqrt(pow(double(origin.first - points[N - 1].first), 2) + pow(double(origin.second - points[N - 1].second), 2));

    cout.precision(10);
    cout << fixed << totalCost << endl;

    return 0;
}
```

This code calculates the total travel cost for Takahashi on a 2D plane by first moving from the origin to the first point, then between all consecutive points, and finally back to the origin. It uses `pair<int, int>` to store the points as coordinates and uses `sqrt` from `cmath` to calculate the Euclidean distance. The output is formatted to 10 decimal places to meet the precision requirements.
