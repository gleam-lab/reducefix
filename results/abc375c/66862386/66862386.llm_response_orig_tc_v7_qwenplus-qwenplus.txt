```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long double total = 0.0;
    int prev_x = 0, prev_y = 0;

    for (int i = 0; i < N; ++i) {
        int x, y;
        cin >> x >> y;
        long double dx = static_cast<long double>(x - prev_x);
        long double dy = static_cast<long double>(y - prev_y);
        total += sqrt(dx * dx + dy * dy);
        prev_x = x;
        prev_y = y;
    }

    // Return to origin
    long double dx = static_cast<long double>(prev_x);
    long double dy = static_cast<long double>(prev_y);
    total += sqrt(dx * dx + dy * dy);

    // Print with sufficient precision
    cout.precision(15);
    cout << fixed << total << '\n';

    return 0;
}
```
