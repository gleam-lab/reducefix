```python
import math

N = int(input())
X = 0
Y = 0
result = 0.0

# Start from origin (0,0)
prev_x, prev_y = 0, 0

for _ in range(N):
    line = input().split()
    x = int(line[0])
    y = int(line[1])
    
    # Add distance from previous point to current point
    dx = x - prev_x
    dy = y - prev_y
    result += math.sqrt(dx*dx + dy*dy)
    
    # Update previous point
    prev_x, prev_y = x, y

# Add distance from last point back to origin
dx = 0 - prev_x
dy = 0 - prev_y
result += math.sqrt(dx*dx + dy*dy)

print(result)
```
