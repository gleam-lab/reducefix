#include <iostream>
#include<algorithm>
#include<queue>
#include<map>
#include<vector>
#include<cstring>
using namespace std;
#define int long long
#define endl '\n'
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, q;
int a[N];

bool compare(int &a, int &b){
    return abs(a)<abs(b);
}

signed main()
{
    ios::sync_with_stdio(false);
    
    cin >> n >> q;
    for (int i = 1; i <= n; i++)cin >> a[i];
    sort(a + 1, a + n + 1);
    
    while (q--) {
        int x, k;
        cin >> x >> k;
        vector<int> distances(n);
        for(int i = 1;i<=n;i++){
            distances[i-1] = abs(a[i]-x);
        }
        sort(distances.begin(),distances.end());
        cout << distances[k-1] << endl;
    }
    
    return 0;
}
