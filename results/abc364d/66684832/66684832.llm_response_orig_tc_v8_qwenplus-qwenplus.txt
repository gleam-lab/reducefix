#include <bits/stdc++.h>
using namespace std;

#define int long long
const int INF = 1e18;

int n, q;
int a[200005];

// Binary search to find the k-th smallest distance
int query(int b, int k) {
    int left = 0, right = 1e8;
    int answer = right;

    while (left <= right) {
        int mid = (left + right) / 2;
        int l = lower_bound(a, a + n, b - mid) - a;
        int r = upper_bound(a, a + n, b + mid) - a;
        if (r - l >= k) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return answer;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> q;
    for (int i = 0; i < n; ++i)
        cin >> a[i];
    sort(a, a + n);

    while (q--) {
        int b, k;
        cin >> b >> k;
        cout << query(b, k) << '\n';
    }

    return 0;
}
