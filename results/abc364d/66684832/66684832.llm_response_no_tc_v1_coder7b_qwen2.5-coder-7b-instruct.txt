#include <iostream>
#include<algorithm>
#include<queue>
#include<map>
#include<vector>
#include<cstring>
using namespace std;
#define int long long
#define endl '\n'
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, q;
int a[N];

int findans(int pos, int k) {
    int l = 0, r = 1e9;
    while (l < r) {
        int mid = (l + r) >> 1;
        int count = 0;
        for (int i = 1; i <= n; ++i) {
            if (abs(a[i] - pos) <= mid) {
                ++count;
                if (count == k) break;
            }
        }
        if (count == k) r = mid;
        else l = mid + 1;
    }
    return l;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 1; i <= n; i++) cin >> a[i];
    sort(a + 1, a + n + 1);

    while (q--) {
        int x, k;
        cin >> x >> k;
        cout << findans(x, k) << endl;
    }

    return 0;
}
