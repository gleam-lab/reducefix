#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define int long long

const int N = 1e5 + 5;

int n, q;
int a[N];

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a, a + n);

    for (int i = 0; i < q; ++i) {
        int b, k;
        cin >> b >> k;

        // Binary search to find the smallest window around b that contains at least k elements
        int left = 0, right = 1e8;
        while (left < right) {
            int mid = (left + right) / 2;

            // Find the index of first element >= (b - mid)
            int low = lower_bound(a, a + n, b - mid) - a;
            // Find the index of first element > (b + mid)
            int high = lower_bound(a, a + n, b + mid + 1) - a;

            if (high - low >= k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }

        cout << left << '\n';
    }

    return 0;
}
