#include <iostream>
#include<algorithm>
#include<queue>
#include<map>
#include<vector>
#include<cstring>
using namespace std;

typedef long long ll;

vector<ll> points;
ll n, q;

// Function to calculate the k-th nearest point to a given point
ll kthNearest(ll target, int k) {
    vector<ll> distances(points.size());
    for (int i = 0; i < points.size(); ++i) {
        distances[i] = abs(target - points[i]);
    }
    nth_element(distances.begin(), distances.begin() + k - 1, distances.end());
    return distances[k - 1];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    points.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i];
    }

    for (int i = 0; i < q; ++i) {
        ll x; int k;
        cin >> x >> k;
        cout << kthNearest(x, k) << '\n';
    }

    return 0;
}

This solution works as follows:

1. **Reading Input**: We read `n` and `q` and then read the coordinates of the `n` points into a vector called `points`.

2. **K-th Nearest Point Calculation**:
   - For each query, we need to find the `k`-th smallest distance from a given point `x`.
   - We create a vector `distances` to store the absolute differences between `x` and each point in `points`.
   - We use the `nth_element` function from the `<algorithm>` library to find the `k`-th smallest element in `distances` without fully sorting it. This ensures efficient computation even for large inputs.

3. **Output**: For each query, we print the `k`-th smallest distance.

This approach ensures that the solution runs efficiently within the given constraints. The time complexity is \(O(Q \cdot (N + \log N))\) due to the `nth_element` call inside the loop, which is efficient enough for the problem constraints.
