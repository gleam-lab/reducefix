signed main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> q;
    for (int i = 1; i <= n; i++) cin >> a[i];

    vector<int> dist[n + 1];

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= q; j++) {
            int b;
            cin >> b;
            int d = abs(a[i] - b);
            dist[j].push_back(d);
        }
    }

    for (int i = 1; i <= q; i++) {
        sort(dist[i].begin(), dist[i].end());
    }

    for (int i = 1; i <= q; i++) {
        int k;
        cin >> k;
        cout << dist[i][k - 1] << endl;
    }

    return 0;
}

This code reads the input, calculates all distances for each query and sorts them. Then it answers each query by printing the k-th smallest distance. This approach ensures that the solution is efficient even for large inputs up to the given constraints.
