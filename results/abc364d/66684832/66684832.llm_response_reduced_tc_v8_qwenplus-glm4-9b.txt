#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;

int main() {
    int n, q;
    cin >> n >> q;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    vector<pair<int, int>> queries;
    for (int i = 0; i < q; ++i) {
        int b, k;
        cin >> b >> k;
        queries.emplace_back(b, k);
    }

    sort(queries.begin(), queries.end());

    vector<int> results(q);
    for (int i = 0; i < q; ++i) {
        int b = queries[i].first;
        int k = queries[i].second;
        int lower = -inf, upper = inf;

        int ans = 0;
        while (lower < upper) {
            int mid = (lower + upper + 1) / 2;
            int count = 0;
            for (int j = 0; j < n; ++j) {
                if (abs(b - a[j]) <= mid) {
                    count++;
                }
            }

            if (count >= k) {
                upper = mid - 1;
                ans = mid;
            } else {
                lower = mid;
            }
        }
        results[i] = ans;
    }

    for (int result : results) {
        cout << result << endl;
    }

    return 0;
}
