#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define int long long
#define endl '\n'

const int N = 2e5 + 9;

int n, q;
int a[N];

// Check if there are at least k points within distance 'dx' from position 'p'
bool check(int p, int dx, int k) {
    // Find the first index >= (p - dx)
    int ln = lower_bound(a + 1, a + n + 1, p - dx) - a;
    // Find the first index > (p + dx), so rn-1 is the last <= (p + dx)
    int rn = upper_bound(a + 1, a + n + 1, p + dx) - a;
    return (rn - ln) >= k;
}

// Binary search for the smallest distance such that at least k points are within that distance
int findans(int pos, int k) {
    int l = 0, r = 2e8; // Maximum possible distance is 2*10^8
    int ans = r;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(pos, mid, k)) {
            ans = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return ans;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    sort(a + 1, a + n + 1);
    
    while (q--) {
        int x, k;
        cin >> x >> k;
        cout << findans(x, k) << endl;
    }

    return 0;
}
