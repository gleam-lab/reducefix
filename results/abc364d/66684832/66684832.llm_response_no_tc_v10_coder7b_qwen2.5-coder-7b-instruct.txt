#include <iostream>
#include<algorithm>
#include<queue>
#include<map>
#include<vector>
#include<cstring>
using namespace std;
#define int long long
#define endl '\n'
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, q;
int a[N];

int findans(int pos, int k) {
    vector<int> dist(n);
    for(int i = 0; i < n; ++i) {
        dist[i] = abs(a[i] - pos);
    }
    sort(dist.begin(), dist.end());
    return dist[k-1];
}

signed main()
{
    ios::sync_with_stdio(false);
    
    cin >> n >> q;
    for (int i = 0; i < n; i++)cin >> a[i];
    
    while (q--) {
        int x, k;
        cin >> x >> k;
        cout << findans(x, k) << endl;
    }
    
    return 0;
}
