#include <iostream>

const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, m, cnt, ans, q;

bool vis[N];
bool check(int p, int dx, int k) {
    int ln = lower_bound(a + 1, a + n + 1, p - dx) - a;
    int rn = upper_bound(a + 1, a + n + 1, p + dx) - a;
    return rn - ln >= k;
}
int findans(int pos, int k) {
    int l = 0, r = 1e8;
    while (l < r) {
        int mid = (l + r) >> 1;
        if (check(pos, mid, k)) r = mid;
        else l = mid + 1;
    }
    return l;
}
