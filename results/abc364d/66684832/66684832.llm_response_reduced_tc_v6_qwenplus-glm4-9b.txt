#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define int long long
const int N = 2e5 + 9;

int main() {
    int n, q;
    cin >> n >> q;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    vector<vector<pair<int, int>>> queries(q);
    for (int i = 0; i < q; i++) {
        int b, k;
        cin >> b >> k;
        queries[i].push_back({b, k});
        queries[i].push_back({b, -k}); // To handle both directions
    }
    
    // Sort queries by b coordinate and k value
    sort(queries.begin(), queries.end(), [](const vector<pair<int, int>>& a, const vector<pair<int, int>>& b) {
        if (a[0].first != b[0].first) return a[0].first < b[0].first;
        return a[0].second < b[0].second;
    });
    
    // Sort a array
    sort(a.begin(), a.end());
    
    // Binary search to find the k-th closest point
    int last_index = 0;
    for (const auto& q : queries) {
        int b = q[0].first;
        int k = q[0].second;
        
        int l = last_index, r = n;
        while (r - l > 1) {
            int mid = (l + r) / 2;
            if (k < 0 ? a[mid] <= b : a[mid] >= b) {
                r = mid;
            } else {
                l = mid;
            }
        }
        
        int ans = 0;
        if (k < 0) {
            if (l >= 0 && r < n) {
                ans = min(abs(a[l] - b), abs(a[r] - b));
            } else if (l >= 0) {
                ans = abs(a[l] - b);
            } else {
                ans = abs(a[r] - b);
            }
        } else {
            if (l >= 0 && r < n) {
                ans = min(abs(a[l] - b), abs(a[r] - b));
            } else if (l >= 0) {
                ans = abs(a[l] - b);
            } else {
                ans = abs(a[r] - b);
            }
        }
        
        cout << ans << '\n';
        
        if (k < 0) {
            last_index = l;
        } else {
            last_index = r;
        }
    }
    
    return 0;
}
