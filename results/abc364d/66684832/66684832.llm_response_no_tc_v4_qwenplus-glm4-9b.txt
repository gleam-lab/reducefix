#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 2e5 + 9;

int n, q;
vector<int> a(N);
vector<pair<int, int>> queries(N); // b_j, k_j

int binary_search(int x, int k) {
    int left = -1, right = n;
    while (left + 1 < right) {
        int mid = left + (right - left) / 2;
        // Calculate the distance between A[mid] and x
        int distance = abs(a[mid] - x);
        // If the distance to A[mid] is the same, we need to find the closest one to B[j]
        if (distance == abs(a[mid - 1] - x)) {
            mid--;
        }
        // Check if the k-th closest point has been found
        if (mid - k >= 0) {
            right = mid;
        } else {
            left = mid;
        }
    }
    // The point at index left is the k-th closest to point B[j]
    return a[left];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> q;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    for (int i = 0; i < q; i++) {
        int x, k;
        cin >> x >> k;
        queries[i] = {x, k};
    }
    sort(queries.begin(), queries.end()); // Sort queries by b_j

    for (int i = 0; i < q; i++) {
        int b_j = queries[i].first;
        int k_j = queries[i].second;
        // Find the k_j-th closest point A_i to B_j
        int result = binary_search(b_j, k_j);
        cout << abs(result - b_j) << "\n";
    }

    return 0;
}
