// Function to find the k-th smallest distance using binary search
int findKthSmallestDistance(int target, int k) {
    int left = 0, right = 2 * 1e8; // Maximum possible distance
    while (left < right) {
        int mid = left + (right - left) / 2;
        int count = 0;
        
        // Count how many points are within the range [target-mid, target+mid]
        for (int i = 1; i <= n; ++i) {
            if (abs(target - a[i]) <= mid) {
                count++;
            }
        }
        
        if (count >= k) {
            right = mid; // Search in the left half
        } else {
            left = mid + 1; // Search in the right half
        }
    }
    return left;
}

    sort(a + 1, a + n + 1); // Sort points A

    while (q--) {
        int x, k;
        cin >> x >> k;
        cout << findKthSmallestDistance(x, k) << endl; // Output the k-th smallest distance
    }
