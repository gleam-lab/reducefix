#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
#define int long long
#define endl '\n'

const int N = 2e5 + 9;

int n, q;
int a[N];

// Binary search to find the smallest distance such that there are at least k points within that distance from pos
int find_kth_closest_distance(int pos, int k) {
    int left = 0, right = 2e14; // max possible distance is ~2*1e8 * 1e8
    while (left < right) {
        int mid = (left + right) / 2;
        // Find first index where a[i] >= pos - mid
        int l = lower_bound(a + 1, a + n + 1, pos - mid) - a;
        // Find first index where a[i] > pos + mid
        int r = upper_bound(a + 1, a + n + 1, pos + mid) - a;
        int count = r - l;
        if (count >= k) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    return left;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    sort(a + 1, a + n + 1);

    while (q--) {
        int b, k;
        cin >> b >> k;
        cout << find_kth_closest_distance(b, k) << endl;
    }

    return 0;
}
