    return x.b < y.b;
}

    vector<int> b_index(n);
    for (int i = 0, j = 0; i < q; ++i) {
        while (j < n && a[j] < queries[i].b) {
            b_index[j++] = i;
        }
    }

    for (int i = 0; i < q; ++i) {

        long long b = queries[i].b, k = queries[i].k;
