```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, q;
    cin >> n >> q;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    vector<pair<int, int>> queries(q);
    for (int i = 0; i < q; ++i) {
        int x, k;
        cin >> x >> k;
        queries[i] = {x, k};
    }

    // Store all distances
    vector<int> distances;
    for (int i = 0; i < n; ++i) {
        for (auto& [x, k] : queries) {
            distances.push_back(abs(a[i] - x));
        }
    }

    // Sort all distances
    sort(distances.begin(), distances.end());

    // Answer each query
    for (auto& [x, k] : queries) {
        int kth_smallest_distance = distances[(k - 1) * n];
        cout << kth_smallest_distance << endl;
    }

    return 0;
}
```

### Explanation:
1. **Input Reading**: We read the number of points `n` and `q`, followed by the coordinates of points `A` and the queries.
2. **Storing Distances**: For each query `(x, k)`, we calculate the absolute difference between `x` and each point `A_i` and store these distances in a vector.
3. **Sorting Distances**: We sort the vector of distances.
4. **Answering Queries**: For each query, we directly access the k-th smallest distance from the sorted list and print it.

This approach ensures that we handle the constraints efficiently and correctly determine the k-th smallest distance for each query.
