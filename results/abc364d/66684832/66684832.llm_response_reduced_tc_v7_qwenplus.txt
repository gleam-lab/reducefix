```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define int long long
const int inf = 1e18;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    vector<int> a(N);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (Q--) {
        int b, k;
        cin >> b >> k;
        
        // Create a vector of distances from B to each A_i
        vector<int> dist;
        for (int i = 0; i < N; i++) {
            dist.push_back(abs(a[i] - b));
        }
        
        // Sort distances and get the k-th smallest
        sort(dist.begin(), dist.end());
        cout << dist[k-1] << '\n';
    }

    return 0;
}
```
