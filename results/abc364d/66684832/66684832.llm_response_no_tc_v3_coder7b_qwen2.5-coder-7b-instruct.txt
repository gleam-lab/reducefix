long long binarySearch(long long target, int k) {
    long long left = 0, right = 1e8, mid;
    while (left < right) {
        mid = (left + right) / 2;
        int count = 0;
        for (int i = 0; i < target; ++i) {

                count++;
            }
        }
        if (count >= k) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    return left;
}
