#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define int long long
#define endl '\n'

const int N = 2e5 + 9;
int n, q;
int a[N];

// Function to count how many points are within distance 'dist' from point 'pos'
int countWithinDistance(int pos, int dist) {
    // Find the leftmost point >= (pos - dist)
    int leftIdx = lower_bound(a + 1, a + n + 1, pos - dist) - a;
    // Find the rightmost point <= (pos + dist)
    int rightIdx = upper_bound(a + 1, a + n + 1, pos + dist) - a - 1;
    
    return rightIdx - leftIdx + 1;
}

// Binary search for the k-th smallest distance from pos to any A_i
int findKthDistance(int pos, int k) {
    int left = 0;
    int right = 200000000; // Maximum possible distance (2 * 10^8)
    
    while (left < right) {
        int mid = (left + right) / 2;
        if (countWithinDistance(pos, mid) >= k) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    
    return left;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    sort(a + 1, a + n + 1);
    
    for (int i = 0; i < q; i++) {
        int b, k;
        cin >> b >> k;
        cout << findKthDistance(b, k) << endl;
    }
    
    return 0;
}
