#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Sort A for binary search purposes
    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        int b;
        int k;
        cin >> b >> k;

        // Binary search on distance
        int low = 0;
        int high = 1e8 + 5; // Max possible distance in constraints

        while (low < high) {
            int mid = (low + high) / 2;

            // Find how many points are within distance `mid` of b
            int left = lower_bound(A.begin(), A.end(), b - mid) - A.begin();
            int right = upper_bound(A.begin(), A.end(), b + mid) - A.begin();

            if (right - left >= k) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        cout << low << "\n";
    }
}

int main() {
    solve();
    return 0;
}
