#include <bits/stdc++.h>
#define endl "\n"
using namespace std;

const int N = 1e5 + 5;
long long a[N], b[N], dist[N];

bool check(int mid, int x, int y) {
    int pos1 = min(x + mid, N - 1);
    int pos2 = max(x - mid - 1, -1);
    if (pos1 >= 0) {
        return dist[pos1] + y >= dist[pos2] + y;
    }
    return dist[pos1] - mid <= dist[pos2] + y;
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i] = abs(a[i]);
    }
    for (int i = 0; i < q; ++i) {
        cin >> b[i] >> dist[i];
        b[i] = abs(b[i]);
    }
    
    sort(a, a + n);
    sort(dist, dist + q);
    
    vector<pair<int, int>> queries(q);
    for (int i = 0; i < q; ++i) {
        queries[i] = {dist[i], i};
    }
    sort(queries.begin(), queries.end());
    
    for (auto &query : queries) {
        long long d = query.first;
        int idx = query.second;
        int lo = 0, hi = n - 1;
        while (lo < hi) {
            int mid = lo + (hi - lo + 1) / 2;
            if (check(a[mid], b[idx], d)) {
                lo = mid;
            } else {
                hi = mid - 1;
            }
        }
        cout << a[lo] + b[idx] << endl;
    }
    
    return 0;
}
