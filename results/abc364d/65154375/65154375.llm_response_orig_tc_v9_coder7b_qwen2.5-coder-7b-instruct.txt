```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> a;

int binary_search(int b, int k) {
    int left = 0, right = a.size() - 1;
    while(left < right) {
        int mid = left + (right - left) / 2;
        if(a[mid] - b < k) left = mid + 1;
        else right = mid;
    }
    return abs(b - a[left]);
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    
    int n, q; cin >> n >> q;
    a.resize(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    for(int i = 0; i < q; i++) {
        int b, k; cin >> b >> k;
        cout << binary_search(b, k) << '\n';
    }
    
    return 0;
}
```
This solution uses a binary search approach to find the k-th smallest distance for each query, which is more efficient than sorting distances for each query individually. This reduces the time complexity significantly, especially for large inputs.
