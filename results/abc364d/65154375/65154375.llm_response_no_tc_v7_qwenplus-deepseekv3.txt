#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {

    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b, k;
        cin >> b >> k;
        vector<ll> distances(N);
        for (int i = 0; i < N; ++i) {
            distances[i] = abs(A[i] - b);
        }
        nth_element(distances.begin(), distances.begin() + k - 1, distances.end());

    }
    return 0;
}
