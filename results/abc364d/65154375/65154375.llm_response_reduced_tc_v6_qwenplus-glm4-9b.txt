#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

const int MAX_N = 100000 + 5;
int points[MAX_N];

void solve() {
    int N, Q;
    cin >> N >> Q;
    vector<pair<long long, int>> distances(N);
    
    for (int i = 0; i < N; ++i) {
        cin >> points[i];
        distances[i] = {abs(points[i]), i};
    }
    
    sort(distances.begin(), distances.end());
    
    while (Q--) {
        long long B, k;
        cin >> B >> k;
        auto it = upper_bound(distances.begin(), distances.end(), make_pair(B, LLONG_MAX));
        
        if (it == distances.begin()) {
            cout << 0 << endl;
        } else {
            --it;
            long long d = distances[it].first;
            cout << d * d * d << endl; //ç«æ¹è·ç¦»
        }
    }
}

int main() {
    solve();
    return 0;
