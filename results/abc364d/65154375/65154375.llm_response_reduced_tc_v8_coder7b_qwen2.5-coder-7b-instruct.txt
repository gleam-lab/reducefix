#include<bits/stdc++.h>
#define endl "\n"
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int N = 2e5 + 5;
vector<int> a;

bool check(int mid, int x, int y){
    auto it = lower_bound(a.begin(), a.end(), x + mid);
    if(it != a.end() && *it - (*prev(it)) >= y){
        return true;
    }
    return false;
}

void solve(void) {
    int n, q; cin >> n >> q;
    a.resize(2*N);
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a[x + N] ++;
    }
    sort(a.begin(), a.end());
    for(int i = 1; i < a.size(); i++){
        a[i] += a[i - 1];
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 2e8;
        while(r > l){
            int mid = l + r >> 1;
            if(check(mid, x + N, y)) r = mid;
            else l = mid + 1;
        }
        cout << r << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
