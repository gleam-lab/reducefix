#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAX_OFFSET = 2e8;
const int MAX_SIZE = 4 * MAX_OFFSET + 10; // Sufficient size for frequency array

// Coordinate compression helper
int getCompressedIndex(ll x) {
    return x + MAX_OFFSET;
}

// Binary search to find smallest distance d where at least k elements are within range
ll findKthSmallestDistance(vector<int>& sortedA, int b, int k) {
    ll left = 0, right = MAX_OFFSET;
    ll answer = -1;

    while (left <= right) {
        ll mid = (left + right) / 2;
        // Find lower and upper bounds around b
        int low = b - mid;
        int high = b + mid;

        // Binary search for the count in sortedA between low and high
        auto l = lower_bound(sortedA.begin(), sortedA.end(), low);
        auto r = upper_bound(sortedA.begin(), sortedA.end(), high);

        int count = r - l;
        if (count >= k) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return answer;
}

void solve() {
    int N, Q;
    cin >> N >> Q;

    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        int b;
        int k;
        cin >> b >> k;

        ll result = findKthSmallestDistance(A, b, k);
        cout << result << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    solve();

    return 0;
}
