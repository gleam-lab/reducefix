```cpp
#include <bits/stdc++.h>
using namespace std;

vector<int> a;
vector<pair<int, int>> queries;

void preprocess() {
    sort(a.begin(), a.end());
}

int query(int target) {
    auto it = lower_bound(a.begin(), a.end(), target);
    return it - a.begin();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    a.resize(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    preprocess();

    for(int i = 0; i < q; ++i) {
        int b, k;
        cin >> b >> k;
        queries.push_back({b, k});
    }

    vector<long long> answers(q);
    for(int i = 0; i < q; ++i) {
        int b = queries[i].first;
        int k = queries[i].second;
        int idx = query(b);
        answers[i] = abs(b - a[idx]);
    }

    for(auto ans : answers) {
        cout << ans << '\n';
    }

    return 0;
}
```
