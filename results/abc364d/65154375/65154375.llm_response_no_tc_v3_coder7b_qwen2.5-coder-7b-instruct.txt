```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 100010;
vector<ll> a(N);

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; i++) {
        ll x; cin >> x;
        a[x]++;
    }
    partial_sum(a.begin(), a.end(), a.begin());
    for(int i = 0; i < q; i++) {
        ll x, k; cin >> x >> k;
        ll pos = lower_bound(a.begin(), a.end(), k) - a.begin();
        cout << abs(pos - x) << '\n';
    }
}

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    solve();
    return 0;
}
```

This solution corrects several issues with the original code:
1. It uses `partial_sum` to calculate prefix sums of distances efficiently instead of manually maintaining them.
2. It uses `lower_bound` to find the position of the k-th smallest distance directly without sorting.
3. The logic to calculate the distance is simplified and corrected.
4. The input/output handling is streamlined for better readability and performance.
