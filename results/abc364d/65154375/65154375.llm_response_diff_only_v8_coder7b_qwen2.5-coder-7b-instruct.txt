#include<bits/stdc++.h>
#define endl "\n"
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int MAXN = 100005;
vector<int> a[MAXN], b(MAXN);

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; i++) {
        int x; cin >> x;
        a[i].push_back(abs(x));
    }
    sort(a, a + n);
    for(int i = 0; i < q; i++) {
        int x, k; cin >> x >> k;
        auto it = upper_bound(a, a + n, abs(x));
        if(it == a) cout << "0" << endl;
        else if(it - a < k) cout << "INF" << endl;
        else cout << *it - abs(x) << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    solve();
    return 0;
}
