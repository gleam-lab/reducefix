However, this approach would be too slow for large inputs (O(Q*N) time complexity). Here's an optimized solution that uses binary search for each query:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        int b, k;
        cin >> b >> k;
        
        int low = 0, high = 2e8;
        int answer = 0;
        
        while (low <= high) {
            int mid = low + (high - low) / 2;
            auto left = lower_bound(A.begin(), A.end(), b - mid);
            auto right = upper_bound(A.begin(), A.end(), b + mid);
            int count = distance(left, right);
            
            if (count >= k) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        
        cout << answer << '\n';
    }
    
    return 0;
}
```

This optimized solution:
1. Sorts the array A once (O(N log N))
2. For each query, performs binary search on possible distances (O(log MAX_DIST))
3. Uses lower_bound and upper_bound to count points within distance mid (O(log N) each)
4. Total complexity is O(N log N + Q log MAX_DIST log N), which should be efficient enough for the constraints

The key improvement is using binary search to find the smallest distance where at least k points are within that distance from b, rather than calculating all distances for each query. This approach is much more efficient for large inputs.
