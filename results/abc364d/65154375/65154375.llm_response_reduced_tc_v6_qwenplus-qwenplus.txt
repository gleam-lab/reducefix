#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query(ll b, int k, int n) {
    int pos = lower_bound(a, a + n, b) - a;
    
    // Use two pointers to find k closest elements
    int l = pos - 1, r = pos;
    multiset<ll> k_closest;
    
    while (k--) {
        ll left_dist = (l >= 0) ? abs(a[l] - b) : LLONG_MAX;
        ll right_dist = (r < n) ? abs(a[r] - b) : LLONG_MAX;
        
        if (left_dist <= right_dist) {
            k_closest.insert(left_dist);
            l--;
        } else {
            k_closest.insert(right_dist);
            r++;
        }
    }
    
    return *k_closest.rbegin(); // Return the largest distance in the set (k-th smallest)
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    
    sort(a, a + N);
    
    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;
        cout << query(b, k, N) << "\n";
    }
    
    return 0;
}
