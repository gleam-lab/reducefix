#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 2e5 + 5;
vector<int> a(N);

int main() {
    int n, q;
    cin >> n >> q;
    a.resize(N);
    for (int i = 0; i < n; ++i) {
        int x;
        cin >> x;
        a[x] = 1; // Increment the count at position x
    }
    for (int i = 1; i < N; ++i) {
        a[i] += a[i - 1]; // Compute prefix sums
    }
    while (q--) {
        int x, k;
        cin >> x >> k;
        int l = 0, r = N - 1;
        while (l < r) {
            int mid = l + (r - l + 1) / 2;
            if (a[mid] >= k) {
                r = mid;
            } else {
                l = mid;
            }
        }
        // Calculate the minimum distance to the point b_j
        int ans = (l - x + 1) * (l - x + 1); // distance^2
        if (l > x) {
            // Check if moving left decreases the distance
            int left = l - 1;
            if (left >= 0 && a[left] >= k) {
                ans = min(ans, (left - x + 1) * (left - x + 1));
            }
        }
        if (l < N - 1 && a[l + 1] >= k) {
            // Check if moving right decreases the distance
            int right = l + 1;
            ans = min(ans, (right - x + 1) * (right - x + 1));
        }
        cout << ans << '\n';
    }
    return 0;
}
