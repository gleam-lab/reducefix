```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        int l = 0, r = N - 1;
        while (l < r) {
            int mid = (l + r) / 2;
            if (A[mid] >= b) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        int pos = l;
        vector<ll> distances;
        int left = max(0, pos - k);
        int right = min(N - 1, pos + k);
        for (int i = left; i <= right; ++i) {
            distances.push_back(abs(A[i] - b));
        }
        sort(distances.begin(), distances.end());
        cout << distances[k - 1] << '\n';
    }
    return 0;
}
```

2. **Sorting**: The array A is sorted to facilitate efficient distance calculations.
3. **Query Processing**: For each query, the code finds the position in the sorted array A where the query point B_j would fit using binary search. It then collects distances from B_j to nearby points in A, sorts these distances, and outputs the k-th smallest distance.
4. **Efficiency**: The sorting step ensures that each query can be processed in logarithmic time relative to the size of the relevant window around the query point, making the solution efficient for large inputs.
