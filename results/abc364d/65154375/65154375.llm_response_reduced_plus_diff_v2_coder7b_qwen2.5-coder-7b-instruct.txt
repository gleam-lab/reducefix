```cpp
#include <bits/stdc++.h>

using namespace std;

vector<int> A;
vector<pair<int,int>> queries;
vector<int> ans;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    
    for(int i = 0; i < n; ++i) {
        int ai; cin >> ai;
        A.push_back(ai);
    }

    for(int i = 0; i < q; ++i) {
        int bi, ki; cin >> bi >> ki;
        queries.push_back({bi, ki});
    }

    sort(A.begin(), A.end());

    for(auto& [bi, ki]: queries) {
        auto it = lower_bound(A.begin(), A.end(), bi);
        int dist = min(abs(bi - *it), abs(bi - *(--it)));
        ans.push_back(dist * dist);
    }

    for(auto a: ans) cout << a << '\n';

    return 0;
}
```

This solution works as follows:

1. We read all inputs and store coordinates of points A into a vector `A` and pairs of coordinates and k-values into another vector `queries`.

2. We sort the vector `A` containing the coordinates of points A.

3. For each query, we use binary search (`lower_bound`) to find the first point in sorted array `A` that is not less than the target point `B`. This gives us the nearest point to `B`. Then we calculate distances to both this and previous points and choose the minimum one.

4. Finally, we output squared distances for all queries.

This approach ensures efficient handling of large inputs up to the specified constraints, with a time complexity dominated by sorting (O(N log N)) and binary searches (O(Q log N)).
