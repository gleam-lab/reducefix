#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    int N, Q;
    cin >> N >> Q;

    vector<ll> a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }

    // Sort the A array for binary search purposes
    sort(a.begin(), a.end());

    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;

        // Binary search on distance
        ll low = 0, high = 2e8 + 5; // Max possible distance is 2*1e8

        while (low < high) {
            ll mid = (low + high) / 2;

            // Find how many points are in [b - mid, b + mid]
            ll left = b - mid;
            ll right = b + mid;

            // Use binary search to find the range
            auto l = lower_bound(a.begin(), a.end(), left);
            auto r = upper_bound(a.begin(), a.end(), right);
            int count = r - l;

            if (count >= k)
                high = mid;
            else
                low = mid + 1;
        }

        cout << low << "\n";
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    solve();

    return 0;
}
