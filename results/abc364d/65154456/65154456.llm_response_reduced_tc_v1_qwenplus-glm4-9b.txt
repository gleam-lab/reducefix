#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9 + 7;
const int N = 2e5 + 100;

int a[N];

void solve() {
    int n, q;
    cin >> n >> q;
    memset(a, 0, sizeof(a));
    
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        a[x + 100000000]++;
    }
    
    for (int i = 1; i < N; i++) {
        a[i] += a[i - 1];
    }
    
    for (int i = 0; i < q; i++) {
        int x, k;
        cin >> x >> k;
        int l = 0, r = N - 1;
        while (l < r) {
            int mid = (l + r + 1) / 2;
            if (a[mid] >= k)
                r = mid - 1;
            else
                l = mid;
        }
        cout << l + x << endl;
    }
}

int main() {
    solve();
    return 0;
}
