    vector<int> a(n), b(n), k(n);

    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        b[i] = a[i] - 100000000; // Store the adjusted indices
    }

    for (int i = 0; i < q; ++i) {
        int x, k_j;
        cin >> x >> k_j;
        k_j--; // Convert k_j to 0-based index
        // For each B_j, find A_i such that d_i (distance from A_i to B_j) is the k_j-th smallest
        // d_i = abs(A_i - B_j) = abs(a[A_i] - x)
        // We need to sort these distances and find the k_j-th smallest one

        // Calculate distances for all points A_i from B_j
        vector<int> distances(n);
        for (int j = 0; j < n; ++j) {
            distances[j] = abs(a[j] - x);
        }

        // Sort and get the k_j-th smallest distance
        sort(distances.begin(), distances.end());
        cout << distances[k_j] << endl;
    }

    return 0;
}
