    return a[min(N - 1, x + mid)] - a[max(0, x - mid - 1)] >= y;
}

void solve(void) {
    int n, q; cin >> n >> q;
    vector<int> points(n);
    for(int i = 0; i < n; i++) cin >> points[i];
    sort(points.begin(), points.end());
    for(int i = 1; i <= q; i++){
        int x, k; cin >> x >> k;
        int l = 0, r = n - 1;
        while(l < r){
            int mid = (l + r) / 2;
            if(abs(points[mid] - x) >= abs(points[(mid + 1) % n] - x)){
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        cout << abs(points[l] - x) << endl;
    }
}
