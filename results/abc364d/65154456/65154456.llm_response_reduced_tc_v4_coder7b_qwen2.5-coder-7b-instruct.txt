#include<bits/stdc++.h>
#include<vector>
#include<algorithm>
#define endl "\n"
using namespace std;
typedef long long ll;

vector<ll> A, B;

bool compare(ll a, ll b){
    return a < b;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; i++){
        ll x; cin >> x;
        A.push_back(x);
    }
    for(int i = 0; i < q; i++){
        ll x, y; cin >> x >> y;
        B.push_back(x);
        A.push_back(x);
    }
    sort(A.begin(), A.end());
    for(int i = 0; i < q; i++){
        ll x = B[i], k = i + 1;
        auto it = lower_bound(A.begin(), A.end(), x);
        int index = it - A.begin();
        if(index == 0){
            cout << abs(x - A[k]) << endl;
        }else if(index == n + q){
            cout << abs(x - A[n + q - 1]) << endl;
        }else{
            cout << min(abs(x - A[index]), abs(x - A[index - 1])) << endl;
        }
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
