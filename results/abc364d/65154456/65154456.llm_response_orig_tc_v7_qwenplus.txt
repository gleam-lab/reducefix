#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    int n, q;
    cin >> n >> q;
    
    vector<ll> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Sort the A points for binary search
    sort(a.begin(), a.end());
    
    for (int i = 0; i < q; i++) {
        ll b;
        int k;
        cin >> b >> k;
        
        // Binary search on the distance
        ll left = 0, right = 2e8;
        ll result = right;
        
        while (left <= right) {
            ll mid = (left + right) / 2;
            
            // Count points in [b-mid, b+mid]
            ll lower_bound = b - mid;
            ll upper_bound = b + mid;
            
            // Find first index >= (b-mid)
            auto it1 = lower_bound(a.begin(), a.end(), lower_bound);
            // Find first index > (b+mid)
            auto it2 = upper_bound(a.begin(), a.end(), upper_bound);
            
            int count = it2 - it1;
            
            if (count >= k) {
                result = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        cout << result << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    
    return 0;
}
