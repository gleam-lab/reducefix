```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;
    vector<ll> a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        ll left = 0, right = 2e14;
        ll answer = 0;
        while (left <= right) {
            ll mid = left + (right - left) / 2;
            auto it_l = lower_bound(a.begin(), a.end(), b - mid);
            auto it_r = upper_bound(a.begin(), a.end(), b + mid);
            int count = it_r - it_l;
            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        cout << answer << endl;
    }

    return 0;
}
```

This approach uses binary search on the possible distances \(d\) and checks how many elements fall within the range \([B_j - d, B_j + d]\). If this count is at least \(k_j\), we try a smaller \(d\); otherwise, we try a larger \(d\). This ensures that we find the smallest \(d\) such that there are at least \(k_j\) elements within \(d\) distance of \(B_j\). The time complexity is \(O(N \log N + Q \log D)\), where \(D\) is the maximum possible distance (up to \(2 \times 10^{14}\) in this case), which is efficient enough for the given constraints.
