#include <iostream>
#include <vector>
#include <algorithm>

    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    for (int i = 0; i < Q; ++i) {
        cin >> B[i] >> K[i];
        --K[i]; // Convert to 0-based index
    }
    
    // Sort points A to find the closest points faster
    sort(A.begin(), A.end());
    
    // Prepare to store distances
    vector<long long> distances(Q);
    
    // For each query, find the closest point in A to B[j]
    for (int j = 0; j < Q; ++j) {
        int dist;
        // If B[j] is before the first point in A, the closest point is the first point in A
        if (B[j] < A[0]) {
            dist = abs(B[j] - A[0]);
        }
        // If B[j] is after the last point in A, the closest point is the last point in A
        else if (B[j] > A[N - 1]) {
            dist = abs(B[j] - A[N - 1]);
        }
        // Otherwise, use binary search to find the closest point in A
        else {
            auto it = lower_bound(A.begin(), A.end(), B[j]);
            dist = min(abs<B[j], *it), abs<B[j], *(it - 1));
        }
        distances[j] = dist * dist; // Store squared distance to avoid sqrt later
    }
    
    // Sort distances to get the k-th closest distances
    sort(distances.begin(), distances.end());
    
    // Output the k-th closest distance for each query
    for (int j = 0; j < Q; ++j) {
        cout << distances[K[j]] << endl;
    }
    
    return 0;
}
