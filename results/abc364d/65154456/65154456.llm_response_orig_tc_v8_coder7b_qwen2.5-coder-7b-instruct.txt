```cpp
#include<bits/stdc++.h>
using namespace std;

#define all(v) v.begin(), v.end()
#define rep(i, a, b) for(int i = a; i < b; i++)
#define repr(i, a, b) for(int i = a; i >= b; i--)
#define trav(a, x) for(auto& a : x)

typedef long long ll;
typedef pair<int, int> pii;
typedef vector<int> vi;
typedef vector<ll> vl;
typedef vector<pii> vpi;

ll dist(ll a, ll b) {
    return abs(a - b);
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    ll n, q; cin >> n >> q;
    vl a(n), b(q);
    unordered_map<ll, ll> mp;
    
    rep(i, 0, n) {
        cin >> a[i];
        mp[a[i]]++;
    }

    sort(all(a));

    rep(j, 0, q) {
        cin >> b[j];
        ll idx = lower_bound(all(a), b[j]) - a.begin();
        ll left = idx == 0 ? 0 : a[idx - 1], right = idx == n ? 0 : a[idx];
        ll diff_l = dist(b[j], left), diff_r = dist(b[j], right);
        ll ans = min(diff_l * mp[left], diff_r * mp[right]);
        cout << ans << '\n';
    }

    return 0;
}
```
This solution sorts the array of points `A` and then uses binary search to find the closest points to `B`. It calculates the distances and outputs the minimum distance found. The time complexity is O((N+Q)logN).
