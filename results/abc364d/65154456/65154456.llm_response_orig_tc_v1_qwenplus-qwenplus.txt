#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAX_OFFSET = 200000000; // To handle values from -1e8 to 1e8

int freq[400000001]; // Frequency array with offset to avoid negative indices

// Binary search for the smallest radius that contains at least k points around b_j
ll find_kth_closest(int b, int k) {
    int left = 0;
    int right = 200000000;
    ll res = 200000000;

    while (left <= right) {
        int mid = (left + right) / 2;
        int low = max(0, b - mid + MAX_OFFSET);
        int high = min(400000000, b + mid + MAX_OFFSET);

        int count = freq[high] - (low > 0 ? freq[low - 1] : 0);

        if (count >= k) {
            res = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return res;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    // Initialize frequency array
    fill(freq, freq + 400000001, 0);

    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        int idx = A[i] + MAX_OFFSET;
        freq[idx]++;
    }

    // Build prefix sum of frequencies
    for (int i = 1; i <= 400000000; ++i) {
        freq[i] += freq[i - 1];
    }

    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;

        cout << find_kth_closest(b, k) << '\n';
    }
}

int main() {
    solve();
    return 0;
}
