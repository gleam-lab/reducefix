#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query_kth_distance(ll b, ll k, int n) {
    // Create array of distances (implicitly)
    // We'll use binary search on answer technique
    
    ll left = 0;
    ll right = 2e8;  // Max possible distance
    
    while (left < right) {
        ll mid = (left + right) / 2;
        
        // Count how many points have distance <= mid from b
        // This is equivalent to counting number of a[i] in [b - mid, b + mid]
        ll lower = b - mid;
        ll upper = b + mid;
        
        int l = lower_bound(a, a + n, lower) - a;
        int r = upper_bound(a, a + n, upper) - a;
        int count = r - l;
        
        if (count >= k) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    
    return left;
}

void solve() {
    int n, q;
    cin >> n >> q;
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    sort(a, a + n);
    
    while (q--) {
        ll b, k;
        cin >> b >> k;
        cout << query_kth_distance(b, k, n) << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    solve();
    
    return 0;
}
