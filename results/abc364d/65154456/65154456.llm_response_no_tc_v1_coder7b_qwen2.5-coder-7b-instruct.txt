#include<bits/stdc++.h>
using namespace std;

vector<int> a;

bool compare(const pair<int,int>& p1,const pair<int,int>& p2){
    return p1.first<p2.first;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a.push_back(abs(x));
    }
    sort(a.begin(),a.end());
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        auto it = lower_bound(a.begin(),a.end(),abs(x));
        int index = it-a.begin()+1;
        cout << abs(a[index-1]-x) << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
