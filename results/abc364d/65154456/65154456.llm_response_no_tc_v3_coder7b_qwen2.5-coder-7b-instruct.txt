```cpp
#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 100005;

vector<int> a[MAXN];
int b[MAXN], k[MAXN];
ll ans[MAXN];

void solve() {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++) {
        cin >> a[i].push_back(b[i]);
    }
    sort(a[a[1].size()].begin(), a[a[1].size()].end());
    for(int i = 1; i <= q; i++) {
        cin >> b[i] >> k[i];
        ans[i] = lower_bound(a[b[i]].begin(), a[b[i]].end(), k[i]) - a[b[i]].begin();
    }
    for(int i = 1; i <= q; i++) {
        cout << ans[i] << '\n';
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    solve();
    return 0;
}
```

This solution sorts all distances of each B_i with all A_j and uses binary search to find the k_j-th smallest distance for each query. This approach ensures correct results within the constraints.
