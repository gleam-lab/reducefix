        cin >> a[i];
    }

        v.push_back({b[i], i});
    }

    sort(v.begin(), v.end());

    int last_index = 0;
    for(auto& [val, index] : v) {
        while(last_index < n && a[last_index] < val) {
            last_index++;
        }
        
        int dist = abs(val - a[last_index]);
        if(last_index > 0) {
            dist = min(dist, abs(val - a[last_index - 1]));
        }
        
        cout << dist << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t = 1; //cin >> t;

Explanation of changes:
1. We use `vector` instead of array because we don't know the size at compile time.
2. We read all the inputs before processing them to improve efficiency.
3. We sort both the points `A` and the queries `B`.
4. For each query, we find the first point `X` that is greater than or equal to `B[j]`. Then, we calculate the minimum distance between `B[j]` and its neighboring points.
5. The sorting operation takes O((N + Q) log(N + Q)) time, which is efficient enough considering the constraints.
