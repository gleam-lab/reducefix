#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

vector<int> a, b, c;
vector<pair<ll, int>> pos;

bool cmp(const pair<ll, int>& p1, const pair<ll, int>& p2) {
    return p1.first < p2.first;
}

void solve() {
    int n, q; cin >> n >> q;
    a.resize(n + 1); b.resize(q + 1); c.resize(q + 1);
    for(int i = 1; i <= n; i++) cin >> a[i];
    for(int i = 1; i <= q; i++) {
        cin >> b[i] >> c[i];
        pos.push_back({b[i], i});
    }
    sort(pos.begin(), pos.end());
    vector<ll> dis(n + 1, INT_MAX);
    for(int i = 1; i <= n; i++) {
        if(i == 1) dis[i] = abs(a[i] - pos[0].first);
        else dis[i] = min(dis[i - 1], abs(a[i] - pos[0].first));
    }
    reverse(pos.begin(), pos.end());
    for(int i = 1; i <= n; i++) {
        if(i == 1) dis[i] = min(dis[i], abs(a[i] - pos[0].first));
        else dis[i] = min(dis[i], min(dis[i - 1], abs(a[i] - pos[0].first)));
    }
    for(auto& p : pos) cout << dis[c[p.second]] << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
