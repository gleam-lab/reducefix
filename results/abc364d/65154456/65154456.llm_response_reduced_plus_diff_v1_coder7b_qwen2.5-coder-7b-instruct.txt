#include<bits/stdc++.h>

const int N = 1e5 + 5;
int a[N];

bool check(int mid, int x, int y){
    return a[min(N - 1, x + mid)] - a[max(0, x - mid - 1)] >= y;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a[x + 100000000] ++;
    }
    for(int i = 1; i < N; i++){
        a[i] += a[i - 1];
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 2e8;
        while(l < r){
            int mid = l + r >> 1;
            if(check(mid, x + 100000000, y)) l = mid + 1;
            else r = mid;
        }
        cout << l << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
