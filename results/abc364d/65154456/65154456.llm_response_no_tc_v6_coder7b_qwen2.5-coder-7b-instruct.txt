#include<bits/stdc++.h>
using namespace std;

vector<int> a, b, ans;
int n, q;

struct Point {
    int pos, index;
};

bool cmp(const Point& p1, const Point& p2) {
    return abs(p1.pos - b[p2.index]) < abs(p2.pos - b[p1.index]);
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    cin >> n >> q;
    a.resize(n);
    for(int i = 0; i < n; ++i)
        cin >> a[i];

    vector<Point> points;
    for(int i = 0; i < q; ++i) {
        int x, k; cin >> x >> k;
        --k; // Adjusting k to zero-based index
        points.push_back({x, i});
    }

    sort(points.begin(), points.end(), cmp);

    int last_x = -1;
    int idx = 0;
    for(int i = 0; i < points.size(); ++i) {
        int cur_x = points[i].pos;
        int k = points[i].index;

        if(cur_x != last_x) {
            last_x = cur_x;
            idx = upper_bound(a.begin(), a.end(), cur_x) - a.begin();
        }

        while(idx > 0 && a[idx - 1] == a[idx])
            --idx;

        ans[k] = abs(b[k] - a[idx - 1]);
    }

    for(int i = 0; i < q; ++i)
        cout << ans[i] << '\n';

    return 0;
}
