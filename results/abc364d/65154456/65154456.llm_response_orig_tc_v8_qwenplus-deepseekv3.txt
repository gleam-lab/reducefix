#include<bits/stdc++.h>
using namespace std;

int main() {

    cin.tie(0);
    cout.tie(0);

    cin >> N >> Q;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());

    while (Q--) {
        int b, k;
        cin >> b >> k;
        int left = 0, right = 1e9;
        int answer = 0;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            auto lower = lower_bound(A.begin(), A.end(), b - mid);
            auto upper = upper_bound(A.begin(), A.end(), b + mid);
            int count = upper - lower;
            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        cout << answer << '\n';
    }

    return 0;
}
