#include<bits/stdc++.h>
using namespace std;

vector<int> a, b;
vector<vector<int>> distances;

void preprocess() {
    int n = a.size();
    for(int i = 0; i < n; ++i) {
        vector<int> temp(n);
        for(int j = 0; j < n; ++j) {
            temp[j] = abs(a[i] - a[j]);
        }
        sort(temp.begin(), temp.end());
        distances.push_back(temp);
    }
}

int query(int idx, int k) {
    return distances[idx][k-1];
}

int main() {
    int n, q; cin >> n >> q;
    a.resize(n); b.resize(q);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for(int i = 0; i < q; ++i) {
        cin >> b[i];
    }

    preprocess();

    for(int i = 0; i < q; ++i) {
        int pos = lower_bound(b.begin(), b.end(), a[i]) - b.begin();
        cout << query(pos, b[i]) << '\n';
    }

    return 0;
}
