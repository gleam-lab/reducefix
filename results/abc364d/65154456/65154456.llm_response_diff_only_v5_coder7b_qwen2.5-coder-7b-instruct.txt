#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 1e5 + 5;
int a[MAXN], b[MAXN], c[MAXN];

int main() {
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a, a + n);
    for(int i = 0; i < q; i++) {
        cin >> b[i] >> c[i];
    }
    for(int i = 0; i < q; i++) {
        int pos = upper_bound(a, a + n, b[i]) - a;
        if(c[i] <= pos) {
            cout << abs(a[c[i] - 1] - b[i]) << endl;
        } else {
            cout << abs(a[n - 1] - b[i]) << endl;
        }
    }
    return 0;
}
