```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int OFFSET = 1e8;  // To handle negative indices

void solve() {
    int N, Q;
    cin >> N >> Q;

    vector<int> freq(2 * (int)1e8 + 2, 0);  // Frequency array with offset

    // Read A values and build frequency map
    for (int i = 0; i < N; ++i) {
        int a;
        cin >> a;
        freq[a + OFFSET]++;
    }

    // Build prefix sum array
    vector<ll> prefix(2 * (int)1e8 + 2, 0);
    for (int i = 1; i <= 2 * (int)1e8; ++i) {
        prefix[i] = prefix[i - 1] + freq[i];
    }

    // Process queries
    while (Q--) {
        int b, k;
        cin >> b >> k;

        // Binary search on distance d
        int left = 0, right = 2 * (int)1e8;
        int answer = 0;

        while (left <= right) {
            int mid = (left + right) / 2;
            int pos = b + OFFSET;
            int l = max(0, pos - mid);
            int r = min((int)prefix.size() - 1, pos + mid);

            ll count = prefix[r] - prefix[l];

            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        cout << answer << "\n";
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
