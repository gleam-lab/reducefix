```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b, k;
        cin >> b >> k;
        int left = 0, right = N - 1;
        vector<ll> distances(N);
        for (int i = 0; i < N; ++i) {
            distances[i] = abs(A[i] - b);
        }
        sort(distances.begin(), distances.end());
        cout << distances[k - 1] << '\n';
    }
    
    return 0;
}
```

### Explanation
1. **Input Handling**: The input is read, including the number of points N and the number of queries Q. The coordinates of the points A are stored in a vector and sorted.
2. **Query Processing**: For each query, the point B_j and the integer k_j are read. The distances from B_j to each point in A are computed and stored in a vector.
3. **Sorting Distances**: The distances from B_j to each A_i are sorted to easily find the k_j-th smallest distance.
4. **Output**: The k_j-th smallest distance is printed for each query.

This approach ensures that the solution is efficient and handles the constraints correctly, with a time complexity of O(N log N) per query due to the sorting of distances. The overall complexity is O(Q N log N), which is feasible given the constraints.
