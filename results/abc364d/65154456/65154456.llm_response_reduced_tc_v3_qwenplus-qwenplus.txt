#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query(ll b, ll k, int n) {
    // We will use two pointers to maintain a sliding window
    ll left = 0, right = 2e8; // possible max distance is up to 1e8
    ll answer = 0;

    while (left <= right) {
        ll mid = (left + right) / 2;
        ll count = 0;
        // Two pointers approach to count how many A[i] are within distance <= mid of b
        for (int i = 0, j = 0; i < n; ++i) {
            while (j < n && abs(a[i] - b) <= mid) ++j;
            count += j - i;
        }

        if (count >= k) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return answer;
}

void solve() {
    int n, q;
    cin >> n >> q;

    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    sort(a, a + n);

    for (int l = 0; l < q; ++l) {
        ll b, k;
        cin >> b >> k;
        cout << query(b, k, n) << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
