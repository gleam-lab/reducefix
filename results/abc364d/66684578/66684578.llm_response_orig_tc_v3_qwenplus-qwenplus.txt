#include <bits/stdc++.h>
using namespace std;

#define ll long long

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());

    while (Q--) {
        ll B;
        int K;
        cin >> B >> K;

        // Binary search on the k-th smallest distance
        ll low = 0, high = 2e8 + 10;
        while (low < high) {
            ll mid = (low + high) / 2;
            ll cnt = 0;
            auto left = lower_bound(A.begin(), A.end(), B - mid);
            auto right = upper_bound(A.begin(), A.end(), B + mid);
            cnt = right - left;
            if (cnt < K)
                low = mid + 1;
            else
                high = mid;
        }

        cout << low << '\n';
    }

    return 0;
}
