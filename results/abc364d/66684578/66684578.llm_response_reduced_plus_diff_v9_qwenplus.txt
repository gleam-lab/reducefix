#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vl vector<ll>

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n, q;
    cin >> n >> q;
    vi a(n);
    rep(i,n) cin >> a[i];
    
    // Sort the A array to enable binary search
    sort(a.begin(), a.end());
    
    rep(_,q){
        int b, k;
        cin >> b >> k;
        
        // Binary search on the distance
        ll left = -1, right = 2e9 + 10;
        while(right - left > 1){
            ll mid = (left + right) / 2;
            
            // Count how many points in A are within distance 'mid' from B
            ll count = 0;
            // Points in [b-mid, b+mid]
            auto it1 = lower_bound(a.begin(), a.end(), b - mid);
            auto it2 = upper_bound(a.begin(), a.end(), b + mid);
            count = it2 - it1;
            
            if(count < k){
                left = mid;
            } else {
                right = mid;
            }
        }
        
        cout << right << '\n';
    }
}
