#include <bits/stdc++.h>

#define vll vector<ll>

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    vi a(N);
    rep(i, N) cin >> a[i];

    // Sort the A points for binary search and distance computation
    sort(a.begin(), a.end());

    rep(qi, Q) {
        int b, k;
        cin >> b >> k;
        k--; // convert to 0-indexed

        // Binary search on the answer (distance)
        ll left = -1, right = 200000000; // max possible distance is 2e8
        while (right - left > 1) {
            ll mid = (left + right) / 2;
            
            // Count how many points in A are within distance 'mid' from B_j
            ll low = b - mid;
            ll high = b + mid;
            
            auto it1 = lower_bound(a.begin(), a.end(), low);
            auto it2 = upper_bound(a.begin(), a.end(), high);
            int count = it2 - it1;
            
            if (count <= k) {
                left = mid;
            } else {
                right = mid;
            }
        }
        
        cout << right << endl;
    }
}
