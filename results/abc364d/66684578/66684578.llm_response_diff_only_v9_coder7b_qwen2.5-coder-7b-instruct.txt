```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, q;
    cin >> n >> q;

    vector<long long> a(n);
    vector<long long> b(q), k(q);

    for (int i = 0; i < n; ++i)
        cin >> a[i];

    for (int i = 0; i < q; ++i) {
        cin >> b[i] >> k[i];
    }

    sort(a.begin(), a.end());

    for (int i = 0; i < q; ++i) {
        auto it = lower_bound(a.begin(), a.end(), b[i]);
        int index = it - a.begin();
        if (index >= k[i]) {
            cout << abs(a[index - k[i]] - b[i]) << '\n';
        } else {
            cout << abs(a[0] - b[i]) + (k[i] - index) << '\n';
        }
    }

    return 0;
}
```
