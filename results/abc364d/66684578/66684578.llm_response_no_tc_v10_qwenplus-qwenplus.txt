#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    vector<long long> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Sort A for binary search usage
    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        long long B;
        int K;
        cin >> B >> K;

        // Create a lambda to compute distance
        auto distance = [&](int idx) {
            return abs(A[idx] - B);
        };

        // Binary search on the k-th smallest distance
        int left = 0, right = N;
        while (left < right) {
            int mid = (left + right) / 2;
            if (mid < K) {
                left = mid + 1;
            } else {
                // Try to narrow down the range
                int l = mid - K + 1;
                int r = l + K - 1;
                if (r <= mid) {
                    right = mid;
                } else {
                    left = mid + 1;
                }
            }
        }

        // After finding the correct position, we need to find the k-th smallest distance
        // We can use a sliding window of size K and find the maximum in that window
        // But for efficiency, we'll use a different approach: binary search with custom comparator

        // Since sorting all distances for each query would be O(N log N) per query -> too slow,
        // we will perform a binary search on possible distances

        long long low = 0, high = 2e8 + 10;
        long long answer = high;

        while (low <= high) {
            long long mid = (low + high) / 2;
            // Find how many points are within distance `mid` from B
            int left_idx = lower_bound(A.begin(), A.end(), B - mid) - A.begin();
            int right_idx = upper_bound(A.begin(), A.end(), B + mid) - A.begin();
            int count = right_idx - left_idx + 1;

            if (count >= K) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        cout << answer << '\n';
    }

    return 0;
}
