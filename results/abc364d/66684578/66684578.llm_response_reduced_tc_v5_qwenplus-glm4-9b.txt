#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i = 0; i < (int)n; i++)
#define vi vector<int>

const int MAXN = 1e5 + 100;
const int INF = 1e9 + 7;

int main() {
    int n, q;
    cin >> n >> q;
    
    vi a(n);
    rep(i, n) cin >> a[i];

    vector<pair<int, int>> queries;
    rep(i, q) {
        int b, k;
        cin >> b >> k;
        queries.emplace_back(b, k);
    }

    // Sort points A in ascending order
    sort(a.begin(), a.end());

    // Sort queries by the coordinate b to optimize the search process
    sort(queries.begin(), queries.end());

    // To store the answers for each query
    vector<ll> ans(q);

    // Binary search to find the k-th closest point
    int l = 0, r = n - 1;
    for (auto& [b, k] : queries) {
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (b - a[mid] <= a[mid + 1] - b) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        // l is now the index of the k-th closest point
        ans[ queries[i - q].second - 1 ] = abs(a[l] - b);
    }

    // Output the results
    rep(i, q) {
        cout << ans[i] << endl;
    }

    return 0;
}
