#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vll vector<ll>
#define pb push_back
#define pii pair<int,int>
#define fi first
#define se second
#define all(a) a.begin(),a.end()

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    vi A(N);
    rep(i, N) cin >> A[i];
    sort(all(A));
    
    // Precompute prefix sum for binary search
    vi prefix_sum(N+1, 0);
    rep(i, N) prefix_sum[i+1] = prefix_sum[i] + A[i];
    
    rep(q, Q) {
        int b, k;
        cin >> b >> k;
        
        // Binary search to find the k-th smallest distance
        int left = 0;
        int right = 2e8; // Max possible distance is 2*1e8
        
        while (left < right) {
            int mid = (left + right) / 2;
            
            // Find range of elements in A whose distance from b is <= mid
            int l = lower_bound(A.begin(), A.end(), b - mid) - A.begin();
            int r = upper_bound(A.begin(), A.end(), b + mid) - A.begin();
            
            if (r - l >= k)
                right = mid;
            else
                left = mid + 1;
        }
        
        cout << left << "\n";
    }
}
