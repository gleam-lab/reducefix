```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN], b[MAXN], k[MAXN];

int main() {
    int n, q;
    cin >> n >> q;
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < q; ++i) cin >> b[i] >> k[i];

    sort(a, a + n);
    multiset<int> dists;
    for(int i = 0; i < n; ++i) {
        dists.insert(abs(a[i] - b[0]));
    }

    for(int i = 1; i < q; ++i) {
        auto it = dists.begin();
        advance(it, k[i] - 1);
        cout << *it << '\n';
        dists.erase(dists.find(abs(a[0] - b[i])));
        dists.insert(abs(a[0] - b[i]));
    }

    return 0;
}
```
