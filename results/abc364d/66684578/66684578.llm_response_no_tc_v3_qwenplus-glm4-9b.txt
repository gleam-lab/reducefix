#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll get(map<ll, ll>& mp, ll m) {
    auto it = mp.lower_bound(m);
    if (it == mp.begin()) return 0;
    --it;
    return it->second;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    ll n, q; cin >> n >> q;
    vector<ll> a(n);
    
    for (ll i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    while (q--) {
        ll b, k; cin >> b >> k;
        ll left = lower_bound(a.begin(), a.end(), b - k) - a.begin();
        ll right = upper_bound(a.begin(), a.end(), b + k) - a.begin();
        
        if (left >= n || right - left < k) {
            cout << 0 << '\n';
        } else {
            cout << a[left + k - 1] - b << '\n';
        }
    }
    
    return 0;
}
