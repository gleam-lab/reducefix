typedef long long ll;

vector<ll> sort_and_get_distances(const vector<ll>& a, const ll& b) {
    vector<ll> distances;
    for (const auto& x : a) {
        distances.push_back(abs(x - b));
    }
    sort(distances.begin(), distances.end());
    return distances;
}
