// Function to find the k-th smallest distance from B to any A[i]
ll findKthDistance(const vll& sortedA, ll b, int k) {
    // We'll use a binary search over possible distances
    ll low = 0;
    ll high = 2e16; // Safe upper bound based on input constraints
    
    while (low < high) {
        ll mid = (low + high) / 2;
        // Find range in sortedA that lies in [b - mid, b + mid]
        ll left = b - mid;
        ll right = b + mid;
        
        // Use binary search to count how many elements are in this interval
        auto l = lower_bound(sortedA.begin(), sortedA.end(), left);
        auto r = upper_bound(sortedA.begin(), sortedA.end(), right);
        int count = r - l;
        
        if (count >= k) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    
    return low;
}
