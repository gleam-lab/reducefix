#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vll vector<ll>

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    
    vi a(N);
    rep(i, N) cin >> a[i];
    
    // Sort the A points for binary search later
    sort(a.begin(), a.end());
    
    rep(qi, Q) {
        int b, k;
        cin >> b >> k;
        k--; // convert to 0-indexed
        
        // We'll collect distances from B_j to all A_i, but we don't want to sort all each time
        // Instead, we can use binary search on answer or two pointers
        
        // Alternative: find the k-th smallest distance using binary search on the distance value
        // But simpler approach: use lower_bound to find closest points around b
        
        // Create array of distances and get k-th smallest
        // Since N, Q up to 1e5, doing O(N log N) per query would be O(Q*N*log N) = 10^10 * log(10^5) -> too slow
        
        // Better: we can do binary search on the distance d such that exactly k points are within distance d
        // But note: we need the k-th smallest distance, not number of points within d
        
        // Correct approach: 
        // We can use nth_element if we generate distances, but worst-case O(N) per query -> 10^10 operations
        
        // Need efficient solution.
        // Observation: sort A once. For each B_j, we can find insertion point, then expand left/right with two pointers.
        
        // Find position where b would be inserted
        auto it = lower_bound(a.begin(), a.end(), b);
        vi distances;
        
        // Collect distances from nearby points
        // We only need at most k nearest points
        // Use two pointers from the split point
        
        int right_idx = it - a.begin();
        int left_idx = right_idx - 1;
        
        distances.reserve(k+1);
        
        while ((int)distances.size() < k) {
            if (left_idx >= 0 && (right_idx >= N || b - a[left_idx] <= a[right_idx] - b)) {
                distances.push_back(b - a[left_idx]);
                left_idx--;
            } else if (right_idx < N) {
                distances.push_back(a[right_idx] - b);
                right_idx++;
            }
        }
        
        // The last added distance is the k-th smallest
        cout << distances.back() << '\n';
    }
}
