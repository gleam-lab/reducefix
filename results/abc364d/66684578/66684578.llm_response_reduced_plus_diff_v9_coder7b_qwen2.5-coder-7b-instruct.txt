```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 1e5 + 5;

int main() {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int n, q;
    cin >> n >> q;
    
    vector<int> a(n), b(q), k(q);
    for (int i = 0; i < n; ++i) cin >> a[i];
    for (int i = 0; i < q; ++i) {
        cin >> b[i] >> k[i];
        --k[i]; // Adjusting to 0-based index
    }

    sort(a.begin(), a.end());
    vector<vector<pii>> dists(n);
    for (int i = 0; i < n; ++i) {
        dists[i].push_back({abs(a[i] - b[0]), 0});
    }
    for (int j = 1; j < q; ++j) {
        for (int i = 0; i < n; ++i) {
            dists[i].push_back({abs(a[i] - b[j]), j});
        }
        sort(dists[i].begin(), dists[i].end());
    }

    for (int i = 0; i < n; ++i) {
        cout << dists[i][k[i]].first << '\n';
    }

    return 0;
}
```
