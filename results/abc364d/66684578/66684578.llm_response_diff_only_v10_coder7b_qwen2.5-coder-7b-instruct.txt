#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    vi a(n), dist(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    for (int j = 0; j < q; ++j) {
        int b, k;
        cin >> b >> k;
        for (int i = 0; i < n; ++i) {
            dist[i] = abs(a[i] - b);
        }
        sort(dist.begin(), dist.end());
        cout << dist[k - 1] << '\n';
    }

    return 0;
}
