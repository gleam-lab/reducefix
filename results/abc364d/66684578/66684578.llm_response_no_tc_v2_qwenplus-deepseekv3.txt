```cpp
#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b, k;
        cin >> b >> k;
        
        ll left = 0, right = 2e14;
        ll answer = 0;
        
        while (left <= right) {
            ll mid = (left + right) / 2;
            auto lower = lower_bound(A.begin(), A.end(), b - mid);
            auto upper = upper_bound(A.begin(), A.end(), b + mid);
            int count = upper - lower;
            
            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        cout << answer << '\n';
    }
    
    return 0;
}
```

This code correctly uses binary search to find the minimal distance `d` such that there are at least `k` elements in A within distance `d` of B_j. The `lower_bound` and `upper_bound` functions are used to count the number of elements in the range [B_j - d, B_j + d], and the binary search adjusts `d` until the correct value is found.
