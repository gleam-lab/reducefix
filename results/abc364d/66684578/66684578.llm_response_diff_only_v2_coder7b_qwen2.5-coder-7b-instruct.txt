```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define rep(i,n) for(int i=0;i<n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>
#define vpii vector<pair<int, int>>

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    
    int n, q;
    cin >> n >> q;
    vi a(n), b(q);
    vi k(q);
    
    rep(i, n)
        cin >> a[i];
    
    rep(j, q) {
        cin >> b[j] >> k[j];
    }

    sort(a.begin(), a.end());
    
    rep(j, q) {
        int pos = upper_bound(a.begin(), a.end(), b[j]) - a.begin();
        int dist = min(pos * b[j] - a[pos], (pos == n ? INT_MAX : a[pos + 1] - b[j]));
        cout << dist << '\n';
    }
    
    return 0;
}
```

This solution works as follows:
1. **Input Reading**: The input values are read into vectors `a` and `b`, and arrays `k`.
2. **Sorting Points A**: The points `A_i` are sorted in non-decreasing order.
3. **Finding Closest Point**:
   - For each point `B_j`, determine its position using binary search (`upper_bound`) in the sorted list of points `A`.
   - Calculate the minimum distance to either side of this position.
4. **Output**: Print the calculated distances for each query.

This approach ensures that we efficiently find the k-th closest point for each query with a time complexity of \(O(N \log N + Q \log N)\), which is suitable given the constraints.
