#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<int>
#define vll vector<ll>
#define pb push_back

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;

    vi A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Sort the A array once
    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;

        // Binary search on distance
        ll low = 0, high = 2e8 + 10;
        while (low < high) {
            ll mid = (low + high) / 2;
            // Find range of elements within distance mid from b
            int left = lower_bound(A.begin(), A.end(), b - mid) - A.begin();
            int right = upper_bound(A.begin(), A.end(), b + mid) - A.begin();
            if (right - left >= k) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        cout << low << "\n";
    }

    return 0;
}
