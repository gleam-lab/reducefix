```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    int n, q;
    cin >> n >> q;
    
    vector<ll> a(n), b(q);
    vector<int> k(q);
    
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < q; ++i) cin >> b[i] >> k[i];
    
    sort(a.begin(), a.end());
    
    for(int i = 0; i < q; ++i) {
        int x = k[i];
        auto it = lower_bound(a.begin(), a.end(), b[i]);
        int pos = it - a.begin();
        
        if(pos >= x) {
            cout << abs(a[pos-x] - b[i]) << '\n';
        } else {
            cout << abs(a[0] - b[i]) << '\n';
        }
    }
    
    return 0;
}
```
