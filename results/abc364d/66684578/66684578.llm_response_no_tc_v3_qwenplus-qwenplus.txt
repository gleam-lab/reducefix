#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vll vector<ll>
#define pb push_back

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n, q;
    cin >> n >> q;
    
    vi a(n);
    rep(i, n) cin >> a[i];
    sort(a.begin(), a.end());
    
    // Precompute prefix sums for binary search counts
    vi prefix(n + 1, 0);
    rep(i, n) prefix[i + 1] = prefix[i];
    
    vector<pair<int, int>> queries(q);
    for (int j = 0; j < q; ++j) {
        cin >> queries[j].first >> queries[j].second;
    }
    
    rep(j, q) {
        int b = queries[j].first;
        int k = queries[j].second;
        
        // Binary search on distance
        int low = 0;
        int high = 2e8 + 10; // Safe upper bound
        while (low < high) {
            int mid = (low + high) / 2;
            
            // Find range in sorted A that are within distance mid from b
            int left = lower_bound(a.begin(), a.end(), b - mid) - a.begin();
            int right = upper_bound(a.begin(), a.end(), b + mid) - a.begin();
            
            if (right - left >= k)
                high = mid;
            else
                low = mid + 1;
        }
        
        cout << low << "\n";
    }
}
