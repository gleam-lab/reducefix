#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for (int i = 0; i < (int)n; i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>
#define vii vector<vi>
#define vs vector<string>
#define pi pair<int, int>

ll inf = (1LL << 60);

ll rui(ll a, ll b) {
    if (b == 0) return 1;
    if (b % 2 == 1) return a * rui(a * a, b / 2);
    return rui(a * a, b / 2);
}

ll get(map<int, int> &mp, int m) {
    if (mp.find(m) != mp.end()) return mp[m];
    return (--mp.lower_bound(m))->second;
}

int main() {
    int n, q; cin >> n >> q;
    vi a(n);
    map<int, int> mp;
    rep(i, n) {
        cin >> a[i];
        mp[a[i]]++;
    }
    mp[inf] = inf;
    ll now = 0;
    for (auto &p : mp) {
        p.second += now;
        now = p.second;
    }
    rep(i, q) {
        int b, k; cin >> b >> k;
        int l = -inf, r = inf;
        while (r - l > 1) {
            int m = (r + l) / 2;
            if (get(mp, b + m) - get(mp, b - m - 1) < k) {
                l = m;
            } else {
                r = m;
            }
        }
        cout << r << endl;
    }
}
