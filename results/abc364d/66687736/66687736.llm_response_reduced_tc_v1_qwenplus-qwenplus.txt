#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<int>
#define vll vector<ll>
#define vpi vector<pair<int, int>>
#define pb push_back
#define fi first
#define se second

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    vi a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    
    // Sort the A array to enable binary search later
    sort(a.begin(), a.end());
    
    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;
        
        // Use binary search to find the k-th closest element
        int pos = lower_bound(a.begin(), a.end(), b) - a.begin();
        
        int left = pos - 1;
        int right = pos;
        
        // Two pointers to simulate finding k-th smallest distance
        while (k > 0) {
            if (right >= N || (left >= 0 && b - a[left] <= a[right] - b)) {
                left--;
            } else {
                right++;
            }
            k--;
        }
        
        // The k-th closest distance will be between current left and right
        // So we check boundaries again to get valid distances
        ll min_dist = 1e18;
        if (left >= 0) min_dist = min(min_dist, (ll)(b - a[left]));
        if (right < N) min_dist = min(min_dist, (ll)(a[right] - b));
        
        cout << min_dist << "\n";
    }

    return 0;
}
