int main() {

    for(int i = 0; i < q; ++i) {
        long long b;
        int k;
        cin >> b >> k;
        for(int j = 0; j < n; ++j) {

        sort(distances.begin(), distances.end());

        distances.clear();
    }

    return 0;
}
