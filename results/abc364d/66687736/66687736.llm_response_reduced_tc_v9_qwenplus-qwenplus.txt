// Function to compute the k-th smallest distance from b to elements in a
ll kth_distance(vll& a_sorted, int k, int b) {
    int n = a_sorted.size();
    // Binary search on the answer (distance)
    ll low = 0, high = 2e8;
    while (low < high) {
        ll mid = (low + high) / 2;
        // Count how many elements are within [b - mid, b + mid]
        int left = lower_bound(all(a_sorted), b - mid) - a_sorted.begin();
        int right = upper_bound(all(a_sorted), b + mid) - a_sorted.begin();
        if ((right - left) >= k) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low;
}
