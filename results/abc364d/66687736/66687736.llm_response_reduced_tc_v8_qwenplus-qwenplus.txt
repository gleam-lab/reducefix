#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define pb push_back

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    vi a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    
    // Sort the A points for binary search usage later
    sort(a.begin(), a.end());
    
    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;
        
        // Find the position where B_j would be inserted in A to keep order
        int pos = lower_bound(a.begin(), a.end(), b) - a.begin();
        
        // We'll use two pointers starting at pos-1 and pos
        int left = pos - 1;
        int right = pos;
        
        int remaining = k;
        
        while (remaining > 0) {
            if (left < 0) {
                right++;
            } else if (right >= N) {
                left--;
            } else if (abs(a[left] - b) <= abs(a[right] - b)) {
                left--;
            } else {
                right++;
            }
            remaining--;
        }
        
        // The k-th closest is now either a[left] or a[right]
        // Let's find the correct one
        vl distances;
        if (left >= 0) distances.pb(abs((ll)a[left] - b));
        if (right < N) distances.pb(abs((ll)a[right] - b));
        
        // We need to collect all candidates within the needed range
        // Reset pointers and find all valid candidates
        left = pos - 1;
        right = pos;
        map<ll, int> distance_count;
        
        while (remaining < k) {
            if (left < 0) {
                ll d = abs((ll)a[right++] - b);
                distance_count[d]++;
                remaining++;
            } else if (right >= N) {
                ll d = abs((ll)a[left--] - b);
                distance_count[d]++;
                remaining++;
            } else if (abs((ll)a[left] - b) <= abs((ll)a[right] - b)) {
                ll d = abs((ll)a[left--] - b);
                distance_count[d]++;
                remaining++;
            } else {
                ll d = abs((ll)a[right++] - b);
                distance_count[d]++;
                remaining++;
            }
        }
        
        // Now find the k-th smallest distance
        int count = 0;
        for (auto &p : distance_count) {
            count += p.second;
            if (count >= k) {
                cout << p.first << "\n";
                break;
            }
        }
    }
}
