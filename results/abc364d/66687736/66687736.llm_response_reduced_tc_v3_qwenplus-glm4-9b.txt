#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for (int i = 0; i < n; i++)
#define vi vector<int>
#define vl vector<long long>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long

int main() {
    ll n, q; cin >> n >> q;
    vl a(n);
    rep(i, n) cin >> a[i];

    vl b(q), k(q);
    rep(i, q) {
        cin >> b[i] >> k[i];
    }

    // Create a dictionary to store the counts of each unique distance
    unordered_map<ll, ll> distanceCount;
    for (ll &b_j : b) {
        for (ll &a_i : a) {
            ll distance = abs(b_j - a_i);
            distanceCount[distance]++;
        }
    }

    // Sort the unique distances
    vl distances;
    for (auto &p : distanceCount) {
        distances.push_back(p.first);
    }
    sort(distances.begin(), distances.end());

    // For each query, find the k-th closest point to B_j
    rep(i, q) {
        ll b_j = b[i];
        ll k_j = k[i];
        ll cumulativeCount = 0;
        for (ll distance : distances) {
            cumulativeCount += distanceCount[distance];
            if (cumulativeCount >= k_j) {
                cout << distance << endl;
                break;
            }
        }
    }

    return 0;
}

This code initializes the necessary variables and input arrays, processes each query to find the k-th closest point to each B_j by calculating distances, and then sorts and counts the unique distances to answer the queries. The code assumes that the number of unique distances is manageable and does not exceed the constraints.
