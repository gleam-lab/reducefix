#include <bits/stdc++.h>

#define ll long long
#define vi vector<int>

#define rep(i, n) for (int i = 0; i < (int)n; ++i)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    vi a(N);
    for (int &x : a) cin >> x;
    
    // Sort the A points
    sort(all(a));
    
    // For each query, perform binary search to find the k-th smallest distance
    rep(q, Q) {
        int b, k;
        cin >> b >> k;
        
        // Use a lambda to compute distance from B_j to A[i]
        auto get_distance = [&](int idx) {
            return abs(a[idx] - b);
        };
        
        // Binary search on the distance
        int lo = 0, hi = 2e8 + 5;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            
            // We need to count how many A_i are in [b-mid, b+mid]
            int left = lower_bound(all(a), b - mid) - a.begin();
            int right = upper_bound(all(a), b + mid) - a.begin();
            int count = right - left;
            
            if (count < k) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        
        cout << lo << '\n';
    }
    
    return 0;
}
