#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vll vector<ll>
#define pb push_back
#define fi first
#define se second
#define all(v) v.begin(),v.end()

// Binary search + prefix sum approach

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, q;
    cin >> n >> q;
    
    // Read A points and sort them
    vi a(n);
    for(int i = 0; i < n; ++i){
        cin >> a[i];
    }
    sort(all(a));
    
    // Precompute prefix sums for computing median distances quickly
    vll prefix_sum(n+1, 0);
    for(int i = 0; i < n; ++i){
        prefix_sum[i+1] = prefix_sum[i] + 1LL * a[i];
    }
    
    auto solve = [&](int b, int k) -> ll {
        // Find the closest point using binary search
        int pos = lower_bound(all(a), b) - a.begin();
        
        // Two pointers to find k-th smallest distance
        int l = pos - 1, r = pos;
        int count = 0;
        while(count < k){
            if(l < 0){
                r++;
            }else if(r >= n){
                l--;
            }else{
                if(abs(a[l] - b) <= abs(a[r] - b)){
                    l--;
                }else{
                    r++;
                }
            }
            count++;
        }
        
        // Now we need to determine which was the k-th one
        // Reset pointers
        l = pos - 1, r = pos;
        int idx = -1;
        count = 0;
        while(count < k){
            if(l < 0){
                idx = r;
                r++;
            }else if(r >= n){
                idx = l;
                l--;
            }else{
                if(abs(a[l] - b) <= abs(a[r] - b)){
                    idx = l;
                    l--;
                }else{
                    idx = r;
                    r++;
                }
            }
            count++;
        }
        return abs(a[idx] - b);
    };
    
    for(int i = 0; i < q; ++i){
        int b, k;
        cin >> b >> k;
        cout << solve(b, k) << "\n";
    }
}
