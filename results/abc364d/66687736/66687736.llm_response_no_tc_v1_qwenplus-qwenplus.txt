#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<int>
#define vll vector<ll>
#define pb push_back

#define se second
#define all(a) a.begin(), a.end()

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    vi A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    // Sort the A array for binary search operations
    sort(all(A));
    
    // Precompute prefix for binary indexed distances
    vi prefix(N + 1, 0); // prefix[i] = number of elements <= A[i-1]
    
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = i + 1;
    }
    
    for (int q = 0; q < Q; ++q) {
        int b_j, k_j;
        cin >> b_j >> k_j;
        
        // Binary search on distance d
        int low = 0, high = 2e8 + 1;
        while (low < high) {
            int mid = (low + high) / 2;
            
            // Find how many points are within distance `mid` from b_j
            int left = lower_bound(all(A), b_j - mid) - A.begin();
            int right = upper_bound(all(A), b_j + mid) - A.begin();
            int count = right - left;
            
            if (count < k_j) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        
        cout << low << "\n";
    }
    
    return 0;
}
