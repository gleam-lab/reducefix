#include <bits/stdc++.h>

#define rep1(i, n) for (int i = 1; i <= (int)n; ++i)
#define all(v) v.begin(), v.end()

// Function to perform binary search on sorted array a
int solveQuery(int b, int k, const vi &a) {
    int left = 0, right = (int)a.size() - 1;
    
    // Find the closest index in a to b using binary search
    while (left < right) {
        int mid = (left + right) / 2;
        if (a[mid] < b) left = mid + 1;
        else right = mid;
    }
    
    // Now left is the index of the closest or first element >= b
    int l = left - 1, r = left;
    vi distances;
    
    // Collect k smallest distances around the closest point
    while (distances.size() < (size_t)k && (l >= 0 || r < a.size())) {
        if (l < 0) distances.push_back(abs(a[r++] - b));
        else if (r >= a.size()) distances.push_back(abs(a[l--] - b));
        else {
            int d1 = abs(a[l] - b), d2 = abs(a[r] - b);
            if (d1 <= d2) distances.push_back(d1), --l;
            else distances.push_back(d2), ++r;
        }
    }
    
    return distances[k - 1];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;

    vi a(N);
    rep(i, N) cin >> a[i];
    sort(all(a));

    rep(q, Q) {
        int b, k;
        cin >> b >> k;
        cout << solveQuery(b, k, a) << "\n";
    }

    return 0;
}
