#include <bits/stdc++.h>

#define vi vector<int>

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n,q;cin >> n >> q;
    vector<ll> a(n), distances(n);
    for(int i=0; i<n; i++) cin >> a[i];
    
    for(int i=0; i<q; i++){
        ll b, k; cin >> b >> k;
        for(int j=0; j<n; j++) distances[j] = abs(a[j] - b);
        sort(distances.begin(), distances.end());
        cout << distances[k-1] << '\n';
    }
    
    return 0;
}

This solution redefines the problem by calculating the absolute differences between each pair of points, sorting these differences, and then directly accessing the k-th smallest difference for each query. This approach ensures both simplicity and efficiency within the given constraints.
