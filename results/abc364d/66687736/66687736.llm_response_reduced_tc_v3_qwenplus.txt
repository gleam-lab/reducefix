#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vll vector<ll>

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    vll a(N);
    rep(i, N) cin >> a[i];
    
    sort(a.begin(), a.end());
    
    rep(qi, Q) {
        ll b;
        int k;
        cin >> b >> k;
        
        // We want to find the k-th smallest distance from B to any A_i
        // Instead of computing all distances, we can binary search on the answer (distance)
        // For a given distance d, count how many points A_i are within distance <= d from b
        // Then find the smallest d such that this count >= k
        
        ll left = -1;
        ll right = 200000000; // max possible coordinate difference is ~4e8, so 2e8 is safe upper bound on distance
        
        while (right - left > 1) {
            ll mid = (left + right) / 2;
            // Count number of A_i such that |a_i - b| <= mid
            ll low_bound = b - mid;
            ll high_bound = b + mid;
            
            auto it1 = lower_bound(a.begin(), a.end(), low_bound);
            auto it2 = upper_bound(a.begin(), a.end(), high_bound);
            int count = distance(it1, it2);
            
            if (count >= k) {
                right = mid;
            } else {
                left = mid;
            }
        }
        
        cout << right << '\n';
    }
}
