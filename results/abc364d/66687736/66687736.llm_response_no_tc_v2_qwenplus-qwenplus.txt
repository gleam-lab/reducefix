// Binary search to find the k-th smallest distance from B_j to A's
ll find_kth_distance(vi& a_sorted, int k, ll b_j) {
    int n = a_sorted.size();
    
    // For each query point b_j, we use binary search on possible distances
    ll low = 0;
    ll high = 2e8 + 10; // Since max coordinate difference is <= 2e8
    
    while (low < high) {
        ll mid = (low + high) / 2;
        ll cnt = 0;
        
        // Use sliding window or binary search per side of b_j
        // Left part: values in a <= b_j
        int l = lower_bound(all(a_sorted), b_j - mid) - a_sorted.data();
        int r = upper_bound(all(a_sorted), b_j + mid) - a_sorted.data();
        cnt = r - l;
        
        if (cnt < k) {
            low = mid + 1;
        } else {
            high = mid;
        }
    }
    
    return low;
}
