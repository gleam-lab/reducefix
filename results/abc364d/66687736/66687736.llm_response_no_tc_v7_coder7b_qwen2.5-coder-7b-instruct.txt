```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    int n, q;
    cin >> n >> q;
    vector<int> a(n), distances(n);
    
    // Read coordinates of points A_i
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    // Process each query
    for(int qi = 0; qi < q; ++qi) {
        int b, k;
        cin >> b >> k;
        
        // Calculate distances from point B_j to all points A_i
        for(int i = 0; i < n; ++i) {
            distances[i] = abs(a[i] - b);
        }
        
        // Sort the distances
        sort(distances.begin(), distances.end());
        
        // The k-th smallest distance
        int result = distances[k-1];
        
        // Output the result
        cout << result << '\n';
    }
    
    return 0;
}
```
