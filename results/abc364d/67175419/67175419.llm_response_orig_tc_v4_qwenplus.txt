```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 1e5 + 5;
int n, q;
long long a[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a, a + n);

    while (q--) {
        long long b;
        int k;
        cin >> b >> k;

        // Find the position where b would be inserted
        auto it = lower_bound(a, a + n, b);
        int pos = it - a;

        // We need to consider k closest points around position pos
        // Use two pointers expanding from pos
        int left = pos - 1;
        int right = pos;
        int count = 0;
        long long result = 0;

        // Expand until we have found the k-th closest point
        while (count < k) {
            long long dist_left = (left >= 0) ? abs(a[left] - b) : LLONG_MAX;
            long long dist_right = (right < n) ? abs(a[right] - b) : LLONG_MAX;

            if (dist_left <= dist_right) {
                result = dist_left;
                left--;
            } else {
                result = dist_right;
                right++;
            }
            count++;
        }

        cout << result << '\n';
    }

    return 0;
}
```
