#include<bits/stdc++.h>
using namespace std;
#define ll long long

struct Point {
    ll x, index;
};

bool cmp(Point &p1, Point &p2) {
    return p1.x < p2.x;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    ll n, q;
    cin >> n >> q;
    
    vector<ll> a(n);
    for(ll i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    vector<Point> points(n);
    for(ll i = 0; i < n; i++) {
        points[i].x = a[i];
        points[i].index = i;
    }
    
    sort(points.begin(), points.end(), cmp);
    
    while(q--) {
        ll b, k;
        cin >> b >> k;
        
        // Binary search for the k-th smallest distance
        ll low = 0, high = 1e18, ans = -1;
        while(low <= high) {
            ll mid = (low + high) / 2;
            
            vector<ll> distances;
            for(ll i = 0; i < n; i++) {
                distances.push_back(abs(points[i].x - b));
            }
            
            sort(distances.begin(), distances.end());
            
            if(distances[k - 1] <= mid) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        
        cout << ans << "\n";
    }
    
    return 0;
}
