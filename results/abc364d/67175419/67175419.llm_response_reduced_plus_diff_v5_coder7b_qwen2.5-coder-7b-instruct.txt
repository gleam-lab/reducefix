const int MAXN = 1e5 + 5;
vector<int> A[MAXN], B[MAXN];

int binary_search(vector<int>& arr, int target) {
    int left = 0, right = arr.size() - 1, result = -1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) {
            left = mid + 1;
            result = mid;
        } else right = mid - 1;
    }
    return result;
}
