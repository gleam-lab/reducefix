#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200000;
int a[MAXN], n, q;
vector<pair<long long, int>> queries;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> q;
    for (int i = 0; i < n; ++i) cin >> a[i];
    sort(a, a + n);

    vector<long long> prefix(n + 1, 0);
    for (int i = 0; i < n; ++i) prefix[i + 1] = prefix[i] + a[i];

    auto query = [&](long long b, int k) -> long long {
        int pos = lower_bound(a, a + n, b) - a;
        int l = pos - 1, r = pos;
        multiset<long long> dists;

        while (k--) {
            if (l >= 0 && (r == n || abs(a[l] - b) <= abs(a[r] - b))) {
                dists.insert(abs(a[l] - b));
                --l;
            } else {
                dists.insert(abs(a[r] - b));
                ++r;
            }
        }

        return *next(dists.begin(), k);
    };

    for (int i = 0; i < q; ++i) {
        long long b;
        int k;
        cin >> b >> k;
        cout << query(b, k) << "\n";
    }

    return 0;
}
