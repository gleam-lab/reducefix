#include <bits/stdc++.h>
using namespace std;

vector<int> sort_and_get_kth(vector<int>& arr, int k) {
    vector<int> sorted_arr = arr;
    sort(sorted_arr.begin(), sorted_arr.end());
    return {sorted_arr[k-1]};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    vector<int> a(n);
    for(int i = 0; i < n; ++i)
        cin >> a[i];

    while(q--) {
        int b, k;
        cin >> b >> k;
        auto result = sort_and_get_kth(a, k);
        cout << result[0] - b << '\n';
    }

    return 0;
}
