#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;

int N, Q;
long long a[MAXN];

// Binary search to find the smallest distance such that there are at least k_j elements within that distance from b_j
long long queryKthSmallestDistance(long long b, int k) {
    // Binary search bounds for the answer
    long long low = 0, high = 2e14; // since coordinates can be up to 1e8 in magnitude, max distance is ~2e8, but we use 2e14 to be safe
    while (low < high) {
        long long mid = (low + high) / 2;
        // Use two pointers approach to count how many A[i] are within distance mid of b
        int count = 0;
        int left = 0;
        for (int right = 0; right < N; ++right) {
            while (a[right] - a[left] > mid) {
                ++left;
            }
            count += right - left + 1;
        }
        if (count >= k) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> N >> Q;
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    sort(a, a + N);
    
    for (int i = 0; i < Q; ++i) {
        long long b;
        int k;
        cin >> b >> k;
        cout << queryKthSmallestDistance(b, k) << "\n";
    }

    return 0;
}
