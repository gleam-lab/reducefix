#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

vector<ll> v;

ll solve(ll target, int k) {
    auto it = lower_bound(v.begin(), v.end(), target);
    if(it == v.end() || *it != target) return -1;
    return it - v.begin() + 1;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) cin >> a[i];

    sort(a.begin(), a.end());

    while(q--) {
        ll b, k;
        cin >> b >> k;
        v.clear();
        for(int i = 0; i < n; i++) v.push_back(abs(a[i] - b));
        sort(v.begin(), v.end());
        cout << solve(b, k) << "\n";
    }

    return 0;
}
