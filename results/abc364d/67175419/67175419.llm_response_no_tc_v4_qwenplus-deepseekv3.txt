3. **Check Exact Distance**: After determining the smallest such distance `d`, ensure that exactly k points are closest by checking the boundaries and possibly adjusting for ties.

const int N = 1e5 + 5;
int a[N];

int countWithinRange(int b, int d) {
    long long left = (long long)b - d;
    long long right = (long long)b + d;
    int lower = lower_bound(a + 1, a + n + 1, left) - a;
    int upper = upper_bound(a + 1, a + n + 1, right) - a;

int findKthDistance(int b, int k) {
    if (n == 1) return abs(a[1] - b);
    
    int left = 0, right = 2e8;
    int answer = right;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        int cnt = countWithinRange(b, mid);
        
        if (cnt >= k) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    return answer;
}
