#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
int a[MAXN];

// Binary search for the k-th smallest distance using prefix sums of sorted distances from B_j

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    sort(a, a + N);

    while (Q--) {
        long long b, k;
        cin >> b >> k;

        // Binary search bounds for the answer
        long long low = 0;
        long long high = 2e8 + 5;

        auto count_less_equal = [&](long long mid) {
            // Use lower and upper bounds to find how many elements are within [b - mid, b + mid]
            int left = lower_bound(a, a + N, b - mid) - a;
            int right = upper_bound(a, a + N, b + mid) - a;
            return right - left;
        };

        // Binary search for the minimal value that has at least k points within distance <= that value
        long long answer = 0;
        while (low <= high) {
            long long mid = (low + high) / 2;
            int cnt = count_less_equal(mid);
            if (cnt < k) {
                low = mid + 1;
            } else {
                answer = mid;
                high = mid - 1;
            }
        }

        cout << answer << "\n";
    }

    return 0;
}
