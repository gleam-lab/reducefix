#include<bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;
int n, q;
long long a[N];

// Binary search on answer: check if there are at least k points within distance 'd' from point b
bool check(long long d, long long b, int k) {
    // Find the leftmost index where a[i] >= b - d
    int left_idx = lower_bound(a, a + n, b - d) - a;
    // Find the rightmost index where a[i] <= b + d (so upper_bound gives first > b+d)
    int right_idx = upper_bound(a, a + n, b + d) - a;
    return (right_idx - left_idx) >= k;
}

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    
    cin >> n >> q;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a, a + n);
    
    while (q--) {
        long long b;
        int k;
        cin >> b >> k;
        
        // Binary search on the distance
        long long l = 0, r = 2e8; // Max possible distance is 2*10^8
        long long ans = r;
        
        while (l <= r) {
            long long mid = (l + r) / 2;
            if (check(mid, b, k)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        
        cout << ans << '\n';
    }
    
    return 0;
}
