#include <bits/stdc++.h>
using namespace std;

const int N = 1 << 17;

int n, q;
vector<long long> a;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> q;
    a.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (q--) {
        long long b, k;
        cin >> b >> k;

        // Binary search on distance
        long long low = 0, high = 2e14; // Larger than max possible distance
        while (low < high) {
            long long mid = (low + high) / 2;
            // Find how many elements have distance <= mid using two pointers
            int count = 0;
            int left = 0, right = 0;
            for (int i = 0; i < n; ++i) {
                while (left < n && abs(a[left] - b) > mid) left++;
                if (left <= i) right = max(right, i + 1);
                while (right < n && abs(a[right] - b) <= mid) right++;
                count += max(0, right - i);
                if (count >= k) break;
            }

            if (count >= k) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        cout << low << "\n";
    }

    return 0;
}
