#include<bits/stdc++.h>
using namespace std;
const int N = 2e5 + 5;
int n, q;
int a[N], b[N], k[N];

int compare(pair<int, int>& left, pair<int, int>& right) {
    return left.second < right.second;
}

int main() {
    cin.tie(0), cout.tie(0);
    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < q; ++i) {
        cin >> b[i] >> k[i];
    }

    // Create a vector of pairs where each pair is (distance, index in a)
    vector<pair<int, int>> distances(n);
    for (int i = 0; i < n; ++i) {
        int dist = abs(a[i] - b[k[i] - 1]);
        distances[i] = {dist, i};
    }

    // Sort distances to get them in ascending order
    sort(distances.begin(), distances.end(), compare);

    // Output the k-th smallest distance for each point B_j
    for (int i = 0; i < q; ++i) {
        cout << distances[k[i] - 1].first << "\n";
    }

    return 0;
}
