#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 1e5 + 5;

struct Point {
    ll x, id;
};

bool cmp(const Point &a, const Point &b) {
    return a.x < b.x;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int n, q;
    cin >> n >> q;

    vector<Point> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i].x;
        A[i].id = i + 1;
    }

    sort(A.begin(), A.end(), cmp);

    while (q--) {
        ll b, k;
        cin >> b >> k;

        int left = 0, right = n - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (A[mid].x >= abs(b - A[0].x)) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        int idx = left - 1;
        if (idx == -1) idx = 0;
        if (idx >= n) idx = n - 1;

        int count = 0;
        for (int i = 0; i < n; ++i) {
            if (abs(A[i].x - b) < abs(A[idx].x - b)) {
                ++count;
                if (count == k) {
                    cout << abs(A[i].x - b) << '\n';
                    break;
                }
            }
        }
    }

    return 0;
}
