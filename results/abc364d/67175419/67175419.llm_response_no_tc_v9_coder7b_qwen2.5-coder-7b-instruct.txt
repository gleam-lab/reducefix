#include<bits/stdc++.h>
using namespace std;
const int N=2e5+5;
int n,q,k,a[N],dist[N];

int findKthClosest(int x, int k) {
    vector<int> distances;
    for(int i = 1; i <= n; ++i) {
        distances.push_back(abs(a[i] - x));
    }
    sort(distances.begin(), distances.end());
    return distances[k-1];
}

int main() {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    cin >> n >> q;
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    while(q--) {
        int x, k;
        cin >> x >> k;
        cout << findKthClosest(x, k) << "\n";
    }
    return 0;
}
