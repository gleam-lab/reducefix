```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 5;

int n, q;
int a[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    sort(a + 1, a + n + 1);

    while (q--) {
        int b, k;
        cin >> b >> k;

        // Binary search on distance
        int left = 0, right = 2e8;
        int answer = -1;

        while (left <= right) {
            int mid = (left + right) / 2;

            // Find first A_i >= b - mid
            int low = lower_bound(a + 1, a + n + 1, b - mid) - a;
            // Find first A_i > b + mid
            int high = upper_bound(a + 1, a + n + 1, b + mid) - a;

            if (high - low >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        cout << answer << "\n";
    }

    return 0;
}
```
