```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;
int n, q, a[N];

// Function to find the k-th smallest distance
int findKthSmallest(int b_j, int k_j) {
    vector<int> distances;
    // Calculate all distances from B_j to all points A_i
    for(int i = 1; i <= n; ++i) {
        distances.push_back(abs(a[i] - b_j));
    }
    // Sort the distances
    sort(distances.begin(), distances.end());
    // Return the k-th smallest distance (1-based index)
    return distances[k_j - 1];
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> q;
    // Read coordinates of points A_i
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    // Process each query
    while(q--) {
        int b_j, k_j;
        cin >> b_j >> k_j;
        // Output the k-th smallest distance
        cout << findKthSmallest(b_j, k_j) << "\n";
    }

    return 0;
}
```
