```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN = 1e5 + 5;

int n, m, k;
ll a[MAXN], b[MAXN];

struct Point {
    ll x, id;
};

bool cmp(const Point &p1, const Point &p2) {
    return p1.x < p2.x;
}

vector<ll> findKthClosest(ll x, vector<Point> &points, int k) {
    sort(points.begin(), points.end(), [x](const Point &p1, const Point &p2) {
        return abs(p1.x - x) < abs(p2.x - x);
    });
    
    vector<ll> result(k);
    for(int i = 0; i < k; ++i) {
        result[i] = abs(points[i].x - x);
    }
    return result;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    cin >> n >> m;
    for(int i = 1; i <= n; ++i) cin >> a[i];
    for(int i = 1; i <= m; ++i) {
        cin >> b[i] >> k;
        vector<Point> points(n);
        for(int j = 0; j < n; ++j) {
            points[j] = {a[j + 1], j + 1};
        }
        vector<ll> distances = findKthClosest(b[i], points, k);
        cout << distances[k - 1] << '\n';
    }

    return 0;
}
```
