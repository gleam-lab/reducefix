```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 5;

int n, q;
int a[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a, a + n);

    while (q--) {
        int b, k;
        cin >> b >> k;

        // Binary search on the distance from b
        int l = 0, r = 2e8;
        int ans = 0;

        while (l <= r) {
            int mid = (l + r) / 2;
            // Find the first index >= b - mid
            int left = lower_bound(a, a + n, b - mid) - a;
            // Find the first index > b + mid
            int right = upper_bound(a, a + n, b + mid) - a;
            int count = right - left;

            if (count >= k) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
```
