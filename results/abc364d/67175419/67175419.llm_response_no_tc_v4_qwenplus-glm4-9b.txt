
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 2e5 + 5;

int main() {
    int n, q;
    cin >> n >> q;
    vector<int> a(N), b(q), k(q);
    
    // Reading the coordinates of points A
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Reading the coordinates of points B and corresponding k_j values
    for (int i = 0; i < q; i++) {
        cin >> b[i] >> k[i];
    }
    
    // Sort points A to prepare for binary search
    sort(a.begin(), a.end());
    
    // For each point B, find the k-th closest point in A
    for (int i = 0; i < q; i++) {
        // Binary search to find the correct distance
        int low = 0, high = n - 1, best_dist = 1e8;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            // Calculate distance and update best distance
            int dist = abs(a[mid] - b[i]);
            if (dist < best_dist) {
                best_dist = dist;
                low = mid + 1; // Continue to the right if not the best
            } else {
                high = mid - 1; // Stop if better distance is not possible on the right
            }
        }
        
        // Output the best distance found
        cout << best_dist << "\n";
    }
    
    return 0;
}
```
