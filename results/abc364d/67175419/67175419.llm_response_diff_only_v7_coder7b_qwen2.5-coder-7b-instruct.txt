```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;
int n, m, k, sum, cnt, a[N];

void solve() {
    cin >> n >> m;
    vector<int> distances(N);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    sort(a + 1, a + n + 1);

    while (m--) {
        int x, k;
        cin >> x >> k;

        // Calculate all distances from point x to other points
        for (int i = 1; i <= n; ++i) {
            distances[i] = abs(a[i] - x);
        }

        // Sort the distances
        sort(distances.begin(), distances.end());

        // Output the k-th smallest distance
        cout << distances[k - 1] << "\n";
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    solve();

    return 0;
}
```
