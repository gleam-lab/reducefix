    vector<m_int> a(n);
    
    for (m_int i = 0; i < n; ++i) {
        cin >> a[i];

        // Priority queue to store the k smallest distances
        priority_queue<pair<m_int, int>, vector<pair<m_int, int>>, greater<>> pq;

            m_int dist = abs(a[i] - b);
            if (pq.size() < k) {
                pq.push({dist, i});
            } else if (dist < pq.top().first) {
                pq.pop();
                pq.push({dist, i});

        }

    }
    
    return 0;
}
