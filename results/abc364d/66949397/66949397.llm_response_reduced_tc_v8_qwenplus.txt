```cpp
#include <vector>
#include <iostream>
#include <algorithm>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;

int main() {
	
	m_int n, q;
	cin >> n >> q;
	vector<m_int> a(n);
	
	for(m_int i = 0; i < n; ++i) {
		cin >> a[i];
	}
	std::sort(a.begin(), a.end());

	while(q--) {
		
		m_int b, k;
		cin >> b >> k;

		auto f = [&](m_int x) -> bool {
			// Count number of A_i such that |a_i - b| <= x
			auto it_l = std::lower_bound(a.begin(), a.end(), b - x);
			auto it_r = std::upper_bound(a.begin(), a.end(), b + x);
			return (it_r - it_l) >= k;
		};

		// The maximum possible distance is up to 2e8, but k-th could be large if spread out
		// However, coordinates are within [-1e8, 1e8], so max distance is about 2e8
		// But we need to allow for worst-case: e.g., one point at -1e8, query at 1e8 => dist ~ 2e8
		// So set r high enough: 300 million should suffice, but let's use 3e8 to be safe
		m_int l = -1;
		m_int r = 300000000; // sufficient for max distance between -1e8 and 1e8

		while(l + 1 < r) {
			m_int mid = (l + r) / 2;
			if (f(mid)) {
				r = mid;
			} else {
				l = mid;
			}
		}
		cout << r << "\n";
	}
}
```
