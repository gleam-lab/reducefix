
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

int main() {
    int n, q;
    cin >> n >> q;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (q--) {
        int b, k;
        cin >> b >> k;

        int dist; // Distance of the k-th closest point
        // Find the k-th closest point
        if (k <= n) {
            dist = abs(a[k - 1] - b);
        } else { // If k > n, the k-th closest point is the farthest A_i (at the end of vector a)
            dist = abs(a[n - 1] - b);
        }
        cout << dist << "\n";
    }

    return 0;
}
```
