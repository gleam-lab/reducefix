```cpp
#include <vector>
#include <iostream>
#include <algorithm>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;

int main() {
    
    m_int n, q; cin >> n >> q;
    vector<m_int> a(n);
    
    for(m_int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    std::sort(a.begin(), a.end());

    while(q--) {
        
        m_int b, k; cin >> b >> k;

        auto f = [&](m_int x) -> bool {
            
            // Count points in [b-x, b+x]
            auto it_l = std::lower_bound(a.begin(), a.end(), b - x);
            auto it_r = std::upper_bound(a.begin(), a.end(), b + x);
            return (it_r - it_l) >= k;
        };

        // Binary search on distance
        m_int l = -1; 
        m_int r = 2e9 + 1;  // Increased upper bound to handle large distances
        
        while(l + 1 < r) {
            m_int mid = l + (r - l) / 2;  // Avoid potential overflow
            if (f(mid)) {
                r = mid;
            } else {
                l = mid;
            }
        }
        cout << r << "\n";
    }
}
```
