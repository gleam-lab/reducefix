```cpp
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;

int main() {
    m_int n, q; cin >> n >> q;
    vector<m_int> a(n);
    
    for (m_int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    std::sort(a.begin(), a.end());

    while (q--) {
        m_int b, k; cin >> b >> k;

        auto f = [&](m_int x) -> bool {
            // count how many points are within distance x of b
            return std::count_if(a.begin(), a.end(), [b, x](m_int ai) { return std::abs(ai - b) <= x; }) >= k;
        };

        m_int l = 0, r = 1e8 + 1;
        while (r - l > 1) {
            m_int mid = (l + r) / 2;
            if (f(mid)) {
                r = mid;
            } else {
                l = mid;
            }
        }
        cout << r << "\n";
    }

    return 0;
}
```

This solution addresses the problem by using binary search to find the minimum distance `x` such that there are at least `k` points within this distance from the target point `B_j`. The function `f(x)` checks how many points are within distance `x` of `B_j`, and the binary search narrows down the possible values of `x` until the correct one is found. This approach ensures that the solution is efficient even for large inputs.
