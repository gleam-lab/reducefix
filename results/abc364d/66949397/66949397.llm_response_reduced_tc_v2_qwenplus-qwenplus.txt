```cpp
#include <vector>
#include <iostream>
#include <algorithm>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;

int main() {
    m_int n, q;
    cin >> n >> q;
    vector<m_int> a(n);
    
    for (m_int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (q--) {
        m_int b, k;
        cin >> b >> k;

        // Binary search on distance
        m_int left = 0;
        m_int right = 2e18; // Large enough upper bound

        auto count_points_within_distance = [&](m_int x) {
            // Find range of values in a where |a_i - b| <= x
            m_int low = b - x;
            m_int high = b + x;
            auto it_l = lower_bound(a.begin(), a.end(), low);
            auto it_r = upper_bound(a.begin(), a.end(), high);
            return static_cast<m_int>(it_r - it_l);
        };

        // Binary search to find smallest distance that has at least k points within
        while (left < right) {
            m_int mid = (left + right) / 2;
            if (count_points_within_distance(mid) >= k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }

        cout << left << "\n";
    }

    return 0;
}
```
