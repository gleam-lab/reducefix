```cpp
#include <vector>
#include <iostream>
#include <queue>
#include <algorithm>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;
using std::priority_queue;

struct Distance {
    m_int dist;
    int index;
};

bool operator<(const Distance& a, const Distance& b) {
    return a.dist > b.dist;
}

int main() {
    m_int n, q;
    cin >> n >> q;
    vector<m_int> a(n);
    
    for (m_int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    while (q--) {
        m_int b, k;
        cin >> b >> k;
        
        // Min-heap to store distances
        priority_queue<Distance> pq;
        
        for (m_int i = 0; i < n; ++i) {
            m_int dist = abs(a[i] - b);
            pq.push({dist, i});
            
            if (pq.size() > k) {
                pq.pop();
            }
        }
        
        // The top of the heap contains the k-th smallest distance
        cout << pq.top().dist << "\n";
    }
    
    return 0;
}
```

### Explanation:
1. **Priority Queue**: We use a min-heap to keep track of the k smallest distances. This ensures that we only keep the relevant points for each query.
2. **Sorting**: For each query, we calculate the distance from `b` to each point in `a`.
3. **Heap Operations**: We push each distance into the heap. If the size of the heap exceeds `k`, we pop the largest element to ensure that the heap always contains the k smallest distances.
4. **Result**: After processing all points, the top element of the heap will be the k-th smallest distance, which is the required answer.

This approach ensures that we handle the constraints efficiently and correctly find the k-th closest point for each query.
