```cpp
#include <vector>
#include <iostream>
#include <algorithm>

using m_int = long long;
using std::cin;
using std::cout;
using std::vector;

int main() {
	
	m_int n, q; 
	cin >> n >> q;
	vector<m_int> a(n);
	
	for(m_int i = 0; i < n; ++i) {
		cin >> a[i];
	}
	std::sort(a.begin(), a.end());

	while(q--) {
		
		m_int b, k;
		cin >> b >> k;

		auto f = [&](m_int x) -> bool {
			// Count number of points A_i such that |a_i - b| <= x
			// This means: b - x <= a_i <= b + x
			auto it_l = std::lower_bound(a.begin(), a.end(), b - x);
			auto it_r = std::upper_bound(a.begin(), a.end(), b + x);
			return (it_r - it_l) >= k;
		};

		// Binary search on the answer (distance)
		m_int l = -1;
		m_int r = 2e9 + 10; // Maximum possible distance is ~2 * 10^8

		while(l + 1 < r) {
			m_int mid = (l + r) / 2;
			if (f(mid)) {
				r = mid;
			} else {
				l = mid;
			}
		}
		cout << r << "\n";
	}
}
```
