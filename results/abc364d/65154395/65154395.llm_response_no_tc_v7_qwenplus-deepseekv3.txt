void solve() {

        cin >> a[i];

        int k;
        cin >> b >> k;
        
        int left = 0, right = N - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (a[mid] <= b) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        int i = right;
        int j = right + 1;
        vector<ll> distances;
        for (int cnt = 0; cnt < k && (i >= 0 || j < N); ) {
            if (i >= 0 && j < N) {
                if (abs(a[i] - b) <= abs(a[j] - b)) {
                    distances.push_back(abs(a[i] - b));

                i--;

                distances.push_back(abs(a[j] - b));
