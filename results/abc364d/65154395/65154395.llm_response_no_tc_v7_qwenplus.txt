#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 1e5 + 5;

int a[MAXN];

void solve() {
    int N, Q;
    cin >> N >> Q;
    
    // Read the A points
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    
    // Sort the A array to enable binary search around B_j
    sort(a, a + N);
    
    // Process each query
    for (int j = 0; j < Q; j++) {
        int b, k;
        cin >> b >> k;
        
        // We'll find the k-th smallest distance from B to any A_i
        // Instead of computing all distances, we can use binary search on answer
        
        // But simpler and efficient enough: find position where b would be inserted
        // Then expand left and right using two pointers to collect k closest points
        
        // Find lower bound of b in sorted a
        auto it = lower_bound(a, a + N, b);
        int pos = it - a;
        
        // Use two pointers: left starting at pos-1, right starting at pos
        int left = pos - 1;
        int right = pos;
        int count = 0;
        ll ans = 0;
        
        // We need to find the k-th smallest distance
        // Simulate merging left and right distances until we reach k-th
        while (count < k) {
            ans = 0;
            ll left_dist = (left >= 0) ? (ll)(b - a[left]) : LLONG_MAX;
            ll right_dist = (right < N) ? (ll)(a[right] - b) : LLONG_MAX;
            
            // Choose the smaller distance
            if (left_dist <= right_dist) {
                ans = left_dist;
                left--;
            } else {
                ans = right_dist;
                right++;
            }
            count++;
        }
        
        cout << ans << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
