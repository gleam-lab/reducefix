#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

// Binary search for the k-th smallest distance to point b in sorted a
ll kth_distance(const vector<ll>& a_sorted, ll b, int k) {
    int n = a_sorted.size();
    // Find the position where b would be inserted (first index >= b)
    int right = lower_bound(a_sorted.begin(), a_sorted.end(), b) - a_sorted.begin();
    int left = right - 1;

    // We'll use two pointers: left and right to simulate a merge of two reverse-sorted arrays
    int count = 0;
    while (count < k) {
        if (left < 0) {
            ++right;
        } else if (right >= n) {
            --left;
        } else if (abs(a_sorted[left] - b) <= abs(a_sorted[right] - b)) {
            --left;
        } else {
            ++right;
        }
        ++count;
    }

    // After loop, we backtrack one step to get the k-th closest
    if (left < 0) {
        return abs(a_sorted[right - 1] - b);
    } else if (right >= n) {
        return abs(a_sorted[left + 1] - b);
    } else if (left >= 0 && right < n) {
        if (abs(a_sorted[left + 1] - b) < abs(a_sorted[right] - b)) {
            return abs(a_sorted[left + 1] - b);
        } else {
            return abs(a_sorted[right] - b);
        }
    }
    return 0; // Should not reach here
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;
        cout << kth_distance(A, b, k) << '\n';
    }

    return 0;
}
