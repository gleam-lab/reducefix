```cpp
#include <bits/stdc++.h>
using namespace std;

vector<int> a, b;
int n, q;

void build(vector<int>& v) {
    sort(v.begin(), v.end());
    vector<int> s(v.size() + 1);
    for (int i = 0; i < v.size(); ++i) s[i + 1] = s[i] + v[i];
    swap(s, v);
}

int query(vector<int>& v, int k) {
    int lo = 0, hi = v.size() - 1;
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (v[mid] < k) lo = mid + 1;
        else hi = mid;
    }
    return lo;
}

int main() {
    cin >> n >> q;
    a.resize(n), b.resize(q);
    for (auto& x : a) cin >> x;
    for (auto& x : b) cin >> x >> a[--x];

    build(a);

    for (int i = 0; i < q; ++i) {
        int pos = query(a, b[i]);
        cout << a[pos] << '\n';
    }

    return 0;
}
```
