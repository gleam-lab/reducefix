```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        vector<ll> distances(N);
        for (int i = 0; i < N; ++i) {
            distances[i] = abs(A[i] - b);
        }
        nth_element(distances.begin(), distances.begin() + (k - 1), distances.end());
        cout << distances[k - 1] << '\n';
    }
    
    return 0;
}
```

1. **Input Handling**: The input is read, including the number of points N and the number of queries Q. The coordinates of points A_i are stored in a vector and sorted once to facilitate efficient query processing.

2. **Query Processing**: For each query, the point B_j and the integer k_j are read. The distances from B_j to each A_i are computed and stored in a temporary vector.

3. **Quickselect for k-th Smallest Distance**: The `nth_element` function is used to partially sort the distances vector such that the k_j-th smallest distance is placed at the (k_j - 1)-th index.
