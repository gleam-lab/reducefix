```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int N = 2e5 + 5;

vector<int> a(N);

bool check(int mid, int x, int y){
    int l = max(0, x - mid), r = min(N - 1, x + mid);
    int sum = a[r] - (l ? a[l - 1] : 0);
    return sum >= y;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; i++){
        int x; cin >> x;
        a[x]++;
    }
    for(int i = 1; i < N; i++){
        a[i] += a[i - 1];
    }
    for(int i = 0; i < q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 1e9;
        while(l < r){
            int mid = l + r >> 1;
            if(check(mid, x, y)) r = mid;
            else l = mid + 1;
        }
        cout << l << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
```

1. Adjusted the array size `N` to match the problem constraints (`10^5`).

4. Corrected the logic inside the `check` function to ensure it correctly counts elements within the specified range.
