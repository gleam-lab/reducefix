#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query(ll b, ll k, int n) {
    // We will use two pointers to maintain a sliding window of size k
    ll res = 0;
    
    // Find the k-th smallest |a[i] - b| using binary search on answer
    ll low = 0, high = 2e8; // Distance can't be more than 2e8 (from coord range)
    while (low < high) {
        ll mid = (low + high) / 2;
        
        // Count how many points have distance <= mid from b
        // This is equivalent to counting points in [b-mid, b+mid]
        int left = lower_bound(a, a + n, b - mid) - a;
        int right = upper_bound(a, a + n, b + mid) - a;
        int count = right - left;
        
        if (count >= k) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    
    sort(a, a + N);
    
    for (int i = 0; i < Q; ++i) {
        ll b, k;
        cin >> b >> k;
        cout << query(b, k, N) << "\n";
    }
    
    return 0;
}
