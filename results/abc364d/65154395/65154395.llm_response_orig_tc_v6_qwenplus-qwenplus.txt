#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

// Binary search for the k-th smallest distance
int binarySearchKthDistance(vector<int>& A, int b, int k) {
    int left = 0;
    int right = (int)1e9; // max possible distance is 2*1e8
    int answer = -1;
    
    while (left <= right) {
        int mid = (left + right) / 2;
        
        // Count how many elements in A are within distance <= mid from b
        int lower = b - mid;
        int upper = b + mid;
        
        auto low = lower_bound(A.begin(), A.end(), lower);
        auto high = upper_bound(A.begin(), A.end(), upper);
        
        int count = high - low;
        
        if (count >= k) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    return answer;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    sort(A.begin(), A.end());
    
    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;
        cout << binarySearchKthDistance(A, b, k) << "\n";
    }
}

int main() {
    solve();
    return 0;
}
