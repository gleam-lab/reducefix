#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 1e5 + 5;
vector<ll> a(MAXN);

void solve(void) {
    int n, q; cin >> n >> q;
    vector<ll> b(q);
    vector<int> k(q);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    for(int i = 0; i < q; i++) {
        cin >> b[i] >> k[i];
    }

    sort(a.begin(), a.end());
    
    for(int i = 0; i < q; i++) {
        auto it = lower_bound(a.begin(), a.end(), b[i]);
        int idx = it - a.begin();
        if(idx == 0) {
            cout << abs(b[i] - a[idx]) << endl;
        } else if(idx == n) {
            cout << abs(b[i] - a[idx - 1]) << endl;
        } else {
            int dist1 = abs(b[i] - a[idx]);
            int dist2 = abs(b[i] - a[idx - 1]);
            if(k[i] == 1 && dist1 < dist2 || k[i] != 1 && dist1 <= dist2) {
                cout << dist1 << endl;
            } else {
                cout << dist2 << endl;
            }
        }
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    solve();
    return 0;
}
