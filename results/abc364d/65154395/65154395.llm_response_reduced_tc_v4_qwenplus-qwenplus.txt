#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query(ll b, ll k, int n) {
    // We will find the smallest radius 'd' such that there are at least 'k' points within distance 'd' of 'b'
    ll left = 0, right = 2e8;
    ll ans = -1;

    while (left <= right) {
        ll mid = (left + right) / 2;
        ll low = b - mid;
        ll high = b + mid;

        // Find number of elements in [low, high] using binary search
        ll* l = lower_bound(a, a + n, low);
        ll* r = upper_bound(a, a + n, high);

        if ((r - l) >= k) {
            ans = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    return ans;
}

void solve() {
    int n, q;
    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a, a + n);

    for (int i = 0; i < q; ++i) {
        ll b, k;
        cin >> b >> k;
        cout << query(b, k, n) << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
