
```cpp
#include <bits/stdc++.h>
#define endl "\n"
using namespace std;
typedef long long ll;

const int N = 2e5 + 5;

int n, q;
int a[N];
int b[N], k[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= q; ++i) {
        cin >> b[i] >> k[i];
    }

    sort(a + 1, a + n + 1);
    for (int i = 0; i < q; ++i) {
        int x = b[i];
        int idx = upper_bound(a + 1, a + n + 1, x) - (a + 1) - k[i] + 1;
        if (idx == 0) {
            cout << 1e17 << endl;
        } else {
            cout << abs(x - a[idx]) << endl;
        }
    }

    return 0;
}
```

This solution correctly sorts the array `a` and then for each query calculates the index using `upper_bound` to find the `k_i`-th closest point to `b_i`. If the index is 0, it means we are looking for the closest point to the left of `b_i`, which would be the first point of the sorted array. If the index is larger than the number of points, it means we are looking for a point outside the range of `a`, which would be infinity. The output is then either the distance to the closest point or 1e17, the maximum possible distance.
