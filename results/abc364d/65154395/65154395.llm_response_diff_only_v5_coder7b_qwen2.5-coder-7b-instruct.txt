```cpp
#include<bits/stdc++.h>
#define endl "\n"
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int MAXN = 100005;
vector<int> a[MAXN];

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr);
    int n, q; cin >> n >> q;
    for(int i = 0; i < n; ++i) {
        int ai; cin >> ai;
        a[ai].push_back(i + 1);
    }
    for(int i = 0; i < MAXN; ++i)
        sort(a[i].begin(), a[i].end());
    
    for(int i = 0; i < q; ++i) {
        int bi, ki; cin >> bi >> ki;
        --ki;
        if(ki < a[bi].size())
            cout << abs(bi - a[bi][ki]) << endl;
        else
            cout << "No valid point found" << endl; // Handle this case appropriately
    }
    return 0;
}
```

### Explanation of Fixes:
1. **Corrected Array Size**: Changed `N` to `MAXN` and adjusted the array declaration accordingly to handle up to 100,000 points.
2. **Input Handling**: Adjusted the loop to correctly read coordinates and store them in the appropriate arrays.
3. **Sorting**: Sorted the lists of indices for each coordinate value.
4. **Query Processing**: For each query, checked if the requested index exists in the sorted list and outputted the corresponding distance or handled the case where no valid point was found.
5. **Efficiency**: The initial approach using binary search and prefix sums was overly complex for this problem. Sorting the indices for each coordinate and then simply retrieving the k-th element was more efficient and straightforward.
