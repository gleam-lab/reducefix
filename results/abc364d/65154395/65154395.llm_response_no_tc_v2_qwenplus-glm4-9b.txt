```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX_N = 100100;
const int INF = 1e9;

vector<int> a(MAX_N);

int main() {
    int N, Q;
    cin >> N >> Q;

    // Read the positions of points A
    for (int i = 0; i < N; ++i) {
        int pos;
        cin >> pos;
        a[pos] = 1;  // Mark the position with 1
    }

    // Read the queries
    for (int i = 0; i < Q; ++i) {
        int b, k;
        cin >> b >> k;

        // Calculate the distance to the k-th closest point A
        // We need to find the k-th smallest distance d such that there are k elements of a <= d
        int low = -INF, high = INF;
        while (high - low > 1) {
            int mid = (low + high) / 2;
            int count = 0;
            // Count how many points are within distance 'mid'
            for (int j = max(0, b - mid); j <= min(N - 1, b + mid); ++j) {
                if (a[j] == 1) count++;
            }
            if (count >= k) {
                high = mid;
            } else {
                low = mid;
            }
        }

        // Output the k-th smallest distance
        cout << high << endl;
    }

    return 0;
}
```

This code has been simplified and made more readable. The main changes include:
1. Removed unnecessary binary indexed tree (BIT) usage.
2. Used a simple linear scan to count the number of points within a certain distance to get the k-th closest distance.
3. The distance range is scanned from `low` to `-INF` and `high` to `INF`, adjusting the range based on the count of points within the range.
