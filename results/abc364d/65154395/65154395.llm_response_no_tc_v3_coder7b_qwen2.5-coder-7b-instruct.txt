bool check(int mid, vector<int>& x, int y){
    int cnt = upper_bound(x.begin(), x.end(), y) - lower_bound(x.begin(), x.end(), y - mid);
    return cnt >= mid;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a[x].push_back(i);
    }
    for(int i = 1; i < N; i++){
        sort(a[i].begin(), a[i].end());
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 1e8;
        while(r > l){
            int mid = l + r >> 1;
            if(check(mid, a[x], y)) r = mid;
            else l = mid + 1;
        }
        cout << r << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
