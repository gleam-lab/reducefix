typedef long long ll;

    int N, Q;

    vector<ll> a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        vector<ll> distances;

        }
        sort(distances.begin(), distances.end());

    }
    
    return 0;
}
