#include<bits/stdc++.h>
using namespace std;

vector<int> A, B;

void solve() {
    int n, q; cin >> n >> q;
    A.resize(n + 1); B.resize(q + 1);
    
    for(int i = 1; i <= n; i++) cin >> A[i];
    sort(A.begin(), A.end());
    
    for(int i = 1; i <= q; i++) {
        cin >> B[i] >> A[i];
        int idx = upper_bound(A.begin(), A.end(), B[i]) - A.begin();
        cout << abs(B[i] - A[idx]) << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    solve();
    return 0;
}
