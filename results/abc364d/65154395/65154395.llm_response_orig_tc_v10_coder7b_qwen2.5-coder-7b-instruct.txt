#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

struct Point{
    int id, x;
}a[100005], b[100005];

int n, m;

bool cmp(const Point&a, const Point&b){return a.x < b.x;}

ll solve(int x){
    sort(b, b+m, cmp);
    multiset<int> s;
    for(int i=0;i<m;i++){
        auto it = s.lower_bound(abs(a[x].x-b[i].x));
        if(it != end(s)){
            if(next(it) != end(s)) s.erase(next(it));
            s.insert(*it + abs(a[x].x-b[i].x));
        }else{
            s.insert(abs(a[x].x-b[i].x));
        }
    }
    auto it = begin(s);
    advance(it, n-1);
    return *it;
}

int main(){
    cin >> n >> m;
    for(int i=0;i<n;i++) cin >> a[i].x, a[i].id = i;
    for(int i=0;i<m;i++) cin >> b[i].x, b[i].id = i;
    sort(a, a+n, cmp);
    for(int i=0;i<n;i++) cout << solve(i) << '\n';
    return 0;
}
