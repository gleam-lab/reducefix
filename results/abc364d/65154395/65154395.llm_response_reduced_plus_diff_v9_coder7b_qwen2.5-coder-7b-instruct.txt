```cpp
#include<bits/stdc++.h>
using namespace std;
#define LL long long

vector<int> A, B;
int K[100010];

bool cmp(const pair<int,int>& a,const pair<int,int>& b){return a.first<b.first;}

LL getdis(vector<pair<int,int>>& v,int k){
    sort(v.begin(),v.end(),cmp);
    return v[k].second;
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int n,q;
    cin>>n>>q;
    A.resize(n+1);B.resize(q+1);
    for(int i=1;i<=n;++i)cin>>A[i],A[i]+=1e8;
    for(int i=1;i<=q;++i)cin>>B[i]>>K[i],B[i]+=1e8;
    
    vector<pair<int,int>> vec(n+1);
    for(int i=1;i<=n;++i)vec[i]={A[i]-B[1],A[i]};
    LL ans=0;
    for(int i=1;i<=q;++i){
        sort(vec.begin()+1,vec.begin()+n+1,cmp);
        ans+=getdis(vec,K[i]);
    }
    cout<<ans;
    return 0;
}
```
