```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
vector<int> a[MAXN], b[MAXN];
vector<pair<int, int>> ans;

int main() {
    int n, q;
    cin >> n >> q;
    for(int i = 1; i <= n; ++i) {
        int x; cin >> x;
        a[x].push_back(i);
    }
    for(int i = 1; i <= q; ++i) {
        int x, k; cin >> x >> k;
        b[k].push_back(x);
    }

    for(int i = 1; i <= n; ++i) sort(a[i].begin(), a[i].end());
    for(int i = 1; i <= q; ++i) sort(b[i].begin(), b[i].end());

    for(int i = 1; i <= q; ++i) {
        int idx = lower_bound(b[i].begin(), b[i].end(), 1) - b[i].begin();
        int pos = upper_bound(b[i].begin(), b[i].end(), n) - b[i].begin() - 1;
        int res = 0;
        if(idx < pos) res = b[i][pos] - b[i][idx];
        ans.push_back({i, res});
    }

    sort(ans.begin(), ans.end());
    for(auto& [id, val]: ans) cout << val << '\n';

    return 0;
}
```
