#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 1e5 + 5;
vector<ll> a, b, c;
vector<int> k;

bool cmp(ll x, ll y) {
    return x < y;
}

void solve() {
    int n, q; cin >> n >> q;
    a.resize(n); b.resize(q); c.resize(n);
    k.resize(q);
    
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        c[i] = abs(a[i]);
    }
    
    sort(c.begin(), c.end());
    
    for(int i = 0; i < q; i++) {
        cin >> b[i] >> k[i];
        b[i] = abs(b[i]);
    }
    
    for(int i = 1; i < n; i++) {
        c[i] += c[i - 1];
    }
    
    for(int i = 0; i < q; i++) {
        int pos = upper_bound(c.begin(), c.end(), b[i]) - c.begin();
        if(pos == 0) {
            cout << abs(b[i] - a[0]) << endl;
        } else if(pos == n) {
            cout << abs(b[i] - a[n - 1]) << endl;
        } else {
            int left = pos - 1, right = pos;
            if(k[i] == 1) {
                cout << min(abs(b[i] - a[left]), abs(b[i] - a[right])) << endl;
            } else {
                cout << abs(b[i] - a[right]) << endl;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    solve();
    return 0;
}
