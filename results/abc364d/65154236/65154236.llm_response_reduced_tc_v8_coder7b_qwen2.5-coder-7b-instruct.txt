#include <bits/stdc++.h>
using namespace std;

vector<int> A, B;
vector<long long> distances;

long long findKthClosestDistance(long long b_j, int k_j) {
    // Calculate all distances from B_j to A_i
    for (auto a_i : A) {
        distances.push_back(abs(a_i - b_j));
    }

    // Sort the distances
    sort(distances.begin(), distances.end());

    // Return the k_j-th smallest distance
    return distances[k_j - 1];
}

int main() {
    int N, Q;
    cin >> N >> Q;

    A.resize(N);
    B.resize(Q);

    // Read coordinates of points A_i
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Read coordinates of points B_j and corresponding k_j values
    for (int i = 0; i < Q; ++i) {
        cin >> B[i] >> A[i]; // Note: A[i] is reused here to store k_j temporarily
    }

    // For each B_j, find the k_j-th closest distance
    for (int i = 0; i < Q; ++i) {
        long long result = findKthClosestDistance(B[i], A[i]);
        cout << result << endl;
    }

    return 0;
}

This solution reads the input values, calculates the distances from each B_j to all A_i, sorts these distances, and then finds the k_j-th smallest distance for each B_j. The main function handles the entire process without using any hard-coded values or special cases, ensuring it meets the requirements specified in the problem statement.
