#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;

    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Sort A for binary search later
    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;

        // Create a list of absolute differences from b to each A[i]
        vector<ll> diffs;
        for (int i = 0; i < N; ++i) {
            diffs.push_back(abs(A[i] - b));
        }

        // Find the k-th smallest distance using nth_element
        nth_element(diffs.begin(), diffs.begin() + (k - 1), diffs.end());
        cout << diffs[k - 1] << '\n';
    }

    return 0;
}
