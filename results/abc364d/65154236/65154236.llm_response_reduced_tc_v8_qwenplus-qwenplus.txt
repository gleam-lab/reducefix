#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search helper to find k-th smallest distance efficiently
ll get_kth_distance(int n, ll b, int k) {
    // Sort the A array for binary search usage
    sort(a, a + n);

    // Define search range
    ll low = 0;
    ll high = abs(a[n - 1] - b); // Max possible distance

    while (low < high) {
        ll mid = (low + high) / 2;
        ll count = 0;

        // For each a[i], find how many elements within distance <= mid from b
        for (int i = 0; i < n; ++i) {
            ll left = b - mid;
            ll right = b + mid;
            // Find first a >= left
            ll l = lower_bound(a, a + n, left) - a;
            // Find last a <= right
            ll r = upper_bound(a, a + n, right) - a;
            count += r - l;
            if (count > k) break; // Optimization
        }

        if (count >= k)
            high = mid;
        else
            low = mid + 1;
    }

    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }

    for (int j = 0; j < Q; ++j) {
        ll b;
        int k;
        cin >> b >> k;
        cout << get_kth_distance(N, b, k) << "\n";
    }

    return 0;
}
