#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;

    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;

        // Binary search to find the position in A where b would be inserted
        int left = 0, right = N - 1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (A[mid] <= b)
                left = mid + 1;
            else
                right = mid - 1;
        }

        // Now, we have two pointers: one on the last element <= b (right), and one on first > b (left)
        // We'll use a sliding window approach to find the k-th closest
        int i = right, j = left;
        int count = 0;
        ll answer = 0;

        while (count < k) {
            if (i >= 0 && j < N) {
                ll diff_i = abs(A[i] - b);
                ll diff_j = abs(A[j] - b);
                if (diff_i <= diff_j) {
                    answer = diff_i;
                    --i;
                } else {
                    answer = diff_j;
                    ++j;
                }
            } else if (i >= 0) {
                answer = abs(A[i--] - b);
            } else if (j < N) {
                answer = abs(A[j++] - b);
            }
            ++count;
        }

        cout << answer << '\n';
    }
}

int main() {
    solve();
    return 0;
}
