#include<bits/stdc++.h>
using namespace std;

vector<int> pointsA, pointsB;
vector<pair<int, int>> queries;

void preprocess() {
    sort(pointsA.begin(), pointsA.end());
}

int findKthClosest(int target, int k) {
    auto it = lower_bound(pointsA.begin(), pointsA.end(), target);
    int index = it - pointsA.begin();
    if(index == pointsA.size()) --index;
    if(index < 0 || abs(target - pointsA[index]) > abs(target - pointsA[(index + 1) % pointsA.size()])) ++index;
    if(abs(target - pointsA[index]) > abs(target - pointsA[(index - 1 + pointsA.size()) % pointsA.size()])) --index;
    if(k == 1) return abs(target - pointsA[index]);
    if(k == pointsA.size()) return abs(target - pointsA[(index + 1) % pointsA.size()]);
    vector<int> distances;
    for(int i = max(0, index - k); i <= min((int)pointsA.size() - 1, index + k); ++i) {
        distances.push_back(abs(target - pointsA[i]));
    }
    sort(distances.begin(), distances.end());
    return distances[k - 1];
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, Q;
    cin >> N >> Q;
    pointsA.resize(N);
    pointsB.resize(Q);
    queries.resize(Q);

    for(int i = 0; i < N; ++i) {
        cin >> pointsA[i];
    }

    preprocess();

    for(int i = 0; i < Q; ++i) {
        cin >> pointsB[i] >> queries[i].second;
        queries[i].first = i;
    }

    sort(queries.begin(), queries.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
        return pointsB[a.first] < pointsB[b.first];
    });

    int idx = 0;
    for(auto& query : queries) {
        while(idx < N && pointsA[idx] <= pointsB[query.first]) ++idx;
        cout << findKthClosest(pointsB[query.first], query.second) << '\n';
    }

    return 0;
}
