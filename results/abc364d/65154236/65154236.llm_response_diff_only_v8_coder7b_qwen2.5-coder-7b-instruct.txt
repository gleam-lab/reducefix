int findKthSmallest(vector<int>& distances, int k) {
    sort(distances.begin(), distances.end());
    return distances[k - 1];
}

        cin >> b_j >> k_j;
        queries.emplace_back(b_j, k_j);
    }

    for(auto& [b_j, k_j] : queries) {
        vector<int> distances;
        for(int a_i : A) {
