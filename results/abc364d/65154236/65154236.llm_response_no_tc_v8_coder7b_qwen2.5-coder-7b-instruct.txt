```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 1e5 + 5;
vector<int> a, b;
vector<pair<ll, int>> queries;

bool compare(pair<ll, int>& p1, pair<ll, int>& p2) {
    return p1.first < p2.first || (p1.first == p2.first && p1.second < p2.second);
}

void solve() {
    int n, q; cin >> n >> q;
    a.resize(n + 1); b.resize(q + 1);
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for(int i = 1; i <= q; i++) {
        cin >> b[i] >> queries[i].second;
        queries[i].first = b[i];
    }

    sort(a.begin(), a.end());
    sort(queries.begin(), queries.end(), compare);

    vector<ll> ans(q + 1);
    int idx = 1;
    for(auto& query : queries) {
        while(idx <= n && abs(query.first - a[idx]) < abs(query.first - a[idx - 1])) {
            idx++;
        }
        ans[query.second] = abs(query.first - a[idx - 1]);
    }

    for(int i = 1; i <= q; i++) {
        cout << ans[i] << '\n';
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    int t = 1;
    while(t--) solve();
    return 0;
}
```

This solution reorganizes the approach to efficiently find the k-th nearest point using sorting and binary search principles. It maintains a list of distances and sorts them along with their respective queries. Then, it iterates through these sorted queries and calculates the k-th nearest distance. This ensures efficiency even within large constraints.
