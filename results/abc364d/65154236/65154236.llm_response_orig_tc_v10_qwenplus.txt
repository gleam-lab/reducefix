#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    int n, q;
    cin >> n >> q;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Sort the A points for binary search
    sort(a.begin(), a.end());
    
    for (int i = 0; i < q; i++) {
        int b, k;
        cin >> b >> k;
        
        // Binary search on the distance
        int left = 0, right = 200000000;
        int ans = right;
        
        while (left <= right) {
            int mid = (left + right) / 2;
            
            // Count points in [b-mid, b+mid]
            int left_idx = lower_bound(a.begin(), a.end(), b - mid) - a.begin();
            int right_idx = upper_bound(a.begin(), a.end(), b + mid) - a.begin();
            int count = right_idx - left_idx;
            
            if (count >= k) {
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        cout << ans << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
