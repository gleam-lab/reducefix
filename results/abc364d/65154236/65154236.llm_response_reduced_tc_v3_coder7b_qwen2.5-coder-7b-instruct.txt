```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>

const int MAXN = 1e5 + 10;

vector<pair<ll, int>> A[MAXN];
vi B;

void preprocess() {
    sort(B.begin(), B.end());
    for (int i = 0; i < B.size(); ++i) {
        B[i] = lower_bound(B.begin(), B.end(), B[i]) - B.begin();
    }
}

int query(int idx, int k) {
    int l = 0, r = A[idx].size() - 1;
    while (l < r) {
        int m = (l + r) / 2;
        if (A[idx][m].second >= k) r = m;
        else l = m + 1;
    }
    return A[idx][l].first;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    for (int i = 0; i < N; ++i) {
        int a;
        cin >> a;
        A[B.size()].push_back({abs(a), i});
        B.push_back(abs(a));
    }

    preprocess();

    for (int i = 0; i < Q; ++i) {
        int b, k;
        cin >> b >> k;
        b = abs(b);
        int idx = lower_bound(B.begin(), B.end(), b) - B.begin();
        cout << query(idx, k) << '\n';
    }

    return 0;
}
```
