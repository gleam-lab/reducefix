#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    ll n, q; cin >> n >> q;
    vi a(n);
    for(ll i = 0; i < n; ++i) cin >> a[i];
    
    for(ll j = 0; j < q; ++j) {
        ll b, k; cin >> b >> k;
        auto it = lower_bound(a.begin(), a.end(), b);
        ll dist = abs(b - *it);
        if(it != a.end() && it != a.begin()) {
            dist = min(dist, abs(b - *(it - 1)));
        } else if(it == a.end()){
            dist = abs(b - *(it - 1));
        }
        
        for(ll m = 1; m < k; ++m) {
            if(it == a.end()) {
                it--;
                dist = abs(b - *(it - m));
            } else if(it == a.begin()) {
                it++;
                dist = abs(b - *(it + m));
            } else {
                ll leftDist = abs(b - *(it - m));
                ll rightDist = abs(b - *(it + m));
                dist = min(leftDist, rightDist);
            }
        }
        cout << dist << '\n';
    }
    return 0;
}
