#include <bits/stdc++.h>
using namespace std;

vector<int> dist;

void solve(vector<int>& A, vector<int>& B, int k) {

    sort(dist.begin(), dist.end());
    cout << dist[k - 1] << '\n';
}

    cin >> n >> q;

    vector<int> A(n), B(q);

    }

    for (int i = 0; i < q; ++i) {

        --k; // Adjusting to zero-based index

}
