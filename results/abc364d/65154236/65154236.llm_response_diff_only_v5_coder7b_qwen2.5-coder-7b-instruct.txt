#include<bits/stdc++.h>
using namespace std;

const int MAXN = 100010;
const int INF = 0x3f3f3f3f;

int a[MAXN], b[MAXN], k[MAXN];
vector<int> dists[MAXN];

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int n, q; cin >> n >> q;
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    for(int j = 0; j < q; ++j) {
        cin >> b[j] >> k[j];
        dists[b[j]].push_back(abs(b[j] - a[i]));
    }

    for(int i = 0; i < MAXN; ++i) {
        sort(dists[i].begin(), dists[i].end());
    }

    for(int j = 0; j < q; ++j) {
        cout << dists[b[j]][k[j] - 1] << '\n';
    }

    return 0;
}
