
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n >> q;
    
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    for (int i = 0; i < q; ++i) {
        long long b, k;
        cin >> b >> k;
        
        // Find k-th closest point to b from a
        sort(a.begin(), a.end());
        
        long long d = a[k - 1] - b;
        
        // Output the distance
        cout << d << endl;
    }
    
    return 0;
}
```
