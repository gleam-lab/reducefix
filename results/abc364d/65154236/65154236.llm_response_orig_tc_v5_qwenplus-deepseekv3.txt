```cpp
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;
    
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        k--; // convert to 0-based index
        
        // Binary search the answer
        ll low = 0, high = 2e14;
        ll answer = 0;
        
        while (low <= high) {
            ll mid = (low + high) / 2;
            
            // Count how many A[i] have |A[i] - b| <= mid
            auto left = lower_bound(A.begin(), A.end(), b - mid);
            auto right = upper_bound(A.begin(), A.end(), b + mid);
            int count = distance(left, right);
            
            if (count > k) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        
        cout << answer << '\n';
    }

    return 0;
}
```

1. Sorted the A array initially (O(N log N))
2. For each query (O(Q)):

   - For each binary search step, use lower_bound and upper_bound to count points within the current distance (O(log N))
3. Overall complexity: O(N log N + Q log N log(max distance)), which is efficient enough for the constraints
