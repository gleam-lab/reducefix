#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int OFFSET = 100000000; // To handle negative coordinates
const int MAX_COORD = 200000000 + 10; // Max possible coordinate after offset

// Binary indexed tree (Fenwick Tree) implementation
class BIT {
public:
    vector<int> tree;
    BIT(int size) : tree(size + 2, 0) {} // Initialize with extra space to avoid overflow
    
    void update(int idx, int delta) {
        while (idx < (int)tree.size()) {
            tree[idx] += delta;
            idx += idx & -idx;
        }
    }
    
    int query(int idx) {
        int res = 0;
        while (idx > 0) {
            res += tree[idx];
            idx -= idx & -idx;
        }
        return res;
    }
    
    // Find smallest x such that prefix sum >= k
    int find_kth(int k) {
        int idx = 0, sum = 0;
        for (int i = 26; i >= 0; i--) { // log2(MAX_COORD)
            int new_idx = idx + (1 << i);
            if (new_idx < (int)tree.size() && sum + tree[new_idx] < k) {
                idx = new_idx;
                sum += tree[new_idx];
            }
        }
        return idx + 1;
    }
};

void solve(void) {
    int N, Q;
    cin >> N >> Q;
    
    vector<int> counts(MAX_COORD + 2, 0); // Coordinate compression isn't needed due to small range
    
    for (int i = 0; i < N; ++i) {
        int x;
        cin >> x;
        counts[x + OFFSET]++;
    }
    
    BIT bit(MAX_COORD + 5);
    // Initialize BIT
    for (int i = 0; i <= MAX_COORD; ++i) {
        if (counts[i] > 0) {
            bit.update(i + 1, counts[i]); // 1-indexed BIT
        }
    }
    
    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;
        
        // We use binary search on the distance
        int left = 0, right = MAX_COORD;
        while (left < right) {
            int mid = (left + right) / 2;
            
            // Count how many points have distance <= mid from b
            int l = max(0, b + OFFSET - mid);
            int r = min(MAX_COORD, b + OFFSET + mid);
            
            int cnt = bit.query(r + 1) - bit.query(l);
            
            if (cnt >= k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        
        cout << left << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
