```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    vector<int> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    vector<pair<int, int>> queries(q);
    for(int i = 0; i < q; ++i) {
        int b, k; cin >> b >> k;
        queries[i] = {b, k};
    }

    sort(a.begin(), a.end());

    for(auto [b, k]: queries) {
        auto it = lower_bound(a.begin(), a.end(), b);
        int idx = it - a.begin();
        int ans = abs(b - a[idx]);
        if(idx > 0) ans = min(ans, abs(b - a[idx - 1]));
        cout << ans << '\n';
    }

    return 0;
}
```
