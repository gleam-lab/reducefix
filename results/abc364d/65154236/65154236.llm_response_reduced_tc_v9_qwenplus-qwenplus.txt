#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    int N, Q;
    cin >> N >> Q;

    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Sort the A array to enable binary search around B_j
    sort(A.begin(), A.end());

    for (int q = 0; q < Q; ++q) {
        ll b;
        int k;
        cin >> b >> k;

        // Binary search to find the k-th closest point
        // We use a sliding window approach on sorted A

        int l = 0, r = N - 1;
        while (r - l + 1 >= k) {
            int left_dist = abs(b - A[l]);
            int right_dist = abs(b - A[r]);

            if (left_dist < right_dist) {
                // Keep left, move right
                r--;
            } else {
                // Keep right, move left
                l++;
            }
        }

        // At this point, l and r define a window of size less than k
        // The next one is our answer
        cout << abs(b - A[l]) << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    solve();
    return 0;
}
