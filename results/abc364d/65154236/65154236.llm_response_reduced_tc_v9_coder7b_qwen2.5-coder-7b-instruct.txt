```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    vector<ll> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    vector<pair<ll, int>> b(q);
    for(int i = 0; i < q; ++i) {
        cin >> b[i].first >> b[i].second;
        --b[i].second; // Adjusting for 0-based index
    }

    sort(a.begin(), a.end());
    
    for(auto& [x, k] : b) {
        auto it = lower_bound(a.begin(), a.end(), x);
        if(it != a.begin()) --it;
        ll ans = abs(*it - x);
        if(ans > abs(*(next(it)) - x)) ++it;
        if(k == 0) {
            cout << ans << '\n';
            continue;
        }
        --k;
        for(; it != a.end(); ++it) {
            if(k-- == 0) {
                ans = abs(*it - x);
                break;
            }
        }
        cout << ans << '\n';
    }

    return 0;
}
```
