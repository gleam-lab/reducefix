#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAX_N = 1e5 + 5;

vector<ll> a;
ll b, k;

void solve() {
    int N, Q;
    cin >> N >> Q;
    
    a.resize(N);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    
    // Sort the A array for binary search and lower/upper bound operations
    sort(a.begin(), a.end());
    
    for (int q = 0; q < Q; q++) {
        cin >> b >> k;
        
        // Binary search on the answer (distance)
        ll left = 0, right = 2e8;
        ll ans = right;
        
        while (left <= right) {
            ll mid = (left + right) / 2;
            
            // Count how many points in A are within distance 'mid' from point b
            ll lower_bound_val = b - mid;
            ll upper_bound_val = b + mid;
            
            auto it1 = lower_bound(a.begin(), a.end(), lower_bound_val);
            auto it2 = upper_bound(a.begin(), a.end(), upper_bound_val);
            
            int count = distance(it1, it2); // number of points within [b-mid, b+mid]
            
            if (count >= k) {
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        cout << ans << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    
    return 0;
}
