#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int MAXN = 100005;
int a[MAXN], cnt[MAXN * 2];

bool check(int mid, int x, int y){
    return cnt[min(MAXN * 2 - 1, x + mid + MAXN)] - cnt[max(0, x - mid - 1 + MAXN)] >= y;
}

void update(int idx, int val){
    while(idx < MAXN * 2){
        cnt[idx] += val;
        idx += lowbit(idx);
    }
}

int query(int idx){
    int res = 0;
    while(idx > 0){
        res += cnt[idx];
        idx -= lowbit(idx);
    }
    return res;
}

void solve(void) {
    int n, q; cin >> n >> q;
    memset(cnt, 0, sizeof(cnt));
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        update(x + MAXN, 1);
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 2 * MAXN;
        while(r > l){
            int mid = l + r >> 1;
            if(check(mid, x + MAXN, y)) r = mid;
            else l = mid + 1;
        }
        cout << l - MAXN << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
