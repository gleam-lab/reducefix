#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAX_OFFSET = 200000000; // Offset to handle negative coordinates

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;

    // Frequency array using offset to avoid negative indices
    vector<int> freq(400000001, 0); // Range from -1e8 to 1e8 (inclusive)

    for (int i = 0; i < N; ++i) {
        int a;
        cin >> a;
        freq[a + MAX_OFFSET]++;
    }

    // Prefix sum to allow fast range count queries
    vector<ll> prefix(400000002, 0LL);
    for (int i = 0; i <= 400000000; ++i)
        prefix[i + 1] = prefix[i] + freq[i];

    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;

        // Binary search on distance d
        int left = 0, right = 200000000;
        int answer = right;

        while (left <= right) {
            int mid = (left + right) / 2;
            int low = max(-200000000, b - mid);
            int high = min(200000000, b + mid);

            // Map to valid index range [0, 400000000]
            int l_idx = low + MAX_OFFSET;
            int r_idx = high + MAX_OFFSET;

            ll count = prefix[r_idx + 1] - prefix[l_idx];

            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        cout << answer << '\n';
    }
}

int main() {
    solve();
    return 0;
}
