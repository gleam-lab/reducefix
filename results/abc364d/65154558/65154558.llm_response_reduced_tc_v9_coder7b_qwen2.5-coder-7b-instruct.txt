#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

vector<ll> a, b, ans;
vector<int> k;

void solve() {
    int n, q; cin >> n >> q;
    a.resize(n); b.resize(q); k.resize(q);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < q; ++i) cin >> b[i] >> k[i];
    
    sort(a.begin(), a.end());
    
    for(int i = 0; i < q; ++i) {
        int idx = upper_bound(a.begin(), a.end(), b[i]) - a.begin();
        if(k[i] == 1) ans.push_back(b[i] - a[idx - 1]);
        else if(k[i] == n) ans.push_back(a[idx] - b[i]);
        else ans.push_back(min(a[idx] - b[i], b[i] - a[idx - 1]));
    }
    
    for(auto& x : ans) cout << x << '\n';
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    solve();
    return 0;
}
