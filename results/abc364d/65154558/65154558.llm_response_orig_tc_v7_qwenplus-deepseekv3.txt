```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());

    while (Q--) {
        ll b, k;
        cin >> b >> k;
        int low = 0, high = 2e8;
        ll ans = 0;
        while (low <= high) {
            ll mid = low + (high - low) / 2;
            auto left = lower_bound(A.begin(), A.end(), b - mid);
            auto right = upper_bound(A.begin(), A.end(), b + mid);
            int count = right - left;
            if (count >= k) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        cout << ans << '\n';
    }

    return 0;
}
```

### Explanation:
1. **Sorting**: The array `A` is sorted to facilitate binary search operations.
2. **Binary Search for Each Query**: For each query point `b` and integer `k`, we perform a binary search on the possible distances from `b` to find the smallest distance `mid` such that at least `k` points in `A` are within `mid` distance from `b`.
3. **Counting Points Within Range**: Using `lower_bound` and `upper_bound`, we efficiently count the number of points in `A` that lie within the range `[b - mid, b + mid]`.
4. **Adjusting Search Range**: Based on whether the count of points within the range meets or exceeds `k`, we adjust the binary search range to find the minimal such `mid`.

This approach ensures that we efficiently handle each query in logarithmic time relative to the range of possible distances, making it suitable for large input sizes as specified in the problem constraints.
