### Final Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());

    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        k--; // 0-based index

        ll low = 0, high = 2e14;
        ll answer = 0;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll left = b - mid;
            ll right = b + mid;
            auto it_left = lower_bound(A.begin(), A.end(), left);
            auto it_right = upper_bound(A.begin(), A.end(), right);
            int count = it_right - it_left;
            if (count <= k) {
                low = mid + 1;
            } else {
                answer = mid;
                high = mid - 1;
            }
        }

        // Now, collect all elements within distance 'answer' from 'b'
        vector<ll> distances;
        ll left_bound = b - answer;
        ll right_bound = b + answer;
        auto it_left = lower_bound(A.begin(), A.end(), left_bound);
        auto it_right = upper_bound(A.begin(), A.end(), right_bound);
        for (auto it = it_left; it != it_right; ++it) {
            distances.push_back(abs(*it - b));
        }
        sort(distances.begin(), distances.end());
        cout << distances[k] << '\n';
    }

    return 0;
}
```

This code:
1. Sorts the array A.
2. For each query, performs a binary search to find the minimal distance d such that there are at least k+1 elements within distance d of B.
3. Collects these elements, sorts their distances, and outputs the k-th smallest distance.

This approach ensures that we efficiently find the k-th closest element for each query, leveraging binary search and the sorted property of the array A. The complexity is O(N log N) for sorting and O(Q * log(max_distance) * log N) for the queries, which is efficient enough for the given constraints.
