#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    int N, Q;
    cin >> N >> Q;
    vector<ll> a(N);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());

    while (Q--) {
        ll b;
        int k;
        cin >> b >> k;
        k--; // convert to 0-based index

        int low = 0, high = N - 1;
        while (low < high) {
            int mid = (low + high) / 2;
            if (a[mid] < b) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        int pos = low;

        int left = pos - 1;
        int right = pos;
        vector<ll> distances;
        for (int i = 0; i < k + 1; ++i) {
            if (left >= 0 && right < N) {
                if (b - a[left] <= a[right] - b) {
                    distances.push_back(b - a[left]);
                    left--;
                } else {
                    distances.push_back(a[right] - b);
                    right++;
                }
            } else if (left >= 0) {
                distances.push_back(b - a[left]);
                left--;
            } else {
                distances.push_back(a[right] - b);
                right++;
            }
        }
        nth_element(distances.begin(), distances.begin() + k, distances.end());
        cout << distances[k] << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}

### Explanation
1. **Sorting the Array**: The array `a` is sorted to allow efficient binary search operations.
2. **Handling Queries**: For each query point `b_j`, we first find its position in the sorted array using binary search. This helps us locate the closest points around `b_j`.
3. **Finding k-th Closest Distance**: We then expand the search around the found position to include the nearest points. The distances of these points from `b_j` are collected and the k-th smallest distance is found using `nth_element`, which efficiently places the k-th smallest element in its correct position.
4. **Output**: The k-th smallest distance is printed for each query.

This approach ensures that we handle each query in logarithmic time relative to the number of points, making it efficient for large inputs as specified in the problem constraints.
