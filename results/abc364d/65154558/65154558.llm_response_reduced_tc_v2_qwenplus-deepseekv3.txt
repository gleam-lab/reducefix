#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {

    cin.tie(nullptr);

    int N, Q;

    vector<ll> A(N);

        cin >> b >> k;
        // Binary search to find the k-th smallest distance

        ll answer = 0;
        while (low <= high) {
            ll mid = low + (high - low) / 2;

    }

    return 0;
}
