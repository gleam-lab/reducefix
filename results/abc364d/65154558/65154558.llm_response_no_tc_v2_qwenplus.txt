```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAX_N = 1e5 + 5;

vector<int> a;
int n, q;

void solve() {
    cin >> n >> q;
    a.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Sort the A points to enable binary search
    sort(a.begin(), a.end());
    
    for (int j = 0; j < q; j++) {
        int b, k;
        cin >> b >> k;
        
        // We want to find the k-th smallest distance from b to any a[i]
        // Instead of computing all distances, we can use binary search on the answer
        // But simpler: find the position where b would be inserted, then check neighbors
        
        // Find the first index where a[i] >= b
        auto it = lower_bound(a.begin(), a.end(), b);
        int pos = it - a.begin();
        
        // We'll consider candidates around this position
        // The k-th closest must be within k elements to the left or right
        vector<int> candidates;
        
        // Add up to k elements before pos
        for (int i = max(0, pos - k); i < pos; i++) {
            candidates.push_back(abs(a[i] - b));
        }
        // Add up to k elements from pos onwards
        for (int i = pos; i < min(n, pos + k); i++) {
            candidates.push_back(abs(a[i] - b));
        }
        
        // Sort distances and pick the k-th smallest (1-indexed)
        sort(candidates.begin(), candidates.end());
        cout << candidates[k - 1] << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
