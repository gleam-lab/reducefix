#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 1e5 + 5;

int n, q;
ll a[MAXN];

// Binary search on answer: find smallest distance d such that there are at least k points within distance d from b
int solve_query(ll b, int k) {
    int left = 0, right = 2e8;
    int ans = right;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        ll lower = b - mid;
        ll upper = b + mid;
        
        // Count number of A points in [lower, upper]
        auto it1 = lower_bound(a, a + n, lower);
        auto it2 = upper_bound(a, a + n, upper);
        int count = it2 - it1;
        
        if (count >= k) {
            ans = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> q;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a, a + n);
    
    for (int i = 0; i < q; i++) {
        ll b;
        int k;
        cin >> b >> k;
        cout << solve_query(b, k) << '\n';
    }
    
    return 0;
}
