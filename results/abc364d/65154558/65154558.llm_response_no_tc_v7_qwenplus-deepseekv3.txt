#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    while (Q--) {
        ll b, k;
        cin >> b >> k;
        
        // Binary search the k-th smallest distance
        ll left = 0, right = 2e14; // Sufficiently large upper bound
        ll answer = 0;
        while (left <= right) {
            ll mid = (left + right) / 2;
            
            // Count the number of elements <= mid distance from b
            auto lower = lower_bound(A.begin(), A.end(), b - mid);
            auto upper = upper_bound(A.begin(), A.end(), b + mid);
            ll count = upper - lower;
            
            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        cout << answer << endl;
    }
}

int main() {

    cin.tie(nullptr);
    solve();
    return 0;
}

### Explanation
1. **Sorting**: The array `A` is sorted to allow efficient range queries using binary search.
2. **Binary Search for Distance**: For each query, we perform a binary search on the possible distances from `B_j` to find the smallest distance `d` such that at least `k_j` points in `A` are within `d` units of `B_j`.
3. **Range Query**: For each midpoint `mid` in the binary search, we determine the number of points in `A` that lie within the range `[B_j - mid, B_j + mid]` using `lower_bound` and `upper_bound`.
4. **Adjust Search Range**: If there are at least `k_j` points within the range, we narrow the search to smaller distances; otherwise, we search larger distances.

This approach ensures that each query is handled in `O(log(max_distance) * log N)` time, making it efficient for large inputs.
