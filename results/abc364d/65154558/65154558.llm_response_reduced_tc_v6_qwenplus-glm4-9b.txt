#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long solve(const vector<pair<long long, int>>& queries, const vector<long long>& points) {
    vector<pair<long long, long long>> sorted_points;
    for (auto& p : points) {
        sorted_points.emplace_back(p, 0);
    }
    sort(sorted_points.begin(), sorted_points.end());

    vector<long long> prefix_sum;
    long long current_sum = 0;
    for (auto& p : sorted_points) {
        current_sum += p.second;
        p.second = current_sum;
    }
    prefix_sum = p.second;

    long long ans = 0;
    for (auto& q : queries) {
        long long b = q.first;
        int kj = q.second - 1; // kj is 0-based, so subtract 1
        if (kj < 0 || kj >= sorted_points.size()) {
            ans = -1; // Should not happen, but let's handle it
            break;
        }
        long long a = sorted_points[kj].first;
        long long d = abs(a - b);
        ans += d * prefix_sum[kj];
    }
    return ans;
}

int main() {
    int n, q;
    cin >> n >> q;

    vector<long long> points(n);
    for (int i = 0; i < n; ++i) {
        cin >> points[i];
    }

    vector<pair<long long, int>> queries(q);
    for (int i = 0; i < q; ++i) {
        cin >> queries[i].first >> queries[i].second;
    }

    cout << solve(queries, points) << endl;

    return 0;
}
