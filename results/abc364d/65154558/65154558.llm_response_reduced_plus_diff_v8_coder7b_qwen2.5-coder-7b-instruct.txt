#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100005;
vector<int> a, b;

struct Point {
    int x, idx;
};

bool cmp(const Point& p1, const Point& p2) {
    return p1.x < p2.x;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    int n, q; cin >> n >> q;
    a.resize(n), b.resize(q);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < q; ++i) cin >> b[i] >> a[i], a[n+i].idx = i;

    sort(a.begin(), a.end());
    vector<int> sorted_a = a;
    
    for(int i = 0; i < q; ++i) {
        int target = b[i], k = a[n+i].idx;
        int left = lower_bound(sorted_a.begin(), sorted_a.end(), target) - sorted_a.begin();
        int right = upper_bound(sorted_a.begin(), sorted_a.end(), target) - sorted_a.begin();

        if(k == left) cout << sorted_a[left] - target << '\n';
        else if(k == right) cout << target - sorted_a[right - 1] << '\n';
        else cout << min(target - sorted_a[k - 1], sorted_a[k] - target) << '\n';
    }

    return 0;
}
