```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    for(int i = 0; i < N; i++){
        cin >> a[i];
    }
    
    sort(a, a + N);
    
    for(int j = 0; j < Q; j++){
        ll b;
        int k;
        cin >> b >> k;
        
        // Create array of distances from B_j to each A_i
        vector<ll> dist(N);
        for(int i = 0; i < N; i++){
            dist[i] = abs(a[i] - b);
        }
        
        // Partial sort to get the k-th smallest distance
        nth_element(dist.begin(), dist.begin() + k - 1, dist.end());
        
        cout << dist[k - 1] << '\n';
    }
    
    return 0;
}
```
