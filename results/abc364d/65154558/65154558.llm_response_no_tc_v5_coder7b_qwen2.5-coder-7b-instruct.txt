```cpp
#include<bits/stdc++.h>
#define endl "\n"
using namespace std;
typedef long long ll;

const int INF = 0x3f3f3f3f;
const int N = 1e5 + 5;
vector<int> a[N];

bool check(int mid, vector<int>& x, int y){
    return x[min(N - 1, mid)] - x[max(0, mid - 1)] >= y;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a[x].push_back(i);
    }
    for(int i = 1; i <= 2 * 1e8; i++){
        sort(a[i].begin(), a[i].end());
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 2e8;
        while(r > l){
            int mid = l + r >> 1;
            if(!check(mid, a[x], y)) r = mid;
            else l = mid + 1;
        }
        cout << l << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}
```
