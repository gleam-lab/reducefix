```cpp
#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<int>
#define vl vector<ll>

struct point{
    ll dist;
    int idx;
};

bool cmp(point p1, point p2){
    if(p1.dist == p2.dist)return p1.idx < p2.idx;
    return p1.dist < p2.dist;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    vl A(n);
    for(int i = 0; i < n; i++)cin >> A[i];

    vl B(q);
    vi K(q);
    for(int i = 0; i < q; i++){
        cin >> B[i] >> K[i];
    }

    for(int i = 0; i < q; i++){
        vector<point> v;
        for(int j = 0; j < n; j++){
            v.push_back({abs(A[j]-B[i]), j});
        }
        sort(v.begin(), v.end(), cmp);
        cout << v[K[i]-1].dist << '\n';
    }

    return 0;
}
```
