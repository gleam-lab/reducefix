#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

const int INF = 1e9 + 7;
typedef long long ll;

int n, q, a[100000 + 10], b[100000 + 10], k[100000 + 10];
ll distance(int left, int right, int x) {
    return (ll)(a[left] - b[x]) * (a[left] - b[x]) + (ll)(a[right] - b[x]) * (a[right] - b[x]);
}

void solve() {
    cin >> n >> q;
    for (int i = 0; i < n; ++i) {
        cin >> a[i + 1];
    }
    for (int i = 0; i < q; ++i) {
        cin >> b[i + 1] >> k[i + 1];
    }
    sort(a + 1, a + n + 1);
    for (int i = 0; i < q; ++i) {
        int l = 1, r = n;
        while (l < r) {
            int mid = (l + r) / 2;
            if (distance(a[mid - 1], a[mid], b[i + 1]) <= distance(a[mid], a[mid + 1], b[i + 1])) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        l = l - k[i] + 1;
        ll ans = distance(a[l - 1], a[l], b[i + 1]);
        cout << ans << endl;
    }
}

int main() {
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
