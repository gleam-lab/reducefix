```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAX_OFFSET = 200000000; // to handle -1e8 to 1e8 range

int freq[400000001]; // frequency array indexed by (value + MAX_OFFSET)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;

    for (int i = 0; i < N; ++i) {
        int a;
        cin >> a;
        freq[a + MAX_OFFSET]++;
    }

    // Build prefix sum of frequencies
    int* prefix = new int[400000001];
    prefix[0] = freq[0];
    for (ll i = 1; i <= 400000000; ++i) {
        prefix[i] = prefix[i - 1] + freq[i];
    }

    for (int q = 0; q < Q; ++q) {
        int b, k;
        cin >> b >> k;

        // Binary search for the smallest distance d such that there are at least k points within distance d from b
        int left = 0;
        int right = MAX_OFFSET * 2;
        int answer = right;

        while (left <= right) {
            int mid = (left + right) / 2;
            int low = b - mid;
            int high = b + mid;

            // Clamp to valid index range
            low = max(-MAX_OFFSET, min(low, (int)MAX_OFFSET));
            high = max(-MAX_OFFSET, min(high, (int)MAX_OFFSET));

            int count = prefix[high + MAX_OFFSET] - (low > -MAX_OFFSET ? prefix[low - 1 + MAX_OFFSET] : 0);

            if (count >= k) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        cout << answer << '\n';
    }

    delete[] prefix;
    return 0;
}
```
