#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

const int INF = 0x3f3f3f3f;
const int MAXN = 1e5 + 5;

vector<ll> a(MAXN * 2);

bool check(int mid, int x, int y){
    return a[min(MAXN - 1, x + mid)] - a[max(0, x - mid - 1)] >= y;
}

void solve(void) {
    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++){
        int x; cin >> x;
        a[MAXN + x]++;
    }
    for(int i = 1; i < 2 * MAXN; i++){
        a[i] += a[i - 1];
    }
    for(int i = 1; i <= q; i++){
        int x, y; cin >> x >> y;
        int l = 0, r = 2e8;
        while(l < r){
            int mid = (l + r) / 2;
            if(check(mid, MAXN + x, y)) r = mid;
            else l = mid + 1;
        }
        cout << l << endl;
    }
}

int main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t = 1; //cin >> t;
    while (t--) solve();
    return 0;
}

2. **Binary Search Range**: The binary search range was set incorrectly. It should start from `0` and go up to `2e8`, inclusive.
3. **Array Size**: The array size was set too large (`N = 2e8`) which is unnecessary and inefficient. It should be adjusted based on the problem constraints (`MAXN = 1e5`).
4. **Output Value**: The output value should be `l` instead of `r` since we need the smallest `mid` that satisfies the condition.
