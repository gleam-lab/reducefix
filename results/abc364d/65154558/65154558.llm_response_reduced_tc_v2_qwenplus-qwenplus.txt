#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 5;

ll a[MAXN];

// Binary search to find the k-th smallest distance from b
ll query(ll b, ll k, int n) {
    // We will find left and right bounds for binary search
    ll left = 0;
    ll right = 2e8; // Maximum possible distance in input range
    
    // We need to build prefix sum of sorted a array
    // But we do it implicitly using lower_bound and upper_bound
    
    // The idea is to count how many distances |a_i - b| <= mid
    // This is equivalent to counting the number of a_i in [b - mid, b + mid]
    
    while (left < right) {
        ll mid = (left + right) / 2;
        // Find how many elements are in [b-mid, b+mid]
        ll l = b - mid;
        ll r = b + mid;
        
        int low = lower_bound(a, a + n, l) - a;
        int high = upper_bound(a, a + n, r) - a;
        int count = high - low;
        
        if (count >= k) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    
    return left;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    
    sort(a, a + N);
    
    for (int i = 0; i < Q; ++i) {
        ll b, k;
        cin >> b >> k;
        cout << query(b, k, N) << "\n";
    }
    
    return 0;
}
