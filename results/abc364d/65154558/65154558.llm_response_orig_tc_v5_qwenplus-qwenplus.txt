#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int OFFSET = 100000000;  // To handle negative coordinates
const int MAX_COORD = 200000000 + 5;

int freq[MAX_COORD * 2 + 10];  // Frequency array to count points at each offset coordinate

// Binary search to find the smallest radius such that there are at least k points within or at that radius from x
bool hasKOrMore(int mid, int x, int k) {
    int left = max(0, x - mid);
    int right = min(MAX_COORD * 2, x + mid);
    int total = freq[right] - freq[left];
    return total >= k;
}

void solve() {
    int N, Q;
    cin >> N >> Q;

    // Initialize frequency array
    fill(freq, freq + MAX_COORD * 2 + 5, 0);

    // Read N A points and update frequency array
    for (int i = 0; i < N; ++i) {
        int a;
        cin >> a;
        freq[a + OFFSET]++;
    }

    // Build prefix sum array
    for (int i = 1; i <= MAX_COORD * 2; ++i) {
        freq[i] += freq[i - 1];
    }

    // Process Q queries
    while (Q--) {
        int b, k;
        cin >> b >> k;

        int low = 0, high = MAX_COORD * 2;
        int result = high;

        // Binary search for the k-th smallest distance
        while (low <= high) {
            int mid = (low + high) / 2;
            int pos = b + OFFSET;

            // Check if between [pos - mid, pos + mid] we have at least k points
            int l = max(0, pos - mid);
            int r = min(MAX_COORD * 2, pos + mid);
            if (freq[r] - freq[l - 1] >= k) {
                result = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        cout << result << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
