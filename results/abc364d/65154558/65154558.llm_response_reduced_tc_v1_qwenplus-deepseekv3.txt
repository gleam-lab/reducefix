#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());

    while (Q--) {
        ll b_j;
        int k_j;
        cin >> b_j >> k_j;

        ll low = 0, high = 2e18;
        ll answer = 0;

        while (low <= high) {
            ll mid = low + (high - low) / 2;
            int count = upper_bound(A.begin(), A.end(), b_j + mid) - lower_bound(A.begin(), A.end(), b_j - mid);
            if (count >= k_j) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        cout << answer << "\n";
    }

    return 0;
}
