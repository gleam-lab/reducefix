import bisect

n, q = map(int, input().split())
a = list(map(int, input().split()))
a.sort()

for _ in range(q):
    b, k = map(int, input().split())
    
    # Find the position where b would be inserted in sorted array a
    pos = bisect.bisect_left(a, b)
    
    # We'll collect distances from points around the insertion point
    # Use two pointers expanding outward from pos-1 and pos
    left = pos - 1
    right = pos
    
    distances = []
    
    # Instead of collecting all distances (which would be O(N)), 
    # we need to find the k-th smallest distance.
    # But since k can be up to N, and N is 10^5, we can collect all distances.
    # Given constraints, O(N) per query might be too slow for Q=10^5 -> O(N*Q)=10^10 worst case.
    # However, let's reconsider: we must compute distances from B_j to all A_i? 
    # Actually, no efficient way around it unless we precompute something.
    # But note: we only need the k-th smallest distance, not all.
    # However, without more structure, we may need to consider all points.
    
    # Since N and Q can be up to 10^5, doing a full scan for each query (O(N*Q)) could be 10^10 operations.
    # That might be borderline in Pyton, but let's try optimizing by not using extra space unnecessarily.
    
    # Actually, we can generate distances and use quickselect or sorting.
    # Sorting all distances per query: O(N log N) per query -> O(Q * N log N) ~ 10^5 * 10^5 * log(10^5) = 10^10 * 17 → too slow.
    
    # Alternative approach: we can binary search on the answer (distance).
    # For a given distance d, count how many points A_i are within distance <= d from B_j.
    # Then find the smallest d such that count >= k_j.
    # This works because the k-th smallest distance is the smallest d where at least k points are within d.
    
    # Let's implement binary search on answer.
    
    # The possible distance ranges from 0 to max(|a_i - b|). Since values are up to 10^8, max distance is up to 2*10^8.
    lo = 0
    hi = 2 * 10**8 + 1
    
    while lo < hi:
        mid = (lo + hi) // 2
        # Count number of A_i such that |a_i - b| <= mid
        # This means: b - mid <= a_i <= b + mid
        left_bound = b - mid
        right_bound = b + mid
        
        # Find leftmost index >= left_bound
        l_idx = bisect.bisect_left(a, left_bound)
        # Find rightmost index <= right_bound
        r_idx = bisect.bisect_right(a, right_bound)
        
        count = r_idx - l_idx
        
        if count >= k:
            hi = mid
        else:
            lo = mid + 1
    
    print(lo)
