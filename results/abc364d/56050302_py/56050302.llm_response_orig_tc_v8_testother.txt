```python
import bisect

n, q = map(int, input().split())
a = list(map(int, input().split()))
a.sort()

for _ in range(q):
    b, k = map(int, input().split())
    
    # Find the position where b would be inserted
    pos = bisect.bisect_left(a, b)
    
    # We need to consider points around the insertion position
    # Collect distances from nearby points
    candidates = []
    
    # Add points to the left of pos
    left = pos - 1
    while left >= 0:
        candidates.append(abs(a[left] - b))
        left -= 1
    
    # Add points to the right of pos (including pos itself)
    right = pos
    while right < n:
        candidates.append(abs(a[right] - b))
        right += 1
    
    # Sort the distances
    candidates.sort()
    
    # The k-th smallest distance (k is 1-indexed)
    print(candidates[k-1])
```
