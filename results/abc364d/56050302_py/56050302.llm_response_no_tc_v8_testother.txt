import bisect

n, q = map(int, input().split())
a = list(map(int, input().split()))
a.sort()

for _ in range(q):
    b, k = map(int, input().split())
    
    # Find the position where b would be inserted in sorted array a
    pos = bisect.bisect_left(a, b)
    
    # We'll collect distances from points around the insertion point
    # But instead of just checking neighbors, we need to consider more points
    
    # Use two pointers expanding from the insertion point
    left = pos - 1
    right = pos
    distances = []
    
    # Instead of collecting all distances (which is O(N)), we can use a smarter approach
    # But since we need the k-th smallest distance, we can expand outward until we have k+1 candidates
    
    # Actually, let's do this properly: collect candidate distances by expanding from the closest point
    candidates = []
    
    # Add indices from both sides in order of increasing distance
    # We'll expand outward from the insertion point
    left_idx = pos - 1
    right_idx = pos
    
    while len(candidates) < k and (left_idx >= 0 or right_idx < n):
        if left_idx >= 0 and right_idx < n:
            left_dist = abs(a[left_idx] - b)
            right_dist = abs(a[right_idx] - b)
            if left_dist <= right_dist:
                candidates.append(left_dist)
                left_idx -= 1
            else:
                candidates.append(right_dist)
                right_idx += 1
        elif left_idx >= 0:
            candidates.append(abs(a[left_idx] - b))
            left_idx -= 1
        else:
            candidates.append(abs(a[right_idx] - b))
            right_idx += 1
    
    # If we still don't have enough candidates, continue adding
    while len(candidates) < k and left_idx >= 0:
        candidates.append(abs(a[left_idx] - b))
        left_idx -= 1
        
    while len(candidates) < k and right_idx < n:
        candidates.append(abs(a[right_idx] - b))
        right_idx += 1
    
    # Sort the candidates and get the k-th smallest
    candidates.sort()
    print(candidates[k-1])
