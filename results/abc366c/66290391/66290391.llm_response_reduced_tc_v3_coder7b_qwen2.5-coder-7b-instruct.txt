```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int q;
    cin >> q;
    set<int> st;
    rep(i, q)
    {
        int type, x;
        cin >> type;
        if (type == 1)
        {
            cin >> x;
            st.insert(x);
        }
        else if (type == 2)
        {
            cin >> x;
            st.erase(st.find(x));
        }
        else if (type == 3)
        {
            cout << st.size() << '\n';
        }
    }

    return 0;
}
```

### Explanation of Fixes
1. **Efficiency Improvement**: The original code used `st.erase(x)` directly, which could lead to undefined behavior since `x` might not exist in the set. Instead, we use `st.erase(st.find(x))` to ensure safety.
2. **Optimization**: Added `ios_base::sync_with_stdio(false); cin.tie(nullptr);` to speed up I/O operations, which can significantly improve performance for large inputs.
3. **Code Readability**: Ensured consistent formatting and readability by adding spaces around operators and after commas where necessary.
