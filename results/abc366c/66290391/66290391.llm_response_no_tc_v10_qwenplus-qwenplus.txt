```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int Q;
    cin >> Q;

    unordered_map<int, int> countMap; // Tracks count of each number in the bag
    int distinctCount = 0; // Number of distinct integers currently in the bag

    for (int i = 0; i < Q; ++i) {
        int type;
        cin >> type;

        if (type == 1) {
            int x;
            cin >> x;
            if (countMap[x] == 0) {
                ++distinctCount;
            }
            ++countMap[x];
        } 
        else if (type == 2) {
            int x;
            cin >> x;
            --countMap[x];
            if (countMap[x] == 0 && x != 0) {
                --distinctCount;
            }
        } 
        else { // type == 3
            cout << distinctCount << endl;
        }
    }

    return 0;
}
```
