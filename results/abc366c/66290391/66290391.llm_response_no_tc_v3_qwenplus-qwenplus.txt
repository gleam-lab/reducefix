```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int q;
    cin >> q;
    map<int, int> countMap; // Tracks count of each integer in the bag

    rep(i, q)
    {
        int type;
        cin >> type;

        if (type == 1)
        {
            int x;
            cin >> x;
            countMap[x]++;
        }
        else if (type == 2)
        {
            int x;
            cin >> x;
            countMap[x]--;
            if (countMap[x] == 0)
            {
                countMap.erase(x);
            }
        }
        else if (type == 3)
        {
            cout << countMap.size() << endl;
        }
    }

    return 0;
}
```
