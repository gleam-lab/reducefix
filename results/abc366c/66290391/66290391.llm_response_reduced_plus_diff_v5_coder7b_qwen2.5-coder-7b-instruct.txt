```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int q;
    cin >> q;
    map<int, int> mp;
    while(q--) {
        int type, x;
        cin >> type;
        if(type == 1) {
            cin >> x;
            mp[x]++;
        } else if(type == 2) {
            cin >> x;
            mp[x]--;
            if(mp[x] == 0) mp.erase(x);
        } else if(type == 3) {
            cout << mp.size() << '\n';
        }
    }
    return 0;
}
```

This solution uses a `std::map` instead of a `std::set`. The map stores each unique value and its count. For each operation:

- If the operation is `1`, we increment the count of the value.
- If the operation is `2`, we decrement the count of the value. If the count becomes zero, we erase the value from the map.
- If the operation is `3`, we output the size of the map, which represents the number of distinct values.

This approach ensures correct handling of all operations and avoids issues related to duplicate entries in the set.
