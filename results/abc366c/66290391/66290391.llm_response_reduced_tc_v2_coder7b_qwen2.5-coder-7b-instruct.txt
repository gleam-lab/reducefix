#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int q;
    cin >> q;
    unordered_set<int> uset; // Using unordered_set for faster insertions and deletions
    rep(i, q)
    {
        int type, x;
        cin >> type;
        if (type == 1)
        {
            cin >> x;
            uset.insert(x);
        }
        else if (type == 2)
        {
            cin >> x;
            uset.erase(x);
        }
        else if (type == 3)
        {
            cout << uset.size() << endl;
        }
    }
}
