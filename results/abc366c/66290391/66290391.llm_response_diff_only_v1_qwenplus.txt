#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main()
{
    int q;
    cin >> q;
    map<int, int> count;
    rep(i, q)
    {
        int type, x;
        cin >> type;
        if (type == 1)
        {
            cin >> x;
            count[x]++;
        }
        else if (type == 2)
        {
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                count.erase(x);
            }
        }
        else if (type == 3)
        {
            cout << count.size() << endl;
        }
    }
}
