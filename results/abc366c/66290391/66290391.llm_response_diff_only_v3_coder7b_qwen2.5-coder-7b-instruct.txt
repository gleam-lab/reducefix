```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int q;
    cin >> q;
    set<int> st;
    rep(i, q)
    {
        int type, x;
        cin >> type;
        if (type == 1)
        {
            cin >> x;
            st.insert(x);
        }
        else if (type == 2)
        {
            cin >> x;
            st.erase(st.find(x));
        }
        else if (type == 3)
        {
            cout << st.size() << endl;
        }
    }
}
```

### Explanation of Changes
- The issue in the original code was related to handling the second type of query (`2 x`). Instead of directly using `st.erase(x)`, which might not work as expected because `x` might not be in the set due to its value being out of the range of the set's elements, we use `st.erase(st.find(x))`. This ensures that the correct element is removed from the set based on its actual position within the set.
