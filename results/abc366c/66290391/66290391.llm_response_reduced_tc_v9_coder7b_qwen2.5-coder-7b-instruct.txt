#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int q;
    cin >> q;
    unordered_set<int> st;
    rep(i, q)
    {
        int type, x;
        cin >> type;
        if (type == 1)
        {
            cin >> x;
            st.insert(x);
        }
        else if (type == 2)
        {
            cin >> x;
            st.erase(x);
        }
        else if (type == 3)
        {
            cout << st.size() << endl;
        }
    }

    return 0;
}

In this solution, I've made two primary changes to fix the original code:

1. Added `ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);` at the beginning of the program to optimize the input/output operations.
2. Changed `set<int> st;` to `unordered_set<int> st;`. This change improves the time complexity of insertions and deletions from average O(log n) to O(1), making it more efficient for large inputs as specified in the problem's constraints.
