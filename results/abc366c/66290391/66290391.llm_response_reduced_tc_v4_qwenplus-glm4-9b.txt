#include <iostream>
#include <set>

int main() {
    int q;
    std::cin >> q;

    std::set<int> st;

    while (q--) {
        int type;
        std::cin >> type;

        if (type == 1) {
            int x;
            std::cin >> x;
            st.insert(x);
        } else if (type == 2) {
            int x;
            std::cin >> x;
            st.erase(x);
        } else if (type == 3) {
            std::cout << st.size() << std::endl;
        }
    }

    return 0;
}
