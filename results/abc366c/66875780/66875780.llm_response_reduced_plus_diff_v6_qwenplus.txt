```cpp
#include <iostream>
#include <map>

using namespace std;

// Fast IO
void fastIO()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
}

void Solve()
{
    int Q;
    cin >> Q;
    
    // Use a map to count frequency of each number
    map<int, int> freq;
    
    while (Q--)
    {
        int type;
        cin >> type;
        
        if (type == 1)
        {
            int x;
            cin >> x;
            freq[x]++;
        }
        else if (type == 2)
        {
            int x;
            cin >> x;
            freq[x]--;
            // Remove the key if count becomes 0 to keep only existing numbers
            if (freq[x] == 0) {
                freq.erase(x);
            }
        }
        else
        {
            // Print number of distinct integers in the bag
            cout << freq.size() << "\n";
        }
    }
}

int main()
{
    fastIO();
    
    Solve();
    
    return 0;
}
```
