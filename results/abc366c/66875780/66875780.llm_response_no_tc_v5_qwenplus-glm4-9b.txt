#include <iostream>
#include <unordered_set>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int Q;
    cin >> Q;

    unordered_set<int> bag;
    vector<int> results;

    for (int q = 0; q < Q; q++) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            bag.insert(x);
        } else if (type == 2) {
            int x;
            cin >> x;
            bag.erase(x);
        } else if (type == 3) {
            results.push_back(bag.size());
        }
    }

    for (int result : results) {
        cout << result << '\n';
    }

    return 0;
}
