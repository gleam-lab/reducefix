        {
            ll x;
            cin >> x;

                s.erase(it);      // Erase only one occurrence

// Revised version with proper frequency tracking
int main()
{
    fastIO();

    ll Q;
    cin >> Q;
    
    unordered_map<ll, ll> freq;  // frequency of each number
    ll distinct = 0;             // current count of distinct numbers
    
    while (Q--)
    {
        ll type;
        cin >> type;
        
        if (type == 1)
        {
            ll x;
            cin >> x;
            if (freq[x] == 0) {
                distinct++;
            }
            freq[x]++;
        }
        else if (type == 2)
        {
            ll x;
            cin >> x;
            freq[x]--;
            if (freq[x] == 0) {
                distinct--;
            }
        }
        else // type == 3
        {
            cout << distinct << '\n';
        }
    }

    return 0;
}
