```cpp
#include <iostream>
#include <set>
#include <unordered_set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0);

    unordered_set<int> s;
    int n; cin >> n;
    while(n--) {
        int op, x; cin >> op >> x;
        if(op == 1) s.insert(x);
        else if(op == 2) s.erase(s.find(x));
        else cout << s.size() << '\n';
    }
    return 0;
}
```

This solution uses an `unordered_set` instead of a `set`. The reason being that insertion, deletion, and search operations are faster in average case (O(1)) compared to a `set` (O(log N)). This would significantly improve performance for large inputs as required by the problem constraints.
