#include <iostream>
#include <unordered_set>

using namespace std;

void Solve() {
    int Q;
    cin >> Q;
    unordered_set<int> s;
    while (Q--) {
        int a;
        cin >> a;
        if (a == 1) {
            int x;
            cin >> x;
            s.insert(x);
        } else if (a == 2) {
            int x;
            cin >> x;
            s.erase(x);
        } else if (a == 3) {
            cout << s.size() << "\n";
        }
    }
}

int main() {
    Solve();
    return 0;
}
