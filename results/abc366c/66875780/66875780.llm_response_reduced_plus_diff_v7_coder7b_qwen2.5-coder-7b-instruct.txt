#include <iostream>
#include <set>
#include <unordered_set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int Q;
    cin >> Q;

    unordered_set<int> s;

    while(Q--) {
        int op, val;
        cin >> op >> val;
        
        if(op == 1) {
            s.insert(val);
        } else if(op == 2) {
            s.erase(s.find(val));
        } else {
            cout << s.size() << '\n';
        }
    }

    return 0;
}
