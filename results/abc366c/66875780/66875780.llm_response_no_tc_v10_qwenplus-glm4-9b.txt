#include <iostream>
#include <unordered_set>
#include <vector>

using namespace std;

int main() {
    int Q;
    cin >> Q;

    unordered_set<int> bag;
    vector<int> results;

    for (int i = 0; i < Q; ++i) {
        int query_type;
        cin >> query_type;

        if (query_type == 1) {
            int x;
            cin >> x;
            bag.insert(x);
        } else if (query_type == 2) {
            int x;
            cin >> x;
            bag.erase(x);
        } else if (query_type == 3) {
            results.push_back(bag.size());
        }
    }

    for (int result : results) {
        cout << result << "\n";
    }

    return 0;
}
