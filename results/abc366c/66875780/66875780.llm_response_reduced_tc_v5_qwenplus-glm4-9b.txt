#include <iostream>
#include <set>
#include <unordered_map>

using namespace std;

int main() {
    long long Q;
    cin >> Q;
    
    set<long long> balls;
    while (Q--) {
        long long query;
        cin >> query;
        
        if (query == 1) {
            long long x;
            cin >> x;
            balls.insert(x);
        } else if (query == 2) {
            long long x;
            cin >> x;
            balls.erase(x);
        } else if (query == 3) {
            cout << balls.size() << "\n";
        }
    }
    return 0;
}
