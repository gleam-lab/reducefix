#include <iostream>
#include <set>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int Q;
    cin >> Q;

    set<int> ballSet;
    while (Q--) {
        int query;
        cin >> query;
        if (query == 1) {
            int x;
            cin >> x;
            ballSet.insert(x);
        } else if (query == 2) {
            int x;
            cin >> x;
            ballSet.erase(x);
        } else if (query == 3) {
            cout << ballSet.size() << "\n";
        }
    }

    return 0;
}
