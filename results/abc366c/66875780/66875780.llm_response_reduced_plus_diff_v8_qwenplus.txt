```cpp
#include <iostream>
#include <map>
using namespace std;

typedef long long ll;

void fastIO()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
}

void Solve()
{
    ll Q;
    cin >> Q;
    map<ll, ll> ballCount; // Maps integer x to its frequency in the bag
    
    while (Q--)
    {
        ll a;
        cin >> a;
        if (a == 1)
        {
            ll x;
            cin >> x;
            ballCount[x]++;
        }
        else if (a == 2)
        {
            ll x;
            cin >> x;
            ballCount[x]--;
            if (ballCount[x] == 0) {
                ballCount.erase(x);
            }
        }
        else
        {
            cout << ballCount.size() << "\n";
        }
    }
}

int main()
{
    fastIO();
    
    int t = 1;
    while (t--)
    {
        Solve();
    }

    return 0;
}
```
