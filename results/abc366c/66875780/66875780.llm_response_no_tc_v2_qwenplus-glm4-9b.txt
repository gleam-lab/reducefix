#include <iostream>
#include <set>

using namespace std;

int main() {
    int Q;
    cin >> Q;

    set<int> s;
    while (Q--) {
        int a;
        cin >> a;
        if (a == 1) {
            int x;
            cin >> x;
            s.insert(x);
        } else if (a == 2) {
            int x;
            cin >> x;
            s.erase(x);
        } else {
            cout << s.size() << '\n';
        }
    }

    return 0;
}
