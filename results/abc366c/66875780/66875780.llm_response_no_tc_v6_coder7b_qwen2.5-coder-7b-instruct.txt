#include <iostream>
#include <set>

using namespace std;

int main() {
    int Q;
    cin >> Q;
    
    set<int> s;
    
    while(Q--) {
        int type;
        cin >> type;
        
        if(type == 1) {
            int x;
            cin >> x;
            s.insert(x);
        } else if(type == 2) {
            int x;
            cin >> x;
            s.erase(s.find(x));
        } else if(type == 3) {
            cout << s.size() << endl;
        }
    }
    
    return 0;
}
