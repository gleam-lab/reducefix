#include <iostream>
#include <set>

using namespace std;

typedef long long ll;

void fastIO()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
}

void Solve()
{
    ll Q;
    cin >> Q;
    map<ll, ll> count; // To keep track of frequency of each number
    while (Q--)
    {
        ll a;
        cin >> a;
        if (a == 1)
        {
            ll x;
            cin >> x;
            count[x]++;
        }
        else if (a == 2)
        {
            ll x;
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                count.erase(x);
            }
        }
        else
        {
            cout << count.size() << "\n";
        }
    }
}

int main()
{
    fastIO();
    Solve();
    return 0;
}
