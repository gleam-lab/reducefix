```cpp
#include <iostream>
#include <unordered_map>
using namespace std;

// Fast IO
void fastIO()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
}

void Solve()
{
    int Q;
    cin >> Q;
    
    unordered_map<int, int> countMap; // To store count of each integer in the bag
    int distinctCount = 0; // To track the number of distinct integers

    while (Q--)
    {
        int queryType;
        cin >> queryType;

        if (queryType == 1)
        {
            int x;
            cin >> x;
            if (countMap[x] == 0)
                distinctCount++;
            countMap[x]++;
        }
        else if (queryType == 2)
        {
            int x;
            cin >> x;
            countMap[x]--;
            if (countMap[x] == 0)
                distinctCount--;
        }
        else // queryType == 3
        {
            cout << distinctCount << "\n";
        }
    }
}

int main()
{
    fastIO();
    int t = 1;
    // cin >> t;
    while (t--)
    {
        Solve();
    }

    return 0;
}
```
