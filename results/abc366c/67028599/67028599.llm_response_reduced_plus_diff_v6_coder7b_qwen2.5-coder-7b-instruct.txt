```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int q;
    cin >> q;
    set<int> s;

    while(q--) {
        int type;
        cin >> type;
        if(type == 1) {
            int x;
            cin >> x;
            s.insert(x);
        } else if(type == 2) {
            int x;
            cin >> x;
            s.erase(s.find(x));
        } else {
            cout << s.size() << '\n';
        }
    }

    return 0;
}
```

This solution uses a `set` from the C++ STL library, which automatically sorts its elements and ensures uniqueness. The operations insert and erase both run in logarithmic time due to the tree-like structure of the `set`. This meets the requirement of handling up to 2 * 10^5 queries efficiently.
