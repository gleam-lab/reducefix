```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {

    int q;
    cin >> q;

    unordered_set<int> h;
    int type, x;
    while( q-- ) {
        cin >> type;
        if ( type == 1 ) {
            cin >> x;
            h.insert(x);
        } else if ( type == 2 ) {
            cin >> x;
            h.erase(h.find(x));
        } else {
            cout << h.size() << endl;
        }
    }

return 0;
}
```
