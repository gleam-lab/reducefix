```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;
    
    unordered_set<int> h; // Using unordered_set for faster insertion and deletion
    int type, x;
    while(q--) {
        cin >> type;
        if(type == 1) {
            cin >> x;
            h.insert(x); // Inserting element if not already present
        } else if(type == 2) {
            cin >> x;
            h.erase(h.find(x)); // Erasing element only if it exists
        } else {
            cout << h.size() << endl; // Printing the size of the set
        }
    }
    return 0;
}
```

This solution uses an `unordered_set` instead of a `set`. The `unordered_set` provides average O(1) time complexity for insertions, deletions, and lookups, making it more efficient for large datasets compared to the `set`, which has O(log n) time complexity for these operations. Additionally, the use of `find` before erasing ensures that the element is only removed if it exists, avoiding any undefined behavior.
