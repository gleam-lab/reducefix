```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> count; // Count frequency of each number
    int distinct = 0;    // Track number of distinct integers
    int type, x;

    while (q--) {
        cin >> type;
        if (type == 1) {
            cin >> x;
            if (count[x] == 0) {
                distinct++; // First occurrence of x
            }
            count[x]++;
        } else if (type == 2) {
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                distinct--; // No more occurrences of x
            }
        } else {
            cout << distinct << endl;
        }
    }

    return 0;
}
```
