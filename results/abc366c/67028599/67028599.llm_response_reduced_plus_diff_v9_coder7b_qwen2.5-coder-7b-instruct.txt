```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;
    
    set<int> s;
    
    while(q--) {
        int type;
        cin >> type;
        
        if(type == 1) {
            int x;
            cin >> x;
            s.insert(x);
        } 
        else if(type == 2) {
            int x;
            cin >> x;
            s.erase(s.find(x));
        } 
        else {
            cout << s.size() << endl;
        }
    }
    
    return 0;
}
```

This solution uses a set from the STL library which automatically sorts its elements and removes duplicates. The operations insert and erase both take logarithmic time, making the overall time complexity O(Q log Q), where Q is the number of queries. This approach efficiently handles all the problem requirements without any hard-coded values or special cases.
