```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> count; // Tracks frequency of each number in the bag
    int distinct = 0;    // Current count of distinct integers

    while (q--) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            if (count[x] == 0) {
                distinct++;
            }
            count[x]++;
        } 
        else if (type == 2) {
            int x;
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                distinct--;
            }
        } 
        else if (type == 3) {
            cout << distinct << endl;
        }
    }

    return 0;
}
```
