The issue with the current code is that it uses a `set` to track the balls in the bag, but this approach doesn't handle duplicate values correctly. When multiple balls with the same integer value are added, the set only stores one instance, and when one is removed, it removes all instances.

To fix this, we need to use a data structure that can count the occurrences of each integer. A `map` or `unordered_map` will work perfectly for this purpose - we can store the count of each integer, incrementing when adding and decrementing when removing.

```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> count;
    int type, x;
    
    while (q--) {
        cin >> type;
        if (type == 1) {
            cin >> x;
            count[x]++;
        } else if (type == 2) {
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                count.erase(x);
            }
        } else {
            cout << count.size() << endl;
        }
    }

    return 0;
}
```
