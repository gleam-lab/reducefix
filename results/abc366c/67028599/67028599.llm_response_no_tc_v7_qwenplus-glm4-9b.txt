#include<bits/stdc++.h>

using namespace std;

int main() {
    int q;
    cin >> q;

    set<int> h; // Using a set to keep track of unique integers
    int type, x;
    while (q--) {
        cin >> type;
        if (type == 1) {
            cin >> x;
            h.insert(x); // Inserting the integer into the set
        } else if (type == 2) {
            cin >> x;
            h.erase(x); // Removing the integer from the set
        } else {
            cout << h.size() << endl; // Output the size of the set, which is the number of unique integers
        }
    }

    return 0;
}
