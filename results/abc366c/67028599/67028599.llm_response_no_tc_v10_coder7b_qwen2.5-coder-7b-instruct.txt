```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {

    int q;
    cin >> q;

    set<int> h;
    int type, x;
    while( q-- ) {
        cin >> type;
        if ( type == 1 ) {
            cin >> x;
            h.insert(x);
        } else if ( type == 2 ) {
            cin >> x;
            h.erase(h.find(x));
        } else {
            cout << h.size() << endl;
        }
    }

return 0;
}
```

**Explanation**: The issue in the original code was in the erase operation for type 2 queries. Instead of directly erasing the value, it should find the element in the set first using `find()` method because `erase()` without finding can lead to undefined behavior. This change ensures the program handles all constraints and correctly implements the functionality described in the problem statement.
