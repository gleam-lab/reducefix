```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {

	int q;
	cin >> q;

	unordered_set<int> h; // Changed set to unordered_set for faster insert and erase operations
	int type, x;
	while( q-- ) {
		cin >> type;
		if ( type == 1 ) {
			cin >> x;
			h.insert(x); // Inserting the element directly
		} else if ( type == 2 ) {
			cin >> x;
			h.erase(h.find(x)); // Finding and erasing the element using find method
		} else {
			cout << h.size() << endl; // Printing the size of the unordered_set
		}
	}

	return 0;
}
```

This solution uses an `unordered_set` instead of a `set`. The reason for this change is that `unordered_set` provides average constant time complexity for insert, erase, and find operations, while `set` provides logarithmic time complexity for these operations. Since we need to perform these operations multiple times, using `unordered_set` will make our program run much faster, especially for large inputs.
