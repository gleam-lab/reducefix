#include<bits/stdc++.h>

using namespace std;

int main() {
    int q;
    cin >> q;

    set<int> h; // Using a set to store unique integers
    while (q--) {
        cin >> type; // Read the type of query
        if (type == 1) {
            cin >> x; // Read the integer to be added
            h.insert(x); // Insert it into the set
        } else if (type == 2) {
            cin >> x; // Read the integer to be removed
            h.erase(x); // Remove it from the set
        } else {
            cout << h.size() << endl; // Output the size of the set
        }
    }

    return 0;
}
