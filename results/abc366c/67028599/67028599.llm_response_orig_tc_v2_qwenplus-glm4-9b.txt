```cpp
#include <iostream>
#include <set>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int q;
    cin >> q;

    set<int> h;
    int type, x;

    while (q--) {
        cin >> type;
        if (type == 1) {
            cin >> x;
            h.insert(x);
        } else if (type == 2) {
            cin >> x;
            h.erase(x);
        } else {
            cout << h.size() << endl;
        }
    }

    return 0;
}
```
