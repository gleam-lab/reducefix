#include<bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;
    unordered_set<int> s; // Using unordered_set for faster operations
    while(q--) {
        int type, x;
        cin >> type;
        if(type == 1) {
            cin >> x;
            s.insert(x); // Inserting element into the set
        } else if(type == 2) {
            cin >> x;
            s.erase(s.find(x)); // Erasing element from the set
        } else {
            cout << s.size() << endl; // Printing the size of the set
        }
    }
    return 0;
}
