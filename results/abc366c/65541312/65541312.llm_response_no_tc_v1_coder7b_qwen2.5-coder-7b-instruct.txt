```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
	int q;
	cin >> q;
	unordered_set<int>st;
	for (int i = 0; i < q; i++)
	{
		int a;
		cin >> a;
		if (a == 1)
		{
			int x;
			cin >> x;
			st.insert(x);
		}
		else if (a == 2)
		{
			int y;
			cin >> y;
			st.erase(st.find(y));

		}
		else if (a == 3)
		{
			cout << st.size() << endl;
		}
	}
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif

	int t = 1;
	//cin >> t;
	while (t--)
	{
		solve();
	}
}
```

Explanation:
- The original code uses a set for storing elements, but sets do not support efficient removal based on value without iterating through them first. This leads to inefficient erase operations.
- To optimize both insertion and deletion, an unordered_set was used instead. Unordered_sets provide average O(1) time complexity for insertions, deletions, and lookups.
- The condition checking for the operation type was also improved for clarity and correctness. Each operation type is now explicitly checked using else if statements to avoid unnecessary checks after finding a match.
