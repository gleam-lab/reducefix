To handle this correctly, we need to use a map (or a dictionary) to keep track of the count of each integer in the bag. When we remove a ball, we decrement the count, and if the count reaches zero, we remove the integer from the set of distinct integers. This ensures that the set only contains integers that have at least one ball in the bag.

void solve() {

    unordered_map<int, int> count;
    set<int> distinct;
    
    for (int i = 0; i < q; i++) {
        int type;
        cin >> type;

            if (count[x] == 1) {
                distinct.insert(x);
            }

                    distinct.erase(x);
                }
            }
        } else if (type == 3) {
            cout << distinct.size() << endl;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);

    int t = 1;
