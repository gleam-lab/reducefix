#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int q;
    cin >> q;
    set<int>st;
    for (int i = 0; i < q; i++)
    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            st.insert(x);
        }
        else if (a == 2)
        {
            int y;
            cin >> y;
            st.erase(st.find(y));

        }
        else if (a == 3)
        {
            cout << st.size() << endl;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t = 1;
    //cin >> t;
    while (t--)
    {
        solve();
    }
}

This solution uses a `set` to store the numbers, which automatically removes duplicates and allows for efficient insertion, deletion, and size querying. The `set` is part of the C++ Standard Library and provides a sorted collection of unique elements.
