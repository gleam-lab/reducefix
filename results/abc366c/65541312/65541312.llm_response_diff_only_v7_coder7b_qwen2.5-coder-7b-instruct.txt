#include<bits/stdc++.h>
using namespace std;

void solve()
{
	int q;
	cin >> q;
	unordered_set<int> st;
	for (int i = 0; i < q; i++)
	{
		int a;
		cin >> a;
		if (a == 1)
		{
			int x;
			cin >> x;
			st.insert(x);
		}
		else if (a == 2)
		{
			int y;
			cin >> y;
			st.erase(st.find(y));
		}
		else if (a == 3)
		{
			cout << st.size() << endl;
		}
	}
}

int main()
{
#ifndef ONLINE_JUDGE

#endif

	int t = 1;
	// cin >> t;

		solve();
	}
}

In the original code, the `erase` operation for removing elements was incorrect because it did not use the `find` method to locate the element before erasing it. This could lead to undefined behavior if the element was not found in the set. The corrected code now uses `find` to ensure that the element exists before attempting to erase it.
