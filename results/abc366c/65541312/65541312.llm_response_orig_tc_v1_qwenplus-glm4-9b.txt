
The problem involves processing a series of queries on a set of integers, and there are three types of queries:

1. Add an integer to the set.
2. Remove an integer from the set.
3. Output the number of unique integers in the set.

1. It is not handling the queries efficiently when there are many operations.
2. It might be causing unnecessary I/O operations by printing the size of the set after every `3` query.

### Potential Fixes

1. **Efficient I/O Operations**: Instead of printing at every `3` query, we can collect all the answers to the third type of query and print them at once at the end of the function. This will minimize the number of I/O operations.

2. **Handling Large Inputs**: If the input size is very large, we can consider using a more efficient way to handle the input (e.g., using `ios::sync_with_stdio(false); cin.tie(NULL);` to speed up input/output operations).

#include <bits/stdc++.h>
using namespace std;

void solve() {
    ios::sync_with_stdio(false); // speed up input
    cin.tie(NULL);

    int q;
    cin >> q;
    set<int> st;
    vector<int> results;

    for (int i = 0; i < q; i++) {
        int a;
        cin >> a;
        if (a == 1) {
            int x;
            cin >> x;
            st.insert(x);
        } else if (a == 2) {
            int y;
            cin >> y;
            st.erase(y);
        } else if (a == 3) {
            results.push_back(st.size());
        }
    }

    for (int result : results) {
        cout << result << endl;
    }
}

int main() {
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
}
