The code fails when multiple balls with the same integer are present because it doesn't account for quantities - it only tracks presence/absence in the set.

Here's the corrected solution that uses a frequency map to track counts of each integer and a separate count for distinct integers:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    unordered_map<int, int> freq;
    int distinct = 0;
    int q;
    cin >> q;
    while (q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int x;
            cin >> x;
            if (freq[x] == 0) {
                distinct++;
            }
            freq[x]++;
        } else if (type == 2) {
            int x;
            cin >> x;
            freq[x]--;
            if (freq[x] == 0) {
                distinct--;
            }
        } else if (type == 3) {
            cout << distinct << '\n';
        }
    }
    return 0;
}
```

1. Changed from using a `set` to an `unordered_map` to track frequencies of each integer
2. Added a `distinct` counter that only increments/decrements when an integer's count changes between 0 and 1
3. Used `ios_base::sync_with_stdio(false)` and `cin.tie(nullptr)` for faster I/O
4. Simplified the logic to properly handle multiple balls with the same integer
5. Maintained O(1) average time complexity for each operation through the use of hash maps
