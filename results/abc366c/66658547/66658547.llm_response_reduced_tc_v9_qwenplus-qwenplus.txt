```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;
    
    map<int, int> freq;  // To store frequency of each number
    int distinct = 0;    // Track current number of distinct integers

    while (q--) {
        int type;
        cin >> type;

        if (type == 1 || type == 2) {
            int x;
            cin >> x;
            
            if (type == 1) {
                if (freq[x] == 0) {
                    distinct++;
                }
                freq[x]++;
            } else { // type == 2
                freq[x]--;
                if (freq[x] == 0) {
                    distinct--;
                }
            }
        } else if (type == 3) {
            cout << distinct << "\n";
        }
    }

    return 0;
}
```
