#include <iostream>
#include <set>
using namespace std;

int main() {
    int q;
    cin >> q;
    set<int> st;
    while (q--) {
        int x;
        cin >> x;
        if (x == 1 || x == 2) {
            cin >> x; // Read the value of x for both 1 and 2 queries
            if (x == 1)
                st.insert(x); // Insert into set for type 1 query
            else if (x == 2)
                st.erase(x); // Erase from set for type 2 query
        } else if (x == 3) {
            cout << st.size() << "\n"; // Output the size of the set for type 3 query
        }
    }
    return 0;
}
