```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    unordered_map<int, int> count;
    int distinct = 0;
    int q;
    cin >> q;
    
    while (q--) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            if (count[x] == 0) distinct++;
            count[x]++;
        } else if (type == 2) {
            int x;
            cin >> x;
            count[x]--;
            if (count[x] == 0) distinct--;
        } else if (type == 3) {
            cout << distinct << '\n';
        }
    }
    
    return 0;
}
```

1. Used `unordered_map` for O(1) operations instead of `set` which has O(log n) operations
2. Tracked the count of each number and the number of distinct numbers separately
3. Added `ios_base::sync_with_stdio(false)` and `cin.tie(nullptr)` for faster I/O
4. Used single quotes for `'\n'` which is slightly more efficient than `"\n"` for single characters
5. Maintained the distinct count by checking when counts transition between 0 and 1
