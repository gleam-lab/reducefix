#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    unordered_set<int> s;
    int q; cin >> q;
    vector<int> ans;

    while(q--) {
        int t; cin >> t;
        if(t == 1) {
            int x; cin >> x;
            s.insert(x);
        } else if(t == 2) {
            int x; cin >> x;
            s.erase(s.find(x));
        } else {
            ans.push_back(s.size());
        }
    }

    for(int a : ans) {
        cout << a << '\n';
    }

    return 0;
}
