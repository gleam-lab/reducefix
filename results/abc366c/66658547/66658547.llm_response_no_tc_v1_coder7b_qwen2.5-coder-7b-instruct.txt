```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    unordered_set<int> uset; // Using unordered_set for faster insertions and deletions
    int q; cin >> q;
    while(q--){
        int type, x; cin >> type;
        if(type == 1){
            cin >> x;
            uset.insert(x); // Inserting element into the set
        } else if(type == 2){
            cin >> x;
            uset.erase(x); // Erasing element from the set
        } else {
            cout << uset.size() << '\n'; // Printing the size of the set
        }
    }
    return 0;
}
```
