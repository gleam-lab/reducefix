#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);

    set<int> s;
    int q; cin >> q;
    while(q--) {
        int type, val;
        cin >> type;
        if(type == 1) {
            cin >> val;
            s.insert(val);
        } else if(type == 2) {
            cin >> val;
            s.erase(s.find(val));
        } else {
            cout << s.size() << '\n';
        }
    }

    return 0;
}
