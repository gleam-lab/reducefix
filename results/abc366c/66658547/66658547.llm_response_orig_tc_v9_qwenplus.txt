The issue with the original code is that it uses a `set` to track the balls in the bag, but when a ball is removed using `st.erase(t)`, it removes *all* occurrences of that value from the set. However, we only want to remove one occurrence at a time.

We need to track the frequency/count of each integer separately. A `map<int, int>` (or `unordered_map`) is suitable for this purpose, where the key is the integer on the ball and the value is its count.

When printing the number of different integers (query type 3), we simply output the size of the map (number of keys with positive count).

Here's the corrected version:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    map<int, int> freq;
    int q;
    cin >> q;
    while(q--){
        int type;
        cin >> type;
        if(type == 1){
            int x;
            cin >> x;
            freq[x]++;
        }
        else if(type == 2){
            int x;
            cin >> x;
            freq[x]--;
            if(freq[x] == 0){
                freq.erase(x);
            }
        }
        else if(type == 3){
            cout << freq.size() << "\n";
        }
    }
    return 0;
}
```
