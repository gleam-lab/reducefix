```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> count_map;  // To keep track of counts of each number
    int distinct_count = 0;   // To store the current number of distinct integers

    while (q--) {
        int type;
        cin >> type;

        if (type == 1 || type == 2) {
            int x;
            cin >> x;

            if (type == 1) {
                if (count_map[x] == 0) {
                    distinct_count++;
                }
                count_map[x]++;
            } else {  // type == 2
                count_map[x]--;
                if (count_map[x] == 0) {
                    distinct_count--;
                }
            }
        } else if (type == 3) {
            cout << distinct_count << "\n";
        }
    }

    return 0;
}
```
