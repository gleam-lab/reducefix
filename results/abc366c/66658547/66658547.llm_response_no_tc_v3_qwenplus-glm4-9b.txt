#include <iostream>
#include <set>

using namespace std;

int main() {
    int q;
    cin >> q;
    set<int> st; // Using a set to store unique integers
    while (q--) {
        int x;
        cin >> x;
        if (x == 1) {
            int t;
            cin >> t;
            st.insert(t); // Insert the integer into the set
        } else if (x == 2) {
            int t;
            cin >> t;
            st.erase(t); // Remove the integer from the set
        } else if (x == 3) {
            cout << st.size() << "\n"; // Output the size of the set
        }
    }
    return 0;
}
