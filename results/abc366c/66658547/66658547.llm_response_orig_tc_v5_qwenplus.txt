Looking at the problem, I need to track the number of distinct integers in a bag as balls are added and removed. The issue with the current code is that it uses a `set` which only stores unique values, but doesn't handle the case where multiple balls with the same integer can be added and then removed one by one.

The problem is that when we add multiple balls with the same value and then remove them, the set approach doesn't work correctly because:
- Adding the same value multiple times to a set has no effect (sets only store unique elements)
- Removing a value removes it entirely from the set

Instead, I need to use a map or unordered_map to count the frequency of each integer, and only add/remove from the distinct count when the frequency goes from 0 to 1 (on addition) or from 1 to 0 (on removal).

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
constexpr ll MOD  = 1000000007;
constexpr int inf = 2000000000;
constexpr ll INF = 9200000000000000000;

int main(){
    map<int, int> freq;
    int q;
    cin >> q;
    
    while(q--){
        int type;
        cin >> type;
        
        if(type == 1){
            int x;
            cin >> x;
            freq[x]++;
        }
        else if(type == 2){
            int x;
            cin >> x;
            freq[x]--;
            if(freq[x] == 0){
                freq.erase(x);
            }
        }
        else if(type == 3){
            cout << freq.size() << "\n";
        }
    }
}
```
