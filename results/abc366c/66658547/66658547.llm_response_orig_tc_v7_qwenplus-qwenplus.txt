```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    unordered_map<int, int> countMap; // Tracks count of each number
    int Q;
    cin >> Q;
    
    while (Q--) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            countMap[x]++;
        } 
        else if (type == 2) {
            int x;
            cin >> x;
            countMap[x]--;
            if (countMap[x] == 0) {
                countMap.erase(x); // Remove the key if count is zero to keep size accurate
            }
        } 
        else if (type == 3) {
            cout << countMap.size() << "\n";
        }
    }

    return 0;
}
```
