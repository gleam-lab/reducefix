#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int q;
    cin >> q;

    multiset<int> balls;
    rep(i, q) {
        int type;
        cin >> type;
        if (type == 1) {
            int x;
            cin >> x;
            balls.insert(x);
        } else if (type == 2) {
            int x;
            cin >> x;
            auto it = balls.find(x);
            if (it != balls.end()) {
                balls.erase(it);
            }
        } else if (type == 3) {
            set<int> unique_balls(balls.begin(), balls.end());
            cout << unique_balls.size() << '\n';
        }
    }

    return 0;
}
