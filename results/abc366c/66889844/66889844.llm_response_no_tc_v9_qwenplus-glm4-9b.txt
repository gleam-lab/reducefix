#include <iostream>
#include <set>

int main() {
    int Q;
    std::cin >> Q;
    
    std::set<int> bag;
    
    for (int i = 0; i < Q; ++i) {
        int query;
        std::cin >> query;
        
        if (query == 1) {
            int value;
            std::cin >> value;
            bag.insert(value);
        } else if (query == 2) {
            int value;
            std::cin >> value;
            bag.erase(value);
        } else if (query == 3) {
            std::cout << bag.size() << std::endl;
        }
    }
    
    return 0;
}
