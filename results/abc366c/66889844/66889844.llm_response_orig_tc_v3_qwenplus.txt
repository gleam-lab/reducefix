```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int main() {
    int q;
    cin >> q;
    
    set<int> bag;
    
    rep(i, q) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            bag.insert(x);
        }
        else if (type == 2) {
            int x;
            cin >> x;
            bag.erase(x);
        }
        else if (type == 3) {
            cout << bag.size() << endl;
        }
    }
    
    return 0;
}
```
