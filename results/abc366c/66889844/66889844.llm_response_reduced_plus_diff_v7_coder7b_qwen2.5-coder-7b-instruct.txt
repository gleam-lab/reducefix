#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n; cin >> n;
  unordered_set<int> s;

  while(n--) {
    int q; cin >> q;
    if(q == 1) {
      int x; cin >> x;
      s.insert(x);
    } else if(q == 2) {
      int x; cin >> x;
      s.erase(s.find(x));
    } else {
      cout << s.size() << '\n';
    }
  }

  return 0;
}
