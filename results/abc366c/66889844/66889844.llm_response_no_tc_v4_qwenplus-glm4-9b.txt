#include <iostream>
#include <set>
using namespace std;

int main() {
    int Q;
    cin >> Q;
    
    set<int> balls;
    
    while (Q--) {
        int query;
        cin >> query;
        
        if (query == 1) {
            int x;
            cin >> x;
            balls.insert(x);
        } else if (query == 2) {
            int x;
            cin >> x;
            balls.erase(x);
        } else if (query == 3) {
            cout << balls.size() << endl;
        }
    }
    
    return 0;
}
