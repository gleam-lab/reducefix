#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);

  int q; cin >> q;
  map<int, int> a;

  while(q--) {
    int type; cin >> type;
    if(type == 1) {
      int x; cin >> x;
      a[x]++;
    } else if(type == 2) {
      int x; cin >> x;
      a[x]--;
      if(a[x] == 0) a.erase(x);
    } else if(type == 3) {
      cout << a.size() << '\n';
    }
  }

  return 0;
}
