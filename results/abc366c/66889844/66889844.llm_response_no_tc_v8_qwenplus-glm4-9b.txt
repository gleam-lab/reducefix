#include <iostream>
#include <unordered_map>
using namespace std;

int main() {
    int Q;
    cin >> Q;

    unordered_map<int, int> bag;
    while (Q--) {
        int op;
        cin >> op;
        if (op == 1) { // Insert operation
            int x;
            cin >> x;
            bag[x]++;
        } else if (op == 2) { // Remove operation
            int x;
            cin >> x;
            bag[x]--;
            if (bag[x] == 0) {
                bag.erase(x);
            }
        } else if (op == 3) { // Query the number of different integers
            cout << bag.size() << endl;
        }
    }

    return 0;
}
