```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  
  int q; cin >> q;
  map<int, int> a;
  
  while(q--) {
    int t, x; cin >> t >> x;
    if(t == 1) {
      a[x]++;
    } else if(t == 2) {
      if(a[x] > 0) {
        a[x]--;
      }
    } else if(t == 3) {
      cout << a.size() << '\n';
    }
  }
  
  return 0;
}
```

Explanation:
- The initial code uses a set to store elements, but sets do not allow duplicate elements, so we need to use a map where each element's count can be tracked.
- We also use `ios::sync_with_stdio(false)` and `cin.tie(0)` to speed up I/O operations.
- For the `3` query, we simply output the size of the map, which represents the number of unique elements in the bag.
