#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);

  int Q; cin >> Q;
  map<int, int> countMap;

  while(Q--) {
    int queryType; cin >> queryType;
    if(queryType == 1) {
      int x; cin >> x;
      countMap[x]++;
    } 
    else if(queryType == 2) {
      int x; cin >> x;
      countMap[x]--;
      if(countMap[x] == 0) {
        countMap.erase(x);
      }
    } 
    else if(queryType == 3) {
      cout << countMap.size() << '\n';
    }
  }

  return 0;
}
