#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int main() {
    int n; cin >> n;
    set<int> a;
    
    rep(i, n) {
        int x; cin >> x;
        if (x == 1) {
            int y; cin >> y;
            a.insert(y);
        } else if (x == 2) {
            int y; cin >> y;
            a.erase(y);
        } else if (x == 3) {
            cout << a.size() << endl;
        }
    }
    
    return 0;
}
