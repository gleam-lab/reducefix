```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  
  int q; cin >> q;
  map<int, int> cnt;
  int distinct = 0;
  
  while(q--){
    int type; cin >> type;
    if(type == 1){
      int x; cin >> x;
      if(cnt[x] == 0) distinct++;
      cnt[x]++;
    } 
    else if(type == 2){
      int x; cin >> x;
      if(--cnt[x] == 0) distinct--;
    } 
    else if(type == 3){
      cout << distinct << '\n';
    }
  }
}
```
