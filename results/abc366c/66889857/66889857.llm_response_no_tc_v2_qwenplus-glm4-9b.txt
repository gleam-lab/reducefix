#include <iostream>
#include <set>
using namespace std;

int main() {
    int Q;
    cin >> Q;
    set<int> a;
    
    for (int i = 0; i < Q; i++) {
        int query;
        cin >> query;
        if (query == 1) {
            int x;
            cin >> x;
            a.insert(x);
        } else if (query == 2) {
            int x;
            cin >> x;
            a.erase(x);
        } else if (query == 3) {
            cout << a.size() << endl;
        }
    }
    
    return 0;
}
