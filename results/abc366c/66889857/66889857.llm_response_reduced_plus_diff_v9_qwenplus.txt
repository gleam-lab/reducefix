#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define rep2(i, s, n) for (int i = (s); i < (int)(n); i++)

int main() {
    int Q;
    cin >> Q;
    
    // Use multiset to handle multiple balls with same number
    multiset<int> bag;
    
    rep(i, Q) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            bag.insert(x);
        }
        else if (type == 2) {
            int x;
            cin >> x;
            // Remove one occurrence of x
            auto it = bag.find(x);
            if (it != bag.end()) {
                bag.erase(it);
            }
        }
        else if (type == 3) {
            // Count distinct integers in the bag
            set<int> distinct(bag.begin(), bag.end());
            cout << distinct.size() << endl;
        }
    }
}
