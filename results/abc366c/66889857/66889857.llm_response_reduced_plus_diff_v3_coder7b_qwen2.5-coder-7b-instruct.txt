#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n; cin >> n;
  map<int, int> mp;

  while(n--) {
    int q; cin >> q;
    if(q == 1) {
      int x; cin >> x;
      mp[x]++;
    } else if(q == 2) {
      int x; cin >> x;
      if(mp[x] > 0) {
        mp[x]--;
        if(mp[x] == 0) {
          mp.erase(x);
        }
      }
    } else {
      cout << mp.size() << '\n';
    }
  }

  return 0;
}
