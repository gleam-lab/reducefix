using namespace std;

int main() {

  cin.tie(nullptr);

  int n; cin >> n;
  unordered_set<int> a;

  while(n--) {
    int type; cin >> type;
    if(type == 1) {
      int x; cin >> x;
      a.insert(x);
    } 
    else if(type == 2) {
