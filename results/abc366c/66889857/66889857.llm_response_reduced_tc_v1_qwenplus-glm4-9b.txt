
```cpp
#include <iostream>
#include <unordered_set>
#include <vector>
#include <string>
using namespace std;

int main() {
    int Q;
    cin >> Q;
    vector<int> queries(Q);
    unordered_set<int> balls;
    vector<int> results;

    for (int i = 0; i < Q; ++i) {
        cin >> queries[i];
        if (queries[i] == 1) {
            int ball; cin >> ball;
            balls.insert(ball);
        } else if (queries[i] == 2) {
            int ball; cin >> ball;
            balls.erase(ball);
        } else if (queries[i] == 3) {
            results.push_back(balls.size());
        }
    }

    for (int result : results) {
        cout << result << endl;
    }

    return 0;
}
```
