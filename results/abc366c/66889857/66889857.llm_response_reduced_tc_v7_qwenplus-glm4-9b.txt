#include <iostream>
#include <set>
using namespace std;

int main() {
    int Q;
    cin >> Q;

    set<int> unique_numbers;
    
    while (Q--) {
        int query;
        cin >> query;
        if (query == 1) {
            int number;
            cin >> number;
            unique_numbers.insert(number);
        } else if (query == 2) {
            int number;
            cin >> number;
            unique_numbers.erase(number);
        } else if (query == 3) {
            cout << unique_numbers.size() << endl;
        }
    }

    return 0;
}
