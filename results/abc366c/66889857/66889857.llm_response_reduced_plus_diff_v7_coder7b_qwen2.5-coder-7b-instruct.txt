```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n; cin >> n;
    unordered_set<int> s;

    while(n--) {
        int t, x;
        cin >> t >> x;
        
        if(t == 1) {
            s.insert(x);
        } else if(t == 2) {
            s.erase(s.find(x));
        } else if(t == 3) {
            cout << s.size() << '\n';
        }
    }

    return 0;
}
```

1. **Use of `unordered_set<int>`**:
   - This ensures that we can insert and remove elements efficiently with average constant time complexity.

2. **Query Handling**:

   - For query type 2 (`2 x`), we find the element `x` in the set and erase it.
   - For query type 3 (`3`), we output the size of the set, which represents the number of distinct elements.
