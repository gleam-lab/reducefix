```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int main() {
    int Q;
    cin >> Q;
    
    map<int, int> ball_count;
    
    rep(i, Q) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int x;
            cin >> x;
            ball_count[x]++;
        }
        else if (type == 2) {
            int x;
            cin >> x;
            ball_count[x]--;
            if (ball_count[x] == 0) {
                ball_count.erase(x);
            }
        }
        else if (type == 3) {
            cout << ball_count.size() << endl;
        }
    }
    
    return 0;
}
```
