```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);

  int n; cin >> n;
  map<int, int> mp;

  while(n--) {
    int t; cin >> t;
    if(t == 1) {
      int x; cin >> x;
      mp[x]++;
    } 
    else if(t == 2) {
      int x; cin >> x;
      mp[x]--;
      if(mp[x] == 0) mp.erase(x);
    } 
    else if(t == 3) {
      cout << mp.size() << '\n';
    }
  }

  return 0;
}
```

This solution uses a `std::map`, which automatically keeps keys sorted and ensures each key is unique. The time complexity of insertion and deletion in `std::map` is logarithmic, making this solution efficient even for large inputs up to 2 * 10^5 as specified in the problem statement.
