#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int q; cin >> q;
  multiset<int> s;

  while(q--) {
    int t; cin >> t;
    if(t == 1) {
      int x; cin >> x;
      s.insert(x);
    } else if(t == 2) {
      int x; cin >> x;
      s.erase(s.find(x));
    } else {
      cout << s.size() << '\n';
    }
  }

  return 0;
}
