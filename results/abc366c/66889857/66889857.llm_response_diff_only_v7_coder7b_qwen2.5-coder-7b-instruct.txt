#include <bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);

  int n; cin >> n;
  map<int, int> freq;
  
  while(n--) {
    int x; cin >> x;
    if(x == 1) {
      int y; cin >> y;
      freq[y]++;
    } else if(x == 2) {
      int y; cin >> y;
      if(freq[y] > 0) {
        freq[y]--;
        if(freq[y] == 0) freq.erase(y);
      }
    } else if(x == 3) {
      cout << freq.size() << '\n';
    }
  }

  return 0;
}
