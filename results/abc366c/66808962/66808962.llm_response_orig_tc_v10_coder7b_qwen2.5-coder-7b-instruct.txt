#include<iostream>
#include<unordered_set>
using namespace std;

int main(){
    unordered_set<int> s;
    int q;
    cin >> q;
    while(q--){
        int n;
        cin >> n;
        if(n == 1){
            int x;
            cin >> x;
            if(s.find(x) == s.end()){
                s.insert(x);
            }
        }
        else if(n == 2){
            int x;
            cin >> x;
            if(s.find(x) != s.end()){
                s.erase(x);
            }
        }
        else {
            cout << s.size() << endl;
        }
    }
    return 0;
}
