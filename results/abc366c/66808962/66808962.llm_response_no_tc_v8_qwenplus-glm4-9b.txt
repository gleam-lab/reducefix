#include <iostream>
#include <map>
using namespace std;

int main() {
    int q;
    cin >> q;
    map<int, int> m;
    int ans = 0; // This should actually be 0, but we'll set it to 1 to account for the empty bag
    // Add the initial 0 ball to correctly count the empty bag scenario
    m[0] = 1;
    ans = 1;

    for (int i = 0; i < q; i++) {
        int n, x;
        cin >> n;

        if (n == 1) {
            cin >> x;
            // Increment the count of the ball x
            m[x]++;
        } else if (n == 2) {
            cin >> x;
            // Decrement the count of the ball x
            m[x]--;
            // If the count drops to zero, remove the ball from the map
            if (m[x] == 0) {
                m.erase(x);
                ans--; // We need to subtract from ans only if an actual new ball is removed
            }
        } else if (n == 3) {
            // Output the current count of different integers
            cout << ans << endl;
        }
    }

    return 0;
}
