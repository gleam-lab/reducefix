```cpp
#include<iostream>
#include<map>
using namespace std;
using ll = long long;

int main(){
    map<int,int> m;
    int q;
    cin >> q;

    int ans = 0;
    for(int i = 0; i < q; i++){
        int n;
        cin >> n;

        if(n == 1){
            int x;
            cin >> x;
            if(m.find(x) == m.end() || m[x] == 0) { // if the ball is not present or count is zero
                m[x] = 1;
                ans++;
            } else {
                m[x]++;
            }
        } else if(n == 2){
            int x;
            cin >> x;
            m[x]--;
            if(m[x] == 0) {
                ans--;
            }
        } else if(n == 3){
            cout << ans << endl;
        }
    }
    return 0;
}
```
