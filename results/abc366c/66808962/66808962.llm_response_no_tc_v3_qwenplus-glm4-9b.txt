
```cpp
#include <iostream>
#include <map>

using namespace std;

int main() {
    map<int, int> m; // map to keep track of balls
    int q;
    cin >> q;

    int ans = 0; // number of different integers written on the balls

    for (int i = 0; i < q; i++) {
        int n, x;
        cin >> n;

        if (n == 1) {
            cin >> x; // read the value of the ball
            m[x]++; // increment the count of this ball
            if (m[x] == 1) {
                ans++; // if this is the first time we're seeing this ball, increment the count of different integers
            }
        } else if (n == 2) {
            cin >> x; // read the value of the ball
            m[x]--; // decrement the count of this ball
            if (m[x] == 0) {
                ans--; // if this is the last time we're seeing this ball, decrement the count of different integers
            }
        } else if (n == 3) {
            cout << ans << endl; // print the number of different integers written on the balls
        }
    }

    return 0;
}
```
