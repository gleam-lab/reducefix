#include <iostream>
#include <unordered_map>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int q;
    std::cin >> q;

    std::unordered_map<int, int> ball_count;
    int unique_count = 0;
    int current_query_type;

    for (int i = 0; i < q; ++i) {
        std::cin >> current_query_type;

        if (current_query_type == 1) {
            int x;
            std::cin >> x;
            if (ball_count.find(x) == ball_count.end()) {
                ball_count[x] = 1;
                unique_count++;
            } else {
                ball_count[x]++;
            }
        } else if (current_query_type == 2) {
            int x;
            std::cin >> x;
            ball_count[x]--;
            if (ball_count[x] == 0) {
                ball_count.erase(x);
                unique_count--;
            }
        } else if (current_query_type == 3) {
            std::cout << unique_count << std::endl;
        }
    }

    return 0;
}
