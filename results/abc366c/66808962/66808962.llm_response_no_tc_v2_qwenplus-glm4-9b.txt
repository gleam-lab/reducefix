
```cpp
#include <iostream>
#include <map>

using namespace std;

int main() {
    map<int, int> m; // map to store the count of each ball
    int q; // number of queries
    cin >> q; // read number of queries

    int ans = 0; // answer to the third type query
    for (int i = 0; i < q; ++i) {
        int n; // type of the current query
        cin >> n;

        if (n == 1) {
            int x; // the value of the ball
            cin >> x;
            if (m[x] == 0) { // if the ball with value x is not in the bag
                m[x] = 1;
                ans++; // increment the answer since we have a new unique ball
            } else {
                m[x]++; // increment the count of the ball with value x
            }
        } else if (n == 2) {
            int x; // the value of the ball
            cin >> x;
            m[x]--; // decrement the count of the ball with value x
            if (m[x] == 0) { // if the count is zero, remove the ball from the map
                ans--; // decrement the answer since we have one less unique ball
            }
        } else if (n == 3) {
            cout << ans << endl; // output the number of unique balls
        }
    }
    return 0;
}
```
