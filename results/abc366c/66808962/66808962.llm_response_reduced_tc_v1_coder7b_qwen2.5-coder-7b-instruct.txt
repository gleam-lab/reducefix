#include<iostream>
#include<vector>
#include<string>
#include<set>
#include<algorithm>
#include<map>
#include<cmath>
#include<math.h>
#include<queue>
#include <iomanip> //setprecisionを使うため
using namespace std;
using ll = long long;
using ull = unsigned long long;

int main(){
    set<int> s;
    int q;
    cin >> q;

    int ans=0;
    for(int i=0;i<q;i++){
        int n,x;
        cin >> n;

        if(n==1){
            cin >> x;
            if(s.insert(x).second) { //まだそのボールが1個もなかった時
                ans++;
            }
        }else if(n==2){
            cin >> x;
            s.erase(x);
            if(s.empty()) ans=0;
            else ans=s.size();
        }else{
            cout << ans << endl;
        }
    }
    return 0;
}
