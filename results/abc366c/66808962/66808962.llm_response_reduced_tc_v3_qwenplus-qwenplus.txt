```cpp
#include<iostream>
#include<map>
using namespace std;

int main(){
    map<int, int> countMap;
    int q;
    cin >> q;
    
    int distinctCount = 0;
    
    for(int i = 0; i < q; ++i){
        int type;
        cin >> type;
        
        if(type == 1 || type == 2){
            int x;
            cin >> x;
            
            if(type == 1){
                if(countMap[x] == 0){
                    distinctCount++;
                }
                countMap[x]++;
            } else { // type == 2
                countMap[x]--;
                if(countMap[x] == 0){
                    distinctCount--;
                }
            }
        } else { // type == 3
            cout << distinctCount << "\n";
        }
    }
    
    return 0;
}
```
