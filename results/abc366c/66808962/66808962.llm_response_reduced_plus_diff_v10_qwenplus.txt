```cpp
#include<iostream>
#include<map>
using namespace std;
using ll = long long;

int main(){
    map<int,int> m;
    int q;
    cin >> q;

    int ans=0;
    for(int i=0;i<q;i++){
        int n,x;
        cin >> n;

        if(n==1){
            cin >> x;
            if(m[x] == 0) { //まだそのボールが1個もなかった時
                m[x] = 1;
                ans++;
            }else{
                m[x]++;
            }
        }else if(n==2){
            cin >> x;
            m[x]--;
            if(m[x] == 0) ans--;
        }else{
            cout << ans << endl;
        }
    }
    return 0;
}
```
