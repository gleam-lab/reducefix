```cpp
#include<iostream>
#include<vector>
#include<string>
#include<set>
#include<algorithm>
#include<map>
#include<cmath>
#include<math.h>
#include<queue>
#include <iomanip> //setprecisionを使うため
using namespace std;
using ll = long long;
using ull = unsigned long long;

int main(){
    set<int> s; // Using set instead of map to keep track of unique elements efficiently
    int q;
    cin >> q;

    int ans=0;
    for(int i=0;i<q;i++){
        int n,x;
        cin >> n;

        if(n==1){
            cin >> x;
            if(s.find(x) == s.end()) { // If the element is not found in the set
                s.insert(x);
                ans++;
            }
        }else if(n==2){
            cin >> x;
            s.erase(x); // Erase the element directly
            if(s.empty()) ans = 0; // If the set becomes empty, reset the count
            else ans--; // Decrement the count as we removed one element
        }else{
            cout << ans << endl;
        }
    }
    // cout << m.size() << endl;
    return 0;
}
```
