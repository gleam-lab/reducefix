#include <iostream>
#include <set>
using namespace std;

int main() {
    int q, x;
    cin >> q;

    set<int> bag;

    for (int i = 0; i < q; i++) {
        cin >> x;
        if (x == 1) {
            bag.insert(x); // Insert the ball into the bag
        } else if (x == 2) {
            bag.erase(x); // Remove the ball from the bag
        } else if (x == 3) {
            cout << bag.size() << endl; // Output the number of different integers in the bag
        }
    }

    return 0;
}
