
```cpp
#include <iostream>
#include <map>

using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> balls;
    int different_balls = 0;

    for (int i = 0; i < q; ++i) {
        int query_type;
        cin >> query_type;

        if (query_type == 1) {
            int x;
            cin >> x;
            if (balls[x] == 0) {
                different_balls++;
            }
            balls[x]++;
        } else if (query_type == 2) {
            int x;
            cin >> x;
            balls[x]--;
            if (balls[x] == 0) {
                different_balls--;
            }
        } else if (query_type == 3) {
            cout << different_balls << endl;
        }
    }

    return 0;
}
```
