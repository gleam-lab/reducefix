#include <iostream>
#include <map>
using namespace std;

int main() {
    int q;
    cin >> q;

    map<int, int> m;
    int ans = 0; // To keep track of the number of distinct integers

    while (q--) {
        int n;
        cin >> n;

        if (n == 1) {
            int x;
            cin >> x;
            if (m.find(x) == m.end()) { // Check if x is not already in the map
                m[x] = 1; // Add x to the map
                ans++;   // Increment the count of distinct integers
            }
        } else if (n == 2) {
            int x;
            cin >> x;
            if (m[x] == 1) { // Check if there is only one occurrence of x
                m.erase(x); // Remove x from the map
                ans--;    // Decrement the count of distinct integers
            } else {
                m[x]--; // Decrement the count of x in the map
            }
        } else if (n == 3) {
            cout << ans << endl; // Output the number of distinct integers
        }
    }

    return 0;
}
