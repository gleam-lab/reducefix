#include<map>
using namespace std;

int main(){

    int q;
    cin >> q;

    int ans = 0;
    for(int i=0;i<q;i++){

            cin >> x;
            if(m[x] == 0) { // If x was not in the bag before, increment ans
                ans++;

            m[x]++;

            cin >> x;
            m[x]--;
            if(m[x] == 0) { // If x is no longer in the bag, decrement ans
                ans--;
            }
        }else{
            cout << ans << endl;
        }
    }
    return 0;
}
