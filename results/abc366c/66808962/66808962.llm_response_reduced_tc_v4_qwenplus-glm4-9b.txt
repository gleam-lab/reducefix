#include <iostream>
#include <map>
using namespace std;

int main() {
    map<int, int> m;
    int q;
    cin >> q;

    while (q--) {
        int n;
        cin >> n;

        if (n == 1) { // Add ball x to the bag
            int x;
            cin >> x;
            m[x]++;
        } else if (n == 2) { // Remove ball x from the bag
            int x;
            cin >> x;
            m[x]--;
            if (m[x] == 0) m.erase(x);
        } else { // Output the number of different integers in the bag
            cout << m.size() << endl;
        }
    }

    return 0;
}
