#include <iostream>
using namespace std;

int main() {
    int q;
    cin >> q;

    set<int> balls; // Use a set to store unique integers
    while (q--) {
        int n;
        cin >> n;
        if (n == 1) {
            int x;
            cin >> x;
            balls.insert(x); // Insertion maintains unique elements
        } else if (n == 2) {
            int x;
            cin >> x;
            balls.erase(x); // Erase removes one occurrence of the element
        } else if (n == 3) {
            cout << balls.size() << endl; // Output the number of unique elements
        }
    }

    return 0;
}
