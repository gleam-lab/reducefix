This leads to `ans` being undercounted.

The solution is to ensure that whenever a ball is added (type 1 query), we increment `ans` only if the count of that integer was zero before the addition. Similarly, when a ball is removed (type 2 query), we decrement `ans` only if the count of that integer becomes zero after the removal. This ensures that `ans` accurately reflects the number of distinct integers in the bag at any point.

#include <map>

using namespace std;

    int ans = 0;

        int n, x;
        cin >> n;

        if (n == 1) {
            cin >> x;
            if (m[x] == 0) { // Only increment ans if the count was zero before adding
                ans++;
            }
            m[x]++;
        } else if (n == 2) {
            cin >> x;
            m[x]--;
            if (m[x] == 0) { // Only decrement ans if the count becomes zero after removal
                ans--;
            }
        } else {
            cout << ans << endl;
        }
    }
    return 0;
}

2. **Processing Queries**:
