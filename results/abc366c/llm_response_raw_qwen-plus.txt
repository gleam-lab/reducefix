import os
import subprocess
from typing import List, Tuple, Optional, Set
import sys

def run_program(program_path: str, input_data: bytes) -> bytes:
    result = subprocess.run(
        [program_path], 
        input=input_data,
        capture_output=True,
        check=True,  
        timeout=10   
    )
    return result.stdout

def parse_input(lines: List[str]) -> Tuple[int, List[Tuple]]:
    Q = int(lines[0])
    queries = []
    for line in lines[1:]:
        parts = line.strip().split()
        if parts[0] == '3':
            queries.append((3,))
        else:
            t, x = parts
            queries.append((int(t), int(x)))
    return Q, queries

def create_input_bytes(Q: int, queries: List[Tuple]) -> bytes:
    lines = [f"{Q}\n"]
    for q in queries:
        if q[0] == 3:
            lines.append("3\n")
        else:
            lines.append(f"{q[0]} {q[1]}\n")
    return "".join(lines).encode("utf-8")

def test_interesting(data: Tuple[int, List[Tuple]]) -> bool:
    Q, queries = data
    input_data = create_input_bytes(Q, queries)
    try:
        wa_output = run_program("./wa", input_data)
        ac_output = run_program("./ac", input_data)
        return wa_output.strip() != ac_output.strip()
    except:
        return True

def reduce_queries_ddmin(original_Q: int, original_queries: List[Tuple]) -> List[Tuple]:
    current_indices = set(range(original_Q))
    granularity = 2
    
    while len(current_indices) >= 1:
        print(f"[Reduce] 当前下标集大小={len(current_indices)}: {sorted(list(current_indices))}")
        print(f"         尝试粒度: {granularity}")
        num_indices = len(current_indices)
        subset_size = max(1, num_indices // granularity)
        made_progress = False
        
        candidate_indices_list = list(current_indices)
        
        # Try removing a subset
        print(f"         1) 尝试『移除』大小约={subset_size} 的分块...")
        for start_idx in range(0, num_indices, subset_size):
            block = candidate_indices_list[start_idx:start_idx + subset_size]
            candidate_indices = set([i for i in current_indices if i not in block])
            
            if not candidate_indices:
                continue
                
            candidate_queries = [original_queries[i] for i in sorted(candidate_indices)]
            if test_interesting((len(candidate_indices), candidate_queries)):
                print(f"成功! 新下标集大小={len(candidate_indices)}")
                current_indices = candidate_indices
                granularity = 2
                made_progress = True
                break
        
        if made_progress:
            continue
        
        # Try keeping only a subset
        print(f"         2) 尝试『只保留』大小约={subset_size} 的分块...")
        for start_idx in range(0, num_indices, subset_size):
            block = candidate_indices_list[start_idx:start_idx + subset_size]
            candidate_indices = set(block)
            
            if not candidate_indices or len(candidate_indices) == num_indices:
                continue
                
            candidate_queries = [original_queries[i] for i in sorted(candidate_indices)]
            if test_interesting((len(candidate_indices), candidate_queries)):
                print(f"成功! 新下标集大小={len(candidate_indices)}")
                current_indices = candidate_indices
                granularity = 2
                made_progress = True
                break
        
        if made_progress:
            continue
        
        # If no progress, increase granularity
        if granularity < num_indices:
            new_gran = min(granularity * 2, num_indices)
            print(f"         -> 无进展，增大粒度至 {new_gran}\n")
            granularity = new_gran
        else:
            print("\n[Reduce] 已无法进一步缩减 —— 到达 1-minimal。")
            break
    
    final_queries = [original_queries[i] for i in sorted(current_indices)]
    print(f"\n[Result] 缩减后查询数 M={len(final_queries)}，保留下标: {sorted(list(current_indices))}")
    return len(final_queries), final_queries

def reduce_input(input_file: str, output_file: str):
    if not os.path.exists(input_file):
        print(f"[Error] 输入文件不存在: {input_file}", file=sys.stderr)
        return 
    
    with open(input_file, "r", encoding="utf-8") as f:
        lines = [line.rstrip("\n") for line in f]
    
    if not lines:
        raise ValueError("输入文件为空")
    
    Q, queries = parse_input(lines)
    
    print(f"[Info] 初始查询数 Q={Q} 已确认导致输出差异，开始 ddmin 缩减...\n")
    reduced_Q, reduced_queries = reduce_queries_ddmin(Q, queries)
    
    output_bytes = create_input_bytes(reduced_Q, reduced_queries)
    
    with open(output_file, "wb") as f:
        f.write(output_bytes)
    
    print(f"[Info] 缩减过程完成，结果已写入: {output_file}")
