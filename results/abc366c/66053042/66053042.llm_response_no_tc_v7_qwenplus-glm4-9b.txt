
```cpp
#include <iostream>
#include <map>

using namespace std;

int main() {
    map<int, int> m; // map to store the count of each number
    int Q; // number of queries
    cin >> Q; // input number of queries

    for (int i = 0; i < Q; i++) {
        int type;
        cin >> type; // read the type of query

        if (type == 1) {
            int x;
            cin >> x; // read the number on the ball
            m[x]++; // increment the count of this number
        } else if (type == 2) {
            int x;
            cin >> x; // read the number on the ball
            m[x]--; // decrement the count of this number
            if (m[x] == 0) {
                m.erase(x); // remove the number from the map if its count drops to zero
            }
        } else if (type == 3) {
            cout << m.size() << endl; // output the number of unique integers in the bag
        }
    }

    return 0;
}
```
