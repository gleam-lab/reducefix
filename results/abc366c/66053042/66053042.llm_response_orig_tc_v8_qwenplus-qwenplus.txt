```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    map<int, int> countMap;
    int uniqueCount = 0;

    int Q;
    cin >> Q;

    for (int i = 0; i < Q; ++i)
    {
        int type;
        cin >> type;

        if (type == 1)
        {
            int x;
            cin >> x;
            if (countMap[x] == 0)
                ++uniqueCount;
            ++countMap[x];
        }
        else if (type == 2)
        {
            int x;
            cin >> x;
            --countMap[x];
            if (countMap[x] == 0)
                --uniqueCount;
        }
        else if (type == 3)
        {
            cout << uniqueCount << endl;
        }
    }

    return 0;
}
```
