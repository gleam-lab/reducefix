```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    unordered_map<int, int> m;
    int Q;
    cin >> Q;
    int n = 0;
    for (int i = 0; i < Q; i++)
    {
        int type;
        cin >> type;

        if (type == 1)
        {
            int x;
            cin >> x;
            if (m[x] == 0)
            {
                n++;
            }
            m[x]++;
        }
        else if (type == 2)
        {
            int x;
            cin >> x;
            m[x]--;
            if (m[x] == 0)
            {
                n--;
                m.erase(x);
            }
        }
        else if (type == 3)
        {
            cout << n << endl;
        }
    }
}
```

Key changes made:
1. Changed `map` to `unordered_map` for better performance (O(1) average case operations)
2. Simplified the insertion logic for type 1 query
3. For type 2 queries, now properly erases the key from the map when count reaches 0
4. The count of unique numbers (`n`) is now correctly maintained by checking if the count was 0 before incrementing (type 1) and properly decremented when count reaches 0 (type 2)
