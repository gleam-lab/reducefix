```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    unordered_map<int, int> m;
    int Q;
    cin >> Q;
    int n = 0;
    for (int i = 0; i < Q; i++)
    {
        int type;
        cin >> type;

        if (type == 1)
        {
            int x;
            cin >> x;
            m[x]++;
            if (m[x] == 1) n++;
        }
        else if (type == 2)
        {
            int x;
            cin >> x;
            m[x]--;
            if (m[x] == 0) n--;
        }
        else if (type == 3)
        {
            cout << n << endl;
        }
    }
}
```

1. Changed `map` to `unordered_map` for better performance (O(1) average case operations)
2. Simplified the logic for adding elements - always increment count and check if it's a new unique element
3. Fixed the removal logic to only decrement `n` when count goes to 0
