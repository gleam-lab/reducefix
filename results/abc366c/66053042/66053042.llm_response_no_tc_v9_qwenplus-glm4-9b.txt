
```cpp
#include <iostream>
#include <unordered_map>
using namespace std;

int main() {
  unordered_map<int, int> m;
  int Q;
  cin >> Q;

  int distinct_count = 0; // This will keep track of the number of different integers
  for (int i = 0; i < Q; i++) {
    int type;
    cin >> type;

    if (type == 1) { // Insert ball with integer x
      int x;
      cin >> x;
      if (m[x] == 0) { // If x is not already in the map, it's the first occurrence, increment distinct_count
        distinct_count++;
      }
      m[x]++; // Increment the count of x in the map
    } else if (type == 2) { // Remove ball with integer x
      int x;
      cin >> x;
      m[x]--; // Decrement the count of x in the map
      if (m[x] == 0) { // If count reaches zero, decrement distinct_count
        distinct_count--;
      }
    } else if (type == 3) { // Output the number of different integers
      cout << distinct_count << endl;
    }
  }

  return 0;
}
```
