#include <iostream>
#include <unordered_map>

using namespace std;

int main() {
    unordered_map<int, int> ball_count;
    int Q;
    cin >> Q;

    for (int i = 0; i < Q; ++i) {
        int query_type;
        cin >> query_type;

        if (query_type == 1) {
            int x;
            cin >> x;
            ball_count[x]++;
        } else if (query_type == 2) {
            int x;
            cin >> x;
            if (--ball_count[x] == 0) {
                ball_count.erase(x);
            }
        } else if (query_type == 3) {
            cout << ball_count.size() << endl;
        }
    }

    return 0;
}
