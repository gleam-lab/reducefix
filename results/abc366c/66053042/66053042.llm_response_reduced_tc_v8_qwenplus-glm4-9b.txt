
```cpp
#include <iostream>
#include <unordered_set>
using namespace std;

int main() {
    unordered_set<int> bag;
    int Q;
    cin >> Q;
    int distinct_count = 0;

    for (int i = 0; i < Q; i++) {
        int type;
        cin >> type;

        if (type == 1) {
            int x;
            cin >> x;
            bag.insert(x);  // Insert into the set, set automatically handles duplicates
            distinct_count = bag.size();  // Update the distinct count
        } else if (type == 2) {
            int x;
            cin >> x;
            bag.erase(x);  // Remove from the set, set automatically handles existence
            distinct_count = bag.size();  // Update the distinct count
        } else if (type == 3) {
            cout << distinct_count << endl;
        }
    }

    return 0;
}
```
