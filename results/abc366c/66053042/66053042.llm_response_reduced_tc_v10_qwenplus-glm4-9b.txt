#include <iostream>
#include <unordered_set>

int main() {
    std::unordered_set<int> balls;
    int Q;
    std::cin >> Q;

    for (int i = 0; i < Q; ++i) {
        int type;
        std::cin >> type;

        if (type == 1) {
            int x;
            std::cin >> x;
            balls.insert(x);
        } else if (type == 2) {
            int x;
            std::cin >> x;
            balls.erase(x);
        } else if (type == 3) {
            std::cout << balls.size() << std::endl;
        }
    }

    return 0;
}
