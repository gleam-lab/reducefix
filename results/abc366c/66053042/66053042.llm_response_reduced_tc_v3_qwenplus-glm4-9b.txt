#include <iostream>
#include <map>

int main() {
    std::map<int, int> m;

    int Q;
    std::cin >> Q;
    while (Q--) {
        int type;
        std::cin >> type;

        if (type == 1) {
            int x;
            std::cin >> x;
            m[x]++;
        } else if (type == 2) {
            int x;
            std::cin >> x;
            m[x]--;
            if (m[x] == 0) m.erase(x);
        } else if (type == 3) {
            std::cout << m.size() << std::endl;
        }
    }

    return 0;
}
