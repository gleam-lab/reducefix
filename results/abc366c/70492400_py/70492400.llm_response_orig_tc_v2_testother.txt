```python
q = int(input())
bag = {}
for i in range(q):
    query = [int(a) for a in input().split()]
    
    if query[0] == 1:
        x = query[1]
        bag[x] = bag.get(x, 0) + 1
    elif query[0] == 2:
        x = query[1]
        bag[x] -= 1
        if bag[x] == 0:
            del bag[x]
    else:
        print(len(bag))
```
