```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int q;
    cin >> q;
    set<int> st;
    for (int i = 0; i < q; i++) {
        int a;
        cin >> a;
        if (a == 1) {
            int x;
            cin >> x;
            st.insert(x);
        } else if (a == 2) {
            int y;
            cin >> y;
            st.erase(st.find(y));
        } else if (a == 3) {
            cout << st.size() << endl;
        }
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif

    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }

    return 0;
}
```

This solution uses a `set` to store the unique numbers in the bag. The `insert` operation adds a number to the set, ensuring all elements are unique. The `erase` operation removes a specific element from the set using `find`, which is efficient as `set` maintains its elements sorted and provides logarithmic time complexity for find operations. The `size` method of the `set` gives the count of unique elements, which directly answers the third type of query. This approach ensures correct handling of all specified constraints and guarantees efficient operations even for large inputs.
