The issue with the original code is that it uses a `set` to track the unique integers in the bag, but it doesn't account for multiple occurrences of the same integer. When we remove a ball with integer `x`, we need to ensure that there are no other balls with the same integer `x` left in the bag before removing `x` from the set of unique integers.

To fix this, we need to maintain a count of how many balls there are for each integer. We can use a `map` or `unordered_map` to keep track of these counts, and only remove an integer from the set when its count reaches zero.

#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int q;

    unordered_map<int, int> count;
    set<int> unique;
    
    for (int i = 0; i < q; i++)
    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            count[x]++;
            if (count[x] == 1) {
                unique.insert(x);
            }
        }
        else if (a == 2)
        {
            int x;
            cin >> x;
            count[x]--;
            if (count[x] == 0) {
                unique.erase(x);
            }
        }
        else if (a == 3)
        {
            cout << unique.size() << endl;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t = 1;
    while (t--)
    {
        solve();
    }
}

3. Modified the removal logic to only remove from the set when the count reaches 0 (last occurrence).
