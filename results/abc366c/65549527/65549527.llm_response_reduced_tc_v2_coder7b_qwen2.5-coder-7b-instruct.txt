```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
	int q;
	cin >> q;
	set<int> st;
	for (int i = 0; i < q; i++)
	{
		int a;
		cin >> a;
		if (a == 1)
		{
			int x;
			cin >> x;
			st.insert(x);
		}
		else if (a == 2)
		{
			int y;
			cin >> y;
			st.erase(st.find(y));
		}
		else if (a == 3)
		{
			cout << st.size() << endl;
		}
	}
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif

	int t = 1;
	// cin >> t;
	while (t--)
	{
		solve();
	}
}
```

In the original code, when handling the second type of query (`2 x`), the `erase` method was directly called without finding the element in the set first. This could lead to undefined behavior if the element was not found. The corrected version uses `st.find(y)` to find the element before erasing it, ensuring safety and correct functionality.
