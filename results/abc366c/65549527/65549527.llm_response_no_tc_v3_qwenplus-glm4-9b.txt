#include<bits/stdc++.h>
using namespace std;

void solve() {
    int q;
    cin >> q;
    set<int> st; // Using set to keep track of unique ball numbers
    for (int i = 0; i < q; i++) {
        int a;
        cin >> a;
        if (a == 1) { // Insert a new ball with number x
            int x;
            cin >> x;
            st.insert(x);
        } else if (a == 2) { // Remove a ball with number x
            int x;
            cin >> x;
            st.erase(x);
        } else if (a == 3) { // Output the number of unique balls
            cout << st.size() << endl;
        }
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif

    int t = 1;
    //cin >> t;
    while (t--) {
        solve();
    }
}
