```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
	int q;
	cin >> q;
	unordered_set<int>st;
	for (int i = 0; i < q; i++)
	{
		int a;
		cin >> a;
		if (a == 1)
		{
			int x;
			cin >> x;
			st.insert(x);
		}
		else if (a == 2)
		{
			int y;
			cin >> y;
			st.erase(st.find(y));
		}
		else if (a == 3)
		{
			cout << st.size() << endl;
		}
	}

}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif

	int t = 1;
	//cin >> t;
	while (t--)
	{
		solve();
	}
}
```

**Explanation:**
- The original code used a `set` for storing the elements, which does not allow efficient removal of elements based on value. A `set::erase` operation with a value requires linear time complexity, making it inefficient for large inputs.
- The corrected code uses an `unordered_set`, which provides average constant time complexity for insertions, deletions, and lookups, thus making it more suitable for this problem with constraints up to \(2 \times 10^5\) operations.
