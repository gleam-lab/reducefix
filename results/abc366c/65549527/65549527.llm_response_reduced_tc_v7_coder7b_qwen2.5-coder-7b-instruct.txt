{

    set<int> st;
    for (int i = 0; i < q; i++)
    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            st.insert(x);
        }
        else if (a == 2)
        {
            int y;
            cin >> y;
            st.erase(st.find(y));
        }
        else if (a == 3)
        {
            cout << st.size() << endl;
        }
    }
}

Explanation:
- The issue was that the erase operation in the original code was incorrect. Instead of using `st.erase(y);`, we should use `st.erase(st.find(y));` to ensure that the correct element is erased.
- This change ensures that the erase operation works correctly, even when multiple elements with the same value exist in the set.
