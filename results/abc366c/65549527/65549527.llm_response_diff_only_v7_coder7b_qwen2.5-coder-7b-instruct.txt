```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int q;
    cin >> q;
    unordered_set<int> st;
    for (int i = 0; i < q; i++)
    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            st.insert(x);
        }
        else if (a == 2)
        {
            int y;
            cin >> y;
            st.erase(y);
        }
        else if (a == 3)
        {
            cout << st.size() << endl;
        }
    }
}

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif

    int t = 1;
    //cin >> t;
    while (t--)
    {
        solve();
    }
}
```

### Explanation of Fixes
1. Changed `set<int>` to `unordered_set<int>`: This change ensures that the insert and erase operations run in average constant time, which is crucial given the constraints of the problem. Sets in C++ are implemented as balanced binary search trees, where each insertion and deletion takes O(log n) time, making them less efficient than unordered sets for large datasets.
