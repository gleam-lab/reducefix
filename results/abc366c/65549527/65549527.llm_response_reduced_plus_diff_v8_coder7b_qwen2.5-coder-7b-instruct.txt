    unordered_set<int> s;
    map<int, int> freq;

    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            if (freq[x] == 0)
                s.insert(x);
            freq[x]++;
        }
        else if (a == 2)
        {
            int y;
            cin >> y;
            freq[y]--;
            if (freq[y] == 0)
                s.erase(y);
        }
        else if (a == 3)
        {
            cout << s.size() << endl;
        }
    }
}

### Explanation of Changes:
1. **Use of `unordered_set`**: To store unique elements efficiently.
