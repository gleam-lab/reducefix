```cpp
#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int q;
    cin >> q;
    set<int>st;
    for (int i = 0; i < q; i++)
    {
        int a;
        cin >> a;
        if (a == 1)
        {
            int x;
            cin >> x;
            st.insert(x);
        }
        else if (a == 2)
        {
            int y;
            cin >> y;
            st.erase(st.find(y));

        }
        else if (a == 3)
        {
            cout << st.size() << endl;
        }
    }
}

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif

    int t = 1;
    //cin >> t;
    while (t--)
    {
        solve();
    }
}
```

This solution uses a set to store unique integers efficiently. For each query, it inserts, deletes, or counts the number of unique elements in the set. The operations on the set are efficient due to its logarithmic time complexity for insertions, deletions, and size queries. This ensures that the overall time complexity remains within the required limits.
