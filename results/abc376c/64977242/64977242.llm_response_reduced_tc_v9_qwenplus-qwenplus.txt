```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in ascending order
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Try to fit smallest N-1 toys into the N-1 boxes
    bool possible = true;
    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) {
            possible = false;
            break;
        }
    }

    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    // The largest toy must go into the new box
    long long max_toy = A.back();

    // Find the smallest box size x such that x >= max_toy and x is not among B
    // We need to find the smallest x that is >= max_toy and not in B
    // To do this, we can just find the smallest number >= max_toy that is not in B

    // First, remove duplicates in B
    sort(B.begin(), B.end());
    B.erase(unique(B.begin(), B.end()), B.end());

    // Binary search for the minimum x >= max_toy that is not in B
    long long low = max_toy;
    long long high = 2e18; // Large upper bound
    long long answer = -1;

    while (low <= high) {
        long long mid = (low + high) / 2;
        // Count how many numbers in B are <= mid
        long long count = upper_bound(B.begin(), B.end(), mid) - B.begin();
        // If there are less than N-1 elements <= mid, then there's a gap we can use
        if (count < N - 1) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << answer << endl;
    return 0;
}
```
