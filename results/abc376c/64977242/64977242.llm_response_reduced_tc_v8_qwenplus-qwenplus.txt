```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in ascending order
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can fit N-1 smallest toys into existing boxes
    bool possible = true;
    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) {
            possible = false;
            break;
        }
    }

    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    // The minimum required size for the new box is the maximum of:
    // - The largest toy that couldn't be placed (which is A[N-1] since we haven't checked it yet)
    // - The minimum increase needed in B to accommodate the other toys
    long long min_box_size = A.back();

    // Now compare the largest N-1 toys with the largest N-1 boxes
    // If any toy is bigger than its corresponding box, we need a new box big enough to hold that toy
    for (int i = 0; i < N - 1; ++i) {
        if (A[i + 1] > B[i]) {
            min_box_size = max(min_box_size, A[i + 1]);
            break;
        }
    }

    cout << min_box_size << endl;
    return 0;
}
```
