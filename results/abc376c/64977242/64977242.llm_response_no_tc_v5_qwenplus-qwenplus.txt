```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in ascending order
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can fit the smallest N-1 toys into the N-1 boxes
    bool possible = true;
    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) {
            possible = false;
            break;
        }
    }

    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    // The required size of the additional box is at least A[N-1]
    // But also, it must be larger than any B[i] that is less than corresponding A[i]
    long long min_required_box = 0;
    for (int i = 0; i < N - 1; ++i) {
        if (A[i + 1] > B[i]) {
            min_required_box = max(min_required_box, A[i + 1]);
        }
    }

    // If all A[i+1] <= B[i], then we still need a box for A[0]
    // In this case, any value >= A[0] works. We choose the minimum: A[0]
    if (min_required_box == 0) {
        min_required_box = A[0];
    }

    cout << min_required_box << endl;

    return 0;
}
```
