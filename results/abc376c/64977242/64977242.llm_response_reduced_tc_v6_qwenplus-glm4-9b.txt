
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int N;
vector<long long> A, B;

int main() {
    cin >> N;
    A.resize(N);
    B.resize(N - 1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end(), greater<long long>());

    long long minBoxSize = LLONG_MAX;
    for (int i = 0; i < N; i++) {
        if (B[i] < A[i]) {
            minBoxSize = min(minBoxSize, A[i]);
        }
    }

    if (minBoxSize == LLONG_MAX) {
        cout << -1 << endl;
    } else {
        cout << minBoxSize << endl;
    }

    return 0;
}
```
