```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int N;
vector<int> A, B;

int main() {
    cin >> N;
    A.resize(N);
    B.resize(N-1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    // Sort A in ascending order and B in descending order
    sort(A.begin(), A.end());
    sort(B.rbegin(), B.rend());

    // Check if we can fit all toys into the available boxes
    int maxBoxSize = 0;
    for (int i = 0; i < N - 1; i++) {
        if (A[i] > B[i]) {
            maxBoxSize = A[i];
            break;
        }
    }

    // If no toy is too big for any box, use the largest toy as the box size
    if (maxBoxSize == 0) {
        maxBoxSize = A[N - 1];
    }

    cout << maxBoxSize << endl;

    return 0;
}
```
