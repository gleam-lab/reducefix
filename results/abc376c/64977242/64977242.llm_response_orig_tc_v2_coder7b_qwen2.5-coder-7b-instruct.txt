```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> A(N), B(N - 1);

    for(int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    for(int i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    for(int i = 0; i < N - 1; ++i) {
        if(A[i] > B[i]) {
            cout << "-1" << endl;
            return 0;
        }
    }

    int minBoxSize = A.back();
    for(int i = 0; i < N - 1; ++i) {
        if(B[i] >= A[i + 1]) {
            minBoxSize = max(minBoxSize, B[i]);
        }
    }

    cout << minBoxSize << endl;
    return 0;
}
```
