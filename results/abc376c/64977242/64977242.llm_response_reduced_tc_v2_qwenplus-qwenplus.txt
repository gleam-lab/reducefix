```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in ascending order
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can fit the smallest N-1 toys into the N-1 boxes
    bool possible = true;
    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) {
            possible = false;
            break;
        }
    }

    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    // The needed size of the extra box is at least A[N-1] (largest toy)
    // But also check if there's a smaller toy that cannot go in any of the given boxes
    long long min_required = A.back();

    // Find the first toy that doesn't fit in the largest available box
    int idx = 0;
    while (idx < N - 1 && A[idx] <= B[idx]) {
        idx++;
    }

    // If all N-1 smallest toys fit in the boxes, then the new box must hold the biggest toy
    // Otherwise, one of the small toys needs the new box
    min_required = max(min_required, A[idx]);

    cout << min_required << endl;
    return 0;
}
```
