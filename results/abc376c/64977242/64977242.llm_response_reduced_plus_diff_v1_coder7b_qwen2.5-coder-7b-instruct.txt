#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < n - 1; ++i) cin >> b[i];
    
    sort(a.rbegin(), a.rend()); // Sort in descending order
    sort(b.rbegin(), b.rend()); // Sort in descending order
    
    priority_queue<int> pq; // Max heap to store the sizes of boxes
    
    for(int i = 0; i < n - 1; ++i) {
        pq.push(b[i]);
    }
    
    int idx = 0;
    while(!pq.empty()) {
        int top = pq.top();
        pq.pop();
        
        if(top >= a[idx]) {
            ++idx;
        } else {
            cout << -1 << endl;
            return 0;
        }
    }
    
    cout << a[0] << endl; // The minimum size needed is the largest toy size
    
    return 0;
}

This solution works as follows:

1. We read the number of toys `n` and their sizes `a`, and the number of boxes `n-1` and their sizes `b`.
2. Both arrays `a` and `b` are sorted in descending order to facilitate easy comparison and selection.
3. We use a max heap (priority queue) to keep track of the available box sizes in descending order.
4. We iterate through the sorted toy sizes. For each toy, we check if the largest available box can accommodate it. If yes, we remove that box from the heap and move to the next toy. If no, we output `-1` and exit.
5. If all toys fit into the available boxes, we output the size of the largest toy, which is the minimum size required to ensure all toys can be stored.
