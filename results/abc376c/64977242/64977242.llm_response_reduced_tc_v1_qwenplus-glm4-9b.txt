#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    vector<long long> A, B;

    cin >> N;
    A.resize(N);
    B.resize(N - 1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Find the maximum size of the toy that can fit into the smallest box
    long long minBoxSize = A[0];
    long long maxFitToySize = 0;
    for (int i = 0; i < N - 1; i++) {
        if (A[i] > B[i]) {
            minBoxSize = max(minBoxSize, A[i]);
            maxFitToySize = max(maxFitToySize, A[i]);
        }
    }

    // The smallest box size that can store all toys is at least 1
    // It also needs to be large enough to store the largest toy that does not fit
    // in any of the existing boxes B.
    if (maxFitToySize == 0) {
        // All toys fit in the existing boxes and there is no need for an additional box
        cout << -1 << endl;
    } else {
        long long minBoxSizeNeeded = maxFitToySize + 1;
        cout << (minBoxSize >= minBoxSizeNeeded ? minBoxSizeNeeded : -1) << endl;
    }

    return 0;
}
