#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> sizes(N, 0);
    
    for (int i = 0; i < N; i++) {
        cin >> sizes[i];
    }
    
    // Split the sizes into toys and boxes
    vector<long long> A(sizes.begin(), sizes.begin() + N);
    vector<long long> B(sizes.begin() + N, sizes.end());
    
    // Sort the toy sizes in descending order
    sort(A.begin(), A.end(), greater<long long>());
    // Sort the box sizes in descending order
    sort(B.begin(), B.end(), greater<long long>());
    
    // If A[0] is larger than B[0] (which should be the case after sorting), it means we cannot find a box
    // that can contain the largest toy. In this case, the minimum box size x should be A[0] + 1.
    // If A[0] is smaller or equal to B[0], we need to check if we can fit all toys in the smallest box B[0].
    // If we can fit all toys, then the answer is B[0], otherwise, it's -1.
    if (A[0] > B[0]) {
        cout << A[0] + 1 << endl;
    } else {
        // Find the first toy that cannot fit in the smallest box
        for (int i = 0; i < N; i++) {
            if (A[i] > B[0]) {
                cout << -1 << endl;
                return 0;
            }
        }
        cout << B[0] << endl;
    }

    return 0;
}
