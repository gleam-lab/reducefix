#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort the sizes of toys and boxes
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Create a combination of sizes where each toy is paired with the minimum possible box size
    vector<pair<long long, long long>> pairs(N);
    for (int i = 0; i < N; ++i) {
        pairs[i] = {A[i], B[i]};
    }

    // Sort the pairs by the toy size to align the largest toys with the smallest box sizes
    sort(pairs.begin(), pairs.end());

    // Check the maximum size of the boxes we need to accommodate the largest toy
    long long maxBoxSize = 0;
    for (int i = 0; i < N; ++i) {
        maxBoxSize = max(maxBoxSize, pairs[i].second);
    }

    // Also align the boxes with the largest toy to ensure they are not larger than necessary
    for (int i = 0; i < N; ++i) {
        maxBoxSize = max(maxBoxSize, pairs[i].first);
    }

    // Output the maximum box size we need
    cout << maxBoxSize << endl;

    return 0;
}
