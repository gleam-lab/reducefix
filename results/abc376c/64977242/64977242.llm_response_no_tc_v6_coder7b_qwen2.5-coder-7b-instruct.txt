#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int N;
vector<long long> A, B;

int main() {
    cin >> N;
    A.resize(N);
    B.resize(N-1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    bool canStore = true;
    for (int i = 0; i < N - 1; i++) {
        if (A[i] > B[i]) {
            canStore = false;
            break;
        }
    }

    if (!canStore) {
        cout << -1 << endl;
        return 0;
    }

    // The minimum box size needed should be at least the largest toy size
    long long minBoxSize = *max_element(A.begin(), A.end());

    // Check if the largest box already meets the requirement
    if (minBoxSize >= *max_element(B.begin(), B.end())) {
        cout << minBoxSize << endl;
        return 0;
    }

    // If not, we need to find the next smallest box that can accommodate the largest toy
    for (long long size = minBoxSize + 1; ; ++size) {
        bool valid = true;
        for (int i = 0; i < N - 1 && valid; ++i) {
            if (A[i] > size) {
                valid = false;
            } else {
                B[i] = size;
            }
        }
        if (valid) {
            cout << size << endl;
            return 0;
        }
    }

    return 0;
}
