```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find which toy cannot be placed in the current boxes
    // and determine the minimal x such that adding a box of size x makes it possible.

    int b_idx = 0;
    vector<long long> unplaced;

    // Try to assign smallest toys to smallest available boxes
    for (int i = 0; i < N; ++i) {
        if (b_idx < N - 1 && A[i] <= B[b_idx]) {
            b_idx++;
        } else {
            unplaced.push_back(A[i]);
        }
    }

    if (unplaced.empty()) {
        cout << 0 << endl;
        return 0;
    }

    // Now we must place all unplaced toys using one additional box.
    // The size of this box must be at least as big as the largest unplaced toy.
    cout << unplaced.back() << endl;
    return 0;
}
```
