#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in ascending order
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if it's possible to place all toys using some additional box
    bool possible = true;
    vector<long long> available_boxes = B;
    // We will use the smallest N-1 boxes to fit N-1 of the toys
    // The largest toy may go into the biggest box or we may swap one toy to the new box

    // Try greedy matching: assign smallest boxes to smallest toys
    // Except one toy that will go into the purchased box
    int j = 0;
    for (int i = 0; i < N; ++i) {
        if (j < N - 1 && A[i] <= B[j]) {
            ++j;
        }
        // Else, this toy must go into the purchased box
    }

    if (j >= N - 1) {
        // We could fit N-1 toys into the existing boxes
        cout << A[N - 1] << endl;
        return 0;
    }

    // Otherwise try to find minimal x such that we can put all toys
    // Binary search on x
    sort(B.begin(), B.end());
    sort(A.begin(), A.end());

    long long low = 1, high = 1e18, answer = -1;
    while (low <= high) {
        long long mid = (low + high) / 2;

        // Try to assign toys to boxes + one box of size mid
        vector<long long> C = B;
        C.push_back(mid);
        sort(C.begin(), C.end());

        int ptr = 0;
        for (int i = 0; i < N; ++i) {
            if (ptr < C.size() && A[i] <= C[ptr]) {
                ++ptr;
            }
        }

        if (ptr == C.size()) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << answer << endl;
    return 0;
}
