#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    // Sort the toy sizes in descending order to consider the largest toy size
    sort(A.begin(), A.end(), greater<long long>());
    // Sort the box sizes in ascending order to consider the smallest existing box size
    sort(B.begin(), B.end());

    // Check if it's already possible to fit all toys in existing boxes
    bool isPossible = true;
    for (int i = 0; i < N && isPossible; i++) {
        if (A[i] > B[i]) {
            isPossible = false;
        }
    }

    if (isPossible) {
        // If all toys can fit in existing boxes, the smallest box size needed is 1
        cout << 1 << endl;
    } else {
        // Find the smallest box size x that allows all toys to be stored
        long long minBoxSize = 0;
        for (int i = 0; i < N; i++) {
            // Update the minimum box size needed by the largest toy that cannot fit in any existing box
            minBoxSize = max(minBoxSize, A[i] - B[i]);
        }
        cout << minBoxSize << endl;
    }

    return 0;
}
