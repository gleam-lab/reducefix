#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> A(N), B(N-1);
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    // Sort both arrays
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Calculate the prefix sums of B
    partial_sum(B.begin(), B.end(), B.begin());

    // Find the minimum x such that we can fit all toys into boxes
    for (int x = 1; x <= A[N-1]; x++) {
        if (x >= A[0]) {
            bool canFit = true;
            for (int i = 0; i < N-1 && canFit; i++) {
                if (B[i] + A[i+1] > x) {
                    canFit = false;
                }
            }
            if (canFit) {
                cout << x << endl;
                return 0;
            }
        }
    }

    // If no valid x found, output -1
    cout << -1 << endl;
    return 0;
}
