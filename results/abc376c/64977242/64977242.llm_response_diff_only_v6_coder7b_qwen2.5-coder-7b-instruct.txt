```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> A(N), B(N-1);

    // Read input values
    for (int i = 0; i < N; i++) cin >> A[i];
    for (int i = 0; i < N - 1; i++) cin >> B[i];

    // Sort the arrays
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can place all toys without any box being too small
    bool canPlace = true;
    for (int i = 0; i < N - 1; i++) {
        if (A[i] > B[i]) {
            canPlace = false;
            break;
        }
    }

    // If we cannot place all toys with the current boxes, output -1
    if (!canPlace) {
        cout << -1 << endl;
        return 0;
    }

    // Find the maximum size needed among the remaining toys after placing them in available boxes
    int maxNeededSize = 0;
    for (int i = N - 1; i >= 0; --i) {
        if (i + 1 < N && A[i] <= B[i]) continue; // This toy fits in the current box
        if (maxNeededSize < A[i]) maxNeededSize = A[i]; // Update the max size needed
    }

    // The minimum size of the new box should be at least the maxNeededSize
    cout << maxNeededSize << endl;

    return 0;
}
```
