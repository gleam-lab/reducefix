```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> toys(A.begin(), A.end());
    multiset<long long> boxes(B.begin(), B.end());

    // Try to fit the smallest toys into the smallest boxes
    for (auto box = boxes.begin(); box != boxes.end(); ++box) {
        auto toy = toys.lower_bound(*box);
        if (toy != toys.end()) {
            toys.erase(toy);
        } else {
            // If we can't place a toy in this box, we must use our custom box here
            // The required size is at least the smallest remaining toy
            cout << *toys.begin() << "\n";
            return 0;
        }
    }

    // If we've made it here, all N-1 boxes have been used successfully
    // We would not need to buy an extra box
    cout << -1 << "\n";

    return 0;
}
```
