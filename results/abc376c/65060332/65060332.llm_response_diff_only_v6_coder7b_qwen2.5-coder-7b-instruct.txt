```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;
ll a[N], b[N];

int main() {
    int n;
    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for(int i = 1; i < n; i++) {
        cin >> b[i];
    }

    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    int j = n - 1;
    for(int i = n; i >= 1; i--) {
        if(j < 1 || b[j] < a[i]) {
            cout << -1 << endl;
            return 0;
        } else {
            j--;
        }
    }

    cout << a[n] << endl;

    return 0;
}
```

1. **Variable Type**: Changed `ll` to `int` since the constraints specify that \( N \) and other variables will not exceed \( 2 \times 10^5 \), which fits within the range of `int`.
2. **Edge Case Handling**: Added a check at the beginning of the loop where we iterate through toys from largest to smallest. If there isn't a big enough box for the current toy, output `-1` immediately.
3. **Output**: Directly output the size of the last toy after sorting, as this ensures that there is always a place for it with the smallest possible box.
