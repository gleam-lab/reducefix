#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N],b[N];
int main(){
	ll n,cnt=0,mx=0;
	cin>>n;
	for(ll i=1;i<=n;i++){
		cin>>a[i];
	}
	for(ll i=1;i<n;i++){
		cin>>b[i];
	}
	sort(a+1,a+n+1);
	sort(b+1,b+n);
	ll l=n,r=n-1;
	while(l>=1 && r>=1){
		if(b[r]>=a[l]){
			r--;
			l--;
		}
		else{
			cnt++;
			mx=max(mx,a[l]);
			l--;
		}
	}
	if(cnt==0){
		cout<<-1;
		return 0;
	}
	cout<<mx;
	return 0;
}

### Explanation
The issue with your original code was that you were checking `cnt > 1` at the end. This means if there is even one box left over after placing all toys, you would output `-1`. However, according to the problem statement, Takahashi wants to ensure that all toys fit into boxes without any leftovers. Therefore, the correct condition should be `cnt == 0`, meaning all toys have been successfully placed into boxes. If this condition is met, we output `-1`, indicating that no additional box is needed. Otherwise, we output the maximum size of the toys that couldn't fit into the available boxes.
