Here is the corrected C++ code:

```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N],b[N];
int main(){
    ll n,cnt=0,mx=0;
    cin>>n;
    for(ll i=1;i<=n;i++){
        cin>>a[i];
    }
    for(ll i=1;i<n;i++){
        cin>>b[i];
    }
    sort(a+1,a+n+1);
    sort(b+1,b+n);
    ll l=n,r=n-1;
    while(l>=1 && r>=1){
        if(b[r]>=a[l]){
            r--;
            l--;
        }
        else{
            cnt++;
            mx=max(mx,a[l]);
            l--;
        }
    }
    if(cnt==1){
        cout<<mx;
    }
    else{
        cout<<-1;
    }
    return 0;
}
```

In this corrected version, I have changed `if(cnt>1)` to `if(cnt==1)`. The reason behind this change is that if cnt is equal to 1 then it means we need to buy a box with size greater than or equal to the maximum size of the toys which could not fit into any of the existing boxes. If cnt is greater than 1 then it means we need to buy more than one box which is not allowed according to the problem statement.
