
```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N = 2e5 + 5;
ll a[N], b[N], c[N];

int main() {
    ll n;
    cin >> n;
    for (ll i = 0; i < n; i++) {
        cin >> a[i];
        c[i] = a[i];
    }
    sort(c, c + n);
    for (ll i = 0; i < n - 1; i++) {
        cin >> b[i];
    }
    ll l = 0, r = n - 1, lo = -1, hi = -1;
    
    while (l <= r) {
        ll mid = (l + r) / 2;
        ll i = 0, j = 0, box = 0;
        while (i < n) {
            if (b[j] >= c[mid] || j >= n - 1) {
                box++;
                j++;
            } else {
                i++;
            }
        }
        if (box == n) {
            lo = mid;
            hi = mid;
            l = mid + 1;
        } else {
            hi = mid;
            r = mid - 1;
        }
    }
    
    if (lo == -1) {
        cout << -1;
    } else {
        cout << c[lo];
    }
    
    return 0;
}
```

### Explanation:
1. The code starts by reading the input values and storing the toy sizes in an array `c` which is sorted.
2. The variable `l` and `r` are set to the low and high bounds for binary search respectively.
3. The binary search is used to find the minimum box size `x` that allows all toys to be placed in boxes.
4. If all toys can be placed with `x = c[mid]`, then `lo` and `hi` are updated and `l` is set to `mid + 1` to search for a potentially smaller valid `x`.
5. If not, `hi` is updated and `r` is set to `mid - 1` to search for a larger valid `x`.
6. The loop continues until `l` exceeds `r`, ensuring the smallest valid `x` is found.
7. If no valid `x` is found (`lo` remains `-1`), `-1` is printed; otherwise, the value of `c[lo]` (the smallest valid `x`) is printed.
