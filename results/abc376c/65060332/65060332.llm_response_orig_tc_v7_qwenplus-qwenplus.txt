```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 2e5 + 5;

ll a[MAXN], b[MAXN];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // We will use binary search to find the minimum x
    // The idea is: can we assign toys to boxes such that all but one toy fit into existing boxes,
    // and one toy fits in our additional box of size x?

    // Use binary search on the answer x
    ll low = 1, high = 1e18;
    ll ans = -1;

    while (low <= high) {
        ll mid = (low + high) / 2;

        // Try to match toys to boxes (including our new one of size mid)
        vector<ll> temp_b(b, b + (n - 1));
        temp_b.push_back(mid);
        sort(temp_b.begin(), temp_b.end());

        bool ok = true;
        for (int i = 0; i < n; ++i) {
            if (a[i] > temp_b[i]) {
                ok = false;
                break;
            }
        }

        if (ok) {
            ans = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << ans << '\n';
    return 0;
}
```
