```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const ll N = 2e5 + 5;
ll a[N], b[N];

int main() {
    ll n;
    cin >> n;
    for (ll i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (ll i = 1; i < n; i++) {
        cin >> b[i];
    }

    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    ll mx = 0;
    for (ll i = 1; i <= n; i++) {
        if (i > n - 1 || b[i] >= a[i]) {
            mx = max(mx, a[i]);
        } else {
            cout << -1 << endl;
            return 0;
        }
    }

    cout << mx << endl;
    return 0;
}
```
