def find_minimum_box_size(n, a, b):
    # Find the maximum size among all toys and existing boxes
    max_toy_size = max(a)
    max_box_size = max(b)

    # If the largest toy can fit in the largest box, no additional box is needed
    if max_toy_size <= max_box_size:
        return -1

    # Binary search to find the smallest size of the extra box needed
    low, high = 0, max_toy_size + 1  # start from 1 to include max_toy_size
    while low < high:
        mid = (low + high) // 2
        # Check if mid size box is sufficient
        if any(toy > mid for toy in a):
            low = mid + 1
        else:
            high = mid

    return low

# Input reading
n = int(input())
a = list(map(int, input().split()))
b = list(map(int, input().split()))

# Function call and output
result = find_minimum_box_size(n, a, b)
print(result)
