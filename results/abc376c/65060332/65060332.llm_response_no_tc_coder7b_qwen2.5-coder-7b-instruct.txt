```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N], b[N];

int main(){
    ll n, mx = 0;
    cin >> n;
    for(ll i = 1; i <= n; i++){
        cin >> a[i];
    }
    for(ll i = 1; i < n; i++){
        cin >> b[i];
    }
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);
    
    priority_queue<ll, vector<ll>, greater<ll>> pq;
    for(ll i = 1; i < n; i++){
        pq.push(b[i]);
    }
    
    for(ll i = 1; i <= n; i++){
        if(!pq.empty() && pq.top() >= a[i]){
            pq.pop();
        } else {
            mx = max(mx, a[i]);
        }
    }
    
    cout << (mx ? mx : -1) << endl;
    return 0;
}
```
