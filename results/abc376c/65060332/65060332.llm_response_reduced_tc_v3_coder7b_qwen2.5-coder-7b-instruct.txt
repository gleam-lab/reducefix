```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N],b[N];
int main(){
    ll n;
    cin>>n;
    for(ll i=1;i<=n;i++){
        cin>>a[i];
    }
    for(ll i=1;i<n;i++){
        cin>>b[i];
    }
    sort(a+1,a+n+1);
    sort(b+1,b+n);
    ll l=1,r=n-1,x=-1;
    while(l<=r){
        ll mid=(l+r)/2;
        vector<ll> extra;
        bool valid=true;
        for(ll i=1;i<=n;i++){
            if(i<=mid){
                if(a[i]>b[i-mid]){
                    valid=false;
                    break;
                }
            }
            else{
                extra.push_back(a[i]);
            }
        }
        if(valid){
            x=b[mid];
            r=mid-1;
        }
        else{
            l=mid+1;
        }
    }
    cout<<x<<endl;
    return 0;
}
```
