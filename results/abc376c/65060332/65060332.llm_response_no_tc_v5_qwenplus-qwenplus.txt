#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll n;
    cin >> n;

    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) {
        cin >> A[i];
    }
    for (ll i = 0; i < n - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Try to assign largest toys to largest boxes
    ll extra_toy = -1;
    int a_idx = n - 1, b_idx = n - 2;

    while (a_idx >= 0 && b_idx >= 0) {
        if (B[b_idx] >= A[a_idx]) {
            --a_idx;
            --b_idx;
        } else {
            if (extra_toy == -1) {
                extra_toy = A[a_idx];
                --a_idx;
            } else {
                cout << -1 << '\n';
                return 0;
            }
        }
    }

    // If there are still toys left and we already used the extra box
    while (a_idx >= 0) {
        if (extra_toy == -1) {
            extra_toy = A[a_idx];
        } else {
            cout << -1 << '\n';
            return 0;
        }
        --a_idx;
    }

    cout << extra_toy << '\n';
    return 0;
}
