#define ll long long

const ll N = 2e5 + 5;
ll a[N], b[N];

    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // Find the smallest box size that can be used for the largest toys
    for(ll i = n - 1; i >= 0; --i) {
        if (a[i] <= b[i]) {
            x = min(x, b[i]);
        } else {
            break;
        }
    }

    if (x == LLONG_MAX) {
        cout << -1;

    return 0;
}
