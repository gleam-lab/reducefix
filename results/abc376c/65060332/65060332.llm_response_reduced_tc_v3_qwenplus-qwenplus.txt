```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;

    ll a[N], b[N];
    for (ll i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (ll i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // Try to fit largest toys into largest boxes
    ll required_toys = 0;
    ll j = n - 1; // pointer for boxes

    // We try to fit the biggest toys into the biggest boxes first
    for (ll i = n - 1; i >= 0 && j >= 0; --i) {
        if (b[j] >= a[i]) {
            j--;
        } else {
            // This toy can't go in any available box, needs our custom box
            required_toys++;
        }
    }

    if (required_toys > 1) {
        cout << -1 << endl;
    } else {
        // If we need one toy in our custom box, its size is the size of that toy
        // Otherwise, we can use the smallest unused toy (though all are used)
        // Actually, we might have multiple candidates due to ties
        // We should check which toy cannot be placed in any box
        // Let's identify the specific toy that couldn't be placed

        // Reset and find exactly which toy can't be placed
        vector<ll> a_sorted(a, a + n);
        vector<ll> b_sorted(b, b + n - 1);
        vector<bool> used(n, false);

        j = n - 2; // index for boxes (n-1 total boxes)
        for (ll i = n - 1; i >= 0 && j >= 0; --i) {
            if (b_sorted[j] >= a_sorted[i]) {
                used[i] = true;
                j--;
            }
        }

        ll result = -1;
        for (ll i = 0; i < n; ++i) {
            if (!used[i]) {
                if (result == -1 || a_sorted[i] < result) {
                    result = a_sorted[i];
                }
            }
        }

        cout << result << endl;
    }

    return 0;
}
```
