```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 200005;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We try to fit the largest toys into the available boxes.
    // The idea is to greedily match the largest toy with the largest available box.
    // If a toy cannot go into any box, we will need to buy a new box for it.

    ll need_x = -1;
    ll j = n - 2; // index for B

    for (ll i = n - 1; i >= 0; --i) {
        if (j >= 0 && B[j] >= A[i]) {
            // This toy can go into a box
            j--;
        } else {
            // Need to buy a box for this toy
            if (need_x != -1) {
                // Already one toy needs the purchased box => impossible
                cout << -1 << "\n";
                return 0;
            }
            need_x = A[i];
        }
    }

    if (need_x == -1) {
        // All toys were matched to existing boxes, so x can be as small as possible (e.g., 1)
        // But we must return the minimum valid x. Since all toys fit in existing boxes,
        // technically x could be anything ≥ some unused toy size. However, we are asked
        // for the smallest x such that the arrangement is possible. If all toys are placed
        // in existing boxes, then Takahashi doesn't need to use his bought box at all.
        // Therefore, he can choose x arbitrarily, and the minimal x is 1.

        // So output 1 or check if any toy is unassigned
        cout << "1\n";
    } else {
        cout << need_x << "\n";
    }

    return 0;
}
```
