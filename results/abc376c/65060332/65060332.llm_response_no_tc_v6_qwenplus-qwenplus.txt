```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 200005;

ll a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;

    for (ll i = 0; i < n; ++i) {
        cin >> a[i];
    }

    for (ll i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // We will use two pointers to match toys to boxes
    ll toy_ptr = 0;
    ll box_ptr = 0;
    ll need_x = -1;

    while (toy_ptr < n && box_ptr < n - 1) {
        if (b[box_ptr] >= a[toy_ptr]) {
            toy_ptr++;
            box_ptr++;
        } else {
            // This toy needs the extra box
            if (need_x == -1) {
                need_x = a[toy_ptr];
            } else {
                // More than one toy can't fit => impossible
                cout << -1 << endl;
                return 0;
            }
            toy_ptr++;
        }
    }

    // If there's one toy left, it must go into the new box
    if (toy_ptr < n) {
        if (need_x == -1 || a[toy_ptr] > need_x) {
            need_x = a[toy_ptr];
        }
    }

    cout << need_x << endl;
    return 0;
}
```
