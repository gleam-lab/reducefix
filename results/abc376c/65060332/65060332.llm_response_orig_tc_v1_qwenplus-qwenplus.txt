```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

ll a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // We need to place all toys in boxes. One toy per box.
    // We can use the smallest N-1 toys with the largest N-1 boxes.
    // The largest toy must go into the new box whose size we want to minimize.

    // So, try to match smallest toys with largest boxes greedily
    int j = n - 2; // pointer for boxes (0-based)
    for (int i = 0; i < n && j >= 0; ++i) {
        if (a[i] <= b[j]) {
            j--;
        } else {
            // cannot fit this toy in any remaining box
            // this will be the one that goes into our purchased box
            cout << a[i] << "\n";
            return 0;
        }
    }

    // If we reach here, all but one toy could be placed in boxes
    // Then the last one must be put in our box
    cout << a[n - 1] << "\n";

    return 0;
}
```
