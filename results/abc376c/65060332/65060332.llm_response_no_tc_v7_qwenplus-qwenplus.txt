#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    ll n;
    cin >> n;
    
    vector<ll> A(n), B(n - 1);
    
    for (ll i = 0; i < n; ++i) {
        cin >> A[i];
    }
    
    for (ll i = 0; i < n - 1; ++i) {
        cin >> B[i];
    }
    
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());
    
    // Try to match toys to boxes greedily
    ll j = 0;
    ll extra_box_needed = -1;
    
    for (ll i = 0; i < n; ++i) {
        if (j < n - 1 && B[j] >= A[i]) {
            j++;
        } else {
            if (extra_box_needed == -1) {
                extra_box_needed = A[i];
            } else {
                cout << -1 << '\n';
                return 0;
            }
        }
    }
    
    if (extra_box_needed == -1) {
        // All toys fit in existing boxes, so we can choose x = 0 or minimal unassigned
        // But per problem statement, x must be positive
        // So we still need a box, even if all toys matched
        // We can use the largest unmatched toy as lower bound
        // Since one box is unused, find smallest size that cannot be matched (already all matched)
        // In this case, any x will work, pick minimal x which is not used (but all are used)
        // Actually, we still need to pick the smallest toy that was matched last
        // We can safely pick x = 1 as it's arbitrary and only one extra box
        cout << 1 << '\n';
    } else {
        cout << extra_box_needed << '\n';
    }

    return 0;
}
