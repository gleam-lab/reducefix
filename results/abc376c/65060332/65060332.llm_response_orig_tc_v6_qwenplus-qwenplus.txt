```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

ll a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // We need to assign toys to boxes such that each box has exactly one toy,
    // and only one toy can go into the new box of size x.

    // The idea:
    // We try to match largest toys with largest boxes.
    // If we cannot fit a large toy in the largest available box, we must put it in the new box.

    ll required_x = 0;
    int j = n - 2; // pointer for boxes (sorted descending)

    for (int i = n - 1; i >= 0; --i) {
        if (j >= 0 && b[j] >= a[i]) {
            // Box can fit this toy, use it
            j--;
        } else {
            // Cannot fit in any existing box, must put in new box
            required_x = max(required_x, a[i]);
        }
    }

    if (required_x == 0) {
        // No toy needed the new box => any x works, but problem says "purchase one box", so return minimum possible
        // But according to problem statement, there should be at least one toy in new box.
        // So we find the smallest toy that couldn't be matched if all were matched
        cout << -1;
    } else {
        cout << required_x;
    }

    return 0;
}
```
