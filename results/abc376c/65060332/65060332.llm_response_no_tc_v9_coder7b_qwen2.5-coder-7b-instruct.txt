```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N = 2e5 + 5;
ll a[N], b[N];

int main() {
    ll n;
    cin >> n;
    for (ll i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (ll i = 1; i < n; i++) {
        cin >> b[i];
    }
    
    // Sort the sizes of toys and boxes
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    // Initialize variables to track the number of extra boxes needed and the maximum toy size
    ll cnt = 0, mx = 0;

    // Iterate through the sorted list of toys and boxes
    for (ll i = 1; i <= n; i++) {
        if (cnt > 0) {
            // If we need an extra box, use it for the current toy
            mx = max(mx, a[i]);
        } else {
            // Check if the current box can fit the current toy
            if (i < n && b[i] >= a[i]) {
                continue;
            } else {
                // If not, we need an extra box
                cnt++;
                mx = max(mx, a[i]);
            }
        }
    }

    // If we need more than one extra box, it's impossible to place all toys
    if (cnt > 1) {
        cout << -1;
    } else {
        cout << mx;
    }

    return 0;
}
```
