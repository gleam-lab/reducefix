```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 200005;

ll a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;

    for (ll i = 1; i <= n; ++i) {
        cin >> a[i];
    }

    for (ll i = 1; i < n; ++i) {
        cin >> b[i];
    }

    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    ll need = 0;

    // Try to fit largest toys into largest boxes
    ll j = n - 1;
    for (ll i = n; i >= 1 && j >= 1;) {
        if (b[j] >= a[i]) {
            --i;
            --j;
        } else {
            // This toy cannot fit in any available box, we will need our custom box
            need = max(need, a[i]);
            --i;
        }
    }

    if (need == 0) {
        // All toys fit into existing boxes
        cout << 1;
    } else {
        cout << need;
    }

    return 0;
}
```
