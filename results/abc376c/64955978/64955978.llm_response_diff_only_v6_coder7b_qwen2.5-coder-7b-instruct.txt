```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> a(N), b(N - 1);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < N - 1; i++) {
        cin >> b[i];
    }

    // Sort toys and boxes in ascending order
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Check if we can place each toy in a box of at least its size
    for (int i = 0; i < N; i++) {
        if (i == N - 1) { // Last toy, check against the largest box
            if (a[i] > b.back()) {
                cout << -1 << endl;
                return;
            }
        } else {
            if (a[i] > b[i]) {
                cout << -1 << endl;
                return;
            }
        }
    }

    // The minimum size of the box needed is the size of the last toy
    cout << a[N - 1] << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
```

### Explanation of Fixes:
1. **Sorting**: Changed sorting to `ascending` order instead of `descending`. This is because we need to ensure that each toy fits into a box of at least its size.
2. **Edge Case Handling**: Added checks to handle the case where the last toy cannot fit into any available box.
3. **Efficiency**: Utilized `vector` instead of arrays for better memory management and flexibility.
4. **Readability**: Improved readability by using `vector` and standard library functions.
