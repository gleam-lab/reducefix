#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> A(N), B(N - 1);

    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in descending order
    sort(A.begin(), A.end(), greater<ll>());
    sort(B.begin(), B.end(), greater<ll>());

    // We will use binary search to find the minimum x
    ll left = 1, right = 1e18, answer = -1;

    // Binary search on the value of x (size of additional box)
    while (left <= right) {
        ll mid = (left + right) / 2;
        vector<ll> tempB = B;
        tempB.push_back(mid);
        sort(tempB.begin(), tempB.end(), greater<ll>());

        bool possible = true;
        for (ll i = 0; i < N; ++i) {
            if (A[i] > tempB[i]) {
                possible = false;
                break;
            }
        }

        if (possible) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    cout << answer << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
