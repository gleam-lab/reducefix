#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> box_sizes;
    int j = 0;
    long long min_required = -1;

    for (int i = 0; i < N; ++i) {
        // Move j to find a box >= A[i]
        while (j < N - 1 && B[j] < A[i]) {
            box_sizes.insert(B[j]);
            ++j;
        }

        if (j < N - 1) {
            // Use the smallest available box that fits A[i]
            auto it = box_sizes.lower_bound(A[i]);
            if (it != box_sizes.end()) {
                box_sizes.erase(it);  // Used this box
            } else {
                // Need to leave this toy for the new box
                if (min_required == -1) {
                    min_required = A[i];
                }
            }
        } else {
            // All boxes have been considered, need new box
            if (min_required == -1) {
                min_required = A[i];
            }
        }
    }

    if (min_required != -1) {
        cout << min_required << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
