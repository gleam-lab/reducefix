```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    int n; cin >> n;
    vector<long long> a(n), b(n-1);
    
    for(int i = 0; i < n; i++) cin >> a[i];
    for(int i = 0; i < n-1; i++) cin >> b[i];
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    int i = 0, j = 0;
    while(i < n && j < n-1) {
        if(a[i] <= b[j]) {
            i++;
            j++;
        } else {
            j++;
        }
    }
    
    if(i == n) cout << "1";
    else cout << "-1";
    
    return 0;
}
```

1. We read the input values for `n`, `a` and `b`.
2. We sort both arrays `a` and `b`. 
3. We use two pointers `i` and `j` to traverse through both arrays. The pointer `i` will traverse through array `a` and the pointer `j` will traverse through array `b`.
4. For each element in `a`, we check if it can fit into any box in `b`. If it can, we increment both pointers. If it cannot, we just move the pointer `j` forward.
5. After the loop, if `i` is equal to `n`, it means all toys can fit into the boxes, so we print `1`. Otherwise, we print `-1`.
