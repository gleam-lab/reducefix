#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> A(N), B(N - 1);

    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can fit all toys using the N-1 boxes
    ll extra_toy_index = -1;
    ll b_ptr = 0;

    for (ll a : A) {
        if (b_ptr < B.size() && a <= B[b_ptr]) {
            b_ptr++;
        } else {
            // This toy cannot be placed in any existing box
            if (extra_toy_index == -1) {
                extra_toy_index = a;
            } else {
                // We already had one toy that didn't fit, now there's another
                cout << -1 << endl;
                return;
            }
        }
    }

    if (extra_toy_index == -1) {
        // All toys fit in current boxes, so x can be zero (but must buy at least one)
        cout << 1 << endl;
        return;
    }

    // Now find the minimum x such that the extra toy fits in the new box
    // and remaining toys fit in the N-1 existing boxes
    cout << extra_toy_index << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
