```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(int i = 0; i < N; i++) {
        cin >> a[i];
    }
    for(int i = 0; i < N-1; i++) {
        cin >> b[i];
    }

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Check if we can place the largest toy in any box
    if(a.back() > b.back()) {
        cout << -1 << endl;
        return;
    }

    // Find the minimum box size needed to place the largest toy
    ll min_box_size = a.back();
    bool found = false;
    for(ll box : b) {
        if(box >= min_box_size) {
            found = true;
            break;
        }
    }

    if(!found) {
        cout << -1 << endl;
        return;
    }

    cout << min_box_size << endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();

    return 0;
}
```

### Explanation of Fixes:
1. **Vector Usage**: Replaced arrays with vectors for better flexibility and easier management.
2. **Sorting**: Used `std::sort` for sorting both `a` and `b`.
3. **Condition Check**: Simplified the condition check to ensure the largest toy can fit in any box.
4. **Efficiency**: The solution iterates through the sorted list of box sizes once to find the minimum suitable box, which is efficient given the constraints.
5. **Output**: Correctly prints `-1` if no suitable box is found and the minimum box size otherwise.
