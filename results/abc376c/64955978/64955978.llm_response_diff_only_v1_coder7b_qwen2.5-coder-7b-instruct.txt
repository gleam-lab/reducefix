#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"

using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> a(N), b(N - 1);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < N - 1; i++) {
        cin >> b[i];
    }

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Find the minimum box size that can accommodate the largest toy
    ll maxToySize = *max_element(a.begin(), a.end());
    ll minBoxSize = *min_element(b.begin(), b.end());

    if (maxToySize <= minBoxSize) {
        cout << maxToySize << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();

    return 0;
}

3. **Condition Check**: Instead of complex nested loops, the code checks if the largest toy can fit into the smallest available box directly.
4. **Output**: The code prints the result based on whether the largest toy fits into any of the existing boxes or not.
