#include<bits/stdc++.h>
#define ll long long
using namespace std;

    // Read toy sizes
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    // Read box sizes
    for (int i = 0; i < N - 1; ++i) {

    sort(A.rbegin(), A.rend());
    sort(B.rbegin(), B.rend());
    
    ll min_x = LLONG_MAX;
    bool can_store_all = true;
    
    // Try to place each toy in the smallest possible box
    for (int i = 0; i < N; ++i) {
        if (i < N - 1 && A[i] <= B[i]) {
            // If the toy can fit in the existing largest box
            min_x = min(min_x, B[i]);

            // Otherwise, we need to buy a new box of size at least A[i]
            min_x = min(min_x, A[i]);

}
