#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(ll i = 0; i < N; i++) {
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; i++) {
        cin >> b[i];
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Find the maximum size among the toys
    ll maxToySize = *max_element(a.begin(), a.end());

    // Check if we can place the largest toy in any of the available boxes
    bool found = false;
    for(ll i = 0; i < N-1; i++) {
        if(maxToySize <= b[i]) {
            found = true;
            break;
        }
    }

    if(found) {
        cout << maxToySize << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
