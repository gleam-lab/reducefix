    sort(b.begin(), b.end());

    ll minBoxSize = INT_MAX;
    bool found = false;

    for(ll i = 0; i < N-1 && !found; i++){
        if(a.back() <= b[i]) {
            minBoxSize = max(minBoxSize, b[i]);
            found = true;
        } else if(a[i] <= b[i] && a[i+1] > b[i]) {
            minBoxSize = max(minBoxSize, b[i]);
            found = true;
        }
    }

        if(a.back() > b[N-2]) {
            cout << -1 << endl;
            return;
        }
        minBoxSize = a.back();
    }

    cout << minBoxSize << endl;
}
