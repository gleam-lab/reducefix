```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];
    
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> b_set;
    for (ll x : B) b_set.insert(x);

    ll required = -1;

    // Try to place all toys except one in the available boxes
    for (ll exclude = 0; exclude < N; ++exclude) {
        bool ok = true;
        multiset<ll> temp = b_set;

        for (ll i = 0; i < N; ++i) {
            if (i == exclude) continue;
            auto it = temp.lower_bound(A[i]);
            if (it == temp.end()) {
                ok = false;
                break;
            }
            temp.erase(it);
        }

        if (!ok) continue;

        // We can use this configuration, so the purchased box size should be at least A[exclude]
        required = A[exclude];
        break;
    }

    cout << required << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
