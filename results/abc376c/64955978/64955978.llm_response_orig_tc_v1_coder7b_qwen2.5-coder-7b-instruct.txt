```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
ll N;
cin>>N;
vector<ll> a(N), b(N-1);
for(ll i=0;i<N;i++){
  cin>>a[i];
}
for(ll i=0;i<N-1;i++){
  cin>>b[i];
}
sort(a.begin(), a.end());
sort(b.begin(), b.end());
reverse(b.begin(), b.end());

ll j = 0;
for(ll i=0;i<N-1 && j < N-1;){
  if(a[j] <= b[i]) {
    j++;
  } else {
    i++;
  }
}

if(j == N-1){
  cout<<a[j-1]<<endl;
} else {
  cout<<-1<<endl;
}
}

int main(){
ios::sync_with_stdio(0);
cin.tie(0);

 solve();

return 0;
}
```

This solution correctly addresses the problem by ensuring that we find the smallest box size `x` such that all toys can be placed in separate boxes without violating the conditions. It sorts both the toys and the available boxes in descending order and then tries to fit as many toys as possible into the available boxes starting from the largest toy. If it successfully fits all toys, it outputs the size of the last toy placed, otherwise, it outputs `-1`. The time complexity is O(N log N) due to sorting operations.
