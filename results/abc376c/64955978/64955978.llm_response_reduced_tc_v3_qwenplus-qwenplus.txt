#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> box_sizes;
    int j = 0;
    for (int i = 0; i < N; ++i) {
        // Skip toys that exactly fit into current boxes
        while (j < N - 1 && B[j] < A[i]) ++j;
        if (j < N - 1) {
            box_sizes.insert(B[j++]);
        }
    }

    if (box_sizes.size() < N) {
        cout << -1 << endl;
        return 0;
    }

    // Now find the minimum x such that we can assign toys to boxes
    // We have one extra box to add of size x

    sort(A.rbegin(), A.rend()); // descending
    sort(B.rbegin(), B.rend()); // descending

    // Try to greedily assign largest toys to largest available boxes
    multiset<long long> available_boxes;
    for (int i = 0; i < N - 1; ++i) {
        available_boxes.insert(B[i]);
    }

    for (int i = 0; i < N; ++i) {
        auto it = available_boxes.lower_bound(A[i]);
        if (it != available_boxes.end()) {
            available_boxes.erase(it);
        } else {
            // Need a new box of at least A[i]
            cout << A[i] << endl;
            return 0;
        }
    }

    // Should never reach here, because we added one box
    cout << -1 << endl;
    return 0;
}
