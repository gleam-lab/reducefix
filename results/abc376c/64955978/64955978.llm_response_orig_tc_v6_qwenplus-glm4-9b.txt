
The problem asks us to find the smallest box size \( x \) such that it's possible to store all N toys in N boxes, where each toy can only fit into a box not smaller than its size.

The logic behind the problem is that in order to place each toy into a box, we can consider the maximum size among all toys and existing boxes. If a toy of size \( A_i \) needs to go into a box and all existing boxes are smaller, we would need a new box of size at least \( A_i \) to store that toy.

Here's a logical way to approach the solution:

1. Compute the maximum size of all toys, let's call it `max_a`.
2. Compute the maximum size of all boxes, let's call it `max_b`.
3. If `max_a` is less than or equal to `max_b`, then the smallest possible box would just be a box of size `max_a` or larger.
4. If `max_a` is greater than `max_b`, then we need to find the smallest box size `x` such that `x` is at least `max_a`.

#include <vector>
#include <algorithm>

using namespace std;

    cin >> N;
    vector<int> A(N);
    vector<int> B(N - 1);

    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    for (int i = 0; i < N - 1; i++) {
        cin >> B[i];
    }

    long long max_a = *max_element(A.begin(), A.end());
    long long max_b = *max_element(B.begin(), B.end());

    // If the largest toy size is less than or equal to the largest box size,
    // then the smallest box needed would be of the size of the largest toy.
    if (max_a <= max_b) {
        cout << max_a << endl;
    } else {
        // Otherwise, we need to find the smallest box that can hold the largest toy.
        // This would be the largest box size from the existing boxes.
        cout << max_b << endl;
    }

    return 0;
}
