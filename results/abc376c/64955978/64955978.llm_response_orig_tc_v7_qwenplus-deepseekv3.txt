```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    int N;
    cin >> N;
    vector<int> A(N), B(N-1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N-1; ++i) cin >> B[i];

    sort(A.begin(), A.end(), greater<int>());
    sort(B.begin(), B.end(), greater<int>());

    vector<bool> used(N-1, false);
    int x = -1;
    int j = 0;
    for (int i = 0; i < N; ++i) {
        if (j < N-1 && A[i] <= B[j]) {
            used[j] = true;
            j++;
        } else {
            if (x == -1) {
                x = A[i];
            } else {
                cout << -1 << endl;
                return 0;
            }
        }
    }

    if (x != -1) {
        cout << x << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
```

### Explanation
1. **Sorting**: Both the toys (`A`) and boxes (`B`) are sorted in descending order to facilitate greedy matching.
2. **Greedy Matching**: We iterate through each toy, trying to match it to the largest available box that can accommodate it. If a toy cannot be placed in any existing box, it is considered for the new box `x`.
3. **Result Determination**: If only one toy remains unmatched, it becomes the value of `x`. If more than one toy is unmatched, it's impossible to satisfy the conditions, and we output `-1`. If all toys are matched to existing boxes, we also output `-1` because no new box is needed, which contradicts the problem's requirement that a new box must be purchased.
