#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    int N;
    cin >> N;
    vector<ll> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We will use binary search to find the minimal x
    ll low = 1, high = 1e18, answer = -1;

    while (low <= high) {
        ll mid = (low + high) / 2;
        bool possible = true;
        int a_idx = 0, b_idx = 0;

        // Count how many toys can fit in existing boxes with size >= toy
        for (int i = 0; i < N - 1; ++i) {
            if (B[b_idx] >= A[a_idx]) {
                a_idx++;
                b_idx++;
            }
        }

        // Check if the remaining toy(s) can fit in the new box of size `mid`
        if (a_idx < N && A[a_idx] <= mid)
            a_idx++;

        if (a_idx == N)
            possible = true;
        else
            possible = false;

        if (possible) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << answer << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
