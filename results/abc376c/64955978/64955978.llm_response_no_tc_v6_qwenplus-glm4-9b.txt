#include<bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> A(N), B(N);
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (ll i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }
    
    // Sort the sizes of toys and boxes in descending order
    sort(A.begin(), A.end(), greater<ll>());
    sort(B.begin(), B.end(), greater<ll>());

    // The minimum box size x must be able to fit the largest toy
    ll x = A[0];
    bool possible = true;

    // Check if all toys can fit into the existing boxes or the newly purchased box of size x
    for (ll i = 0; i < N; ++i) {
        if (A[i] > B[i]) {
            // If the current toy cannot fit into the current box, it must fit into the box of size x
            if (A[i] > x) {
                possible = false;
                break;
            }
        }
    }

    if (possible) {
        cout << x << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();

    return 0;
}
