#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;
    std::vector<int> a(n), b(n - 1);
    for (int i = 0; i < n; ++i) {
        std::cin >> a[i];
    }
    for (int i = 1; i < n; ++i) {
        std::cin >> b[i - 1];
    }
    std::sort(a.begin(), a.end());
    std::sort(b.begin(), b.end());

    // We need to find the smallest box size x such that all toys can be placed
    // into existing boxes or the new box of size x.
    int min_x = *std::max_element(b.begin(), b.end()) + 1; // Start with the largest box size + 1
    bool can_place = true;

    for (int i = 0; i < n; ++i) {
        // Check if the current toy can be placed in the smallest box found so far
        if (a[i] > min_x) {
            can_place = false;
            break;
        }
        // Check if the current toy can be placed in the existing boxes
        if (i > 0 && a[i] <= b[i - 1]) {
            continue;
        }
        // Otherwise, we need to increase the size of the box
        min_x = a[i] + 1;
    }

    if (can_place) {
        std::cout << min_x << std::endl;
    } else {
        std::cout << -1 << std::endl;
    }

    return 0;
}
