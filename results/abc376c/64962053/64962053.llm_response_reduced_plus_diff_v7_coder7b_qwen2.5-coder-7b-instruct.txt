```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;
    std::vector<long long> a(n), b(n - 1);

    // Read input values
    for(int i = 0; i < n; ++i) {
        std::cin >> a[i];
    }
    for(int i = 0; i < n - 1; ++i) {
        std::cin >> b[i];
    }

    // Sort the arrays
    std::sort(a.begin(), a.end());
    std::sort(b.begin(), b.end());

    // Initialize variables
    bool possible = true;
    int idx = 0;

    // Check if there's enough space to fit each toy in the boxes
    for(int i = 0; i < n && possible; ++i) {
        while(idx < n - 1 && b[idx] < a[i]) {
            ++idx;
        }
        if(idx == n - 1 && b[idx] < a[i]) {
            possible = false;
        } else {
            ++idx;
        }
    }

    // Print result based on possibility
    if(possible) {
        std::cout << a.back() + 1 << std::endl;
    } else {
        std::cout << -1 << std::endl;
    }

    return 0;
}
```
