```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find the minimal x such that we can assign each toy to a box:
    // - Each toy fits in its box (box size >= toy size)
    // - Each box holds exactly one toy
    // - One additional box of size x is added

    // We try to match the smallest toys to the smallest boxes.
    // The first toy that cannot be matched will require the new box.

    int i = 0, j = 0;
    vector<long long> unmatched_toys;

    while (i < N && j < N - 1) {
        if (A[i] <= B[j]) {
            // Toy fits in this box, use it
            i++;
            j++;
        } else {
            // Box too small, try next box
            j++;
        }
    }

    if (i == N) {
        // All toys fit, no need for extra box
        cout << -1 << endl;
        return 0;
    }

    // There are still unmatched toys
    for (; i < N; ++i) {
        unmatched_toys.push_back(A[i]);
    }

    if (unmatched_toys.empty()) {
        cout << -1 << endl;
    } else {
        cout << unmatched_toys[0] << endl;
    }

    return 0;
}
```
