```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> bSet;
    for (long long b : B) {
        bSet.insert(b);
    }

    for (int i = 0; i < N; ++i) {
        auto it = bSet.lower_bound(A[i]);
        if (it != bSet.end()) {
            bSet.erase(it); // Assign box to toy
        } else if (i == N - 1) {
            // Last toy will be placed in the new box, so we can proceed
            cout << A[i] << "\n";
            return 0;
        } else {
            // Can't place current toy in any of the existing boxes
            cout << "-1\n";
            return 0;
        }
    }

    // This shouldn't happen as one toy remains unassigned and we have a new box
    cout << "-1\n";
    return 0;
}
```
