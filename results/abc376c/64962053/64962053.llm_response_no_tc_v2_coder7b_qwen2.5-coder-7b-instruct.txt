```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n);
    for(int i = 0; i < n; ++i)
        cin >> a[i];
    for(int i = 0; i < n - 1; ++i)
        cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    long long max_b = accumulate(b.begin(), b.end(), 0LL);
    if(max_b >= a[n - 1]) {
        cout << a[n - 1] << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
```
