```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> A(n);
    vector<long long> B(n - 1);

    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to assign N toys to N boxes: N-1 given + 1 of size x.
    // Each toy must go into a box that is >= its size, and each box can hold at most one toy.

    // Strategy:
    // Try to match largest N-1 toys with the boxes.
    // The unmatched toy will be placed in the purchased box. Its size determines x.

    // To do this efficiently, we try to greedily match largest toys with largest available boxes.
    // If we cannot match all N toys using the existing boxes and one custom box,
    // then it's impossible.

    // Let's try to match largest N-1 toys with the boxes.

    vector<long long> unmatched;

    int j = n - 2; // index for B
    for (int i = n - 1; i >= 0; --i) {
        if (j >= 0 && B[j] >= A[i]) {
            // Match found
            j--;
        } else {
            // Cannot fit this toy into any available box, so store it as unmatched
            unmatched.push_back(A[i]);
        }
    }

    if (unmatched.size() > 1) {
        cout << -1 << endl;
        return 0;
    }

    cout << (unmatched.empty() ? A[0] : unmatched[0]) << endl;
}
```
