#include <bits/stdc++.h>
using namespace std;

int main() {

    cin >> n;
    vector<int> A(n), B(n - 1);
    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n - 1; ++i) cin >> B[i];
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());
    
    int i = 0, j = 0;
    vector<int> unmatched;
    while (i < n && j < n - 1) {
        if (A[i] <= B[j]) {
            ++i;
            ++j;
        } else {
            unmatched.push_back(A[i]);
            ++i;
        }
    }
    while (i < n) {
        unmatched.push_back(A[i]);
        ++i;
    }
    
    if (unmatched.empty()) {
        cout << 1 << endl;
    } else if (unmatched.size() == 1) {
        cout << unmatched[0] << endl;
    } else {
        cout << -1 << endl;
    }
    
    return 0;
}
