    long long maxToySize = *max_element(va.begin(), va.end());

    bool canStoreAllToys = true;

        if (vb[i] < va[i]) {
            canStoreAllToys = false;
        }
    }

    // If all toys can fit in existing boxes, we need a new box as big as the largest toy
    if (canStoreAllToys) {
        cout << maxToySize << endl;

            }
        }
