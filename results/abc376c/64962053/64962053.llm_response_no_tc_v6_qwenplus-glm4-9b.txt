#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> A(n);
    vector<int> B(n - 1);
    
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> B[i];
    }

    // Since we need to find the minimum box size x, we start with the maximum possible value
    // which is the size of the largest toy A[n-1].
    int x = A.back();
    
    // We iterate over the toys from the largest to the smallest to check if we can fit each toy
    // into the existing boxes or the newly chosen box x.
    for (int i = n - 1; i >= 0; --i) {
        // Check if the current toy can fit into box x
        if (A[i] <= x) {
            continue; // If it fits, we move on to the next toy
        } else {
            // If it doesn't fit, we need to buy a new box
            // We update x to be the size of the current toy to ensure it fits
            x = A[i];
        }
    }
    
    // After processing all toys, x should be the minimum box size needed to fit all toys
    cout << x << endl;
    
    return 0;
}
