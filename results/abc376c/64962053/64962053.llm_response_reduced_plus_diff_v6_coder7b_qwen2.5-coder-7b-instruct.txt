#include <bits/stdc++.h>
using namespace std;

bool canFitAllToys(vector<int>& toys, vector<int>& boxes, int x) {
    priority_queue<int, vector<int>, greater<int>> pq(boxes.begin(), boxes.end());
    for (int toy : toys) {
        while (!pq.empty() && pq.top() < toy) pq.pop();
        if (pq.empty()) return false;
        pq.pop();
    }
    return true;
}

int main() {
    int n;
    cin >> n;
    vector<int> toys(n), boxes(n-1);
    for (int i = 0; i < n; ++i) cin >> toys[i];
    for (int i = 0; i < n-1; ++i) cin >> boxes[i];

    sort(toys.begin(), toys.end());
    sort(boxes.begin(), boxes.end());

    if (toys[n-1] <= boxes[n-2]) {
        cout << toys[n-1] << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
