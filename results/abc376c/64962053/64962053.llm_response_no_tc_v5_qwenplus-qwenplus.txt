```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> A(n);
    vector<long long> B(n - 1);

    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to assign N toys into N boxes: N-1 existing + one new.
    // Each toy must go into a box of size >= its size.
    // Exactly one new box is added. We want the minimal x such that this works.

    // Strategy:
    // Try to match all toys with the N-1 boxes, leaving one toy out.
    // The unmatched toy will be placed in the new box of size x.
    // To minimize x, we want the smallest unmatched toy.

    // We greedily try to match largest toys with largest available boxes.
    // If a toy cannot fit in the current largest unused box, then it must go in the new box.

    int j = n - 2; // index for B (boxes), starting from the largest
    long long required_size = -1;

    for (int i = n - 1; i >= 0; --i) {
        if (j >= 0 && A[i] <= B[j]) {
            j--; // use this box
        } else {
            // This toy cannot be placed in any remaining existing box
            // It must go into the new box
            required_size = A[i];
            break;
        }
    }

    cout << required_size << endl;
}
```
