```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> bSet;
    for (long long b : B) bSet.insert(b);

    // Try to match toys with boxes greedily
    for (int i = N - 1; i >= 0; --i) {
        auto it = bSet.lower_bound(A[i]);
        if (it != bSet.end()) {
            bSet.erase(it);
        } else {
            // This toy cannot be placed in any box, so we need the extra box of size A[i]
            cout << A[i] << endl;
            return 0;
        }
    }

    // If all toys were matched, the minimum x can be zero theoretically,
    // but according to problem statement, x must be a positive integer.
    // So Takahashi must buy at least one box, which could be of size 1.
    cout << 1 << endl;
    return 0;
}
```
