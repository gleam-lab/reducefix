```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll N;
    cin >> N;
    vector<ll> A(N), B(N - 1);

    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> bSet;
    for (ll b : B) bSet.insert(b);

    for (ll i = 0; i < N; ++i) {
        auto it = bSet.lower_bound(A[i]);
        if (it == bSet.end()) {
            // Cannot fit this toy in any available box
            cout << -1 << endl;
            return;
        }
        bSet.erase(it);
    }

    // The smallest x is the smallest toy that couldn't be placed in existing boxes
    // Since all toys are placed, x can be as small as the largest leftover toy requirement
    // But we must find the minimum x such that all toys can be placed
    // So x should be at least the last unplaced toy before step completes
    // However, since we used greedy matching, just output the largest toy that needed a new box
    cout << A.back() << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
```
