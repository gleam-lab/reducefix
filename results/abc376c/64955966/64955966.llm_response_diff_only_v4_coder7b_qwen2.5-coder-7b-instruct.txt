```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> a(N), b(N - 1);
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < N - 1; ++i) {
        cin >> b[i];
    }

    // Sort toys in descending order
    sort(a.begin(), a.end(), greater<ll>());
    // Sort boxes in ascending order
    sort(b.begin(), b.end());

    ll max_box_size = 0;
    bool valid = true;

    // Check if we can fit all toys into the boxes
    for (int i = 0; i < N - 1; ++i) {
        if (a[i] > b[i]) {
            max_box_size = max(max_box_size, a[i]);
            valid = false;
        } else {
            break;
        }
    }

    // If the largest toy is too big for the largest box, check the last box
    if (!valid && a[N - 1] > b[N - 2]) {
        max_box_size = max(max_box_size, a[N - 1]);
    }

    // If still invalid, output -1
    if (!valid) {
        cout << -1 << endl;
    } else {
        cout << max_box_size << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
```

### Explanation of Changes:
1. **Vector Usage**: Changed arrays `a` and `b` to vectors for easier manipulation.
2. **Sorting**: Sorted `a` in descending order and `b` in ascending order.
3. **Validation Loop**: Simplified the loop to check if each toy can fit into its corresponding box. If not, update the maximum required box size and mark as invalid.
4. **Final Check**: After the loop, checked if the last toy fits into the second-largest box.
5. **Output**: If valid, print the maximum required box size; otherwise, print `-1`.
