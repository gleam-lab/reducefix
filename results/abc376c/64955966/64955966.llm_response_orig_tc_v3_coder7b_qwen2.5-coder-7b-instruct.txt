```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(ll i = 0; i < N; ++i){
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; ++i){
        cin >> b[i];
    }
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Find the smallest x such that we can fit all toys into boxes
    for(ll x = a[0]; x <= 1e9; ++x){
        bool valid = true;
        vector<bool> used(N-1, false);
        
        for(ll i = 0; i < N; ++i){
            if(i == N-1 && !used.back()){
                if(a[i] > x){
                    valid = false;
                    break;
                }else{
                    continue;
                }
            }
            
            ll j = lower_bound(b.begin(), b.end(), a[i]) - b.begin();
            while(j < N-1 && used[j]){
                ++j;
            }
            
            if(j >= N-1 || a[i] > x){
                valid = false;
                break;
            }else{
                used[j] = true;
            }
        }
        
        if(valid){
            cout << x << endl;
            return;
        }
    }
    
    cout << "-1" << endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();

    return 0;
}
```

This solution iterates through potential values of `x` starting from the maximum toy size up to \(10^9\) and checks if it's possible to place all toys into boxes without exceeding any box's capacity. It uses binary search to find the appropriate box size for each toy and ensures that each toy is placed in a valid box. If a valid `x` is found, it prints the value; otherwise, it outputs `-1`.
