#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Try to fit the smallest N-1 toys into the N-1 boxes.
    int j = 0;
    ll required_size = -1;

    for (int i = 0; i < N; ++i) {
        if (j < N - 1 && A[i] <= B[j]) {
            j++;
        } else {
            // This toy must go into the new box
            if (required_size == -1) {
                required_size = A[i];
            } else {
                // Only one toy can go into the new box
                cout << -1 << endl;
                return;
            }
        }
    }

    if (required_size != -1) {
        cout << required_size << endl;
    } else {
        // All toys fit in existing boxes, but we still need a box for one of them
        // So we can choose x = 1 (minimum possible value)
        cout << 1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
