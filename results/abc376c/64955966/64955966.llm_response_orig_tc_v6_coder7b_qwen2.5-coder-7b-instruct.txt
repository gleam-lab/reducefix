```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(ll i = 0; i < N; ++i){
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; ++i){
        cin >> b[i];
    }
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    bool found = false;
    for(ll i = 0; i < N-1 && !found; ++i){
        if(a[i] <= b[i]){
            found = true;
        }
    }
    
    if(found){
        cout << a.back() << endl;
    }else{
        cout << "-1" << endl;
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();

    return 0;
}
```
This solution checks if there is any box that can hold at least one toy. If such a box exists, it prints the size of the largest toy that can fit into that box. Otherwise, it prints `-1`. The approach ensures that we use the smallest possible box by sorting both the array of toy sizes and the array of box sizes.
