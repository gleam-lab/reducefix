#include <bits/stdc++.h>

using namespace std;

// Function to check if we can place all toys in boxes with minimum size x
bool canPlaceAllToys(vector<int> &a, vector<int> &b, int x) {
    int n = a.size();
    // Sort sizes in descending order
    sort(a.begin(), a.end(), greater<int>());
    sort(b.begin(), b.end(), greater<int>());

    int j = 0; // Index for boxes
    for (int i = 0; i < n && j < b.size(); ++i) {
        if (a[i] <= b[j]) {
            // If current toy fits in current box, move to next toy
            ++j;
        } else {
            // If current toy does not fit, try placing it in a bigger box
            while (j < b.size() && a[i] > b[j]) {
                ++j;
            }
            if (j == b.size()) {
                // No suitable box found
                return false;
            }
            // Place toy in next available box
            ++j;
        }
    }
    // Check if all toys have been placed
    return j == b.size();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);

    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    // Find the minimum size of the box needed
    int l = 1, r = 1e9 + 10;
    while (l < r) {
        int mid = (l + r) / 2;
        if (canPlaceAllToys(a, b, mid)) {
            r = mid;
        } else {
            l = mid + 1;
        }
    }

    // Check if the found size works
    if (canPlaceAllToys(a, b, l)) {
        cout << l << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}

This solution uses binary search to find the minimum size of the box required to store all toys without exceeding their individual sizes. It first sorts both arrays in descending order and then tries to place each toy in the largest available box that can accommodate it. If at any point, no suitable box is found for a toy, it means the current mid-point size is too small, and the search range is adjusted accordingly. The final result is checked using the `canPlaceAllToys` function to ensure correctness.
