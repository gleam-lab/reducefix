#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve() {
    ll N;
    cin >> N;
    vector<ll> a(N), b(N - 1);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < N - 1; i++) {
        cin >> b[i];
    }
    
    // Sort both arrays in ascending order
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    // Initialize variables
    bool placed[N] = {false};
    int minBoxSize = INT_MAX;
    
    // Try placing toys into boxes starting from the largest toy
    for (int i = N - 1; i >= 0; i--) {
        int toySize = a[i];
        bool found = false;
        
        // Find a suitable box for the current toy
        for (int j = 0; j < N - 1; j++) {
            if (!placed[j] && toySize <= b[j]) {
                placed[j] = true;
                found = true;
                break;
            }
        }
        
        // If no suitable box is found, check if we need to buy a new box
        if (!found) {
            if (toySize > b[N - 2]) {
                cout << -1 << endl;
                return;
            } else {
                minBoxSize = max(minBoxSize, toySize);
            }
        }
    }
    
    // Check if all toys fit without needing a new box
    if (minBoxSize == INT_MAX) {
        cout << "1" << endl;
    } else {
        cout << minBoxSize << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    solve();
    
    return 0;
}
