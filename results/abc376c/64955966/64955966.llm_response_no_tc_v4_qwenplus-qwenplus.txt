#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in descending order
    sort(A.begin(), A.end(), greater<ll>());
    sort(B.begin(), B.end(), greater<ll>());

    // We will use binary search to find the minimal x
    ll left = 1, right = 1e18;
    ll answer = -1;

    while (left <= right) {
        ll mid = (left + right) / 2;

        // Create a list of available box sizes including the new one
        vector<ll> boxes = B;
        boxes.push_back(mid);  // Add the new box of size `mid`
        sort(boxes.begin(), boxes.end(), greater<ll>());

        bool possible = true;
        ll b_idx = 0;

        for (ll a : A) {
            if (b_idx < boxes.size() && boxes[b_idx] >= a) {
                b_idx++;
            } else {
                possible = false;
                break;
            }
        }

        if (possible) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    cout << answer << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
