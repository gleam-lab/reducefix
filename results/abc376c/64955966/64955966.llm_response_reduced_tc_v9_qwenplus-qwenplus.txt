```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxes;
    for (long long b : B) {
        boxes.insert(b);
    }

    for (int i = 0; i < N; ++i) {
        auto it = boxes.lower_bound(A[i]);
        if (it == boxes.end()) {
            cout << -1 << endl;
            return 0;
        }
        boxes.erase(it);
    }

    cout << A[0] << endl;
    return 0;
}
```
