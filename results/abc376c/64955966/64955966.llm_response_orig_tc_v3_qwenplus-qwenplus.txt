```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N;
    cin >> N;

    vector<ll> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> b_set;
    for (ll x : B) b_set.insert(x);

    // Try to fit the smallest N-1 toys into boxes
    int j = 0;
    for (int i = 0; i < N - 1; ++i) {
        auto it = b_set.lower_bound(A[i]);
        if (it == b_set.end()) {
            cout << -1 << endl;
            return 0;
        }
        b_set.erase(it);
    }

    // The only remaining toy is A[N-1], we need a box >= A[N-1]
    cout << A[N - 1] << endl;

    return 0;
}
```
