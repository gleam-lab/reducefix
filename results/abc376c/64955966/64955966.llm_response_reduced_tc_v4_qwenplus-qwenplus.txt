#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    int j = 0;
    long long minRequiredBox = -1;

    for (int i = 0; i < N; ++i) {
        // Add all boxes that can potentially fit current toy
        while (j < N - 1 && B[j] < A[i]) {
            boxSizes.insert(B[j]);
            j++;
        }

        // If we still haven't found a suitable box, we need to use the new one
        if (boxSizes.empty()) {
            minRequiredBox = A[i];
            continue;
        }

        // Use the smallest possible box that can fit the current toy
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            boxSizes.erase(it);
        } else {
            // No suitable box found, this toy must go into the new box
            minRequiredBox = A[i];
        }
    }

    if (minRequiredBox == -1) {
        cout << -1 << endl;
    } else {
        cout << minRequiredBox << endl;
    }

    return 0;
}
