```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(ll i = 0; i < N; ++i){
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; ++i){
        cin >> b[i];
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    bool possible = false;
    for(ll i = 0; i < N-1; ++i){
        if(a[i] <= b[i]){
            continue;
        }else if(i + 1 < N - 1 && a[i] <= b[i + 1]){
            continue;
        }else{
            possible = true;
            break;
        }
    }

    if(!possible){
        cout << "-1\n";
        return;
    }

    cout << *max_element(a.begin(), a.end()) << "\n";
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    solve();
    return 0;
}
```

This solution addresses the problem by ensuring that we have enough space to place each toy in a box where its size is not smaller than the toy's size. It first sorts both arrays of toy sizes and box sizes, then checks if it's possible to fit each toy in a box using the existing boxes. If not, it concludes that no such box exists and prints -1. Otherwise, it prints the size of the largest toy that can be stored in any of the existing boxes.
