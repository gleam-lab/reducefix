```cpp
#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(int i = 0; i < N; i++) cin >> a[i];
    for(int i = 0; i < N-1; i++) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    bool found = false;
    for(ll x = a[0]; x <= a[N-1]; ++x){
        vector<bool> used(N-1, false);
        bool valid = true;
        for(int i = 0; i < N; ++i){
            int idx = upper_bound(b.begin(), b.end(), a[i]) - b.begin();
            if(idx == N-1 && !used[idx] && a[i] <= b[idx]){
                used[idx] = true;
            }else if(idx > 0 && !used[idx-1] && a[i] <= b[idx-1]){
                used[idx-1] = true;
            }else{
                valid = false;
                break;
            }
        }
        if(valid){
            cout << x << endl;
            found = true;
            break;
        }
    }
    if(!found) cout << -1 << endl;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
```

### Explanation:
1. **Input Reading**: The input is read as specified, with `N`, followed by the sizes of the `N` toys and then the sizes of the `N-1` boxes.
2. **Sorting**: Both arrays, `a` and `b`, are sorted to facilitate finding suitable box sizes for each toy.
3. **Finding Minimum Box Size**: We iterate through potential box sizes starting from the smallest toy size (`a[0]`) up to the largest toy size (`a[N-1]`). For each potential box size `x`, we check if it's possible to place all toys in the available boxes without violating the constraints.
4. **Placement Check**: Using a greedy approach, we try to place each toy in the smallest available box that can fit it. If we can successfully place all toys, we print `x` and exit. If no valid box size is found within the range, we print `-1`.
5. **Efficiency**: The solution ensures that we efficiently find the minimum box size using sorting and binary search techniques, making it suitable for large inputs within the given constraints.
