#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> b_set;
    ll j = 0;
    ll required = 0;

    // Try to fit toys into boxes greedily
    for (ll i = 0; i < N; ++i) {
        while (j < N - 1 && B[j] < A[i]) {
            b_set.insert(B[j]);
            ++j;
        }
        if (!b_set.empty()) {
            auto it = b_set.lower_bound(A[i]);
            if (it != b_set.end()) {
                b_set.erase(it);  // use this box
                continue;
            }
        }
        // This toy needs the new box
        if (required == 0 || A[i] > required) {
            required = A[i];
        }
    }

    if (required == 0) {
        cout << -1 << endl;
    } else {
        cout << required << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
