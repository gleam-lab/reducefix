#include<bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    ll a[N], b[N-1];
    for(ll i = 0; i < N; i++) {
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; i++) {
        cin >> b[i];
    }
    
    // Sort the toy sizes in descending order
    sort(a, a + N, greater<ll>());
    // Sort the box sizes in descending order
    sort(b, b + N-1, greater<ll>());

    // Initialize the minimum box size to a large value
    ll min_box_size = LLONG_MAX;
    
    // Iterate over each toy and find the minimum box size that can be used for all toys
    for(ll i = 0; i < N; i++) {
        ll max_needed_box_size = 0;
        for(ll j = 0; j < N-1; j++) {
            // If the toy can fit into box j, break and use box j+1 for the rest
            if(a[i] <= b[j]) {
                max_needed_box_size = max(max_needed_box_size, b[j+1]); // We need the box after the current one
                break;
            }
        }
        // If the toy cannot fit into any of the boxes, it's impossible to store all toys
        if(max_needed_box_size == 0) {
            cout << -1 << endl;
            return;
        }
        // Update the minimum box size needed
        min_box_size = min(min_box_size, max_needed_box_size);
    }
    
    // If we found a minimum box size, print it; otherwise, print -1
    if(min_box_size != LLONG_MAX) {
        cout << min_box_size << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
