void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N - 1);
    for(int i = 0; i < N; i++) cin >> a[i];
    for(int i = 0; i < N - 1; i++) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    ll max_a = *max_element(a.begin(), a.end());
    ll min_b = *min_element(b.begin(), b.end());

    if(max_a <= min_b){
        cout << min_b << endl;
    }else{
        cout << -1 << endl;
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}

The original code was attempting to match the largest remaining toy with the largest available box, which was incorrect.
