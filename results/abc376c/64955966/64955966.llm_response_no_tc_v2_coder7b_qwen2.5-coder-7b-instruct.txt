#include<bits/stdc++.h>
#define ll long long
#define yes cout<<"YES\n"
#define no cout<<"NO\n"
#define sp ' '
using namespace std;

void solve(){
    ll N;
    cin >> N;
    vector<ll> a(N), b(N-1);
    for(ll i = 0; i < N; ++i){
        cin >> a[i];
    }
    for(ll i = 0; i < N-1; ++i){
        cin >> b[i];
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    ll x = max(*max_element(a.begin(), a.end()), *max_element(b.begin(), b.end()));
    bool valid = true;

    for(ll i = 0; i < N - 1 && valid; ++i){
        if(a[i] > x || a[i + 1] > x){
            valid = false;
        }
    }

    if(valid){
        cout << x << endl;
    } else {
        cout << -1 << endl;
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
