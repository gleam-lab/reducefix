#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll N;
    cin >> N;
    vector<ll> A(N), B(N - 1);

    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> bSet(B.begin(), B.end());

    ll requiredBox = -1;

    for (ll i = 0, j = 0; i < N; ++i) {
        auto it = bSet.lower_bound(A[i]);
        if (it != bSet.end()) {
            bSet.erase(it);
        } else {
            // This is the toy that will go into the new box
            if (requiredBox == -1)
                requiredBox = A[i];
            else {
                // We already have a candidate for new box but can't fit this one => impossible
                cout << -1 << endl;
                return;
            }
        }
    }

    if (requiredBox == -1) {
        // All toys fit in existing boxes without needing an extra => choose smallest possible box anyway
        // Find smallest box that is not used
        sort(B.begin(), B.end());
        sort(A.begin(), A.end());
        for (ll i = 0; i < N - 1; ++i) {
            if (B[i] >= A[i] && (i == N - 2 || B[i + 1] >= A[i + 1])) {
                requiredBox = A[i + 1];
                break;
            }
        }
        if (requiredBox == -1) {
            cout << -1 << endl;
            return;
        }
    }

    cout << requiredBox << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
