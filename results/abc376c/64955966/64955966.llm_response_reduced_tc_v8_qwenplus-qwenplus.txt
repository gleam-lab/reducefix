```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> b_set(B.begin(), B.end());

    ll required_x = -1;

    for (ll i = 0, j = 0; i < N; ++i) {
        if (j < N - 1 && A[i] <= B[j]) {
            ++j;
        } else {
            required_x = A[i];
            break;
        }
    }

    if (required_x == -1) {
        cout << -1 << endl;
        return;
    }

    // Now check if all other toys can be placed in the boxes excluding one box that will be replaced by x
    // We already have a candidate x = required_x
    // Try to assign each toy to a box >= its size, using the smallest available box possible
    // Use greedy with multiset

    auto check = [&](ll x) -> bool {
        multiset<ll> boxes = b_set;
        boxes.insert(x);

        for (ll a : A) {
            auto it = boxes.lower_bound(a);
            if (it == boxes.end()) return false;
            boxes.erase(it);
        }
        return true;
    };

    // Binary search on x
    ll low = required_x, high = 2e9, answer = -1;
    while (low <= high) {
        ll mid = (low + high) / 2;
        if (check(mid)) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << (answer != -1 ? answer : -1) << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
