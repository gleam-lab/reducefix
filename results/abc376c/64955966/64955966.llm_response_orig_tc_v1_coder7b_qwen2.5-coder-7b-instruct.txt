void solve(){
    ll N;
    cin>>N;
    vector<ll> a(N), b(N-1);
    for(ll &i:a){
        cin>>i;
    }
    for(ll &i:b){
        cin>>i;
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    ll i = 0, j = 0;
    while(i < N && j < N-1){
        if(a[i] <= b[j]) {
            i++;
        } 
        else {
            j++;
        }
    }
    if(i == N){
        cout<<b[j-1]<<endl;
    }
    else{
        cout<<-1<<endl;
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}

This code sorts both arrays and then checks if we can fit all the toys into the available boxes by comparing them sequentially. If at any point, a toy cannot fit into the current box, it moves to the next one. If we successfully fit all toys, it prints the size of the last used box. If not, it prints -1.
