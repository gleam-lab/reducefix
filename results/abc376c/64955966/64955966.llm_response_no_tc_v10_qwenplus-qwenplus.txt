```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    // Sort toys and boxes in descending order
    sort(A.begin(), A.end(), greater<ll>());
    sort(B.begin(), B.end(), greater<ll>());

    ll required_x = -1;
    ll b_idx = 0;

    for (ll i = 0; i < N; ++i) {
        if (b_idx < N - 1 && B[b_idx] >= A[i]) {
            b_idx++;
        } else {
            // This toy must go into the new box
            if (required_x == -1) {
                required_x = A[i];
            } else {
                // We need a bigger box to fit multiple unplaced toys
                required_x = max(required_x, A[i]);
            }
        }
    }

    if (required_x != -1) {
        cout << required_x << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve();
    return 0;
}
```
