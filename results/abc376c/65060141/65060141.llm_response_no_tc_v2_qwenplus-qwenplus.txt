```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> A(n), B(n - 1);

    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Try to assign largest toys to largest boxes
    int a_idx = n - 1;
    int b_idx = n - 2;

    while (a_idx >= 0 && b_idx >= 0) {
        if (B[b_idx] >= A[a_idx]) {
            b_idx--;
            a_idx--;
        } else {
            // Can't fit, need an extra box of at least A[a_idx]
            a_idx--;
            // We can afford only one toy not fitting in existing boxes
            if (n - (a_idx + 1) >= 1) break;
        }
    }

    // If all toys are matched
    if (a_idx < 0) {
        cout << 0 << endl;
        return 0;
    }

    // At this point, we need a box for the largest unmatched toy
    ll required_size = A[a_idx];

    // Check if there is at most one toy that cannot be matched
    int remaining_toys = a_idx + 1;
    if (remaining_toys > 1) {
        cout << -1 << endl;
    } else {
        cout << required_size << endl;
    }

    return 0;
}
```
