```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N],b[N];
int main(){
    ll n,cnt=0,ans=0,mx=0;
    cin>>n;
    for(ll i=1;i<=n;i++){
        cin>>a[i];
    }
    for(ll i=1;i<n;i++){
        cin>>b[i];
    }
    sort(a+1,a+n+1);
    sort(b+1,b+n);
    ll l=n,r=n-1;
    while(l>=1 && r>=1){
        if(b[r]>=a[l]){
            r--;
            l--;
        }
        else{
            cnt++;
            mx=max(mx,a[l]);
            l--;
        }
    }
    if(cnt==1){
        cout<<mx;
    }
    else{
        cout<<-1;
    }
    return 0;
}
```
In this solution, we first read the inputs and then sort both arrays. We keep track of the number of boxes used and the maximum toy size that can fit in those boxes. If we use exactly one box, we output the maximum toy size. Otherwise, we output -1 indicating that it's impossible to place all toys in separate boxes using a single box.
