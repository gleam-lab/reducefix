#include<bits/stdc++.h>
#define ll long long
using namespace std;

const ll N = 2e5 + 5;

ll a[N], b[N];

int main() {
    ll n;
    cin >> n;
    for (ll i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (ll i = 1; i < n; i++) {
        cin >> b[i];
    }

    // Sort the toy sizes and the existing box sizes
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    // Check for compatibility of toys with existing boxes
    bool compatible = true;
    for (ll i = 1; i <= n; i++) {
        if (a[i] > b[i - 1]) {
            compatible = false;
            break;
        }
    }

    // If compatible, the minimum box size x can be the maximum toy size
    if (compatible) {
        ll ans = *max_element(a + 1, a + n + 1);
        cout << ans << endl;
    } else {
        // If not compatible, we need to find the smallest box size x
        // that can accommodate the largest toy size which is not compatible with existing boxes
        ll l = 1, r = *max_element(a + 1, a + n + 1);
        ll mid, ans = -1;
        while (l <= r) {
            mid = (l + r) / 2;
            // Check if there is a box smaller than the mid size that can accommodate all toys
            bool canUseThisBox = true;
            for (ll i = 1; i <= n; i++) {
                if (a[i] > mid) {
                    canUseThisBox = false;
                    break;
                }
            }

            // Check if there is a toy that cannot fit into any of the boxes smaller than mid
            bool canFitIntoSomeBox = false;
            for (ll i = 1; i <= n; i++) {
                if (a[i] > b[i - 1]) {
                    canFitIntoSomeBox = true;
                    break;
                }
            }

            if (canFitIntoSomeBox) {
                r = mid - 1;
            } else {
                ans = mid;
                l = mid + 1;
            }
        }
        cout << ans << endl;
    }
    return 0;
}
