#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll n;
    cin >> n;

    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) {
        cin >> A[i];
    }

    for (ll i = 0; i < n - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We will use binary search to find the minimum required size of the extra box

    // First, check if it's possible with current boxes (to see if we even need an extra box)
    ll a_ptr = 0;
    ll b_ptr = 0;
    ll unplaced = 0;
    ll min_needed = 0;

    while (a_ptr < n && b_ptr < n - 1) {
        if (B[b_ptr] >= A[a_ptr]) {
            a_ptr++;
        }
        b_ptr++;
    }

    if (a_ptr == n) {
        // All toys can be placed in existing boxes
        cout << 0 << endl;
        return 0;
    }

    // Only one toy is left unplaced, we need to find minimal x such that we can place that toy in the new box
    ll remaining_toy = A[a_ptr];

    // Binary search for minimal x such that all toys can be placed
    ll low = 1;
    ll high = 1e18;
    ll answer = -1;

    while (low <= high) {
        ll mid = (low + high) / 2;

        // Try placing all toys in B and one box of size mid
        ll toy_ptr = 0;
        ll box_ptr = 0;
        ll placed = 0;

        while (toy_ptr < n && box_ptr < n - 1) {
            if (B[box_ptr] >= A[toy_ptr]) {
                placed++;
                toy_ptr++;
            }
            box_ptr++;
        }

        // If we can use the extra box to place the next toy
        if (toy_ptr < n && mid >= A[toy_ptr]) {
            placed++;
            toy_ptr++;
        }

        // Check if all toys are placed
        if (placed == n) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << answer << endl;

    return 0;
}
