#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N;
    cin >> N;

    vector<ll> A(N), B(N - 1);

    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (ll i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    ll required = -1;
    ll b_ptr = 0;

    for (ll a : A) {
        if (b_ptr < B.size() && B[b_ptr] >= a) {
            b_ptr++;
        } else {
            // This toy needs the new box
            if (required == -1) {
                required = a;
            } else {
                // More than one toy can't be placed in existing boxes
                cout << -1 << "\n";
                return 0;
            }
        }
    }

    if (required == -1) {
        // All toys fit in existing boxes, so we don't need to buy any box
        // But according to problem statement, we must buy one box
        // So we can just use the smallest unassigned toy size
        // Actually, this case shouldn't happen as per problem constraints

        // Let's still find the minimal x that works
        // We can choose x to be at least the smallest toy not assigned
        // Re-simulate with dummy box list to get correct minimal x
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());

        ll l = 0, r = N - 1;
        while (l < N && r < N - 1) {
            if (B[r] >= A[l]) {
                ++r;
                ++l;
            } else {
                break;
            }
        }
        if (l < N) {
            cout << A[l] << "\n";
        } else {
            cout << -1 << "\n";
        }
    } else {
        cout << required << "\n";
    }

    return 0;
}
