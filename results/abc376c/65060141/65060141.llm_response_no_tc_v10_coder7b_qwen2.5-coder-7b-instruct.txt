```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N],b[N];
int main(){
	ll n,cnt=0,ans=0,mx=0;
	cin>>n;
	for(ll i=1;i<=n;i++){
		cin>>a[i];
	}
	for(ll i=1;i<n;i++){
		cin>>b[i];
	}
	sort(a+1,a+n+1);
	sort(b+1,b+n);
	ll l=n,r=n-1;
	while(l>=1){
		if(r>=1 && b[r]>=a[l]){
			r--;
		}
		else{
			cnt++;
			mx=max(mx,a[l]);
		}
		l--;
	}
	if(cnt>1){
		cout<<-1;
		return 0;
	}
	cout<<mx;
	return 0;
}
```

This solution addresses the problem by sorting both arrays of toy sizes and box sizes. It then iterates through the sorted arrays to determine if a single box of a certain size can accommodate all the toys without violating any constraints. If it finds that more than one box is needed, it prints -1; otherwise, it prints the size of the smallest box required.
