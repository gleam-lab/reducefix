#include<bits/stdc++.h>
#define ll long long
using namespace std;

const ll N = 2e5 + 5;
ll a[N], b[N];

int main() {
    ll n;
    cin >> n;
    for (ll i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (ll i = 1; i < n; i++) {
        cin >> b[i];
    }
    
    // Sort toys and boxes by size to facilitate the matching process
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);

    // Variables to keep track of toys and boxes
    ll toys = n, boxes = n - 1;
    bool invalid = false;

    // Iterate from the largest to the smallest toy
    while (toys > 0 && boxes > 0) {
        // If the current toy can fit into the current box, place it and move to the next toy and box
        if (a[toys] <= b[boxes]) {
            toys--;
            boxes--;
        } else {
            // If the current toy cannot fit into the current box, check if we can buy a new box of appropriate size
            if (boxes == 0 || a[toys] > b[boxes]) {
                invalid = true;
                break;
            }
            // Otherwise, this toy cannot fit into any of the remaining boxes, but we can buy a new one
            toys--;
            boxes--;
        }
    }

    // If it is not possible to fit all toys into boxes, output -1
    if (invalid) {
        cout << -1;
    } else {
        // Otherwise, the minimum size of the box we can buy is the size of the last toy
        cout << a[toys + 1];
    }
    return 0;
}
