#include<bits/stdc++.h>
#define ll long long
using namespace std;

bool canStoreAll(int x, vector<int>& a, vector<int>& b) {
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    int i = 0, j = 0;
    while (i < a.size() || j < b.size()) {
        if (j >= b.size() || (i < a.size() && a[i] <= x)) {
            // If we can place toy i in the current box or we run out of boxes, move to the next toy
            i++;
        } else {
            // Otherwise, we need a new box
            x = max(x, a[i]);
            i++;
        }
    }
    return true;
}

int minBoxSize(vector<int>& a, vector<int>& b) {
    int low = *max_element(a.begin(), a.end()); // Minimum size of a toy
    int high = 1e9; // Maximum possible size of a box
    
    while (low < high) {
        int mid = (low + high) / 2;
        if (canStoreAll(mid, a, b)) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low == *max_element(a.begin(), a.end()) ? -1 : low;
}

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n-1);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < n-1; ++i) cin >> b[i];
    
    cout << minBoxSize(a, b) << endl;
    return 0;
}
