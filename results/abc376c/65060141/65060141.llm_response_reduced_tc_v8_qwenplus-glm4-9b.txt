
```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=2e5+5;
ll a[N], b[N];
int main() {
    ll n;
    cin >> n;
    for (ll i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (ll i = 1; i < n; ++i) {
        cin >> b[i];
    }
    
    // Sort toys and boxes sizes
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n);
    
    // Find the minimum box size x that can accommodate all toys
    bool possible = true;
    ll min_x = LLONG_MAX;
    for (ll i = 1; i <= n; ++i) {
        // If the i-th toy can fit in the i-th box or the next box
        if (a[i] <= b[i-1]) {
            min_x = min(min_x, b[i-1]);
        }
        // If no box can fit the i-th toy, then it's impossible to perform the task
        else if (a[i] > b[i-1] && (i == n || a[i] <= b[i])) {
            possible = false;
            break;
        }
    }
    
    // Output the minimum x or -1 if it's not possible
    if (possible) {
        cout << min_x;
    } else {
        cout << -1;
    }
    
    return 0;
}
```
