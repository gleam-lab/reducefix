```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 2e5 + 5;

ll a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a, a + n);
    sort(b, b + n - 1);

    // We need to assign toys to boxes such that each box contains at most one toy,
    // and only one toy can go into the new box.

    // Strategy: Try to match largest toys with largest available boxes

    int i = n - 1, j = n - 2;
    int extra_box_needed = 0;

    while (i >= 0 && j >= 0) {
        if (b[j] >= a[i]) {
            // Box can fit this toy, use it
            --i;
            --j;
        } else {
            // Cannot fit current toy in biggest available box, must save this toy for new box
            ++extra_box_needed;
            --i;
        }
    }

    if (extra_box_needed > 1) {
        cout << -1 << '\n';
    } else {
        // If we used all toys, then x=0 is OK
        // Otherwise, we need a box just big enough to fit the unmatched toy
        cout << (extra_box_needed == 1 ? a[i + 1] : 0) << '\n';
    }

    return 0;
}
```
