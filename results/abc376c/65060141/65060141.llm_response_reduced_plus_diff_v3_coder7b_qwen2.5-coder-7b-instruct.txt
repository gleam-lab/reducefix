```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < n - 1; ++i) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int ans = 0;
    for(int i = n - 1, j = n - 2; i >= 0; --i) {
        while(j >= 0 && a[i] > b[j]) --j;
        if(j >= 0) {
            ans = max(ans, abs(a[i] - b[j]));
            --j;
        } else break;
    }

    cout << (ans ? ans : -1) << endl;
    return 0;
}
```
