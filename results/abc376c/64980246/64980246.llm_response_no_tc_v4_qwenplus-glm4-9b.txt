#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int solve() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (int &tt : a) cin >> tt;
    for (int &tt : b) cin >> tt;
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    int x = 0;
    for (int i = 0; i < n; ++i) {
        if (a[i] > b[i]) {
            if (x == 0) {
                x = a[i];
            } else if (a[i] != x) {
                x = 0;
                break;
            }
        }
    }
    
    if (x != 0) {
        cout << x << endl;
    } else {
        cout << -1 << endl;
    }
    
    return 0;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
