#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<ll> a(n), b(n - 1);
    for (auto &x : a)
        cin >> x;
    for (auto &x : b)
        cin >> x;

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // We will binary search on the answer
    ll left = 1, right = 2e9 + 10;
    ll answer = -1;

    while (left <= right)
    {
        ll mid = (left + right) / 2;

        // Check if it's possible to assign toys to boxes with an additional box of size mid
        int i = 0, j = 0, k = 0; // i: toy index, j: box index, k: count of placed toys
        bool possible = true;

        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;

        // Try greedy matching using two pointers and min-heap
        while (i < n || !pq.empty())
        {
            // No more toys to place and no more boxes
            if (i == n && pq.empty())
                break;

            // Place as many toys as possible in current box
            while (!pq.empty() && (j < n - 1 ? pq.top().first <= b[j] : pq.top().first <= mid))
            {
                auto [toy_size, idx] = pq.top();
                pq.pop();

                if (j < n - 1)
                {
                    ++j;
                }
                else
                {
                    mid = max(mid, toy_size); // Need this toy in our purchased box
                }
                ++k;
            }

            // If we couldn't place the toy in any box, it's impossible
            if (!pq.empty())
            {
                possible = false;
                break;
            }

            // Add next toy to the queue
            if (i < n)
            {
                pq.push({a[i], i});
                ++i;
            }
        }

        // Final check for remaining toys
        while (!pq.empty() && j < n - 1)
        {
            auto [toy_size, idx] = pq.top();
            pq.pop();
            if (toy_size <= b[j])
            {
                ++j;
                ++k;
            }
            else
            {
                possible = false;
                break;
            }
        }

        while (!pq.empty())
        {
            auto [toy_size, idx] = pq.top();
            pq.pop();
            if (toy_size <= mid)
            {
                ++k;
            }
            else
            {
                possible = false;
                break;
            }
        }

        if (possible && k == n)
        {
            answer = mid;
            right = mid - 1;
        }
        else
        {
            left = mid + 1;
        }
    }

    cout << answer << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
