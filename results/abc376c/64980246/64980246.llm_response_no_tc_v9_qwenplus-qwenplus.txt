#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int x = -1;
    int i = 0, j = 0;
    while (i < n && j < n - 1)
    {
        if (a[i] <= b[j])
        {
            ++i;
            ++j;
        }
        else
        {
            // This toy cannot fit in the current box
            // We need to assign this toy to the additional box
            if (x == -1)
                x = a[i];
            else
            {
                // More than one toy needs to go to the extra box => not possible
                cout << -1 << endl;
                return;
            }
            ++i;
        }
    }

    // After loop, if there is a remaining toy and we haven't assigned an extra box yet
    if (i < n && x == -1)
        x = a[i];

    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
