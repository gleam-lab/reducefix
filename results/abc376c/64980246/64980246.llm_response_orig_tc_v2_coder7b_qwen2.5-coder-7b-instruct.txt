#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    // Sort sizes in descending order
    sort(a.rbegin(), a.rend());
    sort(b.rbegin(), b.rend());

    // Check if we can place all toys using the largest box
    int maxBoxSize = *max_element(a.begin(), a.end());
    int index = 0;
    for (int i = 0; i < n - 1; ++i) {
        if (index >= n || b[i] < a[index]) {
            cout << -1 << endl;
            return;
        }
        ++index;
    }

    cout << maxBoxSize << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
