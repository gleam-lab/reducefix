#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    multiset<int> boxes;
    int j = 0;
    int res = -1;

    for (int i = 0; i < n; ++i)
    {
        // Add all boxes that can potentially fit a[i]
        while (j < n - 1 && b[j] < a[i])
        {
            boxes.insert(b[j]);
            j++;
        }

        if (boxes.empty())
        {
            // No box available to fit this toy, need new box
            if (res == -1 || a[i] > res)
                res = a[i];
        }
        else
        {
            // Use the smallest possible box that fits the toy
            auto it = boxes.lower_bound(a[i]);
            if (it != boxes.end())
            {
                boxes.erase(it); // use this box
            }
            else
            {
                // No box fits this toy, we need to buy one
                if (res == -1 || a[i] > res)
                    res = a[i];
            }
        }
    }

    cout << res << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
