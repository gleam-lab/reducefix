```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int x = -1;
    int i = 0, j = 0;
    while (i < n && j < n - 1)
    {
        if (a[i] <= b[j])
        {
            ++i;
            ++j;
        }
        else
        {
            // This toy needs a new box
            if (x == -1)
                x = a[i];
            else
            {
                // Can't have more than one extra box
                cout << -1 << endl;
                return;
            }
            ++i;
        }
    }

    // If we have unmatched toys and already assigned a box, it's OK
    // because we can put the last unmatched toy in our purchased box
    if (x == -1 && i < n)
        x = a[i];  // We just need any box for the last unmatched toy

    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
```
