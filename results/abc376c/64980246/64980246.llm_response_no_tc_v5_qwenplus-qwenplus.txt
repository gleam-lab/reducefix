#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    // Sort toys and boxes in descending order
    sort(a.begin(), a.end(), greater<int>());
    sort(b.begin(), b.end(), greater<int>());

    // Try to fit the largest toy in the largest available box
    int x = -1;
    int i = 0, j = 0;
    while (i < n && j < n - 1)
    {
        if (a[i] <= b[j])
        {
            ++i; // Toy placed in this box
        }
        else
        {
            // This toy cannot go into any box so far
            // We need to assign it to the extra box
            if (x == -1)
                x = a[i];
            else
            {
                // Can't have more than one unassigned large toy
                cout << -1 << endl;
                return;
            }
            ++i;
            continue;
        }
        ++j;
    }

    // If there are still toys left, they must go to the new box
    while (i < n)
    {
        if (x == -1)
            x = a[i];
        else
        {
            cout << -1 << endl;
            return;
        }
        ++i;
    }

    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
