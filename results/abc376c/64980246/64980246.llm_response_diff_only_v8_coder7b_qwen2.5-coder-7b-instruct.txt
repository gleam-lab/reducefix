#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    // Sort both arrays in ascending order
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Initialize variables to track the minimum box size needed
    int x = -1;
    int j = 0;

    // Iterate through the toys
    for (int i = 0; i < n; ++i) {
        // Find a suitable box for the current toy
        while (j < n - 1 && b[j] < a[i]) {
            ++j;
        }
        
        // If we found a suitable box, check its size
        if (j < n - 1 && b[j] >= a[i]) {
            if (x == -1) {
                x = b[j];
            } else {
                x = max(x, b[j]);
            }
        } else {
            // If no suitable box was found, break the loop
            break;
        }
    }

    // Output the result
    if (x != -1) {
        cout << x << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // Uncomment the line below to read multiple test cases
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}

### Explanation of Changes:
1. **Sorting Arrays**: The arrays `a` and `b` are sorted in ascending order instead of descending order. This allows us to efficiently find the smallest box that can fit each toy.
2. **Finding Suitable Box**: The inner loop now finds the first box that can accommodate the current toy by iterating through the `b` array until a suitable box is found.
3. **Updating Minimum Box Size**: When a suitable box is found, the code updates `x` with the maximum size encountered so far. This ensures that we keep the smallest possible box size that fits all toys.
4. **Handling Edge Cases**: If no suitable box is found for any toy, the loop breaks, and `-1` is printed, indicating that it's impossible to fit all toys under the given constraints.

This approach ensures that the solution is efficient and handles the problem requirements correctly.
