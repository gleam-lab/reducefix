#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // We'll use a multiset to efficiently match toys to boxes
    multiset<int> boxSet;
    for (int i = 0, j = 0; i < n; ++i) {
        // Add all boxes smaller than current toy to the set
        while (j < n - 1 && b[j] < a[i]) {
            boxSet.insert(b[j]);
            ++j;
        }
        
        // If we have a box that can fit this toy, use it
        if (!boxSet.empty()) {
            boxSet.erase(boxSet.begin()); // Use the smallest available box
        } else {
            // No box can fit this toy, so we need to buy one
            cout << a[i] << endl;
            return;
        }
    }

    // If we reach here, all toys could be placed in existing boxes
    // So no additional box is needed
    cout << -1 << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
