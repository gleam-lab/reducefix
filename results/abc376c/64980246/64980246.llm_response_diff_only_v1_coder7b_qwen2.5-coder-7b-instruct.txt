using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    int x = -1;
    for (int i = 0, j = 0; i < n && j < n - 1; ++i)
    {
        if (a[i] <= b[j])
        {
            if (x == -1)
                x = b[j];
            else
            {
                cout << -1 << endl;
                return;
            }
        }
        else
            ++j;
    }
    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}

In the original code, the comparison logic was incorrect. The goal was to place each toy in a box where the box's size is not less than the toy's size. Therefore, the condition `if (a[i] > b[j])` should have been `if (a[i] <= b[j])`. Additionally, the sorting order should be ascending for both vectors to ensure that we can always find a suitable box for each toy.
