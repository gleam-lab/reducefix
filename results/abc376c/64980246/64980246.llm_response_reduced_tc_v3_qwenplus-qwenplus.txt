#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // We will use a multiset to efficiently find matching boxes
    multiset<int> box_sizes(b.begin(), b.end());
    
    int required_x = -1;

    for (int i = 0; i < n; ++i) {
        // Find the smallest box that can fit toy a[i]
        auto it = box_sizes.lower_bound(a[i]);
        
        if (it != box_sizes.end()) {
            // Box found, use it
            box_sizes.erase(it);
        } else {
            // No box can fit this toy
            if (required_x == -1) {
                required_x = a[i];
            } else {
                // Already had one required x, can't satisfy with just one extra box
                cout << -1 << endl;
                return;
            }
        }
    }

    cout << required_x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
