#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (int &i : a) cin >> i;
    for (int &i : b) cin >> i;
    
    // Use sets to find the maximum size required for the new box
    set<int> a_set(a.begin(), a.end());
    set<int> b_set(b.begin(), b.end());
    
    // The minimum size of the new box must be at least the maximum size of any toy
    int min_needed_size = *max_element(a.begin(), a.end());
    
    // Check if this size is already available in one of the boxes
    auto it = b_set.lower_bound(min_needed_size);
    if (it != b_set.end() && *it == min_needed_size) {
        cout << min_needed_size << endl;
        return;
    }
    
    // If not, find the smallest box size from the existing boxes that can accommodate the largest toy
    it = b_set.lower_bound(min_needed_size);
    if (it == b_set.begin()) {
        cout << -1 << endl;
        return;
    }
    
    // Move back one element to find the largest box that is smaller than the largest toy
    --it;
    cout << *it << endl;
}

int main() {
    int t = 1;
    cin >> t;
    while (t--) solve();
    return 0;
}
