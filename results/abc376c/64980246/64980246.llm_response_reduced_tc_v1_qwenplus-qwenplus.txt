#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // We will use a pointer approach to match toys to boxes
    int i = 0, j = 0;
    int x = -1;

    while (i < n && j < n - 1)
    {
        if (a[i] > b[j])
        {
            // This toy can't fit in any remaining box, must go to our purchased box
            if (x == -1)
                x = a[i];
            else
            {
                // We already had to put one toy in our box, can't do it again
                cout << -1 << endl;
                return;
            }
            i++;
        }
        else
        {
            // This toy fits in current box, so assign it here
            i++;
            j++;
        }
    }

    // If we have more than one toy left unassigned, we can't proceed
    if (i < n)
    {
        if (x != -1 || n - i > 1)
            cout << -1 << endl;
        else
            cout << a.back() << endl;
    }
    else
    {
        if (x == -1)
            x = a.back();
        cout << x << endl;
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
