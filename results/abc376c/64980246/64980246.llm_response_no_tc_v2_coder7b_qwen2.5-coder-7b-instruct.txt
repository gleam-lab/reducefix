```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    
    // Sort both arrays in ascending order
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    int x = -1;
    int j = 0;
    for (int i = 0; i < n; ++i) {
        // If the current toy cannot fit in the current box
        if (a[i] > b[j]) {
            // Check if we have already found a suitable box
            if (x != -1) {
                cout << -1 << endl;
                return;
            } else {
                // Mark this toy's size as the minimum required box size
                x = a[i];
                // Move to the next box since this one is too small
                ++j;
            }
        } else {
            // If the toy fits, move to the next toy
            ++i;
            // Move to the next box
            ++j;
        }
    }
    cout << x << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```

### Explanation of Fixes
1. **Sorting**: The original code incorrectly sorted the arrays in descending order. This was incorrect because it would lead to placing the largest toy in the smallest box, which is not allowed.
2. **Index Management**: The loop condition should correctly handle the traversal of both `a` and `b`.
3. **Edge Case Handling**: The code now properly handles the case where no suitable box can be found after iterating through the toys.
4. **Efficiency**: The solution remains efficient with a time complexity of \(O(N \log N)\) due to sorting, making it suitable for the problem constraints.
