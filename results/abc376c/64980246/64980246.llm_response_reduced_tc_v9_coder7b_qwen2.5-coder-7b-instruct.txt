The issue was that you were trying to place each toy into a box in descending order, which doesn't guarantee that the largest toy will fit into the largest available box. Instead, we need to sort the toys and boxes in ascending order and try to fit the smallest toy into the smallest available box. If at any point a toy cannot fit into any of the available boxes, then it's impossible to place all toys.

```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    int x = -1;
    for (int i = 0, j = 0; i < n && j < n - 1; ++i)
    {
        if (a[i] <= b[j])
        {
            if (x == -1)
                x = b[j];
            else
            {
                cout << -1 << endl;
                return;
            }
        }
        else
            ++j;
    }
    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
```

This code sorts both the toys and the boxes in ascending order and tries to place the smallest toy into the smallest available box. If at any point a toy cannot fit into any of the available boxes, it returns `-1`. Otherwise, it prints the size of the smallest box needed to fit all the toys.
