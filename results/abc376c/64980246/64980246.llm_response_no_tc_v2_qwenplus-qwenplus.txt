#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    // Sort toys and boxes in ascending order
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Try to fit the smallest toy into the smallest box
    int x = -1;
    multiset<int> b_set(b.begin(), b.end());

    for (int i = 0, j = 0; i < n; ++i)
    {
        auto it = b_set.lower_bound(a[i]);
        if (it != b_set.end())
        {
            b_set.erase(it); // Use this box
        }
        else
        {
            if (x == -1)
                x = a[i];
            else
            {
                cout << -1 << endl;
                return;
            }
        }
    }

    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
