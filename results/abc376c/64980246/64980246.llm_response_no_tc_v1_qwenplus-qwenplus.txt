#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve()
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (auto &tt : a)
        cin >> tt;
    for (auto &tt : b)
        cin >> tt;

    // Sort toys and boxes in descending order
    sort(a.begin(), a.end(), greater<int>());
    sort(b.begin(), b.end(), greater<int>());

    int x = -1;

    // Try to fit largest toys into largest available boxes
    int i = 0, j = 0;
    while (i < n && j < n - 1)
    {
        if (a[i] > b[j])
        {
            // This toy can't fit in any box - must go to the new box
            if (x == -1)
                x = a[i];
            else
            {
                // Can't have two toys needing special box
                cout << -1 << endl;
                return;
            }
            ++i;  // this toy is placed in the purchased box
        }
        else
        {
            // Place toy `a[i]` into box `b[j]`
            ++i;
            ++j;
        }
    }

    // If we've used all boxes and still have toys left, they need the new box
    while (i < n)
    {
        if (x == -1)
            x = a[i];
        else
        {
            cout << -1 << endl;
            return;
        }
        ++i;
    }

    cout << x << endl;
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
    {
        solve();
    }
    return 0;
}
