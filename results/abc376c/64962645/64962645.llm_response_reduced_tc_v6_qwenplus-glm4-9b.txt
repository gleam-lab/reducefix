#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    vector<int> va(n), vb(n - 1);
    for (int i = 0; i < n; i++)
        cin >> va[i];
    for (int i = 0; i < n - 1; i++)
        cin >> vb[i];
    
    // Pairing toy sizes with their original indices and sorting them
    vector<pair<int, int>> toy_sizes;
    for (int i = 0; i < n; i++)
        toy_sizes.emplace_back(va[i], i);
    sort(toy_sizes.begin(), toy_sizes.end());
    
    // Pairing box sizes with their original indices and sorting them
    vector<pair<int, int>> box_sizes;
    for (int i = 0; i < n - 1; i++)
        box_sizes.emplace_back(vb[i], i);
    sort(box_sizes.begin(), box_sizes.end());
    
    // Merging the two sorted vectors and checking conditions
    vector<pair<int, int>> merged;
    for (const auto& toy : toy_sizes)
        merged.emplace_back(toy.first, toy.second);
    for (const auto& box : box_sizes)
        merged.emplace_back(box.first, box.second);
    
    // Sort merged vector
    sort(merged.begin(), merged.end());
    
    // Check if the toys can fit in the boxes (with 1 extra space for the new box)
    bool can_fit = true;
    int last_box_size = -1; // No box has been placed yet
    for (const auto& item : merged)
    {
        if (item.second < n - 1) // This is a box
        {
            if (last_box_size > -1 && item.first < last_box_size)
                can_fit = false;
            last_box_size = item.first;
        }
        else // This is a toy
        {
            if (last_box_size == -1 || item.first <= last_box_size)
                last_box_size = -1; // Toy can be placed in the last box
            else
                can_fit = false; // No box is large enough for this toy
        }
    }
    
    // Output the result based on whether all toys can be placed
    if (can_fit)
        cout << merged.back().first + 1 << endl; // +1 for the new box
    else
        cout << -1 << endl;
}
