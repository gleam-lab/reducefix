#include <bits/stdc++.h>
using namespace std;

bool canFit(const vector<int>& sizes, int boxSize) {
    int currentBoxSize = 0;
    for (const auto& size : sizes) {
        if (size > boxSize) {
            return false;
        }
        currentBoxSize += size;
        if (currentBoxSize > boxSize) {
            return false;
        }
    }
    return true;
}
