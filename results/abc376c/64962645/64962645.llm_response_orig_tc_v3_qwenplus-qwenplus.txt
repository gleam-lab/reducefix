```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) boxSizes.insert(b);

    int extra_toys = 0;
    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            // We can place toy A[i] in a box
            boxSizes.erase(it);
        } else {
            // Cannot place this toy, need an extra box
            extra_toys++;
        }
    }

    if (extra_toys == 0) {
        cout << "-1\n";
    } else {
        // We need to find the smallest size x that can fit the largest unplaced toy
        // Sort A descending and find the (extra_toys-1)-th element
        sort(A.rbegin(), A.rend());
        cout << A[extra_toys - 1] << "\n";
    }

    return 0;
}
```
