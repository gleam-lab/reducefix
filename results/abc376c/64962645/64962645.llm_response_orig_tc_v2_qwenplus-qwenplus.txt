```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> b_set;
    for (int i = 0; i < N - 1; ++i) {
        b_set.insert(B[i]);
    }

    for (int i = 0; i < N; ++i) {
        auto it = b_set.lower_bound(A[i]);
        if (it != b_set.end()) {
            // Can use this box for current toy
            b_set.erase(it);
        } else {
            // Need to use the extra box
            cout << A[i] << '\n';
            return 0;
        }
    }

    // All toys can be placed without needing an extra box, but we must choose one
    // So output maximum possible toy size that could go into the extra box
    cout << "-1\n";
    return 0;
}
```
