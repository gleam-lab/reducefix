#include <bits/stdc++.h>
using namespace std;

bool canPlace(vector<int>& va, vector<int>& vb, int x) {
    priority_queue<int> pq;
    for(auto& b : vb) pq.push(b);
    for(auto& a : va) {
        while(!pq.empty() && pq.top() < a) pq.pop();
        if(pq.empty()) return false;
        pq.pop();
    }
    return true;
}

int main() {
    int n;
    cin >> n;
    vector<int> va(n), vb(n-1);
    for(int i=0; i<n; i++) cin >> va[i];
    for(int i=0; i<n-1; i++) cin >> vb[i];
    sort(va.begin(), va.end(), greater<int>());
    sort(vb.begin(), vb.end());
    int l = 0, r = 1e9 + 5, ans = -1;
    while(l <= r) {
        int mid = (l + r) / 2;
        if(canPlace(va, vb, mid)) {
            ans = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    cout << ans << endl;
    return 0;
}
