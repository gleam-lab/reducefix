#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    vector<int> va(n);
    vector<int> vb(n);
    for (int i = 0; i < n; i++)
        cin >> va[i];
    for (int i = 0; i < n - 1; i++)
        cin >> vb[i];
    // Sort the toy sizes and box sizes in descending order
    sort(va.rbegin(), va.rend());
    sort(vb.rbegin(), vb.rend());
    // If the largest toy is smaller than the largest box, we can directly place all toys
    if (va[0] <= vb[0]) {
        cout << vb[0] << endl;
        return 0;
    }
    // We need to find the smallest box size that can accommodate the largest toy
    // without exceeding the confines of the other boxes
    for (int i = 0; i < n - 1; i++) {
        // Check if the current box can accommodate the largest toy without exceeding
        // the size of the other boxes
        if (va[0] <= vb[i] && va[1] <= vb[i + 1]) {
            cout << vb[i] << endl;
            return 0;
        }
    }
    // If no such box exists, output -1
    cout << -1 << endl;
    return 0;
}
