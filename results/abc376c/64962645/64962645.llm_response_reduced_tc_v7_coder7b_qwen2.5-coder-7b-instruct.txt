#include <bits/stdc++.h>
using namespace std;

bool canPlaceAllToys(vector<int>& toys, vector<int>& boxes, int x) {
    int i = toys.size() - 1, j = boxes.size() - 1;
    while (i >= 0 && j >= 0) {
        if (toys[i] <= x && toys[i] <= boxes[j]) {
            --i;
        } else if (boxes[j] > x) {
            return false;
        } else {
            --j;
        }
    }
    return i == -1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<int> toys(n), boxes(n - 1);
    for (int i = 0; i < n; ++i) cin >> toys[i];
    for (int i = 0; i < n - 1; ++i) cin >> boxes[i];

    sort(toys.begin(), toys.end());
    sort(boxes.begin(), boxes.end());

    int left = toys.back(), right = toys.back();
    while (left < right) {
        int mid = (left + right) / 2;
        if (canPlaceAllToys(toys, boxes, mid)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }

    if (canPlaceAllToys(toys, boxes, left)) {
        cout << left << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
