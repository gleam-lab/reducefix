```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    // Sort both arrays in descending order
    sort(A.begin(), A.end(), greater<>());
    sort(B.begin(), B.end(), greater<>());

    // Try to fit the largest toy into the largest boxes
    // We need to find the minimal x such that we can place all toys

    // The idea is to match the smallest box + one extra box to the toys
    // We check where the toy is too big for the available box
    // and that will be the size of the needed additional box.

    long long required_size = -1;

    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) {
            required_size = A[i];
            break;
        }
    }

    // If required_size is not found in the loop, it means all toys up to N-2 fit
    // Then the last toy (smallest one) can go into the purchased box
    if (required_size == -1) {
        cout << A.back() << endl;
    } else {
        cout << required_size << endl;
    }

    return 0;
}
```
