```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N);
    for (int i = 0; i < N; ++i) cin >> A[i];

    vector<long long> B(N - 1);
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if it's possible to place N-1 toys in the existing boxes
    // If not, then we need to find which toy can't fit and needs the new box
    int i = 0, j = 0;
    int unmatched = 0;

    while (i < N && j < N - 1) {
        if (A[i] <= B[j]) {
            i++;
            j++;
        } else {
            unmatched++;
            i++;
        }
    }

    // If more than one toy can't be placed in existing boxes, return -1
    if (unmatched > 1) {
        cout << -1 << endl;
        return 0;
    }

    // The minimum size of the additional box needed
    if (unmatched == 0) {
        // All toys fit in existing boxes, we just need a box of size of the smallest unpaired toy
        cout << A[0] << endl;
        return 0;
    }

    // Find which toy needs the extra box
    i = 0, j = 0;
    while (i < N && j < N - 1) {
        if (A[i] <= B[j]) {
            i++;
            j++;
        } else {
            cout << A[i] << endl;
            return 0;
        }
    }

    // If we reach here, the largest toy needs the extra box
    cout << A[N - 1] << endl;
}
```
