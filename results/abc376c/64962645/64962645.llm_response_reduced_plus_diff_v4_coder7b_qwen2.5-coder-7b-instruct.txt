```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> va(n), vb(n-1);
    for(int i=0; i<n; ++i) cin >> va[i];
    for(int i=0; i<n-1; ++i) cin >> vb[i];

    sort(va.begin(), va.end(), greater<int>());
    sort(vb.begin(), vb.end());

    for(int i=0; i<n-1; ++i) {
        if(va[i] < vb[i]) {
            cout << -1 << endl;
            return 0;
        }
    }

    cout << va[n-1] << endl;

    return 0;
}
```
