#include <bits/stdc++.h>
using namespace std;

bool canPlace(vector<int>& toys, vector<int>& boxes, int size) {
    vector<bool> used(boxes.size(), false);
    for (int i = 0; i < toys.size(); ++i) {
        int idx = -1;
        for (int j = 0; j < boxes.size(); ++j) {
            if (!used[j] && boxes[j] >= toys[i]) {
                idx = j;
                break;
            }
        }
        if (idx == -1 || idx + 1 > size) return false;
        used[idx] = true;
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n;
    cin >> n;
    vector<int> toys(n), boxes(n - 1);
    for (int i = 0; i < n; ++i)
        cin >> toys[i];
    for (int i = 0; i < n - 1; ++i)
        cin >> boxes[i];

    sort(toys.begin(), toys.end());
    sort(boxes.begin(), boxes.end());

    int l = 0, r = n;
    while (l < r) {
        int m = (l + r) / 2;
        if (canPlace(toys, boxes, m))
            r = m;
        else
            l = m + 1;
    }

    cout << (l <= n ? l : -1) << endl;
    return 0;
}
