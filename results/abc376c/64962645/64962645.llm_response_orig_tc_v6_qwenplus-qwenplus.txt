```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.rbegin(), A.rend());
    sort(B.rbegin(), B.rend());

    // We need to find the minimal x such that we can assign each toy to a box:
    // - Each toy must fit in its box (A[i] <= box size)
    // - All boxes are distinct and used at most once

    // Try greedy matching:
    // For each toy from largest to smallest, try to assign to largest available box
    // If any toy cannot be assigned, then we need to put it in the new box.
    // The answer is the minimum size of that new box.

    int ptr = 0; // pointer to B array
    multiset<long long> extra_boxes;
    long long needed_box = -1;

    for (int i = 0; i < N; ++i) {
        // Add B[i] to extra_boxes if it exists
        if (ptr < N - 1) {
            extra_boxes.insert(B[ptr]);
            ptr++;
        }

        // Find smallest box >= A[i]
        auto it = extra_boxes.lower_bound(A[i]);

        if (it != extra_boxes.end()) {
            // Found a valid box, remove it
            extra_boxes.erase(it);
        } else {
            // Cannot store this toy in any existing box, need to use our purchased box
            needed_box = max(needed_box, A[i]);
        }
    }

    cout << (needed_box == -1 ? -1 : needed_box) << endl;

    return 0;
}
```
