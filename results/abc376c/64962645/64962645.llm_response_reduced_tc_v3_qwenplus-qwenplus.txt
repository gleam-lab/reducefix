#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if it's possible to assign toys to boxes without the extra box
    // We will try to match largest toy with largest available box or extra box

    int j = N - 2; // index for B (boxes), sorted ascending
    multiset<long long> ms(A.begin(), A.end()); // store toys in a multiset

    // Try to fit as many toys as possible into the largest boxes
    for (int i = N - 1; i >= 0 && j >= 0; --i) {
        auto it = ms.upper_bound(B[j]);
        if (it != ms.begin()) {
            --it;
            ms.erase(it);
            --j;
        }
    }

    if (ms.size() == 0) {
        cout << 0 << endl;
        return 0;
    }

    // If only one toy is left, that's the minimum size of additional box needed
    if (ms.size() == 1) {
        cout << *ms.begin() << endl;
        return 0;
    }

    // If more than one toy remains, it's not possible to place them in only one additional box
    cout << -1 << endl;

    return 0;
}
