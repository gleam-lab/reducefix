```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;
    std::vector<long long> A(n), B(n-1);
    for(int i = 0; i < n; ++i) std::cin >> A[i];
    for(int i = 0; i < n-1; ++i) std::cin >> B[i];

    std::sort(A.begin(), A.end());
    std::sort(B.begin(), B.end());

    for(int i = 0; i < n-1; ++i) {
        if(A[i] > B[i]) {
            std::cout << -1 << '\n';
            return 0;
        }
    }

    std::cout << A.back() << '\n';

    return 0;
}
```
