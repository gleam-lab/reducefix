```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) {
        boxSizes.insert(b);
    }

    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            // Can place toy A[i] in this box
            boxSizes.erase(it);
        } else {
            // Need a new box of size at least A[i]
            cout << A[i] << "\n";
            return 0;
        }
    }

    // All toys fit into existing boxes, which should not happen per constraints
    cout << "-1\n";
    return 0;
}
```
