
```cpp
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    vector<int> a(n);
    vector<int> b(n - 1);
    for (int i = 0; i < n; i++)
        cin >> a[i];
    for (int i = 0; i < n - 1; i++)
        cin >> b[i];

    // Sort toy sizes in descending order and box sizes in descending order
    sort(a.rbegin(), a.rend());
    sort(b.rbegin(), b.rend());

    // Check if there's a box that can fit the largest toy
    bool possible = false;
    for (int i = 0; i < n - 1; i++) {
        if (a[i] > b[i]) {
            possible = true;
            break;
        }
    }

    // If no box can fit the largest toy, output -1
    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    // Check for the minimum box size that can fit the second largest toy
    for (int i = 0; i < n - 2; i++) {
        if (a[i + 1] <= b[i]) {
            cout << a[i + 1] << endl;
            return 0;
        }
    }

    // If no box can fit the second largest toy, the answer is the size of the largest toy
    cout << a[0] << endl;
    return 0;
}
```
