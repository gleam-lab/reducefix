```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) {
        boxSizes.insert(b);
    }

    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            // We can place this toy in an existing box, so we remove the box
            boxSizes.erase(it);
        } else {
            // Need to use the new box for this toy
            if (i == N - 1) {
                cout << A[i] << "\n";
                return 0;
            } else {
                cout << -1 << "\n";
                return 0;
            }
        }
    }

    // All toys fit into existing boxes, no need for a new box
    cout << -1 << "\n";

    return 0;
}
```
