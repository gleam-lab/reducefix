```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> va(n), vb(n - 1);
    for (int i = 0; i < n; i++) cin >> va[i];
    for (int i = 0; i < n - 1; i++) cin >> vb[i];

    sort(va.begin(), va.end()); // Sort toys by size in ascending order
    sort(vb.begin(), vb.end()); // Sort boxes by size in ascending order

    int maxBoxSize = *max_element(vb.begin(), vb.end()); // Find the largest box size
    bool canStore = true;

    // Check if the largest box can accommodate all toys
    for (int i = 0; i < n && canStore; ++i) {
        if (va[i] > maxBoxSize) canStore = false;
    }

    // If the largest box cannot accommodate all toys, find the next largest box that can
    if (!canStore) {
        for (int i = n - 2; i >= 0 && !canStore; --i) {
            if (va.back() <= vb[i]) canStore = true;
        }
        if (!canStore) {
            cout << -1 << endl; // No suitable box found
            return 0;
        }
        maxBoxSize = vb[n - 2]; // Update maxBoxSize to the next largest box
    }

    cout << maxBoxSize << endl; // Print the minimum box size needed
    return 0;
}
```
