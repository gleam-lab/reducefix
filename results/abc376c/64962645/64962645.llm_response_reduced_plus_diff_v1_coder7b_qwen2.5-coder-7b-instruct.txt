#include <bits/stdc++.h>
using namespace std;

bool canPlaceToys(vector<int>& va, vector<int>& vb, long long x) {
    multiset<long long> boxes(vb.begin(), vb.end());
    for (long long toySize : va) {
        auto it = boxes.lower_bound(toySize);
        if (it == boxes.end()) {
            return false;
        }
        boxes.erase(it);
    }
    return true;
}

        if (canPlaceToys(va, vb, mid)) {
            ans = mid;
