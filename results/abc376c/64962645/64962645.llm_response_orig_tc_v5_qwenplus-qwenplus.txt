```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (auto &x : A) cin >> x;
    for (auto &x : B) cin >> x;

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) {
        boxSizes.insert(b);
    }

    // Try to place each toy in a suitable box
    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            // Found a suitable box
            boxSizes.erase(it);
        } else {
            // No suitable box found, we need a new one of size A[i]
            cout << A[i] << endl;
            return 0;
        }
    }

    // If we reach here, all toys could be placed without needing an extra box
    // But we must still output the minimum possible x that allows this configuration
    // Since no additional box is needed, the answer is not valid per problem statement
    // So we return -1
    cout << -1 << endl;

    return 0;
}
```
