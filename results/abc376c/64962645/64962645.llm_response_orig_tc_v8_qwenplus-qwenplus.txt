```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) boxSizes.insert(b);

    int used = 0;
    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            boxSizes.erase(it);
            used++;
        }
    }

    if (used == N - 1) {
        // All but one toy fit in existing boxes, so we need a box for the largest toy that couldn't fit
        cout << A[N - 1] << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
```
