```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    vector<long long> va(n), vb(n);
    
    for(int i = 0; i < n; ++i)
        cin >> va[i];
    
    for(int i = 0; i < n - 1; ++i)
        cin >> vb[i];

    sort(va.rbegin(), va.rend()); // Sort in descending order
    sort(vb.begin(), vb.end());

    long long maxBoxSize = 0;
    for(int i = 0; i < n - 1; ++i) {
        if(vb[i] >= va[i]) {
            maxBoxSize = max(maxBoxSize, vb[i]);
        } else {
            break;
        }
    }

    if(maxBoxSize >= va[n - 1])
        cout << maxBoxSize << '\n';
    else
        cout << -1 << '\n';

    return 0;
}
```

This solution first sorts the arrays `va` and `vb`. It then iterates through `vb`, checking if the current box size can accommodate the largest remaining toy (`va`). If at any point a box cannot accommodate the next largest toy, it breaks out of the loop and outputs `-1`. If it successfully finds a suitable box size for all toys, it prints that size.
