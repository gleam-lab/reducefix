```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> a(n), b(n);
    for(int i = 0; i < n; ++i)
        cin >> a[i];
    for(int i = 0; i < n - 1; ++i)
        cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    long long max_box_size = -1;
    for(int i = 0; i < n - 1; ++i) {
        if(b[i] >= a[i]) {
            max_box_size = max(max_box_size, b[i]);
        } else {
            max_box_size = -1;
            break;
        }
    }

    if(max_box_size == -1) {
        cout << max_box_size << endl;
    } else {
        cout << max(max_box_size, a[n - 1]) << endl;
    }

    return 0;
}
```
