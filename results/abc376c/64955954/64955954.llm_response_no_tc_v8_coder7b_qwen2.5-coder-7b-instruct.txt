```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> a(n), b(n - 1);

    for (int i = 0; i < n; ++i)
        cin >> a[i];

    for (int i = 0; i < n - 1; ++i)
        cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    long long max_size = *max_element(a.begin(), a.end());
    for (long long x = max_size; x >= 1; --x) {
        int j = 0;
        for (int i = 0; i < n - 1 && j < n; ++i) {
            while (j < n && a[j] <= x)
                ++j;
            if (j == n || b[i] < a[j]) {
                cout << x << '\n';
                return 0;
            }
        }
    }

    cout << -1 << '\n';
    return 0;
}
```
