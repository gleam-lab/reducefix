```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> b_set(B.begin(), B.end());

    // Try to assign each toy to a box
    for (int i = 0; i < N; ++i) {
        // Find the smallest box that can fit A[i]
        auto it = b_set.lower_bound(A[i]);
        if (it == b_set.end()) {
            // No suitable box found, this toy must go into the new box
            // So x must be at least A[i], and we need to check the rest can be assigned
            cout << A[i] << "\n";
            return 0;
        }
        b_set.erase(it); // Assign this box to the current toy
    }

    // If we reach here, all toys could be placed in existing boxes, which shouldn't happen
    // as we have N toys and only N-1 boxes without the extra one.
    // So theoretically unreachable unless there's a bug.
    cout << "-1\n";
    return 0;
}
```
