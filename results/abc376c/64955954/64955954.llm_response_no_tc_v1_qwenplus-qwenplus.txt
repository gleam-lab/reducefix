```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<long long> B(N - 1);
    for (int i = 0; i < N - 1; ++i) {
        cin >> B[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to assign N toys to N boxes: N-1 existing and one new box of size x.
    // The idea is to try placing the largest toy in the new box x, and see what's the minimal x.

    // We try to match A[0], A[1], ..., A[i-1], A[i+1], ... A[N-1] with B[0], B[1], ..., B[N-2]
    // So we leave out A[i] to be placed in the new box x = A[i]

    // Find the smallest A[i] such that all other A[j] can be matched with B[k] where A[j] <= B[k]

    // To do this efficiently:
    // For each i from 0 to N-1:
    //   compare A[0..i-1] + A[i+1..N-1] with B[0..N-2]
    //   if A[j] <= B[j] for all j != i, then x = A[i] is valid

    // We can do this efficiently by pre-checking prefix and suffix

    vector<bool> prefix(N), suffix(N);

    // prefix[i] == true means A[0..i] can be matched with B[0..i]
    for (int i = 0; i < N - 1; ++i) {
        if (A[i] > B[i]) break;
        prefix[i] = true;
    }

    // suffix[i] == true means A[i..N-1] can be matched with B[i-1..N-2]
    for (int i = N - 1; i >= 1; --i) {
        if (A[i] > B[i - 1]) break;
        suffix[i] = true;
    }

    // Now check for any i where prefix[i-1] and suffix[i+1] are true
    // That means A[0..i-1] matches B[0..i-1], A[i+1..N-1] matches B[i..N-2]
    // Then A[i] is the only unmatched toy which will go into the new box

    long long min_x = -1;
    if (suffix[N - 1]) { // i = 0
        min_x = A[0];
    }
    if (prefix[N - 2]) { // i = N-1
        if (min_x == -1 || A[N - 1] < min_x)
            min_x = A[N - 1];
    }

    for (int i = 1; i < N - 1; ++i) {
        if (prefix[i - 1] && suffix[i + 1]) {
            if (min_x == -1 || A[i] < min_x)
                min_x = A[i];
        }
    }

    cout << min_x << endl;

    return 0;
}
```
