```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Use a multiset to efficiently manage available boxes
    multiset<long long> box_sizes;
    for (long long b : B) box_sizes.insert(b);

    for (int i = 0; i < N; ++i) {
        // Find the smallest box that can fit toy A[i]
        auto it = box_sizes.lower_bound(A[i]);
        if (it == box_sizes.end()) {
            // No suitable box found, need to buy one
            cout << A[i] << "\n";
            return 0;
        }
        // Use this box
        box_sizes.erase(it);
    }

    // All toys could be placed in existing boxes
    cout << "-1\n";
    return 0;
}
```
