```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to match toys to boxes.
    // One toy will go into the new box of size x.
    // The rest must be matched to boxes in B such that each toy fits.

    // Try to greedily match smallest toys to smallest boxes

    // Count how many toys can fit in each box
    multiset<long long> boxSizes;
    for (long long b : B) boxSizes.insert(b);

    // Try to assign toys to boxes
    for (int i = 0; i < N; ++i) {
        // Find first box in multiset >= A[i]
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            boxSizes.erase(it); // Use this box
        } else {
            // Can't place this toy, so we need to put it in the new box
            cout << A[i] << "\n";
            return 0;
        }
    }

    // If we reach here, all toys were placed without needing the extra box
    // which is impossible since we must have one extra box
    cout << "-1\n";
    return 0;
}
```
