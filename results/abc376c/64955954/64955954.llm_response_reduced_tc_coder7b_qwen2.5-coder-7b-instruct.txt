#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> A(n), B(n-1);
    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n-1; ++i) cin >> B[i];
    
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());
    
    int x = -1;
    for (int i = 0; i < n-1; ++i) {
        if (A[i] > B[i]) {
            x = max(x, A[i]);
        }
    }
    if (x == -1) {
        x = A[n-1];
    } else {
        bool found = false;
        for (int i = 0; i < n-1; ++i) {
            if (B[i] >= x) {
                found = true;
                break;
            }
        }
        if (!found) x = A[n-1];
    }
    
    if (x == -1 || (x > B.back())) {
        cout << -1 << endl;
    } else {
        cout << x << endl;
    }
}
