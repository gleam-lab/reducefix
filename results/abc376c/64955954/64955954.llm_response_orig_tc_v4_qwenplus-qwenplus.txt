```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> box_sizes;
    int j = 0;
    for (int i = 0; i < N - 1; ++i) {
        while (j < N && A[j] <= B[i]) {
            box_sizes.insert(A[j]);
            j++;
        }
        if (box_sizes.empty()) {
            cout << -1 << endl;
            return 0;
        }
        // Use the largest possible toy that fits in this box to save smaller toys for bigger boxes
        box_sizes.erase(prev(box_sizes.end()));
    }

    // Now, one of the remaining toys can go into the new box
    // The minimal x is the smallest toy that hasn't been placed yet
    cout << A[j] << endl;

    return 0;
}
```
