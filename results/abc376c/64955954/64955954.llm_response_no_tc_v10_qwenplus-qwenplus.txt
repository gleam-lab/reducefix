#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N);
    vector<long long> B(N - 1);

    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Check if we can fit the smallest N-1 toys in the boxes without using the extra box
    // and assign the largest toy to the extra box.
    // If at any point B[i] < A[i], then we need an extra box of size A[N - 1]

    long long required_x = A.back();  // Start with the biggest toy as candidate for x
    bool possible = true;

    for (int i = 0; i < N - 1; ++i) {
        if (B[i] < A[i]) {
            possible = false;
            break;
        }
    }

    if (possible) {
        // We might be able to do better than A.back()
        // Find the minimal x such that after inserting x into B, we can match all A sorted
        // Binary search on x

        long long low = 1, high = *max_element(A.begin(), A.end());
        while (low < high) {
            long long mid = (low + high) / 2;

            // Create a temporary B' including mid
            vector<long long> tempB = B;
            tempB.push_back(mid);
            sort(tempB.begin(), tempB.end());

            bool ok = true;
            for (int i = 0; i < N; ++i) {
                if (tempB[i] < A[i]) {
                    ok = false;
                    break;
                }
            }

            if (ok) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        cout << low << endl;
    } else {
        // Even without adding a box, the N-1 smallest toys cannot be placed in the existing boxes
        cout << -1 << endl;
    }

    return 0;
}
