```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> box_sizes;
    int j = 0;
    for (int i = 0; i < N - 1; ++i) {
        while (j < N && A[j] <= B[i]) {
            box_sizes.insert(A[j]);
            j++;
        }
        if (!box_sizes.empty()) {
            box_sizes.erase(box_sizes.begin()); // Use the smallest possible toy for this box
        } else {
            cout << -1 << endl;
            return 0;
        }
    }

    // After placing N-1 toys in boxes, one toy remains
    if (j < N) {
        cout << A[j] << endl;
    } else if (!box_sizes.empty()) {
        cout << *box_sizes.begin() << endl;
    } else {
        cout << -1 << endl;
    }

    return 0;
}
```
