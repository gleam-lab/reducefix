```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<long long> A(N), B(N - 1);
    for (int i = 0; i < N; ++i) cin >> A[i];
    for (int i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> boxSizes;
    for (long long b : B) {
        boxSizes.insert(b);
    }

    // Try to assign the smallest possible toys to boxes first
    for (int i = 0; i < N; ++i) {
        auto it = boxSizes.lower_bound(A[i]);
        if (it != boxSizes.end()) {
            // Found a suitable box, remove it
            boxSizes.erase(it);
        } else {
            // Cannot place this toy in any existing box, must go to new box
            cout << A[i] << '\n';
            return 0;
        }
    }

    // All toys can be placed in existing boxes, so no need for new box
    cout << -1 << '\n';
    return 0;
}
```
