```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> a(n), b(n-1);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < n-1; ++i) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    long long minBoxSize = -1;
    for(int i = 0; i < n-1; ++i) {
        if(b[i] >= a[i+1]) {
            minBoxSize = max(minBoxSize, a[i+1]);
        } else {
            break;
        }
    }

    if(minBoxSize == -1) {
        if(b[n-2] >= a[0]) minBoxSize = a[0];
        else minBoxSize = -1;
    }

    cout << minBoxSize << endl;

    return 0;
}
```
