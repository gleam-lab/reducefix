```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n-1);
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=0;i<n-1;i++)cin>>b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int j = 0;
    for(int i = 0; i < n; ++i){
        while(j < n-1 && a[i] > b[j])++j;
        if(j == n-1 && a[i] > b[j])break;
    }

    if(j == n-1)cout<<-1<<endl;
    else cout<<max(a[n-1], b[j])<<endl;

    return 0;
}
```
