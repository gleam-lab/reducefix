bool canStoreAllToys(vector<int>& toys, vector<int>& boxes) {
    sort(toys.begin(), toys.end());
    sort(boxes.begin(), boxes.end());
    
    int i = 0, j = 0;
    while (i < toys.size() && j < boxes.size()) {
        if (toys[i] <= boxes[j]) {
            i++;
        } else {
            j++;
        }
    }
    return i == toys.size();
}
