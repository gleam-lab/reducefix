```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N;
    cin >> N;
    vector<long long> A(N), B(N - 1);
    for (auto &x : A) cin >> x;
    for (auto &x : B) cin >> x;

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<long long> box_sizes;
    int j = 0;
    for (int i = 0; i < N - 1; ++i) {
        while (j < N && A[j] <= B[i]) {
            box_sizes.insert(A[j]);
            ++j;
        }
        if (box_sizes.empty()) {
            cout << -1 << endl;
            return 0;
        }
        box_sizes.erase(box_sizes.begin()); // use smallest available toy
    }

    // After assigning toys to all boxes, check if any toy remains
    if (j >= N) {
        cout << -1 << endl;
        return 0;
    }

    // Find the minimal x such that remaining toy can fit in new box
    long long min_x = A[j];
    cout << min_x << endl;
    return 0;
}
```
