#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // We need to find the smallest box size x such that all toys can be placed.
    // x must be larger than the largest toy size, which is a[n-1].
    // We need to find the smallest x such that:
    // - For all i from 0 to n-2: b[i] >= a[i]
    // - b[n-2] >= a[n-1]
    // We can achieve this by using a binary search for the smallest x.

    int low = 0;
    int high = a[n - 1]; // The largest toy size is the upper bound.

    while (low < high) {
        int mid = low + (high - low) / 2;
        bool canPlaceAllToys = true;

        // Check if all toys can be placed with box size of mid.
        for (int i = 0; i < n - 1; ++i) {
            if (b[i] < a[i] || b[n - 2] < a[n - 1]) {
                canPlaceAllToys = false;
                break;
            }
        }

        if (canPlaceAllToys) {
            high = mid; // Try a smaller box size.
        } else {
            low = mid + 1; // Increase the box size.
        }
    }

    // After the loop, low must be the smallest possible value of x.
    cout << low << endl;

    return 0;
}
