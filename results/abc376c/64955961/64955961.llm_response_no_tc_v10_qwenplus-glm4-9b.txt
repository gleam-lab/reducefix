#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int solve() {
    int n;
    cin >> n;
    vector<long long> a(n);
    vector<long long> b(n - 1);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n - 1; ++i) {
        cin >> b[i];
    }
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    long long min_x = 0;
    long long last_box_size = 0;
    int valid_boxes = 0;
    
    for (int i = 0; i < n; ++i) {
        if (a[i] > last_box_size) {
            if (valid_boxes > 0) {
                min_x = max(min_x, a[i]);
            } else {
                // No valid boxes found yet
                min_x = a[i];
            }
            valid_boxes = 0; // Reset since we found a new toy size that needs new box
        } else {
            // Use existing boxes for the toy
            valid_boxes++;
            if (a[i] > b[valid_boxes - 1]) {
                // If the toy is larger than the last used box size, it needs a new box
                if (valid_boxes == 1 && a[i] > b[0]) {
                    // If this is the first box we're considering, it means no single box can contain this toy, so we need a new box
                    min_x = a[i];
                }
            } else {
                // If the toy can fit into one of the smaller boxes we've already considered, update last_box_size
                last_box_size = b[valid_boxes - 1];
            }
        }
    }
    
    if (min_x > 0) {
        cout << min_x << endl;
    } else {
        cout << -1 << endl;
    }
    
    return 0;
}

int main() {
    solve();
    return 0;
}
