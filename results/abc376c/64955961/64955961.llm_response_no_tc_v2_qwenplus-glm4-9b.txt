#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> A(n), B(n - 1);
    for (int i = 0; i < n; i++) {
        cin >> A[i];
    }
    for (int i = 0; i < n - 1; i++) {
        cin >> B[i];
    }

    // Sort both arrays to properly handle the conditions
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find the minimum size x that allows us to place all toys
    // We have n-1 boxes and we need to add at least one box of size x
    // The strategy is to try placing the toys in the existing boxes first
    // and then decide if we need a new box of size x

    // We iterate from the smallest to the largest toy size
    for (int i = 0; i < n; i++) {
        int foundBox = false;
        // Check each box to see if the current toy can fit
        for (int j = 0; j < n - 1; j++) {
            if (A[i] <= B[j]) {
                foundBox = true;
                break;
            }
        }
        // If we cannot find a box for the current toy, we need a new box
        if (!foundBox) {
            // The size of the new box must be at least the size of the current toy
            // We are looking for the minimum such size
            cout << A[i] << endl;
            return 0;
        }
    }

    // If all toys can fit in the existing boxes, then the minimum size x is 1
    // because we could place any toy in any box (as long as it fits)
    cout << 1 << endl;
    return 0;
}
