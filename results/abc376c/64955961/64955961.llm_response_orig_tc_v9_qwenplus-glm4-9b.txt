1. Store the maximum toy size in `mx`.
2. Iterate through the boxes. For each box, if a toy's size is larger than the box size, then `x` must be at least as large as the toy size.
3. After the iteration, `x` should be the maximum of `mx` and the sizes of the boxes that couldn't fit any toy to ensure that all toys can be placed, and no larger box is needed.

#include<bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> a(n), b(n);
    for (ll i = 0; i < n; i++) {
        cin >> a[i];
    }
    for (ll i = 0; i < n - 1; i++) {
        cin >> b[i];
    }
    ll mx = 0;
    for (ll i = 0; i < n - 1; i++) {
        for (ll j = 0; j < n; j++) {

                mx = max(mx, a[j]);
            }
        }

}

}
