#include<bits/stdc++.h>
using namespace std;
#define ll long long

void solve()
{
    ll n;
    cin >> n;
    vector<ll> a(n), b(n-1);
    for(ll i = 0; i < n; i++) cin >> a[i];
    for(ll i = 0; i < n-1; i++) cin >> b[i];
    
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    ll i = n-1, j = n-2, mx = 0;
    while(i >= 0 && j >= 0){
        if(a[i] <= b[j]){
            i--; j--;
        }else{
            mx = max(mx, a[i]);
            i--;
        }
    }
    if(i == 0) cout << mx << endl;
    else cout << -1 << endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}

This solution corrects the logic of the original code by using a greedy approach. It iterates through both arrays from the end, trying to place each toy into a box that can accommodate it without exceeding the box's capacity. If a toy cannot fit into any remaining box, it records the toy's size as a potential answer. Finally, if there is at least one toy left after placing them all, it outputs the maximum recorded toy size; otherwise, it outputs `-1`. This ensures that we always use the smallest possible box size to store all toys.
