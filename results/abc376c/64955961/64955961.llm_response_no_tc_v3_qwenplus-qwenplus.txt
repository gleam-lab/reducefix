#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll N;
    cin >> N;

    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find the smallest x such that we can assign all toys to boxes
    // with one toy per box, and each toy fits in its assigned box (toy_size <= box_size)
    // One of the boxes is our purchased box of size x.

    // Greedy strategy:
    // - Assign the largest toy to the largest available box or our purchased box
    // - If the toy fits in the current B box, use it
    // - Else, we must use our own box. Only one such assignment is allowed.

    ll need_x = -1;
    ll j = N - 2; // index for B (boxes)

    for (ll i = N - 1; i >= 0; --i) {
        if (j >= 0 && B[j] >= A[i]) {
            j--; // use this box
        } else {
            // Need to use our box, only allow once
            if (need_x != -1) {
                cout << -1 << endl;
                return;
            }
            need_x = A[i];
        }
    }

    if (need_x == -1) {
        // All toys fit in existing boxes
        cout << 0 << endl;
    } else {
        cout << need_x << endl;
    }
}

int main() {
    solve();
}
