#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

bool check(vector<ll>& a, vector<ll>& b, ll x){
    priority_queue<pair<ll,ll>,vector<pair<ll,ll>>,greater<pair<ll,ll>>> pq;
    for(auto &i:a)pq.push({i,i});
    for(auto &i:b)pq.push({i,i});
    while(!pq.empty()){
        auto [s,c]=pq.top(); pq.pop();
        if(s>x)return false;
        c++;
        s+=c;
        pq.push({s,c});
    }
    return true;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0),cout.tie(0);
    ll n;
    cin >> n;
    vector<ll> a(n),b(n-1);
    for(ll i=0;i<n;i++)cin >> a[i];
    for(ll i=0;i<n-1;i++)cin >> b[i];
    sort(a.begin(),a.end());
    sort(b.begin(),b.end());
    ll l=1,r=*max_element(a.begin(),a.end())+*max_element(b.begin(),b.end()),mid,ans=-1;
    while(l<=r){
        mid=(l+r)>>1;
        if(check(a,b,mid)){
            r=mid-1;
            ans=mid;
        }else l=mid+1;
    }
    cout << ans << endl;
    return 0;
}
