
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> A(n), B(n - 1);
    for (int i = 0; i < n; i++) cin >> A[i];
    for (int i = 0; i < n - 1; i++) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    int min_x = 0;
    int i = n - 1;
    int j = n - 2;
    while (i >= 0 && j >= 0) {
        if (A[i] <= B[j]) {
            i--;
        } else {
            min_x = A[i];
            i--;
            j--;
        }
    }

    // If min_x is still 0, it means no valid x could be found that fits all toys.
    if (min_x == 0) {
        cout << -1 << endl;
    } else {
        cout << min_x << endl;
    }

    return 0;
}
```
