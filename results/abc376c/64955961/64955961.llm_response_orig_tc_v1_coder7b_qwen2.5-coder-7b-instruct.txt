#include <bits/stdc++.h>
using namespace std;

bool canPlaceAllToys(vector<int>& toys, vector<int>& boxes, int maxSize) {
    sort(toys.begin(), toys.end());
    sort(boxes.begin(), boxes.end());

    int toyIndex = 0, boxIndex = 0;
    while (toyIndex < toys.size() && boxIndex < boxes.size()) {
        if (toys[toyIndex] <= maxSize && toys[toyIndex] <= boxes[boxIndex]) {
            toyIndex++;
        } else if (toys[toyIndex] > maxSize) {
            return false;
        }
        boxIndex++;
    }

    // Check if all toys are placed
    return toyIndex == toys.size();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<int> toys(N), boxes(N - 1);
    for (int i = 0; i < N; ++i) cin >> toys[i];
    for (int i = 0; i < N - 1; ++i) cin >> boxes[i];

    int left = *max_element(toys.begin(), toys.end()), right = 1e9 + 1;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (canPlaceAllToys(toys, boxes, mid)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }

    cout << (left > *max_element(toys.begin(), toys.end()) ? -1 : left) << endl;

    return 0;
}
