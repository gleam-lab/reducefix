2. **Greedy Matching**: Iterate through the toys and boxes to find the largest toy that cannot be matched to any of the existing boxes. The idea is that the smallest possible `x` is the size of the largest unmatched toy.
3. **Check Validity**: Ensure that only one toy remains unmatched after attempting to match all others to the existing boxes. If more than one toy remains unmatched, it's impossible to place all toys in separate boxes, and we should return `-1`.

#include<bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> a(n), b(n-1);
    for (ll i = 0; i < n; i++) {
        cin >> a[i];
    }
    for (ll i = 0; i < n-1; i++) {
        cin >> b[i];
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    ll unmatched = 0;
    ll x = -1;
    ll i = 0, j = 0;
    while (i < n && j < n-1) {
        if (a[i] <= b[j]) {
            i++;
            j++;
        } else {
            if (x == -1) {
                x = a[i];
                unmatched++;
                i++;
            } else {
                unmatched++;
                i++;
            }
        }
    }
    while (i < n) {
        if (x == -1) {
            x = a[i];
            unmatched++;
        } else {
            unmatched++;
        }
        i++;
    }
    
    if (unmatched > 1) {
        cout << -1 << endl;
    } else {
        cout << x << endl;
    }
}

int main() {
    solve();
    return 0;
}

### Explanation
- **Sorting**: Both the toy sizes (`a`) and box sizes (`b`) are sorted to facilitate a greedy matching approach.
- **Greedy Matching**: The algorithm iterates through the sorted arrays, attempting to match each toy to a box of sufficient size. If a toy cannot be matched, it is considered as a candidate for the new box (`x`).
- **Unmatched Toys**: The algorithm checks if there's exactly one unmatched toy, which would be the size of the new box (`x`). If there are more than one unmatched toys, it's impossible to place all toys, and the output is `-1`.
