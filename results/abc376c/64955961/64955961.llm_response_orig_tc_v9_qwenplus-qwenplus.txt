#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);
    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> boxSizes;
    for (ll b : B) boxSizes.insert(b);

    ll requiredBox = -1;

    // Try to assign toys from largest to smallest
    for (ll i = n - 1; i >= 0; --i) {
        auto it = boxSizes.upper_bound(A[i]);
        if (it != boxSizes.begin()) {
            --it;
            boxSizes.erase(it);
        } else {
            // No suitable existing box, need a new one
            if (requiredBox == -1) {
                requiredBox = A[i];
            } else {
                // Already used a new box, can't use more than one
                cout << -1 << endl;
                return;
            }
        }
    }

    if (requiredBox == -1) {
        // All toys fit in existing boxes
        cout << *min_element(A.begin(), A.end()) << endl;
    } else {
        cout << requiredBox << endl;
    }
}

int main() {
    solve();
    return 0;
}
