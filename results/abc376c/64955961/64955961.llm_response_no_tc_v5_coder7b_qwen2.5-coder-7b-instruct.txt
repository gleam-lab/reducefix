```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> a(n), b(n - 1);
    for (ll i = 0; i < n; ++i) cin >> a[i];
    for (ll i = 0; i < n - 1; ++i) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    ll max_b = *max_element(b.begin(), b.end());
    if (a.back() > max_b) {
        cout << -1 << endl;
        return;
    }

    ll ans = 0;
    for (ll i = 0, j = 0; i < n && j < n - 1;) {
        if (a[i] <= b[j]) {
            ++i;
            ++j;
        } else {
            ans = a[i];
            ++i;
        }
    }

    if (ans == 0) {
        cout << -1 << endl;
    } else {
        cout << ans << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    solve();

    return 0;
}
```
