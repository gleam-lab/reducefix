#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Use binary search to find the minimum required size of the extra box
    ll left = 0, right = n - 1;
    ll ans = -1;

    // We want to find how many toys can be placed in the available boxes
    ll j = 0;
    ll matched = 0;

    for (ll i = 0; i < n; ++i) {
        if (j < n - 1 && A[i] <= B[j]) {
            ++matched;
            ++j;
        }
    }

    if (matched == n - 1) {
        // One toy will go into the new box, which is the largest unmatched one
        // So the answer is the largest unmatched toy
        cout << A.back() << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
