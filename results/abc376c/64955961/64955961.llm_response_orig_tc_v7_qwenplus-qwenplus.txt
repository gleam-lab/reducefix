#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    multiset<ll> boxSizes;
    ll j = 0;
    for (ll i = 0; i < n; ++i) {
        while (j < n - 1 && B[j] < A[i]) {
            boxSizes.insert(B[j]);
            j++;
        }
        if (j < n - 1) {
            j++; // use one box of size >= A[i]
        } else if (!boxSizes.empty()) {
            auto it = boxSizes.lower_bound(A[i]);
            if (it != boxSizes.end()) {
                boxSizes.erase(it); // use a previously stored box
            } else {
                cout << -1 << endl;
                return;
            }
        } else {
            cout << -1 << endl;
            return;
        }
    }

    // The smallest x that can be used to fit the largest toy not covered by existing boxes
    cout << A.back() << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
