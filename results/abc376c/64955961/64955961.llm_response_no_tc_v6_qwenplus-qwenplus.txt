#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve() {
    ll N;
    cin >> N;
    
    vector<ll> A(N), B(N - 1);
    for (ll i = 0; i < N; ++i) cin >> A[i];
    for (ll i = 0; i < N - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find the smallest x such that we can assign all toys to boxes
    // with one toy per box, and each toy fits in its assigned box.

    // The purchased box will replace the one toy that cannot be matched greedily

    ll required = -1;
    ll j = 0;
    for (ll i = 0; i < N && j < N - 1; ++i) {
        if (A[i] <= B[j]) {
            ++j;  // this toy fits in current box, use it
        } else {
            // this toy is too big for current box, we'll need a special box for it
            // keep track of the largest unmatched toy
            required = A[i];
            ++i;  // skip this toy (we'll handle it with the new box)
        }
    }

    // If we have matched all boxes, then we can just use the biggest toy as fallback
    if (j == N - 1) {
        cout << A[N - 1] << endl;
    } else {
        cout << -1 << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
