```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n-1);
    for(int i = 0; i < n; ++i) cin >> a[i];
    for(int i = 0; i < n-1; ++i) cin >> b[i];

    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    int i = 0, j = 0;
    while(i < n && j < n-1) {
        if(a[i] <= b[j]) {
            ++i;
        } else {
            ++j;
        }
    }

    if(i == n) cout << a.back() << endl;
    else cout << -1 << endl;

    return 0;
}
```
