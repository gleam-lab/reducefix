#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve() {
    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);

    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Use binary search to find the minimum required x
    ll left = 0, right = n - 1;
    ll answer = -1;

    // We want to find the smallest toy that cannot be placed in any box
    // We compare largest toys with largest boxes
    multiset<ll> boxes(B.begin(), B.end());

    // Try to place the toys from largest to smallest
    for (ll i = n - 1; i >= 0; --i) {
        auto it = boxes.lower_bound(A[i]);
        if (it != boxes.end()) {
            // Found a suitable box, use it
            boxes.erase(it);
        } else {
            // Cannot place this toy, must go to our purchased box
            if (answer == -1) {
                answer = A[i];
            } else {
                // If more than one toy can't be placed, not possible
                cout << -1 << endl;
                return;
            }
        }
    }

    if (answer == -1) {
        // All toys could be placed, so we don't need to buy a box
        // But according to problem statement, we must buy one box
        // So we can just buy a box of size 1 (minimum possible)
        cout << 1 << endl;
    } else {
        cout << answer << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
