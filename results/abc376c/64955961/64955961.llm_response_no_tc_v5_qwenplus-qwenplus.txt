#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve() {
    ll n;
    cin >> n;
    vector<ll> A(n), B(n - 1);
    for (ll i = 0; i < n; ++i) cin >> A[i];
    for (ll i = 0; i < n - 1; ++i) cin >> B[i];

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // We need to find the smallest x such that we can assign all toys to boxes
    // Try to match largest toys with largest available boxes, and see what's left

    ll j = n - 2; // index for B (boxes)
    ll required_box_size = -1;

    // Go from biggest toy to smallest
    for (ll i = n - 1; i >= 0; --i) {
        if (j >= 0 && A[i] <= B[j]) {
            // This box can fit this toy - use it
            j--;
        } else {
            // Can't fit this toy in any unused box - this is the one we must put in our custom box
            if (required_box_size != -1) {
                // More than one toy can't be placed -> impossible
                cout << -1 << endl;
                return;
            }
            required_box_size = A[i];
        }
    }

    if (required_box_size == -1) {
        // All toys could be placed in existing boxes, so we don't need an extra one
        cout << 1 << endl; // minimum possible x is 1
    } else {
        cout << required_box_size << endl;
    }
}

int main() {
    solve();
}
