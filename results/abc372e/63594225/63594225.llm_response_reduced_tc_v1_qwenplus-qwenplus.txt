#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

// We'll maintain each DSU set's members in a sorted vector,
// but to make merging efficient, we'll always merge smaller into larger.
vector<vector<int>> components;
vector<int> parent;

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return;

    // Always merge smaller component into larger for efficiency
    if (components[a].size() < components[b].size())
        swap(a, b);

    // Merge the smaller vector into the larger one
    for (int x : components[b]) {
        components[a].push_back(x);
    }

    // Maintain sorted order by sorting and keeping only top 10 elements
    sort(components[a].rbegin(), components[a].rend()); // Sort descending
    while (components[a].size() > 100) {
        components[a].pop_back();
    }

    parent[b] = a;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    components.resize(n+1);
    parent.resize(n+1);
    
    for (int i = 1; i <= n; ++i) {
        parent[i] = i;
        components[i] = {i};
    }

    while (q--) {
        int t, u, v;
        cin >> t >> u >> v;
        if (t == 1) {
            unite(u, v);
        } else {
            int rep = find(u);
            if (components[rep].size() < (size_t)v) {
                cout << -1 << '\n';
            } else {
                cout << components[rep][v-1] << '\n';
            }
        }
    }

    return 0;
}
