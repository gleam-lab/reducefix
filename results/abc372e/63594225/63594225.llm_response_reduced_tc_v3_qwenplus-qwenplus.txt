#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

// Instead of storing only top 100 elements, we'll store all elements in a sorted set
vector <set<int>> components;

vector <int> p;

int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    
    // Always merge smaller set into larger set to optimize time complexity
    if (components[a].size() < components[b].size()) 
        swap(a, b);

    // Merge the sets
    for (auto val : components[b]) {
        components[a].insert(val);
    }

    // Update parent
    p[b] = a;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    p.resize(n+1);
    components.resize(n+1);
    for(int i = 1; i <= n; i++) {
        p[i] = i;
        components[i].insert(i);
    }
    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if(t == 1) {
            uni(a, b);
        }
        else {
            int root = par(a);
            if ((int)components[root].size() < b) {
                cout << -1 << '\n';
            } else {
                // Get k-th largest using reverse iterator
                auto it = components[root].rbegin();
                advance(it, b-1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
