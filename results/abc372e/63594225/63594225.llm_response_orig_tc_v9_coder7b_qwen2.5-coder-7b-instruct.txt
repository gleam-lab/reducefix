#include <bits/stdc++.h>

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

const int N = 2e5 + 5;

vector<vector<int>> adj[N];
vector<bool> visited(N, false);
priority_queue<int, vector<int>, greater<int>> pq[N];

void dfs(int node) {
    visited[node] = true;
    for(auto &child : adj[node]) {
        if(!visited[child]) {
            dfs(child);
            while(!pq[child].empty()) {
                pq[node].push(pq[child].top());
                pq[child].pop();
            }
        }
    }
    pq[node].push(node);
}

int getKthLargest(int node, int k) {
    if(k > pq[node].size()) return -1;
    while(--k) pq[node].pop();
    return pq[node].top();
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    
    for(int i = 1; i <= n; i++) adj[i].push_back(i);

    while(q--) {
        int t, u, v;
        cin >> t >> u >> v;
        if(t == 1) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        } else {
            if(!visited[u]) dfs(u);
            cout << getKthLargest(v, k) << '\n';
        }
    }

    return 0;
}
