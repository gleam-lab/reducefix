
```cpp
#include <bits/stdc++.h>

using namespace std;

#define ll long long

int const N = 2e5 + 5;

vector<int> g[N];
vector<int> all_vertices;
vector<ll> subtree_sizes;

void dfs(int n, int p) {
    all_vertices.push_back(n);
    subtree_sizes.push_back(1);
    for (int next : g[n]) {
        if (next == p) continue;
        dfs(next, n);
        subtree_sizes.back() += subtree_sizes[next];
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    for (int i = 2; i <= n; ++i) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    dfs(1, -1);

    sort(all_vertices.begin(), all_vertices.end(), [&](int u, int v) {
        return subtree_sizes[u] > subtree_sizes[v];
    });

    for (int i = 1; i <= q; ++i) {
        int v, k;
        cin >> v >> k;
        if (k > subtree_sizes[v]) {
            cout << -1 << '\n';
        } else {
            cout << all_vertices[k - 1] << '\n';
        }
    }

    return 0;
}
```
