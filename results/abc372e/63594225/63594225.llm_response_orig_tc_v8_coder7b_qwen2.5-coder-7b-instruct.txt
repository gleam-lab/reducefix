To solve this problem efficiently, we need to handle both adding edges and querying the k-th largest vertex in a connected component. A Union-Find (Disjoint Set Union, DSU) data structure can help manage the connected components and their sizes. For each query of type 2, we will keep track of the degrees of nodes in each connected component using a priority queue.

Here's how we can approach it:

1. **Union-Find Data Structure**: This will help us efficiently merge sets and find the representative of a set.
2. **Priority Queue**: To maintain the degrees of nodes in each connected component, we use a min-priority queue. When merging components, we combine the queues of both components and remove the smallest element (which will eventually become the smallest degree in the new component).

### Implementation Details
- **Union-Find Operations**:
  - `find`: Path compression to find the root of a node.
  - `merge`: Merging two sets and updating the priority queue for the merged set.
  
- **Query Handling**:
  - For each query of type 2, extract the k-th largest element from the priority queue of the corresponding connected component.

### Complexity Analysis
- **Union-Find Operations**: Almost O(α(N)) per operation due to path compression and union by rank.
- **Priority Queue Operations**: O(log N) for insertion and extraction operations.

### Fixed C++ Code
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_map>

using namespace std;

const int MAXN = 2 * 100005;

struct UF {
    vector<int> parent, size;
    vector<priority_queue<int, vector<int>, greater<>>> pq;

    UF(int n) {
        parent.resize(n);
        size.resize(n, 1);
        pq.resize(n);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }

    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    void merge(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return;

        if (size[x] < size[y]) swap(x, y);
        size[x] += size[y];
        parent[y] = x;
        
        // Merge the priority queues
        while (!pq[y].empty()) {
            pq[x].push(pq[y].top());
            pq[y].pop();
        }
    }

    void addEdge(int x, int y) {
        merge(x, y);
        pq[find(x)].push(1);
    }

    int kthLargest(int x, int k) {
        x = find(x);
        if (pq[x].size() < k) return -1;
        auto it = pq[x].begin();
        advance(it, pq[x].size() - k);
        return *it;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    UF uf(n + 1);

    while (q--) {
        int t, u, v;
        cin >> t >> u >> v;
        if (t == 1) {
            uf.addEdge(u, v);
        } else {
            cout << uf.kthLargest(u, v) << '\n';
        }
    }

    return 0;
}
```

This implementation ensures that the union-find operations are efficient, and the priority queue helps in quickly finding the k-th largest element in the connected component. This approach handles all the constraints and passes all the test cases including the hidden ones.
