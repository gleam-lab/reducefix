#include <iostream>
#include <vector>
#include <set>

using namespace std;

const int MAX_N = 2 * 100005;
vector<int> parent(MAX_N);
vector<set<int>> adjList(MAX_N);

int findParent(int node) {
    if (parent[node] != node) {
        parent[node] = findParent(parent[node]);
    }
    return parent[node];
}

void unionSets(int node1, int node2) {
    int p1 = findParent(node1);
    int p2 = findParent(node2);
    if (p1 != p2) {
        if (adjList[p1].size() < adjList[p2].size()) {
            swap(p1, p2);
        }
        for (int neighbor : adjList[p2]) {
            adjList[p1].insert(neighbor);
        }
        parent[p2] = p1;
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        parent[i] = i;
        adjList[i].insert(i);
    }
    while (q--) {
        int type, u, v;
        cin >> type >> u >> v;
        if (type == 1) {
            unionSets(u, v);
        } else {
            int pu = findParent(u);
            if (adjList[pu].size() < v) {
                cout << -1 << "\n";
            } else {
                auto it = adjList[pu].rbegin();
                advance(it, v - 1);
                cout << *it << "\n";
            }
        }
    }
    return 0;
}
