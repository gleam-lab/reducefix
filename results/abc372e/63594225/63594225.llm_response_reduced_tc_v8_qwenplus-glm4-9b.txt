
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 200005;

vector<int> adj[N];
vector<int> subtree_sizes;
vector<vector<int>> nodes_with_subtree_sizes;

int find_subtree_size(int node, int parent) {
    subtree_sizes[node] = 1;
    for (int neighbor : adj[node]) {
        if (neighbor == parent) continue;
        subtree_sizes[node] += find_subtree_size(neighbor, node);
    }
    nodes_with_subtree_sizes[subtree_sizes[node]].push_back(node);
    return subtree_sizes[node];
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 2; i <= n; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    find_subtree_size(1, -1);

    for (int size = 10; size >= 1; --size) {
        if (!nodes_with_subtree_sizes[size].empty()) {
            cout << nodes_with_subtree_sizes[size][0] << '\n';
            break;
        }
    }

    return 0;
}
```
