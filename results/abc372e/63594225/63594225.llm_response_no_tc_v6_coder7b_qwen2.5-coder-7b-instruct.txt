```cpp
#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector <vector <int>> v;

vector <int> p;

int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    v[a].insert(v[a].end(), v[b].begin(), v[b].end());
    sort(v[a].begin(), v[a].end(), greater<int>());
    while(SZ(v[a]) > 10) {
        v[a].pop_back();
    }
    p[b] = a;
    return;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    p.resize(n+1);
    v.resize(n+1);
    for(int i = 1; i <= n; i++) {
        p[i] = i;
        v[i].push_back(i);
    }
    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if(t == 1) {
            uni(a, b);
        }
        else {
            a = par(a);
            cout << (SZ(v[a]) < b ? -1 : v[a][b-1]) << '\n';
        }
    }

    return 0;
}
```

This solution maintains a list of vertices connected to each vertex using union-find data structure. When adding an edge, it merges the lists of both vertices and sorts them in descending order to maintain the k-th largest element at the correct position. This approach ensures that the k-th largest element can be accessed efficiently in O(1) time after sorting. The constraints are handled appropriately, ensuring that the operations remain efficient even for large inputs.
