After analyzing the code and the test case, I notice that the issue seems to be related to the handling of the k-th largest element in the connected components. The original code maintains sorted lists of up to 100 elements, which might not be sufficient for some test cases. Here's the fixed version that properly maintains the sorted lists and handles the queries correctly:

```cpp
#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector<set<int>> v;
vector<int> p;

int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    
    if(v[a].size() < v[b].size()) swap(a, b);
    
    for(int x : v[b]) {
        v[a].insert(x);
    }
    v[b].clear();
    p[b] = a;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    p.resize(n+1);
    v.resize(n+1);
    for(int i = 1; i <= n; i++) {
        p[i] = i;
        v[i].insert(i);
    }
    
    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if(t == 1) {
            uni(a, b);
        }
        else {
            a = par(a);
            if(v[a].size() < b) {
                cout << -1 << '\n';
            }
            else {
                auto it = v[a].end();
                for(int i = 0; i < b; i++) it--;
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
```

Key improvements made:
1. Changed from vectors to sets to automatically maintain sorted order of elements
2. Simplified the union operation by using sets and merging smaller sets into larger ones (using union-by-size heuristic)
3. For query type 2, directly access the k-th largest element from the set by using reverse iteration
4. Removed the arbitrary limit of 100 elements, now storing all elements in each component's set
5. The complexity is now O(Q α(N) log N) where α is the inverse Ackermann function, which should be efficient enough for the constraints
