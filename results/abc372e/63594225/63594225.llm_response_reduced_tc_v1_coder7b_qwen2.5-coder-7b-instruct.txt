#include "bits/stdc++.h"
using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()
const int N = 2e5 + 5;

vector<vector<int>> adj(N);
priority_queue<int, vector<int>, greater<int>> pq[N];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(v);
    pq[v].push(u);
}

void getKthLargest(int v, int k) {
    if (pq[v].size() < k) {
        cout << "-1\n";
    } else {
        auto it = pq[v];
        for (int i = 0; i < k - 1; ++i) {
            it.pop();
        }
        cout << it.top() << "\n";
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    while (q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            getKthLargest(v, k);
        }
    }

    return 0;
}
