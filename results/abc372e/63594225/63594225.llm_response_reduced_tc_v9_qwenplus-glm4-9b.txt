#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 2e5 + 5;

vector<int> graph[N];
vector<int> rank(N);

int find_root(int a) {
    if (rank[a] == rank[a]) return a;
    rank[a] = find_root(rank[a]);
    return rank[a];
}

void merge(int a, int b) {
    int aRoot = find_root(a), bRoot = find_root(b);
    if (aRoot != bRoot) {
        if (rank[aRoot] < rank[bRoot]) {
            swap(aRoot, bRoot);
        }
        rank[aRoot] += rank[bRoot];
        graph[aRoot].insert_end(bRoot);
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        rank[i] = 1;
    }

    for (int i = 0; i < q; ++i) {
        int t, a, b;
        cin >> t >> a >> b;
        if (t == 1) {
            merge(a, b);
        } else {
            int root = find_root(b);
            vector<int> vec = graph[root];

    }

}
