int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    vector <int> v1;
    int ai = 0, bi = 0;
    while(ai < SZ(v[a]) or bi < SZ(v[b])) {
        if(bi == SZ(v[b]) or v[a][ai] < v[b][bi]) { // Changed comparison operator
            v1.push_back(v[a][ai]);
            ai++;
        } else {
            v1.push_back(v[b][bi]);
            bi++;
        }
    }
    v[a] = v1;

        v[a].pop_back();
    }
    p[b] = a;
    return;
}
