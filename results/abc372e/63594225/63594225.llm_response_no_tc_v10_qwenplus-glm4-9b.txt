```cpp
#include <bits/stdc++.h>

using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
vector<int> ans;
priority_queue<int> pq;
vector<int> size(N, 1);

void merge(int u, int v) {
    if (size[u] < size[v]) swap(u, v);
    size[u] += size[v];
    g[u].insert(g[u].end(), g[v].begin(), g[v].end());
}

int findKthLargest(int v, int k) {
    auto it = upper_bound(g[v].rbegin(), g[v].rend(), -k);
    return g[v].rend() - it;
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 0; i < q; ++i) {
        int t, u, v;
        cin >> t >> u >> v;
        if (t == 1) {
            merge(u, v);
        } else if (t == 2) {
            ans.push_back(findKthLargest(v, k));
        }
    }
    for (int &a : ans) {
        cout << a << '\n';
    }
    return 0;
}
```

This fixed code uses a Union-Find structure with path compression (not shown here for simplicity) and a priority queue to efficiently find the k-th largest vertex among the vertices connected to a given vertex. The `merge` function is called to combine the sizes of two vertices' neighbor lists when an edge is added. The `findKthLargest` function uses the end iterator of the list to find the (k-1)-th largest element, which is the k-th largest due to the descending order of the list. The result for each Type 2 query is stored in the `ans` vector and printed at the end.
