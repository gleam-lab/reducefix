```cpp
#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector<vector<int>> v(N);

int par[N];

int find(int x) {
    if(par[x] == x) return x;
    return par[x] = find(par[x]);
}

void unite(int a, int b) {
    a = find(a), b = find(b);
    if(a == b) return;
    v[a].insert(end(v[a]), begin(v[b]), end(v[b]));
    sort(begin(v[a]), end(v[a]));
    v[a].resize(min((int)v[a].size(), 10));
    par[b] = a;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    for(int i = 1; i <= n; i++) {
        par[i] = i;
        v[i].push_back(i);
    }
    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if(t == 1) {
            unite(a, b);
        } else {
            a = find(a);
            if(SZ(v[a]) < b) {
                cout << -1 << '\n';
            } else {
                cout << v[a][b-1] << '\n';
            }
        }
    }

    return 0;
}
```
