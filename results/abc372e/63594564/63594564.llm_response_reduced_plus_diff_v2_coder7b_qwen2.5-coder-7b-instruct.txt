int find(int u) {

bool unite(int u, int v) {
    int pu = find(u), pv = find(v);
    if (pu == pv) return false;
    if (rank[pu] > rank[pv])
        swap(pu, pv);
    parent[pu] = pv;
    size[pv] += size[pu];
    if (rank[pu] == rank[pv])
        rank[pv]++;
    return true;
}

int kthLargest(int u, int k) {
    int pu = find(u);
    if (size[pu] < k) return -1;
    priority_queue<int> pq;
    for (int i = 1; i <= size[pu]; ++i) {
        int v = find(i);

            pq.push(-i);
        }
    }
    while (--k && !pq.empty()) pq.pop();
    return -pq.top();
