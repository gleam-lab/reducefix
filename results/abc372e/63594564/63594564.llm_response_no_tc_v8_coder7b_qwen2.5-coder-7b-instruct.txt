```cpp
#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector <vector <int>> v;

vector <int> p;

int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    v[a].insert(end(v[a]), begin(v[b]), end(v[b]));
    sort(begin(v[a]), end(v[a]));
    v[a].resize(min(10, (int)v[a].size()));
    while(SZ(v[a]) > 10) {
        v[a].pop_back();
    }
    p[b] = a;
    return;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    p.resize(n+1);
    v.resize(n+1);
    for(int i = 1; i <= n; i++) {
        p[i] = i;
        v[i].push_back(i);
    }
    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if(t == 1) {
            uni(a, b);
        }
        else {
            a = par(a);
            cout << (SZ(v[a]) < b ? -1 : v[a][b-1]) << '\n';
        }
    }

    return 0;
}
```

This solution uses a union-find data structure with path compression and union by rank to efficiently manage the connectivity of vertices. It maintains a sorted list of connected vertices for each component and ensures that only the top 10 largest vertices are kept. This approach avoids redundant elements and ensures efficient updates and queries.
