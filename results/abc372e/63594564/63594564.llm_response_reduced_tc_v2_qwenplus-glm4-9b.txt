#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 2e5 + 5;

vector<int> graph[N];
vector<int> vertexSizes;
int parent[N];

int findRoot(int v) {
    if (parent[v] == v) return v;
    parent[v] = findRoot(parent[v]);
    return parent[v];
}

void unionVertices(int u, int v) {
    int rootU = findRoot(u);
    int rootV = findRoot(v);
    if (rootU != rootV) {
        if (vertexSizes[rootU] < vertexSizes[rootV]) {
            swap(rootU, rootV);
        }
        parent[rootV] = rootU;
        graph[rootU].insert(graph[rootV].end(), graph[rootV].begin(), graph[rootV].end());
        vertexSizes[rootU] += vertexSizes[rootV];
        graph[rootV].clear();
        vertexSizes[rootV] = 0;
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        parent[i] = i;
        vertexSizes.push_back(1);
    }
    while (q--) {
        int t, u, v;
        cin >> t >> u >> v;
        if (t == 1) {
            unionVertices(u, v);
        } else {
            int rootV = findRoot(v);
            if (vertexSizes[rootV] < v) {
                cout << -1 << '\n';
            } else {
                vector<int> connectedVertices = graph[rootV];
                sort(connectedVertices.rbegin(), connectedVertices.rend());
                cout << connectedVertices[v - 1] << '\n';
            }
        }
    }
    return 0;
}
