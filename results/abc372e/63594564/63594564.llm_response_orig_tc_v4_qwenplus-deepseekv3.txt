#include <bits/stdc++.h>

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector<vector<int>> v;
vector<int> p;

int par(int x) {
    if(p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if(a == b) return;
    
    if (v[a].size() < v[b].size()) {
        swap(a, b);
    }
    
    vector<int> merged;
    merged.reserve(v[a].size() + v[b].size());
    merge(v[a].begin(), v[a].end(), 
          v[b].begin(), v[b].end(), 
          back_inserter(merged), greater<int>());
    
    if (merged.size() > 10) {
        merged.resize(10);
    }
    v[a] = merged;
    p[b] = a;
}
