#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

// Stores the representative set for each component, but only up to 10 largest nodes
vector<set<int>> components;
vector<int> parent;

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int a, int b) {
    int pa = find(a), pb = find(b);
    if (pa == pb) return;

    // Always merge smaller into larger for efficiency
    if (components[pa].size() < components[pb].size())
        swap(pa, pb);

    // Merge pb's elements into pa's component
    for (int val : components[pb]) {
        components[pa].insert(val);
        if (components[pa].size() > 10) {
            // Remove the smallest (since we want top 10 largest)
            components[pa].erase(*components[pa].begin());
        }
    }

    // Update parent pointers
    parent[pb] = pa;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    parent.resize(n+1);
    components.resize(n+1);

    // Initialize DSU
    for (int i = 1; i <= n; ++i) {
        parent[i] = i;
        components[i].insert(i);
    }

    while(q--) {
        int t, u, v;
        cin >> t >> u >> v;
        if(t == 1) {
            unite(u, v);
        } else {
            int p = find(u);
            if ((int)components[p].size() < v) {
                cout << -1 << '\n';
            } else {
                // Get the k-th largest: advance from begin by (size - k)
                auto it = components[p].end();
                advance(it, -v);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
