#include "bits/stdc++.h"

using namespace std;

#define ll long long
#define N 200010

vector<int> adj[N];
vector<int> values[N];
vector<int> rank(N, 0);

void unite(int a, int b) {
    if (rank[a] < rank[b]) swap(a, b);
    if (rank[a] == rank[b]) rank[a]++;
    if (rank[a] == 0) rank[a] = rank[b] + 1;
    for (int num : values[b]) {
        adj[a].push_back(num);
        values[a].push_back(num);
        sort(values[a].begin(), values[a].end(), greater<int>());
        if (values[a].size() > 10) values[a].pop_back();
    }
    for (int num : adj[b]) {
        adj[a].push_back(num);
        values[a].push_back(num);
        sort(values[a].begin(), values[a].end(), greater<int>());
        if (values[a].size() > 10) values[a].pop_back();
    }
    adj[b].clear();
    values[b].clear();
}

int find_kth_largest(int vertex, int k) {
    if (k > values[vertex].size()) return -1;
    return values[vertex][k - 1];
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; i++) values[i].push_back(i);
    for (int i = 1; i <= n; i++) sort(values[i].begin(), values[i].end(), greater<int>());
    
    while (q--) {
        int t, u, v, k;
        cin >> t >> u >> v;
        if (t == 1) {
            unite(u, v);
        } else {
            cout << find_kth_largest(u, k) << '\n';
        }
    }

    return 0;
}
