After analyzing the problem and the incorrect code, I identified the main issue is with the union operation's efficiency and correctness. The current approach merges all elements into a single vector and keeps only the top 10, which is too slow for large inputs (N=2e5) and can lose important information.

Here's the fixed code that uses a more efficient approach with Union-Find and maintains separate sorted lists for each component, merging only when necessary and keeping track of the largest elements:

```cpp
#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector<multiset<int>> v;
vector<int> p;
vector<int> sz;

int par(int x) {
    if (p[x] == x) return x;
    return p[x] = par(p[x]);
}

void uni(int a, int b) {
    a = par(a), b = par(b);
    if (a == b) return;
    if (sz[a] < sz[b]) swap(a, b);
    
    // Merge smaller into larger
    v[a].insert(v[b].begin(), v[b].end());
    while (v[a].size() > 10) {
        v[a].erase(v[a].begin());
    }
    
    p[b] = a;
    sz[a] += sz[b];
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    p.resize(n + 1);
    sz.resize(n + 1, 1);
    v.resize(n + 1);
    
    for (int i = 1; i <= n; i++) {
        p[i] = i;
        v[i].insert(i);
    }
    
    while (q--) {
        int t, a, b;
        cin >> t >> a >> b;
        if (t == 1) {
            uni(a, b);
        } else {
            a = par(a);
            if (v[a].size() < b) {
                cout << -1 << '\n';
            } else {
                auto it = v[a].end();
                advance(it, -b);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
```

Key improvements:
1. Changed from vector<vector<int>> to vector<multiset<int>> for efficient merging and maintaining sorted order
2. Added size tracking (sz) for union by size optimization
3. Modified the union operation to merge smaller sets into larger ones (Union-Find with path compression and union by size)
4. Maintain only the top 10 elements in each set for efficient k-th largest queries
5. Used multiset's reverse iterator to efficiently find k-th largest element

This approach ensures:
- O(α(N)) amortized time per union operation (α is inverse Ackermann)
- O(log N) time for k-th largest query (as we maintain sorted elements)
- Efficient memory usage by keeping at most 10 elements per component
