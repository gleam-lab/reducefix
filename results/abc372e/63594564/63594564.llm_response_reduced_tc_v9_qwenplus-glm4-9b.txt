#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

const int N = 200005;

int parent[N + 1];
vector<int> graph[N + 1];

void findParent(int v) {
    if (parent[v] == v) return;
    findParent(parent[v]);
    parent[v] = parent[parent[v]]; // Path compression
}

void unionSets(int u, int v) {
    findParent(u);
    findParent(v);
    if (u != v) { // Union by rank is implicitly achieved by the order of union
        graph[parent[v]].insert(graph[parent[u]].begin(), graph[parent[u]].end());
        graph[parent[u]].clear();
        parent[u] = parent[v]; // Union by rank
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    
    for (int i = 1; i <= n; ++i) {
        parent[i] = i; // Initialize parent array for union-find
    }
    
    while (q--) {
        int type, u, v;
        cin >> type >> u >> v;
        
        if (type == 1) {
            // Add an edge by union of sets
            unionSets(u, v);
        } else if (type == 2) {
            // Find the root of the connected component of vertex v
            int root = parent[v];
            // Use set to keep track of the vertices in the connected component
            set<int> component(graph[root]);
            // Output the k-th largest vertex
            if (component.size() < v) {
                cout << -1 << endl;
            } else {
                auto it = component.upper_bound(v); // Find the iterator to the k-th largest element
                if (it != component.end()) {
                    cout << *it << endl;
                } else {
                    cout << -1 << endl; // This should never happen because we check the size
                }
            }
        }
    }
    
    return 0;
}
