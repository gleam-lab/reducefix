#include "bits/stdc++.h"

using namespace std;

#define ff first
#define ss second
#define ll long long
#define SZ(s) (int)s.size()

int const N = 2e5 + 5;

vector <set<int>> component;
vector <int> parent;

int find(int x) {
    if(parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

void unite(int a, int b) {
    int pa = find(a), pb = find(b);
    if(pa == pb) return;

    // Merge smaller set into larger one to keep tree shallow
    if(component[pa].size() < component[pb].size()) swap(pa, pb);

    for(int node : component[pb]) {
        component[pa].insert(node);
        parent[node] = pa;
    }

    component[pb].clear();
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    component.resize(n+1);
    parent.resize(n+1);

    for(int i = 1; i <= n; ++i) {
        component[i].insert(i);
        parent[i] = i;
    }

    while(q--) {
        int t, a, b;
        cin >> t >> a >> b;

        if(t == 1) {
            unite(a, b);
        } else {
            int root = find(a);
            if((int)component[root].size() < b) {
                cout << -1 << '\n';
            } else {
                auto it = component[root].rbegin(); // reverse iterator for descending order
                for(int i = 1; i < b && it != component[root].rend(); ++i) ++it;
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
