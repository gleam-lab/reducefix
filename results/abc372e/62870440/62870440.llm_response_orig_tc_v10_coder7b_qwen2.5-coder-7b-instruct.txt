```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
priority_queue<int, vector<int>, greater<int>> pq[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void updatePriorityQueue(int v, int newWeight) {
    pq[v].pop();
    pq[v].push(newWeight);
    while (!pq[v].empty() && pq[v].top() != INT_MAX) break;
}

void solveQueries(int n, int q) {
    for (int i = 1; i <= n; ++i) {
        pq[i].push(INT_MAX); // Initialize with infinity
    }

    for (int i = 0; i < q; ++i) {
        int type, u, v, k;
        cin >> type;
        if (type == 1) {
            cin >> u >> v;
            addEdge(u, v);
        } else {
            cin >> v >> k;
            if (adj[v].size() < k) {
                cout << "-1" << endl;
            } else {
                int res = -1;
                for (int neighbor : adj[v]) {
                    if (!pq[neighbor].empty()) {
                        res = pq[neighbor].top();
                        if (res >= k) {
                            cout << res << endl;
                            break;
                        }
                    }
                }
                if (res == -1) {
                    cout << "-1" << endl;
                }
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    solveQueries(n, q);

    return 0;
}
```

This solution uses adjacency lists to represent the graph and priority queues to efficiently manage the k-th largest elements. When adding an edge, it updates the priority queue for both vertices involved. For each query, it checks the size of the adjacency list and either prints -1 or finds the k-th largest element using the priority queues. This approach ensures that the solution handles all types of queries within the given constraints.
