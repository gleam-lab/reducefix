To solve this problem efficiently, we need to:

1. **Maintain connected components** of an undirected graph dynamically as edges are added.
2. For each **Type 2 query**, return the k-th largest vertex number in the component that includes a given vertex `v`. If there are fewer than `k` vertices in that component, return `-1`.

---

### Strategy

We use **Disjoint Set Union (DSU)** with path compression and union by rank for efficient component management.

To support the Type 2 queries (finding the k-th largest node), we maintain a sorted list (or vector) of nodes in descending order for each component root.

- When merging two components, we merge their lists in O(M + N), where M and N are the sizes of the lists (but limited to top 20 or so).
- We keep only the **top 20 largest values per set**, since `k ≤ 10`, and we don't need more.

---

### Time Complexity

- DSU operations: O(α(N)) per operation
- Maintaining top elements: O(20) per union
- Querying k-th largest: O(1)

This is efficient enough for constraints up to 2 × 10⁵.

---

### Final Working Code

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
int parent[MAXN];
vector<int> top_vals[MAXN];

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int x, int y) {
    int fx = find(x);
    int fy = find(y);
    if (fx == fy) return;

    // Merge smaller set into larger one
    if (top_vals[fx].size() < top_vals[fy].size())
        swap(fx, fy);

    // Merge the top values
    for (int val : top_vals[fy]) {
        top_vals[fx].push_back(val);
    }

    // Keep only top 20 values
    sort(top_vals[fx].begin(), top_vals[fx].end(), greater<int>());
    if (top_vals[fx].size() > 20)
        top_vals[fx].resize(20);

    parent[fy] = fx;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        top_vals[i].push_back(i);
    }

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if (k > (int)top_vals[root].size()) {
                cout << -1 << '\n';
            } else {
                cout << top_vals[root][k - 1] << '\n';
            }
        }
    }

    return 0;
}
```
