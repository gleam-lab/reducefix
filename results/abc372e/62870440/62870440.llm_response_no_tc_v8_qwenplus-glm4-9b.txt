#include<bits/stdc++.h>
using namespace std;
int parent[200005];
vector<int> adj[200005];
int find(int x) {
    if (parent[x] == x) return x;
    parent[x] = find(parent[x]);
    return parent[x];
}
void union_set(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);
    if (rootX != rootY) {
        parent[rootY] = rootX;
    }
}
int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        parent[i] = i;
    }
    for (int i = 0; i < q; i++) {
        int type, u, v, k;
        cin >> type;
        if (type == 1) {
            cin >> u >> v;
            union_set(u, v);
        } else if (type == 2) {
            cin >> v >> k;
            int rootV = find(v);
            vector<int> nodes = adj[rootV];
            sort(nodes.rbegin(), nodes.rend()); // Sort descending
            if (k > nodes.size()) {
                cout << -1 << endl;
            } else {
                cout << nodes[k - 1] << endl;
            }
        }
    }
    return 0;
}
