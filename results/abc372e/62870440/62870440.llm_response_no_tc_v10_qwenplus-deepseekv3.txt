const int MAX_N = 200005;
vector<int> parent(MAX_N);
vector<vector<int>> top_elements(MAX_N);

int find_root(int x) {
    if (parent[x] != x) {
        parent[x] = find_root(parent[x]);
    }
    return parent[x];

void merge(int u, int v) {

    int root_v = find_root(v);
    if (root_u == root_v) return;
