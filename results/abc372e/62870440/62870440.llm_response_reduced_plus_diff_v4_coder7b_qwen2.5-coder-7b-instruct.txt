using namespace std;

const int MAXN = 2 * 1e5 + 5;
vector<int> adj[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void dfs(int node, int parent, vector<int>& dist) {
    for(auto child : adj[node]) {
        if(child != parent) {
            dist[child] = dist[node] + 1;
            dfs(child, node, dist);
        }
    }
}
