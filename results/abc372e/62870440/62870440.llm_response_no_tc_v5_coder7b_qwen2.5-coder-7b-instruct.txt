#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void updatePQ(int v) {
    pq[v].push(v);
    for (int neighbor : adj[v]) {
        if (!pq[neighbor].empty()) {
            pq[v].push(pq[neighbor].top());
            pq[neighbor].pop();
        }
    }
    while (pq[v].size() > 10) pq[v].pop();
}

int getKthLargest(int v, int k) {
    if (pq[v].size() < k) return -1;
    return pq[v].nth_element(k-1);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    for (int i = 1; i <= n; ++i) {
        updatePQ(i);
    }

    for (int i = 0; i < q; ++i) {
        int type, u, v;
        cin >> type >> u >> v;
        if (type == 1) {
            addEdge(u, v);
            updatePQ(u);
            updatePQ(v);
        } else {
            cout << getKthLargest(v, k) << '\n';
        }
    }

    return 0;
}
