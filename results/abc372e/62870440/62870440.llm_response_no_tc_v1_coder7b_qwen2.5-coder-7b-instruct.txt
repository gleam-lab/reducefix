```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
set<int> st[200005];

void dfs(int node, int parent) {
    for (auto child : adj[node]) {
        if (child == parent) continue;
        dfs(child, node);
        st[node].insert(st[child].begin(), st[child].end());
        while (st[node].size() > 10) st[node].erase(st[node].begin());
    }
    st[node].insert(node);
    while (st[node].size() > 10) st[node].erase(st[node].begin());
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int n, q; cin >> n >> q;
    vector<pair<int, pair<int, int>>> queries(q);
    for (int i = 0; i < q; ++i) {
        int type; cin >> type;
        queries[i] = {type, {0, 0}};
        if (type == 1) {
            int u, v; cin >> u >> v;
            queries[i].second.first = min(u, v);
            queries[i].second.second = max(u, v);
        } else {
            int v, k; cin >> v >> k;
            queries[i].second.first = v;
            queries[i].second.second = k;
        }
    }

    for (int i = 1; i <= n; ++i) adj[i].clear();
    for (int i = 0; i < q; ++i) {
        if (queries[i].first == 1) {
            int u = queries[i].second.first, v = queries[i].second.second;
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    dfs(1, -1);

    for (int i = 0; i < q; ++i) {
        if (queries[i].first == 2) {
            int v = queries[i].second.first, k = queries[i].second.second;
            if (k > st[v].size()) cout << "-1" << endl;
            else cout << *next(st[v].rbegin(), k - 1) << endl;
        }
    }

    return 0;
}
```

This solution uses Depth First Search (DFS) to build a tree from the graph and then maintains a set of the 10 largest elements for each node. When processing a Type 2 query, it checks if the required element exists in the set and prints it or -1 otherwise. This approach ensures that the solution is efficient and handles all constraints and edge cases appropriately.
