```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2 * 100005;
const int MAXK = 10;

vector<int> adj[MAXN];
priority_queue<int, vector<int>, greater<int>> pq[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void processQueries(int n, int q) {
    for (int i = 0; i < q; ++i) {
        int type, v, k;
        cin >> type >> v;
        if (type == 1) {
            int u;
            cin >> u;
            addEdge(v, u);
            for (int w : adj[u]) {
                pq[w].push(v);
                if (pq[w].size() > MAXK) pq[w].pop();
            }
        } else {
            cin >> k;
            if (pq[v].size() >= k) cout << pq[v].top() << "\n";
            else cout << "-1\n";
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    processQueries(n, q);

    return 0;
}
```

### Explanation
1. **Graph Representation**: We use an adjacency list to represent the graph and priority queues to keep track of the k-th largest elements efficiently.
2. **Adding Edges**: When adding an edge between two vertices, we update the priority queue for both vertices to include the new connection.
3. **Processing Queries**:
   - For Type 1 queries, we add the edge and update the priority queues accordingly.
   - For Type 2 queries, we check if there are at least k elements in the priority queue and output the k-th largest element. If not, we output `-1`.
4. **Efficiency**: Using priority queues ensures that each insertion and extraction operation is efficient, maintaining a logarithmic time complexity for these operations.
