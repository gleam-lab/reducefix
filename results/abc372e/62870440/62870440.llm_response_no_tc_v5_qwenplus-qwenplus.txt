#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int K_MAX = 10;

int fa[MAXN];
set<int, greater<int>> group[MAXN];  // Store up to k largest nodes in descending order

// Find with path compression
int find(int x) {
    if (fa[x] != x)
        fa[x] = find(fa[x]);
    return fa[x];
}

// Merge smaller set into larger one for efficiency
void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    
    if (root_u == root_v)
        return;
        
    // Ensure we merge smaller into larger
    if (group[root_u].size() < group[root_v].size()) {
        swap(root_u, root_v);
    }
    
    // Merge elements from root_v into root_u
    for (auto it : group[root_v]) {
        if (group[root_u].size() >= K_MAX) {
            // If already have K_MAX elements, only insert if larger than smallest in set
            int smallest = *group[root_u].rbegin();
            if (it > smallest) {
                group[root_u].erase(smallest);
                group[root_u].insert(it);
            }
        } else {
            group[root_u].insert(it);
        }
    }
    
    // Update parent pointer
    fa[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU
    for (int i = 1; i <= N; ++i) {
        fa[i] = i;
        group[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } 
        else if (type == 2) {
            int v, k;
            cin >> v >> k;
            
            int root = find(v);
            
            if (k <= group[root].size()) {
                auto it = group[root].begin();
                advance(it, k - 1);
                cout << *it << endl;
            } else {
                cout << -1 << endl;
            }
        }
    }

    return 0;
}
