#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i=0;i<(n);i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()

using ll = long long;
using vi = vector<int>;
using vvi = vector<vi>;

class UnionFind {
public:
    vi parent, rank, size;
    
    UnionFind(int n) : parent(n), rank(n, 0), size(n, 1) {
        iota(all(parent), 0);
    }

    int findRoot(int x) {
        if (parent[x] != x) parent[x] = findRoot(parent[x]);
        return parent[x];
    }

    void unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return;
        
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            size[ry] += size[rx];
        } else {
            parent[ry] = rx;
            size[rx] += size[ry];
            if (rank[rx] == rank[ry]) rank[rx]++;
        }
    }
};

class Node {
public:
    int val;
    set<int> neighbors;
    
    Node(int _val) : val(_val) {}
    
    bool addNeighbor(int neighbor) {
        if (!neighbors.count(neighbor)) {
            neighbors.insert(neighbor);
            return true;
        }
        return false;
    }
    
    void removeNeighbor(int neighbor) {
        neighbors.erase(neighbor);
    }
    
    int kthLargest(int k) {
        if (k > neighbors.size()) return -1;
        auto it = neighbors.rbegin();
        advance(it, k - 1);
        return *it + 1;
    }
};

int main() {

    vector<Node> nodes(n);

                uf.unite(ru, rv);
                
                for (auto nei : nodes[v].neighbors) {
                    nodes[u].addNeighbor(nei);
                    nodes[nei].removeNeighbor(v);

            }
