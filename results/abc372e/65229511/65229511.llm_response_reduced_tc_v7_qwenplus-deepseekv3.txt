To fix this code, we need to address the issue in handling the connected components and correctly maintaining the set of vertices in each component. The original code attempts to merge sets when components are united, but it incorrectly clears the old set and only merges one side, leading to incomplete information about the connected components. 

### Approaches
1. **Union-Find with Path Compression and Union by Rank**: This helps efficiently manage the connected components and ensures that the operations are nearly constant time.
2. **Maintaining Sorted Lists**: For each component, maintain a sorted list (or a set) of vertices. When two components are merged, merge their corresponding sorted lists. To optimize, always merge the smaller list into the larger one to keep the operations efficient.
3. **Handling Queries**: For type 2 queries, access the k-th largest element in the sorted list of the connected component of the queried vertex. If the size of the list is smaller than k, return -1.

#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i=0;i<(n);i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()
using ll = long long;

struct UnionFind {
    vector<int> parent, rank;
    vector<set<int>> components;
    UnionFind(int n) : parent(n), rank(n, 0), components(n) {
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            components[i].insert(i);
        }
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    void unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            components[ry].insert(components[rx].begin(), components[rx].end());
            components[rx].clear();
        } else {
            parent[ry] = rx;
            components[rx].insert(components[ry].begin(), components[ry].end());
            components[ry].clear();
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
    }
};
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<int> ans;
    rep(qi, q) {
        int com;
        cin >> com;
        if(com==1) {
            int u, v;
            cin >> u >> v;
            u--; v--;
            uf.unite(u, v);
        }
        if(com==2) {
            int v, k;
            cin >> v >> k;
            v--;
            int root = uf.findRoot(v);
            if((int)uf.components[root].size() < k) {
                ans.push_back(-1);
            } else {
                auto it = uf.components[root].rbegin();
                advance(it, k-1);
                ans.push_back(*it + 1);
            }
        }
    }
    for(auto x:ans) cout << x << endl;
    return 0;
}

### Explanation
1. **Union-Find Data Structure**: The `UnionFind` class manages the connected components with path compression and union by rank to ensure efficient operations.
2. **Components Maintenance**: Each component's vertices are maintained in a `set` within the `UnionFind` structure. This allows for efficient merging of components and quick access to sorted elements.
3. **Query Handling**: For type 2 queries, the code checks the size of the component set for the queried vertex. If the size is sufficient, it retrieves the k-th largest element by accessing the set in reverse order.
4. **Efficiency**: Merging smaller sets into larger ones ensures that each element is moved at most O(log n) times, maintaining an overall efficient time complexity for the operations.

This approach ensures the correct handling of dynamic graph connectivity and efficient answering of queries regarding the k-th largest vertex in connected components.
