struct UnionFind {
    vector<int> parent, rank;
    vector<set<int>>* sets;
    UnionFind(int n, vector<set<int>>& s) : parent(n), rank(n, 0), sets(&s) {
        for (int i = 0; i < n; i++)

        if (parent[x] != x)

        return parent[x];

    void unite(int x, int y) {

        if (rx == ry) return;
        if (rank[rx] < rank[ry]) swap(rx, ry);
        parent[ry] = rx;

        (*sets)[rx].insert((*sets)[ry].begin(), (*sets)[ry].end());
        (*sets)[ry].clear();
    }
};
