#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void dfs(int node, int parent, vector<int>& degree, priority_queue<int, vector<int>, greater<int>>& pq) {
    pq.push(degree[node]);
    if(pq.size() > 10) pq.pop();
    for(auto neighbor : adj[node]) {
        if(neighbor == parent) continue;
        dfs(neighbor, node, degree, pq);
    }
}

    for(int i = 1; i <= n; ++i) {
        dfs(i, -1, degree, pq);
    }

        } else {

This solution uses a depth-first search (DFS) approach to maintain a priority queue (min-heap) that keeps track of the top 10 degrees of nodes connected to each node. This allows efficient retrieval of the k-th largest degree for any node in response to a query. The `addEdge` function can be used to handle edge additions, though it's not directly called in the provided solution due to the nature of the problem statement where edges are added implicitly through the initial degree input.
