#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void sortAdjList() {
    for(int i = 1; i <= n; i++) {
        sort(adj[i].begin(), adj[i].end());
    }
}

int findKthLargest(int v, int k) {
    if(k > adj[v].size()) return -1;
    else return adj[v][adj[v].size() - k];
}

int main() {
    cin >> n >> q;
    
    while(q--) {
        int type, u, v, k;
        cin >> type;
        
        if(type == 1) {
            cin >> u >> v;
            addEdge(u, v);
        } 
        else if(type == 2) {
            cin >> v >> k;
            cout << findKthLargest(v, k) << endl;
        }
    }
    
    return 0;
}
