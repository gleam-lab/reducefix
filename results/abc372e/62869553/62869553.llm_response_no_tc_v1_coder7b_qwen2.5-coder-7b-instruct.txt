```cpp
#include<bits/stdc++.h>
using namespace std;
vector<int> adj[200005];
int n,q;

void dfs(int node, int parent, vector<int>& temp) {
    temp.push_back(node);
    for(auto child : adj[node]) {
        if(child != parent) {
            dfs(child, node, temp);
        }
    }
}

int main() {
    cin >> q >> n;
    while(q--) {
        int op, ix, iy;
        cin >> op >> ix >> iy;
        if(op == 1) {
            adj[ix].push_back(iy);
            adj[iy].push_back(ix);
        } else {
            vector<int> temp;
            dfs(iy, -1, temp);
            sort(temp.begin(), temp.end());
            if(iy > temp.size()) cout << "-1\n";
            else cout << temp[temp.size()-iy] << "\n";
        }
    }
    return 0;
}
```
