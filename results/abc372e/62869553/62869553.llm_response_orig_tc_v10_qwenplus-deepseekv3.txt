int parent[MAXN];
vector<int> adj[MAXN];  // Stores top 10 largest elements for each set

int find_set(int v) {
    if (v == parent[v])
        return v;
    return parent[v] = find_set(parent[v]);
}

void union_sets(int a, int b) {
    a = find_set(a);
    b = find_set(b);
    if (a != b) {
        if (adj[a].size() < adj[b].size())

        if (adj[a].size() > MAXK-1)
            adj[a].resize(MAXK-1);
        
        parent[b] = a;
    }
}
