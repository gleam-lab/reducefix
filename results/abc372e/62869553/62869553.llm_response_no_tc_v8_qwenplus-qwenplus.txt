#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int K_MAX = 10;

int fa[MAXN];
set<int, greater<int>> groups[MAXN];  // Store top 10 largest nodes in each group

int find(int x) {
    if (fa[x] != x) fa[x] = find(fa[x]);
    return fa[x];
}

void unite(int u, int v) {
    int fu = find(u);
    int fv = find(v);
    if (fu == fv) return;

    // Always merge smaller group into larger one for efficiency
    if (groups[fu].size() < groups[fv].size()) swap(fu, fv);

    // Merge the two sets keeping only top 10 elements
    for (int node : groups[fv]) {
        groups[fu].insert(node);
        if (groups[fu].size() > K_MAX) {
            auto last = prev(groups[fu].end());
            groups[fu].erase(last);
        }
    }

    // Update DSU parent
    fa[fv] = fu;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and sets
    for (int i = 1; i <= N; ++i) {
        fa[i] = i;
        groups[i].insert(i);
    }

    while (Q--) {
        int op;
        cin >> op;

        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            
            if (k > (int)groups[root].size()) {
                cout << -1 << '\n';
            } else {
                auto it = groups[root].begin();
                advance(it, k-1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
