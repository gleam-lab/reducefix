#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MAX_N = 200005;
const int MAX_EDGE = 20; // Maximum number of edges a vertex can have

int parent[MAX_N];
vector<int> adjList[MAX_N];

void make_set(int n) {
    fill_n(parent, n + 1, -1);
}

int find_set(int x) {
    if (parent[x] < 0) return x;
    parent[x] = find_set(parent[x]);
    return parent[x];
}

bool unite(int x, int y) {
    x = find_set(x);
    y = find_set(y);
    if (x == y) return false;
    if (parent[x] > parent[y]) swap(x, y);
    parent[x] += parent[y];
    parent[y] = x;
    return true;
}

int main() {
    int n, q;
    cin >> n >> q;

    make_set(n);

    while (q--) {
        int op, u, v, k;
        cin >> op >> u >> v;
        
        if (op == 1) {
            // Union operation
            unite(u, v);
        } else if (op == 2) {
            // Find the representative of v
            int rep = find_set(v);
            // Extract all vertices in the same set as v sorted in descending order
            vector<int> sortedVertices;
            for (int neighbor : adjList[rep]) {
                if (find_set(neighbor) == rep) {
                    sortedVertices.push_back(neighbor);
                }
            }
            sort(sortedVertices.begin(), sortedVertices.end(), greater<int>());

            if (k <= sortedVertices.size()) {
                cout << sortedVertices[k - 1] << endl;
            } else {
                cout << -1 << endl;
            }
        }
    }

    return 0;
}
