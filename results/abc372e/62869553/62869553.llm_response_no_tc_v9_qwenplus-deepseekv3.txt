const int MAXN = 200005;
vector<int> top[MAXN]; // Stores top 10 largest vertices in each component's root
int fa[MAXN];

int find(int x) {
    if (fa[x] != x) {
        fa[x] = find(fa[x]);
    }
    return fa[x];
}

void unite(int u, int v) {
    u = find(u);
    v = find(v);
    if (u == v) return;
    if (top[u].size() < top[v].size()) swap(u, v);

        top[u].push_back(num);
    }
    sort(top[u].rbegin(), top[u].rend());
    if (top[u].size() > 10) {
        top[u].resize(10);
    }
    fa[v] = u;
}
