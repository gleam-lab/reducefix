#include<bits/stdc++.h>
using namespace std;

// Define the maximum number of vertices and queries
#define MAXN 200005
#define MAXK 10

// Union-find structure
int parent[MAXN];
int size[MAXN];
int rank[MAXN];

void initialize(int n) {
    for (int i = 0; i <= n; ++i) {
        parent[i] = i;
        size[i] = 1;
        rank[i] = 0;
    }
}

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]); // Path compression
    }
    return parent[x];
}

void union_sets(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);
    if (rootX != rootY) {
        if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
            size[rootX] += size[rootY];
        } else if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
            size[rootY] += size[rootX];
        } else {
            parent[rootY] = rootX;
            rank[rootX] += 1;
            size[rootX] += size[rootY];
        }
    }
}

// Graph representation
vector<int> graph[MAXN];

// Query results
vector<int> answers;

int main() {
    int n, q;
    cin >> n >> q;
    initialize(n);

    for (int i = 0; i < q; ++i) {
        int op, u, v, k;
        cin >> op >> u >> v;
        if (op == 1) {
            // Union operation
            union_sets(u, v);
            graph[find(v)].push_back(find(u)); // Store edges for connectivity
        } else if (op == 2) {
            // Find k-th largest vertex connected to v
            int root_v = find(v);
            sort(graph[root_v].rbegin(), graph[root_v].rend()); // Sort in descending order
            int count = 0;
            for (int node : graph[root_v]) {
                if (find(node) == root_v) { // Check if node is still a valid vertex
                    count++;
                    if (count == k) {
                        answers.push_back(node);
                        break;
                    }
                }
            }
            if (count < k) {
                answers.push_back(-1);
            }
        }
    }

    for (int ans : answers) {
        cout << ans << endl;
    }

    return 0;
}
