#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(v);
    pq[v].push(u);
}

void removeEdge(int u, int v) {
    auto it = find(adj[u].begin(), adj[u].end(), v);
    if (it != adj[u].end()) {
        adj[u].erase(it);
        pq[v].erase(find(pq[v].begin(), pq[v].end(), u));
    }
    it = find(adj[v].begin(), adj[v].end(), u);
    if (it != adj[v].end()) {
        adj[v].erase(it);
        pq[u].erase(find(pq[u].begin(), pq[u].end(), v));
    }
}

int kthLargest(int v, int k) {
    while (!pq[v].empty() && pq[v].size() > k) pq[v].pop();
    return pq[v].size() >= k ? pq[v].top() : -1;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q; cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        pq[i].push(-1);
    }

    for (int i = 0; i < q; ++i) {
        int type; cin >> type;
        if (type == 1) {
            int u, v; cin >> u >> v;
            addEdge(u, v);
        } else {
            int v, k; cin >> v >> k;
            cout << kthLargest(v, k) << '\n';
        }
    }

    return 0;
}
