```cpp
#include <bits/stdc++.h>
using namespace std;

struct UnionFind {
    vector<int> parent;
    vector<vector<int>> size;

    explicit UnionFind(int n) : parent(n + 1), size(n + 1) {
        for (int i = 1; i <= n; ++i) {
            parent[i] = i;
            size[i].push_back(i);
        }
    }

    int find(int x) {
        if (parent[x] == x) return x;
        int root = find(parent[x]);
        parent[x] = root; // Path compression
        return root;
    }

    void unite(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX != rootY) {
            int maxSize;
            if (size[rootX].size() > size[rootY].size()) {
                maxSize = rootX;
                size[rootX].insert(size[rootX].end(), size[rootY].begin(), size[rootY].end());
            } else {
                maxSize = rootY;
                size[rootY].insert(size[rootY].end(), size[rootX].begin(), size[rootX].end());
            }
            parent[rootX] = parent[rootY] = maxSize;
        }
    }
};

int main() {
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);

    while (q--) {
        int op, u, v, k;
        cin >> op >> u >> v;
        if (op == 1) {
            uf.unite(u, v);
        } else if (op == 2) {
            int root = uf.find(v);
            if (k > uf.size[root].size()) {
                cout << -1 << endl;
            } else {
                cout << uf.size[root][uf.size[root].size() - k] << endl;
            }
        }
    }

    return 0;
}
```

Explanation:
1. I've used the Union-Find data structure with path compression to manage the union operations. This helps in efficiently finding and uniting sets.
2. Each union operation also maintains a size vector for each set to keep track of the elements in the set.
3. For each type 2 query, I find the root of the set containing `v`, then check if `k` is within the size of the set. If it is, I print the `k`-th largest element; otherwise, I print `-1`.
This code is efficient and handles the constraints of the problem well.
