#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(v);
    pq[v].push(u);
}

int kthLargest(int v, int k) {
    while (!pq[v].empty() && pq[v].size() > k) pq[v].pop();
    if (pq[v].size() < k) return -1;
    return pq[v].top();
}
