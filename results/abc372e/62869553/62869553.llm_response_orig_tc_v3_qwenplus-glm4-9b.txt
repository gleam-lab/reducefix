int parent[MAXN];          // Parent of each node

// Find function with path compression
int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

// Union function with sizes of components
void union_set(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);
    if (rootX != rootY) {
        if (sz[rootX] < sz[rootY]) {
            swap(rootX, rootY);
        }
        parent[rootY] = rootX;
        sz[rootX] += sz[rootY];
    }
}
