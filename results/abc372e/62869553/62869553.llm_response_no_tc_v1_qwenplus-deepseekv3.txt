const int MAXN = 200005;
int parent[MAXN];
vector<int> top[MAXN]; // Each component's top 10 largest vertices in descending order

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

void unite(int x, int y) {
    x = find(x);
    y = find(y);
    if (x == y) return;

    vector<int> merged;

}
