```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int TOP = 10;

int fa[MAXN];
set<int, greater<int>> group[MAXN]; // Store top 10 largest nodes in descending order

// Find with path compression
int find(int x) {
    if (fa[x] != x)
        fa[x] = find(fa[x]);
    return fa[x];
}

// Merge two sets: merge small into large
void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    
    if (root_u == root_v)
        return;
        
    // Always merge smaller set into larger one
    if (group[root_u].size() < group[root_v].size()) {
        swap(root_u, root_v);
        swap(fa[root_u], fa[root_v]);
    }
    
    // Merge the elements from group_v to group_u
    for (auto it : group[root_v]) {
        if (group[root_u].size() >= TOP) break;
        group[root_u].insert(it);
    }
    
    // Clear group_v since it's merged
    group[root_v].clear();
    fa[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and group data
    for (int i = 1; i <= N; ++i) {
        fa[i] = i;
        group[i].insert(i);
    }

    while (Q--) {
        int op;
        cin >> op;
        
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root_v = find(v);
            
            if (group[root_v].size() < (size_t)k) {
                cout << -1 << '\n';
            } else {
                auto it = group[root_v].begin();
                advance(it, k - 1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
```
