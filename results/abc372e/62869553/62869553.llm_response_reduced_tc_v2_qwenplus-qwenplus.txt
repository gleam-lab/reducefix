#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
int parent[MAXN];
set<int> component[MAXN];

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

void unite(int u, int v, int size_limit = 10) {
    int root_u = find(u);
    int root_v = find(v);
    if (root_u == root_v) return;

    // Always merge smaller into larger for efficiency
    if (component[root_u].size() < component[root_v].size()) {
        swap(root_u, root_v);
    }

    // Move elements from root_v to root_u
    for (int node : component[root_v]) {
        component[root_u].insert(node);
        if (component[root_u].size() > size_limit) {
            // Remove the smallest element to keep top 10
            component[root_u].erase(*component[root_u].begin());
        }
    }

    // Clear and update parent
    component[root_v].clear();
    parent[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and sets
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        component[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else if (type == 2) {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if (component[root].size() < k) {
                cout << -1 << endl;
            } else {
                auto it = component[root].rbegin(); // reverse iterator (descending)
                for (int i = 1; i < k && it != component[root].rend(); ++i) {
                    ++it;
                }
                cout << *it << endl;
            }
        }
    }

    return 0;
}
