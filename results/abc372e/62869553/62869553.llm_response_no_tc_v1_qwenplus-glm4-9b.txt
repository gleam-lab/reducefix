#include <bits/stdc++.h>
using namespace std;

const int MAX_N = 200005;
const int MAX_K = 10;

int parent[MAX_N];
vector<int> graph[MAX_N];
vector<int> order;
vector<int> vertexDegree[MAX_N];

void initialize(int n) {
    fill(parent, parent + n + 1, -1);
    order.clear();
    for (int i = 1; i <= n; ++i) {
        vertexDegree[i].clear();
    }
}

int findRoot(int x) {
    if (parent[x] == -1) return x;
    parent[x] = findRoot(parent[x]);
    return parent[x];
}

void unite(int x, int y) {
    int rootX = findRoot(x);
    int rootY = findRoot(y);
    if (rootX != rootY) {
        parent[rootX] = rootY;
    }
}

void addEdge(int x, int y) {
    int rootX = findRoot(x);
    int rootY = findRoot(y);
    if (rootX != rootY) {
        graph[rootX].push_back(rootY);
        graph[rootY].push_back(rootX);
        vertexDegree[rootX].push_back(y);
        vertexDegree[rootY].push_back(x);
    }
}

int kthLargest(int v, int k) {
    int root = findRoot(v);
    sort(vertexDegree[root].begin(), vertexDegree[root].end(), greater<int>());
    if (k > vertexDegree[root].size()) return -1;
    return vertexDegree[root][k - 1];
}

int main() {
    int n, q;
    cin >> n >> q;
    
    initialize(n);
    
    for (int i = 0; i < q; ++i) {
        int op, u, v, k;
        cin >> op >> u >> v;
        if (op == 1) {
            addEdge(u, v);
        } else if (op == 2) {
            cout << kthLargest(v, k) << endl;
        }
    }
    
    return 0;
}
