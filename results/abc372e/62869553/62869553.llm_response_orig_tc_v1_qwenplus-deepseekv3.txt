#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
const int MAXK = 10;

int parent[MAXN];
vector<int> topk[MAXN];

int find(int u) {
    if (parent[u] != u) {
        parent[u] = find(parent[u]);
    }
    return parent[u];
}

void merge(int u, int v) {
    u = find(u);
    v = find(v);
    if (u == v) return;

    if (topk[u].size() < topk[v].size()) {
        swap(u, v);
    }

    vector<int> merged;
    merge(topk[u].begin(), topk[u].end(), 
          topk[v].begin(), topk[v].end(),
          back_inserter(merged), greater<int>());
    
    topk[u].clear();
    for (int i = 0; i < min(MAXK, (int)merged.size()); ++i) {
        topk[u].push_back(merged[i]);
    }

    parent[v] = u;
}

int main() {

    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        topk[i].push_back(i);
    }

    while (Q--) {
        int op, u, v;

        if (op == 1) {
