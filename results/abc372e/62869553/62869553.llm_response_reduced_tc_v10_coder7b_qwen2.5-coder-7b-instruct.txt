#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void updatePQ(int node, int value) {
    pq[node].push(value);
    if(pq[node].size() > 10) pq[node].pop();
}

int kthLargest(int node, int k) {
    while(k-- && !pq[node].empty()) {
        int top = pq[node].top();
        pq[node].pop();
        if(!pq[node].empty() && pq[node].top() == top) continue;
        return top;
    }
    return -1;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int n, q; cin >> n >> q;
    for(int i = 1; i <= n; i++) addEdge(i, i);

    int u, v, k;
    for(int i = 0; i < q; i++) {
        cin >> u >> v;
        if(u != v) {
            addEdge(u, v);
            if(u > v) swap(u, v);
            updatePQ(u, v);
            updatePQ(v, u);
        } else {
            cin >> k;
            cout << kthLargest(u, k) << "\n";
        }
    }

    return 0;
}
