Here's the corrected implementation that addresses these issues:

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int K = 10;

int parent[MAXN];
vector<int> top[MAXN];

int find(int u) {
    while (parent[u] != u) {
        parent[u] = parent[parent[u]];
        u = parent[u];
    }
    return u;
}

void unite(int u, int v) {
    u = find(u);
    v = find(v);
    if (u == v) return;
    
    if (top[u].size() < top[v].size()) {
        swap(u, v);
    }
    
    vector<int> merged;
    merge(top[u].rbegin(), top[u].rend(), 
          top[v].rbegin(), top[v].rend(), 
          back_inserter(merged), greater<int>());
    
    if (merged.size() > K) {
        merged.resize(K);
    }
    top[u].assign(merged.begin(), merged.end());
    parent[v] = u;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        top[i] = {i};
    }
    
    while (Q--) {
        int op, u, v;
        cin >> op >> u >> v;
        
        if (op == 1) {
            unite(u, v);
        } else {
            int root = find(u);
            if (v > top[root].size()) {
                cout << "-1\n";
            } else {
                cout << top[root][v-1] << "\n";
            }
        }
    }
    
    return 0;
}
```

Key improvements made:

1. Correctly read N and Q in the proper order
2. Implemented proper Union-Find with path compression
3. Maintained a sorted list of up to K largest elements for each component
4. Proper merging of sorted lists during union operations
5. Efficient querying by directly accessing the precomputed top K elements

The algorithm now:
1. Uses a standard Union-Find structure with path compression
2. For each component, maintains a sorted list of up to 10 largest elements
3. When merging two components, combines their top elements and keeps only the largest 10
4. For queries, simply checks if the requested index is within bounds and returns the corresponding element

This approach ensures O(Q α(N)) time complexity for the Union-Find operations and O(K) time per query, where α is the inverse Ackermann function. The space complexity is O(NK).
