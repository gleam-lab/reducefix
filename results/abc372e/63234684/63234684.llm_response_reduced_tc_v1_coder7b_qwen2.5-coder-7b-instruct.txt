#include<bits/stdc++.h>
using namespace std;

struct UF {
    vector<int> parent, rank;
    UF(int n): parent(n), rank(n, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    void unite(int x, int y) {
        x = find(x); y = find(y);
        if (x == y) return;
        if (rank[x] < rank[y]) swap(x, y);
        parent[y] = x;
        if (rank[x] == rank[y]) rank[x]++;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q; cin >> n >> q;
    UF uf(n);
    map<int, set<int>> mp;

    for (int i = 0; i < q; ++i) {
        int type; cin >> type;
        if (type == 1) {
            int u, v; cin >> u >> v;
            --u; --v;
            uf.unite(u, v);
            mp[u].insert(v);
            mp[v].insert(u);
        } else {
            int v, k; cin >> v >> k;
            --v;
            int id = uf.find(v);
            auto &s = mp[id];
            if (k > s.size()) cout << -1 << '\n';
            else cout << *next(s.rbegin(), k - 1) + 1 << '\n';
        }
    }
}
