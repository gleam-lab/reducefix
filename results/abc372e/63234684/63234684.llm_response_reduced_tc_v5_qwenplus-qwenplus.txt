#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (int i = 0; i < (n); i++)
using P = pair<int, int>;
using T = tuple<int, int, int>;

struct Union_Find {
    vector<int> sz, par;
    vector<set<int>> components;

    Union_Find(int n) {
        sz.resize(n, 1);
        par.resize(n, -1);
        components.resize(n);
        for (int i = 0; i < n; ++i) {
            components[i].insert(i + 1); // 1-based numbering
        }
    }

    int root(int v) {
        if (par[v] == -1) return v;
        return par[v] = root(par[v]);
    }

    bool issame(int x, int y) {
        return root(x) == root(y);
    }

    void merge(int x, int y) {
        x = root(x), y = root(y);
        if (x == y) return;

        // Always merge smaller into larger
        if (components[x].size() < components[y].size()) swap(x, y);

        // Move all elements from y to x
        for (auto val : components[y]) {
            components[x].insert(val);
        }
        components[y].clear();

        sz[x] += sz[y];
        sz[y] = 0;
        par[y] = x;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, q;
    cin >> n >> q;
    Union_Find uf(n);

    while (q--) {
        int ty, a, b;
        cin >> ty >> a >> b;
        a--; // Convert to 0-based index

        if (ty == 1) {
            uf.merge(a, b - 1);
        } else {
            int r = uf.root(a);
            auto& comp = uf.components[r];

            if (b > (int)comp.size()) {
                cout << -1 << "\n";
            } else {
                // Get k-th largest element (descending order)
                auto it = comp.rbegin();
                advance(it, b - 1);
                cout << *it << "\n";
            }
        }
    }

    return 0;
}
