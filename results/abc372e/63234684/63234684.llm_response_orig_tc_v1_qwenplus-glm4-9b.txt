struct UnionFind {
  vector<int> parent, size, rank;

  UnionFind(int n) {
    parent.resize(n);
    size.resize(n);
    rank.resize(n);
    for (int i = 0; i < n; i++) {
      parent[i] = i;
      size[i] = 1;
    }
  }

  int find(int x) {
    if (parent[x] == x) return x;
    parent[x] = find(parent[x]);
    return parent[x];
  }

  bool unionSets(int x, int y) {
    x = find(x);
    y = find(y);
    if (x == y) return false;
    if (rank[x] < rank[y]) swap(x, y);
    if (rank[x] == rank[y]) rank[x]++;
    parent[y] = x;
    size[x] += size[y];
    return true;
  }
};
