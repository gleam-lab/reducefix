#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i = 0; i < n; i++)
using P = pair<int, int>;

struct UnionFind {
    vector<int> sz, par, rank, max_node;
    vector<vector<int>> uni;

    UnionFind(int n) {
        sz.resize(n, 1);
        par.resize(n, -1);
        rank.resize(n, 0);
        max_node.resize(n, n);  // Store the maximum node in the subtree
        uni.resize(n);
        for (int i = 0; i < n; i++) {
            uni[i].emplace_back(i);
        }
    }

    int root(int v) {
        if (par[v] == -1) return v;
        else return par[v] = root(par[v]);
    }

    bool issame(int x, int y) {
        return root(x) == root(y);
    }

    void merge(int x, int y) {
        x = root(x), y = root(y);
        if (x == y) return;
        if (rank[x] > rank[y]) swap(x, y);
        if (rank[x] == rank[y]) rank[x]++;
        par[x] = y;
        sz[y] += sz[x];
        max_node[y] = max(max_node[y], max_node[x]);
        uni[y].insert(uni[y].end(), uni[x].begin(), uni[x].end());
        sort(uni[y].rbegin(), uni[y].rend());
        uni[y].resize(10);  // Limit the size to 10 for efficiency
    }
};

int main() {
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<int> ans;

    for (int i = 0; i < q; i++) {
        int ty, a, b;
        cin >> ty >> a >> b;
        a--, b--;
        if (ty == 1) {
            uf.merge(a, b);
        } else {
            int root_v = uf.root(b);
            if (uf.uni[root_v].size() < b) {
                ans.push_back(-1);
            } else {
                ans.push_back(uf.uni[root_v][b] + 1);
            }
        }
    }

    for (int i = 0; i < ans.size(); i++) {
        cout << ans[i] << endl;
    }

    return 0;
}
