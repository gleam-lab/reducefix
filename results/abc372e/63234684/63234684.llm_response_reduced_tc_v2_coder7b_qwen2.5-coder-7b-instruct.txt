#include<bits/stdc++.h>
using namespace std;

class UF {
public:
    vector<int> parent, rank, size;
    UF(int n) : parent(n), rank(n, 1), size(n, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int u) {
        return parent[u] == u ? u : parent[u] = find(parent[u]);
    }

    void unite(int u, int v) {
        int pu = find(u), pv = find(v);
        if (pu != pv) {
            if (rank[pu] < rank[pv]) swap(pu, pv);
            parent[pv] = pu;
            size[pu] += size[pv];
            if (rank[pu] == rank[pv]) rank[pu]++;
        }
    }

    int get_size(int u) {
        return size[find(u)];
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);
    int n, q; cin >> n >> q;
    UF uf(n);

    priority_queue<int, vector<int>, greater<int>> pq[n];

    while(q--) {
        int type; cin >> type;
        if(type == 1) {
            int u, v; cin >> u >> v; --u; --v;
            uf.unite(u, v);
        } else {
            int v, k; cin >> v >> k; --v;
            int p = uf.find(v);
            if(k > pq[p].size()) cout << -1 << '\n';
            else cout << pq[p].top() + 1 << '\n';
        }
    }

    return 0;
}
