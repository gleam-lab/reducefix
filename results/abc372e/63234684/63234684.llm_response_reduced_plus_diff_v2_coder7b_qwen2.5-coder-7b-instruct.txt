#include<bits/stdc++.h>
using namespace std;

struct DSU {
    vector<int> parent, rank, size, max_val;
    int components;

    DSU(int n) {
        parent.resize(n + 1);
        iota(parent.begin(), parent.end(), 0);
        rank.resize(n + 1, 0);
        size.resize(n + 1, 1);
        max_val.resize(n + 1, INT_MIN);
        components = n;
    }

    int find(int u) {
        if (parent[u] != u) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }

    bool unite(int u, int v) {
        int pu = find(u), pv = find(v);
        if (pu == pv) return false;
        if (rank[pu] < rank[pv]) {
            swap(pu, pv);
        }
        parent[pv] = pu;
        if (rank[pu] == rank[pv]) rank[pu]++;
        size[pu] += size[pv];
        max_val[pu] = max(max_val[pu], max_val[pv]);
        components--;
        return true;
    }

    int get_size(int u) {
        return size[find(u)];
    }

    int get_max_value(int u) {
        return max_val[find(u)];
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    DSU dsu(n);

    while(q--) {
        int t;
        cin >> t;
        if (t == 1) {
            int u, v;
            cin >> u >> v;
            dsu.unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = dsu.find(v);
            if (k > dsu.get_size(root)) {
                cout << -1 << '\n';
            } else {
                cout << dsu.get_max_value(root) - k + 1 << '\n';
            }
        }
    }

    return 0;
}
