To solve this problem efficiently, we need to handle two types of queries on an initially disconnected graph:

1. **Adding an edge** between two nodes.
2. **Querying the k-th largest node number** in the connected component that includes a given node.

We can use a **Disjoint Set Union (DSU)** data structure with path compression and union by size/rank for efficient merging of components. However, since we also need to support **k-th largest queries**, we must maintain additional information: a sorted list (in descending order) of the vertices in each connected component.

### Key Optimizations:
- Maintain a `vector<int>` within each DSU set that stores up to the **top 10 largest vertex values** in descending order. This is sufficient because `k <= 10` for type 2 queries.
- When two sets are merged, merge their top lists, sort them in descending order, and retain only the top 10 elements.

This ensures:
- Efficient merging (`O(α(N))` time per operation due to DSU optimizations).
- Efficient query (`O(1)` access to the k-th element).

Here is the corrected and optimized C++ code:

```cpp
#include <bits/stdc++.h>
using namespace std;

struct UnionFind {
    vector<int> parent, size;
    vector<vector<int>> top_nodes;

    UnionFind(int n) {
        parent.resize(n);
        size.resize(n, 1);
        top_nodes.resize(n);
        for (int i = 0; i < n; ++i) {
            top_nodes[i].push_back(i);
        }
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    void unite(int x, int y) {
        int root_x = find(x);
        int root_y = find(y);
        if (root_x == root_y) return;

        // Always merge smaller into larger for efficiency
        if (size[root_x] < size[root_y]) swap(root_x, root_y);

        // Merge the top nodes list
        for (int val : top_nodes[root_y]) {
            top_nodes[root_x].push_back(val);
        }

        // Sort and keep at most top 10 largest
        sort(top_nodes[root_x].rbegin(), top_nodes[root_x].rend());
        if (top_nodes[root_x].size() > 10) {
            top_nodes[root_x].resize(10);
        }

        // Update size
        size[root_x] += size[root_y];
        parent[root_y] = root_x;
    }

    vector<int>& get_top(int x) {
        return top_nodes[find(x)];
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    UnionFind uf(N);

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            --u, --v;
            uf.unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            --v;
            auto& top = uf.get_top(v);
            if (k > (int)top.size()) {
                cout << -1 << '\n';
            } else {
                cout << top[k - 1] + 1 << '\n'; // Convert from 0-indexed to 1-based
            }
        }
    }

    return 0;
}
```

### ✅ Explanation of Fixes:
- The original implementation had incorrect handling of merge operations and sorting logic.
- Merging small sets into large ones helps reduce overhead.
- Keeping only the top 10 largest nodes avoids unnecessary computation and memory usage.
- Accessing the k-th largest is done via direct indexing after ensuring the list is sorted and valid.

This solution is both **correct** and **efficient** under constraints (`N, Q ≤ 2 * 10^5`).
