1. **Union-Find Structure**: Instead of maintaining a list of vertices for each root, you can maintain a size for each component. When merging, you can add the size of the smaller component to the larger one. This way, you can find the component quickly using the parent pointers.

2. **Adding Edges**: When adding an edge, if the vertices are in different components, merge the two components.

3. **Querying for k-th Largest**: Find the root of the vertex and traverse the list of vertices in that component. Keep track of the size of the component and the index of the vertex in the traversal. If you reach the k-th vertex, return its number. If the size of the component is less than k, return -1.

4. **Efficiency**: You should make sure the operations are efficient. For the union-find, path compression and union by size are commonly used techniques to make the operations nearly constant time.

struct UnionFind {
    vector<int> par, rank, size;
    UnionFind(int n) {
        par.resize(n);
        rank.resize(n, 0);
        size.resize(n, 1);
        for (int i = 0; i < n; i++) {
            par[i] = i;
        }
    }
    int find(int u) {
        if (par[u] != u) par[u] = find(par[u]);
        return par[u];
    }
    void merge(int u, int v) {
        u = find(u);
        v = find(v);
        if (u == v) return;
        if (rank[u] < rank[v]) swap(u, v);
        size[u] += size[v];
        par[v] = u;
        if (rank[u] == rank[v]) rank[u]++;
    }
};
