#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (int i = 0; i < (n); i++)
using P = pair<int, int>;
using T = tuple<int, int, int>;

struct Union_Find {
    vector<int> par, sz;
    vector<set<int>> components;

    Union_Find(int n) {
        par.resize(n);
        sz.resize(n, 1);
        components.resize(n);
        for (int i = 0; i < n; ++i) {
            par[i] = i;
            components[i].insert(i);
        }
    }

    int root(int x) {
        if (par[x] != x) par[x] = root(par[x]);
        return par[x];
    }

    bool same(int x, int y) {
        return root(x) == root(y);
    }

    void unite(int u, int v) {
        int ru = root(u), rv = root(v);
        if (ru == rv) return;

        // Merge smaller component into larger one
        if (components[ru].size() < components[rv].size()) swap(ru, rv);

        // Move all elements from rv to ru
        for (int node : components[rv]) {
            components[ru].insert(node);
        }
        components[rv].clear();
        par[rv] = ru;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    Union_Find uf(N);

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            --u, --v;
            uf.unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            --v;
            int root_v = uf.root(v);
            if (k > (int)uf.components[root_v].size()) {
                cout << -1 << '\n';
            } else {
                // Accessing the k-th largest element in a set
                auto it = uf.components[root_v].rbegin();
                for (int i = 0; i < k - 1; ++i) ++it;
                cout << (*it) + 1 << '\n';  // Convert 0-based to 1-based
            }
        }
    }

    return 0;
}
