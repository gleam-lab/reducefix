#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for(int i = 0; i < n; i++)
using P = pair<int, int>;
using T = tuple<int, int, int>;

struct Union_Find {
    vector<int> sz, par;
    vector<vector<int>> edges; // This vector will store edges connected to each vertex

    Union_Find(int n) {
        sz.resize(n, 1);
        par.resize(n, -1);
        edges.resize(n);
    }

    int root(int v) {
        if (par[v] == -1) return v;
        else return par[v] = root(par[v]);
    }

    bool issame(int x, int y) {
        return root(x) == root(y);
    }

    void merge(int x, int y) {
        x = root(x), y = root(y);
        if (x == y) return;
        if (sz[x] < sz[y]) swap(x, y);
        sz[x] += sz[y];
        sz[y] = 0;
        par[y] = x;
        edges[x].insert(edges[x].end(), edges[y].begin(), edges[y].end());
        edges[y].clear();
    }
};

int main() {
    int n, q;
    cin >> n >> q;
    Union_Find uf(n);

    while (q--) {
        int ty, a, b;
        cin >> ty >> a >> b;
        a--, b--;
        if (ty == 1) {
            uf.merge(a, b);
        } else {
            int r = uf.root(a);
            if (uf.edges[r].size() < b) {
                cout << -1 << endl;
            } else {
                cout << uf.edges[r][b] + 1 << endl;
            }
        }
    }

    return 0;
}
