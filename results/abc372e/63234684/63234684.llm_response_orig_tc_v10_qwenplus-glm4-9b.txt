### Solution Overview

1. **Data Structures**:
    - **Union-Find**: To manage connected components efficiently.
    - **Priority Queue**: To keep track of vertices in each connected component for efficient retrieval of the k-th largest vertex.

2. **Union-Find Operations**:
    - `root(v)`: Find the root of vertex v, with path compression.
    - `issame(x, y)`: Check if vertices x and y are in the same connected component.
    - `merge(x, y)`: Merge the components of x and y and update the priority queue accordingly.

3. **Handling Queries**:
    - For type 1 queries, add the edge and merge the components of the two vertices using `merge()`.
    - For type 2 queries, if the number of vertices in the component of v is less than k, print `-1`. Otherwise, retrieve and print the k-th largest vertex from the priority queue.

#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i = 0; i < n; i++)
using P = pair<int, int>;
using T = tuple<int, int, int>;

struct Union_Find {
    vector<int> sz, par, rank, pq;
    vector<vector<int>> uni;

    Union_Find(int n) {
        par.resize(n, -1);
        sz.resize(n, 1);
        pq.resize(n);
        rank.resize(n, 0);
        rep(i, n) pq[i] = (priority_queue<int, vector<int>, greater<int>>) {i};
        uni.resize(n);
        rep(i, n) uni[i].push_back(i);
    }
    
    int root(int v) {
        if (par[v] == -1) return v;
        else par[v] = root(par[v]);
        return par[v];
    }
    
    void merge(int x, int y) {
        x = root(x), y = root(y);
        if (x == y) return;
        if (rank[x] < rank[y]) swap(x, y);
        sz[x] += sz[y];
        if (rank[x] == rank[y]) rank[x]++;
        par[y] = x;
        auto &q1 = pq[x], &q2 = pq[y];
        while (!q2.empty()) {
            q1.push(q2.top());
            q2.pop();
        }
        pq[y] = q1;
        sort(q1.begin(), q1.end());
        while (q1.size() > 9) q1.pop();
    }
    
    int size(int v) {
        return sz[root(v)];
    }
    
    int kth_largest(int v, int k) {
        int r = root(v);
        if (size(root(v)) < k) return -1;
        auto &q = pq[r];
        for (int i = 0; i < k; i++) {
            if (q.empty()) return -1;
            if (i == k - 1) return q.top() + 1; // Return 1-based index
            q.pop();
        }
        return -1;
    }
};

}
