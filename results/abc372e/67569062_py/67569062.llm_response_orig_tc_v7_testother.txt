class UnionFind:
    def __init__(self, n):
        self.par = list(range(n))
        self.size = [1] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]

    def unite(self, x, y):
        x = self.find(x)
        y = self.find(y)
        if x == y:
            return False
        if self.size[x] < self.size[y]:
            x, y = y, x
        self.par[y] = x
        self.size[x] += self.size[y]
        return True

    def same(self, x, y):
        return self.find(x) == self.find(y)

N, Q = map(int, input().split())

# Store the vertices in each component (as sorted list by descending order for k-th largest)
components = [[] for _ in range(N + 1)]
for i in range(1, N + 1):
    components[i] = [i]

uf = UnionFind(N + 1)

ans = []
for _ in range(Q):
    query = list(map(int, input().split()))
    if query[0] == 1:
        u, v = query[1], query[2]
        if not uf.same(u, v):
            root_u = uf.find(u)
            root_v = uf.find(v)
            # Always merge smaller into larger (by root index, but we can use size or just maintain consistency)
            if uf.size[root_u] < uf.size[root_v]:
                u, v = v, u
                root_u, root_v = root_v, root_u
            
            # Merge components: add all vertices from root_v to root_u
            merged = components[root_u] + components[root_v]
            # Sort in descending order to get largest first
            merged.sort(reverse=True)
            # Keep at most 10 largest elements as k <= 10
            merged = merged[:10]
            new_root = uf.find(u)  # After union, this should be the root
            uf.unite(u, v)
            final_root = uf.find(u)
            components[final_root] = merged
    else:
        v, k = query[1], query[2]
        root = uf.find(v)
        comp = components[root]
        if k <= len(comp):
            ans.append(comp[k - 1])  # k-th largest is at index k-1
        else:
            ans.append(-1)

for a in ans:
    print(a)
