class UnionFind:
    def __init__(self, n):
        self.par = list(range(n))
        self.size = [1] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]

    def unite(self, x, y):
        x = self.find(x)
        y = self.find(y)
        if x == y:
            return False
        if self.size[x] < self.size[y]:
            x, y = y, x
        self.par[y] = x
        self.size[x] += self.size[y]
        return True

    def same(self, x, y):
        return self.find(x) == self.find(y)

N, Q = map(int, input().split())

# For each component root, maintain a sorted list (in descending order) of the largest at most 10 vertex numbers
mem = [[] for _ in range(N+1)]
for i in range(1, N+1):
    mem[i] = [i]

ans = []
uf = UnionFind(N+1)

for _ in range(Q):
    query = list(map(int, input().split()))
    if query[0] == 1:
        u, v = query[1], query[2]
        if not uf.same(u, v):
            root_u = uf.find(u)
            root_v = uf.find(v)
            # Always merge smaller into larger (by root index, but we want to keep the tree shallow)
            # But we need to merge the lists: combine and keep top 10 largest values
            uf.unite(u, v)
            new_root = uf.find(u)
            
            # Get the two lists
            list_u = mem[root_u]
            list_v = mem[root_v]
            
            # Merge the two lists and keep at most 10 largest elements
            merged = []
            i = j = 0
            # Since both lists are maintained in descending order, we can merge them like merge sort
            while len(merged) < 10 and (i < len(list_u) or j < len(list_v)):
                if i >= len(list_u):
                    merged.append(list_v[j])
                    j += 1
                elif j >= len(list_v):
                    merged.append(list_u[i])
                    i += 1
                else:
                    if list_u[i] > list_v[j]:
                        merged.append(list_u[i])
                        i += 1
                    else:
                        merged.append(list_v[j])
                        j += 1
            
            mem[new_root] = merged
            # Clear the old root's list to avoid using it again
            if root_u != new_root:
                mem[root_u] = []
            if root_v != new_root:
                mem[root_v] = []
    else:  # query[0] == 2
        v, k = query[1], query[2]
        root = uf.find(v)
        component_list = mem[root]
        if k <= len(component_list):
            ans.append(component_list[k-1])
        else:
            ans.append(-1)

for a in ans:
    print(a)
