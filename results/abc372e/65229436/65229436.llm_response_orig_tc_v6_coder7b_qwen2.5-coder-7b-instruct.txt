To solve this problem, we need to handle two types of queries efficiently:
1. Adding an edge between two vertices.
2. Finding the k-th largest vertex number among the vertices connected to a given vertex.

Given the constraints, using a union-find structure to manage the connectivity of vertices is essential. Additionally, using a sorted set to keep track of the neighbors' degrees can help quickly find the k-th largest element.

Here's the corrected implementation:

#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i=0;i<(n);i++)
#define all(a) a.begin(), a.end()

using ll = long long;

struct UnionFind {
    vector<int> parent, rank;
    UnionFind(int n) : parent(n), rank(n, 0) {
        for (int i = 0; i < n; i++) 
            parent[i] = i;
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return false;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
        } else {
            parent[ry] = rx;
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<set<int>> cnt(n);
    rep(i, n) cnt[i].insert(i);

    vector<int> ans;
    rep(qi, q) {
        int com;
        cin >> com;
        if (com == 1) {
            int u, v;
            cin >> u >> v;
            u--; v--;
            int ru = uf.findRoot(u);
            int rv = uf.findRoot(v);
            if (ru != rv) {
                if (cnt[ru].size() < cnt[rv].size()) swap(ru, rv);
                uf.unite(ru, rv);
                int rnew = uf.findRoot(ru);
                for (int x : cnt[rv]) cnt[rnew].insert(x);
                cnt[rv].clear();
            }

            int v, k;

            v--;
            if (cnt[uf.findRoot(v)].size() < k) ans.push_back(-1);

                advance(it, k - 1);
                ans.push_back(*it + 1);
            }
        }
    }

}

### Explanation:
1. **Union-Find Structure**: This is used to manage the connectivity of vertices. It supports finding the root of a set and uniting two sets.
2. **Sorted Sets for Neighbors**: For each vertex, we maintain a sorted set of its neighbors' indices. This allows us to efficiently find the k-th largest neighbor.
3. **Handling Queries**:
   - **Type 1 Query**: Unites two sets and merges their neighbor sets.
   - **Type 2 Query**: Finds the k-th largest neighbor of a given vertex using the sorted set.

This approach ensures that both operations are efficient and handle the constraints effectively.
