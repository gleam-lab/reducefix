To fix this code, we need to address the issue where the incorrect output was generated for the given test case. The problem involves maintaining connected components in an undirected graph and answering queries about the k-th largest vertex in a component. The initial approach used a Union-Find data structure with sets to keep track of vertices in each component, but it failed to correctly handle the merging of sets and the retrieval of the k-th largest element.

### Approaches
1. **Union-Find with Path Compression and Union by Rank**: This helps in efficiently managing the connected components.
2. **Maintaining Sorted Lists for Each Component**: Instead of using sets, which can be slower for large operations, we maintain sorted lists and merge them in a way that keeps the larger list, ensuring efficient merging operations.
3. **Handling Queries Efficiently**: For each query, we find the root of the component and directly access the k-th largest element from the sorted list, leveraging the fact that the list is maintained in sorted order.

### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i = 0; i < (n); i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()
using ll = long long;
using pi = pair<int, int>;
const int INF = 1e9;

struct UnionFind {
    vector<int> parent, rank;
    vector<vector<int>> components;
    UnionFind(int n) : parent(n), rank(n, 0), components(n) {
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            components[i].push_back(i);
        }
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    void unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            components[ry].insert(components[ry].end(), all(components[rx]));
            sort(rall(components[ry]));
            components[rx].clear();
        } else {
            parent[ry] = rx;
            components[rx].insert(components[rx].end(), all(components[ry]));
            sort(rall(components[rx]));
            components[ry].clear();
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<int> ans;
    rep(qi, q) {
        int com;
        cin >> com;
        if (com == 1) {
            int u, v;
            cin >> u >> v;
            u--; v--;
            uf.unite(u, v);
        } else if (com == 2) {
            int v, k;
            cin >> v >> k;
            v--;
            int root = uf.findRoot(v);
            if (uf.components[root].size() < k) {
                ans.push_back(-1);
            } else {
                ans.push_back(uf.components[root][k - 1] + 1);
            }
        }
    }
    for (auto x : ans) cout << x << endl;
    return 0;
}
```

### Explanation
1. **Union-Find Structure**: The `UnionFind` class is used to manage the connected components with path compression and union by rank to ensure efficient operations.
2. **Component Tracking**: Each component is maintained as a sorted list in descending order. When two components are merged, the larger component absorbs the smaller one, and the combined list is kept sorted.
3. **Query Handling**: For each Type 2 query, the root of the vertex is found, and the k-th largest element is directly accessed from the sorted list of the component. If the component size is insufficient, `-1` is returned.
4. **Efficiency**: The merging of components is optimized by always merging into the larger component, and the sorted lists allow O(1) access to the k-th largest element, ensuring the solution handles the constraints efficiently.
