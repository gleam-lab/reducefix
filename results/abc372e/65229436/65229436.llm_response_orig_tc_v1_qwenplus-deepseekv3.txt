3. **Tracking Connected Vertices**: For each component, maintain a set of vertices sorted in descending order to quickly access the k-th largest vertex when needed. This ensures O(log N) time for insertion and deletion operations.

#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i = 0; i < (n); i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()
using ll = long long;
using pi = pair<int, int>;
const int INF = 1e9;

struct UnionFind {
    vector<int> parent, rank;
    vector<set<int, greater<int>>> vertices;
    UnionFind(int n) : parent(n), rank(n, 0), vertices(n) {
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            vertices[i].insert(i);
        }
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    void unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            vertices[ry].insert(all(vertices[rx]));
            vertices[rx].clear();
        } else {
            parent[ry] = rx;
            vertices[rx].insert(all(vertices[ry]));
            vertices[ry].clear();
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
    }
};

int main() {
    ios::sync_with_stdio(false);

    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<int> ans;
    while (q--) {

            u--; v--;

            v--;
            int root = uf.findRoot(v);

            }
        }
    }

    return 0;
}

### Explanation
1. **UnionFind Structure**: The `UnionFind` class manages the connected components using path compression and union by rank to ensure efficient operations.
2. **Tracking Vertices**: Each root maintains a set of vertices in descending order. This allows efficient access to the k-th largest vertex in the component.
3. **Unite Operation**: When merging two components, the vertices of the smaller component are merged into the larger component's set, ensuring the set remains sorted.
4. **Query Handling**: For type 2 queries, the root of the vertex is found, and the set is checked for size. If the size is sufficient, the k-th largest vertex is retrieved; otherwise, -1 is returned.

This approach ensures that the solution is both efficient and correct, handling all edge cases and large inputs within the constraints.
