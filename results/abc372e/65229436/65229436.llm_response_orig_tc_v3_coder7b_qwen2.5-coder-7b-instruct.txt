To solve this problem, we need to efficiently manage the addition of edges and querying the k-th largest connected node for each vertex. Here's a step-by-step breakdown of the solution:

1. **Union-Find Data Structure**: We use a union-find structure to keep track of connected components. This allows us to efficiently add edges and find the root of any vertex.

2. **Priority Queue for Connected Nodes**: For each component, we maintain a priority queue (max-heap) to store the sizes of the connected nodes. This helps in quickly finding the k-th largest node.

3. **Update Operations**:
   - When adding an edge, merge the two sets and update the size in the priority queue.
   - When querying the k-th largest node, simply fetch the top element from the priority queue of the relevant component.

#define rep(i, n) for(int i=0;i<(n);i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()
using ll = long long;

struct UnionFind {
    vector<int> parent, rank, size;
    UnionFind(int n) : parent(n), rank(n, 0), size(n, 1) {
        iota(all(parent), 0);
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return false;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            size[ry] += size[rx];
        } else {
            parent[ry] = rx;
            size[rx] += size[ry];
            if (rank[rx] == rank[ry]) 
                rank[rx]++;
        }
        return true;
    }
};
