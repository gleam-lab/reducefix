#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

const int MAX_N = 200100;
vector<int> parent(MAX_N);
vector<int> rank(MAX_N);

void initUnionFind(int n) {
    for (int i = 0; i < n; ++i) {
        parent[i] = i;
        rank[i] = 0;
    }
}

int findRoot(int x) {
    if (parent[x] != x) {
        parent[x] = findRoot(parent[x]);
    }
    return parent[x];
}

void unite(int x, int y) {
    int rootX = findRoot(x);
    int rootY = findRoot(y);
    if (rootX != rootY) {
        if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;

    initUnionFind(n);
    vector<set<int>> graph(n);

    while (q--) {
        int query_type;
        cin >> query_type;
        if (query_type == 1) {
            int u, v;
            cin >> u >> v;
            --u, --v;
            unite(u, v);
            graph[u].insert(v);
            graph[v].insert(u);
        } else if (query_type == 2) {
            int v, k;
            cin >> v >> k;
            --v;
            int root_v = findRoot(v);
            if (graph[root_v].size() < k) {
                cout << -1 << endl;
            } else {
                auto it = graph[root_v].rbegin();
                advance(it, k - 1);
                cout << *it + 1 << endl;
            }
        }
    }

    return 0;
}
