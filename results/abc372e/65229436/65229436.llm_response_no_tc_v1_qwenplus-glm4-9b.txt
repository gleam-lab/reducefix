struct UnionFind {
    vector<int> parent, rank;
    UnionFind(int n) : parent(n, -1) {
        for (int i = 0; i < n; i++) 
            parent[i] = i;
    }
    int findRoot(int x) {
        if (parent[x] < 0) return x;
        else return parent[x] = findRoot(parent[x]);
    }
    bool unite(int x, int y) {
        x = findRoot(x);
        y = findRoot(y);
        if (x == y) return false;
        if (rank[x] > rank[y]) swap(x, y);
        if (rank[x] == rank[y]) rank[x]++;
        parent[x] = y;
        return true;
    }
};

            int root = uf.findRoot(v);

### Explanation:
1. **UnionFind Structure**: Improved the UnionFind structure to use path compression and union by rank to ensure efficient find and unite operations.
2. **Adjacency List**: Instead of maintaining a set of connected vertices, we use an adjacency list to store all connected vertices.
3. **Query Handling**: For each Type 2 query, after finding the root of the connected component for vertex `v`, we iterate through all vertices in the adjacency list and collect all 1-based indices of the vertices in the same connected component.
4. **Output**: For each Type 2 query, if the number of vertices in the connected component is less than `k`, output `-1`. Otherwise, output the `k`-th largest vertex number.
