```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i=0;i<(n);i++)
#define all(a) a.begin(), a.end()
using ll = long long;
const int INF = 1e9;

// A structure to represent a union-find set
struct UnionFind {
    vector<int> parent, rank;
    UnionFind(int n) : parent(n), rank(n, 0) {
        for (int i = 0; i < n; i++) 
            parent[i] = i;
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return false;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
        } else {
            parent[ry] = rx;
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, q;
    cin >> n >> q;
    UnionFind uf(n);
    vector<vector<int>> connectedVertices(n);
    
    while (q--) {
        int com;
        cin >> com;
        if (com == 1) {
            int u, v;
            cin >> u >> v;
            u--; v--;
            uf.unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            v--;
            vector<int> vertices = connectedVertices[v];
            if (vertices.size() < k) {
                cout << -1 << endl;
            } else {
                // Output the k-th largest element (0-indexed in vector)
                cout << vertices[k-1] + 1 << endl;
            }
        }
    }
    
    return 0;
}
```

### Explanation:
1. **Union-Find Data Structure**: The `UnionFind` structure is used to keep track of connected components in the graph.
2. **.connectedVertices Array**: Instead of using a `set` for each component to store vertices, a `vector` is used. This is because we are interested in finding the k-th largest element, and `vector` allows random access which is more efficient than `set`.
3. **Handling Queries**:
   - For type 1 queries, we call `unite` on the union-find structure to merge components.
   - For type 2 queries, we add the vertex to the list of connected vertices for the given vertex `v`. We then directly access the `k-1`th element to get the k-th largest vertex (considering 0-based indexing).
4. **Output**: For each type 2 query, we output the result directly or `-1` if there are fewer than `k` vertices connected to `v`.
