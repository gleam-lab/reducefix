#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <algorithm>

using namespace std;

const int MAX_N = 200000;

int find_set(vector<int>& parents, int v) {
    if (parents[v] == v) return v;
    parents[v] = find_set(parents, parents[v]);
    return parents[v];
}

void unite_set(vector<int>& parents, vector<int>& ranks, int u, int v) {
    u = find_set(parents, u);
    v = find_set(parents, v);
    if (u == v) return;

    if (ranks[u] < ranks[v]) swap(u, v);

    parents[v] = u;

    if (ranks[u] == ranks[v])
        ranks[u]++;
}

int main() {
    int n, q;
    cin >> n >> q;

    vector<int> parents(MAX_N);
    vector<int> ranks(MAX_N, 0);
    map<int, vector<int>> adj_list;

    for (int i = 0; i < MAX_N; ++i) {
        parents[i] = i;
    }

    while (q--) {
        int type, v, k;
        cin >> type >> v >> k;

        if (type == 1) {
            int u;
            cin >> u;
            unite_set(parents, ranks, u - 1, v - 1);
        } else {
            int root = find_set(parents, v - 1);
            if (k > adj_list[root].size()) {
                cout << -1 << endl;
            } else {
                cout << adj_list[root][adj_list[root].size() - k] + 1 << endl;
            }
        }
    }

    return 0;
}
