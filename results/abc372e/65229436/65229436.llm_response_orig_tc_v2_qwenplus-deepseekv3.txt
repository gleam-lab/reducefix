### Approaches

2. **Maintaining Sorted Lists**: For each component, maintain a set or a sorted list of vertices to efficiently answer k-th largest queries. When two components are merged, the smaller set is merged into the larger one to maintain efficiency.
3. **Handling Queries**: For each type 2 query, check if the component size is at least k. If it is, retrieve the k-th largest element from the sorted list of the component; otherwise, return -1.

#include <bits/stdc++.h>

#define all(a) a.begin(), a.end()

struct UnionFind {
    vector<int> parent, rank;
    vector<set<int>> components;
    UnionFind(int n) : parent(n), rank(n, 0), components(n) {
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            components[i].insert(i);
        }
    }
    int findRoot(int x) {
        if (parent[x] != x) 
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }
    void unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
            for (int elem : components[rx]) 
                components[ry].insert(elem);
            components[rx].clear();
        } else {
            parent[ry] = rx;
            for (int elem : components[ry]) 
                components[rx].insert(elem);
            components[ry].clear();
            if (rank[rx] == rank[ry])
                rank[rx]++;
        }
    }
};

int main() {

}
