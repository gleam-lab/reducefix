#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

struct UnionFind {
    vector<int> parent, rank;
    UnionFind(int n) : parent(n), rank(n, 0), size(n, 1) {
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    
    int findRoot(int x) {
        if (parent[x] != x) {
            parent[x] = findRoot(parent[x]);
        }
        return parent[x];
    }
    
    bool unite(int x, int y) {
        x = findRoot(x);
        y = findRoot(y);
        if (x == y) return false;
        if (rank[x] < rank[y]) {
            swap(x, y);
        }
        if (rank[x] == rank[y]) rank[x]++;
        parent[y] = x;
        size[x] += size[y];
        return true;
    }
    
    int getSize(int x) {
        return size[findRoot(x)];
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    UnionFind uf(N);
    vector<set<int>> graph(N);
    
    int query;
    while (cin >> query) {
        if (query == 1) {
            int u, v;
            cin >> u >> v;
            uf.unite(u, v);
        } else if (query == 2) {
            int v, k;
            cin >> v >> k;
            int root = uf.findRoot(v - 1);
            int size = uf.getSize(root);
            if (k > size) {
                cout << -1 << endl;
            } else {
                auto it = graph[root].upper_bound(-k);
                cout << *it << endl;
            }
        }
    }
    
    return 0;
}
