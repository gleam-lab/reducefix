#include <bits/stdc++.h>

#define all(a) a.begin(), a.end()

using ll = long long;

struct UnionFind {
    vector<int> parent, rank;
    vector<set<int>> components;

    UnionFind(int n) : parent(n), rank(n, 0), components(n) {
        for (int i = 0; i < n; ++i) {
            parent[i] = i;
            components[i].insert(i);
        }
    }

    int findRoot(int x) {
        if (parent[x] != x)
            parent[x] = findRoot(parent[x]);
        return parent[x];
    }

    bool unite(int x, int y) {
        int rx = findRoot(x), ry = findRoot(y);
        if (rx == ry) return false;

        // Always merge smaller set into larger one
        if (components[rx].size() < components[ry].size())
            swap(rx, ry);

        // Merge ry's component into rx's component
        for (int val : components[ry])
            components[rx].insert(val);
        components[ry].clear();
        parent[ry] = rx;

        return true;
    }

    set<int>* getComponent(int x) {
        return &components[findRoot(x)];
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, q;
    cin >> n >> q;

    UnionFind uf(n);

    vector<int> ans;

    while (q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            u--; v--;
            uf.unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            v--;
            auto comp = uf.getComponent(v);
            if (comp->size() < (size_t)k) {
                ans.push_back(-1);
            } else {
                auto it = comp->rbegin();
                advance(it, k - 1);
                ans.push_back(*it + 1);
            }
        }
    }

    for (auto x : ans)
        cout << x << "\n";

    return 0;
}
