#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int KMAX = 10;

int parent[MAXN];
set<int, greater<int>> group[MAXN]; // Store top 10 largest vertices in descending order

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int u, int v) {
    int rootU = find(u);
    int rootV = find(v);
    
    if (rootU == rootV)
        return;

    // Merge smaller group into larger group for efficiency
    if (group[rootU].size() < group[rootV]].size())
        swap(rootU, rootV);

    // Transfer elements from rootV to rootU
    for (auto it : group[rootV]) {
        if (group[rootU].size() >= KMAX)
            group[rootU].erase(--group[rootU].end());
        group[rootU].insert(it);
    }

    parent[rootV] = rootU;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;

    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        group[i].insert(i);
    }

    while (Q--) {
        int op;
        cin >> op;

        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else if (op == 2) {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            
            if (k > (int)group[root].size()) {
                cout << -1 << '\n';
            } else {
                auto it = group[root].begin();
                advance(it, k - 1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
