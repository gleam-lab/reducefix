To solve this problem, we need to maintain a list of connected vertices for each vertex and efficiently handle both adding edges and querying the k-th largest vertex. We can use a combination of Union-Find (Disjoint Set Union) to manage the connected components and priority queues to keep track of the vertices' degrees.

Here's the corrected code:

```cpp
#include<bits/stdc++.h>
using namespace std;

struct DSU {
    vector<int> parent, rank, size, pq;
    int n;

    DSU(int n) : n(n), parent(n + 1), rank(n + 1, 0), size(n + 1, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    void merge(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return;
        if (rank[x] > rank[y]) swap(x, y);
        parent[x] = y;
        rank[y] += (rank[x] == rank[y]);
        size[y] += size[x];
        pq[y].insert(pq[x].begin(), pq[x].end());
        pq[y].erase(unique(pq[y].begin(), pq[y].end()), pq[y].end());
        pq[y].push_back(size[y]);
    }

    vector<int> getLargestK(int root, int k) {
        auto &p = pq[root];
        if (p.size() < k) return {-1};
        return {*(--p.end())};
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    DSU dsu(n);

    while (q--) {
        int op, ix, iy;
        cin >> op >> ix >> iy;
        if (op == 1) {
            dsu.merge(ix, iy);
        } else {
            auto result = dsu.getLargestK(dsu.find(ix), iy);
            cout << result[0] << "\n";
        }
    }

    return 0;
}
```

### Explanation:
1. **DSU Class**: 
   - `find` method to find the root of a set.
   - `merge` method to union two sets and update the priority queue.
   - `getLargestK` method to retrieve the k-th largest element from the priority queue of a set.

2. **Main Function**:
   - Read input values.
   - Initialize the DSU structure.
   - Process each query:
     - For type 1 queries, call `merge`.
     - For type 2 queries, call `getLargestK` and output the result.

This approach ensures that both operations (union and query) are efficient, maintaining the required time complexity.
