#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
const int MAXK = 10;

vector<int> adj[MAXN];

void dfs(int node, int parent, vector<int>& size, vector<int>& depth, vector<int>& result, int& pos) {
    size[node] = 1;
    for (auto child : adj[node]) {
        if (child != parent) {
            depth[child] = depth[node] + 1;
            dfs(child, node, size, depth, result, pos);
            size[node] += size[child];
        }
    }
    while (pos >= 0 && size[node] > result[pos]) pos--;
    result.insert(result.begin() + ++pos, node);
}

void preprocess(vector<int>& size, vector<int>& depth, vector<vector<int>>& result) {
    for (int i = 1; i <= n; ++i) {
        dfs(i, -1, size, depth, result[i], pos);
    }
}

int kthLargest(int node, int k, vector<int>& depth, vector<vector<int>>& result) {
    int pos = lower_bound(result[node].begin(), result[node].end(), node) - result[node].begin();
    int idx = pos + k - 1;
    if (idx >= result[node].size()) return -1;
    return result[node][idx];
}

int main() {
    cin >> n >> q;
    vector<int> size(n + 1), depth(n + 1), result[n + 1];
    int pos = 0;

    for (int i = 0; i < q; ++i) {
        int type, u, v;
        cin >> type >> u >> v;
        if (type == 1) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        } else {
            cout << kthLargest(u, v, depth, result) << endl;
        }
    }

    return 0;
}
