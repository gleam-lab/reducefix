```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void processQueries(int n, int q) {
    for (int i = 0; i < q; ++i) {
        int type, v, k;
        cin >> type >> v;
        if (type == 1) {
            cin >> k;
            addEdge(v, k);
            pq[v].push(k);
            pq[k].push(v);
        } else if (type == 2) {
            cin >> k;
            while (!pq[v].empty() && pq[v].top() != v) pq[v].pop();
            if (pq[v].size() >= k) cout << pq[v].nth(k - 1) << '\n';
            else cout << -1 << '\n';
        }
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    processQueries(n, q);
    return 0;
}
```

This solution uses adjacency lists and priority queues to manage the connections and efficiently handle the queries. The `addEdge` function adds an edge between two vertices, updating their respective adjacency lists and priority queues. The `processQueries` function processes each query, either adding an edge or finding the k-th largest adjacent vertex. This approach ensures that the operations are performed in an efficient manner, even under the given constraints.
