#include<bits/stdc++.h>
using namespace std;

const int MAX_N = 2e5 + 5;
const int MAX_K = 10;

vector<int> adj[MAX_N];
int parent[MAX_N]; // To keep track of the parent of each vertex for union-find
int size[MAX_N]; // To keep track of the size of each component

void init(int n) {
    for (int i = 1; i <= n; i++) {
        parent[i] = i;
        size[i] = 1;
    }
}

int find(int u) {
    if (parent[u] != u) parent[u] = find(parent[u]);
    return parent[u];
}

void unionSets(int u, int v) {
    int rootU = find(u);
    int rootV = find(v);

    if (rootU != rootV) {
        if (size[rootU] < size[rootV]) swap(rootU, rootV);
        parent[rootV] = rootU;
        size[rootU] += size[rootV];
    }
}

void processQuery1(int u, int v) {
    unionSets(u, v);
}

void processQuery2(int v, int k) {
    int rootV = find(v);
    int connectedVertices = 0;
    vector<int> vertices;

    // Find all connected vertices
    for (int neighbor : adj[rootV]) {
        if (find(neighbor) == rootV) {
            vertices.push_back(neighbor);
            connectedVertices++;
        }
    }

    // Sort the vertices in descending order
    sort(vertices.begin(), vertices.end(), greater<int>());

    // Output the k-th largest vertex if possible
    if (connectedVertices < k) {
        cout << -1 << endl;
    } else {
        cout << vertices[k - 1] << endl;
    }
}

int main() {
    int n, q;
    cin >> n >> q;

    init(n);

    while (q--) {
        int op, u, v;
        cin >> op >> u >> v;

        if (op == 1) {
            // Edge case: if u and v are already connected
            if (find(u) != find(v)) {
                adj[u].push_back(v);
                adj[v].push_back(u);
                processQuery1(u, v);
            }
        } else if (op == 2) {
            processQuery2(v, k);
        }
    }

    return 0;
}
