#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 10;
vector<int> adj[MAXN];
set<int> st[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    st[u].insert(v);
    st[v].insert(u);
}

int kthLargest(int u, int k) {
    if(k > st[u].size()) return -1;
    auto it = st[u].rbegin();
    advance(it, k-1);
    return *it;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int n, q;
    cin >> n >> q;

    while(q--) {
        int type, u, v;
        cin >> type >> u >> v;

        if(type == 1) {
            addEdge(u, v);
        } else {
            cout << kthLargest(u, v) << '\n';
        }
    }

    return 0;
}
