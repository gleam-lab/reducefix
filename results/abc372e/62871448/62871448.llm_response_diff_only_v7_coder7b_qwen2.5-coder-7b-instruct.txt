#include<bits/stdc++.h>
using namespace std;
vector<int> adj[200005];
int as[200005][60];
int fa[200005];

int gtf(int x) {
    if (x == fa[x]) return x;
    return fa[x] = gtf(fa[x]);
}

void merge(int &a, int b) {
    vector<int> tmp;
    while (!adj[a].empty() && !adj[b].empty()) {
        if (adj[a].back() > adj[b].back()) {
            tmp.push_back(adj[a].back());
            adj[a].pop_back();
        } else {
            tmp.push_back(adj[b].back());
            adj[b].pop_back();
        }
    }
    while (!adj[a].empty()) {
        tmp.push_back(adj[a].back());
        adj[a].pop_back();
    }
    while (!adj[b].empty()) {
        tmp.push_back(adj[b].back());
        adj[b].pop_back();
    }
    for (auto it = tmp.rbegin(); it != tmp.rend(); ++it) {
        adj[a].push_back(*it);
    }
}

int main() {
    cin >> q >> n;
    for (int i = 1; i <= n; i++) {
        fa[i] = i;
        memset(as[i], -1, sizeof(as[i]));
        as[i][0] = i;
    }
    int op, ix, iy;
    for (int i = 0; i < q; i++) {
        cin >> op >> ix >> iy;
        if (op == 1) {
            int px = gtf(ix), py = gtf(iy);
            if (px != py) {
                merge(px, py);
                fa[py] = px;
            }
        } else {
            cout << adj[gtf(ix)].size() >= iy ? adj[gtf(ix)][iy - 1] : -1 << "\n";
        }
    }
    return 0;
}
