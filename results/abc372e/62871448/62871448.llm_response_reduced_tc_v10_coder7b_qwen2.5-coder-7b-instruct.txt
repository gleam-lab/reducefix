```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
int deg[MAXN];

struct Node {
    int val;
    int idx;
    bool operator<(const Node& other) const {
        return val > other.val;
    }
};

priority_queue<Node> pq[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    deg[u]++;
    deg[v]++;
}

void removeEdge(int u, int v) {
    auto itu = find(adj[u].begin(), adj[u].end(), v);
    if(itu != adj[u].end()) {
        adj[u].erase(itu);
        deg[u]--;
    }

    auto itv = find(adj[v].begin(), adj[v].end(), u);
    if(itv != adj[v].end()) {
        adj[v].erase(itv);
        deg[v]--;
    }
}

void updatePQ(int node) {
    pq[node].push({node, node});
    for(auto neighbor : adj[node]) {
        pq[node].push({deg[neighbor], neighbor});
    }
    sort(pq[node].begin(), pq[node].end());
    while(pq[node].size() > 10) {
        pq[node].pop();
    }
}

void solve() {
    int n, q;
    cin >> n >> q;

    for(int i = 1; i <= n; ++i) {
        updatePQ(i);
    }

    for(int i = 0; i < q; ++i) {
        int type, u, k;
        cin >> type >> u;
        if(type == 1) {
            cin >> k;
            addEdge(u, k);
            updatePQ(u);
            updatePQ(k);
        } else {
            cin >> k;
            if(deg[u] < k) cout << "-1\n";
            else cout << pq[u].top().idx << "\n";
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    solve();

    return 0;
}
```

This solution uses adjacency lists to store the graph and priority queues to efficiently manage the degrees and indices of connected nodes. It updates these structures after each edge addition or removal to ensure correct results for the degree-based queries.
