#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int TOP_K = 10;

int parent[MAXN];
set<int, greater<int>> component[MAXN];

int find(int x) {
    if (parent[x] != x)
        return parent[x] = find(parent[x]);
    return x;
}

void unite(int x, int y) {
    int fx = find(x), fy = find(y);
    if (fx == fy)
        return;
    
    // Always merge smaller into larger for efficiency
    if (component[fx].size() < component[fy].size())
        swap(fx, fy);

    // Merge the elements from fy into fx
    component[fx].insert(component[fy].begin(), component[fy].end());
    
    // Clear and update parent
    component[fy].clear();
    parent[fy] = fx;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        component[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else if (type == 2) {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if (component[root].size() < (size_t)k) {
                cout << -1 << '\n';
            } else {
                auto it = component[root].begin();
                advance(it, k - 1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
