#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int K_MAX = 10;

int parent[MAXN];
set<int, greater<int>> components[MAXN]; // Stores top K nodes in descending order for each root

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

void merge(int u, int v) {
    int ru = find(u);
    int rv = find(v);
    if (ru == rv) return;

    // Always merge smaller set into larger one
    if (components[ru].size() < components[rv].size()) {
        swap(ru, rv);
    }

    // Merge the top K elements from both sets
    vector<int> merged;
    auto it1 = components[ru].begin();
    auto it2 = components[rv].begin();

    while (merged.size() < K_MAX && (it1 != components[ru].end() || it2 != components[rv].end())) {
        if (it1 == components[ru].end()) {
            merged.push_back(*it2++);
        } else if (it2 == components[rv].end()) {
            merged.push_back(*it1++);
        } else {
            if (*it1 > *it2) {
                merged.push_back(*it1++);
            } else {
                merged.push_back(*it2++);
            }
        }
    }

    components[ru].clear();
    for (int node : merged) {
        components[ru].insert(node);
    }

    // Clear the smaller component after merging
    components[rv].clear();
    parent[rv] = ru;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and components
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        components[i].insert(i);
    }

    while (Q--) {
        int op;
        cin >> op;
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            merge(u, v);
        } else if (op == 2) {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if (k <= (int)components[root].size()) {
                auto it = components[root].begin();
                advance(it, k - 1);
                cout << *it << "\n";
            } else {
                cout << "-1\n";
            }
        }
    }

    return 0;
}
