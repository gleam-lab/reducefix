#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int TOP_K = 10;

int parent[MAXN];
set<int> component_info[MAXN]; // Store top 10 largest nodes in each component

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    
    if (root_u == root_v)
        return;

    // Always merge smaller component into larger one
    if (component_info[root_u].size() < component_info[root_v].size())
        swap(root_u, root_v);

    // Merge the sets
    for (int node : component_info[root_v]) {
        component_info[root_u].insert(node);
        if (component_info[root_u].size() > TOP_K)
            component_info[root_u].erase(prev(component_info[root_u].end()));
    }

    component_info[root_v].clear();
    parent[root_v] = root_u;
}

int get_kth_largest(int v, int k) {
    int root = find(v);
    if (component_info[root].size() < k)
        return -1;
    
    auto it = component_info[root].rbegin();
    advance(it, k - 1);
    return *it;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and component info
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        component_info[i].insert(i);
    }

    while (Q--) {
        int op;
        cin >> op;

        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            cout << get_kth_largest(v, k) << '\n';
        }
    }

    return 0;
}
