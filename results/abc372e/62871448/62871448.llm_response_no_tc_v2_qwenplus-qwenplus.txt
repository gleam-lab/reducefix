```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int TOP_K = 10;

int fa[MAXN];
set<int> components[MAXN]; // Store top 10 largest nodes in each component
bool is_root[MAXN];

int find(int x) {
    if (fa[x] != x)
        fa[x] = find(fa[x]);
    return fa[x];
}

void merge(int u, int v) {
    int ru = find(u);
    int rv = find(v);
    if (ru == rv)
        return;
    
    // Always merge smaller component into larger one
    if (components[ru].size() < components[rv].size()) 
        swap(ru, rv);
    
    // Merge the sets
    for (auto it = components[rv].begin(); it != components[rv].end(); ++it) {
        components[ru].insert(*it);
        if (components[ru].size() > TOP_K)
            components[ru].erase(prev(components[ru].end()));
    }
    
    // Update parent
    fa[rv] = ru;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;
    
    // Initialize DSU
    for (int i = 1; i <= N; ++i) {
        fa[i] = i;
        components[i].insert(i);
        is_root[i] = true;
    }
    
    while (Q--) {
        int op;
        cin >> op;
        
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            merge(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            
            int root = find(v);
            if (k > (int)components[root].size()) {
                cout << -1 << endl;
            } else {
                auto it = components[root].end();
                for (int i = 0; i < k; ++i)
                    --it;
                cout << *it << endl;
            }
        }
    }
    
    return 0;
}
```
