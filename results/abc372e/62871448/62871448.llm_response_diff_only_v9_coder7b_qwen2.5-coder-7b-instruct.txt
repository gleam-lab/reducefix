```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void processQueries(int q, int n) {
    for (int i = 0; i < q; ++i) {
        int type, v, k;
        cin >> type >> v;
        if (type == 1) {
            cin >> k;
            addEdge(v, k);
            pq[v].push(k);
            if (pq[v].size() > 10) pq[v].pop();
        } else {
            if (pq[v].empty()) cout << -1 << '\n';
            else {
                cout << pq[v].top() << '\n';
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, q;
    cin >> n >> q;
    processQueries(q, n);
    return 0;
}
```

This solution uses adjacency lists and priority queues to manage the connections and find the k-th largest element efficiently. It handles both types of queries and ensures that the constraints are respected.
