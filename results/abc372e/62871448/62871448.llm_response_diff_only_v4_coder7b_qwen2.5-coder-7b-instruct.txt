```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
priority_queue<int> pq[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void updatePQ(int v) {
    pq[v].push(-v);
    for (int u : adj[v]) {
        pq[v].push(-u);
    }
    while (pq[v].size() > 10) pq[v].pop();
}

int getKthLargest(int v, int k) {
    if (pq[v].size() < k) return -1;
    priority_queue<int> temp = pq[v];
    while (--k) temp.pop();
    return -temp.top();
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, q; cin >> n >> q;
    
    for (int i = 1; i <= n; ++i) updatePQ(i);
    
    for (int i = 0; i < q; ++i) {
        int type, u, v; cin >> type >> u;
        if (type == 1) {
            cin >> v;
            addEdge(u, v);
            updatePQ(u);
            updatePQ(v);
        } else {
            cout << getKthLargest(u, v) << '\n';
        }
    }
    
    return 0;
}
```
