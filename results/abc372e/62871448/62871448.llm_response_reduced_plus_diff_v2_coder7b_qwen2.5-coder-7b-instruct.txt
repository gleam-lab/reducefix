#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

int kthLargest(int u, int k) {
    vector<int> degs(adj[u].size());
    iota(degs.begin(), degs.end(), 0);
    sort(degs.begin(), degs.end(), [&](int i, int j){
        return adj[u][i] > adj[u][j];
    });
    
    if(k > adj[u].size()) return -1;
    
    return adj[u][degs[k-1]];
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    int n, m;
    cin >> n >> m;
    
    while(m--) {
        int type, u, v;
        cin >> type >> u >> v;
        
        if(type == 1) {
            addEdge(u, v);
        } else {
            cout << kthLargest(u, v) << "\n";
        }
    }
    
    return 0;
}
