#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
const int KMAX = 10;

int N, Q;
int parent[MAXN];
set<int> components[MAXN]; // Store top K elements in descending order

// Find the root of a node with path compression
int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

// Merge smaller component into larger one to maintain efficiency
void unite(int x, int y) {
    int fx = find(x);
    int fy = find(y);
    if (fx == fy)
        return;
    
    // Always merge smaller set into larger set for better performance
    if (components[fx].size() < components[fy].size()) {
        swap(fx, fy);
    }
    
    // Merge the two sets by inserting all elements
    for (auto val : components[fy]) {
        components[fx].insert(val);
        if (components[fx].size() > KMAX) {
            components[fx].erase(prev(components[fx].end()));
        }
    }
    parent[fy] = fx;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> N >> Q;
    
    // Initialize DSU and components
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        components[i].insert(i);
    }
    
    while (Q--) {
        int type;
        cin >> type;
        
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else if (type == 2) {
            int v, k;
            cin >> v >> k;
            
            int fv = find(v);
            if (components[fv].size() < k) {
                cout << -1 << '\n';
            } else {
                auto it = components[fv].rbegin();
                for (int i = 1; i < k; ++i)
                    ++it;
                cout << *it << '\n';
            }
        }
    }
    
    return 0;
}
