```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
int fa[200005];

int gtf(int x) {
    if (x == fa[x]) return x;
    return fa[x] = gtf(fa[x]);
}

void addEdge(int u, int v) {
    u = gtf(u), v = gtf(v);
    if (u != v) {
        fa[u] = v;
        adj[v].insert(adj[v].end(), adj[u].begin(), adj[u].end());
        adj[u].clear();
    }
}

int findKthLargest(int v, int k) {
    v = gtf(v);
    auto it = adj[v].rbegin();
    for (int i = 0; i < k - 1 && it != adj[v].rend(); ++it, ++i);
    return it == adj[v].rend() ? -1 : *it;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) fa[i] = i;
    
    while (q--) {
        int op, u, v;
        cin >> op >> u >> v;
        if (op == 1) addEdge(u, v);
        else cout << findKthLargest(v, k) << '\n';
    }
    return 0;
}
```
