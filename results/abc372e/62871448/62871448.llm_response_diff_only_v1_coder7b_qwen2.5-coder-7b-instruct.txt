#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(-v);
    pq[v].push(-u);
}

void updatePQ(int node, int value) {
    pq[node].pop();
    pq[node].push(value);
}

int getKthLargest(int node, int k) {
    if (pq[node].size() < k) return -1;
    for (int i = 0; i < k - 1; ++i) {
        pq[node].pop();
    }
    return -pq[node].top();
}

int main() {
    int n, q;
    cin >> n >> q;
    while(q--) {
        int type;
        cin >> type;
        if(type == 1) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            cout << getKthLargest(v, k) << endl;
        }
    }
    return 0;
}
