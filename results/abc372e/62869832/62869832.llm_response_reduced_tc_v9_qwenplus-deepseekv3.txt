I've analyzed the issue and realized the original code was incorrectly merging the adjacency lists and not maintaining the top 10 largest elements properly. Here's the corrected version that properly handles the union-find operations and maintains the top 10 largest elements in each connected component:

```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
int fa[MAXN];
int n, q;

int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        fa[i] = i;
        adj[i].push_back(i);
    }

    while (q--) {
        int op, u, v, k;
        cin >> op;
        if (op == 1) {
            cin >> u >> v;
            int fu = find(u), fv = find(v);
            if (fu != fv) {
                if (adj[fu].size() < adj[fv].size()) {
                    swap(fu, fv);
                }
                fa[fv] = fu;
                adj[fu].insert(adj[fu].end(), adj[fv].begin(), adj[fv].end());
                adj[fv].clear();
                sort(adj[fu].rbegin(), adj[fu].rend());
                if (adj[fu].size() > 10) {
                    adj[fu].resize(10);
                }
            }
        } else {
            cin >> v >> k;
            int fv = find(v);
            if (adj[fv].size() < k) {
                cout << "-1\n";
            } else {
                cout << adj[fv][k-1] << "\n";
            }
        }
    }

    return 0;
}
```

Key improvements:
1. Used proper union-find with path compression
2. Maintained adjacency lists for each component
3. When merging, combined the lists and kept only top 10 elements
4. Optimized by always merging smaller list into larger one
5. Used proper indexing for the k-th largest element (0-based vs 1-based)
6. Added faster I/O for large input sizes

The algorithm now correctly:
- Maintains the connected components
- Tracks the largest elements in each component
- Answers queries efficiently
- Handles all edge cases properly
