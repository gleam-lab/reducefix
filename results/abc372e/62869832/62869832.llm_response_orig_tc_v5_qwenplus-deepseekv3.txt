using namespace std;

const int MAXN = 200005;
vector<int> parent(MAXN);
vector<priority_queue<int, vector<int>, greater<int>>> top_k(MAXN);

int find(int u) {
    if (parent[u] != u) {
        parent[u] = find(parent[u]);
    }
    return parent[u];
}

void merge(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    if (root_u == root_v) return;

        }
    }
    parent[root_v] = root_u;

### Explanation
1. **Union-Find Data Structure**: This is used to manage the connected components of the graph. The `find` function uses path compression to optimize future queries.
2. **Priority Queues**: Each component maintains a min-heap (priority queue) of its top 10 largest elements. This ensures efficient access and merging.
3. **Merge Operation**: When merging two components, the smaller heap is merged into the larger one to maintain efficiency. The merged heap is kept to a maximum size of 10.
