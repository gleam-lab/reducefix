#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int KMAX = 10;

int parent[MAXN];
set<int> components[MAXN]; // Each component maintains a sorted set of vertices

// Find the root of the set with path compression
int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

// Union by size to maintain balanced trees
void unite(int x, int y) {
    int fx = find(x), fy = find(y);
    if (fx == fy)
        return;

    // Always merge smaller into larger
    if (components[fx].size() < components[fy].size())
        swap(fx, fy);

    for (auto val : components[fy]) {
        components[fx].insert(val);
    }

    parent[fy] = fx;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and sets
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        components[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;

        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else if (type == 2) {
            int v, k;
            cin >> v >> k;

            int root = find(v);
            auto& comp = components[root];

            if ((int)comp.size() < k) {
                cout << -1 << '\n';
            } else {
                auto it = comp.rbegin(); // reverse iterator for descending order
                for (int i = 1; i < k; ++i)
                    ++it;
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
