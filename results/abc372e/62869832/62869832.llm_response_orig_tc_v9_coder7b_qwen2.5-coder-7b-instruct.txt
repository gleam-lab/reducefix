```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void dfs(int node, vector<bool>& visited, priority_queue<int, vector<int>, greater<int>>& pq) {
    visited[node] = true;
    for(auto neighbor : adj[node]) {
        if(!visited[neighbor]) {
            dfs(neighbor, visited, pq);
            pq.push(-neighbor); // Store negative values to use min-heap as max-heap
        }
    }
}

int kthLargest(vector<int>& pq, int k) {
    while(k--) pq.pop();
    if(pq.empty()) return -1;
    return -pq.top(); // Return positive value
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    vector<bool> visited(n + 1, false);
    priority_queue<int, vector<int>, greater<int>> pq;

    for(int i = 0; i < q; ++i) {
        int type;
        cin >> type;
        if(type == 1) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            visited.assign(n + 1, false);
            pq = {};
            dfs(v, visited, pq);
            cout << kthLargest(pq, k - 1) << "\n";
        }
    }

    return 0;
}
```

This solution uses Depth First Search (DFS) to traverse the graph and store the connected nodes in a min-heap (priority queue). It then retrieves the k-th largest element using this heap, ensuring that it handles all edge cases and maintains a time complexity of O(N log N), which is efficient enough given the problem constraints.
