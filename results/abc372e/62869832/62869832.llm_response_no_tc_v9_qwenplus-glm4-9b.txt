#include<bits/stdc++.h>
using namespace std;

const int MAX_VERTICES = 200005;
const int MAX_K = 10;

int parent[MAX_VERTICES]; // Parent array for Union-Find
vector<int> adj[MAX_VERTICES]; // Adjacency list for the graph
vector<int> rank(MAX_VERTICES, 0); // Rank array for Union-Find

// Function to find the root of the set containing v
int find(int v) {
    if (parent[v] != v) {
        parent[v] = find(parent[v]); // Path compression
    }
    return parent[v];
}

// Function to union two sets containing u and v
void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    if (root_u != root_v) {
        if (rank[root_u] > rank[root_v]) {
            parent[root_v] = root_u;
        } else if (rank[root_u] < rank[root_v]) {
            parent[root_u] = root_v;
        } else {
            parent[root_v] = root_u;
            rank[root_u] += 1;
        }
    }
}

// Function to process query type 1: Add edge between u and v
void add_edge(int u, int v) {
    if (find(u) != find(v)) {
        unite(u, v);
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
}

// Function to process query type 2: Find and print the k-th largest vertex connected to v
void find_kth_largest(int v, int k) {
    if (find(v) == k) { // Find the root of the connected component of v
        sort(adj[v].begin(), adj[v].end()); // Sort the vertices in descending order
        if (adj[v].size() < k) {
            cout << -1 << '\n';
        } else {
            cout << adj[v][k - 1] << '\n';
        }
    } else {
        cout << -1 << '\n';
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    
    // Initialize Union-Find structure
    for (int i = 1; i <= n; i++) {
        parent[i] = i;
        rank[i] = 0;
    }
    
    int op, u, v;
    int q2_count = 0; // Counter for Type 2 queries
    
    for (int i = 0; i < q; i++) {
        cin >> op >> u >> v;
        if (op == 1) {
            add_edge(u, v);
        } else if (op == 2) {
            find_kth_largest(v, k);
            q2_count++;
            if (q2_count % 1000 == 0) { // Flush the output every 1000 queries
                cout.flush();
            }
        }
    }
    
    return 0;
}
