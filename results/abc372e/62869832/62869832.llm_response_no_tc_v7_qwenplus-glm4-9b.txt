#include<bits/stdc++.h>
using namespace std;
int parent[200005];
int rank[200005];
int size[200005];

// Function to find the root of the vertex x
int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);
    }
    return parent[x];
}

// Function to union the sets of x and y
void unionSets(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);
    if (rootX != rootY) {
        if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
            size[rootX] += size[rootY];
        } else if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
            size[rootY] += size[rootX];
        } else {
            parent[rootY] = rootX;
            size[rootX] += size[rootY];
            rank[rootX]++;
        }
    }
}

// Function to add an edge between vertices u and v
void addEdge(int u, int v) {
    unionSets(u, v);
}

// Function to get the k-th largest vertex connected to vertex v
int getKthLargest(int v, int k) {
    int root = find(v);
    return size[root] >= k ? size[root] - k + 1 : -1;
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        parent[i] = i;
        rank[i] = 0;
        size[i] = 1;
    }
    int op, u, v;
    while (q--) {
        cin >> op >> u >> v;
        if (op == 1) {
            addEdge(u, v);
        } else {
            cout << getKthLargest(v, v - k + 1) << endl;
        }
    }
    return 0;
}
