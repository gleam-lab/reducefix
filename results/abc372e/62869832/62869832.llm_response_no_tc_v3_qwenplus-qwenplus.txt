#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int KMAX = 10;

int fa[MAXN];
set<int, greater<int>> groups[MAXN]; // Store top K nodes in descending order

// Find representative with path compression
int find(int x) {
    if (fa[x] != x)
        fa[x] = find(fa[x]);
    return fa[x];
}

// Merge two sets: small-to-large heuristic
void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    
    if (root_u == root_v)
        return;
        
    // Always merge smaller set into larger one
    if (groups[root_u].size() < groups[root_v].size()) {
        swap(root_u, root_v);
        swap(groups[root_u], groups[root_v]);
    }
    
    // Merge the elements
    for (int val : groups[root_v]) {
        groups[root_u].insert(val);
        if (groups[root_u].size() > KMAX) {
            groups[root_u].erase(prev(groups[root_u].end()));
        }
    }
    
    // Union-Find update
    fa[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    // Initialize DSU and groups
    for (int i = 1; i <= N; ++i) {
        fa[i] = i;
        groups[i].insert(i);
    }
    
    while (Q--) {
        int op;
        cin >> op;
        
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            
            if (groups[root].size() < k) {
                cout << -1 << '\n';
            } else {
                auto it = groups[root].begin();
                advance(it, k-1);
                cout << *it << '\n';
            }
        }
    }
    
    return 0;
}
