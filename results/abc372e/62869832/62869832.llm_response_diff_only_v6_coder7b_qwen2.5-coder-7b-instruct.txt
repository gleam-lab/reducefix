```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(v);
    pq[v].push(u);
    if(pq[u].size() > 10) pq[u].pop();
    if(pq[v].size() > 10) pq[v].pop();
}

int getKthLargest(int v, int k) {
    while(!pq[v].empty() && pq[v].top() < k) pq[v].pop();
    return pq[v].size() >= k ? pq[v].top() : -1;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    
    for(int i = 0; i < q; ++i) {
        int type, u, v, k;
        cin >> type;
        
        if(type == 1) {
            cin >> u >> v;
            addEdge(u, v);
        } else {
            cin >> u >> k;
            cout << getKthLargest(u, k) << "\n";
        }
    }

    return 0;
}
```

### Explanation:
1. **Adjacency List Representation**: We use an adjacency list to represent the graph.
2. **Priority Queue for K Largest Elements**: For each vertex, we maintain a min-heap (priority queue) of its neighbors. This allows us to efficiently keep track of the k largest elements.
3. **Adding Edges**: When adding an edge, we add both vertices to each other's adjacency list and push them into each other's priority queues. If the size exceeds k, we remove the smallest element.
4. **Finding K-th Largest Element**: For a given vertex and k, we pop elements from the priority queue until the top element is at least k or the queue is empty. If the queue has at least k elements, we return the top element; otherwise, we return -1.

This approach ensures that each operation is efficient and handles the constraints effectively.
