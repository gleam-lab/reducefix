#include<bits/stdc++.h>

const int MAXN = 200005;
const int MAXK = 10;

struct DSU {
    vector<int> parent;
    vector<array<int, MAXK>> top;

    DSU(int n) : parent(n+1), top(n+1) {
        for(int i=1; i<=n; i++) {
            parent[i] = i;
            top[i][0] = i;
            for(int j=1; j<MAXK; j++) top[i][j] = -1;
        }
    }

    int find(int u) {
        if(parent[u] != u) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }

    void unite(int u, int v) {
        u = find(u);
        v = find(v);
        if(u == v) return;

        if(u > v) swap(u, v);
        parent[v] = u;

        array<int, 2*MAXK> merged;
        int i = 0, j = 0, k = 0;
        while(k < MAXK && (i < MAXK || j < MAXK)) {
            int a = (i < MAXK) ? top[u][i] : -1;
            int b = (j < MAXK) ? top[v][j] : -1;
            if(a > b) {
                if(k == 0 || a != merged[k-1]) {
                    merged[k++] = a;
                }
                i++;
            } else {
                if(b != -1 && (k == 0 || b != merged[k-1])) {
                    merged[k++] = b;
                }
                j++;
            }
        }

        for(int x=0; x<MAXK; x++) {
            top[u][x] = (x < k) ? merged[x] : -1;
        }
    }

    int query(int u, int k) {
        u = find(u);
        if(k-1 < MAXK && k-1 >= 0) {
            return top[u][k-1];
        }
        return -1;
    }
};
