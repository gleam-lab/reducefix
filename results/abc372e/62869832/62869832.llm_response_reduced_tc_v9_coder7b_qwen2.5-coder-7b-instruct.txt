#include<bits/stdc++.h>
using namespace std;

#define MAXN 200005
vector<int> adj[MAXN];

void addEdge(int u, int v){
    adj[u].push_back(v);
    adj[v].push_back(u);
}

int dfs(int node, int parent, vector<int>& visited){
    visited[node] = 1;
    int count = 1;
    for(auto child : adj[node]){
        if(child != parent && !visited[child]){
            count += dfs(child, node, visited);
        }
    }
    return count;
}

int findKthLargest(vector<int>& arr, int k){
    sort(arr.begin(), arr.end());
    reverse(arr.begin(), arr.end());
    if(k > arr.size()) return -1;
    else return arr[k-1];
}

int main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> queries(m);
    for(int i = 0; i < m; i++){
        string type;
        cin >> type;
        if(type == "1"){
            int u, v;
            cin >> u >> v;
            queries[i] = {type, u, v};
        }else{
            int v, k;
            cin >> v >> k;
            queries[i] = {type, v, k};
        }
    }

    vector<int> result(m, -1);
    vector<int> visited(n + 1, 0);
    
    for(int i = 0; i < m; i++){
        if(queries[i][0] == "1"){
            int u = queries[i][1], v = queries[i][2];
            addEdge(u, v);
        }else{
            int v = queries[i][1], k = queries[i][2];
            if(!visited[v]){
                visited[v] = 1;
                vector<int> temp;
                queue<int> q;
                q.push(v);
                
                while(!q.empty()){
                    int node = q.front();
                    q.pop();
                    
                    for(auto child : adj[node]){
                        if(!visited[child]){
                            visited[child] = 1;
                            q.push(child);
                            temp.push_back(child);
                        }
                    }
                }
                result[i] = findKthLargest(temp, k);
            }else{
                result[i] = findKthLargest(adj[v], k);
            }
        }
    }
    
    for(int i = 0; i < m; i++){
        if(queries[i][0] == "2"){
            cout << result[i] << "\n";
        }
    }
    
    return 0;
}
