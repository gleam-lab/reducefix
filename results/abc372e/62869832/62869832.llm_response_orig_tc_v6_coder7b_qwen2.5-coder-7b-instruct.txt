#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

int dfs(int node, int parent, vector<bool>& visited, vector<vector<int>>& rank) {
    visited[node] = true;
    rank[node][0] = 1;
    for(auto child : adj[node]) {
        if(child == parent) continue;
        int r = dfs(child, node, visited, rank);
        rank[node][0] += r;
        for(int j = 0; j < 10; j++) {
            if(j + 1 < 10) rank[node][j + 1] += rank[child][j];
        }
    }
    return rank[node][0];
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n >> q;
    
    for(int i = 0; i < q; i++) {
        int type;
        cin >> type;
        if(type == 1) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            vector<bool> visited(n + 1, false);
            vector<vector<int>> rank(n + 1, vector<int>(10, 0));
            dfs(v, -1, visited, rank);
            if(k > rank[v][0]) cout << "-1\n";
            else cout << rank[v][k - 1] << "\n";
        }
    }

    return 0;
}
