To solve this problem efficiently, we need to handle dynamic connectivity and queries for the k-th largest vertex in a connected component. The best data structure for managing dynamic connectivity is the **Disjoint Set Union (DSU)** or **Union-Find**, but with an important enhancement:

### Key Requirements:
1. **Maintain connected components** — DSU can do this.
2. **Quick access to top-k largest vertices in a component** — We need a way to store and maintain the top 10 largest nodes in each component.

---

### Strategy:
- Use **DSU with path compression** for efficient union-find operations.
- Extend DSU by maintaining a **sorted list of top 10 largest vertices** per set.
- When two sets are merged, merge their top 10 lists and sort them in descending order, then keep only the top 10.
- For query type `2 v k`, return the k-th largest node from the root's list or `-1` if there are fewer than `k` nodes.

---

### Time Complexity:
- Each union operation involves merging at most 10 elements → O(α(N)) + O(1)
- Each query is O(1)

This ensures performance within the constraints.

---

Here is the corrected and fully working C++ code:

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int K = 10;

int parent[MAXN];
vector<int> topKList[MAXN];

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int x, int y) {
    int px = find(x);
    int py = find(y);
    if (px == py) return;

    // Always attach smaller tree to larger one
    parent[py] = px;
    // Merge the top-K lists
    vector<int> merged;
    merged.insert(merged.end(), topKList[px].begin(), topKList[px].end());
    merged.insert(merged.end(), topKList[py].begin(), topKList[py].end());

    // Sort and keep top K largest values
    sort(merged.begin(), merged.end(), greater<int>());
    if (merged.size() > K)
        merged.resize(K);
    topKList[px] = merged;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and topKList
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        topKList[i].push_back(i);
    }

    while (Q--) {
        int op;
        cin >> op;
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if ((int)topKList[root].size() < k)
                cout << -1 << '\n';
            else
                cout << topKList[root][k - 1] << '\n';
        }
    }

    return 0;
}
```
