#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

void updatePQ(int v, int k) {
    pq[v].push(k);
    if (pq[v].size() > 10) {
        pq[v].pop();
    }
}

int getKthLargest(int v, int k) {
    priority_queue<int, vector<int>, greater<int>> temp = pq[v];
    for (int i = 0; i < k - 1 && !temp.empty(); ++i) {
        temp.pop();
    }
    return temp.empty() ? -1 : temp.top();
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, q;
    cin >> n >> q;

    for (int i = 0; i < q; ++i) {
        int type, u, v;
        cin >> type >> u >> v;

        if (type == 1) {
            addEdge(u, v);
            updatePQ(u, v);
            updatePQ(v, u);
        } else if (type == 2) {
            cout << getKthLargest(u, v) << endl;
        }
    }

    return 0;
}

This solution uses adjacency lists to represent the graph and priority queues to keep track of the top 10 largest neighbors for each vertex efficiently. It handles both types of queries as specified and ensures that the operations are performed within the constraints provided.
