```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    pq[u].push(v);
    pq[v].push(u);
}

int getKthLargest(int v, int k) {
    while (pq[v].size() > k) pq[v].pop();
    if (pq[v].size() < k) return -1;
    return pq[v].top();
}

int main() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        pq[i].push(-1); // Initialize with a dummy value
    }
    for (int i = 0; i < q; ++i) {
        int op, ix, iy;
        cin >> op >> ix >> iy;
        if (op == 1) {
            addEdge(ix, iy);
        } else {
            cout << getKthLargest(ix, iy) << '\n';
        }
    }
    return 0;
}
```

### Explanation:
1. **Graph Representation**: Instead of using a union-find data structure, we use an adjacency list (`adj`) to store the connections between nodes and a priority queue (`pq`) to efficiently keep track of the smallest elements for quick access.
   
2. **Adding Edges**: The `addEdge` function updates both the adjacency list and the priority queues for each node involved in the edge.

3. **Finding K-th Largest**: The `getKthLargest` function ensures that the priority queue for a given node contains only the k largest elements by popping smaller elements until it reaches the correct size. If there are fewer than k elements, it returns `-1`.

This approach maintains the constraints and efficiently handles the operations required by the problem statement.
