```cpp
#include<bits/stdc++.h>
using namespace std;
using vi = vector<int>;

int parent[200005];  // parent array for union-find
vi adj[200005];      // adjacency list for more efficient traversal

int findRoot(int x) {
    if (x == parent[x]) return x;
    parent[x] = findRoot(parent[x]);  // path compression
    return parent[x];
}

int main() {
    int n, q;
    cin >> n >> q;
    
    // Initialize union-find structure
    for (int i = 1; i <= n; ++i) {
        parent[i] = i;
    }
    
    vi results;
    for (int i = 0; i < q; ++i) {
        int type, u, v, k;
        cin >> type >> u >> v;
        
        if (type == 1) {
            // Union operation
            int root_u = findRoot(u);
            int root_v = findRoot(v);
            if (root_u != root_v) {
                // Only union if they are in different sets
                parent[root_v] = root_u;
                adj[root_u].push_back(root_v);  // Add to adjacency list
                adj[root_v].push_back(root_u);  // Because the graph is undirected
            }
        } else if (type == 2) {
            // Find k-th largest vertex
            int root_v = findRoot(v);
            sort(adj[root_v].begin(), adj[root_v].end(), greater<int>());  // Sort in descending order
            if (k > adj[root_v].size()) {
                results.push_back(-1);
            } else {
                results.push_back(adj[root_v][k - 1]);
            }
        }
    }
    
    for (int result : results) {
        cout << result << '\n';
    }
    
    return 0;
}
```

This code performs union-find operations to keep track of connected components and uses adjacency lists to efficiently find and sort the neighbors of a vertex. The `findRoot` function with path compression ensures that the union-find structure is efficient. For each type 2 query, it computes the k-th largest connected vertex by sorting the adjacency list of the vertex's root. The results for all type 2 queries are collected in the `results` vector and printed at the end.
