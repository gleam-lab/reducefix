```cpp
#include<bits/stdc++.h>
using namespace std;

vector<int> adj[200005];
int degree[200005];
priority_queue<int, vector<int>, greater<int>> pq[200005];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    degree[u]++;
    degree[v]++;
}

void updatePriorityQueue(int v) {
    pq[v].push(-degree[v]);
    for(auto child : adj[v]) {
        if(degree[child] > degree[v])
            pq[child].push(-degree[v]);
    }
}

void removeEdge(int u, int v) {
    auto it = find(adj[u].begin(), adj[u].end(), v);
    adj[u].erase(it);
    it = find(adj[v].begin(), adj[v].end(), u);
    adj[v].erase(it);

    degree[u]--;
    degree[v]--;
    pq[u].pop();
    pq[v].pop();

    updatePriorityQueue(u);
    updatePriorityQueue(v);
}

int getKthLargest(int v, int k) {
    while(k--) {
        if(pq[v].size() == 0)
            return -1;
        int val = -pq[v].top();
        pq[v].pop();
        pq[v].push(val);
    }
    return -pq[v].top();
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, q;
    cin >> n >> q;
    for(int i = 1; i <= n; i++) {
        pq[i].push(-degree[i]);
    }

    while(q--) {
        int op, ix, iy;
        cin >> op >> ix >> iy;
        if(op == 1) {
            addEdge(ix, iy);
        } else {
            cout << getKthLargest(ix, iy) << '\n';
        }
    }
    return 0;
}
```

This solution uses priority queues to keep track of the degrees of nodes in the graph. When an edge is added, the degrees of the adjacent nodes are updated and their priority queues are adjusted accordingly. When a query is asked, the k-th largest degree node is retrieved from its priority queue. This approach ensures that each operation runs efficiently and handles all edge cases appropriately.
