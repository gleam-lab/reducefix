```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
const int KMAX = 10;

int parent[MAXN];
set<int, greater<int>> groups[MAXN];  // Stores top K elements in descending order

// Find function for DSU (Union-Find)
int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

// Merge two sets: merge smaller set into larger set for efficiency
void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);
    
    if (root_u == root_v)
        return;

    // Always merge smaller group into larger group to optimize performance
    if (groups[root_u].size() < groups[root_v].size()) {
        swap(root_u, root_v);
    }

    // Move elements from root_v to root_u
    for (int x : groups[v]) {
        groups[root_u].insert(x);
        if (groups[root_u].size() > KMAX) {
            auto last = prev(groups[root_u].end());
            groups[root_u].erase(last);
        }
    }
    parent[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU and each group with individual nodes
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        groups[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;

        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root_v = find(v);
            
            if (k > (int)groups[root_v].size()) {
                cout << -1 << '\n';
            } else {
                auto it = groups[root_v].begin();
                advance(it, k - 1);
                cout << *it << '\n';
            }
        }
    }

    return 0;
}
```
