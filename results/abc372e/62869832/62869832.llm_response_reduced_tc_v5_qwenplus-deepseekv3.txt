const int MAXN = 2e5 + 5;
int parent[MAXN];
vector<int> component[MAXN];

int find_set(int v) {
    if (v == parent[v])
        return v;
    return parent[v] = find_set(parent[v]);
}

void union_sets(int a, int b) {
    a = find_set(a);
    b = find_set(b);
    if (a != b) {

        parent[b] = a;

        component[b].clear();
        sort(component[a].rbegin(), component[a].rend());
        if (component[a].size() > 10)
            component[a].resize(10);
    }
}
