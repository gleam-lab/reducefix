#include<bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
int degree[MAXN];

void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
    degree[u]++;
    degree[v]++;
}

void dfs(int node, int parent, vector<int>& path) {
    path.push_back(node);
    for(auto child : adj[node]) {
        if(child != parent) {
            dfs(child, node, path);
        }
    }
}

vector<int> preprocess() {
    vector<int> path;
    dfs(1, -1, path);
    return path;
}

vector<int> buildSparseTable(vector<int>& arr, int n) {
    vector<vector<int>> st(n, vector<int>(20));
    for(int i = 0; i < n; ++i) {
        st[i][0] = arr[i];
    }
    for(int j = 1; (1 << j) <= n; ++j) {
        for(int i = 0; i + (1 << j) - 1 < n; ++i) {
            st[i][j] = max(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
        }
    }
    return st;
}

int getMax(int L, int R, vector<vector<int>>& st) {
    int len = R - L + 1;
    int h = log2(len);
    return max(st[L][h], st[R - (1 << h) + 1][h]);
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int n, m;
    cin >> n >> m;

    vector<int> path = preprocess();

    vector<vector<int>> sparseTable(n, vector<int>(20));

    for(int i = 0; i < n; ++i) {
        sparseTable[path[i]] = {degree[path[i]]};
        for(int j = 1; j < 20; ++j) {
            if(path[i] >= (1 << j)) {
                sparseTable[path[i]][j] = max(sparseTable[path[i]][j - 1], sparseTable[path[i] - (1 << j)][j - 1]);
            } else {
                sparseTable[path[i]][j] = sparseTable[path[i]][j - 1];
            }
        }
    }

    while(m--) {
        int type, v, k;
        cin >> type >> v >> k;
        --v;
        if(type == 1) {
            // Handle edge addition here
        } else {
            int index = upper_bound(path.begin(), path.end(), v) - path.begin();
            if(index + k > n) {
                cout << "-1\n";
            } else {
                cout << sparseTable[index + k - 1][log2(k)] << "\n";
            }
        }
    }

    return 0;
}
