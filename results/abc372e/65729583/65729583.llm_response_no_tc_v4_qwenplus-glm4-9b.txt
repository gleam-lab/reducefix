#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long int ll;

const int N = 2e5 + 9;
vector<int> graph[N];
vector<int> vertices;
vector<int> results;

int findRoot(int v) {
    if (vertices[v] == v) return v;
    vertices[v] = findRoot(vertices[v]);
    return vertices[v];
}

void unionSets(int v1, int v2) {
    int root1 = findRoot(v1), root2 = findRoot(v2);
    if (root1 != root2) {
        if (roots[root1] < roots[root2]) {
            vertices[root2] = root1;
        } else if (roots[root1] > roots[root2]) {
            vertices[root1] = root2;
        } else {
            vertices[root2] = root1;
            roots[root1]++;
        }
    }
}

void kruskal() {
    sort(vertices.begin(), vertices.end(), greater<int>());
    for (int i = 0; i < (int)vertices.size(); i++) {
        int u = vertices[i];
        for (auto &v : graph[u]) {
            if (i < (int)results.size() && v == results[i]) {
                cout << -1 << endl;
                return;
            }
        }
    }
}

int main() {
    int n, q;
    cin >> n >> q;
    vertices.resize(n + 1);
    for (int i = 1; i <= n; ++i) {
        vertices[i] = i;
    }
    
    for (int i = 0; i < q; ++i) {
        int op;
        cin >> op;
        if (op == 1) {
            int u, v;
            cin >> u >> v;
            unionSets(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int rootV = findRoot(v);
            vector<int> connectedVertices;
            for (auto &u : graph[rootV]) {
                connectedVertices.push_back(u);
            }
            if (k > connectedVertices.size()) {
                cout << -1 << endl;
            } else {
                cout << connectedVertices[k - 1] << endl;
            }
        }
    }
    return 0;
}
