#include <iostream>
#include <vector>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;
using namespace __gnu_pbds;

typedef long long int ll;
typedef unsigned long long int ull;

#define pb push_back
#define pf push_front

typedef tree<int, null_type, less<int>, rb_tree_tag, tree_order_statistics_node_update> ordered_set;

const int N = 227500 + 9; // Adjusted to be safe for max node index

int U[N];
ordered_set V[N];

int Find(int u) {
    if (U[u] == u) return u;
    return U[u] = Find(U[u]);
}

void Union(int u, int v) {
    int x = Find(u), y = Find(v);
    if (x == y) return;
    
    // Always merge smaller set into larger one
    if (V[x].size() > V[y].size()) swap(x, y);

    for (auto it : V[x]) {
        V[y].insert(it);
    }
    U[x] = y;
}

void cumbia420PaLosCheems() {
    int n, q;
    cin >> n >> q;

    for (int i = 1; i <= n; ++i) {
        U[i] = i;
        V[i].insert(i); // Store actual vertex values in the set
    }

    while (q--) {
        int cmd;
        cin >> cmd;
        if (cmd == 1) {
            int u, v;
            cin >> u >> v;
            Union(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = Find(v);
            if (V[root].size() < (size_t)k) {
                cout << -1 << endl;
            } else {
                // Use find_by_order with descending order logic
                auto it = V[root].end();
                advance(it, -k);
                cout << *it << endl;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cumbia420PaLosCheems();
    return 0;
}
