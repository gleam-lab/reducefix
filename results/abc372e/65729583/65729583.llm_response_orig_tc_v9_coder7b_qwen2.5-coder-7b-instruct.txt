#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 200005;

int parent[MAXN];
int rank[MAXN];

vector<pii> edges;
vector<vector<int>> adjacencyList(MAXN);
set<int> adjacencySet[MAXN];

void makeSet(int v) {
    parent[v] = v;
    rank[v] = 0;
}

int findSet(int v) {
    if (parent[v] != v)
        parent[v] = findSet(parent[v]);
    return parent[v];
}

void unionSets(int u, int v) {
    u = findSet(u);
    v = findSet(v);
    if (u != v) {
        if (rank[u] < rank[v])
            swap(u, v);
        parent[v] = u;
        if (rank[u] == rank[v])
            rank[u]++;
    }
}

void addEdge(int u, int v) {
    edges.push_back({u, v});
    adjacencyList[u].push_back(v);
    adjacencyList[v].push_back(u);
    adjacencySet[u].insert(v);
    adjacencySet[v].insert(u);
    unionSets(u, v);
}

vector<int> getKthLargest(int u, int k) {
    set<int> &s = adjacencySet[u];
    auto it = s.rbegin();
    advance(it, k - 1);
    return {*(it)};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    for (int i = 1; i <= n; ++i)
        makeSet(i);
    
    for (int i = 0; i < m; ++i) {
        int op, u, v;
        cin >> op >> u >> v;
        if (op == 1) {
            addEdge(u, v);
        } else {
            int k;
            cin >> k;
            auto result = getKthLargest(u, k);
            if (!result.empty())
                cout << result[0] << "\n";
            else
                cout << "-1\n";
        }
    }
    
    return 0;
}
