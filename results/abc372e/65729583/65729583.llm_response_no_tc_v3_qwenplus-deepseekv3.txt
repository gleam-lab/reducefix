The original code uses a Union-Find data structure with ordered sets to keep track of connected components. However, it has several issues:
1. It inserts negative values to maintain descending order which is inefficient
2. It limits set sizes to 30 arbitrarily which can cause incorrect answers
3. The approach is too slow for N,Q up to 2×10^5 as ordered_set operations are O(log n) per operation

Here's the fixed version using Union-Find with path compression and maintaining only the top 10 largest elements (since k ≤ 10) in each component's set:

```cpp
#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

const int N = 2e5 + 5;

int parent[N];
vector<int> comp[N];

void init() {
    for (int i = 0; i < N; i++) {
        parent[i] = i;
        comp[i].push_back(i);
    }
}

int find(int u) {
    if (parent[u] != u) {
        parent[u] = find(parent[u]);
    }
    return parent[u];
}

void unite(int u, int v) {
    u = find(u);
    v = find(v);
    if (u == v) return;
    
    if (comp[u].size() < comp[v].size()) {
        swap(u, v);
    }
    
    parent[v] = u;
    comp[u].insert(comp[u].end(), comp[v].begin(), comp[v].end());
    sort(comp[u].rbegin(), comp[u].rend());
    if (comp[u].size() > 10) {
        comp[u].resize(10);
    }
    comp[v].clear();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    init();
    
    int n, q;
    cin >> n >> q;
    
    while (q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            v = find(v);
            if (comp[v].size() < k) {
                cout << -1 << '\n';
            } else {
                cout << comp[v][k-1] << '\n';
            }
        }
    }
    
    return 0;
}
```

Key improvements:
1. Uses standard Union-Find with path compression
2. Each component maintains only its top 10 largest elements (since k ≤ 10)
3. Merges components efficiently by always merging smaller into larger
4. Uses vectors instead of ordered_set for better performance
5. Sorts and truncates the merged list to keep only top 10 elements
6. More efficient memory usage by clearing smaller components after merging

This solution should handle the upper constraint limits efficiently while providing correct answers to all queries.

The time complexity is O(Q * α(N)) for the Union-Find operations plus O(Q * 10 log 10) for the sorting operations when merging, which is efficient enough for the given constraints.
