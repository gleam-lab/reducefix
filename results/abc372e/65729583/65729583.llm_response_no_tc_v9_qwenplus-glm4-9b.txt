#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 2e5 + 9;

vector<int> parent(N), size(N, 1);
vector<vector<int>> adj(N);

// Find with path compression
int find(int u) {
    if (parent[u] != u) parent[u] = find(parent[u]);
    return parent[u];
}

// Union by size
void unionSets(int u, int v) {
    int rootU = find(u);
    int rootV = find(v);
    if (rootU != rootV) {
        if (size[rootU] < size[rootV]) {
            swap(rootU, rootV);
        }
        parent[rootV] = rootU;
        size[rootU] += size[rootV];
        // Merge the two sets by size
        for (int neighbor : adj[rootV]) {
            adj[rootU].push_back(neighbor);
            parent[neighbor] = rootU;
            size[rootU]++;
        }
        adj[rootV].clear();
    }
}

int main() {

    cin >> n >> q;
    for (int i = 0; i < q; i++) {
        int type;
        cin >> type;
        if (type == 1) {
