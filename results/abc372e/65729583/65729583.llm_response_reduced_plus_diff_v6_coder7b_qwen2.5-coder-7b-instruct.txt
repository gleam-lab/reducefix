#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <cassert>

using namespace std;

typedef long long ll;

class DSU {
public:
    vector<int> parent, rank;
    DSU(int n) : parent(n), rank(n, 1) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int u) {
        if (parent[u] != u) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }

    void unite(int u, int v) {
        u = find(u);
        v = find(v);
        if (rank[u] < rank[v]) {
            swap(u, v);
        }
        parent[v] = u;
        if (rank[u] == rank[v]) {
            rank[u]++;
        }
    }
};

class Graph {
private:
    int n;
    vector<vector<pair<int, int>>> adj;
    DSU dsu;

public:
    Graph(int n) : n(n), adj(n), dsu(n) {}

    void addEdge(int u, int v) {
        adj[u].emplace_back(v, adj[u].size());
        adj[v].emplace_back(u, adj[v].size());
        dsu.unite(u, v);
    }

    int getKthLargest(int u, int k) {
        u = dsu.find(u);
        set<int> neighbors;
        for (auto& [neighbor, index] : adj[u]) {
            neighbors.insert(index);
        }
        auto it = next(neighbors.rbegin(), k - 1);
        if (it == neighbors.rend()) {
            return -1;
        }
        return it->second + 1;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    Graph g(n);

    while (q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            g.addEdge(u - 1, v - 1);
        } else {
            int u, k;
            cin >> u >> k;
            cout << g.getKthLargest(u - 1, k) << "\n";
        }
    }

    return 0;
}
