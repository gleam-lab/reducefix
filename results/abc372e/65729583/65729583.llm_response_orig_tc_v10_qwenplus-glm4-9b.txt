#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

const int N = 2e5 + 9;
vector<int> parent(N);
vector<int> size(N);
vector<vector<int>> adj(N);

void find(int u) {
    if (parent[u] != u) {
        parent[u] = find(parent[u]);
    }
    return parent[u];
}

void unite(int u, int v) {
    u = find(u);
    v = find(v);
    if (u != v) {
        if (size[u] < size[v]) {
            swap(u, v);
        }
        parent[v] = u;
        size[u] += size[v];
        // Add all edges from v's component to u's component
        for (int to : adj[v]) {
            adj[u].push_back(to);
        }
        adj[v].clear();
    }
}
