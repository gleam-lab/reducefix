#include <iostream>
#include <vector>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;
using namespace __gnu_pbds;

typedef long long ll;
const int MAXN = 2e5 + 5;

#define debug(x) //cerr << #x << " = " << x << endl;

// We'll store negative numbers to allow k-th largest queries in a set that sorts in ascending order
typedef tree<int, null_type, less<int>, rb_tree_tag,
             tree_order_statistics_node_update>
    ordered_set;

int parent[MAXN];
ordered_set components[MAXN];

int find(int u) {
    if (parent[u] != u)
        parent[u] = find(parent[u]);
    return parent[u];
}

void unite(int u, int v) {
    int root_u = find(u);
    int root_v = find(v);

    if (root_u == root_v)
        return;

    // Always merge smaller set into larger one for better performance
    if (components[root_u].size() < components[root_v].size()) {
        swap(root_u, root_v);
    }

    // Merge components
    for (auto val : components[root_v]) {
        components[root_u].insert(val);
    }

    // Update parent
    parent[root_v] = root_u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, Q;
    cin >> N >> Q;

    // Initialize DSU
    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        components[i].insert(-i);  // Store as negative for k-th largest functionality
    }

    while (Q--) {
        int type;
        cin >> type;

        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            
            if (components[root].size() < k) {
                cout << -1 << "\n";
            } else {
                int val = -(*components[root].find_by_order(k - 1));
                cout << val << "\n";
            }
        }
    }

    return 0;
}
