#include <iostream>
#include <vector>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;
using namespace __gnu_pbds;

typedef long long ll;

#define ordered_set tree<int, null_type, less<int>, rb_tree_tag, tree_order_statistics_node_update>

int N, Q;
int parent[200005];
ordered_set components[200005];

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

void unite(int u, int v) {
    int pu = find(u);
    int pv = find(v);
    if (pu == pv)
        return;

    // Merge smaller set into larger one for better performance
    if (components[pu].size() < components[pv].size()) {
        swap(pu, pv);
    }

    for (auto val : components[pv]) {
        components[pu].insert(val);
    }
    parent[pv] = pu;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> N >> Q;

    for (int i = 1; i <= N; ++i) {
        parent[i] = i;
        components[i].insert(i);
    }

    while (Q--) {
        int type;
        cin >> type;
        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unite(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int root = find(v);
            if ((int)components[root].size() < k) {
                cout << -1 << '\n';
            } else {
                // Since we store in increasing order, k-th largest is size - k
                int sz = components[root].size();
                cout << *components[root].find_by_order(sz - k) << '\n';
            }
        }
    }

    return 0;
}
