const int N = 2e5 + 9;

int parent[N];

int find(int u) {
    if (parent[u] != u) {
        parent[u] = find(parent[u]);
    }
    return parent[u];
}

void unionSets(int u, int v) {
    u = find(u), v = find(v);
    if (u != v) {
        parent[v] = u;
    }
}

            vector<int> connectedVertices;
            for (int neighbor : adjList[v]) {
