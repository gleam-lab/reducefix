#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

typedef long long int ll;

const int N = 2e5 + 9;

vector<int> adjList[N];
int parent[N];

int findSet(int v) {
    if (parent[v] != v)
        parent[v] = findSet(parent[v]);
    return parent[v];
}

void unionSets(int u, int v) {
    int rootU = findSet(u);
    int rootV = findSet(v);
    if (rootU != rootV) {
        parent[rootV] = rootU;
    }
}

int main() {
    int n, q;
    cin >> n >> q;

    for (int i = 1; i <= n; i++) {
        parent[i] = i;
    }

    while (q--) {
        int type;
        cin >> type;

        if (type == 1) {
            int u, v;
            cin >> u >> v;
            unionSets(u, v);
        } else {
            int v, k;
            cin >> v >> k;
            int rootV = findSet(v);
            set<int> &connected = adjList[rootV];

            if (connected.size() < k) {
                cout << -1 << endl;
            } else {
                // Find the k-th largest value in the set of connected vertices
                auto it = connected.lower_bound(-k);
                cout << -(*it) << endl;
            }
        }
    }

    return 0;
}
