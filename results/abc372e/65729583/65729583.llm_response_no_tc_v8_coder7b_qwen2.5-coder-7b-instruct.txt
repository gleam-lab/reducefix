#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

typedef long long int ll;

const int N = 2e5 + 9;

int U[N];
set<int> V[N];

int Find(int u) {
    if (U[u] == u) return u;
    return U[u] = Find(U[u]);
}

void Union(int u, int v) {
    u = Find(u), v = Find(v);
    if (u != v) {
        U[v] = u;
        if (V[v].size() > V[u].size()) swap(V[u], V[v]);
        for (auto num : V[v]) {
            V[u].insert(num);
            if (V[u].size() >= 10) break;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, q;
    cin >> n >> q;

    for (int i = 1; i <= n; ++i) {
        U[i] = i;
        V[i].insert(i);
    }

    while (q--) {
        int cmd, u, v, k;
        cin >> cmd;
        if (cmd == 1) {
            cin >> u >> v;
            Union(u, v);
        } else {
            cin >> u >> k;
            u = Find(u);
            auto it = V[u].rbegin();
            if (k <= V[u].size()) {
                cout << *it << endl;
            } else {
                cout << -1 << endl;
            }
        }
    }

    return 0;
}
