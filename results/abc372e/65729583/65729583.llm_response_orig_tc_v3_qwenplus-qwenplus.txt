#include <iostream>

const int MAXN = 2 * 1e5 + 5;

int U[MAXN];
multiset<int> V[MAXN];

int Find(int u) {
    if (U[u] == u) return u;
    return U[u] = Find(U[u]);
}

void Union(int u, int v) {
    int fu = Find(u);
    int fv = Find(v);
    if (fu == fv) return;
    // Merge smaller set into larger one for efficiency
    if (V[fu].size() < V[fv].size()) swap(fu, fv);
    for (auto x : V[fv]) {
        V[fu].insert(x);
    }
    U[fv] = fu;
    V[fv].clear();
}

}
