```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to find the minimum possible value of (max(B) - min(B)) where B is any subsequence of size N-K obtained by deleting K elements.
    // The key observation is that since we want to minimize max(B)-min(B), it's best to consider B to be a window of N-K elements in the sorted array.

    sort(A.begin(), A.end());

    int minDiff = numeric_limits<int>::max();
    for (int i = 0; i + (N - K) - 1 < N; ++i) {
        int j = i + (N - K);
        if (j > N) break;
        minDiff = min(minDiff, A[j - 1] - A[i]);
    }

    cout << minDiff << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
