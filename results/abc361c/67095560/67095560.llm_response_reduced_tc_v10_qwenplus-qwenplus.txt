#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

typedef long long ll;

void solve() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will maintain a sliding window of size N-K
    // and find the minimum value of max(B) - min(B)
    int res = numeric_limits<ll>::max();
    deque<int> minQ, maxQ;

    for (int i = 0; i < N; ++i) {
        // Maintain min queue
        while (!minQ.empty() && A[minQ.back()] >= A[i])
            minQ.pop_back();
        minQ.push_back(i);

        // Maintain max queue
        while (!maxQ.empty() && A[maxQ.back()] <= A[i])
            maxQ.pop_back();
        maxQ.push_back(i);

        // Remove elements out of window [i-K+1, i]
        if (minQ.front() <= i - (N - K)) {
            minQ.pop_front();
        }
        if (maxQ.front() <= i - (N - K)) {
            maxQ.pop_front();
        }

        // Once we have a valid window of size N-K
        if (i >= N - K - 1) {
            res = min(res, A[maxQ.front()] - A[minQ.front()]);
        }
    }

    cout << res << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
