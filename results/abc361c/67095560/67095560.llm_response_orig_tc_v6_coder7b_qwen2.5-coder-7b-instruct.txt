#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>
#define pb push_back
#define sz size()
#define MOD 1000000007

void solve() {
    ll n, k; cin >> n >> k;
    vi arr(n);
    for(ll i = 0; i < n; i++) cin >> arr[i];

    sort(arr.begin(), arr.end());
    ll ans = LLONG_MAX;
    for(ll i = 0; i <= n-k; i++) {
        ans = min(ans, arr[i+k-1] - arr[i]);
    }

    cout << ans << '\n';
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);
    ll T = 1;
    while(T--) solve();
    return 0;
}
