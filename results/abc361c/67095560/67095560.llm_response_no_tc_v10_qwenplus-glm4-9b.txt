#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    long long n, k;
    cin >> n >> k;
    vector<long long> A(n);
    for (long long &x : A) cin >> x;

    // Sort the array to easily find the minimum and maximum in the remaining sequence
    sort(A.begin(), A.end());

    // If k is even, we need to remove pairs to minimize the max-min difference
    // If k is odd, we need to remove one element from both ends and one single element from the middle
    if (k % 2 == 0) {
        // Remove pairs from both ends
        long long max_diff = A[0] - A[n - 1];
        for (int i = 0; i < k / 2; ++i) {
            max_diff = min(max_diff, A[i] - A[n - k + 2 * i - 1]);
        }
        cout << max_diff << endl;
    } else {
        // Remove one element from each end and one single element from the middle
        long long max_diff = A[0] - A[n - 1];
        for (int i = 0; i < k / 2; ++i) {
            max_diff = min(max_diff, A[i] - A[n - k + 2 * i - 1]);
        }
        max_diff = min(max_diff, A[n - k / 2] - A[n - k / 2 - 1]);
        cout << max_diff << endl;
    }

    return 0;
}
