```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    sort(a.begin(), a.end());

    int ans = INT_MAX;
    for (int i = 0; i <= k; ++i) {
        int min_val = a[i];
        int max_val = a[n - k + i - 1];
        ans = min(ans, max_val - min_val);
    }

    cout << ans << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
