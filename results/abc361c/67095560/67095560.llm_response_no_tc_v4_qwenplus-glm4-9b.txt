#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    // Sort the array to make it easier to find the minimum and maximum
    sort(a.begin(), a.end());

    // If k is even, we do not want to remove the smallest and largest k/2 elements.
    // If k is odd, we do not want to remove the smallest (k//2) elements.
    int removeFromEnd = (k % 2 == 1) ? k / 2 : (k / 2) + 1;
    int removeFromStart = k / 2;

    // Calculate the new minimum of B
    int newMin = a[removeFromStart];

    // Calculate the new maximum of B
    int newMax = a[n - removeFromEnd - 1];

    // The result is the difference between the new maximum and new minimum
    cout << newMax - newMin << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
