#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We need to find the minimum possible value of max(B) - min(B)
    // where B is formed by removing exactly K elements from A,
    // and keeping the remaining elements in order.

    // Since the order must be preserved, we are choosing a subsequence
    // of length N-K from A. The idea is to find all windows of size (N-K)
    // in the array and compute min(max - min) over those windows.

    deque<int> minDeque, maxDeque;
    int res = numeric_limits<int>::max();

    for (int i = 0; i < n; ++i) {
        // Maintain deque for min and max in window of size (N-K)

        while (!minDeque.empty() && A[i] <= A[minDeque.back()])
            minDeque.pop_back();
        while (!maxDeque.empty() && A[i] >= A[maxDeque.back()])
            maxDeque.pop_back();

        minDeque.push_back(i);
        maxDeque.push_back(i);

        // Remove elements out of current window
        if (i >= (n - k)) {
            while (!minDeque.empty() && minDeque.front() <= i - (n - k))
                minDeque.pop_front();
            while (!maxDeque.empty() && maxDeque.front() <= i - (n - k))
                maxDeque.pop_front();

            res = min(res, A[maxDeque.front()] - A[minDeque.front()]);
        }
    }

    cout << res << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
