### Correct Approach
The main idea is to minimize the range (max - min) of the new sequence B. To do this, we should focus on removing elements that are far apart in value to minimize the maximum minus minimum range.

1. Sort the sequence A.
2. We can consider removing elements from both ends of the sorted sequence to reduce the range. After sorting, the smallest elements will be at the beginning and the largest elements will be at the end.
3. By removing K elements, we can start by removing the largest or smallest K elements (or some combination of both), and then choose the minimum possible range from the remaining elements.
4. We can use binary search to find the optimal split point in the sorted array to remove exactly K elements, and then calculate the range from the remaining elements.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<long long> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }
    
    // Sort the array
    sort(A.begin(), A.end());
    
    // Binary search to find the best split point
    long long low = 0, high = n - 1;
    while (low < high) {
        int mid = (low + high) / 2;
        // Check if removing K elements from the start or end produces a smaller range
        long long minRange1 = (mid <= k) ? (A[mid] - A[0]) : (A[n - 1] - A[k - 1]);
        long long minRange2 = (n - 1 - mid <= k) ? (A[n - 1] - A[n - 1 - k]) : (A[mid] - A[0]);
        if (minRange1 <= minRange2) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    
    // Output the minimum range found
    cout << (A[low] - A[low - k]) << endl;
    
    return 0;
}
