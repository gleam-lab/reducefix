```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // Sort the array to group smallest and largest values
    sort(A.begin(), A.end());

    // We want to choose a window of size (n - k) with minimum possible (max - min)
    // Try all windows of size (n - k)
    int min_diff = numeric_limits<int>::max();
    for (int i = 0; i + (n - k) <= n; ++i) {
        int left = i;
        int right = i + (n - k) - 1;
        min_diff = min(min_diff, A[right] - A[left]);
    }

    cout << min_diff << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
