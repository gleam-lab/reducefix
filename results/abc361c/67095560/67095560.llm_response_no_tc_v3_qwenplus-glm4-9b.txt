#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define pb push_back

typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    cin >> n >> k;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // Sort the array to easily find the smallest and largest elements
    sort(a.begin(), a.end());

    // If k is odd, we remove the smallest and largest elements alternately
    // If k is even, we remove the smallest and largest elements consecutively
    // This ensures that the remaining elements have the smallest possible range
    if (k % 2 == 1) {
        // Remove smallest and largest alternately
        int i = 0, j = n - 1;
        while (i < j && k > 0) {
            a.erase(a.begin() + i);
            a.pop_back(); // Remove the largest element
            i++;
            j--;
            k -= 2;
        }
        // If we still have an odd number of elements to remove
        if (k % 2 == 1) {
            a.erase(a.begin() + i);
        }
    } else {
        // Remove smallest and largest consecutively
        int i = 0, j = n - 1;
        while (i < j && k > 0) {
            a.erase(a.begin() + i);
            a.pop_back(); // Remove the largest element
            k -= 2;
        }
    }

    // The range of the remaining elements
    int min_val = *min_element(a.begin(), a.end());
    int max_val = *max_element(a.begin(), a.end());
    cout << max_val - min_val << endl;

    return 0;
}
