```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"

void solve() {
    int n, k;
    cin >> n >> k;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    sort(a.begin(), a.end());

    long long ans = LLONG_MAX;
    for (int i = 0; i <= k; ++i) {
        ans = min(ans, a[k + n - i - 1] - a[i]);
    }

    cout << ans << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```

1. **Sorting**: The array `a` is sorted to facilitate finding the minimum difference between the maximum and minimum values after removing `K` elements.
2. **Sliding Window**: We use a sliding window approach to find the minimum possible value of the difference between the maximum and minimum values of the remaining elements. The window size is `(n - k)`.
3. **Update Minimum Difference**: For each valid window, we calculate the difference between the maximum and minimum values and update the answer with the minimum difference found.

This approach ensures that we efficiently find the minimum possible value of the required expression within the constraints.
