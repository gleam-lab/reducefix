```cpp
#include<bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll n, k;
    cin >> n >> k;
    vector<ll> a(n);
    for(ll i = 0; i < n; i++) {
        cin >> a[i];
    }

    sort(a.begin(), a.end());

    ll ans = LLONG_MAX;
    for(ll i = 0; i <= n-k; i++) {
        ans = min(ans, a[i+k-1] - a[i]);
    }

    cout << ans << '\n';
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```

This solution involves sorting the array and then iterating through it to find the minimum difference between the maximum and minimum values of any subarray of size `n-k`. This approach ensures that we are considering all possible ways to remove `k` elements to minimize the resulting difference. The time complexity is dominated by the sorting step, making it O(N log N), which is efficient enough for the given constraints.
