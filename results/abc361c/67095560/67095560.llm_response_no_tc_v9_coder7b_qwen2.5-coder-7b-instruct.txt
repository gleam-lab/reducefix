```cpp
#include<bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n, k;
    cin >> n >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    sort(a.begin(), a.end());

    ll ans = LLONG_MAX;
    for(int i = 0; i <= n - k; ++i) {
        ans = min(ans, a[i + k - 1] - a[i]);
    }

    cout << ans << '\n';
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```

This solution sorts the array and then iterates over possible subarrays of size `n-k` to find the minimum difference between the maximum and minimum values in any such subarray. This approach ensures that we efficiently find the minimum possible value of the desired expression while adhering to the constraints provided.
