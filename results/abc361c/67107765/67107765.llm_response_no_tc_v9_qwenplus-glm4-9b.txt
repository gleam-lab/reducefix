    vector<int> num(n);

    }

    // The middle element of the remaining sequence will be the maximum
    // of the minimum possible B (when removing K elements).
    int middleIndex = n - k / 2;
    // The farthest elements removed will be the minimum and the second minimum
    // of the remaining sequence, which will help in finding the maximum difference.
    int maxDiff = num[middleIndex + k / 2];
    int minDiff = num[middleIndex];
    // Output the difference between the maximum and minimum of the remaining sequence.
    cout << maxDiff - minDiff << endl;
