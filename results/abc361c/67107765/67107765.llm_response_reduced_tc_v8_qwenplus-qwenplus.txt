#include <bits/stdc++.h>
using namespace std;

#define fi first
#define se second
#define pb push_back
const int INF = 1001001001;
const ll LINF = 1001002003004005006ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will keep N - K elements in B
    int m = n - k;

    // We need to consider all intervals of size m in the original array
    // and find the one with minimum (max - min)

    // To do this efficiently:
    // Use two sliding windows:
    // 1. A window of size m to find candidate subarrays
    // 2. In each window, find min and max

    // Optimized using two deques to maintain indices of potential mins/maxes

    deque<int> min_dq, max_dq;

    ll ans = LINF;

    for (int i = 0; i < n; ++i) {
        // Remove elements out of window [i-m+1, i]
        if (i >= m) {
            while (!min_dq.empty() && min_dq.front() <= i - m) min_dq.pop_front();
            while (!max_dq.empty() && max_dq.front() <= i - m) max_dq.pop_front();
        }

        // Add current element to deques
        while (!min_dq.empty() && A[min_dq.back()] >= A[i]) min_dq.pop_back();
        while (!max_dq.empty() && A[max_dq.back()] <= A[i]) max_dq.pop_back();

        min_dq.push_back(i);
        max_dq.push_back(i);

        if (i >= m - 1) {
            ll curr_diff = (ll)A[max_dq.front()] - (ll)A[min_dq.front()];
            ans = min(ans, curr_diff);
        }
    }

    cout << ans << endl;
}
