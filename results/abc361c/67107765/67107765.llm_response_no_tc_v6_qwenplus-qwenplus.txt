```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<int, int>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will use a sliding window to choose N-K elements from A
    // such that max(B) - min(B) is minimized

    int window_size = n - k;
    deque<int> min_deq;
    deque<int> max_deq;
    ll ans = LLONG_MAX;

    for (int i = 0; i < n; ++i) {
        // Maintain min_deque (increasing)
        while (!min_deq.empty() && A[min_deq.back()] >= A[i]) min_deq.pop_back();
        min_deq.push_back(i);

        // Maintain max_deque (decreasing)
        while (!max_deq.empty() && A[max_deq.back()] <= A[i]) max_deq.pop_back();
        max_deq.push_back(i);

        // Remove elements out of window
        if (i >= window_size - 1) {
            while (min_deq.front() <= i - window_size) min_deq.pop_front();
            while (max_deq.front() <= i - window_size) max_deq.pop_front();

            ans = min(ans, (ll)A[max_deq.front()] - (ll)A[min_deq.front()]);
        }
    }

    cout << ans << endl;
}
```
