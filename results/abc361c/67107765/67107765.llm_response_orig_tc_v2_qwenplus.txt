```cpp
#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
#define chmin(x,y) x = min(x,y)
using ll = long long;
const int INF = 1e9 + 10;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    int ans = INF;
    
    // Try all possible contiguous segments of length 'keep'
    // The optimal remaining sequence must be a contiguous subarray in the sorted array
    // because if we skip an element in the middle, we could potentially get a smaller range
    for (int i = 0; i + keep - 1 < n; i++) {
        chmin(ans, a[i + keep - 1] - a[i]);
    }
    
    cout << ans << endl;
}
```
