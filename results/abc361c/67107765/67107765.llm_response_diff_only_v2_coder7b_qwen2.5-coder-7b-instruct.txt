```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> num(n);
    for (int i = 0; i < n; i++) {
        cin >> num[i];
    }
    sort(num.begin(), num.end());
    int ans = num[n-k-1] - num[k]; // The optimal strategy is always to remove the smallest k elements and the largest k elements
    cout << ans << endl;
    return 0;
}
```
