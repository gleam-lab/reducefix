#include <bits/stdc++.h>
using namespace std;

#define fi first
#define se second
#define pb push_back
#define chmax(x, y) x = max(x, y)
#define chmin(x, y) x = min(x, y)

using ll = long long;
using P = pair<int, int>;

const int INF = 1001001001;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We need to choose N-K elements to keep such that:
    // max(B) - min(B) is minimized
    int m = n - k;

    // Strategy:
    // To minimize max-min, we should pick a contiguous range of values as tightly packed as possible.
    // Since the array is not guaranteed to be sorted, but removing K elements gives us flexibility,
    // we sort A and then find the minimal difference over all windows of size m.

    sort(A.begin(), A.end());

    ll ans = 1e18;
    for (int i = 0; i + m - 1 < n; ++i) {
        chmin(ans, 1LL * A[i + m - 1] - A[i]);
    }

    cout << ans << "\n";
}
