#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
using ll = long long;
using P = pair<int, int>;
const ll LINF = 1001002003004005006ll;
const int INF = 1001001001;
#define chmin(x,y) x = min(x,y)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We are going to remove exactly K elements.
    // The remaining sequence has length n - k.
    // We want to minimize (max of remaining - min of remaining).
    // Since the array is sorted, any valid remaining subsequence must be contiguous?
    // But note: we can only remove arbitrary elements, but the order remains.
    // However, since we care only about min and max, and the rest can be ignored,
    // the optimal solution will keep a contiguous segment in the sorted array.
    // Why? Because if you skip an element inside, you could include it without increasing range.
    // So we can try all possible contiguous segments of length (n-k).

    int len = n - k;
    int ans = a[len - 1] - a[0]; // first window
    for (int i = 1; i + len - 1 < n; i++) {
        chmin(ans, a[i + len - 1] - a[i]);
    }
    cout << ans << endl;
}
