#include <bits/stdc++.h>
using namespace std;

#define fi first
#define se second
#define pb push_back
#define chmin(x, y) x = min(x, y)
using ll = long long;
using P = pair<int, int>;

const int INF = 1001001001;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We'll consider the minimum possible value of max(B) - min(B)
    // Since we can remove exactly K elements, we want to find a window of size N-K
    // in the original array such that the difference between the max and min in that window is minimized.

    int window_size = N - K;
    multiset<int> ms;

    // Sliding window approach using multiset
    for (int i = 0; i < window_size; ++i) {
        ms.insert(A[i]);
    }

    int ans = *ms.rbegin() - *ms.begin();
    
    for (int i = window_size; i < N; ++i) {
        ms.erase(ms.find(A[i - window_size])); // Remove element exiting the window
        ms.insert(A[i]);                      // Add new element entering the window
        int current_diff = *ms.rbegin() - *ms.begin();
        chmin(ans, current_diff);
    }

    cout << ans << endl;
}
