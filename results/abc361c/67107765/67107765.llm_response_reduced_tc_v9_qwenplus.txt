#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
#define chmin(x, y) x = min(x, y)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    // The optimal way is to remove some prefix and/or suffix.
    // Try all possible ways: remove i from left and (k - i) from right.
    int ans = a[n - 1] - a[0]; // worst case: no removal improvement
    for (int i = 0; i <= k; i++) {
        int left_removed = i;
        int right_removed = k - i;
        if (left_removed + right_removed > n) continue;
        if (left_removed >= n || right_removed >= n) continue;
        int st = left_removed;
        int en = n - 1 - right_removed;
        if (st <= en) {
            chmin(ans, a[en] - a[st]);
        }
    }
    
    cout << ans << endl;
}
