#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
#define chmin(x, y) x = min(x, y)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    // The optimal strategy is to remove some prefix and suffix such that total removed = k.
    // Try all possible ways: remove i from left, k-i from right, for i in [0, k]
    int ans = INT_MAX;
    int keep = n - k;
    
    for (int i = 0; i <= k; i++) {
        int left_remove = i;
        int right_remove = k - i;
        
        int left_idx = left_remove;
        int right_idx = n - 1 - right_remove;
        
        if (left_idx <= right_idx) {
            chmin(ans, a[right_idx] - a[left_idx]);
        }
    }
    
    cout << ans << endl;
}
