```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> num(n);
    for (int i = 0; i < n; i++) {
        cin >> num[i];
    }
    sort(num.begin(), num.end());
    int del = k / 2;
    int md = k % 2;
    int st = del;
    int gl = (n - 1) - del;
    if(md == 0){
        cout << num[gl] - num[st] << endl;
    } else {
        int ans = min((num[gl - 1] - num[st]), (num[gl] - num[st + 1]));
        cout << ans << endl;
    }
    return 0;
}
```
