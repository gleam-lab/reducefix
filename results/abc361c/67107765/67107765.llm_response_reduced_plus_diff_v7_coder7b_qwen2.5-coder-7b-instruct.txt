```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<long long> num(n);
    for(int i=0; i<n; ++i) cin >> num[i];

    sort(num.begin(), num.end());

    // Calculate the minimum possible difference after removing K elements
    long long ans = LLONG_MAX;
    for(int i=0; i<=n-k; ++i) {
        ans = min(ans, num[i+k-1] - num[i]);
    }

    cout << ans << '\n';
    return 0;
}
```
