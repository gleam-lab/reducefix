```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> nums(n);
    for(int i = 0; i < n; ++i) cin >> nums[i];
    
    sort(nums.begin(), nums.end());
    
    // Calculate the difference between the largest and smallest numbers after removing k elements
    int result = nums[n - 1] - nums[k];
    
    cout << result << '\n';
    
    return 0;
}
```
