```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;

    vector<long long> num(n);
    for(int i = 0; i < n; ++i) {
        cin >> num[i];
    }

    sort(num.begin(), num.end());

    // Calculate the minimum possible difference
    long long result = LLONG_MAX;
    for(int i = 0; i <= n - k; ++i) {
        result = min(result, num[i + k - 1] - num[i]);
    }

    cout << result << '\n';
    return 0;
}
```

This solution sorts the array first and then calculates the minimum possible difference between the maximum and minimum values of any subarray of size `n-k+1`. This ensures that we find the optimal way to remove `k` elements to minimize the difference. The time complexity is O(N log N) due to the sorting step, which is efficient enough given the problem constraints.
