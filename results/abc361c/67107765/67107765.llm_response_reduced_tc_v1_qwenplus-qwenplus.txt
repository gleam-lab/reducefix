#include <bits/stdc++.h>
using namespace std;

#define fi first
#define se second
#define pb push_back
using ll = long long;
using P = pair<int, int>;

const int INF = 1001001001;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will try all possible windows of size (n - k)
    int window_size = n - k;
    deque<int> min_deque, max_deque;
    int min_result = INT_MAX;

    for (int i = 0; i < n; ++i) {
        // Maintain deque for min
        while (!min_deque.empty() && min_deque.front() <= i - window_size)
            min_deque.pop_front();
        while (!min_deque.empty() && A[min_deque.back()] >= A[i])
            min_deque.pop_back();
        min_deque.push_back(i);

        // Maintain deque for max
        while (!max_deque.empty() && max_deque.front() <= i - window_size)
            max_deque.pop_front();
        while (!max_deque.empty() && A[max_deque.back()] <= A[i])
            max_deque.pop_back();
        max_deque.push_back(i);

        if (i >= window_size - 1) {
            int current_max = A[max_deque.front()];
            int current_min = A[min_deque.front()];
            min_result = min(min_result, current_max - current_min);
        }
    }

    cout << min_result << endl;
}
