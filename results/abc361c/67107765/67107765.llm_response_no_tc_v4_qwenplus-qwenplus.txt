```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<int, int>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We need to choose N - K elements in order and minimize (max - min)
    // Since N can be up to 2e5, we need an O(N log N) or O(N) approach

    // Sliding window: consider all subarrays of length N-K
    // For each window, compute max - min
    // Use two deques to maintain min and max efficiently

    deque<int> min_deq, max_deq;
    ll res = LLONG_MAX;

    auto add = [&](int x, int i) {
        while (!min_deq.empty() && A[min_deq.back()] >= x)
            min_deq.pop_back();
        min_deq.push_back(i);
        
        while (!max_deq.empty() && A[max_deq.back()] <= x)
            max_deq.pop_back();
        max_deq.push_back(i);
    };

    auto remove = [&](int x, int i) {
        if (!min_deq.empty() && min_deq.front() == i)
            min_deq.pop_front();
        
        if (!max_deq.empty() && max_deq.front() == i)
            max_deq.pop_front();
    };

    for (int i = 0; i < n; ++i) {
        add(A[i], i);
        
        // Remove index i - (n - k) from the window
        if (i >= (n - k)) {
            remove(A[i - (n - k)], i - (n - k));
        }
        
        // Once window is full, compute max - min
        if (i >= (n - k) - 1) {
            res = min(res, 1LL * A[max_deq.front()] - A[min_deq.front()]);
        }
    }

    cout << res << endl;
}
```
