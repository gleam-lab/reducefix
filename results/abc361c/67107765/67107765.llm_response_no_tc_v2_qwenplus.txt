#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
#define chmin(x,y) x = min(x,y)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    // We need to remove exactly K elements, leaving (n-k) elements
    // The remaining elements must be contiguous in the sorted array to minimize max-min
    // Try all possible contiguous subarrays of length (n-k)
    
    int len = n - k;
    int ans = a[len - 1] - a[0]; // initial value: first window
    
    for (int i = 1; i + len - 1 < n; i++) {
        chmin(ans, a[i + len - 1] - a[i]);
    }
    
    cout << ans << endl;
}
