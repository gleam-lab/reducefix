#define chmax(x,y) x = max(x,y)
#define chmin(x,y) x = min(x,y)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We remove exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    int ans = a[keep - 1] - a[0]; // Initial candidate: take first 'keep' elements
    
    // Try all possible windows of length 'keep'
    // The idea is that after removing K elements, the remaining sequence must be a contiguous segment in the sorted array
    // because we want to minimize (max - min), so it's optimal to take a consecutive block
    for (int i = 0; i + keep - 1 < n; i++) {
        chmin(ans, a[i + keep - 1] - a[i]);
    }
    
    cout << ans << endl;
}
