#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define pb push_back
using ll = long long;
using P = pair<int, int>;
const ll LINF = 1001002003004005006ll;
const int INF = 1001001001;
#define chmin(x,y) x = min(x,y)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    int rem = n - k; // remaining elements
    int ans = a[rem - 1] - a[0]; // initial window [0, rem-1]
    
    // try all possible windows of length 'rem'
    for (int i = 1; i + rem - 1 < n; i++) {
        chmin(ans, a[i + rem - 1] - a[i]);
    }
    
    cout << ans << endl;
}
