#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<int, int>;

#define chmin(a, b) if ((a) > (b)) (a) = (b)
#define chmax(a, b) if ((a) < (b)) (a) = (b)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N, K;
    cin >> N >> K;

    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to choose a subsequence of length N-K by removing K elements
    // and minimize (max - min) in the resulting sequence.

    // Since we want to minimize max - min,
    // we can consider all windows of size (N-K) in the array,
    // and find the window with minimum (max - min).

    // To do this efficiently, use sliding window with multiset
    multiset<int> window;
    int res = numeric_limits<int>::max();

    for (int i = 0; i < N; ++i) {
        window.insert(A[i]);

        if (i >= N - K - 1) {
            // Window size reached N-K
            int mn = *window.begin();
            int mx = *window.rbegin();
            chmin(res, mx - mn);
            // Remove the element going out of the window
            window.erase(window.find(A[i - (N - K - 1)]));
        }
    }

    cout << res << endl;
}
