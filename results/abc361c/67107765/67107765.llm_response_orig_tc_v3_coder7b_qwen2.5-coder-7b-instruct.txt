```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<long long> num(n);
    for (int i = 0; i < n; i++) {
        cin >> num[i];
    }
    
    sort(num.begin(), num.end());
    
    // Calculate the difference between the largest and smallest elements after removing k elements
    long long result = num[n-k-1] - num[k];
    
    cout << result << endl;
    
    return 0;
}
```
