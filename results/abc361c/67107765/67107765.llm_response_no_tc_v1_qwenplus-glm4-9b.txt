int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, k;
    cin >> n >> k;
    vector<int> num(n);
    for (int i = 0; i < n; ++i) {
        cin >> num[i];
    }
    sort(num.begin(), num.end());
    
    // If we remove the smallest k elements, the max is at the end of the remaining elements (n-k), and min is at the start (0)
    // If we remove the largest k elements, the max is at the kth element, and min is at the kth from the end (n-k)
    // We need to consider the scenario where we remove exactly k elements and calculate the difference in max and min
    int max_after_removal = num[n - k];
    int min_after_removal = num[k - 1];
    int max_diff = max_after_removal - min_after_removal;

    // We also need to consider removing k elements that might include the border between the smallest and largest k elements
    // This could potentially create a smaller difference between the max and min of the remaining elements
    // The potential new minimums after removal are elements at positions k and n-k-1
    int potential_new_min = min(num[k - 1], num[n - k - 1]);
    int new_max_diff = num[n - 1] - potential_new_min;

    // The result is the minimum of the two differences calculated
    int result = min(max_diff, new_max_diff);
    cout << result << endl;
}
