#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    
    // Reading the array elements
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Sorting the array to help find the minimum difference
    sort(a.begin(), a.end());
    
    // The minimum difference initialized to a large value
    long long min_diff = LLONG_MAX;
    
    // We need to choose k elements to remove, so we consider removing k-1 elements from the start
    // and 1 element from the end, and then calculate the difference of the maximum and minimum of the remaining elements
    for (int i = 0; i <= k - 1; i++) {
        // Calculate the difference of the largest remaining element and the smallest remaining element
        long long diff = a[i + k - 1] - a[i];
        // Update the minimum difference found so far
        min_diff = min(min_diff, diff);
    }
    
    // Output the minimum difference
    cout << min_diff << endl;
    
    return 0;
}
