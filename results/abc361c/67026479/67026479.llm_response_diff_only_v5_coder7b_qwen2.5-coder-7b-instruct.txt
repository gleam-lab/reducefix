```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = LLONG_MAX;
    for (int i = 0; i <= k; ++i) {
        ans = min(ans, static_cast<long long>(a[i + n - k - 1]) - a[i]);
    }
    
    cout << ans << endl;
    
    return 0;
}
```
