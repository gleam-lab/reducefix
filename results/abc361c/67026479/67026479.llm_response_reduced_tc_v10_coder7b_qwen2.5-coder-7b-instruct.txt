    sort(a.begin(), a.end());

    long long min_diff = LLONG_MAX;
    for(int i = 0; i <= n - k; ++i) {
        min_diff = min(min_diff, a[i + k - 1] - a[i]);

    cout << min_diff << endl;
    return 0;
}
