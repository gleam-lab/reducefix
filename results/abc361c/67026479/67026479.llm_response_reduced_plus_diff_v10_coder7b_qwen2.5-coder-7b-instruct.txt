```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<long long> a(n);
    
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    sort(a.begin(), a.end());

    long long min_diff = LLONG_MAX;

    for(int i = 0; i <= n - k; ++i) {
        min_diff = min(min_diff, a[i + k - 1] - a[i]);
    }

    cout << min_diff << endl;

    return 0;
}
```

This solution sorts the array first, which takes O(N log N). Then it iterates through the sorted array to find the minimum difference between the largest and smallest numbers among any subarray of size N-K, which also takes O(N). Therefore, the overall time complexity is O(N log N), which meets the problem's constraints. The space complexity is O(1) since we only use a few extra variables regardless of the input size.
