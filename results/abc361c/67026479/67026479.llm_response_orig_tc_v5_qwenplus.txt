```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    int m = n - k; // number of remaining elements
    long long dmin = (long long)a[n-1] - a[0]; // initialize with worst case
    
    // Try all contiguous subarrays of length m in the sorted array
    for(int i = 0; i <= n - m; i++) {
        long long d = (long long)a[i + m - 1] - a[i];
        dmin = min(dmin, d);
    }
    
    cout << dmin << endl;
}
```
