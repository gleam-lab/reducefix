#include <bits/stdc++.h>
using namespace std;
int main() {
  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  for (int i = 0; i < n; i++) cin >> a[i];

  // Pair each element with its original index and sort by value
  vector<pair<int, int>> indexed_a(n);
  for (int i = 0; i < n; i++) indexed_a[i] = {a[i], i};
  sort(indexed_a.begin(), indexed_a.end());

  // Calculate the minimum possible difference by iterating over the sorted array
  long long min_diff = LLONG_MAX;
  for (int i = 0; i <= k; i++) {
    // Calculate the maximum of the remaining elements and the minimum of the removed elements
    long long max_of_remaining = indexed_a[i + k - 1].first;
    long long min_of_removed = indexed_a[i].first;
    min_diff = min(min_diff, max_of_remaining - min_of_removed);
  }

  cout << min_diff << endl;
}
