```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k, m;
  cin >> n >> k;
  m = n - k;
  
  vector<int> a(n);
  for (int i = 0; i < n; ++i) {
    cin >> a[i];
  }

  // We need to find the minimum difference between min and max
  // over all contiguous subsequences of length m in the original array
  deque<int> minQ, maxQ;
  int res = INT_MAX;

  for (int i = 0; i < n; ++i) {
    // Maintain deque for min
    while (!minQ.empty() && minQ.front() < i - m + 1) minQ.pop_front();
    while (!minQ.empty() && a[minQ.back()] >= a[i]) minQ.pop_back();
    minQ.push_back(i);

    // Maintain deque for max
    while (!maxQ.empty() && maxQ.front() < i - m + 1) maxQ.pop_front();
    while (!maxQ.empty() && a[maxQ.back()] <= a[i]) maxQ.pop_back();
    maxQ.push_back(i);

    // Start calculating result when we have a window of size m
    if (i >= m - 1) {
      res = min(res, a[maxQ.front()] - a[minQ.front()]);
    }
  }

  cout << res << endl;
}
```
