```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  for (int i = 0; i < n; ++i) {
    cin >> a[i];
  }

  // The size of the subsequence we need to keep
  int m = n - k;

  // If only one element is kept, the difference is zero
  if (m == 1) {
    cout << 0 << endl;
    return 0;
  }

  // We'll maintain two sliding windows:
  // One for min and max in the window of size m

  deque<int> maxdq;  // stores indices of elements in decreasing order (max at front)
  deque<int> mindq;  // stores indices of elements in increasing order (min at front)

  long long min_diff = numeric_limits<long long>::max();

  for (int i = 0; i < n; ++i) {
    // Maintain maxdq
    while (!maxdq.empty() && a[i] > a[maxdq.back()]) {
      maxdq.pop_back();
    }
    maxdq.push_back(i);

    // Remove out-of-window indices
    while (maxdq.front() <= i - m) {
      maxdq.pop_front();
    }

    // Maintain mindq
    while (!mindq.empty() && a[i] < a[mindq.back()]) {
      mindq.pop_back();
    }
    mindq.push_back(i);

    // Remove out-of-window indices
    while (mindq.front() <= i - m) {
      mindq.pop_front();
    }

    // Once we have processed at least m elements, start computing the difference
    if (i >= m - 1) {
      int current_max = a[maxdq.front()];
      int current_min = a[mindq.front()];
      min_diff = min(min_diff, (long long)(current_max - current_min));
    }
  }

  cout << min_diff << endl;
}
```
