
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    
    // Read the array elements
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    // Sort the array to facilitate easy calculation of the maximum difference
    sort(a.begin(), a.end());
    
    // The minimum possible value of the maximum value of B minus the minimum value of B
    // is the difference between the k-th smallest and the (n-k)-th smallest element
    int min_possible_value = a[k - 1] - a[n - k];
    
    // Output the result
    cout << min_possible_value << endl;
    
    return 0;
}
```
