
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

int main() {
    int n, k;
    std::cin >> n >> k;
    std::vector<int> a(n);
    
    // Read the array elements
    for (int i = 0; i < n; ++i) {
        std::cin >> a[i];
    }
    
    // Sort the array to find the minimum and maximum values easily
    std::sort(a.begin(), a.end());
    
    // Initialize the result with the difference between the largest and smallest elements
    long long result = a[n - 1] - a[0];
    
    // If k is 0, no elements are removed, so the answer is 0
    if (k == 0) {
        std::cout << 0 << std::endl;
        return 0;
    }
    
    // Calculate the possible results by removing elements
    for (int i = 0; i + k <= n; ++i) {
        long long max_val = a[n - 1]; // Maximum value is the last element after possible removal
        long long min_val = a[i + k]; // Minimum value is the element at index i+k after removal
        
        // Update the result if the current difference is smaller
        result = std::min(result, max_val - min_val);
    }
    
    // Output the result
    std::cout << result << std::endl;
    
    return 0;
}
```
