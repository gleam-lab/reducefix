```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // We will use two sliding windows to find min and max in window of size m = n - k
    int m = n - k;
    deque<int> min_deq, max_deq;
    long long min_diff = LLONG_MAX;

    auto slide_window = [&](function<void(int)> process) {
        min_deq.clear();
        max_deq.clear();
        for (int i = 0; i < n; ++i) {
            // Maintain min deque
            while (!min_deq.empty() && a[i] <= a[min_deq.back()])
                min_deq.pop_back();
            min_deq.push_back(i);

            // Remove elements out of window
            if (min_deq.front() <= i - m)
                min_deq.pop_front();

            // Maintain max deque
            while (!max_deq.empty() && a[i] >= a[max_deq.back()])
                max_deq.pop_back();
            max_deq.push_back(i);

            if (max_deq.front() <= i - m)
                max_deq.pop_front();

            if (i >= m - 1)
                process(i);
        }
    };

    slide_window([&](int i) {
        min_diff = min(min_diff, (long long)a[max_deq.front()] - (long long)a[min_deq.front()]);
    });

    cout << min_diff << "\n";
    return 0;
}
```
