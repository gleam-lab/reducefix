```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    int m = n - k; // length of the remaining sequence
    long long dmin = a[m - 1] - a[0]; // initialize with first window
    
    for (int i = 1; i + m - 1 < n; i++) {
        dmin = min(dmin, static_cast<long long>(a[i + m - 1] - a[i]));
    }
    
    cout << dmin << endl;
    
    return 0;
}
```
