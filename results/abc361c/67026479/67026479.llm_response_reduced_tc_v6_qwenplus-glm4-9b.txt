#include <bits/stdc++.h>
using namespace std;
int main() {
  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  for (int i = 0; i < n; i++) cin >> a[i];

  // We create a min-heap to keep track of the smallest K elements that we might not remove
  priority_queue<int, vector<int>, greater<int>> min_heap;
  // We create a max-heap to keep track of the largest K elements that we might remove
  priority_queue<int> max_heap;

  // Push the first K elements into the min-heap and the rest into the max-heap
  for (int i = 0; i < n; i++) {
    min_heap.push(a[i]);
    if (i >= k) min_heap.pop();
    if (i >= k) max_heap.push(a[i - k]);
  }

  // The maximum element in the max-heap is the one we want to potentially remove
  int max_element_to_remove = max_heap.top();
  // The minimum element in the min-heap is the one we keep
  int min_element_kept = min_heap.top();

  // If k is even, we can remove the middle element without changing the min and max difference
  // If k is odd, we need to consider the next smallest element in the min-heap
  int min_element_kept_next = (k % 2 == 0) ? min_element_kept : min_heap.top();

  // The minimum possible value of the maximum minus the minimum of B is the minimum of
  // the difference between the max element to remove and the smallest kept element,
  // and the difference between the largest kept element and the next smallest kept element.
  int result = min(max_element_to_remove - min_element_kept, min_element_kept_next - min_element_kept);
  cout << result << endl;
}
