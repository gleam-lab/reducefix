#include <bits/stdc++.h>
using namespace std;
int main() {
  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  for (int i = 0; i < n; i++) cin >> a[i];
  
  // If k is 0, we don't remove any elements, so the range is the whole array
  if (k == 0) {
    int min_val = *min_element(a.begin(), a.end());
    int max_val = *max_element(a.begin(), a.end());
    cout << max_val - min_val << endl;
    return 0;
  }

  sort(a.begin(), a.end());
  
  // To find the minimum possible value of the maximum value of B minus the minimum value of B,
  // we need to consider the impact of removing k elements.
  // This can be achieved by checking how the min and max values of the array change when we
  // remove elements. We need to maximize the difference between the max and min of the remaining elements.
  
  // We will consider the effect of removing k-1 elements from both ends of the sorted array.
  vector<long long> prefix_sum(n + 1, 0);
  vector<long long> suffix_sum(n + 1, 0);

  // Compute prefix sums
  for (int i = 0; i < n; i++) {
    prefix_sum[i + 1] = prefix_sum[i] + a[i];
  }

  // Compute suffix sums
  for (int i = n - 1; i >= 0; i--) {
    suffix_sum[i] = suffix_sum[i + 1] + a[i];
  }

  long long result = LLONG_MAX;
  
  // Check the difference from removing k-1 elements from the start
  for (int i = 0; i <= min(k - 1, n - k); i++) {
    long long min_val = prefix_sum[i] / i;
    long long max_val = (prefix_sum[n] - prefix_sum[i]) / (n - i - 1);
    result = min(result, max_val - min_val);
  }
  
  // Check the difference from removing k-1 elements from the end
  for (int i = 0; i <= min(k - 1, n - k); i++) {
    long long min_val = suffix_sum[n - i - 1] / (n - i - 1);
    long long max_val = (suffix_sum[n] - suffix_sum[n - i - 1]) / i;
    result = min(result, max_val - min_val);
  }
  
  cout << result << endl;
}
