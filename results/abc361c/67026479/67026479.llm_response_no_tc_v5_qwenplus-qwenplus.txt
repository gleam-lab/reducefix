```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  
  vector<int> a(n);
  for (int i = 0; i < n; ++i) {
    cin >> a[i];
  }

  // We need to remove exactly K elements
  // So we will keep N - K elements in the final sequence B
  int m = n - k;
  
  // Edge case: if only one element is left, the value is 0
  if (m == 1) {
    cout << 0 << endl;
    return 0;
  }

  // To find the minimum possible value of max(B) - min(B),
  // we slide a window of size m over the sorted array,
  // because the minimal difference between max and min in any subsequence
  // occurs when the elements are as close together as possible.
  sort(a.begin(), a.end());

  int min_diff = INT_MAX;
  for (int i = 0; i + m <= n; ++i) {
    min_diff = min(min_diff, a[i + m - 1] - a[i]);
  }

  cout << min_diff << endl;
  return 0;
}
```
