```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  
  for(int i = 0; i < n; i++)
    cin >> a[i];
  
  sort(a.begin(), a.end());
  
  long long min_diff = LLONG_MAX;
  
  for(int i = 0; i <= n - k; i++) {
    min_diff = min(min_diff, static_cast<long long>(a[i + k - 1]) - a[i]);
  }
  
  cout << min_diff << endl;
  
  return 0;
}
```

This solution correctly addresses the problem by sorting the array first, which allows us to easily find the smallest difference between any two values that could be the result of removing `K` elements. The loop iterates through all possible subarrays of size `N-K`, calculating the difference between the largest and smallest element in each subarray and keeping track of the minimum difference found. This approach ensures that we efficiently find the optimal solution within the constraints provided.
