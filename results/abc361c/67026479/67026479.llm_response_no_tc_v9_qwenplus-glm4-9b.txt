#include <bits/stdc++.h>
using namespace std;
int main(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    
    // Sort the array to be able to easily find the maximum and minimum of the remaining elements
    sort(a.begin(), a.end());
    
    // We need to find the minimum possible value of the max - min after removing k elements
    // We can do this by considering the scenarios where we are removing the smallest or largest elements
    // and checking the difference between the max and min of the remaining elements after each removal
    
    // Initialize the answer with a large value
    long long ans = LLONG_MAX;
    
    // We consider removing k elements, so we have n - k elements left
    // We need to find the minimum difference between the maximum of the remaining elements and the minimum
    // after considering all possible removals of k elements
    
    // We do this by considering the sliding window of size n-k on the sorted array
    // and finding the difference between the first and last element in each window
    for (int i = 0; i <= n - k; i++) {
        long long max_val = a[i + k - 1];
        long long min_val = a[i];
        ans = min(ans, max_val - min_val);
    }
    
    cout << ans << endl;
}
