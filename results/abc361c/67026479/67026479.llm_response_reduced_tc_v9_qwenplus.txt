```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    int m = n - k; // number of remaining elements
    sort(a.begin(), a.end());
    
    // If only one element remains, the difference is 0
    if (m == 1) {
        cout << 0 << endl;
        return 0;
    }
    
    long long dmin = LLONG_MAX;
    // Check all possible contiguous subarrays of length m in sorted array
    for (int i = 0; i <= n - m; i++) {
        long long d = a[i + m - 1] - a[i];
        dmin = min(dmin, d);
    }
    
    cout << dmin << endl;
    return 0;
}
```
