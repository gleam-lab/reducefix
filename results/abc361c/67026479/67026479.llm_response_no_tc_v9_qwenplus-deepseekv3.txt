#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    int m = n - k;
    if (m == 1) {
        cout << 0 << endl;
        return 0;
    }
    
    int left = 0;
    int right = *max_element(a.begin(), a.end()) - *min_element(a.begin(), a.end());
    int answer = right;
    
    while (left <= right) {
        int mid = (left + right) / 2;
        bool found = false;
        
        // Check if there's a subsequence of length >=m with difference <=mid
        // We can use a sliding window to find the longest window where max - min <=mid
        // Then check if the window length >=m
        
        deque<int> minDeque, maxDeque;
        int start = 0;
        found = false;
        for (int end = 0; end < n; ++end) {
            // Maintain the minDeque
            while (!minDeque.empty() && a[end] < a[minDeque.back()]) {
                minDeque.pop_back();
            }
            minDeque.push_back(end);
            
            // Maintain the maxDeque
            while (!maxDeque.empty() && a[end] > a[maxDeque.back()]) {
                maxDeque.pop_back();
            }
            maxDeque.push_back(end);
            
            // Ensure the window [start..end] has max - min <= mid
            while (a[maxDeque.front()] - a[minDeque.front()] > mid) {
                start++;
                while (!minDeque.empty() && minDeque.front() < start) {
                    minDeque.pop_front();
                }
                while (!maxDeque.empty() && maxDeque.front() < start) {
                    maxDeque.pop_front();
                }
            }
            
            if (end - start + 1 >= m) {
                found = true;
                break;
            }
        }
        
        if (found) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    cout << answer << endl;
    return 0;
}

This code uses binary search to find the smallest possible difference 'd' for which there exists a subsequence of length at least m where the maximum minus the minimum is <= 'd'. The feasibility check is done using a sliding window approach with deques to maintain the minimum and maximum efficiently. 

This approach ensures that we find the correct answer efficiently with a time complexity of O(n log (max_diff)), which is feasible for the given constraints.
