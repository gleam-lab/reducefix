```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // We need to keep exactly m elements
    int m = n - k;

    // Edge case: if only one element needs to be kept
    if (m == 1) {
        cout << 0 << endl;
        return 0;
    }

    // Sort the array to find smallest range containing m elements
    sort(a.begin(), a.end());

    long long min_diff = LLONG_MAX;
    for (int i = 0; i + m - 1 < n; ++i) {
        min_diff = min(min_diff, (long long)a[i + m - 1] - a[i]);
    }

    cout << min_diff << endl;
}
```
