#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    // Sort the array to easily find the minimum difference between any two elements
    sort(a.begin(), a.end());
    
    int min_diff = INT_MAX;
    
    // Iterate through all possible subarrays of size n-k
    for (int i = 0; i <= n - k; ++i) {
        int diff = a[i + k - 1] - a[i];
        min_diff = min(min_diff, diff);
    }
    
    cout << min_diff << endl;
    
    return 0;
}
