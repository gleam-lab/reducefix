```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    sort(a.begin(), a.end());

    long long min_diff = LLONG_MAX;
    for (int i = 0; i <= n - k; ++i) {
        min_diff = min(min_diff, a[i + k - 1] - a[i]);
    }

    cout << min_diff << endl;

    return 0;
}
```

This solution sorts the array first, which takes \(O(N \log N)\) time. Then it iterates through the sorted array to find the minimum difference between the largest and smallest elements in any subarray of size \(N - K\). This step takes \(O(N)\) time. Therefore, the overall time complexity is \(O(N \log N)\), which is efficient enough given the problem constraints. The space complexity is \(O(1)\) as we only use a few extra variables.
