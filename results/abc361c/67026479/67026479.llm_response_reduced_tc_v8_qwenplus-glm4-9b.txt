
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    // We need to remove exactly k elements to minimize the max-min difference.
    // To achieve this, we should remove the largest and the smallest k/2 elements
    // and then calculate the difference between the maximum and minimum of the remaining elements.
    
    // Sort the array to easily find the k/2 largest and smallest elements.
    sort(a.begin(), a.end());
    
    // Calculate the indices of the elements that are not removed.
    int start = n - k;
    int min_val = a[start];
    int max_val = a[start + k/2 - 1]; // We take k/2 elements in the middle if k is even, k/2 + 1 if k is odd.
    
    // Output the difference between the max and min of the remaining elements.
    cout << max_val - min_val << endl;
    
    return 0;
}
```
