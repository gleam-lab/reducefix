```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<long long> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    sort(a.begin(), a.end());

    long long min_diff = LLONG_MAX;
    for(int i = 0; i <= n - k; ++i) {
        long long diff = a[i + k - 1] - a[i];
        min_diff = min(min_diff, diff);
    }

    cout << min_diff << endl;
    return 0;
}
```

It sorts the array and iterates through possible subarrays of size `n-k` to find the minimum difference between the maximum and minimum elements in any such subarray. This approach ensures that we consider all possible ways to remove `k` elements and form the resulting sequence with the smallest range. The time complexity is dominated by the sorting step, making it efficient even for large inputs up to the constraint limits.
