```python
import sys

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    k = int(data[1])
    A = list(map(int, data[2:2+n]))
    
    # We need to remove exactly K elements, so we keep (n - k) elements.
    # To minimize (max(B) - min(B)), we should consider keeping a contiguous segment
    # in the sorted array, because any non-contiguous selection can be improved by
    # taking a contiguous block with smaller range.
    
    A.sort()
    keep = n - k
    min_diff = float('inf')
    
    # Try all contiguous segments of length 'keep'
    for i in range(n - keep + 1):
        j = i + keep - 1
        diff = A[j] - A[i]
        if diff < min_diff:
            min_diff = diff
    
    print(min_diff)

solve()
```
