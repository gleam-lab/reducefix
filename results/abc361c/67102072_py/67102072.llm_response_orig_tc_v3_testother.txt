```python
import sys

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    k = int(data[1])
    A = list(map(int, data[2:2+n]))
    
    # We are removing exactly K elements, so we keep (n - k) elements.
    # To minimize (max(B) - min(B)), we should consider contiguous subarrays
    # of length (n - k) in the sorted array, but preserving original order isn't needed
    # because we can choose any K elements to remove. So optimal strategy:
    # Sort the array, then try all windows of size (n-k) in sorted order.
    # Why? Because the minimal range will always come from a set of values that are
    # consecutive in sorted order.
    
    A.sort()
    window_size = n - k
    min_diff = float('inf')
    
    for i in range(n - window_size + 1):
        j = i + window_size - 1
        diff = A[j] - A[i]
        if diff < min_diff:
            min_diff = diff
    
    print(min_diff)

solve()
```
