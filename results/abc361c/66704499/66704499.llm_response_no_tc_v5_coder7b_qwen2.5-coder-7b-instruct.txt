```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int median = nums[(r+l+1) / 2];
        int dl = abs(median - nums[l]);
        int dr = abs(median - nums[r]);

        if (dl <= dr){
            l++;
        }
        else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```

### Explanation of Changes
- The condition inside the while loop was incorrectly checking `if (dl > dr)` instead of `if (dl <= dr)`. This change ensures that we always try to minimize the range by removing the larger end element when possible.
- No other changes were necessary as the logic was fundamentally correct but needed a slight adjustment to ensure the optimal solution is chosen consistently.
