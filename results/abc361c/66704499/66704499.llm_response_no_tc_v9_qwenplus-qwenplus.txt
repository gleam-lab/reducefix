```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will consider all intervals of size n - k in the sorted array
    // because to minimize max(B) - min(B), we should choose a contiguous
    // block of n - K elements in the sorted array that has minimal spread

    vector<int> sorted_A = A;
    sort(sorted_A.begin(), sorted_A.end());

    int min_diff = numeric_limits<int>::max();
    int window_size = n - k;

    for (int i = 0; i + window_size <= n; ++i) {
        int diff = sorted_A[i + window_size - 1] - sorted_A[i];
        min_diff = min(min_diff, diff);
    }

    cout << min_diff << endl;
    return 0;
}
```
