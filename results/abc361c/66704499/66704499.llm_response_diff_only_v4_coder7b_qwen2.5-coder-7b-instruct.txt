    while (k--) {
        int dl = abs(nums[l + 1] - nums[l]);
        int dr = abs(nums[r] - nums[r - 1]);

        if (dl <= dr){
            l++;
        }
        else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
