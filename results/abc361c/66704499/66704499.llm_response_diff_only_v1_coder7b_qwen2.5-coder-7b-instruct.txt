```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int median = nums[(r+l+1) / 2];
        int dl = abs(median - nums[l]);
        int dr = abs(median - nums[r]);

        if (dl <= dr){ // Corrected condition to ensure removal of larger values when possible
            l++;
        }
        else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```

In the original code, there was a logical error where the decision to increment or decrement `l` or `r` was based on the absolute difference with the median. The condition was incorrectly set to check for greater than instead of less than or equal to, which led to incorrect removals. This has been corrected in the provided solution.
