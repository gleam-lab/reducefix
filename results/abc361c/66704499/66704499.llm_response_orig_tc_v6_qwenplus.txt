```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = a[n - 1] - a[0];
    
    // We are removing exactly K elements
    // Try all possible ways to remove x from left and (k-x) from right
    for (int left_removed = 0; left_removed <= k; left_removed++) {
        int right_removed = k - left_removed;
        
        if (left_removed + right_removed > n) continue;
        if (n - left_removed - right_removed == 0) continue;
        
        int l = left_removed;
        int r = n - 1 - right_removed;
        
        ans = min(ans, (long long)(a[r] - a[l]));
    }
    
    cout << ans << endl;
    
    return 0;
}
```
