```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main() {
    int n, k;
    cin >> n >> k;
    
    for (int i = 0; i < n; i++) {
        cin >> nums[i];
    }
    
    sort(nums, nums + n);
    
    // We need to remove exactly K elements
    // The remaining sequence will have (n - k) elements
    // We want to minimize (max - min) of the remaining elements
    
    // Since array is sorted, we can consider all possible contiguous subarrays
    // of length (n-k) in the sorted array
    // This works because removing K elements optimally means keeping a contiguous
    // segment in the sorted array to minimize the range
    
    int min_range = nums[n-1] - nums[0];
    int remaining = n - k;
    
    for (int i = 0; i + remaining - 1 < n; i++) {
        int j = i + remaining - 1;
        min_range = min(min_range, nums[j] - nums[i]);
    }
    
    cout << min_range << endl;
    
    return 0;
}
```
