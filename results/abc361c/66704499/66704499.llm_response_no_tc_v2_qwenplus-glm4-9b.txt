
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // Sort the array to easily find the minimum and maximum
    sort(A.begin(), A.end());

    // If k is 0, no elements are removed, and the range is the whole array
    if (k == 0) {
        cout << A.back() - A.front() << endl;
        return 0;
    }

    // The best case scenario after removing k elements is to remove the largest k elements
    // Because we want to minimize the maximum value in B and the maximum value in B will be
    // the last element remaining after removing k elements and the minimum value in B will be
    // the first element remaining after removing k elements
    int min_val = A[k];  // The smallest element that will not be removed
    int max_val = A[n - 1];  // The largest element in the array

    cout << max_val - min_val << endl;

    return 0;
}
```
