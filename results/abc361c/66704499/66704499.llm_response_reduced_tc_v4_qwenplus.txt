```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    int min_diff = INT_MAX;
    // We remove exactly K elements, so we keep (n - k) elements
    int keep = n - k;
    
    // Try all possible contiguous subarrays of length 'keep'
    // The optimal solution will always be a contiguous segment in sorted array
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        min_diff = min(min_diff, a[j] - a[i]);
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
