#include <bits/stdc++.h>
using namespace std;

int nums[200000];

    int n, k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    // Select the smallest K+1 elements and the largest K+1 elements
    int min_val = min(nums[k], nums[n-k-1]);

    return 0;
}

4. **Calculating Difference**: The minimum possible difference between the maximum and minimum values of the resulting sequence is calculated and printed.
