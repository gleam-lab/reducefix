```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will use a sliding window approach to find the minimum range
    // after removing exactly K elements

    // Sort the array to analyze values in increasing order
    sort(A.begin(), A.end());

    // Maintain two pointers: left and right
    // such that there are (N - K) elements in the window
    int min_diff = INT_MAX;
    for (int i = 0; i + (N - K) - 1 < N; ++i) {
        int diff = A[i + (N - K) - 1] - A[i];
        min_diff = min(min_diff, diff);
    }

    cout << min_diff << endl;

    return 0;
}
```
