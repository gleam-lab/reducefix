```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        // Calculate the potential reduction in the range by removing the current max or min
        int rm_l = nums[l+k] - nums[l]; 
        int rm_r = nums[r] - nums[r-k-1]; 

        // Decide whether to remove from the left or right to minimize the range difference
        if (rm_l <= rm_r) {
            l++;
        } else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```
