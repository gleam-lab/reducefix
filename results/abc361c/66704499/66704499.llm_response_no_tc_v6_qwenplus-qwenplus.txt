```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will use a sliding window approach to keep N - K elements
    // The goal is to minimize (max - min) in the remaining elements

    // Since we need to keep N - K consecutive elements (in original order), we can slide a window of size N - K over the array
    int window_size = n - k;
    int min_diff = numeric_limits<int>::max();

    for (int i = 0; i + window_size <= n; ++i) {
        int j = i + window_size - 1;
        int min_val = *min_element(A.begin() + i, A.begin() + j + 1);
        int max_val = *max_element(A.begin() + i, A.begin() + j + 1);
        min_diff = min(min_diff, max_val - min_val);
    }

    cout << min_diff << endl;
    return 0;
}
```
