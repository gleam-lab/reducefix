```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = 1e18;
    // Try removing k elements: left i elements and right (k-i) elements
    for (int i = 0; i <= k; i++) {
        int left = i;
        int right = n - 1 - (k - i);
        if (left <= right) {
            ans = min(ans, (long long)(a[right] - a[left]));
        }
    }
    
    cout << ans << endl;
    
    return 0;
}
```
