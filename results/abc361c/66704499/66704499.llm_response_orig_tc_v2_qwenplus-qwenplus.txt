```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will use a sliding window approach to find the minimum range
    // After removing exactly K elements from the array

    // The idea is that we can remove x elements from the front and (K - x) elements from the back
    // where x ranges from 0 to K

    int min_diff = numeric_limits<int>::max();
    multiset<int> left, right;

    // Initialize prefix of size (N - K)
    for (int i = 0; i < N - K; ++i) {
        right.insert(A[i]);
    }

    auto get_range = [&]() {
        return *prev(right.end()) - *right.begin();
    };

    min_diff = get_range();

    // Slide the window by moving one element from right to left each time
    for (int i = N - K; i < N; ++i) {
        // Move A[i] into left set
        left.insert(A[i]);

        // Move A[window_start - 1] out of right set
        right.erase(right.find(A[i - (N - K)]));

        // Rebalance sets so that right contains the current window
        while (!left.empty()) {
            int l_max = *prev(left.end());
            int r_min = *right.begin();
            if (l_max < r_min) break;

            // Move max of left to right
            right.insert(l_max);
            left.erase(left.find(l_max));
        }

        min_diff = min(min_diff, get_range());
    }

    cout << min_diff << endl;
    return 0;
}
```
