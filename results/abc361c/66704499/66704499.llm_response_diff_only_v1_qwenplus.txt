```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    // We remove exactly K elements, so we keep (n - k) elements
    // The optimal strategy is to remove some prefix and/or suffix
    // Try all possible ways: remove i elements from left and (k-i) from right
    long long min_diff = 1e18;
    
    for (int i = 0; i <= k; i++) {
        int left = i;           // Remove i elements from the left
        int right = n - 1 - (k - i);  // Remove (k-i) elements from the right
        
        if (left <= right) {
            min_diff = min(min_diff, (long long)(a[right] - a[left]));
        }
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
