#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define pb push_back
#define mp make_pair
#define eb emplace_back
#define ff first
#define ss second 
#define inf INT_MAX
#define all(v) v.begin(), v.end()
#define rep(i,x) for(int i=0;i<(int)(x);i++)
#define rev(v) reverse(v.begin(),v.end())

typedef long long ll;
typedef pair<int, int> pii;

void solve(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    rep(i, n) cin >> a[i];
    
    sort(all(a));
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    if (keep == 1) {
        cout << 0 << endl;
        return;
    }
    
    // Try all possible contiguous subarrays of length 'keep'
    // because the optimal remaining sequence must be contiguous in sorted order
    // to minimize max - min.
    ll ans = LLONG_MAX;
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        ans = min(ans, (ll)(a[j] - a[i]));
    }
    
    cout << ans << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
