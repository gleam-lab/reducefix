#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define pb push_back
#define mp make_pair
#define ff first
#define ss second 
#define all(v) v.begin(), v.end()
#define rep(i,x) for(int i=0;i<(int)(x);i++)
#define int long long

typedef long long ll;
typedef pair<int, int> pii;

void solve(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    rep(i, n) cin >> a[i];
    
    sort(all(a));
    
    // We are removing exactly K elements
    // So we are keeping (n - k) consecutive elements from the sorted array
    // To minimize (max - min), we try all possible contiguous subarrays of length (n-k)
    
    int keep = n - k;
    int ans = a[keep - 1] - a[0]; // initial value
    
    for (int i = 1; i + keep - 1 < n; i++) {
        ans = min(ans, a[i + keep - 1] - a[i]);
    }
    
    cout << ans << endl;
}

signed main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
