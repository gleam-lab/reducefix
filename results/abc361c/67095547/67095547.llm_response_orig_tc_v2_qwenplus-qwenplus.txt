```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // We need to remove K elements such that the difference between max and min in remaining is minimized
    // Since order must be preserved but we only care about final min and max, sorting helps

    sort(a.begin(), a.end());

    // We try all possible windows of size (n - k)
    int window_size = n - k;
    int min_diff = numeric_limits<int>::max();

    for (int i = 0; i + window_size <= n; ++i) {
        min_diff = min(min_diff, a[i + window_size - 1] - a[i]);
    }

    cout << min_diff << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
