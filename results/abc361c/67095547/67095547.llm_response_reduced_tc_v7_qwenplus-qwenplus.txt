```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to remove K elements to minimize the difference between
    // max and min of the remaining N-K elements.
    // The minimal difference is obtained by choosing a window of size N-K in the sorted array.

    sort(A.begin(), A.end());
    int min_diff = numeric_limits<int>::max();
    int M = N - K;

    for (int i = 0; i + M <= N; ++i) {
        min_diff = min(min_diff, A[i + M - 1] - A[i]);
    }

    cout << min_diff << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
