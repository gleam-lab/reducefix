```cpp
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define ll long long

void solve() {
    ll n, k;
    cin >> n >> k;
    vector<ll> a(n);
    for (ll i = 0; i < n; ++i) cin >> a[i];

    // Sort the array to easily find the min and max
    sort(a.begin(), a.end());

    // The minimum possible value of the difference between the max and min
    // will be achieved by removing the largest k elements and keeping the rest,
    // or removing the smallest k elements and keeping the rest.
    ll ans = min(a[n - 1] - a[k], a[n - k - 1] - a[0]);

    cout << ans << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
