#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define int long long

void solve() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all windows of size (N - K) and find min difference between max and min
    int windowSize = N - K;
    deque<int> maxDeque, minDeque;
    int res = LLONG_MAX;

    for (int i = 0; i < N; ++i) {
        // Maintain deque for maximum values
        while (!maxDeque.empty() && A[maxDeque.back()] <= A[i]) {
            maxDeque.pop_back();
        }
        maxDeque.push_back(i);

        // Maintain deque for minimum values
        while (!minDeque.empty() && A[minDeque.back()] >= A[i]) {
            minDeque.pop_back();
        }
        minDeque.push_back(i);

        // Remove elements outside the window
        if (i >= windowSize - 1) {
            while (maxDeque.front() <= i - windowSize) {
                maxDeque.pop_front();
            }
            while (minDeque.front() <= i - windowSize) {
                minDeque.pop_front();
            }
            res = min(res, A[maxDeque.front()] - A[minDeque.front()]);
        }
    }

    cout << res << endl;
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
