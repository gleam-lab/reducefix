void solve(){
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(all(a));
    
    // We remove exactly K elements, so we keep (n - k) elements.
    // The remaining elements must be contiguous in the sorted array to minimize range.
    // Try all possible contiguous subarrays of length (n - k)
    int min_range = a[n-1] - a[0]; // worst case: no removal helps
    int len = n - k;
    
    if (len == 1) {
        cout << 0 << endl;
        return;
    }
    
    for (int i = 0; i + len - 1 < n; i++) {
        int j = i + len - 1;
        min_range = min(min_range, a[j] - a[i]);
    }
    
    cout << min_range << endl;
}

signed main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
