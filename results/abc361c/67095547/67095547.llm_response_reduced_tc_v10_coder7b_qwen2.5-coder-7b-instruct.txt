#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<long long> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    sort(a.begin(), a.end());
    
    // Calculate the minimum possible value of the difference between the maximum and minimum values of the resulting sequence
    long long min_diff = LLONG_MAX;
    for(int i = 0; i <= k; ++i) {
        long long diff = a[n - 1 - i] - a[k - i];
        min_diff = min(min_diff, diff);
    }
    
    cout << min_diff << endl;
    
    return 0;
}

For each position `i`, we calculate the difference between the maximum element (`a[n-1-i]`) and the minimum element (`a[k-i]`) of the remaining subarray. We keep track of the minimum difference encountered.

4. **Outputting the Result**: Finally, we print the minimum difference found.

This approach ensures that we find the optimal way to remove `k` elements such that the difference between the maximum and minimum values of the resulting sequence is minimized. The time complexity of this solution is \(O(N \log N)\), which is efficient enough given the constraints.
