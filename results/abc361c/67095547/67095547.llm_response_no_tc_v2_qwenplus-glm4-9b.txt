#include <bits/stdc++.h>
using namespace std;

#define endl "\n"

void solve() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    // Use a multiset to maintain the sorted order of elements
    multiset<int> s(a.begin(), a.end());

    // Remove the K smallest or largest elements
    auto it = next(s.begin(), k);
    if (k <= n / 2) {
        s.erase(s.begin(), it);
    } else {
        s.erase(it, s.end());
    }

    // The answer is the difference between the maximum and minimum of the remaining elements
    cout << *(s.rbegin()) - *(s.begin()) << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
