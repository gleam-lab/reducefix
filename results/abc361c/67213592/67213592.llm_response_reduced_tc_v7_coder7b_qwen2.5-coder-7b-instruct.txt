```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto &val : v) {
        cin >> val;
    }

    sort(v.begin(), v.end());

    int result = INT_MAX;
    for (int i = 0; i <= k; ++i) {
        int left = i;
        int right = n - k + i - 1;
        result = min(result, v[right] - v[left]);
    }

    cout << result << endl;
    return 0;
}
```

For each segment, we calculate the difference between the maximum and minimum values and update the `result` accordingly.
