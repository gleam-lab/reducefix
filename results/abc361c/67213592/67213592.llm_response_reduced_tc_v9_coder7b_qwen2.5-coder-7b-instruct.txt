```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto &val : v) {
        cin >> val;
    }

    sort(v.begin(), v.end());

    long long ans = LLONG_MAX;
    for (int i = 0; i <= k; ++i) {
        ans = min(ans, static_cast<long long>(v[n - 1 - i] - v[i]));
    }

    cout << ans << endl;
    return 0;
}
```

This solution sorts the array first, then iterates through all possible pairs of indices `(i, n - 1 - i)` where `i` ranges from `0` to `k`. For each pair, it calculates the difference between the element at index `n - 1 - i` and the element at index `i`, updating the minimum difference found so far. This approach ensures that we consider all possible ways to remove `K` elements to minimize the difference between the maximum and minimum values of the resulting array. The time complexity of this solution is \(O(K + N \log N)\), which is efficient enough given the problem constraints.
