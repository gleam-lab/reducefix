```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We remove exactly K elements, so we keep (n - k) elements.
    // To minimize (max - min) of the remaining subsequence,
    // we should consider contiguous segments of length (n - k) in the sorted array.
    int min_diff = v[n - 1] - v[0];
    int keep = n - k;
    
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        min_diff = min(min_diff, v[j] - v[i]);
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
