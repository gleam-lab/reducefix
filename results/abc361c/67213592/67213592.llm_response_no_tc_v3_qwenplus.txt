```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    // The optimal strategy is to remove some prefix and/or suffix.
    // Try all possible combinations: remove i from left, (k-i) from right.
    long long min_diff = 1e18;
    for (int i = 0; i <= k; i++) {
        int left_idx = i;
        int right_idx = n - 1 - (k - i);
        if (left_idx <= right_idx) {
            min_diff = min(min_diff, (long long)(v[right_idx] - v[left_idx]));
        }
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
