```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    long long min_diff = 1e18;
    
    // Try all possible contiguous subarrays of length 'keep'
    // because the optimal remaining sequence must be contiguous in sorted order
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        min_diff = min(min_diff, (long long)(v[j] - v[i]));
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
