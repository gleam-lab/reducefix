```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }

    sort(v.begin(), v.end());

    int min_diff = INT_MAX;
    for (int i = 0; i <= k; ++i) {
        min_diff = min(min_diff, v[n-k+i-1] - v[i]);
    }

    cout << min_diff << endl;

    return 0;
}
```

This solution correctly identifies the optimal way to remove `K` elements from the array `A` to minimize the difference between the maximum and minimum values in the resulting array `B`. The approach involves sorting the array and then iterating through possible removals to find the minimum difference efficiently. This ensures that the solution handles all constraints and edge cases appropriately.
