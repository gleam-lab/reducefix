```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We are removing exactly K elements
    // So we are keeping (n - k) elements
    int keep = n - k;
    int min_diff = v[n-1] - v[0]; // worst case: entire array
    
    // Try all possible contiguous subarrays of length 'keep'
    // because the optimal remaining sequence must be contiguous in sorted order
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        min_diff = min(min_diff, v[j] - v[i]);
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
