```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, k;
    std::cin >> n >> k;

    std::vector<int> v(n);
    for (auto& val : v) {
        std::cin >> val;
    }

    std::sort(v.begin(), v.end());

    long long min_value = std::numeric_limits<long long>::max();
    for (int i = 0; i <= n - k; ++i) {
        min_value = std::min(min_value, static_cast<long long>(v[i + k - 1]) - v[i]);
    }

    std::cout << min_value << std::endl;

    return 0;
}
```

This solution sorts the array first, which takes \(O(N \log N)\) time. Then it iterates through the sorted array once to find the minimum possible difference between the maximum and minimum values of any subarray of length \(N-K\). This step takes \(O(N)\) time. Therefore, the overall time complexity is \(O(N \log N)\), which meets the problem's constraints. The space complexity is \(O(1)\) beyond the input storage.
