```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    long long min_diff = v[n-1] - v[0]; // worst case: keep everything
    
    // Try all possible contiguous segments of length 'keep'
    // The optimal remaining sequence must be a contiguous block in the sorted array
    for (int left = 0; left + keep - 1 < n; left++) {
        int right = left + keep - 1;
        min_diff = min(min_diff, (long long)(v[right] - v[left]));
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
