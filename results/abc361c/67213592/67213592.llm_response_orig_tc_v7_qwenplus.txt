```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto& val : v) {
        cin >> val;
    }
    
    sort(v.begin(), v.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    // The optimal way is to remove some prefix and/or suffix.
    // Try all possible combinations: remove i from left, and (k-i) from right.
    long long min_diff = 1e18;
    for (int i = 0; i <= k; i++) {
        int left_removed = i;
        int right_removed = k - i;
        int left_index = left_removed;
        int right_index = n - 1 - right_removed;
        
        if (left_index <= right_index) {
            long long diff = v[right_index] - v[left_index];
            if (diff < min_diff) {
                min_diff = diff;
            }
        }
    }
    
    cout << min_diff << endl;
    
    return 0;
}
```
