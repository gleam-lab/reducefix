```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (auto &val : v) {
        cin >> val;
    }

    sort(v.begin(), v.end());

    int best = INT_MAX;
    for (int i = 0; i <= k; ++i) {
        best = min(best, v[n-k+i-1] - v[i]);
    }

    cout << best << endl;
    return 0;
}
```
