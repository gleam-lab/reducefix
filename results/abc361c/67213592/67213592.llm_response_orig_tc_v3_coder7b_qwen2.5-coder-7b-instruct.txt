```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, k;
    std::cin >> n >> k;

    std::vector<int> v(n);
    for (auto& val : v) {
        std::cin >> val;
    }

    std::sort(v.begin(), v.end());

    int min_diff = v[n-k-1] - v[0];

    for (int i = 1; i <= k; ++i) {
        min_diff = std::min(min_diff, v[n-k+i-1] - v[i]);
    }

    std::cout << min_diff;
}
```
