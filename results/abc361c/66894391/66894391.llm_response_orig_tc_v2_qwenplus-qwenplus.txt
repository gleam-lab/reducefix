```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size (n - K)
    // That is, find the minimum range of (n-K) elements in sorted array
    int window_size = n - k;
    sort(A.begin(), A.end());

    long long min_diff = numeric_limits<long long>::max();
    for (int i = 0; i + window_size <= n; ++i) {
        min_diff = min(min_diff, (long long)(A[i + window_size - 1] - A[i]));
    }

    cout << min_diff << endl;

    return 0;
}
```
