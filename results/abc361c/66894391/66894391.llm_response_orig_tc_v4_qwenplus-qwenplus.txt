```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size (N - K)
    int window_size = N - K;
    int min_diff = numeric_limits<int>::max();

    // Sliding window approach
    multiset<int> window;
    for (int i = 0; i < N; ++i) {
        window.insert(A[i]);

        // Maintain the window size
        if (window.size() > window_size)
            window.erase(window.find(A[i - window_size]));

        // When the window is full, compute the difference
        if (i >= window_size - 1)
            min_diff = min(min_diff, *window.rbegin() - *window.begin());
    }

    cout << min_diff << endl;
    return 0;
}
```
