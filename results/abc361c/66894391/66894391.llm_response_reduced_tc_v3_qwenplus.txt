```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = a[n - 1] - a[0];
    
    // We remove exactly k elements
    // Try all possible ways: remove i from left and (k-i) from right
    for (int i = 0; i <= k; i++) {
        int left_removed = i;
        int right_removed = k - i;
        
        int l = left_removed;
        int r = n - 1 - right_removed;
        
        if (l <= r) {
            ans = min(ans, (long long)(a[r] - a[l]));
        }
    }
    
    cout << ans << endl;
    
    return 0;
}
```
