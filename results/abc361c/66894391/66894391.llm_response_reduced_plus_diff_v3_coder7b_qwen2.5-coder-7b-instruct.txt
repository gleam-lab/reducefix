```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int median = nums[(l+r)/2]; // Corrected median calculation
        int dl = abs(median - nums[l]);
        int dr = abs(median - nums[r]);

        if (dl > dr){
            l++;
        } else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```

### Explanation of Changes:
- The issue was with the median calculation in the loop. It should be `(l+r)/2` instead of `(r+l+1)/2`. This ensures that we always take the lower middle element when there's an even number of elements left to consider.
- No other changes were needed as the logic and approach were correct.
