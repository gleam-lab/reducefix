#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;

    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size (n - k)
    int window_size = n - k;

    // To efficiently find the min and max in sliding window,
    // we use two deques: one for max, one for min
    deque<int> max_deque;
    deque<int> min_deque;
    
    int result = numeric_limits<int>::max();

    for (int i = 0; i < n; ++i) {
        // Maintain max_deque for current window [i-window_size+1, i]
        while (!max_deque.empty() && A[i] > A[max_deque.back()]) {
            max_deque.pop_back();
        }
        max_deque.push_back(i);

        // Maintain min_deque for current window [i-window_size+1, i]
        while (!min_deque.empty() && A[i] < A[min_deque.back()]) {
            min_deque.pop_back();
        }
        min_deque.push_back(i);

        // Remove elements out of window
        if (max_deque.front() <= i - window_size) {
            max_deque.pop_front();
        }
        if (min_deque.front() <= i - window_size) {
            min_deque.pop_front();
        }

        // Update result once window is formed
        if (i >= window_size - 1) {
            result = min(result, A[max_deque.front()] - A[min_deque.front()]);
        }
    }

    cout << result << endl;

    return 0;
}
