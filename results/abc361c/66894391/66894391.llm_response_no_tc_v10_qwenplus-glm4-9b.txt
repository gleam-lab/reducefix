#include <bits/stdc++.h>
using namespace std;

int main(void){
    int n, k;
    cin >> n >> k;

    vector<int> nums(n);
    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    // Copy the elements to a new vector for manipulation
    vector<int> nums_to_remove(nums.size());
    for (int i = 0; i < n; i++) {
        nums_to_remove[i] = nums[i];
    }

    // Sort the vector for easy manipulation
    sort(nums_to_remove.begin(), nums_to_remove.end());

    // We need to remove K elements to minimize the range of the remaining elements
    // We should remove the smallest K elements to achieve this
    nums_to_remove.erase(nums_to_remove.begin(), nums_to_remove.begin() + k);

    // The minimum possible value of the maximum value of remaining elements minus
    // the minimum value of remaining elements is the difference between the maximum
    // and minimum of the remaining elements
    int min_value = *min_element(nums_to_remove.begin(), nums_to_remove.end());
    int max_value = *max_element(nums_to_remove.begin(), nums_to_remove.end());

    cout << max_value - min_value << endl;

    return 0;
}
