```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int dl = nums[l + 1] - nums[l];
        int dr = nums[r] - nums[r - 1];
        if (dl <= dr){
            l++;
        }
        else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```

This solution fixes the logic error in determining which end of the sorted array to trim based on whether removing the left or right element would minimize the difference between the maximum and minimum values in the resulting array. The corrected version now checks the difference between consecutive elements at the current bounds and trims the side that does not increase the gap between the largest and smallest elements.
