```cpp
#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int dl = nums[l + k] - nums[l]; // Calculate difference with the leftmost element after removals
        int dr = nums[r] - nums[r - k]; // Calculate difference with the rightmost element after removals
        if (dl < dr){ // Choose the side that minimizes the maximum difference
            l++;
        } else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
```
