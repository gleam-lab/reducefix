#include <bits/stdc++.h>
using namespace std;

int nums[200000];

int main(void){
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; i++){
        cin >> nums[i];
    }

    sort(nums, nums + n);

    int l = 0;
    int r = n - 1;

    while (k--) {
        int median = nums[(r+l+1) / 2];
        int dl = abs(median - nums[l]);
        int dr = abs(median - nums[r]);

        if (dl > dr){
            l++;
        }
        else {
            r--;
        }
    }

    cout << nums[r] - nums[l] << endl;

    return 0;
}
