```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = 1e18;
    
    // We remove exactly K elements. So we keep (n - k) consecutive elements.
    // Try all possible windows of length (n - k) in the sorted array.
    for (int i = 0; i + (n - k) - 1 < n; i++) {
        int j = i + (n - k) - 1;
        ans = min(ans, (long long)(a[j] - a[i]));
    }
    
    cout << ans << endl;
    
    return 0;
}
```
