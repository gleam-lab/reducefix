```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = 1e18;
    // We are removing exactly K elements, so we keep (n - k) elements.
    // Try all possible contiguous segments of length (n - k)
    for (int i = 0; i + (n - k) - 1 < n; i++) {
        int j = i + (n - k) - 1;
        ans = min(ans, (long long)(a[j] - a[i]));
    }
    
    cout << ans << endl;
    
    return 0;
}
```
