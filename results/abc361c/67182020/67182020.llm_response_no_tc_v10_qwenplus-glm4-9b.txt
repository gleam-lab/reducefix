
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // If K is 0, we cannot remove any elements, so the answer is 0.
    if (K == 0) {
        cout << 0 << endl;
        return 0;
    }

    // If we remove all elements, the answer is also 0.
    if (K == N) {
        cout << 0 << endl;
        return 0;
    }

    sort(A.begin(), A.end());

    // If removing K elements results in a sequence with less than 2 elements,
    // we cannot calculate the max - min, so we need special handling.
    if (N - K < 2) {
        cout << "0" << endl;
        return 0;
    }

    // We need to consider the case where we remove K elements
    // and calculate the min possible max - min difference.
    long long minDiff = LLONG_MAX;

    // We can remove elements from either end of the sorted array.
    // We need to consider the ranges [l, r] where 0 <= l <= r <= N - K - 1.
    for (int l = 0; l <= N - K - 1; ++l) {
        int r = l + K;
        // Calculate the max and min of the remaining elements.
        int maxVal = *max_element(A.begin() + l, A.begin() + r);
        int minVal = *min_element(A.begin() + l, A.begin() + r);
        // Update the minimum difference.
        minDiff = min(minDiff, (long long)maxVal - minVal);
    }

    cout << minDiff << endl;
    return 0;
}
```
