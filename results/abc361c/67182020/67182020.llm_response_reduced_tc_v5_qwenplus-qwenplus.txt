#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    cin.tie(nullptr);
    ios_base::sync_with_stdio(false);
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all windows of size (N - K)
    int window_size = N - K;
    int min_diff = numeric_limits<int>::max();

    // Create a sliding window with efficient min/max tracking
    deque<int> max_deq, min_deq;

    auto add_to_window = [&](int i) {
        // Maintain deque for maximums
        while (!max_deq.empty() && A[i] >= A[max_deq.back()]) {
            max_deq.pop_back();
        }
        max_deq.push_back(i);

        // Maintain deque for minimums
        while (!min_deq.empty() && A[i] <= A[min_deq.back()]) {
            min_deq.pop_back();
        }
        min_deq.push_back(i);
    };

    // Initialize the first window
    for (int i = 0; i < window_size; ++i) {
        add_to_window(i);
    }

    for (int i = window_size; i < N; ++i) {
        // Calculate current window's max-min
        min_diff = min(min_diff, A[max_deq.front()] - A[min_deq.front()]);

        // Remove elements that are out of the window
        if (max_deq.front() <= i - window_size) {
            max_deq.pop_front();
        }
        if (min_deq.front() <= i - window_size) {
            min_deq.pop_front();
        }

        // Add new element to the window
        add_to_window(i);
    }

    // Check last window
    min_diff = min(min_diff, A[max_deq.front()] - A[min_deq.front()]);

    cout << min_diff << endl;
    return 0;
}
