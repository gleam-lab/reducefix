```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for(int i = 0; i < N; i++) cin >> A[i];

    sort(A.begin(), A.end());

    int minDiff = INT_MAX;
    for(int i = 0; i <= N-K; i++){
        int diff = A[i+K-1] - A[i];
        minDiff = min(minDiff, diff);
    }

    cout << minDiff << endl;

    return 0;
}
```

This code sorts the array and then iterates through it, calculating the difference between the Kth largest and smallest numbers for each subarray of size N-K. It keeps track of the minimum difference found. The time complexity is O(N log N) due to sorting and O(N) for iterating, which is efficient for the given constraints.
