```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size N-K in the original array
    // Since we can't reorder elements, we must look at contiguous subarrays of length N-K
    int min_diff = numeric_limits<int>::max();
    deque<int> min_deq, max_deq;

    // Sliding window to maintain min and max in O(1) per step
    auto remove_from_deques = [&](int x) {
        if (!min_deq.empty() && min_deq.front() == x)
            min_deq.pop_front();
        if (!max_deq.empty() && max_deq.front() == x)
            max_deq.pop_front();
    };

    auto add_to_deques = [&](int x) {
        while (!min_deq.empty() && min_deq.back() > x)
            min_deq.pop_back();
        min_deq.push_back(x);

        while (!max_deq.empty() && max_deq.back() < x)
            max_deq.pop_back();
        max_deq.push_back(x);
    };

    // Initialize the first window
    for (int i = 0; i < N - K; ++i) {
        add_to_deques(A[i]);
    }

    min_diff = max_deq.front() - min_deq.front();

    // Slide the window
    for (int i = N - K; i < N; ++i) {
        remove_from_deques(A[i - (N - K)]);
        add_to_deques(A[i]);
        min_diff = min(min_diff, max_deq.front() - min_deq.front());
    }

    cout << min_diff << endl;
    return 0;
}
```
