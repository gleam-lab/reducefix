2. Sorting the array A helps, as we can then look at the elements from both ends of the array, which give us the potential maximum and minimum values of the remaining sequence.

The initial approach is to consider the middle K elements of A as they are a good starting point for removing elements. The reason is that by removing the middle elements, we are removing both high and low values, potentially lowering the maximum and raising the minimum of the remaining sequence.

However, the problem statement has a constraint: exactly K elements must be removed. The solution needs to consider the edge cases:

- If K is even, then the best strategy might be to remove K/2 elements from both ends of the array.
- If K is odd, then the best strategy is to remove K/2 - 1 elements from one end and K/2 + 1 elements from the other end.

#include <bits/stdc++.h>
#include <vector>
#include <iostream>
using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    // When K is even, removing K/2 elements from both ends.
    if (K % 2 == 0) {
        int min_val = A[0];
        int max_val = A[N - 1];
        cout << max_val - min_val << endl;
    } else { // When K is odd, removing K/2 - 1 elements from one end and K/2 + 1 elements from the other end.
        int min_val = min({A[0], A[N - 1]});

        cout << max_val - min_val << endl;
    }

    return 0;
}
