```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    // We are removing exactly K elements, so we keep (N - K) elements.
    // The optimal strategy is to remove some number of elements from the left and right ends
    // since the array is sorted. We try all valid combinations:
    // Remove i elements from the left and (K - i) elements from the right, for i in [0, K].
    // Then the remaining segment is from index i to N-1-(K-i) = N-K+i-1.
    // The value is A[N-K+i-1] - A[i].

    int min_diff = INT_MAX;
    for (int i = 0; i <= K; ++i) {
        int left_removed = i;
        int right_removed = K - i;
        int remaining_start = left_removed;
        int remaining_end = N - 1 - right_removed;
        
        if (remaining_start <= remaining_end) {
            min_diff = min(min_diff, A[remaining_end] - A[remaining_start]);
        }
    }

    cout << min_diff << endl;

    return 0;
}
```
