```cpp
#include <bits/stdc++.h>
using namespace std;

long long minMaxDifference(vector<int>& A, int K) {
    sort(A.begin(), A.end());
    int n = A.size();
    long long res = LONG_MAX;
    for (int i = 0; i <= K; ++i) {
        res = min(res, static_cast<long long>(A[n - 1 - i] - A[i]));
    }
    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    cout << minMaxDifference(A, K) << endl;

    return 0;
}
```

This solution sorts the array first and then iterates through all possible subarrays of length `N - K` to find the minimum difference between the maximum and minimum values in any such subarray. This ensures we consider all possible ways to remove `K` elements and get the desired result. The time complexity of this solution is \(O(N \log N)\) due to sorting, which is efficient enough given the problem constraints.
