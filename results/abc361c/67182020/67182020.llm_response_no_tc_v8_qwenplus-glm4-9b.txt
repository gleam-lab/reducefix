#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main() {
    cin.tie(nullptr);
    ios_base::sync_with_stdio(false);
    int N, K;
    cin >> N >> K;
    vector<long long> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    // Use two pointers to find the minimum possible value of the maximum - minimum of B
    int left = 0, right = N - K - 1;
    long long min_val = A[left], max_val = A[right];

    while (left < right) {
        if (A[left] < A[right]) {
            min_val = min(min_val, A[left]);
            left++;
        } else {
            max_val = max(max_val, A[right]);
            right--;
        }
        if ((right - left + 1) % 2 == 1) {
            // If there's an odd number of elements to remove, check both ends
            if (A[left] < A[right]) {
                max_val = max(max_val, A[left]);
                min_val = min(min_val, A[left]);
                left++;
            } else {
                max_val = max(max_val, A[right]);
                min_val = min(min_val, A[right]);
                right--;
            }
        }
    }

    cout << max_val - min_val << endl;
    return 0;
}
