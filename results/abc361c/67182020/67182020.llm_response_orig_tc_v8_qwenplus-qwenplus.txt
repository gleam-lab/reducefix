```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define FASTIO ios_base::sync_with_stdio(false); cin.tie(nullptr);
#define all(v) (v).begin(), (v).end()

int main() {
    FASTIO
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to remove K elements such that the remaining sequence has minimum (max - min)
    // Since the elements are not required to be contiguous, we can sort and choose any N-K elements
    // The optimal way is to minimize (A[i + (N-K-1)] - A[i]) for i from 0 to K

    sort(all(A));
    
    long long ans = LLONG_MAX;
    for (int i = 0; i <= K; ++i) {
        ans = min(ans, 1LL * A[i + (N - K - 1)] - A[i]);
    }
    
    cout << ans << "\n";
    return 0;
}
```
