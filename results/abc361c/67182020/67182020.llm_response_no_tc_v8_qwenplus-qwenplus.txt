```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size N-K in the array
    // and compute max - min for each window.
    // Then return the minimum among these values.

    deque<int> maxDq;  // stores indices of elements in decreasing order (max at front)
    deque<int> minDq;  // stores indices of elements in increasing order (min at front)
    int result = INT_MAX;

    for (int i = 0; i < N; ++i) {
        // Remove elements out of window
        while (!maxDq.empty() && maxDq.front() <= i - (N - K)) {
            maxDq.pop_front();
        }
        while (!minDq.empty() && minDq.front() <= i - (N - K)) {
            minDq.pop_front();
        }

        // Remove elements smaller than current from maxDq
        while (!maxDq.empty() && A[maxDq.back()] <= A[i]) {
            maxDq.pop_back();
        }
        // Remove elements larger than current from minDq
        while (!minDq.empty() && A[minDq.back()] >= A[i]) {
            minDq.pop_back();
        }

        maxDq.push_back(i);
        minDq.push_back(i);

        // When window is complete
        if (i >= (N - K - 1))            {
            int currentMax = A[maxDq.front()];
            int currentMin = A[minDq.front()];
            result = min(result, currentMax - currentMin);
        }
    }

    cout << result << endl;
    return 0;
}
```
