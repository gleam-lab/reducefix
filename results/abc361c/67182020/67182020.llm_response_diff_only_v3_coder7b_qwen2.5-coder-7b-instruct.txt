```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    cin >> N >> K;
    vector<long long> A(N);
    for(int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    long long ans = LLONG_MAX;
    for(int i = 0; i <= N - K; ++i) {
        ans = min(ans, A[i + K - 1] - A[i]);
    }

    cout << ans << '\n';

    return 0;
}
```
