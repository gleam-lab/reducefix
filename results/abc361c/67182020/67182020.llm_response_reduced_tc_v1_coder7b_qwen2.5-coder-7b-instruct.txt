#include <bits/stdc++.h>
using namespace std;

int main() {

    cin.tie(NULL);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    int minDiff = INT_MAX;
    for (int i = 0; i <= N - K; ++i) {
        minDiff = min(minDiff, A[i + K - 1] - A[i]);
    }

    cout << minDiff << endl;
    return 0;
}

4. **Output**: Finally, we output the minimum difference.

This approach ensures that we efficiently find the minimum possible value of the difference between the maximum and minimum values of the resulting sequence after removing `K` elements. The time complexity is \(O(N \log N)\) due to the sorting step, which is efficient enough for the given constraints.
