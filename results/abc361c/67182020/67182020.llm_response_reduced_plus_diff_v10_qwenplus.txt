```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    // We are removing exactly K elements.
    // The remaining sequence has length N - K.
    // We want to minimize (max - min) of the remaining elements.
    // So we try all possible contiguous segments of length N-K in the sorted array.

    int min_diff = INT_MAX;
    for (int i = 0; i + (N - K) <= N; ++i) {
        int j = i + (N - K) - 1;
        min_diff = min(min_diff, A[j] - A[i]);
    }

    cout << min_diff << endl;

    return 0;
}
```
