```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to choose N-K elements such that the difference between max and min is minimized
    // So we remove K elements in a way that leaves us with a contiguous subsequence of length N-K
    // where the difference between max and min is minimized

    int L = N - K;
    int result = numeric_limits<int>::max();

    // We slide a window of size L over the array after sorting it
    sort(A.begin(), A.end());
    for (int i = 0; i + L - 1 < N; ++i) {
        result = min(result, A[i + L - 1] - A[i]);
    }

    cout << result << endl;
    return 0;
}
```
