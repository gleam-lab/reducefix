#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size (N - K)
    int window_size = N - K;
    int min_diff = numeric_limits<int>::max();

    // Sliding window approach
    deque<int> max_deque, min_deque;
    for (int i = 0; i < N; ++i) {
        // Maintain deque for maximum values
        while (!max_deque.empty() && A[max_deque.back()] <= A[i])
            max_deque.pop_back();
        max_deque.push_back(i);

        // Maintain deque for minimum values
        while (!min_deque.empty() && A[min_deque.back()] >= A[i])
            min_deque.pop_back();
        min_deque.push_back(i);

        // Remove elements out of the current window
        if (max_deque.front() <= i - window_size)
            max_deque.pop_front();
        if (min_deque.front() <= i - window_size)
            min_deque.pop_front();

        // Calculate the difference once the window is full
        if (i >= window_size - 1) {
            int diff = A[max_deque.front()] - A[min_deque.front()];
            min_diff = min(min_diff, diff);
        }
    }

    cout << min_diff << endl;
    return 0;
}
