```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    cin >> N >> K;
    vector<long long> A(N);
    for (int i = 0; i < N; ++i) cin >> A[i];

    sort(A.begin(), A.end());

    long long ans = LLONG_MAX;
    for (int i = 0; i <= K; ++i) {
        long long min_val = A[i];
        long long max_val = A[N - K + i - 1];
        ans = min(ans, max_val - min_val);
    }

    cout << ans << endl;
    return 0;
}
```

This solution sorts the array first, which takes O(N log N) time. Then it iterates through all possible subarrays of length `N-K`, calculating the difference between the maximum and minimum values in each subarray and updating the answer with the smallest difference found. This step takes O(N) time. Therefore, the overall time complexity is dominated by the sorting step, making it O(N log N). The space complexity is O(1) since we only use a few extra variables.
