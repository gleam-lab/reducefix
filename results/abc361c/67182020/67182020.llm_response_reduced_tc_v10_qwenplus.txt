```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(A.begin(), A.end());

    // We are removing exactly K elements, so we keep M = N - K elements.
    int M = N - K;
    ll min_diff = 1e18;

    // Try all possible contiguous segments of length M in the sorted array.
    // Since we can only remove elements (not reorder), the optimal remaining elements
    // must form a contiguous segment in the sorted array to minimize max - min.
    for (int i = 0; i + M - 1 < N; ++i) {
        int j = i + M - 1;
        min_diff = min(min_diff, (ll)(A[j] - A[i]));
    }

    cout << min_diff << endl;

    return 0;
}
```
