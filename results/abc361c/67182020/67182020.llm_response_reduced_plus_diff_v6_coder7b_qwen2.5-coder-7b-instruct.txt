```cpp
#include <bits/stdc++.h>
using namespace std;

long long solve(vector<long long> &a, int k) {
    sort(a.begin(), a.end());
    int n = a.size();
    long long ans = LLONG_MAX;
    for(int i = 0; i <= n-k; ++i) {
        ans = min(ans, a[i+k-1] - a[i]);
    }
    return ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    cin >> n >> k;
    vector<long long> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    cout << solve(a, k) << '\n';

    return 0;
}
```

This solution sorts the array first and then iterates through all possible subarrays of size `N-K+1` to find the minimum difference between the maximum and minimum values in each subarray. The result is the minimum of these differences. This approach ensures that we consider all possible ways to remove `K` elements and find the optimal solution. The time complexity of this solution is O(N log N) due to the sorting step, which is efficient enough given the problem constraints.
