#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will keep exactly (n - K) elements in the final array B
    int m = n - K;

    // We'll use a sliding window approach to find the minimum possible (max - min)
    deque<int> min_deque, max_deque;
    int result = numeric_limits<int>::max();

    for (int i = 0; i < n; ++i) {
        // Maintain deque for min: remove elements out of window and those >= current element
        while (!min_deque.empty() && min_deque.front() <= i - m) {
            min_deque.pop_front();
        }
        while (!min_deque.empty() && A[min_deque.back()] >= A[i]) {
            min_deque.pop_back();
        }
        min_deque.push_back(i);

        // Maintain deque for max: remove elements out of window and those <= current element
        while (!max_deque.empty() && max_deque.front() <= i - m) {
            max_deque.pop_front();
        }
        while (!max_deque.empty() && A[max_deque.back()] <= A[i]) {
            max_deque.pop_back();
        }
        max_deque.push_back(i);

        // Once we have processed at least m elements, start computing results
        if (i >= m - 1) {
            result = min(result, A[max_deque.front()] - A[min_deque.front()]);
        }
    }

    cout << result << endl;
    return 0;
}
