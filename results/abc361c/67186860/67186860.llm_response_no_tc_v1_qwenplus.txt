```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We need to remove exactly K elements
    // The remaining sequence will have n-k elements
    // To minimize (max - min) of remaining elements,
    // we should consider all possible ways to remove K elements from ends
    // Since array is sorted, optimal removal is always from the two ends
    
    int ans = INT_MAX;
    int remaining = n - k;
    
    // Try all possible splits: remove i elements from left, (k-i) from right
    for (int i = 0; i <= k; i++) {
        int left_removed = i;
        int right_removed = k - i;
        
        // Remaining elements are from index [left_removed] to [n-1-right_removed]
        int left_idx = left_removed;
        int right_idx = n - 1 - right_removed;
        
        if (left_idx <= right_idx) {
            ans = min(ans, a[right_idx] - a[left_idx]);
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
