```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We'll consider the minimum possible value of (max(B) - min(B)) over all possible sequences B
    // where B is formed by removing exactly K elements from A.

    // The key observation is that in order to minimize (max(B) - min(B)), we want to keep
    // the values as close together as possible. So we should remove the most extreme values.

    // One correct approach:
    // Sort A and consider all N-K length windows in sorted A.
    // For each such window, it represents a possible set of remaining elements with min and max difference.
    // Take the minimum among all these differences.

    sort(A.begin(), A.end());

    long long result = numeric_limits<long long>::max();
    for (int i = 0; i + (n - k) - 1 < n; ++i) {
        result = min(result, (long long)A[i + (n - k) - 1] - (long long)A[i]);
    }

    cout << result << endl;
    return 0;
}
```
