```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<long long> a(n);
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  
  // Sort the array
  sort(a.begin(), a.end());
  
  // Calculate the minimum difference between the maximum and minimum of all possible subarrays after removing k elements
  long long ans = LLONG_MAX;
  for (int i = 0; i <= n-k; i++) {
    ans = min(ans, a[i+k-1] - a[i]);
  }
  
  cout << ans << endl;
  return 0;
}
```
