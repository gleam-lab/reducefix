```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to find the minimum possible value of (max - min) after removing K elements.
    // So we are keeping (N - K) elements in B.

    int L = N - K;

    // To minimize max(B) - min(B), we can consider all intervals of length L in the sorted array.
    // Because B is formed by deleting K elements, its elements must remain in order,
    // but for minimal range, best strategy is to sort A and slide a window of size L.

    sort(A.begin(), A.end());

    long long min_diff = numeric_limits<long long>::max();
    for (int i = 0; i + L <= N; ++i) {
        min_diff = min(min_diff, (long long)(A[i + L - 1] - A[i]));
    }

    cout << min_diff << endl;
    return 0;
}
```
