#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> A(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i];
  }
  
  // We need to choose K elements to remove, so the size of B will be n - k
  // To minimize the maximum value of B - minimum value of B, we should remove
  // the K smallest or the K largest elements from the sequence.
  
  // We will sort the array to easily remove the smallest or largest elements
  sort(A.begin(), A.end());
  
  // If K is odd, we can't have an equal split, so we need to remove the middle K elements
  // If K is even, we can remove the middle K elements and consider the split to be even
  vector<int> lower_half(A.begin(), A.begin() + n - k);
  vector<int> upper_half(A.begin() + n - k, A.end());
  
  // Sort both halves to find the minimum possible difference
  sort(lower_half.begin(), lower_half.end());
  sort(upper_half.begin(), upper_half.end());
  
  // Minimum difference can be one of the following:
  // - The difference between the smallest element of the upper half and the largest element of the lower half
  // - The difference between the second smallest element of the upper half and the second largest element of the lower half
  int min_diff = min(lower_half.back() - upper_half.front(),
                     lower_half[lower_half.size() - 2] - upper_half[upper_half.size() - 2]);
  
  cout << min_diff << endl;
  return 0;
}
