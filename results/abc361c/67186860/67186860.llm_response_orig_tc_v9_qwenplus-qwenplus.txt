#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We need to remove exactly K elements
    // So we will keep N - K elements in B
    int window_size = n - k;

    // Compute prefix min and max
    vector<int> min_prefix(n), max_prefix(n);
    deque<int> min_deque, max_deque;

    auto add_to_deques = [&](int i) {
        while (!min_deque.empty() && A[min_deque.back()] >= A[i]) min_deque.pop_back();
        min_deque.push_back(i);
        while (!max_deque.empty() && A[max_deque.back()] <= A[i]) max_deque.pop_back();
        max_deque.push_back(i);
    };

    auto remove_from_deques = [&](int i) {
        if (!min_deque.empty() && min_deque.front() == i) min_deque.pop_front();
        if (!max_deque.empty() && max_deque.front() == i) max_deque.pop_front();
    };

    // Initialize deques for first window
    for (int i = 0; i < window_size; ++i) {
        add_to_deques(i);
    }

    long long result = LLONG_MAX;

    for (int i = window_size; i <= n; ++i) {
        result = min(result, (long long)(A[max_deque.front()] - A[min_deque.front()]));
        if (i < n) {
            add_to_deques(i);
            remove_from_deques(i - window_size);
        }
    }

    cout << result << endl;
    return 0;
}
