```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<long long> v(n);
  for (int i = 0; i < n; i++) {
    cin >> v[i];
  }
  
  // Sort the array to facilitate finding the minimum and maximum values efficiently
  sort(v.begin(), v.end());
  
  // The optimal strategy is to always remove the smallest and largest elements to minimize the difference between max and min of the resulting array
  long long minDiff = LLONG_MAX; // Initialize with the maximum possible long long value
  
  // Iterate through all possible pairs of indices that could be removed
  for (int i = 0; i <= n - k; ++i) {
    // Calculate the difference between the maximum and minimum values after removing k elements
    long long currentDiff = v[n - 1 - (k - i)] - v[i];
    // Update the minimum difference found so far
    minDiff = min(minDiff, currentDiff);
  }
  
  // Output the minimum difference
  cout << minDiff << endl;
  
  return 0;
}
```
