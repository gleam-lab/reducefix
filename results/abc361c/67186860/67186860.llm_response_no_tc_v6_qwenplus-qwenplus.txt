#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> A(n);
  for (int i = 0; i < n; ++i) {
    cin >> A[i];
  }

  // We will consider all windows of size (n - K) in the array,
  // and find the window where max - min is minimized.
  int window_size = n - k;
  
  // Use a sliding window with two deques to maintain min and max efficiently
  deque<int> min_deq, max_deq;
  int result = numeric_limits<int>::max();

  for (int i = 0; i < n; ++i) {
    // Maintain min_deq: elements are in increasing order
    while (!min_deq.empty() && A[min_deq.back()] >= A[i]) {
      min_deq.pop_back();
    }
    min_deq.push_back(i);

    // Maintain max_deq: elements are in decreasing order
    while (!max_deq.empty() && A[max_deq.back()] <= A[i]) {
      max_deq.pop_back();
    }
    max_deq.push_back(i);

    // Remove elements out of the current window
    if (min_deq.front() <= i - window_size) {
      min_deq.pop_front();
    }
    if (max_deq.front() <= i - window_size) {
      max_deq.pop_front();
    }

    // Once the window is full, compute range
    if (i >= window_size - 1) {
      result = min(result, A[max_deq.front()] - A[min_deq.front()]);
    }
  }

  cout << result << endl;
  return 0;
}
