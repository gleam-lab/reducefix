```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> arr(n);
  for (int i = 0; i < n; i++) {
    cin >> arr[i];
  }
  sort(arr.begin(), arr.end());
  
  int ans = INT_MAX;
  for (int i = 0; i <= k; i++) {
    int max_val = arr[min(n - 1, i + n - k - 1)];
    int min_val = arr[i];
    ans = min(ans, max_val - min_val);
  }
  cout << ans << endl;
  return 0;
}
```
