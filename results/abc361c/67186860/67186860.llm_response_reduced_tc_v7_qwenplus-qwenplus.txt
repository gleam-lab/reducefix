#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
    }

    // We will consider all possible windows of size (n - K)
    int window_size = n - k;
    int min_diff = numeric_limits<int>::max();

    // Maintain indices of elements in current window in a deque
    // such that we can efficiently get min and max
    deque<int> max_deque, min_deque;

    // Sliding window approach
    for (int i = 0; i < n; ++i) {
        // Remove elements outside the window
        while (!max_deque.empty() && max_deque.front() <= i - window_size)
            max_deque.pop_front();
        while (!min_deque.empty() && min_deque.front() <= i - window_size)
            min_deque.pop_front();

        // Remove smaller elements from the end for max_deque
        while (!max_deque.empty() && A[max_deque.back()] <= A[i])
            max_deque.pop_back();
        // Remove larger elements from the end for min_deque
        while (!min_deque.empty() && A[min_deque.back()] >= A[i])
            min_deque.pop_back();

        max_deque.push_back(i);
        min_deque.push_back(i);

        // Once the window is valid (reached required size), calculate difference
        if (i >= window_size - 1) {
            min_diff = min(min_diff, A[max_deque.front()] - A[min_deque.front()]);
        }
    }

    cout << min_diff << endl;
    return 0;
}
