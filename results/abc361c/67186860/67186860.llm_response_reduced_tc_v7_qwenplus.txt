```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    int ans = INT_MAX;
    // We remove exactly K elements, so we keep (n - k) elements.
    // The remaining elements must be a contiguous subarray in the sorted array
    // because otherwise we could get a smaller range by choosing adjacent values.
    int window_size = n - k;
    
    for (int i = 0; i + window_size - 1 < n; i++) {
        ans = min(ans, a[i + window_size - 1] - a[i]);
    }
    
    cout << ans << endl;
    return 0;
}
```
