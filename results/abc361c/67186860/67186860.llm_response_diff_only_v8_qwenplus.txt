```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We need to remove exactly K elements
    // The remaining sequence has length n-k
    // We want to minimize (max - min) of remaining elements
    // Since array is sorted, we can consider contiguous subarrays of length n-k
    
    int ans = INT_MAX;
    int remaining = n - k;
    
    // Try all possible contiguous subarrays of length (n-k)
    // These represent keeping a contiguous block in the sorted array
    for (int i = 0; i + remaining - 1 < n; i++) {
        int j = i + remaining - 1;
        ans = min(ans, a[j] - a[i]);
    }
    
    cout << ans << endl;
    return 0;
}
```
