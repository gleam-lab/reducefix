```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We are removing exactly K elements, so we keep (n - k) elements.
    int keep = n - k;
    long long ans = 1e18;
    
    // Try all possible contiguous subarrays of length 'keep' in the sorted array.
    // Since we want to minimize (max - min), and we can choose any K elements to remove,
    // the optimal remaining elements will form a contiguous segment in the sorted array.
    for (int i = 0; i + keep - 1 < n; i++) {
        int j = i + keep - 1;
        ans = min(ans, (long long)(a[j] - a[i]));
    }
    
    cout << ans << endl;
    return 0;
}
```
