#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> A(n);
  for (int i = 0; i < n; ++i) {
    cin >> A[i];
  }

  // We will consider all possible windows of size (n - K)
  int window_size = n - k;
  int min_diff = numeric_limits<int>::max();

  // Sliding window using two deques to maintain min and max in the window
  deque<int> max_deque, min_deque;

  for (int i = 0; i < n; ++i) {
    // Maintain max_deque for maximum in current window
    while (!max_deque.empty() && A[max_deque.back()] <= A[i]) {
      max_deque.pop_back();
    }
    max_deque.push_back(i);

    // Maintain min_deque for minimum in current window
    while (!min_deque.empty() && A[min_deque.back()] >= A[i]) {
      min_deque.pop_back();
    }
    min_deque.push_back(i);

    // Remove elements out of the current window
    if (max_deque.front() <= i - window_size) {
      max_deque.pop_front();
    }
    if (min_deque.front() <= i - window_size) {
      min_deque.pop_front();
    }

    // Once we have a full window, compute difference
    if (i >= window_size - 1) {
      min_diff = min(min_diff, A[max_deque.front()] - A[min_deque.front()]);
    }
  }

  cout << min_diff << endl;
  return 0;
}
