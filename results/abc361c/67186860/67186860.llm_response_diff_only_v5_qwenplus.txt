```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    // We need to remove exactly K elements, so we keep (n-k) elements
    int keep = n - k;
    if (keep <= 0) {
        cout << 0 << endl;
        return 0;
    }
    
    // Try all possible ways to remove K elements by choosing how many to remove from left and right
    // If we remove i elements from the left, we must remove (k-i) elements from the right
    // Then the remaining array is a[i] to a[n-1-(k-i)] = a[n-k+i-1]
    long long ans = 1e18;
    for (int i = 0; i <= k; i++) {
        int left_removed = i;
        int right_removed = k - i;
        
        // Check if this removal pattern is valid
        if (left_removed + right_removed > n) continue;
        if (n - left_removed - right_removed < 1) continue;
        
        int left_idx = left_removed;
        int right_idx = n - 1 - right_removed;
        
        if (left_idx <= right_idx) {
            ans = min(ans, (long long)(a[right_idx] - a[left_idx]));
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
