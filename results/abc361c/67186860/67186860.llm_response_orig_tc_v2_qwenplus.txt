```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    
    int remaining = n - k;
    int ans = INT_MAX;
    
    // Try all possible contiguous segments of length (n-k) in sorted array
    for (int i = 0; i <= k; i++) {
        int j = i + remaining - 1;
        if (j < n) {
            ans = min(ans, a[j] - a[i]);
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
