```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> vec(n);
  for (int i = 0; i < n; i++) {
    cin >> vec[i];
  }
  sort(vec.begin(), vec.end());
  
  int ans = INT_MAX;
  for (int i = 0; i <= n-k; i++) {
    ans = min(ans, vec[i+k-1] - vec[i]);
  }
  
  cout << ans << endl;
  return 0;
}
```
