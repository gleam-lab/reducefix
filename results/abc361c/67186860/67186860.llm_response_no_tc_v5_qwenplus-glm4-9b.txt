
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> a(n);
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  
  // Sort the array to easily find the minimum and maximum values after removing K elements
  sort(a.begin(), a.end());
  
  // We need to consider the two scenarios:
  // 1. Remove K elements from the beginning of the sorted array
  // 2. Remove K elements from the end of the sorted array
  
  // Scenario 1: Remove K elements from the beginning
  int max_diff1 = a[k-1] - a[0]; // max of the remaining elements - min of the remaining elements
  // Scenario 2: Remove K elements from the end
  int max_diff2 = a[n-k] - a[n-k-1];
  
  // The answer is the maximum of the differences from the two scenarios
  cout << max(max_diff1, max_diff2) << endl;
  return 0;
}
```
