```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<long long> v(n);
  for (int i = 0; i < n; i++) {
    cin >> v[i];
  }
  
  // Sort the array to easily find the smallest and largest elements after removal
  sort(v.begin(), v.end());
  
  // Initialize the result with the difference between the largest and smallest elements
  long long res = v[n - 1] - v[0];
  
  // Iterate through the sorted array and consider removing k elements
  for (int i = 0; i <= n - k; i++) {
    // Calculate the difference after removing k elements
    res = min(res, v[i + k - 1] - v[i]);
  }
  
  // Print the minimum difference found
  cout << res << endl;
  
  return 0;
}
```
