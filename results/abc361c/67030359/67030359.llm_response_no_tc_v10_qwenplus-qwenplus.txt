#include <bits/stdc++.h>
using namespace std;

#define all(x) x.begin(), x.end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We will use a sliding window of size N - K to find the minimal (max - min)
    // Since we are removing K elements, we must keep N-K elements in order
    int res = numeric_limits<int>::max();
    deque<int> minQ, maxQ;

    // Sliding window function
    auto removeOutWindow = [&](int left) {
        if (!minQ.empty() && minQ.front() == left) minQ.pop_front();
        if (!maxQ.empty() && maxQ.front() == left) maxQ.pop_front();
    };

    auto addInWindow = [&](int right) {
        while (!minQ.empty() && A[minQ.back()] >= A[right]) minQ.pop_back();
        while (!maxQ.empty() && A[maxQ.back()] <= A[right]) maxQ.pop_back();
        minQ.push_back(right);
        maxQ.push_back(right);
    };

    for (int i = 0; i < N; ++i) {
        removeOutWindow(i - (N - K));
        addInWindow(i);
        if (i >= N - K - 1) {
            res = min(res, A[maxQ.front()] - A[minQ.front()]);
        }
    }

    cout << res << endl;
    return 0;
}
