
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    // Edge case when K is 0, no elements are removed.
    if (K == 0) {
        cout << 0 << endl;
        return 0;
    }

    // Sort the array to find the minimum and maximum values easily.
    sort(A.begin(), A.end());

    // To minimize the difference between max and min, we should remove the largest K elements or the smallest K elements.
    // We need to consider both cases and take the minimum difference.

    // Case 1: Remove the largest K elements
    int maxAfterRemoval1 = A[N - K - 1]; // The maximum value in the remaining sequence
    int minAfterRemoval1 = A[0]; // The minimum value in the remaining sequence (which is A[N-K] after removal of K largest elements)

    // Case 2: Remove the smallest K elements
    int maxAfterRemoval2 = A[N - 1]; // The maximum value in the remaining sequence (which is A[N-K-1] after removal of K smallest elements)
    int minAfterRemoval2 = A[K]; // The minimum value in the remaining sequence

    // The minimum possible value of the maximum minus the minimum of the new sequence
    cout << min((maxAfterRemoval1 - minAfterRemoval1), (maxAfterRemoval2 - minAfterRemoval2)) << endl;

    return 0;
}
```
