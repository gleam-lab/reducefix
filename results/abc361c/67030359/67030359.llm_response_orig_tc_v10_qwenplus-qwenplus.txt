#include <bits/stdc++.h>
using namespace std;

#define all(x) x.begin(), x.end()
#define eb emplace_back
#define vi vector<int>

template<typename T> inline bool chmin(T &a, T b) {
  if (a > b) {
    a = b;
    return true;
  }
  return false;
}

int main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);

  int N, K;
  cin >> N >> K;
  vi A(N);
  for (int i = 0; i < N; ++i) {
    cin >> A[i];
  }

  // We need to choose N-K elements to keep such that max(B) - min(B) is minimized
  int window_size = N - K;
  sort(all(A));

  int min_diff = numeric_limits<int>::max();
  for (int i = 0; i + window_size - 1 < N; ++i) {
    chmin(min_diff, A[i + window_size - 1] - A[i]);
  }

  cout << min_diff << endl;
  return 0;
}
