#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    
    // Read the array elements
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    
    // Sort the array to make it easier to find the minimum and maximum values
    sort(A.begin(), A.end());
    
    // We need to remove K elements, so we consider removing elements from both ends
    // nums: the minimum and maximum values of B
    int nums[2][2] = {{A[0], A[N-1]}, {A[1], A[N-2]}};
    
    // Try removing up to K elements from both ends and calculate the min and max of B
    for (int x = 0; x <= K; ++x) {
        for (int y = 0; y <= K - x; ++y) {
            if (x + y <= K) {
                // Calculate the min and max of the new sequence B
                int minB = min(nums[0][0], A[x + y]);
                int maxB = max(nums[1][1], A[x + y]);
                
                // Update the global minimum of the difference between maxB and minB
                int diffB = maxB - minB;
                if (diffB < nums[1][0] - nums[0][0]) {
                    nums[0][0] = minB;
                    nums[1][0] = maxB;
                }
            }
        }
    }
    
    // Output the minimum possible value of the difference between the maximum and minimum of B
    cout << nums[1][0] - nums[0][0] << endl;
    
    return 0;
}
