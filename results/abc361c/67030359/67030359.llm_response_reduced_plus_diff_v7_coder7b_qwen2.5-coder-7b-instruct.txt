```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;

    vector<long long> A(N);
    for(int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    sort(begin(A), end(A));
    
    // The optimal strategy is to remove K smallest and K largest elements
    // This way, we minimize the difference between the maximum and minimum of the remaining elements
    long long result = A[K - 1] - A[N - K];

    cout << result << '\n';

    return 0;
}
```
