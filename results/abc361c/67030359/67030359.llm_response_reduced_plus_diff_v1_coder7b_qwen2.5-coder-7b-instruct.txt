```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int INF = 1e9 + 5;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n, k; cin >> n >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    // Sort the array to easily find min/max differences
    sort(a.begin(), a.end());

    // Calculate the difference between consecutive elements
    vector<ll> diff(n - 1);
    for(int i = 1; i < n; ++i) diff[i - 1] = a[i] - a[i - 1];

    // Sort the differences in ascending order
    sort(diff.begin(), diff.end());

    // Remove k smallest differences (largest elements will be left)
    for(int i = 0; i < k; ++i) {
        diff.pop_back();
    }

    // The result is the sum of remaining differences
    ll res = accumulate(diff.begin(), diff.end(), 0LL);

    cout << res << '\n';
    return 0;
}
```
