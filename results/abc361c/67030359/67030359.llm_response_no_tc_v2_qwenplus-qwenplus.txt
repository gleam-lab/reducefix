```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);

  int N, K;
  cin >> N >> K;
  vector<int> A(N);
  for (int i = 0; i < N; ++i) {
    cin >> A[i];
  }

  // We will try to keep (N - K) elements such that the min and max are as close as possible
  int window_size = N - K;
  
  // Sort the array to easily find contiguous windows of size (N-K)
  vector<pair<int, int>> sorted_A(N); // Pair: (value, original index)
  for (int i = 0; i < N; ++i) {
    sorted_A[i] = {A[i], i};
  }
  sort(sorted_A.begin(), sorted_A.end());

  // Sliding window on the sorted values to find a window of size (N-K)
  // where the positions in original array are valid (in order)
  int min_diff = numeric_limits<int>::max();
  for (int i = 0; i + window_size - 1 < N; ++i) {
    int j = i + window_size - 1;
    // Get all original indices in this window
    vector<int> indices;
    for (int k = i; k <= j; ++k) {
      indices.push_back(sorted_A[k].second);
    }
    // Check if these indices can be kept in the original order (i.e., appear in increasing order)
    vector<int> sorted_indices = indices;
    sort(sorted_indices.begin(), sorted_indices.end());
    
    // If the indices are already in order, we can form such a sequence
    if (sorted_indices == indices) {
      min_diff = min(min_diff, sorted_A[j].first - sorted_A[i].first);
    }
  }

  cout << min_diff << endl;
  return 0;
}
```
