```cpp
#include <bits/stdc++.h>

#define all(x) x.begin(), x.end()

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // We need to choose N-K elements to keep, minimizing max(B) - min(B)
    // Since we must preserve the order, but can skip up to K elements,
    // we cannot sort the array as that changes the order.
    // So this approach is fundamentally wrong:
    //   sort(A), then find window of size N-K with minimal diff

    // Correct idea: Use sliding window on sorted indices based on values
    // Maintain a window of N-K elements in original array, preserving order
    // To do this efficiently:
    // 1. Sort indices by their value
    // 2. Slide a window of size N-K over these indices
    // 3. For each window, check if the corresponding elements in A form a valid subsequence
    //    (i.e., appear in order)
    // 4. The minimum difference between max and min among all such windows is our answer

    vector<int> idx(N);
    iota(all(idx), 0);
    sort(idx.begin(), idx.end(), [&](int i, int j) { return A[i] < A[j]; });

    deque<int> dq; // Will store indices to maintain window in order
    ll res = numeric_limits<ll>::max();
    int left = 0;

    // Sliding window on sorted indices by value
    for (int right = 0; right < N; ++right) {
        // Insert idx[right] into deque maintaining increasing order of positions
        while (!dq.empty() && dq.back() > idx[right]) dq.pop_back();
        dq.push_back(idx[right]);

        // Window size == N-K ?
        if (right - left + 1 == N - K) {
            res = min(res, (ll)(A[idx[right]] - A[idx[left]]));
            // Remove element at left from deque if needed
            if (dq.front() == idx[left]) dq.pop_front();
            ++left;
        }
    }

    cout << res << endl;
    return 0;
}
```
