#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'S'}, {'S', 'P'}, {'P', 'R'}}; // Corrected: R beats S, S beats P, P beats R
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 1) {
        cout << 1 << endl;
        return 0;
    }

    // Strategy 1: Start by playing the move that beats the first opponent's move
    int total_1 = 1;
    char last_move = beats[s[0]]; // Our first move beats opponent's first move
    for (int i = 1; i < n; i++) {
        // If our previous move beats opponent's current move, we can repeat the same strategy
        if (beats[last_move] == s[i]) {
            // We continue with the same winning pattern
            continue;
        } else {
            // We need to change our move to beat the current opponent's move
            last_move = beats[s[i]];
            total_1++;
        }
    }

    // Strategy 2: Start by playing the same as opponent's first move
    int total_2 = 1;
    last_move = s[0]; // Our first move is same as opponent's first move
    for (int i = 1; i < n; i++) {
        if (beats[last_move] == s[i]) {
            continue;
        } else {
            last_move = beats[s[i]];
            total_2++;
        }
    }

    cout << max(total_1, total_2) << endl;

    return 0;
}
