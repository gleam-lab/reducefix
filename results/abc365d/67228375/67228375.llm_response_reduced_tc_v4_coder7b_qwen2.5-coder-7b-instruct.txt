```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"

int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int win_count = 0;
    char last_move = beats[s[n-1]];
    for (int i = n-1; i >= 0; i--) {
        if (s[i] != last_move) {
            win_count++;
            last_move = s[i];
        }
    }
    cout << win_count << endl;

    return 0;

}
```
