#include <bits/stdc++.h>
using namespace std;
#define endl "\n"

int main() {

    ios_base::sync_with_stdio(false);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int total_1 = 1;
    char last_move = s[0];
    for (int i = 1; i < n; i++) {
        if (s[i] != last_move) {
            last_move = s[i];
            total_1 += 1;
        }
    }
    int total_2 = 1;
    last_move = s[n-1];
    for (int i = n-2; i >= 0; i--) {
        if (s[i] != last_move) {
            last_move = s[i];
            total_2 += 1;
        }
    }
    cout << max(total_1, total_2) << endl;

    return 0;

}
