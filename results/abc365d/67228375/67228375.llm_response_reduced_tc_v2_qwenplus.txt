```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    // Try all possible starting moves for the player
    int max_rounds = 1;
    
    // We'll try every possible initial move that can beat the first opponent move
    vector<char> possible_moves = {'R', 'P', 'S'};
    
    for (char start_move : possible_moves) {
        // If this move doesn't beat the first opponent move, skip it
        if (beats[s[0]] != start_move) continue;
        
        int rounds = 1;
        char last_player_move = start_move;
        
        for (int i = 1; i < n; i++) {
            // If our last move beats current opponent move, we win and repeat same move
            if (beats[s[i]] == last_player_move) {
                // Continue with same move
                continue;
            }
            // Otherwise, we need to change our move to beat current opponent
            else {
                last_player_move = beats[s[i]];
                rounds++;
            }
        }
        max_rounds = max(max_rounds, rounds);
    }
    
    // Also consider the case where we might start with a move that loses or ties
    // Actually, we must start by winning the first round, so only moves that beat s[0] are valid
    
    cout << max_rounds << endl;

    return 0;
}
```
