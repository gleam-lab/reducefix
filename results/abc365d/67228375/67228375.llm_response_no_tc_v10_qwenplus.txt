#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'S'}, {'P', 'R'}, {'S', 'P'}}; // Correct mapping: what each move beats
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 1) {
        cout << 1 << endl;
        return 0;
    }

    // Strategy 1: Start by playing the move that beats the first opponent's move
    int total_1 = 1;
    char my_move = beats[s[0]]; // My first move beats opponent's first move
    for (int i = 1; i < n; i++) {
        if (beats[my_move] == s[i]) {
            // I win against opponent's current move, continue with same move
            continue;
        } else {
            // I don't win, so I need to change to beat the current move
            my_move = beats[s[i]];
            total_1++;
        }
    }

    // Strategy 2: Start by playing the same as opponent's first move
    int total_2 = 1;
    my_move = s[0]; // My first move is same as opponent's first move
    for (int i = 1; i < n; i++) {
        if (beats[my_move] == s[i]) {
            // I win against opponent's current move, continue with same move
            continue;
        } else {
            // I don't win, so I need to change to beat the current move
            my_move = beats[s[i]];
            total_2++;
        }
    }

    cout << max(total_1, total_2) << endl;

    return 0;
}
