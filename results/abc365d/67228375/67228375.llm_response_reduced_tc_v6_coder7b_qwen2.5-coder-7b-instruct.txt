```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    // Calculate the maximum number of consecutive moves where each move is beaten by the next one
    int max_beaten_moves = 1;
    int current_beaten_moves = 1;
    for (int i = 1; i < n; i++) {
        if (beats[s[i]] == s[i-1]) {
            current_beaten_moves++;
        } else {
            current_beaten_moves = 1;
        }
        max_beaten_moves = max(max_beaten_moves, current_beaten_moves);
    }

    cout << max_beaten_moves << endl;

    return 0;
}
```
