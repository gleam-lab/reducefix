```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    // Initialize variables to store the maximum scores for both players
    int score1 = 0, score2 = 0;
    
    // Iterate through the string to calculate scores
    for (int i = 0; i < n; ++i) {
        if (s[i] == beats[s[(i + 1) % n]]) {
            score1++;
        } else {
            score2++;
        }
    }
    
    // Output the maximum score
    cout << max(score1, score2) << endl;

    return 0;
}
```
