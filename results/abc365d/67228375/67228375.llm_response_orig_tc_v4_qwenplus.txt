#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    // Try both possible starting moves
    auto countWinStreaks = [&](char firstMove) {
        int streaks = 1;
        char lastOpponentMove = s[0];
        char currentWinner = (beats[lastOpponentMove] == firstMove) ? firstMove : s[0];

        for (int i = 1; i < n; i++) {
            char opponentMove = s[i];
            char predictedWinner = (beats[opponentMove] == currentWinner) ? currentWinner : opponentMove;
            
            if (predictedWinner != currentWinner) {
                streaks++;
                currentWinner = predictedWinner;
            }
            lastOpponentMove = opponentMove;
        }
        return streaks;
    };

    // Try starting with beating the first move and starting with the first move itself
    char beatFirst = beats[s[0]];
    int option1 = countWinStreaks(beatFirst);
    int option2 = countWinStreaks(s[0]);
    
    cout << max(option1, option2) << endl;

    return 0;
}
