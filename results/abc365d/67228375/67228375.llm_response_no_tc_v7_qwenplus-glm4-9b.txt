#include <iostream>
#include <string>
#include <unordered_map>

using namespace std;
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    unordered_map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    // Initialize counts for each possible outcome
    int countR = 0, countP = 0, countS = 0;
    char currentPattern = s[0];
    countR = (currentPattern == 'R') ? 1 : 0;
    countP = (currentPattern == 'P') ? 1 : 0;
    countS = (currentPattern == 'S') ? 1 : 0;

    // Count occurrences of each character
    for (char c : s) {
        if (beats[c] == 'R') {
            countR++;
        } else if (beats[c] == 'P') {
            countP++;
        } else if (beats[c] == 'S') {
            countS++;
        }
    }

    // Output the maximum count
    cout << max({countR, countP, countS}) << endl;

    return 0;
}
