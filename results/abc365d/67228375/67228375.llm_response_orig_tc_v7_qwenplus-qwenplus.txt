```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    // Strategy: We simulate the optimal strategy in both directions
    // dp[i] = maximum wins up to position i using greedy strategy

    // Two strategies:
    // 1. Start by countering first move
    // 2. Start with the same move as first character

    auto calculate_max = [&](bool counter_start) {
        int total = 0;
        char last_move = counter_start ? beats[s[0]] : s[0];
        if (counter_start || last_move == s[0]) total++;  // we always count first move

        for (int i = 1; i < n; ++i) {
            char next_counter = beats[s[i]];
            if (next_counter == last_move) {
                // Can continue without changing
                last_move = s[i];  // opponent's move becomes our new move
            } else {
                // Need to change move
                last_move = next_counter;
                total++;
            }
        }
        return total;
    };

    int res1 = calculate_max(true);
    int res2 = calculate_max(false);

    cout << max(res1, res2) << endl;

    return 0;
}
```
