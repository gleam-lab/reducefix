#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007


int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    int total_1 = 0;
    char last_move = 'Z'; // Make sure it's not a valid move
    for (int i = 0; i < n; i++) {
        char current_move = beats[s[i]];
        if (current_move != last_move) {
            total_1++;
            last_move = current_move;
        }
    }
    
    int total_2 = 0;
    last_move = 'Z'; // Reset for second calculation
    for (int i = 0; i < n; i++) {
        char current_move = beats[s[(i + 1) % n]]; // Start from the next character and wrap around
        if (current_move != last_move) {
            total_2++;
            last_move = current_move;
        }
    }
    
    cout << max(total_1, total_2) << endl;

    return 0;

}
