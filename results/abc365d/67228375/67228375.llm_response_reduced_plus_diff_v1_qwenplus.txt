#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 1) {
        cout << 1 << endl;
        return 0;
    }

    // dp[i][c] = maximum number of rounds we can win ending with move c at position i
    vector<map<char, int>> dp(n);
    
    // Base case: first round
    dp[0]['R'] = 1;
    dp[0]['P'] = 1;
    dp[0]['S'] = 1;

    for (int i = 1; i < n; i++) {
        char opponent_move = s[i];
        char winning_move = beats[opponent_move];
        
        // For each possible current move
        for (char my_move : {'R', 'P', 'S'}) {
            int wins = 0;
            if (my_move == winning_move) {
                wins = 1;
            }
            
            // Transition from previous state
            int best_prev = 0;
            for (auto &[prev_move, prev_wins] : dp[i-1]) {
                best_prev = max(best_prev, prev_wins);
            }
            
            dp[i][my_move] = best_prev + wins;
        }
    }

    int result = 0;
    for (auto &[move, wins] : dp[n-1]) {
        result = max(result, wins);
    }
    
    cout << result << endl;

    return 0;
}
