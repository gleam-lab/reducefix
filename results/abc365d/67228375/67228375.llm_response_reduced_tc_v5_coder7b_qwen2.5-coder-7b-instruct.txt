```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int wins = 0;
    char last_move = ' ';
    for (int i = 0; i < n; i++) {
        if (last_move != ' ' && beats[last_move] == s[i]) {
            // If the last move beats the current move, skip this move
            continue;
        } else {
            // Otherwise, play the move that beats the current move
            last_move = beats[s[i]];
            wins++;
        }
    }
    cout << wins << endl;

    return 0;
}
```
