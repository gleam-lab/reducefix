#include <iostream>
#include <map>
#include <string>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {'R': 'P', 'P': 'S', 'S': 'R'};
    int n;
    string s;
    cin >> n >> s;
    int total_1 = 0;
    char last_move = tolower(s[0]); // Convert to lowercase to handle case insensitivity

    // First pass to count the maximum beats for the first character
    for (int i = 0; i < n; ++i) {
        if (beats[s[i]] == last_move) {
            total_1++;
        }
    }

    // Reset last_move for the second pass
    last_move = tolower(s[0]);

    // Second pass to count the maximum beats for the last character
    for (int i = 1; i < n; ++i) {
        if (beats[s[i]] == last_move) {
            last_move = tolower(s[i]);
            total_1++;
        }
    }

    cout << total_1 << endl;

    return 0;
}
