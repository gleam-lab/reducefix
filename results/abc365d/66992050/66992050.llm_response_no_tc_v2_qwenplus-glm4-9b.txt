
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    int ans = 0;
    char prevState = 'N';
    
    for (char c : S) {
        if (c == 'P' || c == 'R' || c == 'S') {
            if (prevState != c) {
                ans++;
                prevState = c;
            }
        } else {
            // In case the string contains characters other than 'P', 'R', 'S'
            prevState = c;
        }
    }
    
    cout << ans;
    return 0;
}
```
