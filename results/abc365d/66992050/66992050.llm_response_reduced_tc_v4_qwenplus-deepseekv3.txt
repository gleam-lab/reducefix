#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {

    char past = 'N';
    int ans = 0;
    for (char c : S) {
        if (c == 'P' && past == 'R') {
            ans++;
        } else if (c == 'R' && past == 'S') {
            ans++;

}
