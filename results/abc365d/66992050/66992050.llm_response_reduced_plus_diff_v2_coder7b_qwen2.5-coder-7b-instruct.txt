```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string S;
    cin >> N >> S;

    vector<int> last(3, -1); // To store the last occurrence of 'P', 'R', and 'S'
    int count = 0; // Count of valid sequences

    for (int i = 0; i < N; ++i) {
        char c = S[i];
        if (c == 'P') {
            count += i - max(last[1], last[2]);
        } else if (c == 'R') {
            count += i - max(last[0], last[2]);
        } else { // c == 'S'
            count += i - max(last[0], last[1]);
        }
        last[c - 'P'] = i;
    }

    cout << count << endl;
    return 0;
}
```
