#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  char last_move = 'N'; // Initial state, no move yet
  int changes = 0;

  for (char c : S) {
    if (c == 'P') { // Paper
      if (last_move != 'S') {
        changes++;
        last_move = 'S'; // Beats Paper, so we use Scissors
      }
    } else if (c == 'R') { // Rock
      if (last_move != 'P') {
        changes++;
        last_move = 'P'; // Beats Rock, so we use Paper
      }
    } else if (c == 'S') { // Scissors
      if (last_move != 'R') {
        changes++;
        last_move = 'R'; // Beats Scissors, so we use Rock
      }
    }
  }

  cout << changes << endl;
  return 0;
}
