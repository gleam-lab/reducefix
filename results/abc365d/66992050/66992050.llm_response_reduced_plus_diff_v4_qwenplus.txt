```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    // We need to simulate the process of replacing consecutive identical characters
    // The key insight: we want to minimize the number of operations to make all characters equal
    // Each operation can replace a maximal contiguous segment of one character with another
    
    // We can think greedily: we want to find the minimum number of changes needed
    // This is equivalent to counting the number of "segments" minus 1, but we can choose which character to keep
    
    // Alternate approach: use dynamic programming
    // dp[i][c] = minimum operations needed for prefix i ending with character c
    // where c = 0 for 'P', 1 for 'R', 2 for 'S'
    
    if (N == 0) {
        cout << 0;
        return 0;
    }
    
    vector<vector<int>> dp(N, vector<int>(3, 1e9));
    
    // Initialize first character
    for (int c = 0; c < 3; c++) {
        dp[0][c] = (S[0] != "PRS"[c]);
    }
    
    for (int i = 1; i < N; i++) {
        for (int curr = 0; curr < 3; curr++) {
            char curr_char = "PRS"[curr];
            int cost = (S[i] != curr_char);
            
            // We can extend any previous state
            for (int prev = 0; prev < 3; prev++) {
                dp[i][curr] = min(dp[i][curr], dp[i-1][prev] + cost);
            }
            
            // But if current character matches previous, we might not need extra operation
            // Actually, our formulation already handles this correctly
        }
    }
    
    int ans = *min_element(dp[N-1].begin(), dp[N-1].end());
    cout << ans;
    
    return 0;
}
```
