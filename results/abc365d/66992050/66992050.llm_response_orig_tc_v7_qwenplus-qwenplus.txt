#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  // This problem can be modeled as finding the maximum number of non-overlapping turns
  // that can be taken according to the rules of the game.

  // We'll use a greedy approach:
  // - Track the cooldown period after each turn (2 seconds)
  // - Only take a turn when allowed and it's the specified action

  // The key is that after using an action, we must wait 2 more steps before we can act again.
  // So we track the earliest time we can act next.

  // 'past' keeps track of when we can act next
  int past = -3; // start with being able to act immediately
  int ans = 0;
  
  for (int i = 0; i < N; i++) {
    char c = S[i];
    
    // Check if we're past the cooldown period
    if (i - past >= 3) {
      // We can choose this action and update our cooldown
      ans++;
      past = i;
    }
  }
  
  cout << ans << endl;
  return 0;
}
