```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  // This problem can be viewed as finding the minimal number of sequences such that
  // each sequence follows the RPS cycle rule: R -> P -> S -> R...

  // We'll use a greedy approach to assign minimal number of sequences.

  // last_occurrence['R'] = last index where 'R' was seen
  // similarly for 'P' and 'S'
  unordered_map<char, int> last_occurrence;

  // Initialize all to -1 (not seen yet)
  last_occurrence['R'] = -1;
  last_occurrence['P'] = -1;
  last_occurrence['S'] = -1;

  // answer will be the maximum number of sequences needed at any point
  int ans = 0;

  // current sequence index
  int curr_seq = 0;

  for (char c : S) {
    // Determine which character should precede current one in RPS cycle
    char prev_char;
    if (c == 'R') prev_char = 'S';
    else if (c == 'P') prev_char = 'R';
    else if (c == 'S') prev_char = 'P';

    // If the previous character in cycle hasn't occurred or occurred before current position,
    // we can reuse the same sequence
    if (last_occurrence[prev_char] == -1) {
      // Just add this character to current sequence
      last_occurrence[c] = curr_seq;
    } else {
      // Need a new sequence
      curr_seq++;
      last_occurrence[c] = curr_seq;

      // Update max sequences used
      ans = max(ans, curr_seq + 1);
    }
  }

  cout << ans << endl;
  return 0;
}
```
