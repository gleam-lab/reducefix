#include <iostream>
#include <string>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    int ans = 1; // Start with 1 because the first character always starts a new sequence
    char past = S[0]; // Initialize the past character with the first character of the string
    
    for (size_t i = 1; i < S.length(); ++i) {
        if (S[i] == past) {
            continue; // Skip if the current character is the same as the past one
        }
        
        if (S[i] == 'P' && past == 'S') {
            ans++; // Increment the answer if it's a new pair
        } else if (S[i] == 'R' && past == 'P') {
            ans++; // Increment the answer if it's a new pair
        } else if (S[i] == 'S' && past == 'R') {
            ans++; // Increment the answer if it's a new pair
        }
        
        past = S[i]; // Update the past character
    }
    
    cout << ans;
    return 0;
}
