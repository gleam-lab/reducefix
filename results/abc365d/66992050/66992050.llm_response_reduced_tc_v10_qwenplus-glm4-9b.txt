
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    int ans = 0;
    char past = 'N';
    
    for (char c : S) {
        if (c == 'P' || c == 'R' || c == 'S') {
            if (past == 'N') {
                past = c;
                ans++;
            } else if (past == c) {
                past = 'N';
            } else {
                past = c;
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
