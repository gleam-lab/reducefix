        if (current == 'R' && past == 'S') {
            ans++;
            past = current;
        } else if (current == 'S' && past == 'P') {
            ans++;
            past = current;
        } else if (current == 'P' && past == 'R') {

        } else {

        }
    }
    cout << ans;
    return 0;
}

2. **Loop Through Characters**: The loop starts from the second character (index 1) and compares each character with the previous one (`past`).
3. **Rock-Paper-Scissors Rules**: The code checks if the current character beats `past` according to the rules:
