```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    char last = 'N'; // Last move played by Alice
    int ans = 0;
    
    for (char c : S) {
        if (c == 'P') {
            if (last != 'S') {
                ans++;
                last = 'S'; // To beat P, play S
            } else {
                last = 'P'; // Can't win, so play P
            }
        } else if (c == 'R') {
            if (last != 'P') {
                ans++;
                last = 'P'; // To beat R, play P
            } else {
                last = 'R'; // Can't win, so play R
            }
        } else if (c == 'S') {
            if (last != 'R') {
                ans++;
                last = 'R'; // To beat S, play R
            } else {
                last = 'S'; // Can't win, so play S
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
