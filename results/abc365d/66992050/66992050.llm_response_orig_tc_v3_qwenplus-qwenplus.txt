```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  char last_move = 'N'; // None initially
  int count = 0;

  for (char c : S) {
    if (c == 'R') { // Rock beats Scissors
      if (last_move != 'P') {
        count++;
        last_move = 'P'; // Played Paper after Rock
      } else {
        last_move = 'R';
      }
    } else if (c == 'P') { // Paper beats Rock
      if (last_move != 'S') {
        count++;
        last_move = 'S'; // Played Scissors after Paper
      } else {
        last_move = 'P';
      }
    } else if (c == 'S') { // Scissors beats Paper
      if (last_move != 'R') {
        count++;
        last_move = 'R'; // Played Rock after Scissors
      } else {
        last_move = 'S';
      }
    }
  }

  cout << count << endl;
  return 0;
}
```
