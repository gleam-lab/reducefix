```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  // Mapping of move to its defeating move: R -> S, S -> P, P -> R
  unordered_map<char, char> defeats = {
      {'R', 'S'},
      {'S', 'P'},
      {'P', 'R'}
  };

  char last_move = 'N'; // Initially no move taken
  int ans = 0;

  for (char c : S) {
    if (last_move == 'N' || last_move == defeats[c]) {
      // We can take this move to defeat the opponent or start fresh
      ans++;
      last_move = c;
    } else {
      // Cannot use same or ineffective move, skip
      continue;
    }
  }

  cout << ans << endl;
  return 0;
}
```
