```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string S;
    cin >> N >> S;

    vector<int> ps(4), rs(4);
    for (char c : S) {
        ++ps[c - 'P'];
        ++rs[c - 'R'];
    }

    int ans = 0;
    for (char c : S) {
        int p = ps[c - 'P'], r = rs[c - 'R'];
        if (c == 'P') {
            if (r > 0) {
                --rs['R' - 'R'];
                ++ans;
            }
        } else if (c == 'R') {
            if (p > 0) {
                --ps['P' - 'P'];
                ++ans;
            }
        } else {
            if (min(p, r) > 0) {
                --ps['P' - 'P'];
                --rs['R' - 'R'];
                ++ans;
            }
        }
        ++ps[c - 'P'];
        ++rs[c - 'R'];
    }

    cout << ans << '\n';

    return 0;
}
```
