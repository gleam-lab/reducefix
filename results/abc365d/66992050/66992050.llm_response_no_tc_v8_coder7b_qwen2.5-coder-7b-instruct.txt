```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
int main(){
  int N;
  string S;
  cin >> N >> S;
  char past = '.';
  int ans = 0;
  for(char c : S) {
    if(past == '.') { // Initial case
      past = c;
      continue;
    }
    if((c == 'P' && past != 'S') || (c == 'R' && past != 'P') || (c == 'S' && past != 'R')) {
      ans++;
      past = c;
    } else {
      past = c;
    }
  }
  cout << ans;
  return 0;
}
```
