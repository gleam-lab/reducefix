```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<char> S(N);
    for(int i = 0; i < N; ++i) {
        cin >> S[i];
    }

    int ans = 0;
    unordered_map<char, char> nextMove = {{'P', 'R'}, {'R', 'S'}, {'S', 'P'}};
    unordered_set<char> visited;
    
    for(int i = 0; i < N; ++i) {
        if(visited.count(S[i])) continue;
        
        int j = i + 1;
        while(j < N && S[j] != nextMove[S[i]]) {
            ++j;
        }
        
        if(j == N) break;
        
        visited.insert(nextMove[S[i]]);
        ++ans;
    }

    cout << ans << '\n';
    return 0;
}
```
