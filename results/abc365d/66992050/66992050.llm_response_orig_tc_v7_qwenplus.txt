#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    // We want to minimize the number of changes such that no two adjacent characters are the same.
    // Actually, we are counting the minimum number of changes needed so that S becomes alternating in a cyclic way: P->R->S->P...
    // But note: the problem is equivalent to finding the minimal replacements so that the string has no adjacent equal characters.
    // However, from the logic, it seems we are simulating a state machine where we expect a cycle: P -> R -> S -> P...
    // But the original code has a flawed logic.

    // Let's rethink:
    // The intended behavior appears to be: we traverse the string and whenever the current character doesn't follow
    // the expected cycle (P after S, R after P, S after R), we "fix" it by counting a change and updating the past accordingly.
    // But the original condition is inverted and flawed.

    // Correct idea:
    // We maintain the last character we have (after possible replacement).
    // For each new character c:
    // - If c follows the cycle after 'past', then we don't need to change it. Update past = c.
    // - Else, we must change c to the next in the cycle after 'past'. Then set past to that next character.

    // The cycle: 
    //   after 'P' comes 'R'
    //   after 'R' comes 'S'
    //   after 'S' comes 'P'

    char past = 'N'; // none at start
    int ans = 0;

    for (char c : S) {
        if (past == 'N') {
            // first character: no change needed, just set past to c
            past = c;
        } else {
            char expected = ' ';
            if (past == 'P') expected = 'R';
            else if (past == 'R') expected = 'S';
            else if (past == 'S') expected = 'P';

            if (c == expected) {
                // no change needed
                past = c;
            } else {
                // we need to change c to expected
                ans++;
                past = expected; // because we changed it
            }
        }
    }

    cout << ans;
    return 0;
}
