#include <bits/stdc++.h>
using namespace std;
using ll = long long;
int main(){
  int N;
  string S;
  cin >> N >> S;
  char past = 'N';
  int ans = 0;
  for(char c:S){
    if(past == 'N' || c == 'P' || c == 'R' || c == 'S') {
      past = c;
      ans++;
    }
  }
  cout << ans - 1;
  return 0;
}
