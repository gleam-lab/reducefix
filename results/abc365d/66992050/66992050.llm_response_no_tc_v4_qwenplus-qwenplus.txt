```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  char last_move = 'N'; // N for None (before the first move)
  int win_count = 0;

  for (char c : S) {
    if (c == 'R') { // Rock
      if (last_move != 'P') {
        win_count++;
        last_move = 'P'; // Paper beats Rock
      } else {
        last_move = 'R';
      }
    } else if (c == 'S') { // Scissors
      if (last_move != 'R') {
        win_count++;
        last_move = 'R'; // Rock beats Scissors
      } else {
        last_move = 'S';
      }
    } else if (c == 'P') { // Paper
      if (last_move != 'S') {
        win_count++;
        last_move = 'S'; // Scissors beats Paper
      } else {
        last_move = 'P';
      }
    }
  }

  cout << win_count << endl;
  return 0;
}
```
