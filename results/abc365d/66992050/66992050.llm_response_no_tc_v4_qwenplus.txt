#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    char past = 'N'; // Track the last gesture we used
    int ans = 0;
    
    for(char c : S){
        if(c == 'P'){
            if(past != 'S'){
                ans++;
                past = 'S'; // To beat P, we use S
            }
            else{
                past = 'P'; // Can't use S again, so use P (but this loses)
            }
        }
        else if(c == 'R'){
            if(past != 'P'){
                ans++;
                past = 'P'; // To beat R, we use P
            }
            else{
                past = 'R'; // Can't use P again, so use R (but this loses)
            }
        }
        else if(c == 'S'){
            if(past != 'R'){
                ans++;
                past = 'R'; // To beat S, we use R
            }
            else{
                past = 'S'; // Can't use R again, so use S (but this loses)
            }
        }
    }
    
    cout << ans;
    return 0;
}
