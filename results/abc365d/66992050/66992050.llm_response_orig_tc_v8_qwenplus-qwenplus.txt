```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  // Each state transition forces a change, so we can simulate optimally by always choosing the move that avoids repetition.
  char last_move = 'N'; // initial state, no previous move
  int ans = 0;

  for (char c : S) {
    if (c == 'R') {
      if (last_move != 'P') {
        ans++;
        last_move = 'P'; // R beats P
      } else {
        last_move = 'R';
      }
    } else if (c == 'S') {
      if (last_move != 'R') {
        ans++;
        last_move = 'R'; // S beats R
      } else {
        last_move = 'S';
      }
    } else if (c == 'P') {
      if (last_move != 'S') {
        ans++;
        last_move = 'S'; // P beats S
      } else {
        last_move = 'P';
      }
    }
  }

  cout << ans << endl;
  return 0;
}
```
