```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;
  
  // Initialize answer and past character tracker
  int ans = 0;
  char past = 'N';  // Initial state, no previous move

  for (char c : S) {
    if (c == 'R') {
      // R beats S, so if previous wasn't R, we win
      if (past != 'R') {
        ans++;
        past = 'R';
      }
    } else if (c == 'P') {
      // P beats R, so if previous wasn't P, we win
      if (past != 'P') {
        ans++;
        past = 'P';
      }
    } else if (c == 'S') {
      // S beats P, so if previous wasn't S, we win
      if (past != 'S') {
        ans++;
        past = 'S';
      }
    }
  }

  cout << ans;
  return 0;
}
```
