```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    char last_move = 'N'; // Last move played by the observer
    int ans = 0;
    
    for (char c : S) {
        if (c == 'P') {
            // To beat Paper, we need Scissors
            if (last_move != 'S') {
                ans++;
                last_move = 'S';
            } else {
                last_move = 'P';
            }
        } else if (c == 'R') {
            // To beat Rock, we need Paper
            if (last_move != 'P') {
                ans++;
                last_move = 'P';
            } else {
                last_move = 'R';
            }
        } else if (c == 'S') {
            // To beat Scissors, we need Rock
            if (last_move != 'R') {
                ans++;
                last_move = 'R';
            } else {
                last_move = 'S';
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
