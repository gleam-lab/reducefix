```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
  int N;
  string S;
  cin >> N >> S;
  
  // Initialize past as a character that doesn't match any of R, P, S
  char past = 'X';
  int ans = 0;
  
  for(char c : S){
    if(c == 'P'){
      if(past != 'S'){
        ans++;
        past = 'P';
      } else {
        past = 'S';
      }
    } 
    else if(c == 'R'){
      if(past != 'P'){
        ans++;
        past = 'R';
      } else {
        past = 'P';
      }
    } 
    else if(c == 'S'){
      if(past != 'R'){
        ans++;
        past = 'S';
      } else {
        past = 'R';
      }
    }
  }
  
  cout << ans << endl;
  return 0;
}
```
