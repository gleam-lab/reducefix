```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int N;
  string S;
  cin >> N >> S;

  char last_move = 'N'; // N as neutral, not part of RPS
  int wins = 0;

  for (char c : S) {
    if (c == 'R') {
      if (last_move != 'P') {
        wins++;
        last_move = 'P'; // Win by playing P against R
      } else {
        last_move = 'R'; // Tie, play R
      }
    } else if (c == 'P') {
      if (last_move != 'S') {
        wins++;
        last_move = 'S'; // Win by playing S against P
      } else {
        last_move = 'P'; // Tie, play P
      }
    } else if (c == 'S') {
      if (last_move != 'R') {
        wins++;
        last_move = 'R'; // Win by playing R against S
      } else {
        last_move = 'S'; // Tie, play S
      }
    }
  }

  cout << wins << endl;
  return 0;
}
```
