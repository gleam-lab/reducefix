Here's the fixed code that correctly counts the number of times a character beats the previous one according to Rock-Paper-Scissors rules:

```cpp
#include <bits/stdc++.h>

using ll = long long;

    string S;
    cin >> N >> S;
    
    if (N == 0) {
        cout << 0;
        return 0;
    }
    
    char past = S[0];
    int ans = 0;
    
    for (int i = 1; i < N; i++) {
        char current = S[i];
        if (current == 'P' && past == 'R') {
            ans++;
            past = current;
        } else if (current == 'R' && past == 'S') {
            ans++;
            past = current;
        } else if (current == 'S' && past == 'P') {
            ans++;
            past = current;
        } else {
            past = current;
        }
    }
    
    cout << ans;
    return 0;
}
```

1. Initialize `past` with the first character of the string instead of 'N'
2. Start the loop from the second character (index 1)
3. Only increment the counter when the current character beats the previous one according to Rock-Paper-Scissors rules
4. Always update `past` to the current character after each comparison
