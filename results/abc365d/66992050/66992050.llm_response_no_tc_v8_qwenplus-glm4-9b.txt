#include <iostream>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    int ans = 0;
    char past = 'N'; // Initialize to 'N' to handle the first character properly without extra conditions.
    
    for (char c : S) {
        if (c == 'P') {
            if (past == 'S' || past == 'N') { // Only increment if we are not from 'P' or still neutral.
                ans++;
                past = 'P';
            } else {
                past = 'P';
            }
        } else if (c == 'R') {
            if (past == 'P' || past == 'N') { // Only increment if we are not from 'R' or still neutral.
                ans++;
                past = 'R';
            } else {
                past = 'R';
            }
        } else if (c == 'S') {
            if (past == 'R' || past == 'N') { // Only increment if we are not from 'S' or still neutral.
                ans++;
                past = 'S';
            } else {
                past = 'S';
            }
        }
    }
    
    cout << ans;
    return 0;
}
