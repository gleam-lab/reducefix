#include <iostream>
#include <string>
using namespace std;

int countNeededPatches(int N, string S) {
    char previous = 'N'; // Initialize to 'N' as no character has been processed yet
    int patchesNeeded = 0;

    for (char c : S) {
        if (c == 'P' || c == 'R' || c == 'S') { // Only consider patches, ignore 'N'
            if (previous == 'N') {
                // If the last character was neither 'P', 'R', nor 'S', reset to current patch
                previous = c;
            } else {
                // If the last character was the same as the current one, increment patches needed
                if (previous == c) {
                    patchesNeeded++;
                    previous = 'N'; // Reset to 'N' until a new patch character is encountered
                } else {
                    // If the last character was a different patch, update the previous to the current character
                    previous = c;
                }
            }
        }
    }

    return patchesNeeded;
}

int main() {
    int N;
    string S;
    cin >> N >> S;
    cout << countNeededPatches(N, S);
    return 0;
}
