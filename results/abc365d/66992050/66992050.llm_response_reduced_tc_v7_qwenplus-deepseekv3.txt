Here's the fixed code that properly counts the number of wins in a Rock-Paper-Scissors game sequence:

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    if (N == 0) {
        cout << 0;
        return 0;
    }
    
    int ans = 0;
    char prev = S[0];
    
    for (int i = 1; i < N; i++) {
        char current = S[i];
        // Check if current move beats previous move
        if ((prev == 'R' && current == 'P') ||
            (prev == 'P' && current == 'S') ||
            (prev == 'S' && current == 'R')) {
            ans++;
            prev = current; // Only update when we win
        }
        // If same move, continue the streak
        else if (prev == current) {
            continue;
        }
        // If lost, reset the streak
        else {
            prev = current;
        }
    }
    
    cout << ans;
    return 0;
}
```

1. Changed the logic to compare each move with the previous move in the sequence
2. Only increment the counter when the current move beats the previous move (Rock-Paper-Scissors rules)
3. Reset the streak when losing, but continue it when moves are the same
4. Added handling for empty input (though the problem guarantees N ≥ 1)
5. The initial move is considered as the first element in the sequence
