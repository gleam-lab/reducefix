                wins++;
                lastMove = 'P';

                lastMove = 'S';
            }

            if(lastMove != 'S') { // Takahashi plays Scissors

                lastMove = 'S';
            } else { // Takahashi plays Rock
                lastMove = 'R';
            }
        } else if(currentMove == 'S') {

                wins++;
                lastMove = 'R';
            } else { // Takahashi plays Paper
                lastMove = 'P';
            }
        }
    }
    
    // Output the result

    return 0;
}
