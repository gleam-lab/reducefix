#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    // We want to minimize the number of changes such that no two adjacent characters are the same.
    // This is equivalent to counting how many times we need to change a character so that 
    // S[i] != S[i-1] for all i >= 1.
    
    if (N == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    int ans = 0;
    char prev = S[0];
    
    for (int i = 1; i < N; i++) {
        if (S[i] == prev) {
            // Need to change S[i] to something different from both prev and (if possible) next
            ans++;
            // After changing S[i], the new "prev" should be a character different from current S[i]
            // But we don't actually modify S, so we just update prev to any character different from S[i]
            // However, we must ensure that the next comparison works correctly.
            // The key insight: after a change, the effective previous character becomes one that is not S[i] nor the original next.
            // But since we can choose optimally, we pick a letter different from S[i] and (if i+1 exists) different from S[i+1].
            // However, there are 3 letters: 'R', 'P', 'S'
            // So we can always pick one that's different from S[i] and possibly from S[i+1].
            
            // But note: we only care about the state for the next iteration.
            // If we changed S[i], then the new value at position i is some c != S[i].
            // To maximize flexibility, we would choose c such that c != S[i] and c != S[i+1] (if i+1 exists).
            // That way, we avoid creating a new conflict.
            
            // So:
            if (i+1 < N) {
                // Choose a character different from S[i] and S[i+1]
                for (char c : {'R', 'P', 'S'}) {
                    if (c != S[i] && c != S[i+1]) {
                        prev = c;
                        break;
                    }
                }
            } else {
                // Last character, just choose anything different from S[i]
                for (char c : {'R', 'P', 'S'}) {
                    if (c != S[i]) {
                        prev = c;
                        break;
                    }
                }
            }
        } else {
            prev = S[i];
        }
    }
    
    cout << ans << endl;
    return 0;
}
