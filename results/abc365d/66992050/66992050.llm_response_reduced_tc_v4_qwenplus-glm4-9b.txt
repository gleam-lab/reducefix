
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    char prev = 'N';
    int ans = 0;
    for (char c : S) {
        if (c == 'P' || c == 'R' || c == 'S') {
            if (prev == 'N') {  // First character is not 'N'
                prev = c;
                ans++;
            } else if (prev == c) {  // Same character as previous
                prev = 'N';
            } else {  // Different character from the previous
                prev = c;
                ans++;
            }
        }
    }
    cout << ans;
    return 0;
}
```
