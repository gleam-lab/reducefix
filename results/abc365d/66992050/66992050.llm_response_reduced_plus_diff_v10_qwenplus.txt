```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    // We need to count the minimum number of button presses
    // The idea is to use dynamic programming where:
    // dp[i][c] = minimum total button presses up to position i, ending with character c
    // where c is 'P', 'R', or 'S'
    
    const int INF = 1e9;
    vector<map<char, int>> dp(N);
    
    // Initialize first character
    dp[0]['P'] = (S[0] == 'P') ? 0 : 1;
    dp[0]['R'] = (S[0] == 'R') ? 0 : 1;
    dp[0]['S'] = (S[0] == 'S') ? 0 : 1;
    
    // Define the cyclic order: P -> R -> S -> P
    auto cost = [](char from, char to) {
        if (from == to) return 0;
        if ((from == 'P' && to == 'R') || 
            (from == 'R' && to == 'S') || 
            (from == 'S' && to == 'P')) {
            return 1;
        }
        return 2; // go backward in cycle
    };
    
    for (int i = 1; i < N; i++) {
        // For each possible current state
        for (char curr : {'P', 'R', 'S'}) {
            dp[i][curr] = INF;
            // Try all possible previous states
            for (char prev : {'P', 'R', 'S'}) {
                if (dp[i-1].count(prev)) {
                    int add_cost = (S[i] == curr) ? 0 : 1;
                    int transition_cost = cost(prev, curr);
                    dp[i][curr] = min(dp[i][curr], dp[i-1][prev] + add_cost + transition_cost);
                }
            }
        }
    }
    
    // Find minimum among all final states
    int ans = INF;
    for (auto& [state, val] : dp[N-1]) {
        ans = min(ans, val);
    }
    
    cout << ans;
    return 0;
}
```
