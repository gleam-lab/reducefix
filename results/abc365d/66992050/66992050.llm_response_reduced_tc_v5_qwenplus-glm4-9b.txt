
```cpp
#include <iostream>
using namespace std;
using ll = long long;

int main() {
    int N;
    string S;
    cin >> N >> S;
    char past = S[0];
    ll ans = 1; // Start with 1 because at least one transition will occur

    for (int i = 1; i < N; ++i) {
        char c = S[i];
        if ((past == 'P' && c == 'S') || (past == 'S' && c == 'R') || (past == 'R' && c == 'P')) {
            ans++;
            past = c;
        } else {
            past = c;
        }
    }

    cout << ans;
    return 0;
}
```
