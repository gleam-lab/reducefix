```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main(){
    int N;
    string S;
    cin >> N >> S;
    
    char last = 'X'; // No move yet
    int ans = 0;
    
    auto required = [](char c) {
        if(c == 'P') return 'S';
        if(c == 'R') return 'P';
        if(c == 'S') return 'R';
        return 'X';
    };
    
    for(char c : S){
        char need = required(c);
        if(last != need){
            ans++;
            last = need;
        }
    }
    
    cout << ans;
    return 0;
}
```
