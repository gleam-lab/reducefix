```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    char last_move = 'N'; // Last move we made (initially none)
    int ans = 0;
    
    for (char c : S) {
        if (c == 'P') {
            // Opponent plays Paper
            // We should play Scissors to win, unless we just played Scissors
            if (last_move != 'S') {
                ans++;
                last_move = 'S';
            } else {
                last_move = 'P'; // Play Paper to avoid same move
            }
        } else if (c == 'R') {
            // Opponent plays Rock
            // We should play Paper to win, unless we just played Paper
            if (last_move != 'P') {
                ans++;
                last_move = 'P';
            } else {
                last_move = 'R'; // Play Rock to avoid same move
            }
        } else if (c == 'S') {
            // Opponent plays Scissors
            // We should play Rock to win, unless we just played Rock
            if (last_move != 'R') {
                ans++;
                last_move = 'R';
            } else {
                last_move = 'S'; // Play Scissors to avoid same move
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
