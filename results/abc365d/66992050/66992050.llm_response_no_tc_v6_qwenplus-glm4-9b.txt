
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    char past = 'N';
    int ans = 0;
    
    for (char c : S) {
        if (c == 'P') {
            if (past != 'P') { // Only increment if the previous character was not 'P'
                ans++;
                past = 'P';
            }
        } else if (c == 'R') {
            if (past != 'R') { // Only increment if the previous character was not 'R'
                ans++;
                past = 'R';
            }
        } else if (c == 'S') {
            if (past != 'S') { // Only increment if the previous character was not 'S'
                ans++;
                past = 'S';
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
