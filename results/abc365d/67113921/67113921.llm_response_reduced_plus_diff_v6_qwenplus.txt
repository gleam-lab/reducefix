#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char prev = 'a'; // previous move chosen by us
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // If we can use the winning move without repeating, do it
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // Otherwise, we must choose a different move that isn't the same as previous
            // We can pick any move except 'prev'. Pick one that beats opponent or at least doesn't lose
            if (c == 'R') {
                // We would have liked to play 'P', but if 'P' == prev, we can play 'R' (tie) or 'S' (lose)
                // To avoid losing unnecessarily, we prefer tie over loss
                if (prev != 'R') {
                    prev = 'R';
                } else {
                    prev = 'S'; // forced to lose
                }
            } else if (c == 'P') {
                if (prev != 'P') {
                    prev = 'P';
                } else {
                    prev = 'R';
                }
            } else { // c == 'S'
                if (prev != 'S') {
                    prev = 'S';
                } else {
                    prev = 'P';
                }
            }
        }
    }   
    cout << wins;
}
