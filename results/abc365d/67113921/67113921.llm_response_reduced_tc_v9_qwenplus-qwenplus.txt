```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove, prev = 'a'; // prev stores the last move we made
    for (char c : S){
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        if (winmove != prev){
            wins++;
            prev = winmove;
        } else {
            // We must have tied on previous round; choose a different move
            // Since each move only beats one other move, we can always choose a move that beats current c
            // We just need to pick a new move different from previous
            if (winmove == 'R') {
                if (c == 'P') winmove = 'S';
                else winmove = 'P';
            } else if (winmove == 'P') {
                if (c == 'S') winmove = 'R';
                else winmove = 'S';
            } else { // winmove == 'S'
                if (c == 'R') winmove = 'P';
                else winmove = 'R';
            }
            prev = winmove;
            wins++;
        }
    }   
    cout << wins;
}
```
