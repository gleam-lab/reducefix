        } else if (c == 'P') {
            wins += (c == 'P') ? 1 : 0;
        } else if (c == 'S') {

    // Now, we need to count the optimal wins by always choosing the best move
    wins = 0;
    for (char c : S) {
        if (c == 'R') {
            wins++; // P beats R
        } else if (c == 'P') {
            wins++; // S beats P
        } else if (c == 'S') {

        }
    }

    return 0;
