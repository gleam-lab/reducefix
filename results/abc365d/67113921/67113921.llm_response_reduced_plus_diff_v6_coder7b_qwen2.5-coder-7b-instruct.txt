```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    
    int wins = 0;
    char prev = 'A'; // Initialize with a character not present in the problem's set {'R', 'P', 'S'}
    
    for (char c : S) {
        char winmove;
        if (c == 'R') winmove = 'P';
        else if (c == 'P') winmove = 'S';
        else winmove = 'R'; // If c is 'S'
        
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            prev = c;
        }
    }
    
    cout << wins << endl;
    return 0;
}
```
