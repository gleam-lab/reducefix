```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char last_played = ' '; // No move played initially
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // Only make a move if we aren't repeating the same winning move
        if (last_played != winmove) {
            wins++;
            last_played = winmove;
        } else {
            // If we would repeat the same winning move, we don't need to change our move
            // We can keep playing the same winning move, so no additional move needed
            // The opponent changes, but we don't have to if our current move still wins
            // So we keep last_played as is
        }
    }   
    cout << wins;
    return 0;
}
```
