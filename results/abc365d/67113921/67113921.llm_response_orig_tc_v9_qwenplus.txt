#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    
    int wins = 0;
    char prev = 'a'; // Previous move played by us
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // If we can play the winning move (different from previous), do it and count win
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // We cannot play the winning move because it's same as previous
            // So we play the losing move (which doesn't matter for score)
            // But we must update prev to the move we actually played
            // We can play either of the two remaining moves, but we should choose one that minimizes future conflicts
            // However, greedy choice: just pick any non-winning move different from what opponent played
            if (c == 'R') {
                // Opponent played R, winning move is P (but we can't play P)
                // So choose between S and R. Choose S (losing) or R (tie). Doesn't add to wins.
                // To minimize constraint for next move, we choose arbitrarily: let's pick S
                prev = 'S';
            } else if (c == 'P') {
                // Opponent played P, winning move is S (but we can't play S)
                // Choose R (losing) 
                prev = 'R';
            } else { // c == 'S'
                // Opponent played S, winning move is R (but we can't play R)
                // Choose P (losing)
                prev = 'P';
            }
        }
    }   
    cout << wins;
}
