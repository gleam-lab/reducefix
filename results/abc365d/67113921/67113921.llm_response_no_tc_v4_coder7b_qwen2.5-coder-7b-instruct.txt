```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // Initialize with a value not in 'RPS'
    for (int i = 0; i < N; ++i){
        char c = S[i];
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (winmove != prev && i > 0){ // Ensure not at the start and move differs
            wins++;
            prev = winmove;
        } else{
            prev = c;
        }
    }   
    cout << wins;

}
```
