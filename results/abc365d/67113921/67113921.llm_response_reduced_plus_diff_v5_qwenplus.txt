```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    long long wins = 0;
    char last_move = ' '; // No move initially

    for (char c : S) {
        char best_counter = ' ';
        if (c == 'R') best_counter = 'P';
        else if (c == 'P') best_counter = 'S';
        else if (c == 'S') best_counter = 'R';

        // If we can use the best counter (i.e., it's different from last move), use it
        if (best_counter != last_move) {
            wins++;
            last_move = best_counter;
        } else {
            // Otherwise, we must use a non-optimal move, so choose any move that isn't c or last_move
            // But importantly: we don't win this round
            // We just need to pick a valid move to set last_move for next iteration
            for (char m : {'R', 'P', 'S'}) {
                if (m != c && m != last_move) {
                    last_move = m;
                    break;
                }
            }
        }
    }   

    cout << wins;
}
```
