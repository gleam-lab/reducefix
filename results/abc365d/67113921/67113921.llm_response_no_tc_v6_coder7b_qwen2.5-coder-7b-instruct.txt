#include <iostream>
#include <string>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'X'; // Initialize with a character that won't be in the input string
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        }
        if (c == prev) { // Check if the current move is the same as the previous one
            wins++; // Increment wins because we need to change our move
        }
        prev = winmove; // Update the previous move to the winning move
    }
    cout << wins;

    return 0;
}
