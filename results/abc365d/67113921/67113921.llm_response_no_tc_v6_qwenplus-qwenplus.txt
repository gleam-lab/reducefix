```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;

    int wins = 0;
    char my_move = 'a'; // previous move, initially invalid

    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }

        if (winmove != my_move) {
            wins++;
            my_move = winmove;
        } else {
            my_move = c; // we tie, so we played same as opponent
        }
    }

    cout << wins;
    return 0;
}
```
