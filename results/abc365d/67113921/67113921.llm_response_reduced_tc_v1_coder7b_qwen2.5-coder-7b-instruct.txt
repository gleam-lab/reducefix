```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove, prev = 'a';
    for (int i = 0; i < N; ++i){
        if (S[i] == 'R'){
            winmove = 'P';
        } else if (S[i] == 'P'){
            winmove = 'S';
        } else if (S[i] == 'S'){
            winmove = 'R';
        }
        if (winmove != prev && (i == 0 || winmove != S[i-1])){
            wins++;
            prev = winmove;
        } else{
            prev = S[i];
        }
    }   
    cout << wins;

}
```
