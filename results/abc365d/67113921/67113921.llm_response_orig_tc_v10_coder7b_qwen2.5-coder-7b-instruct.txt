```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    int wins = 0;
    char winmove, prev = 'A'; // Initialize prev with a character not in the game ('A' is chosen arbitrarily here)

    for (char c : S) {
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        }
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            prev = c;
        }
    }

    cout << wins << endl;
    return 0;
}
```

The initialization of `prev` with a character not in the game (in this case, 'A') helps handle the initial state properly without causing any issues. This approach maintains the algorithmic correctness and handles all edge cases as required.
