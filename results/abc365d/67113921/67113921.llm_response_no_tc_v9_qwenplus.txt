```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char last_played = ' '; // No move played initially
    
    for (char opponent_move : S) {
        char best_counter;
        
        // Determine the move that beats the opponent's move
        if (opponent_move == 'R') {
            best_counter = 'P';
        } else if (opponent_move == 'P') {
            best_counter = 'S';
        } else { // opponent_move == 'S'
            best_counter = 'R';
        }
        
        // If we can play the counter move (i.e., it's different from our last move), do it
        if (last_played != best_counter) {
            wins++;
            last_played = best_counter;
        } 
        // Otherwise, we have to play something else (can't repeat moves)
        // In this case, we play the opponent's move (which won't win, but allows us to change state)
        else {
            last_played = opponent_move;
        }
    }   
    
    cout << wins;
    return 0;
}
```
