```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // no move has been played initially

    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }

        // We can choose the winning move unless we just used it
        if (winmove != prev) {
            wins++;
            prev = winmove; // we play the winning move
        } else {
            // We cannot play the same winning move twice in a row
            // So we play the losing move (i.e., the opponent's move)
            prev = c;
        }
    }   
    cout << wins;
}
```
