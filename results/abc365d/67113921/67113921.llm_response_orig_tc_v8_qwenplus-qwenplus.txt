#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    int wins = 0;
    char winmove, prev = 'a'; // Initialize with dummy value

    for (char c : S) {
        // Determine the winning move for current character
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }

        // If current winning move is different from previous move, we can play it
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // Otherwise, we have to play the original character to avoid repeating the same move
            prev = c;
        }
    }

    cout << wins << endl;
}
