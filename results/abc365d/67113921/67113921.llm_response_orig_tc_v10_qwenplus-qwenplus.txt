```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    int wins = 0;
    char prev_move = 'a'; // Initially invalid to ensure first move always changes

    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }

        if (winmove != prev_move) {
            wins++;
            prev_move = winmove;
        } else {
            // If we can't use the winning move, play the current character's move
            // But only change if it's different from previous move
            if (c != prev_move) {
                prev_move = c;
            }
        }
    }

    cout << wins << endl;
}
```
