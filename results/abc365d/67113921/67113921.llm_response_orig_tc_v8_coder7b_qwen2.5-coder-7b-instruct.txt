int main() {

    string S;
    cin >> N;

    int wins = 0;
    char winmove, prev = ' ';
    for (char c : S) {
        if (prev == ' ') {
            prev = c;
            continue;
        }
        if ((c == 'R' && prev == 'P') || (c == 'P' && prev == 'S') || (c == 'S' && prev == 'R')) {
            wins++;
            prev = c;
        } else {
            prev = c;
        }
    }
    cout << wins;
    return 0;
}
