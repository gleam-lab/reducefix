```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    int wins = 0;
    char prev = 'a'; // previous move we made

    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }

        // If our winning move is different from the previous move we played, we can use it
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } 
        // Otherwise, we are forced to play the opponent's move (or any move, but it won't win)
        // So we set prev to the current opponent's move (or effectively lose/draw without changing strategy)
        else {
            prev = c;
        }
    }   
    cout << wins;
}
```
