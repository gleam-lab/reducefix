```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // no move played initially

    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }

        // We can choose to play the winning move unless it's the same as previous
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // If winning move is same as previous, we must play something else
            // But we don't want to lose, so we play the safe option: the one that ties
            prev = c;
        }
    }   
    cout << wins;
}
```
