#include <iostream>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N; // Read the number of characters in the string
    cin >> S; // Read the string consisting of 'R', 'P', and 'S'
    
    int wins = 0;
    char winmove;
    char prev = S[0]; // Initialize prev with the first character of the string
    
    for (int i = 1; i < N; ++i) {
        switch (S[i]) {
            case 'R':
                winmove = 'P'; // Rock beats Scissors
                break;
            case 'P':
                winmove = 'S'; // Paper beats Rock
                break;
            case 'S':
                winmove = 'R'; // Scissors beats Paper
                break;
        }
        
        if (winmove != prev) {
            wins++; // Increment wins if the winning move is different from the previous move
            prev = winmove; // Update the previous move to the current winning move
        } else {
            prev = S[i]; // Update the previous move to the current character if no change
        }
    }   
    cout << wins;

    return 0;
}
