```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    int wins = 0;
    char prev_move = 'a'; // No impact initially, will be updated

    for (char c : S) {
        char win_move;
        if (c == 'R') win_move = 'P';
        else if (c == 'P') win_move = 'S';
        else win_move = 'R';

        if (win_move != prev_move) {
            wins++;
            prev_move = win_move;
        } else {
            // We cannot use the same winning move, so we must use a different strategy.
            // In this case, choose the move that doesn't interfere with the next decision.
            // For simplicity, we just use the current opponent's move as our move.
            prev_move = c;
        }
    }

    cout << wins << endl;
}
```
