```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove, prev = 'a'; // prev stores the last move we made

    for (char c : S){
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }

        if (winmove != prev){
            wins++;
            prev = winmove;
        } else {
            // We have to pick a different move; we cannot repeat same move
            // So instead of incrementing win, we just change our move to current character
            // but this doesn't guarantee a win, so we do not increment.
            prev = c;
        }
    }   
    cout << wins;
}
```
