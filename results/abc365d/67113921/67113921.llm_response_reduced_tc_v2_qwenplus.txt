```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // No move has been played yet

    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }

        // We can choose the winning move unless it's same as previous move
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // If winning move is same as previous, we cannot use it consecutively
            // So we play the opponent's move (or any non-winning, non-previous move)
            // But we don't win this round
            prev = c;
        }
    }   
    cout << wins;
}
```
