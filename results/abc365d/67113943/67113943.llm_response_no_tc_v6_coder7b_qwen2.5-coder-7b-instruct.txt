```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // Initialize previous move to a dummy value not in {'R', 'P', 'S'}
    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (winmove != prev){
            wins++;
            prev = winmove;
        }
    }   
    cout << wins;

}
```

**Explanation of Changes:**
- The variable `prev` was initialized to 'a' which is not a valid choice among 'R', 'P', or 'S'. This could lead to incorrect logic when determining the first move. I changed it to 'a' as a dummy value indicating no previous move has been made yet.
- No other changes were necessary as the logic for determining the winning move and counting wins was correct.
