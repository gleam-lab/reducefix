```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove = 'a', prev = 'a';
    for (char c : S){
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (winmove != prev){
            wins++;
            prev = winmove;
        } else {
            // Opponent will now play the move that beats our previous move
            // So we need to switch to beat their new move
            if (prev == 'R') {
                prev = 'P';  // They might play S, so we play P
            } else if (prev == 'P') {
                prev = 'S';  // They might play R, so we play S
            } else if (prev == 'S') {
                prev = 'R';  // They might play P, so we play R
            }
        }
    }   
    cout << wins;
}
```
