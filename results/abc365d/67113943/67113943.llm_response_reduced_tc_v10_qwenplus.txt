#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'a'; // previous move played by us
    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        // We can win this round if we play winmove
        // But we cannot play the same move twice in a row
        if (winmove != prev){
            wins++;
            prev = winmove;
        } else {
            // We are forced to play winmove but we already played it last time
            // So we must play something else, meaning we cannot win
            // We choose any move that is not prev and not winning (to avoid winning? no, we just need to make a valid move)
            // Actually, we just need to pick a different move. It doesn't matter which one since we won't win.
            // But we must update prev to what we actually played
            // The key: we don't win, so we just need to pick a non-winning, non-repeating move
            // However, we can pick any move except 'prev'. Even if it loses or ties, we don't win.
            // So we pick an arbitrary move different from prev
            if (winmove == 'R') {
                prev = 'P'; // could also be 'S', doesn't matter
            } else if (winmove == 'P') {
                prev = 'S';
            } else {
                prev = 'R';
            }
        }
    }   
    cout << wins;
}
