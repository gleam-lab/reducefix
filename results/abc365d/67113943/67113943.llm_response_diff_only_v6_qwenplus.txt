```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char opponent_move, my_last_move = 'X'; // 'X' indicates no previous move
    
    for (int i = 0; i < N; i++) {
        opponent_move = S[i];
        char best_response;
        
        // Determine the move that beats the opponent's current move
        if (opponent_move == 'R') {
            best_response = 'P';
        } else if (opponent_move == 'P') {
            best_response = 'S';
        } else { // opponent_move == 'S'
            best_response = 'R';
        }
        
        // If we can make a winning move (or it's the first move), do it
        if (my_last_move != best_response) {
            wins++;
            my_last_move = best_response;
        } else {
            // We can't repeat our move, so we must choose a non-winning option
            // This means we don't win this round, and our move becomes the opponent's move
            my_last_move = opponent_move;
        }
    }
    
    cout << wins;
    return 0;
}
```
