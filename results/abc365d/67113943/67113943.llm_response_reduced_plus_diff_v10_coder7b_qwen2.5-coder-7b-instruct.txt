#include <iostream>
#include <string>

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::string S;
    std::cin >> N;
    std::cin >> S;

    int wins = 0;
    char prev = 'A'; // Initialize with a character not present in the problem's set {R, P, S}

    for (char c : S) {
        char winmove;
        if (c == 'R') winmove = 'P';
        else if (c == 'P') winmove = 'S';
        else winmove = 'R';

        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            prev = c;
        }
    }

    std::cout << wins;
    return 0;
}
