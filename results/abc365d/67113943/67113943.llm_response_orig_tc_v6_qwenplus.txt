```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char prev = ' '; // No previous move initially
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // If we can use the winning move without repeating the previous move, do it
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // Otherwise, we must use a different move (but still valid)
            // We don't win this round, so just pick any move that isn't prev
            // This maintains state for the next iteration
            prev = c; // Using opponent's move is safe since it differs from winmove which equals prev
        }
    }   
    cout << wins;
}
```
