#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove = 'A'; // Initialize winmove to an invalid character
    char prev = 'A';    // Initialize prev to an invalid character

    for (char c : S) {
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        }
        if (winmove != prev && prev != 'A') { // Check if prev is not initialized
            wins++;
            prev = winmove;
        } else {
            prev = c;
        }
    }
    cout << wins;
}
