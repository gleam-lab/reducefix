#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);

    for (char c : S){
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (winmove != prev && (winmove == c || (c == 'R' && winmove == 'S') || (c == 'S' && winmove == 'P') || (c == 'P' && winmove == 'R'))){
            wins++;
            prev = winmove;
        } else{
            prev = c;
        }
    }   
    cout << wins;

}
