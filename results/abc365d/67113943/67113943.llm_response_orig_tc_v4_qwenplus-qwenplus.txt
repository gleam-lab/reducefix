```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;

    int wins = 0;
    char prev_move = 'a';  // Track previous move, initially invalid

    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else {  // c == 'S'
            winmove = 'R';
        }

        if (winmove != prev_move) {
            wins++;
            prev_move = winmove;
        } else {
            // We cannot use the same winning move twice in a row,
            // so we must use the current character (c) as the move
            prev_move = c;
        }
    }

    cout << wins << endl;
}
```
