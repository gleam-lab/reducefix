```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prevWinMove = ' '; // Initialize to something not in R, P, S
    for (char c : S){
        char currentWinMove;
        if (c == 'R'){
            currentWinMove = 'P';
        } else if (c == 'P'){
            currentWinMove = 'S';
        } else if (c == 'S'){
            currentWinMove = 'R';
        }
        if (currentWinMove != prevWinMove){
            wins++;
            prevWinMove = currentWinMove;
        }
    }   
    cout << wins;
}
```

- For each move in the sequence, we determine the counter move that would win.
- We then check if this counter move is different from the previous one we chose. If it is, we count this as a win and update the previous move to this counter move. If it's the same, we skip it.
- This ensures that we maximize the number of wins by not repeating the same counter move consecutively.
