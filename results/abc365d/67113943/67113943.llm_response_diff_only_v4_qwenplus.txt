```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    int wins = 0;
    char opponent_move, my_move, prev_my_move = 'X'; // X indicates no previous move
    
    for (int i = 0; i < N; i++) {
        opponent_move = S[i];
        
        // Determine the move that would beat the opponent's current move
        if (opponent_move == 'R') {
            my_move = 'P';
        } else if (opponent_move == 'P') {
            my_move = 'S';
        } else { // opponent_move == 'S'
            my_move = 'R';
        }
        
        // Only count as a win if we can make a different move than our previous one
        if (prev_my_move == 'X' || my_move != prev_my_move) {
            wins++;
            prev_my_move = my_move;
        } else {
            // If we can't change our move, we have to stick with opponent's move
            // which means we don't win this round
            prev_my_move = opponent_move;
        }
    }
    
    cout << wins;
    return 0;
}
```
