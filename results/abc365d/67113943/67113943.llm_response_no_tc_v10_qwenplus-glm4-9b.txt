#include <iostream>
#include <string>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    
    int wins = 0;
    char prev = 'X'; // Initialize previous option as 'X' to handle the first character separately.
    
    for (char c : S) {
        char winmove;
        if (c == 'R') winmove = 'P'; // Rock beats Scissors
        else if (c == 'P') winmove = 'S'; // Paper beats Rock
        else winmove = 'R'; // Scissors beats Paper
        
        if (prev != 'X') { // Skip the first character comparison
            if (winmove == prev) {
                prev = c; // Update the previous character
            } else {
                wins++; // Increment wins if the current move beats the previous move
                prev = winmove; // Update the previous character to the winning move
            }
        }
    }
    
    cout << wins;
}
