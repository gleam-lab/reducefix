```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove = 'a', prev = 'a';
    for (char c : S){
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (winmove != prev){
            wins++;
            prev = winmove;
        } else {
            // We need to change strategy here:
            // If the winning move is the same as previous, we can choose a different non-losing move
            // So instead of playing the winning move, play a different one that doesn't violate the constraint
            // In this case, since we can't play the same winning move twice in a row,
            // we play the opponent's move directly
            prev = c;  // we play the same as opponent's current move
            // This results in either a loss or tie, but avoids violating the rule
            // Note: This doesn't contribute to the win count
        }
    }   
    cout << wins;
}
```
