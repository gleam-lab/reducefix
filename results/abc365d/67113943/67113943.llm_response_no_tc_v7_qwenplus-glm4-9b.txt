
```cpp
#include <iostream>
#include <string>

using namespace std;

int main() {
    int N;
    string S;
    cin >> N;
    cin >> S;
    
    int wins = 0;
    char prev = 'a';  // Initialize to an invalid character to handle the first non-'R'/'P'/'S' input

    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        } else {
            continue;  // Skip non-'R', 'P', 'S' characters
        }
        
        if (winmove != prev) {
            wins++;
            prev = winmove;
        }
    }
    
    cout << wins;
    return 0;
}
```
