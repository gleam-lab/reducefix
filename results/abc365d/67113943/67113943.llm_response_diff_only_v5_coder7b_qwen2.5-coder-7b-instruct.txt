#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char prev = 'N'; // Initialize to a character that is not R, P, or S
    for (char c : S){
        char winmove;
        if (c == 'R'){
            winmove = 'P';
        } else if (c == 'P'){
            winmove = 'S';
        } else if (c == 'S'){
            winmove = 'R';
        }
        if (prev == 'N' || winmove != prev){ // Check if it's the first move or if the current move is different from the previous winning move
            wins++;
            prev = winmove;
        } else {
            prev = c; // Update prev to the current move if it doesn't lead to a win
        }
    }   
    cout << wins;

}
