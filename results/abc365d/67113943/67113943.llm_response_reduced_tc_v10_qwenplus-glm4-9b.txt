#include <iostream>
#include <string>

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::string S;
    std::cin >> N;
    std::cin >> S;

    int wins = 0;
    char winmove = 'a';  // Initialize to avoid issues when S starts with 'R'
    for (char c : S) {
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        }

        if (c == winmove) {
            wins++;
            winmove = 'a';  // Reset winmove after a winning move
        }
    }
    
    std::cout << wins;

    return 0;
}
