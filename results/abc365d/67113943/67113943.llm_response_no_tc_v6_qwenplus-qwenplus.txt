```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;

    int wins = 0;
    char prev_move = 'X'; // Initialize to an invalid value

    for (char c : S) {
        char win_move;
        if (c == 'R') {
            win_move = 'P';
        } else if (c == 'P') {
            win_move = 'S';
        } else { // c == 'S'
            win_move = 'R';
        }

        if (win_move != prev_move) {
            wins++;
            prev_move = win_move;
        } else {
            // We cannot choose the same move, so we must lose/draw this round
            // According to problem logic, we skip incrementing and move on
            prev_move = c; // Opponent played c, we can update our "last move" to opponent's move
        }
    }

    cout << wins << endl;
    return 0;
}
```
