    cin >> N >> S;
    int wins = 0;
    char prev_move = '\0';

        char current_move;
        if (c == 'R') {
            current_move = 'P';
        } else if (c == 'P') {
            current_move = 'S';
        } else { // 'S'
            current_move = 'R';
        }

            wins++;
            prev_move = current_move;
        }
    }
    
    cout << wins << endl;
    return 0;
}
