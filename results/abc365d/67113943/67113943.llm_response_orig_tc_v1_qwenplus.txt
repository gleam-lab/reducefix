```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N >> S;
    
    long long wins = 0;
    char prev_move = 'X'; // Use 'X' to indicate no previous move
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // If the winning move is different from our last move, we can use it
        if (winmove != prev_move) {
            wins++;
            prev_move = winmove;
        } 
        // Otherwise, we have to play what the opponent played (to reset)
        else {
            prev_move = c;
        }
    }
    
    cout << wins;
    return 0;
}
```
