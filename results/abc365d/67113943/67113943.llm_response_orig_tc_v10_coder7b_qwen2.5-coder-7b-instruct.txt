```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;

    int wins = 0;
    char prevMove = 'A'; // Initial move can be any since we don't know the previous move of the opponent

    for (char move : S) {
        char optimalResponse;
        if (move == 'R') optimalResponse = 'P';
        else if (move == 'P') optimalResponse = 'S';
        else optimalResponse = 'R';

        if (optimalResponse != prevMove) {
            wins++;
            prevMove = optimalResponse;
        } else {
            prevMove = move;
        }
    }

    cout << wins << endl;
    return 0;
}
```
