```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    
    int wins = 0;
    char prev = 'a'; // Previous move played by us
    
    for (char c : S) {
        char winmove;
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else { // c == 'S'
            winmove = 'R';
        }
        
        // If our winning move is different from previous move, we can play it and win
        if (winmove != prev) {
            wins++;
            prev = winmove;
        } else {
            // We cannot repeat moves, so we have to play something else
            // Since we can't win in this case (would require repeating), we choose a non-winning move
            // But the key is: we just need to update our last move to something valid
            // We set prev to opponent's move (or any losing/draw move), but importantly:
            // We don't win this round, and we free up the ability to win next time if needed
            prev = c; // This effectively plays a losing move (or draw), but allows flexibility next turn
        }
    }   
    cout << wins;
}
```
