#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;
    int wins = 0;
    char winmove = 'a', prev = 'a';
    map<char, char> winMoves = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    for (char c : S){
        winmove = winMoves[c];
        if (winmove != prev){
            wins++;
            prev = winmove;
        }
    }   
    cout << wins;

    return 0;
}
