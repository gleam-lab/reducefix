#include <iostream>
#include <string>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    string S;
    cin >> N;
    cin >> S;

    int wins = 0;
    char winmove;
    char prev = 'a'; // Start with any non-playing character

    for (char c : S) {
        // Determine the winning move for the current move
        if (c == 'R') {
            winmove = 'P';
        } else if (c == 'P') {
            winmove = 'S';
        } else if (c == 'S') {
            winmove = 'R';
        }

        // Check if the current winning move is different from the previous one
        if (winmove != prev) {
            wins++;
            prev = winmove; // Update the previous move to the current winning move
        }
    }

    cout << wins;
    return 0;
}
