#include <iostream>
#include <string>
using namespace std;

// Function to determine the winner of a single round
// Returns 1 if player1 wins, -1 if player2 wins, 0 for draw
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return 0;
}

// Function to get the winning move against a given move
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Invalid case
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int ans = 0;
    char last_move = 'X'; // No move initially

    for (int i = 0; i < n; ++i) {
        char current_move = s[i];
        char winning_move = win(current_move);

        // If we can play the winning move against the current symbol
        if (last_move != winning_move) {
            ans++;
            last_move = winning_move;
        } else {
            // Otherwise, we have to play the same symbol as previous
            last_move = 'X'; // Resetting since consecutive moves are not playable
        }
    }

    cout << ans << endl;
    return 0;
}
