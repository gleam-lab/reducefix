```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n;
    string s;
    cin >> n >> s;

    // dp[i][c] = minimum number of changes needed for prefix ending at i with last character c
    // c: 0='R', 1='P', 2='S'
    const int INF = 1e9;
    vector<vector<int>> dp(n, vector<int>(3, INF));
    
    // Initialize first character
    for (int c = 0; c < 3; c++) {
        char ch = "RPS"[c];
        dp[0][c] = (s[0] != ch);
    }
    
    // Transition: ensure no two consecutive same characters
    for (int i = 1; i < n; i++) {
        for (int curr = 0; curr < 2; curr++) {
            for (int prev = 0; prev < 3; prev++) {
                if (curr == prev) continue; // No consecutive same
                
                char curr_ch = "RPS"[curr];
                int cost = (s[i] != curr_ch);
                dp[i][curr] = min(dp[i][curr], dp[i-1][prev] + cost);
            }
        }
        
        // Handle 'S' separately since it's index 2
        for (int prev = 0; prev < 3; prev++) {
            if (prev == 2) continue; // No consecutive S
            
            char curr_ch = 'S';
            int cost = (s[i] != curr_ch);
            dp[i][2] = min(dp[i][2], dp[i-1][prev] + cost);
        }
    }
    
    int ans = min({dp[n-1][0], dp[n-1][1], dp[n-1][2]});
    cout << ans << endl;
    
    return 0;
}
```
