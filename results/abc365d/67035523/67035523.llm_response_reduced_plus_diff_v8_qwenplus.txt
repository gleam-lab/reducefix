using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return x;
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    vector<int> dp(n + 1, 0);
    char last = 'X'; // no move initially

    for (int i = 0; i < n; ++i) {
        char preferred = win(s[i]);
        
        if (last != preferred) {
            dp[i + 1] = dp[i] + 1;
            last = preferred;
        } else {
            dp[i + 1] = dp[i];
            // continue with same move
        }
    }

    cout << dp[n] << endl;
}
