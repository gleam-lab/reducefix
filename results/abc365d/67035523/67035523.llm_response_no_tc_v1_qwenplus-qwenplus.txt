```cpp
#include <iostream>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Should not happen for valid input
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int ans = 0;
    char last_play = 'X'; // No move yet

    for (int i = 0; i < n; ++i) {
        char to_win = win(s[i]);
        if (to_win != last_play) {
            ans++;
            last_play = to_win;
        }
    }

    cout << ans << endl;
    return 0;
}
```
