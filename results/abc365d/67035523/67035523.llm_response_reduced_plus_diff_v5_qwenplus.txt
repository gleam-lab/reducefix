int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100;
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    int ans = 1; // At least one move is needed
    char current = win(s[0]); // First move must be winning against s[0]
    
    for (int i = 1; i < n; i++) {
        char beat_next = win(s[i]);
        if (current != beat_next) {
            ans++;
            current = beat_next;
        }
        // If current already beats s[i], we don't need to change
    }
    
    cout << ans << endl;
    return 0;
}
