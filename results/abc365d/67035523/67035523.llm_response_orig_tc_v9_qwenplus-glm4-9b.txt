
### Problem Analysis
The problem involves analyzing a sequence of characters representing rounds in a game where the characters 'P', 'R', and 'S' represent the outcomes of the game. The goal is to count the number of transitions between different game states based on a predefined winning condition for each pair of characters.

### Solution Overview
The solution uses a comparison function `cmp` to determine the relative order of characters based on the game's rules. It then uses this function to count the number of transitions that result in a change of the winning character over the sequence. The winning function `win` determines the character that would win against a given character in the sequence. The main part of the code reads the input, processes the sequence to count the transitions, and outputs the result.

#include <iostream>
#include <string>

using namespace std;

int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100;
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    return 'S';
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int ans = 0;
    char prev = 'X'; // Initialize a non-existent game state

    for (int i = 0; i < n; i++) {
        char next = s[i];
        if (cmp(prev, next) != 0) {
            ans++;
            prev = next;
        }
    }

    cout << ans << endl;
    return 0;
}
