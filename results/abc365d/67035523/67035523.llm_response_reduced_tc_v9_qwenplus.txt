using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int ans = 1; // At least one round is needed
    char current_form = win(s[0]); // The form the player must take to beat first opponent

    for (int i = 1; i < n; i++) {
        char needed_form = win(s[i]);
        if (needed_form != current_form) {
            ans++;
            current_form = needed_form;
        }
    }

    cout << ans << endl;
    return 0;
}
