```cpp
#include <iostream>
#include <string>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    int ans = 1; // At least one move is needed
    char current_hand = win(s[0]); // First optimal hand to play
    
    for (int i = 1; i < n; i++) {
        char desired_hand = win(s[i]);
        if (desired_hand != current_hand) {
            ans++;
            current_hand = desired_hand;
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
