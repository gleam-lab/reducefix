int cmp(char x, char y) {
    if (x == 'P') return (y == 'P') ? 0 : (y == 'R') ? 1 : -1;
    if (x == 'R') return (y == 'P') ? -1 : (y == 'R') ? 0 : 1;
    if (x == 'S') return (y == 'P') ? -1 : (y == 'R') ? 1 : 0;
    return -100; // X is considered least important for comparison
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    return 'S'; // P always wins against S
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    int ans = 0;
    char prev = 'A'; // Initialize to a non-game character

    for (char c : s) {
        if (prev != win(c)) {
            ans++;
            prev = c;
        }
    }

    cout << ans;
    return 0;
}
