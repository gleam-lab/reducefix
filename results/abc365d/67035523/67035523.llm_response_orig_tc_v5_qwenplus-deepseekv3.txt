        return 0;

    vector<pair<char, int>> segments;

    int count = 1;
    for (int i = 1; i < n; ++i) {
        if (s[i] == current) {
            count++;
        } else {
            segments.emplace_back(current, count);
            current = s[i];
            count = 1;
        }
    }
    segments.emplace_back(current, count);

    int ans = 0;
    char prev = 'X';
    for (size_t i = 0; i < segments.size(); ++i) {
        char move = segments[i].first;
        char w = win(move);
        if (w != prev) {
            ans += segments[i].second;
            prev = w;
        } else {
            if (i + 1 < segments.size()) {
                char next_move = segments[i + 1].first;
                char next_w = win(next_move);
                if (next_w != prev) {
                    prev = next_w;
                    ans += segments[i + 1].second;
