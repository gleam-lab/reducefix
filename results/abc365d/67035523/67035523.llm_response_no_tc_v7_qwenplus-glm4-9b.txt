int cmp(char x, char y) {

}

int main() {
    int n;
    string s;
    cin >> n >> s;
    int ans = 0;
    char prev = 'X'; // Initialize prev to 'X' to handle the first comparison

    for (char c : s) {
        if (cmp(prev, c) != 0) {
            ans++;
            prev = c;
        }
    }

    cout << ans;

}
