#include <iostream>
using namespace std;

// Function to determine the winner of a single game of Rock-Paper-Scissors
// Returns 1 if player1 wins, -1 if player2 wins, 0 for a tie
int cmp(char player1, char player2) {
    if (player1 == player2) return 0;
    
    if ((player1 == 'R' && player2 == 'S') ||
        (player1 == 'P' && player2 == 'R') ||
        (player1 == 'S' && player2 == 'P')) {
        return 1; // player1 wins
    }
    
    return -1; // player2 wins
}

// Function to return the winning move against a given move
char win(char move) {
    if (move == 'R') return 'P';
    if (move == 'P') return 'S';
    return 'R'; // S
}

int main() {
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    int max_moves = 0;
    char last_move = 'X'; // Initially set to an invalid character
    
    for (int i = 0; i < n; ++i) {
        // Always play the move that beats the opponent's current move
        char current_move = win(s[i]);
        
        // Only count a move change
        if (current_move != last_move) {
            max_moves++;
            last_move = current_move;
        }
    }
    
    cout << max_moves << endl;
    
    return 0;
}
