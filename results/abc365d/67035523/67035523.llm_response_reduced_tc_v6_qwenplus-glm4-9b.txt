#include <iostream>
#include <string>
#include <vector>

using namespace std;

int cmp(char x, char y) {
    if (x == 'P') return (y == 'R') - (y == 'S');
    if (x == 'R') return (y == 'S') - (y == 'P');
    if (x == 'S') return (y == 'P') - (y == 'R');
    return -100;
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    int ans = 0;
    char prev = 'X';  // Start with an impossible previous character to ensure the first comparison is always true.

    for (int i = 0; i < n; i++) {
        if (prev != win(s[i])) {
            ans++;
            prev = win(s[i]);
        } else {
            prev = s[i];
        }
    }

    cout << ans;
    return 0;
}
