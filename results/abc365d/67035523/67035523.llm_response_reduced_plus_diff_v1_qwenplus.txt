```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100;
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int ans = 1; // At least one round is always played
    char current_winner = win(s[0]); // First move's winning counter

    for (int i = 1; i < n; i++) {
        char opponent_move = s[i];
        char beat_opponent = win(opponent_move);

        // If current winner can't beat this opponent move, we need a new champion
        if (cmp(current_winner, opponent_move) != 1) {
            ans++;
            current_winner = beat_opponent;
        }
        // Otherwise, current winner continues
    }

    cout << ans << endl;
    return 0;
}
```
