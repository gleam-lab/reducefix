```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

// Function to determine the outcome of a move against a previous one
bool is_winning_move(char prev, char curr) {
    if ((prev == 'R' && curr == 'P') || 
        (prev == 'P' && curr == 'S') ||
        (prev == 'S' && curr == 'R')) {
        return true;
    }
    return false;
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Edge case: no moves
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int ans = 1;  // At least one move needed
    char last_move = s[0];  // First move is always played

    for (int i = 1; i < n; ++i) {
        if (!is_winning_move(last_move, s[i])) {
            ++ans;
            last_move = s[i];
        }
    }

    cout << ans << endl;

    return 0;
}
```
