#include <iostream> // cout, endl, cin

#include<iomanip>
using namespace std;

int main(){
    int n;
    string s;
    cin >> n >> s;
    
    int wins = 0;
    char lastMove = 'X'; // Initial value does not matter since we will update it
    
    for (int i = 0; i < n; ++i) {
        if (lastMove == 'X' || s[i] == lastMove || (s[i] == 'R' && lastMove == 'S') || (s[i] == 'S' && lastMove == 'P') || (s[i] == 'P' && lastMove == 'R')) {
            // If Takahashi can play the same move or win against Aoki's move without losing, he plays the same move
            lastMove = s[i];
        } else {
            // Otherwise, he plays the move that would beat Aoki's current move
            if (s[i] == 'R') lastMove = 'P';
            else if (s[i] == 'S') lastMove = 'R';
            else lastMove = 'S';
            ++wins; // Increment the win count as Takahashi has now won
        }
    }

    cout << wins;
}
