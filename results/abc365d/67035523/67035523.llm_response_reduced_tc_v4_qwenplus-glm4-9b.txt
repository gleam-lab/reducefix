#include <iostream>
#include <string>

using namespace std;

int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100; // Assuming 'X' is not a valid character for comparison
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    if (n == 0) {
        cout << 0;
        return 0;
    }

    char prev = s[0];
    int ans = 0;
    for (int i = 1; i < n; ++i) {
        if (cmp(s[i], prev) != 0) {
            ans++;
            prev = s[i];
        }
    }

    cout << ans;
    return 0;
}
