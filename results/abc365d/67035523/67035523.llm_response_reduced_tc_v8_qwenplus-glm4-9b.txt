#include <iostream>
#include <string>
using namespace std;

int cmp(char x, char y) {
    if (x == 'P') return y == 'S' ? 1 : -1;
    if (x == 'R') return y == 'P' ? 1 : -1;
    if (x == 'S') return y == 'R' ? 1 : -1;
    return -100; // 'X' never wins
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // 'X' never wins
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    int prev = 3; //initialize to a non-winnable state
    int wins = 0;
    
    for (int i = 0; i < n; i++) {
        if (cmp(s[i], char('X' + prev)) == 1) { // Check if current move wins
            wins++;
            prev = win(s[i]); // Update the winning character
        } else {
            prev = (int)s[i]; // Update the character if it doesn't win
        }
    }
    
    cout << wins;
    return 0;
}
