```cpp
#include <iostream>
#include <string>
using namespace std;

// Function to determine the winning move
char win(char x) {
    if (x == 'R') return 'P';  // Paper beats Rock
    if (x == 'S') return 'R';  // Rock beats Scissors
    if (x == 'P') return 'S';  // Scissors beats Paper
    return 'X';  // Invalid case
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int ans = 0;
    char last_move = 'X';  // Initially no move played

    for (int i = 0; i < n; ++i) {
        char current_move = s[i];
        char desired_move = win(current_move);

        // Check if we can play the desired move based on the previous move
        if (last_move != desired_move) {
            ans++;
            last_move = desired_move;
        } else {
            // If not, just pass the turn with the same move
            last_move = current_move;
        }
    }

    cout << ans << endl;
    return 0;
}
```
