```cpp
#include <iostream>
#include <string>
using namespace std;

// Function to determine the winner between two moves
// Returns 1 if x beats y, -1 if x loses to y, 0 if draw
int cmp(char x, char y) {
    if (x == y) return 0;
    if ((x == 'R' && y == 'S') || (x == 'S' && y == 'P') || (x == 'P' && y == 'R'))
        return 1;
    return -1;
}

// Function to return the winning move against x
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    return 'S';
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int ans = 0;
    char current_move = 'X';  // Start with an invalid move

    for (char c : s) {
        // If our current move doesn't beat the opponent's move, switch to the winning move
        if (cmp(current_move, c) != 1) {
            current_move = win(c);
            ans++;
        }
    }

    cout << ans << endl;
    return 0;
}
```
