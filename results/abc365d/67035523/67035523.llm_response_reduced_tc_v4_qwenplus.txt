using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    int ans = 1; // Start with first move always counted
    char last_played = win(s[0]); // First action must be the winning move against first opponent
    
    for (int i = 1; i < n; i++) {
        char needed = win(s[i]);
        if (needed != last_played) {
            ans++;
            last_played = needed;
        }
        // If same as last played, we can use the same action, no new move needed
    }
    
    cout << ans << endl;
    return 0;
}
