#include <iostream>
#include <string>
using namespace std;

// Function to determine the outcome of a single match
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return 0; // Should never reach here
}

// Returns the winning move against given move
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Invalid case
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    int max_wins = 0;
    char current_move = 'X'; // Start with invalid move

    for (int i = 0; i < n; ++i) {
        char opponent_move = s[i];
        char next_move = win(opponent_move);

        if (current_move != next_move) {
            max_wins++;
            current_move = next_move;
        } else {
            // We need to change strategy if we can
            // Try to find a better move that doesn't repeat
            char alt_move1 = (next_move == 'R') ? 'P' : 'R';
            char alt_move2 = (next_move == 'R') ? 'S' : 'R';

            if (cmp(alt_move1, opponent_move) > 0) {
                current_move = alt_move1;
            } else if (cmp(alt_move2, opponent_move) > 0) {
                current_move = alt_move2;
            } else {
                // Stay with current move if no better option
            }
        }
    }

    cout << max_wins << endl;
    return 0;
}
