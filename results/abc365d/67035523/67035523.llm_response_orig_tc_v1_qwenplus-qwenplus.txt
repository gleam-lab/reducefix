#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <unordered_map>
#include <cmath>

using namespace std;

// Function to determine the winner between two moves
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100; // for placeholder 'X'
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // shouldn't happen
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // We will simulate the game optimally
    // Our strategy is to greedily play the move that beats the current opponent's move,
    // but if we just did that, we'd violate the no-consecutive-same rule

    vector<char> our_move(n);
    int ans = 0;

    char last_played = 'X'; // initially different from any valid move

    for (int i = 0; i < n; ++i) {
        char to_play = win(s[i]); // what beats opponent's move

        if (to_play != last_played) {
            ans++;
            last_played = to_play;
        } else {
            // We played same as previous, so we need to find a different move
            // Since we are trying to maximize wins, we try another move that doesn't lose
            // If possible, choose a non-losing move different from last move
            if (cmp('R', s[i]) >= 0 && 'R' != last_played) {
                last_played = 'R';
                ans++;
            } else if (cmp('P', s[i]) >= 0 && 'P' != last_played) {
                last_played = 'P';
                ans++;
            } else if (cmp('S', s[i]) >= 0 && 'S' != last_played) {
                last_played = 'S';
                ans++;
            } else {
                // No way to avoid repeating or losing, repeat and not gain
                // So we don't increment answer
            }
        }
    }

    cout << ans << endl;
}
