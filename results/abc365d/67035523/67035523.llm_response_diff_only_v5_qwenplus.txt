```cpp
#include <iostream>
#include <string>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    int ans = 0;
    char current_held = 'X'; // No symbol initially held
    
    for (int i = 0; i < n; i++) {
        char desired = win(s[i]);
        
        if (current_held != desired) {
            ans++;
            current_held = desired;
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
