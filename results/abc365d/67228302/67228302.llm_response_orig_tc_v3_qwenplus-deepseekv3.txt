    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;

    cin >> s;
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    int total = 1;
    char last_move = beats[s[0]];
    for (int i = 1; i < n; i++) {
        if (beats[s[i]] != last_move) {
            total++;
            last_move = beats[s[i]];
        }
    }
    cout << total << endl;

    return 0;
}
