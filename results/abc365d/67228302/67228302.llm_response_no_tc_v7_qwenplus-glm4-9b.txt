#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007


int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int total = 1;
    char last_move = 'N'; // Initialize to 'N' or any neutral character not in the beats map
    for (int i = 0; i < n; i++) {
        if (beats.find(s[i]) == beats.end()) {
            last_move = s[i]; // If s[i] is not in the beats map, start a new sequence
        } else if (beats[s[i]] != last_move) {
            last_move = beats[s[i]]; // Update last_move to the beat of the current move
            total++;
        }
    }
    cout << total << endl;





    return 0;

}
