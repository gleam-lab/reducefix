
The logic of your code reads the first character of the string and sets it as the `last_move`. Then, in the loop, it checks if the current character `s[i]` beats the `last_move`. If it does, it increments the `total` and updates the `last_move`. This is not correct because if the current character `s[i]` is the same as `s[i-1]`, it should not increment `total`.

```cpp
#include <bits/stdc++.h>

#define endl "\n"
#define MOD 1000000007


int main() {

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int total = 1;
    char last_move = beats[s[0]];
    for (int i = 1; i < n; i++) {
        if (s[i] == s[i-1]) {
            last_move = s[i];
            continue;
        }
        if (beats[s[i]] == last_move) {
            last_move = s[i];
            continue;
        }
        total++;
        last_move = s[i];
    }
    cout << total << endl;





    return 0;

}

```

This corrected code will increment `total` every time you find a new character in the string that does not beat the `last_move`. If the current character is the same as the previous one, it just updates the `last_move` without incrementing `total`.
