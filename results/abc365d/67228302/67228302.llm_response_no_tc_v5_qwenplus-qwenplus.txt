```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    int total = 1; // At least one move needed
    char last_choice = beats[s[0]]; // Our first choice

    for (int i = 1; i < n; i++) {
        char current_beat = beats[s[i]];
        if (current_beat == last_choice) {
            // Same move can be used
            continue;
        } else {
            // Need to change move
            total++;
            last_choice = current_beat;
        }
    }

    cout << total << endl;

    return 0;
}
```
