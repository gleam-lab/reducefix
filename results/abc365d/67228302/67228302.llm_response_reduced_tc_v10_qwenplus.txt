```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int total = 1;
    char last_win = beats[s[0]]; // what we played to beat opponent's first move

    for (int i = 1; i < n; i++) {
        char current_beat = beats[s[i]];
        // If our previous winning move also beats current opponent move
        if (last_win == current_beat) {
            // We can use same strategy
            continue;
        } else {
            // Need to change strategy
            total++;
            last_win = current_beat;
        }
    }

    cout << total << endl;

    return 0;
}
```
