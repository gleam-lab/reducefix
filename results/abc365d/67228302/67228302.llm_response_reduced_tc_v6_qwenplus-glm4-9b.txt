#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007


int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    int total = 0;
    char last_move = 'A';  // Initialize with a value that is not in the beats map
    total += (beats[s[0]] != last_move);  // Check if the first character beats the last move
    last_move = s[0];

    for (int i = 1; i < n; i++) {
        if (beats[s[i]] == last_move) {
            continue;  // Skip if the current character cannot beat the last move
        }
        last_move = s[i];
        total += 1;  // Increment total when a new move is made
    }

    cout << total << endl;

    return 0;

}
