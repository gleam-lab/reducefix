```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int total = 1;
    char last_move = s[0]; // The move that was played in the current round

    for (int i = 1; i < n; i++) {
        // If current opponent's move is the same as the one we're countering, continue current sequence
        if (beats[s[i]] == last_move) {
            continue;
        }
        // Otherwise, we need a new strategy (new round)
        total++;
        last_move = s[i];
    }

    cout << total << endl;

    return 0;
}
```
