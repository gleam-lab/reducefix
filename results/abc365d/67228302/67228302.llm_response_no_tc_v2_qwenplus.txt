```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'S'}, {'P', 'R'}, {'S', 'P'}}; // Corrected: what each move beats
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int total = 1;
    char opponent_last = s[0]; // Last move by opponent
    
    for (int i = 1; i < n; i++) {
        // We want to find the minimum number of players such that each can beat the previous player
        // A player with move X can beat a player with move Y if X beats Y
        
        // Current opponent move is s[i]
        // Check if our current sequence's last player can beat this opponent
        if (beats[opponent_last] != s[i]) {
            // If not, we need a new player in our team
            total++;
        }
        // Update last opponent move
        opponent_last = s[i];
    }
    
    cout << total << endl;
    return 0;
}
```
