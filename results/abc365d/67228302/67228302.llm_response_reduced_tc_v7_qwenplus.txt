```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    int total = 1;
    char opponent_move = s[0];
    char my_move = beats[opponent_move];

    for (int i = 1; i < n; i++) {
        char current_opponent = s[i];
        // If my current move beats the previous opponent's move but loses to current,
        // or is not optimal, we may need a new strategy
        if (beats[my_move] == current_opponent) {
            // My current move loses to opponent's current move
            // Need to change strategy: choose move that beats current opponent
            my_move = beats[current_opponent];
            total++;
        }
        // Otherwise, continue with same move (it either wins or draws)
        // No need to change strategy
    }

    cout << total << endl;

    return 0;
}
```
