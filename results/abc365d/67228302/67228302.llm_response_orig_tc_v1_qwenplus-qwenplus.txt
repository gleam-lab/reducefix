#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    int total = 1;
    vector<char> moves;
    moves.push_back(beats[s[0]]);

    for (int i = 1; i < n; ++i) {
        char curr_move = beats[s[i]];
        if (curr_move == moves.back()) {
            continue;
        }
        // Check if current move can be merged with previous one
        if (!moves.empty() && beats[curr_move] == moves.back()) {
            moves.pop_back();
        } else {
            moves.push_back(curr_move);
        }
    }

    cout << moves.size() << endl;

    return 0;
}
