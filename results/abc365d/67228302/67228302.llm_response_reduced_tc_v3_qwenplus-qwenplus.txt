```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;

    int total = 1;
    char last_move = beats[s[0]];

    for (int i = 1; i < n; i++) {
        char current_beat = beats[s[i]];
        if (current_beat == last_move) {
            // If the current beat is same as last move, we continue without incrementing
            // but update last_move to current character's beaten one
            last_move = current_beat;
        } else {
            // If different, we increase the count and update last_move
            total++;
            last_move = current_beat;
        }
    }

    cout << total << endl;

    return 0;
}
```
