```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;
    cin >> s;
    
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    int total = 1;
    char last_move = s[0]; // The actual move made by opponent
    
    for (int i = 1; i < n; i++) {
        // Check if current opponent's move beats our previous winning move
        if (beats[last_move] == s[i]) {
            total++;
        }
        last_move = s[i];
    }
    
    cout << total << endl;
    return 0;
}
```
