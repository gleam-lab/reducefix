#include <bits/stdc++.h>
using namespace std;
#define endl "\n"
#define MOD 1000000007

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    map<char, char> beats = {{'R', 'P'}, {'P', 'S'}, {'S', 'R'}};
    int n;
    cin >> n;
    string s;

    // Check for invalid input
    if (n < 1 || n > 100000) {
        cout << "Invalid input" << endl;
        return 0;
    }

    cin >> s;
    int total = 1;
    char last_move = beats[s[0]];
    for (int i = 1; i < n; i++) {
        char current_move = s[i];
        if (beats[current_move] == last_move) {
            continue; // This should never happen as per the problem statement
        }
        last_move = current_move;
        total++;
    }
    cout << total << endl;

    return 0;
}
