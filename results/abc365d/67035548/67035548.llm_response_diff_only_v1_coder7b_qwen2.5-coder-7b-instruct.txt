// Function to compare two moves
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    if (x == 'X') return -100; // Assuming 'X' is not a valid move for comparison
}

// Function to determine the winning move
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S'; 
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    int ans = 0;
    char prev = 'A'; // Initial value that is not 'R', 'P', or 'S'
    
    // Calculate the number of changes needed from left to right
    for (int i = 0; i < n; ++i) {
        if (prev != win(s[i])) {
            ++ans;
            prev = win(s[i]);
        } else {
            prev = s[i];
        }
    }
    
    prev = 'A'; // Reset previous character for the second pass
    
    // Calculate the number of changes needed from right to left
    int sub = 0;
    for (int i = n - 1; i >= 0; --i) {
        if (prev != win(s[i])) {
            ++sub;
            prev = win(s[i]);
        } else {
            prev = s[i];
        }
    }
    
    // Output the maximum of the two calculated values
    cout << max(ans, sub);
    
    return 0;
}
