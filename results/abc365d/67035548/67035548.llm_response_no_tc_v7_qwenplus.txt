#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    // Try both possible starting moves: either play the winning move against first opponent move
    // or play the same as opponent's first move
    int max_score = 0;
    
    // Case 1: Start by playing the winning move against s[0]
    char curr_move = win(s[0]);
    int score1 = 1; // First move counts
    for (int i = 1; i < n; i++) {
        if (curr_move != win(s[i])) {
            // Need to change our move
            curr_move = win(s[i]);
            score1++;
        }
        // Otherwise keep same move, no increment
    }
    
    // Case 2: Start by playing the same as s[0] (not countering)
    curr_move = s[0];
    int score2 = 1; // First move counts
    for (int i = 1; i < n; i++) {
        if (curr_move != win(s[i])) {
            // Need to change our move
            curr_move = win(s[i]);
            score2++;
        }
    }
    
    // The answer is the maximum of these two strategies
    cout << max(score1, score2) << endl;
    
    return 0;
}
