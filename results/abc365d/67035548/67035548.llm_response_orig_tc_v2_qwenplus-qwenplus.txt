```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

// Determines the winning move against the given move
char winning_move(char c) {
    if (c == 'R') return 'P';
    if (c == 'P') return 'S';
    if (c == 'S') return 'R';
    return 'X'; // Invalid, shouldn't happen
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Forward pass: greedy from start
    int count1 = 0;
    char last_move = 'X';
    for (int i = 0; i < n; ++i) {
        char desired = winning_move(s[i]);
        if (desired != last_move) {
            ++count1;
            last_move = desired;
        }
    }

    // Backward pass: greedy from end
    int count2 = 0;
    last_move = 'X';
    for (int i = n - 1; i >= 0; --i) {
        char desired = winning_move(s[i]);
        if (desired != last_move) {
            ++count2;
            last_move = desired;
        }
    }

    cout << max(count1, count2) << endl;
    return 0;
}
```
