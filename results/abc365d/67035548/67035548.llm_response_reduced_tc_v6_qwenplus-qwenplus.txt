#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

// Returns 1 if x beats y, -1 if x loses to y, 0 if tie
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100; // Shouldn't reach here
}

char winning_move(char c) {
    if (c == 'R') return 'P';
    if (c == 'P') return 'S';
    if (c == 'S') return 'R';
    return 'X'; // Invalid input
}

int solve(const string& s) {
    int n = s.size();
    vector<int> dp(n + 1, 0);
    
    // We track the best we can do by ending with a particular move at each position
    // States: best[move] = max score achievable ending with 'move' at current position
    // where move can be P, R, S
    
    for (int i = 0; i < n; ++i) {
        char c = s[i];
        char beat = winning_move(c);
        
        // For current position i, try playing the move that beats s[i]
        // and see what's the best previous score we can build upon
        // We can transition from any previous move, as long as it's not the same as beat
        int max_prev = 0;
        for (int j = i - 1; j >= max(0, i - 2); --j) {
            if (s[j] != beat) {
                max_prev = max(max_prev, dp[j]);
            }
        }
        
        dp[i + 1] = max_prev + 1;
    }
    
    return *max_element(dp.begin(), dp.end());
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    cout << solve(s) << endl;
    return 0;
}
