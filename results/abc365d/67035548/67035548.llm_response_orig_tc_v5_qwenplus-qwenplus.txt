#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Placeholder for invalid input
}

int compute_max_strategy(const string& s) {
    int n = s.size();
    int ans = 0;
    char prev = 'X'; // Initially not matching anything

    for (int i = 0; i < n; ++i) {
        char best_move = win(s[i]);
        if (best_move != prev) {
            ans++;
            prev = best_move;
        } else {
            prev = s[i]; // Cannot win, so just continue with current move
        }
    }

    return ans;
}

int compute_reverse_strategy(const string& s) {
    int n = s.size();
    int ans = 0;
    char prev = 'X';

    for (int i = n - 1; i >= 0; --i) {
        char best_move = win(s[i]);
        if (best_move != prev) {
            ans++;
            prev = best_move;
        } else {
            prev = s[i];
        }
    }

    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    string s;
    cin >> n >> s;

    int forward = compute_max_strategy(s);
    int backward = compute_reverse_strategy(s);

    cout << max(forward, backward) << endl;

    return 0;
}
