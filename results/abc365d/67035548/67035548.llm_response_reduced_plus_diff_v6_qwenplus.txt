```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // shouldn't happen
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // dp[i][c] = minimum number of changes needed for prefix [0:i] ending with move c
    // But we can do greedy: at each step, we can either continue the previous winning move or switch
    // Actually, we need to simulate both forward and backward and take max

    auto solve = [&](const string& seq) -> int {
        if (seq.empty()) return 0;
        char last = win(seq[0]); // first must be beaten by our move
        int count = 1;
        for (int i = 1; i < seq.size(); i++) {
            char need = win(seq[i]);
            if (need != last) {
                count++;
                last = need;
            } else {
                last = seq[i]; // opponent repeats, we can repeat too
            }
        }
        return count;
    };

    // Try forward and backward sequences
    int forward = solve(s);
    
    string reversed = s;
    reverse(reversed.begin(), reversed.end());
    int backward = solve(reversed);

    cout << max(forward, backward) << endl;
    
    return 0;
}
```
