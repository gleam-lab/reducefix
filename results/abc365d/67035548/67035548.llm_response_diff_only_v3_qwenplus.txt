#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

char beats(char c) {
    if (c == 'R') return 'P';
    if (c == 'S') return 'R';
    if (c == 'P') return 'S';
    return c;
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    // Forward pass: start with first character's winner
    int forward = 0;
    char current = beats(s[0]);
    forward++; // We can always beat the first one
    
    for (int i = 1; i < n; i++) {
        if (beats(s[i]) == current) {
            // Continue the streak with same winning move
            continue;
        } else {
            // Need to change our move
            current = beats(s[i]);
            forward++;
        }
    }
    
    // Backward pass: start from the end
    int backward = 0;
    current = beats(s[n-1]);
    backward++; // We can always beat the last one
    
    for (int i = n-2; i >= 0; i--) {
        if (beats(s[i]) == current) {
            // Continue the streak with same winning move
            continue;
        } else {
            // Need to change our move
            current = beats(s[i]);
            backward++;
        }
    }
    
    cout << max(forward, backward) << endl;
    
    return 0;
}
