#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

// Determine outcome of a single RSP round: 1 = win, 0 = tie, -1 = loss
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    return -100; // Should never reach here
}

// Return the winning move against a given move
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Should never reach here
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Forward pass
    int forward_wins = 0;
    char current_move = 'X'; // No move yet
    for (int i = 0; i < n; ++i) {
        char opponent_move = s[i];
        if (cmp(win(opponent_move), opponent_move) == 1) {
            // If our move beats opponent's move
            forward_wins++;
            current_move = win(opponent_move);
        } else {
            // If not, pick the move that beats this opponent's move
            current_move = win(opponent_move);
        }
    }

    // Backward pass
    int backward_wins = 0;
    current_move = 'X'; // No move yet
    for (int i = n - 1; i >= 0; --i) {
        char opponent_move = s[i];
        if (cmp(win(opponent_move), opponent_move) == 1) {
            backward_wins++;
            current_move = win(opponent_move);
        } else {
            current_move = win(opponent_move);
        }
    }

    cout << max(forward_wins, backward_wins) << endl;
    
    return 0;
}
