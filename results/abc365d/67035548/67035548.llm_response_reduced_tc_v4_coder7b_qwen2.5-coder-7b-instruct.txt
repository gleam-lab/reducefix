#include<iomanip>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n;
    string s;
    cin >> n >> s;

    char last = 'X';
    int wins = 0;

    for (int i = 0; i < n; ++i) {
        if (last == 'X' || s[i] != last) {
            ++wins;
            last = s[i] == 'R' ? 'P' : s[i] == 'P' ? 'S' : 'R';
        }
    }

    cout << wins << '\n';

    return 0;
}
