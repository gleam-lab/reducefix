#include <iostream>
#include <string>
using namespace std;

// Function to determine the winner of a single RPS match
// Returns 1 if player1 wins, -1 if player2 wins, 0 for draw
int rpsBattle(char player1, char player2) {
    if (player1 == player2) return 0;
    if ((player1 == 'R' && player2 == 'S') ||
        (player1 == 'S' && player2 == 'P') ||
        (player1 == 'P' && player2 == 'R')) {
        return 1; // Player 1 wins
    }
    return -1; // Player 2 wins
}

// Function to return the winning move against given move
char winningMove(char move) {
    if (move == 'R') return 'P';
    if (move == 'P') return 'S';
    return 'R'; // S
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Strategy 1: Greedy from start
    int strategy1 = 0;
    char last_move = 'X'; // No move yet
    
    for (int i = 0; i < n; ++i) {
        char opponent_move = s[i];
        char current_move = winningMove(opponent_move);
        
        if (current_move != last_move) {
            strategy1++;
            last_move = current_move;
        }
    }

    // Strategy 2: Greedy from end
    int strategy2 = 0;
    last_move = 'X'; // Reset
    
    for (int i = n - 1; i >= 0; --i) {
        char opponent_move = s[i];
        char current_move = winningMove(opponent_move);
        
        if (current_move != last_move) {
            strategy2++;
            last_move = current_move;
        }
    }

    cout << max(strategy1, strategy2) << endl;
    return 0;
}
