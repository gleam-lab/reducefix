```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    // Calculate forward simulation
    int forward = 0;
    char current = 'X'; // No previous action
    
    for (int i = 0; i < n; i++) {
        char winner_move = win(s[i]);
        if (current != winner_move) {
            forward++;
            current = winner_move;
        } else {
            current = s[i]; // Continue with the same move
        }
    }
    
    // Calculate backward simulation
    int backward = 0;
    current = 'X'; // Reset for backward pass
    
    for (int i = n - 1; i >= 0; i--) {
        char winner_move = win(s[i]);
        if (current != winner_move) {
            backward++;
            current = winner_move;
        } else {
            current = s[i];
        }
    }
    
    cout << max(forward, backward) << endl;
    
    return 0;
}
```
