#include <iostream>
#include <string>
using namespace std;

// Function to determine the outcome of a move against the previous one
int outcome(char current, char next) {
    if (current == next) return 0; // same move, no win
    if ((current == 'R' && next == 'S') ||
        (current == 'S' && next == 'P') ||
        (current == 'P' && next == 'R')) {
        return 1; // current beats next, win
    }
    return -1; // next beats current, loss
}

// Function to get the winning move
char winningMove(char opponent) {
    if (opponent == 'R') return 'P';
    if (opponent == 'S') return 'R';
    return 'S'; // P -> S
}

int solve(const string& s) {
    int n = s.size();
    int count = 1;
    char last_move = winningMove(s[0]);

    for (int i = 1; i < n; ++i) {
        char desired_move = winningMove(s[i]);
        if (desired_move != last_move) {
            count++;
            last_move = desired_move;
        }
    }

    return count;
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Solve forward and backward to find max number of matches we can win
    int forward = solve(s);

    // Reverse the string and solve again
    reverse(s.begin(), s.end());
    int backward = solve(s);

    cout << max(forward, backward) << endl;

    return 0;
}
