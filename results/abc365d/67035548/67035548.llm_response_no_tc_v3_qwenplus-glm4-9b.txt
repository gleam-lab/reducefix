#include <iostream>
#include <string>
#include <vector>
using namespace std;

int cmp(char x, char y) {
    if (x == 'P') return y == 'R' ? -1 : 1;
    if (x == 'R') return y == 'P' ? 1 : -1;
    if (x == 'S') return y == 'R' ? -1 : 1;
    return -100; // Assuming 'X' is not a valid input for comparison
}

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X'; // Assuming 'X' is not a valid input for winning
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    vector<int> left(n + 1, -1);
    vector<int> right(n + 1, -1);

    int maxLeft = 0, maxRight = 0;
    int currentLeft = 0, currentRight = 0;

    // Calculate max sequence from left
    for (int i = 0; i < n; ++i) {
        if (s[i] == 'X') {
            currentLeft = 0;
        } else {
            currentLeft++;
            maxLeft = max(maxLeft, currentLeft);
        }
    }

    // Calculate max sequence from right
    for (int i = n - 1; i >= 0; --i) {
        if (s[i] == 'X') {
            currentRight = 0;
        } else {
            currentRight++;
            maxRight = max(maxRight, currentRight);
        }
    }

    // Calculate max sequence from both sides
    for (int i = 0; i < n; ++i) {
        if (s[i] == 'X') {
            left[i + 1] = left[i];
            right[i] = right[i + 1];
        } else {
            left[i + 1] = max(left[i], right[i] + 1);
            right[i] = max(right[i], left[i] + 1);
        }
    }

    // Calculate the answer
    int ans = 0;
    for (int i = 0; i <= n; ++i) {
        ans = max(ans, left[i] + right[i] - 1);
    }

    cout << ans << endl;

    return 0;
}
