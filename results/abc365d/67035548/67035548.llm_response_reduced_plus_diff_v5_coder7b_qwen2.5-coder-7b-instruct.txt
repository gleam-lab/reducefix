```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    string s;
    cin >> n >> s;

    int wins = 0;
    char lastWin = 'A';

    for (int i = 0; i < n; ++i) {
        if (s[i] == 'R' && lastWin != 'P') {
            wins++;
            lastWin = 'P';
        } else if (s[i] == 'P' && lastWin != 'S') {
            wins++;
            lastWin = 'S';
        } else if (s[i] == 'S' && lastWin != 'R') {
            wins++;
            lastWin = 'R';
        } else {
            lastWin = s[i];
        }
    }

    cout << wins << endl;
    return 0;
}
```
