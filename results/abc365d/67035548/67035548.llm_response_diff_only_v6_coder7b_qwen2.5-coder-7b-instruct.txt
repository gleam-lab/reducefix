#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

// Function to determine the result of a game between two choices
int cmp(char x, char y) {
    if (x == 'P') {
        if (y == 'P') return 0;
        if (y == 'R') return 1;
        if (y == 'S') return -1;
    }
    if (x == 'R') {
        if (y == 'P') return -1;
        if (y == 'R') return 0;
        if (y == 'S') return 1;
    }
    if (x == 'S') {
        if (y == 'P') return 1;
        if (y == 'R') return -1;
        if (y == 'S') return 0;
    }
    if (x == 'X') return -100; // Assuming 'X' is a special character or marker
}

// Function to determine the winning move for a given move
char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    // Initialize variables
    int siz[300009] = {0};
    char rsp[300009];
    int pos = 0;
    int c = 1;
    siz[n] = 'X';
    char prev = 'A';
    int ans = 0;
    
    // First pass to count the number of moves needed to change from the previous winning move
    for (int i = 0; i < n; i++) {
        if (prev != win(s[i])) {
            ans++;
            prev = win(s[i]);
        } else {
            prev = s[i];
        }
    }
    
    // Reset variables for the second pass
    prev = 'A';
    int sub = 0;
    
    // Second pass to count the number of moves needed to change from the previous losing move
    for (int i = n - 1; i >= 0; i--) {
        if (prev != win(s[i])) {
            sub++;
            prev = win(s[i]);
        } else {
            prev = s[i];
        }
    }
    
    // Output the maximum of the two counts
    cout << max(ans, sub);
    return 0;
}
