char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int solve(const string& s) {
    int n = s.size();
    if (n == 0) return 0;
    
    // Try all possible starting gestures
    vector<char> gestures = {'R', 'P', 'S'};
    int max_wins = 0;
    
    for (char start : gestures) {
        char current_gesture = start;
        int wins = 0;
        
        for (int i = 0; i < n; i++) {
            if (current_gesture == win(s[i])) {
                wins++;
                // After winning, opponent will play the gesture that beats our current one
                // So we switch to the one that beats what they'll play
                // If they played x, they'll now play win(x), so we play win(win(x))
                current_gesture = win(win(s[i]));
            } else {
                // We lose or tie, so we adapt to beat their current move
                current_gesture = win(s[i]);
            }
        }
        max_wins = max(max_wins, wins);
    }
    
    return max_wins;
}

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    cout << solve(s) << endl;
    
    return 0;
}
