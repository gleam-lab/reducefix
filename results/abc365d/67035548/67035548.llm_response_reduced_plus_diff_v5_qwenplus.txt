```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

char win(char x) {
    if (x == 'R') return 'P';
    if (x == 'S') return 'R';
    if (x == 'P') return 'S';
    return 'X';
}

int main() {
    int n;
    string s;
    cin >> n >> s;

    // Forward pass: start with first winner
    vector<char> forward;
    char current = win(s[0]);
    forward.push_back(current);
    
    for (int i = 1; i < n; i++) {
        if (win(s[i]) != current) {
            current = win(s[i]);
            forward.push_back(current);
        }
    }

    // Backward pass: start from end
    vector<char> backward;
    current = win(s[n-1]);
    backward.push_back(current);
    
    for (int i = n-2; i >= 0; i--) {
        if (win(s[i]) != current) {
            current = win(s[i]);
            backward.push_back(current);
        }
    }

    // Answer is maximum of the two sequences
    cout << max(forward.size(), backward.size()) << endl;
    
    return 0;
}
```
