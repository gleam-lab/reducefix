ll gcd(ll a, ll b)
{
    while (b)
    {
        a %= b;
        swap(a, b);
    }
    return a;
}
ll lcm(ll a, ll b)
{
    return a / gcd(a, b) * b;
}

{
    ll n;

    string s;

    ll ans = 0;
    ll cnt = 1; // Start with count as 1
    for (int i = 1; i < n; i++)
    {
        if (s[i] == s[i - 1])
        {
            cnt++; // If same character, increment count
        }
        else
        {
            ans += (cnt + 1) / 2; // Calculate answer for previous sequence
            cnt = 1; // Reset count for new sequence
        }
    }
    ans += (cnt + 1) / 2; // Calculate answer for last sequence
    cout << ans << "\n";
}
