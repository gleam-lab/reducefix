void solve()
{
    ll n;
    cin >> n;
    string s;
    cin >> s;
    
    // Count the number of contiguous segments of same characters
    ll ans = 0;
    ll cnt = 1; // Start with first character
    
    for (int i = 1; i < n; i++)
    {
        if (s[i] == s[i-1])
        {
            cnt++;
        }
        else
        {
            // For each segment, we need ceil(cnt/2) operations
            ans += (cnt + 1) / 2;
            cnt = 1;
        }
    }
    // Don't forget the last segment
    ans += (cnt + 1) / 2;
    
    cout << ans << "\n";
}
