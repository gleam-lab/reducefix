void solve() {
    ll n;
    cin >> n;
    string str;
    cin >> str;

    if (n == 0) {
        cout << 0 << "\n";
        return;
    }

    vector<ll> groups;
    ll count = 1;

    for (ll i = 1; i < n; ++i) {
        if (str[i] == str[i - 1]) {
            count++;
        } else {
            groups.push_back(count);
            count = 1;
        }
    }
    groups.push_back(count);

    ll total_moves = 0;
    for (ll g : groups) {
        total_moves += (g + 1) / 2;
    }

    // Optimization: reduce moves when we have alternating single-character groups
    for (size_t i = 1; i < groups.size() - 1; ++i) {
        if (groups[i] == 1 && groups[i - 1] > 1 && groups[i + 1] > 1) {
            total_moves--;
        }
    }

    cout << total_moves << "\n";
}
