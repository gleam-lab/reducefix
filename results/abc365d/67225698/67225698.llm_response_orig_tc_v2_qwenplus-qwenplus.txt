void solve() {
    ll n;
    string s;
    cin >> n >> s;

    if (n == 0) {
        cout << 0 << "\n";
        return;
    }

    vll groups;
    ll cnt = 1;
    for (ll i = 1; i < n; ++i) {
        if (s[i] == s[i - 1]) {
            cnt++;
        } else {
            groups.push_back(cnt);
            cnt = 1;
        }
    }
    groups.push_back(cnt);

    ll ans = 0;
    for (ll g : groups) {
        ans += (g + 1) / 2;
    }

    // Optimization: reduce the count when there's a peak of single character surrounded by longer runs
    for (size_t i = 1; i < groups.size() - 1; ++i) {
        if (groups[i] == 1 && groups[i - 1] > 1 && groups[i + 1] > 1) {
            ans--;
        }
    }

    cout << ans << "\n";
}
