void solve()
{
    ll n;
    cin >> n;
    string s;
    cin >> s;

    vll groups;
    ll cnt = 1;
    ll total = 0;

    // Group consecutive characters and compute initial pairs
    for (ll i = 1; i < n; ++i)
    {
        if (s[i] == s[i - 1])
        {
            cnt++;
        }
        else
        {
            groups.push_back(cnt);
            total += (cnt + 1) / 2;
            cnt = 1;
        }
    }
    groups.push_back(cnt);
    total += (cnt + 1) / 2;

    // Apply special condition to reduce count where applicable
    for (ll i = 1; i < groups.size() - 1; ++i)
    {
        if (groups[i] == 1 && groups[i - 1] > 1 && groups[i + 1] > 1)
        {
            total -= 1;
        }
    }

    cout << total << "\n";
}
