   - It subtracts `1` from the answer if there are two consecutive blocks of characters.
4. **Output**: The function prints the computed answer.

### Issues in the Code:

1. The condition to subtract `1` from the answer is not properly handled because `cnt` goes to zero after every block. The new character starts and `cnt` becomes 1, which could be part of a block of more than one character.
2. The subtraction should only occur if the new character starts after exactly one character different from the previous one.

void solve()
{
    ll n;
    cin >> n;
    string s;
    cin >> s;
    ll ans = 0;
    ll cnt = 0;
    vll v;
    for (int i = 0; i < n; i++)
    {
        if (i == 0 || s[i] == s[i - 1])
        {
            cnt++;
        }
        else
        {
            ans += (cnt + 1) / 2;
            v.push_back(cnt);
            cnt = 1;
        }
    }
    ans += (cnt + 1) / 2;  // Add last block result
    v.push_back(cnt);

    // Subtract 1 from the answer if there are at least two blocks with more than one character
    for (int i = 1; i < v.size() - 1; i++)
    {
        if (v[i] > 1 && v[i - 1] > 1)
        {
            ans -= 1;
        }
    }
    cout << ans << "\n";
}
