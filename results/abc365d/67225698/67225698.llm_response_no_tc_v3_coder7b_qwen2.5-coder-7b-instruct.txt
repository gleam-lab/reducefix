#include <bits/stdc++.h>
using namespace std;

void solve() {
    ll n;
    cin >> n;
    string s;
    cin >> s;
    
    ll wins = 0;
    for (int i = 0; i < n; ++i) {
        char prev = (i > 0) ? s[i - 1] : '\0';
        char curr = s[i];
        
        if (prev != curr && ((curr == 'R' && prev == 'S') || (curr == 'S' && prev == 'P') || (curr == 'P' && prev == 'R'))) {
            ++wins;
        }
    }
    
    cout << wins << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t = 1;
    cin >> t;
    while (t--) {
        solve();
    }

    return 0;
}
