```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    vector<char> hand = {'R', 'P', 'S'};
    auto getWinner = [](char a, char b) -> int {
        if ((a == 'R' && b == 'S') || (a == 'P' && b == 'R') || (a == 'S' && b == 'P')) {
            return 1; // win
        } else if (a == b) {
            return 0; // tie
        } else {
            return -1; // lose
        }
    };
    
    int maxScore = 0;
    
    // Try all possible starting moves
    for (char firstMove : hand) {
        vector<char> available = hand;
        // Remove the first move from available options for future rounds
        available.erase(remove(available.begin(), available.end(), firstMove), available.end());
        
        int score = 0;
        
        // First round
        if (getWinner(firstMove, S[0]) == 1) {
            score++;
        }
        
        // Subsequent rounds
        for (int i = 1; i < N; i++) {
            char bestMove = available[0];
            int bestResult = getWinner(available[0], S[i]);
            
            // Check other available moves to find the best outcome
            for (char move : available) {
                int result = getWinner(move, S[i]);
                if (result > bestResult) {
                    bestResult = result;
                    bestMove = move;
                }
            }
            
            // Update score
            if (bestResult == 1) {
                score++;
            }
            
            // Update available moves by removing the one we just used
            vector<char> newAvailable;
            for (char move : available) {
                if (move != bestMove) {
                    newAvailable.push_back(move);
                }
            }
            available = newAvailable;
        }
        
        maxScore = max(maxScore, score);
    }
    
    cout << maxScore << endl;
    return 0;
}
```
