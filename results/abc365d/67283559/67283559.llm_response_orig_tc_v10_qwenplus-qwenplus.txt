```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string S;
    cin >> N >> S;

    // We will simulate both scenarios:
    // 1. Starting with all possible options
    // 2. Starting after skipping the first move

    auto simulate = [&](const vector<char>& initial_options) -> int {
        int score = 0;
        vector<char> current_options = initial_options;
        for (char c : S) {
            bool won = false;
            for (char choice : current_options) {
                if ((choice == 'R' && c == 'S') ||
                    (choice == 'P' && c == 'R') ||
                    (choice == 'S' && c == 'P')) {
                    score++;
                    current_options.clear();
                    for (char ch : {'R', 'P', 'S'}) {
                        if (ch != choice) current_options.push_back(ch);
                    }
                    won = true;
                    break;
                }
            }
            if (!won) {
                current_options.push_back(c); // Add opponent's move as our next option
                for (auto it = current_options.begin(); it != current_options.end();) {
                    if (*it == c) {
                        it = current_options.erase(it);
                    } else {
                        ++it;
                    }
                }
            }
        }
        return score;
    };

    // Try starting fresh
    int res1 = simulate({'R', 'P', 'S'});
    
    // Try skipping the first move
    int res2 = 0;
    if (N > 0) {
        vector<char> initial;
        char first_move = S[0];
        for (char c : {'R', 'P', 'S'}) {
            if (c != first_move) initial.push_back(c);
        }
        res2 = simulate(initial);
    }

    cout << max(res1, res2) << endl;
    return 0;
}
```
