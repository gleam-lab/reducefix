```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string S;
    cin >> N >> S;

    // Rock beats Scissors, Paper beats Rock, Scissors beats Paper
    auto wins = [](char a, char b) -> bool {
        return (a == 'R' && b == 'S') ||
               (a == 'P' && b == 'R') ||
               (a == 'S' && b == 'P');
    };

    // For each starting choice, simulate the game
    vector<char> choices = {'R', 'P', 'S'};
    int max_score = 0;

    for (char first_choice : choices) {
        int score = 0;
        char current_choice = first_choice;

        for (int i = 0; i < N; i++) {
            if (wins(current_choice, S[i])) {
                score++;
            }
            // Update strategy: choose what would beat the opponent's next move
            // But we can only choose from remaining options after elimination
            vector<char> available;
            for (char c : choices) {
                if (c != current_choice) {
                    available.push_back(c);
                }
            }

            // If only one option remains, we must use it
            if (available.size() == 1) {
                current_choice = available[0];
            } else {
                // Choose the one that would win against opponent's move if possible
                char best_choice = available[0];
                bool found_winner = false;
                for (char c : available) {
                    if (wins(c, S[i])) {
                        best_choice = c;
                        found_winner = true;
                        break;
                    }
                }
                if (!found_winner) {
                    // If none wins, avoid losing if possible
                    for (char c : available) {
                        if (c != S[i]) {  // doesn't lose (could tie)
                            best_choice = c;
                            break;
                        }
                    }
                }
                current_choice = best_choice;
            }
        }
        max_score = max(max_score, score);
    }

    cout << max_score << endl;
    return 0;
}
```
