#include <bits/stdc++.h>
using namespace std;

int solve(const string& S, const vector<char>& hand) {
    int N = S.size();
    vector<char> available = hand;
    int score = 0;
    
    for (char c : S) {
        bool won = false;
        for (size_t i = 0; i < available.size(); ++i) {
            char move = available[i];
            if ((move == 'R' && c == 'S') || 
                (move == 'P' && c == 'R') || 
                (move == 'S' && c == 'P')) {
                score++;
                available.clear();
                for (char h : hand) {
                    if (h != move) {
                        available.push_back(h);
                    }
                }
                won = true;
                break;
            }
        }
        if (!won) {
            char move = available[0];
            available.clear();
            for (char h : hand) {
                if (h != move) {
                    available.push_back(h);
                }
            }
        }
    }
    return score;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    vector<char> hand = {'R', 'P', 'S'};
    
    // Two options: start with all options or remove first character's best counter
    int res1 = solve(S, hand);
    
    // Try alternative strategy - force different first move
    int res2 = 0;
    vector<char> available;
    char first_move;
    
    // Choose the best move that beats S[0]
    if (S[0] == 'R') first_move = 'P';
    else if (S[0] == 'P') first_move = 'S';
    else first_move = 'R';
    
    available.clear();
    for (char h : hand) {
        if (h != first_move) {
            available.push_back(h);
        }
    }
    
    res2 = 1; // First move always wins
    for (int i = 1; i < N; ++i) {
        bool won = false;
        for (size_t j = 0; j < available.size(); ++j) {
            char move = available[j];
            if ((move == 'R' && S[i] == 'S') ||
                (move == 'P' && S[i] == 'R') ||
                (move == 'S' && S[i] == 'P')) {
                res2++;
                vector<char> new_available;
                for (char h : hand) {
                    if (h != move) {
                        new_available.push_back(h);
                    }
                }
                available = new_available;
                won = true;
                break;
            }
        }
        if (!won) {
            char move = available[0];
            vector<char> new_available;
            for (char h : hand) {
                if (h != move) {
                    new_available.push_back(h);
                }
            }
            available = new_available;
        }
    }
    
    cout << max(res1, res2);
    return 0;
}
