```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    if (N == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    vector<char> hand = {'R', 'P', 'S'};
    
    // Try all three possible starting moves
    int max_score = 0;
    
    for (char first_move : hand) {
        vector<char> available = hand;
        // Start with all moves available, then remove the one we use
        auto it = find(available.begin(), available.end(), first_move);
        if (it != available.end()) {
            available.erase(it);
        }
        
        int score = 0;
        char prev_use = first_move;
        
        // If first move beats first opponent move, score++
        if ((first_move == 'R' && S[0] == 'S') ||
            (first_move == 'P' && S[0] == 'R') ||
            (first_move == 'S' && S[0] == 'P')) {
            score++;
        }
        
        for (int i = 1; i < N; i++) {
            int best_score = 0;
            char best_move = available[0]; // default
            
            // Try each available move
            for (char move : available) {
                int current_score = 0;
                if ((move == 'R' && S[i] == 'S') ||
                    (move == 'P' && S[i] == 'R') ||
                    (move == 'S' && S[i] == 'P')) {
                    current_score = 1;
                }
                
                // Prefer winning if possible, otherwise pick any
                if (current_score > best_score) {
                    best_score = current_score;
                    best_move = move;
                }
            }
            
            score += best_score;
            prev_use = best_move;
            
            // Update available moves: remove the one we just used
            available.clear();
            for (char h : hand) {
                if (h != prev_use) {
                    available.push_back(h);
                }
            }
        }
        
        max_score = max(max_score, score);
    }
    
    cout << max_score << endl;
    return 0;
}
```
