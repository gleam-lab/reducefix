#include <bits/stdc++.h>
using namespace std;

int simulate(const string& S, const vector<char>& hand, char first_move) {
    int total = 0;
    char current_move = first_move;
    
    for (char c : S) {
        if ((current_move == 'R' && c == 'S') ||
            (current_move == 'P' && c == 'R') ||
            (current_move == 'S' && c == 'P')) {
            total++;
        }
        
        // Update next available moves
        vector<char> next_move;
        for (char h : hand) {
            if (h != current_move) {
                next_move.push_back(h);
            }
        }
        
        // Find the best move for next round
        if (next_move.size() == 1) {
            current_move = next_move[0];
        } else {
            // If multiple options, pick the one that beats opponent's next
            bool found = false;
            for (char n : next_move) {
                if ((n == 'R' && S[0] == 'S') ||
                    (n == 'P' && S[0] == 'R') ||
                    (n == 'S' && S[0] == 'P')) {
                    current_move = n;
                    found = true;
                    break;
                }
            }
            if (!found) {
                current_move = next_move[0]; // fallback
            }
        }
    }
    return total;
}

        vector<char> available = hand;
        
        // Remove first move from available
        available.erase(remove(available.begin(), available.end(), current), available.end());
        
        for (char c : S) {
            if ((current == 'R' && c == 'S') ||
                (current == 'P' && c == 'R') ||
                (current == 'S' && c == 'P')) {
                score++;
            }
            
            // Choose next move
            if (available.size() == 1) {
                current = available[0];
            } else if (!available.empty()) {
                // Prefer winning move if possible
                bool found = false;
                for (char n : available) {
                    if ((n == 'R' && S[0] == 'S') ||
                        (n == 'P' && S[0] == 'R') ||
                        (n == 'S' && S[0] == 'P')) {
                        current = n;
                        found = true;
                        break;
                    }
                }
                if (!found) {
                    current = available[0]; // fallback
                }
            }
            
            // Remove chosen move
            if (available.size() > 1) {
                available.erase(remove(available.begin(), available.end(), current), available.end());
            }
        }
        max_score = max(max_score, score);
    }
    
    // Case 2: Opponent chooses first move (we respond)
    int opponent_first = 0;

        if (first == 'R') current = 'P';
        else if (first == 'P') current = 'S';

            score++;
        }
        
        vector<char> available = hand;
        available.erase(remove(available.begin(), available.end(), current), available.end());

            char next_move;

            bool win_move_found = false;

                if ((a == 'R' && c == 'S') ||
