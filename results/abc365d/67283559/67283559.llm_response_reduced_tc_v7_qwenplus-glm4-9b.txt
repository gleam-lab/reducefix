
```cpp
#include <iostream>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    // Convert the string to a vector of characters for easier manipulation
    vector<char> moves(S.begin(), S.end());
    
    // Possible moves for Rock, Paper, Scissors
    char choices[] = {'R', 'P', 'S'};
    vector<char> valid_moves(3);
    
    // Calculate the first sequence of valid moves
    int score1 = 0;
    for (int i = 0; i < N; ++i) {
        valid_moves.fill('0');
        for (int j = 0; j < 3; ++j) {
            if ((choices[j] == 'R' && moves[i] == 'S') ||
                (choices[j] == 'P' && moves[i] == 'R') ||
                (choices[j] == 'S' && moves[i] == 'P')) {
                valid_moves[j] = choices[j];
                score1++;
                break;
            }
        }
    }
    
    // Calculate the second sequence of valid moves
    int score2 = 0;
    for (int i = 1; i < N; ++i) {
        valid_moves.fill('0');
        for (int j = 0; j < 3; ++j) {
            if ((choices[j] == 'R' && moves[i] == 'S') ||
                (choices[j] == 'P' && moves[i] == 'R') ||
                (choices[j] == 'S' && moves[i] == 'P')) {
                valid_moves[j] = choices[j];
                score2++;
                break;
            }
        }
    }
    
    // Output the maximum of the two scores
    cout << max(score1, score2) << endl;
    
    return 0;
}
```
