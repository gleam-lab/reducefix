```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<char> choices = {'R', 'P', 'S'};
    auto getWinner = [](char a, char b) {
        if ((a == 'R' && b == 'S') || (a == 'P' && b == 'R') || (a == 'S' && b == 'P')) 
            return 1; // a wins
        else if (a == b)
            return 0; // draw
        else
            return -1; // b wins
    };

    int maxScore = 0;

    // Try all possible starting hand choices
    for (char firstHand : choices) {
        vector<char> available = choices;
        // Remove the chosen hand
        available.erase(remove(available.begin(), available.end(), firstHand), available.end());
        
        int score = 0;
        char currentHand = firstHand;

        for (int i = 0; i < N; i++) {
            // Check if current hand wins
            if (getWinner(currentHand, S[i]) == 1) {
                score++;
            }
            
            // Update available hands for next round (remove current hand choice)
            vector<char> nextAvailable;
            for (char c : choices) {
                if (c != currentHand) {
                    nextAvailable.push_back(c);
                }
            }
            
            // Choose the best hand from remaining for next round
            char bestNext = nextAvailable[0];
            int bestResult = -2; // worse than loss
            
            for (char nextHand : nextAvailable) {
                int result = getWinner(nextHand, S[min(i+1, N-1)]);
                if (result > bestResult) {
                    bestResult = result;
                    bestNext = nextHand;
                }
            }
            
            currentHand = bestNext;
        }
        
        maxScore = max(maxScore, score);
    }

    cout << maxScore << endl;
    return 0;
}
```
