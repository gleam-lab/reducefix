#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    vector<char> hand = {'R', 'P', 'S'};
    vector<char> next;
    int total = 0;
    int rounds = 0;

    // First pass to calculate total points from hand to S
    for (int i = 0; i < S.length(); ++i) {
        for (char c : hand) {
            if ((c == 'R' && S[i] == 'S') ||
                (c == 'P' && S[i] == 'R') ||
                (c == 'S' && S[i] == 'P')) {
                total++;
                break;
            }
        }
        rounds++;
    }

    if (rounds != N - 1) { // If not all characters are matched, adjust
        total = 0;
        for (int i = 0; i < S.length(); ++i) {
            for (char c : hand) {
                if ((c == 'R' && S[i] == 'S') ||
                    (c == 'P' && S[i] == 'R') ||
                    (c == 'S' && S[i] == 'P')) {
                    total++;
                    break;
                }
            }
        }
    }

    cout << total << endl;
    return 0;
}
