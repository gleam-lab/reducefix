#include <bits/stdc++.h>

int calculateMaxScore(const string& S) {
    int N = S.size();
    if (N == 0) return 0;
    
    int score = 0;
    vector<char> available = {'R', 'P', 'S'};
    char lastUsed = '\0';

        char bestMove = '\0';

        if (opponent == 'R') bestMove = 'P';

        }
    }
    
    return score;
}
