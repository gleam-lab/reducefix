```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    // Initialize variables
    vector<int> count(3, 0); // Count of R, P, S
    for (char c : S) {
        switch (c) {
            case 'R': count[0]++; break;
            case 'P': count[1]++; break;
            case 'S': count[2]++; break;
        }
    }

    // Determine the optimal move based on counts
    char bestMove = '\0';
    int maxScore = 0;
    for (int i = 0; i < 3; ++i) {
        int currentScore = 0;
        for (int j = 0; j < 3; ++j) {
            if ((i + 1) % 3 == j || (i + 2) % 3 == j) {
                currentScore += count[j];
            }
        }
        if (currentScore > maxScore) {
            maxScore = currentScore;
            bestMove = "RPS"[i];
        }
    }

    // Calculate the maximum score
    int result = 0;
    for (char c : S) {
        if (c == bestMove) {
            result++;
        }
    }

    cout << result << endl;
    return 0;
}
```
