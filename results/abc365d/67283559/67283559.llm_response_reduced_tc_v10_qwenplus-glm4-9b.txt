#include <iostream>
#include <vector>
#include <unordered_set>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N;
    cin >> S;
    
    unordered_set<char> choices = {'R', 'P', 'S'};
    int total = 0;
    
    // Calculate scores from left to right
    for (int i = 0; i < N; ++i) {
        char expected = 'R';
        // Determine the expected move to get a point
        if (S[i] == 'R') {
            expected = 'P';
        } else if (S[i] == 'P') {
            expected = 'S';
        } else if (S[i] == 'S') {
            expected = 'R';
        }
        
        if (choices.count(expected)) {
            total++;
            choices.erase(expected);
        }
    }
    
    choices.clear();
    choices = {'R', 'P', 'S'};
    int reverseTotal = 0;
    
    // Calculate scores from right to left
    for (int i = N - 1; i >= 0; --i) {
        char expected = 'R';
        // Determine the expected move to get a point
        if (S[i] == 'R') {
            expected = 'P';
        } else if (S[i] == 'P') {
            expected = 'S';
        } else if (S[i] == 'S') {
            expected = 'R';
        }
        
        if (choices.count(expected)) {
            reverseTotal++;
            choices.erase(expected);
        }
    }
    
    cout << max(total, reverseTotal);
    return 0;
}
