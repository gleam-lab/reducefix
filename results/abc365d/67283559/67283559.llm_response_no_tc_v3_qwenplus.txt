```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    vector<char> choices = {'R', 'P', 'S'};
    auto getWinner = [](char a, char b) -> int {
        if ((a == 'R' && b == 'S') || (a == 'P' && b == 'R') || (a == 'S' && b == 'P')) {
            return 1; // a wins
        } else if (a == b) {
            return 0; // tie
        } else {
            return -1; // b wins
        }
    };
    
    int maxScore = 0;
    
    // Try all possible starting hands
    for (char firstHand : choices) {
        int score = 0;
        vector<char> available = choices;
        
        // Remove the hand we're using in this simulation
        available.erase(remove(available.begin(), available.end(), firstHand), available.end());
        
        char currentHand = firstHand;
        
        for (int i = 0; i < N; i++) {
            // Check if current hand wins
            if (getWinner(currentHand, S[i]) == 1) {
                score++;
            }
            
            // Update available hands after using currentHand
            if (i < N - 1) { // No need to update after last move
                // If opponent plays S[i], we eliminate that choice for next round
                auto it = find(available.begin(), available.end(), S[i]);
                if (it != available.end()) {
                    available.erase(it);
                    
                    // The next hand must be one of the remaining available hands
                    // If we have a winning move among available hands, use it
                    bool foundWinning = false;
                    for (char c : available) {
                        if (getWinner(c, S[i+1]) == 1) {
                            currentHand = c;
                            foundWinning = true;
                            break;
                        }
                    }
                    if (!foundWinning && !available.empty()) {
                        currentHand = available[0];
                    }
                }
            }
        }
        
        maxScore = max(maxScore, score);
    }
    
    cout << maxScore << endl;
    return 0;
}
```
