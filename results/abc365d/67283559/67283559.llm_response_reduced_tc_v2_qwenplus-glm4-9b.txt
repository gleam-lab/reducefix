#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    vector<char> hand = {'R', 'P', 'S'};
    int total = 0;
    int win = 0;

    for (int i = 0; i < N; ++i) {
        char current = S[i];
        char optimalReact = 'R';
        if (current == 'R') {
            optimalReact = 'P';
        } else if (current == 'P') {
            optimalReact = 'S';
        } else if (current == 'S') {
            optimalReact = 'R';
        }

        // Find the optimal move to win
        for (char h : hand) {
            if (h == optimalReact) {
                win++;
                break;
            }
        }
    }
    cout << win << endl;

    return 0;
}
