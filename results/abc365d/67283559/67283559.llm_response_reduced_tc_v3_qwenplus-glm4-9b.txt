
```cpp
#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<char> hand = {'R', 'P', 'S'};
    int total = 0;

    for (int i = 0; i < N; ++i) {
        int score = 0;
        char use;
        for (char d : hand) {
            if (d == 'R' && S[i] == 'S') {
                score = 1;
                use = 'R';
            } else if (d == 'P' && S[i] == 'R') {
                score = 1;
                use = 'P';
            } else if (d == 'S' && S[i] == 'P') {
                score = 1;
                use = 'S';
            }
        }

        if (score == 0) use = S[i];

        for (char d : hand) {
            if (d != use) {
                hand.push_back(d);
            }
        }

        total += score;
        hand.erase(unique(hand.begin(), hand.end()), hand.end());
    }

    cout << total;
}
```
