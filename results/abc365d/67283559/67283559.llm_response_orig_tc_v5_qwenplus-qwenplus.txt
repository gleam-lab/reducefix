#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    string S;
    cin >> S;

    // We can win at most once after each round, and must choose a different move each time
    // So we simulate the game with two possible starting moves and take the better outcome

    auto compute_score = [&](char first_move) {
        int score = 0;
        char current_move = first_move;
        for (char c : S) {
            if ((current_move == 'R' && c == 'S') ||
                (current_move == 'P' && c == 'R') ||
                (current_move == 'S' && c == 'P')) {
                score++;
            }
            // Choose a different move than the one used
            vector<char> next_moves;
            for (char m : {'R', 'P', 'S'}) {
                if (m != current_move) {
                    next_moves.push_back(m);
                }
            }
            // Prefer to pick a move that would beat the opponent's last move
            bool found = false;
            for (char m : next_moves) {
                if ((m == 'R' && c == 'S') ||
                    (m == 'P' && c == 'R') ||
                    (m == 'S' && c == 'P')) {
                    current_move = m;
                    found = true;
                    break;
                }
            }
            if (!found) {
                current_move = next_moves[0];
            }
        }
        return score;
    };

    // Try both possible starting moves and take the maximum
    int result = max(compute_score('R'), compute_score('P'));
    result = max(result, compute_score('S'));

    cout << result << endl;

    return 0;
}
