#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    string S;
    cin >> N >> S;
    
    vector<char> choices = {'R', 'P', 'S'};
    auto getWinner = [](char a, char b) -> int {
        if ((a == 'R' && b == 'S') || (a == 'P' && b == 'R') || (a == 'S' && b == 'P')) 
            return 1; // a wins
        if ((b == 'R' && a == 'S') || (b == 'P' && a == 'R') || (b == 'S' && a == 'P')) 
            return -1; // b wins
        return 0; // tie
    };
    
    int maxScore = 0;
    
    // Try all possible starting hands
    for (char firstMove : choices) {
        vector<char> available = {'R', 'P', 'S'};
        // Remove the chosen move
        available.erase(remove(available.begin(), available.end(), firstMove), available.end());
        
        int score = 0;
        char current = firstMove;
        
        for (int i = 0; i < N; i++) {
            if (getWinner(current, S[i]) == 1) {
                score++;
            }
            
            // Update available moves: remove the one we just used
            available.clear();
            for (char c : choices) {
                if (c != current) {
                    available.push_back(c);
                }
            }
            
            // Choose next move optimally from remaining options
            current = ' '; // reset
            // First try to win
            for (char c : available) {
                if ((c == 'R' && S[i] == 'S') || 
                    (c == 'P' && S[i] == 'R') || 
                    (c == 'S' && S[i] == 'P')) {
                    current = c;
                    break;
                }
            }
            // If can't win, avoid losing (tie or lose)
            if (current == ' ') {
                // Prefer tie over loss if possible
                bool foundTie = false;
                for (char c : available) {
                    if (c == S[i]) {
                        current = c;
                        foundTie = true;
                        break;
                    }
                }
                // Otherwise pick any
                if (current == ' ') {
                    current = available[0];
                }
            }
        }
        
        maxScore = max(maxScore, score);
    }
    
    cout << maxScore << endl;
    return 0;
}
