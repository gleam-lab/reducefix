```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<char> hand = {'R', 'P', 'S'};
    auto getScore = [&](const vector<char>& choices, const string& moves) {
        int total = 0;
        vector<char> next = choices;
        for (int i = 0; i < moves.size(); i++) {
            char opponent = moves[i];
            bool won = false;
            char use = opponent; // default: play same as opponent (tie)
            // Try to win if possible with available choices
            for (char c : next) {
                if ((c == 'R' && opponent == 'S') ||
                    (c == 'P' && opponent == 'R') ||
                    (c == 'S' && opponent == 'P')) {
                    won = true;
                    use = c;
                    break;
                }
            }
            if (won) total++;
            // Update next moves: remove the one we used
            vector<char> newNext;
            for (char c : hand) {
                if (c != use) {
                    newNext.push_back(c);
                }
            }
            next = newNext;
        }
        return total;
    };

    // Try all three possible starting moves
    int maxScore = 0;
    for (char start : hand) {
        vector<char> initial;
        for (char c : hand) {
            if (c != start) {
                initial.push_back(c);
            }
        }
        int score = getScore(initial, S);
        maxScore = max(maxScore, score);
    }

    cout << maxScore << endl;
    return 0;
}
```
