#include <bits/stdc++.h>
using namespace std;

int playGame(const string& S, const vector<char>& hand, char first_move) {
    int N = S.size();
    vector<char> next = hand;
    int total = 0;
    
    // Use the specified first move for the initial round
    char use = first_move;
    int i = 0;
    
    // Check if we can use the desired first move
    bool valid_first = false;
    for (char c : next) {
        if (c == use) {
            valid_first = true;
            break;
        }
    }
    
    if (!valid_first) {
        // If the first move is not available, pick any available move
        use = next[0];
    }
    
    // Determine score for the first round
    if ((use == 'R' && S[i] == 'S') || 
        (use == 'P' && S[i] == 'R') || 
        (use == 'S' && S[i] == 'P')) {
        total++;
    }
    
    i++; // Move to next round
    
    // Update available moves after first selection
    vector<char> new_next;
    for (char c : hand) {
        if (c != use) new_next.push_back(c);
    }
    next = new_next;
    
    // Play remaining rounds
    for (; i < N; i++) {
        bool won = false;
        // Try each available move to see if we can win
        for (char n : next) {
            if ((n == 'R' && S[i] == 'S') || 
                (n == 'P' && S[i] == 'R') || 
                (n == 'S' && S[i] == 'P')) {
                total++;
                use = n;
                break;
            }
        }
        
        // Update available moves
        new_next.clear();
        for (char c : hand) {
            if (c != use) new_next.push_back(c);
        }
        next = new_next;
    }
    
    return total;
}

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    vector<char> hand = {'R', 'P', 'S'};
    
    // Try all possible first moves and take the maximum score
    int max_score = 0;
    for (char first_move : hand) {
        int score = playGame(S, hand, first_move);
        max_score = max(max_score, score);
    }
    
    cout << max_score << endl;
    return 0;
}
