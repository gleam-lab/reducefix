```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    // Define the winning moves
    auto wins = [](char move) {
        if (move == 'R') return 'P';
        if (move == 'P') return 'S';
        return 'R'; // 'S' -> 'R'
    };

    // dp[i][last] = maximum score up to index i, where last is the last move used (0=R,1=P,2=S)
    vector<vector<int>> dp(N + 1, vector<int>(3, 0));
    vector<char> toChar = {'R', 'P', 'S'};

    for (int i = 0; i < N; i++) {
        char opponent = S[i];
        char winner = wins(opponent);

        for (int last = 0; last < 3; last++) {
            char lastMove = toChar[last];
            if (lastMove == winner) {
                // We win this round
                dp[i + 1][last] = max(dp[i + 1][last], dp[i][last] + 1);
            } else {
                // We don't win, but we can still transition
                dp[i + 1][last] = max(dp[i + 1][last], dp[i][last]);
            }

            // Try switching to a different move
            for (int next = 0; next < 3; next++) {
                if (next == last) continue;
                char nextMove = toChar[next];
                if (nextMove == winner) {
                    dp[i + 1][next] = max(dp[i + 1][next], dp[i][last] + 1);
                } else {
                    dp[i + 1][next] = max(dp[i + 1][next], dp[i][last]);
                }
            }
        }
    }

    int result = *max_element(dp[N].begin(), dp[N].end());
    cout << result << endl;

    return 0;
}
```
