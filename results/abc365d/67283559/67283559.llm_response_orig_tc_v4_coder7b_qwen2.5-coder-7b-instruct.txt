#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<int> dp(3, 0), next_dp(3, 0);
    vector<int> count(3, 0);
    
    for (char c : S) {
        int idx = c == 'R' ? 0 : (c == 'P' ? 1 : 2);
        next_dp[idx] = (dp[(idx + 1) % 3] + dp[(idx + 2) % 3]) % MOD;
        count[idx]++;
    }

    long long ans = 0;
    for (int i = 0; i < N; ++i) {
        ans = (ans + next_dp[S[i] == 'R' ? 0 : (S[i] == 'P' ? 1 : 2)]) % MOD;
        swap(dp, next_dp);
        fill(next_dp.begin(), next_dp.end(), 0);
    }

    cout << ans << endl;
    return 0;
}
