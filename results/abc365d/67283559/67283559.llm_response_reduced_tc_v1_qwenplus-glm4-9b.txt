#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<int> outcomes{0, 0, 0}; // R, P, S
    int total = 0;

    // First pass: calculate optimal outcomes for each character in S
    for (int i = 0; i < N; ++i) {
        int score = 0;
        char use = S[i];
        for (int j = 0; j < 3; ++j) {
            if ((use == 'R' && outcomes[j] == 0) ||
                (use == 'P' && outcomes[j] == 1) ||
                (use == 'S' && outcomes[j] == 2)) {
                score++;
                outcomes[j]++;
                break;
            }
        }
        if (score == 0) outcomes[use - 'R']++;
        total += score;
    }

    // Second pass: calculate optimal outcomes for the first character of S
    vector<int> outcomes2{0, 0, 0};
    for (int i = 0; i < 3; ++i) {
        if (i != S[0] - 'R') outcomes2[i]++;
    }

    int total2 = 0;
    for (int i = 1; i < N; ++i) {
        int score = 0;
        char use = S[i];
        for (int j = 0; j < 3; ++j) {
            if ((use == 'R' && outcomes2[j] == 0) ||
                (use == 'P' && outcomes2[j] == 1) ||
                (use == 'S' && outcomes2[j] == 2)) {
                score++;
                outcomes2[j]++;
                break;
            }
        }
        if (score == 0) outcomes2[use - 'R']++;
        total2 += score;
    }

    cout << max(total, total2) << endl;

    return 0;
}
