  vector<char> moves = {'R', 'P', 'S'};
  vector<int> scores(3, 0);
  
  // Calculate scores for each move based on the previous move
  for (int i = 1; i < N; ++i) {
    for (int j = 0; j < 3; ++j) {
      if ((moves[j] == 'R' && S[i - 1] == 'S') ||
          (moves[j] == 'P' && S[i - 1] == 'R') ||
          (moves[j] == 'S' && S[i - 1] == 'P')) {
        scores[j]++;
      }
    }
  }
  
  // Determine the best move to maximize the score
  int max_score = *max_element(scores.begin(), scores.end());
  char best_move = moves[max_element(scores.begin(), scores.end()) - scores.begin()];
  
  // Calculate the final score using the best move
  int final_score = 0;
  for (char c : S) {
    if ((best_move == 'R' && c == 'S') ||
        (best_move == 'P' && c == 'R') ||
        (best_move == 'S' && c == 'P')) {
      final_score++;
    }
  }
  
  cout << final_score << endl;
  return 0;
}
