```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  string S;
  cin >> S;
  
  vector<char> hand = {'R', 'P', 'S'};
  vector<int> scores(3, 0); // Scores for R, P, S
  
  for(int i = 0; i < N; i++) {
    for(int j = 0; j < 3; j++) {
      if((hand[j] == 'R' && S[i] == 'S') ||
         (hand[j] == 'P' && S[i] == 'R') ||
         (hand[j] == 'S' && S[i] == 'P')) {
        scores[j]++;
        break;
      }
    }
  }
  
  int maxScore = *max_element(scores.begin(), scores.end());
  cout << maxScore << endl;
  
  return 0;
}
```
