```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    string S;
    cin >> S;

    // Strategy: For each position, we track the best possible hand choices that maximize wins.
    // dp[i][c] = max number of wins up to position i using character c as the last move (c in 'R', 'P', 'S')

    vector<unordered_map<char, int>> dp(N, unordered_map<char, int>());
    vector<char> moves = {'R', 'P', 'S'};
    unordered_map<char, char> beats = {{'R', 'S'}, {'P', 'R'}, {'S', 'P'}};

    for (char c : moves) {
        if (c == beats[S[0]]) {
            dp[0][c] = 1;  // Win
        } else if (c == S[0]) {
            dp[0][c] = 0;  // Draw
        } else {
            dp[0][c] = 0;  // Lose
        }
    }

    for (int i = 1; i < N; ++i) {
        for (char curr : moves) {
            for (char prev : moves) {
                if (curr == prev) continue;  // Cannot reuse same move
                int score = 0;
                if (curr == beats[S[i]]) score = 1;  // Win
                else if (curr == S[i]) score = 0;     // Draw
                else score = 0;                       // Lose

                dp[i][curr] = max(dp[i][curr], dp[i - 1][prev] + score);
            }
        }
    }

    // Find the maximum score at the end
    int result = 0;
    for (auto& p : dp[N - 1]) {
        result = max(result, p.second);
    }

    cout << result << endl;

    return 0;
}
```
