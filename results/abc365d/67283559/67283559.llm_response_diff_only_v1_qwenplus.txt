```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;
    
    // The key insight: we can choose any move initially, then after each round,
    // we must avoid the move we just used (so we have two choices next time).
    // We want to maximize wins.
    
    vector<char> moves = {'R', 'P', 'S'};
    auto getWinningMove = [](char opponent) -> char {
        if (opponent == 'R') return 'P';
        if (opponent == 'P') return 'S';
        return 'R';
    };
    
    int maxWins = 0;
    
    // Try all possible starting moves
    for (char firstMove : moves) {
        int wins = 0;
        char currentMove = firstMove;
        
        for (int i = 0; i < N; i++) {
            // Check if current move beats opponent's move
            if ((currentMove == 'R' && S[i] == 'S') ||
                (currentMove == 'P' && S[i] == 'R') ||
                (currentMove == 'S' && S[i] == 'P')) {
                wins++;
            }
            
            // Choose next move - we need to pick from the two moves that are not currentMove
            // To maximize future wins, we should choose the move that beats S[i+1] if possible
            if (i < N - 1) {
                char bestNext = getWinningMove(S[i + 1]);
                // If the best move is different from current move, use it
                if (bestNext != currentMove) {
                    currentMove = bestNext;
                } else {
                    // Otherwise pick any other move (doesn't matter which)
                    for (char m : moves) {
                        if (m != currentMove) {
                            currentMove = m;
                            break;
                        }
                    }
                }
            }
        }
        
        maxWins = max(maxWins, wins);
    }
    
    cout << maxWins << endl;
    return 0;
}
```
