#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  string S;
  cin >> S;
  
  // Initialize possible moves
  vector<char> hand = {'R', 'P', 'S'};
  vector<int> scores(3, 0); // To store scores for each move
  
  // Calculate initial scores based on the first character of S
  for (char c : hand) {
    if ((c == 'R' && S[0] == 'S') || (c == 'P' && S[0] == 'R') || (c == 'S' && S[0] == 'P')) {
      scores[c - 'R']++; // Update score for the move that beats the first character
    }
  }
  
  // Iterate through the string to calculate total scores
  for (int i = 0; i < N; ++i) {
    int maxScore = 0;
    char bestMove;
    for (int j = 0; j < 3; ++j) {
      if (scores[j] > maxScore) {
        maxScore = scores[j];
        bestMove = j + 'R'; // Convert index back to move
      }
    }
    
    // Update scores based on the current character and the best move
    for (int j = 0; j < 3; ++j) {
      if ((bestMove == 'R' && S[i] == 'S') || (bestMove == 'P' && S[i] == 'R') || (bestMove == 'S' && S[i] == 'P')) {
        scores[j]++;
      } else if (bestMove != S[i]) {
        scores[j]--;
      }
    }
  }
  
  // Find the maximum score among all possible starting moves
  int maxTotal = 0;
  for (int score : scores) {
    maxTotal = max(maxTotal, score);
  }
  
  cout << maxTotal << endl;
  return 0;
}
