#include <bits/stdc++.h>
using namespace std;

int simulate(const string& S, const vector<char>& hand, char first_move) {
    int total = 0;
    vector<char> next = hand;
    bool found = false;
    for (auto c : next) {
        if (c == first_move) {
            found = true;
            break;
        }
    }
    if (!found) return 0; // Invalid first move

    char use = first_move;
    for (auto it = next.begin(); it != next.end();) {
        if (*it == use) {
            it = next.erase(it);
        } else {
            ++it;
        }
    }

    for (char c : S) {
        int score = 0;
        for (auto n : next) {
            if ((n == 'R' && c == 'S') || (n == 'P' && c == 'R') || (n == 'S' && c == 'P')) {
                score = 1;
                use = n;
                break;
            }
        }
        if (score == 0) {
            use = c;
        }
        total += score;
        next.clear();
        for (auto h : hand) {
            if (h != use) {
                next.push_back(h);
            }
        }
    }
    return total;
}

int main() {
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<char> hand = {'R', 'P', 'S'};
    
    // Try all possible first moves and take the maximum score
    int max_score = 0;
    for (char move : hand) {
        int score = simulate(S, hand, move);
        max_score = max(max_score, score);
    }
    
    cout << max_score;
    return 0;
}
