#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    string S;
    cin >> S;

    // For each position, we want to choose a move that either wins or avoids elimination
    // We can use dynamic programming with state: current available moves (as bitmask or set)
    // However, note: at each step, we remove the move we used from future options

    // Instead of simulating all possibilities, we can use greedy with backtracking idea:
    // But the problem is asking for maximum wins.

    // Alternate Insight:
    // We are allowed to choose any move initially, then after using a move, it's banned.
    // So we have a sequence of distinct moves (at most 3 different ones).

    // Since there are only 3 possible moves, we can try all permutations of {'R','P','S'} 
    // as the order in which we use our moves. Once a move is used, we cannot use it again.

    vector<char> moves = {'R', 'P', 'S'};
    int max_wins = 0;

    // Try all 6 permutations of move orders
    sort(moves.begin(), moves.end());
    do {
        int wins = 0;
        vector<bool> used(3, false); // track which moves we've used

        for (int i = 0; i < N; i++) {
            char opponent = S[i];
            char chosen = 0;
            bool found_win = false;

            // Try moves in the fixed permutation order
            for (int j = 0; j < 3; j++) {
                if (used[j]) continue;
                char my_move = moves[j];

                // Check if this move beats opponent
                if ((my_move == 'R' && opponent == 'S') ||
                    (my_move == 'P' && opponent == 'R') ||
                    (my_move == 'S' && opponent == 'P')) {
                    chosen = my_move;
                    found_win = true;
                    break;
                }
            }

            // If no winning move available, pick the first unused move (to save better ones?)
            if (!found_win) {
                for (int j = 0; j < 3; j++) {
                    if (!used[j]) {
                        chosen = moves[j];
                        break;
                    }
                }
            }

            // Mark this move as used
            for (int j = 0; j < 3; j++) {
                if (moves[j] == chosen) {
                    used[j] = true;
                    break;
                }
            }

            // Count win
            if ((chosen == 'R' && opponent == 'S') ||
                (chosen == 'P' && opponent == 'R') ||
                (chosen == 'S' && opponent == 'P')) {
                wins++;
            }
        }

        max_wins = max(max_wins, wins);
    } while (next_permutation(moves.begin(), moves.end()));

    cout << max_wins << endl;

    return 0;
}
