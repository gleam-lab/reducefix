#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int N;
    std::cin >> N;
    std::string S;
    std::cin >> S;

    // Initialize variables
    const std::vector<char> hand = {'R', 'P', 'S'};
    int total = 0;

    // Iterate over each character in the string
    for (int i = 0; i < N; ++i) {
        int best_score = 0;
        char best_choice = S[i];

        // Check all choices against the current character
        for (char choice : hand) {
            if ((choice == 'R' && S[i] == 'S') ||
                (choice == 'P' && S[i] == 'R') ||
                (choice == 'S' && S[i] == 'P')) {
                best_score++;
                best_choice = choice;
                break;
            }
        }

        // If no better choice found, use the current character itself
        if (best_score == 0) best_choice = S[i];

        // Update total score and remove the chosen character from the set of available choices
        total += best_score;
        std::vector<char> next(hand.begin(), hand.end());
        next.erase(std::remove(next.begin(), next.end(), best_choice), next.end());

        // Clear the next vector after processing the first character
        if (i == 0) next.clear();
    }

    // Calculate the maximum score using the second approach
    int total2 = 0;
    std::vector<char> next2(hand.begin(), hand.end());
    next2.erase(std::remove(next2.begin(), next2.end(), S[0]), next2.end());

    for (int i = 1; i < N; ++i) {
        int best_score = 0;
        char best_choice = S[i];

        // Check all choices against the current character
        for (char choice : next2) {
            if ((choice == 'R' && S[i] == 'S') ||
                (choice == 'P' && S[i] == 'R') ||
                (choice == 'S' && S[i] == 'P')) {
                best_score++;
                best_choice = choice;
                break;
            }
        }

        // If no better choice found, use the current character itself
        if (best_score == 0) best_choice = S[i];

        // Update total score and remove the chosen character from the set of available choices
        total2 += best_score;
        next2.erase(std::remove(next2.begin(), next2.end(), best_choice), next2.end());
    }

    // Output the maximum of both scores
    std::cout << std::max(total, total2) << std::endl;

    return 0;
}
