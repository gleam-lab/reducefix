#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<int> cnt(3), dp(3, INT_MIN);
    dp[0] = dp[1] = dp[2] = 0;

    for (char c : S) {
        vector<int> new_dp(3, INT_MIN);
        if (c == 'R') {
            new_dp[0] = max(new_dp[0], dp[1]);
            new_dp[1] = max(new_dp[1], dp[2]);
            new_dp[2] = max(new_dp[2], dp[0] + 1);
        } else if (c == 'P') {
            new_dp[0] = max(new_dp[0], dp[2]);
            new_dp[1] = max(new_dp[1], dp[0]);
            new_dp[2] = max(new_dp[2], dp[1] + 1);
        } else if (c == 'S') {
            new_dp[0] = max(new_dp[0], dp[0]);
            new_dp[1] = max(new_dp[1], dp[1]);
            new_dp[2] = max(new_dp[2], dp[2] + 1);
        }
        dp = new_dp;
    }

    int ans = max({dp[0], dp[1], dp[2]});
    cout << ans << endl;

    return 0;
}
