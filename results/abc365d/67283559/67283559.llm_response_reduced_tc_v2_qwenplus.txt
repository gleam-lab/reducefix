#include <bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin >> N;
    string S;
    cin >> S;

    vector<char> hand = {'R','P','S'};
    auto getScore = [&](const string& s) {
        vector<char> next = hand;
        int total = 0;
        for (int i = 0; i < s.size(); i++) {
            char opponent = s[i];
            char use = 0;
            int score = 0;

            // Try to pick a move from available options that beats opponent
            for (char c : next) {
                if ((c == 'R' && opponent == 'S') ||
                    (c == 'P' && opponent == 'R') ||
                    (c == 'S' && opponent == 'P')) {
                    score = 1;
                    use = c;
                    break;
                }
            }

            // If no winning move, use the opponent's move (or any valid one)
            if (score == 0) {
                use = opponent;
            }

            // Update next set of available moves: remove the one we used
            vector<char> new_next;
            for (char c : hand) {
                if (c != use) {
                    new_next.push_back(c);
                }
            }
            next = new_next;
            total += score;
        }
        return total;
    };

    // Try all three possible starting moves
    int max_score = 0;
    for (char first_move : hand) {
        vector<char> next;
        for (char c : hand) {
            if (c != first_move) {
                next.push_back(c);
            }
        }
        int score = 0;
        for (int i = 0; i < N; i++) {
            char opponent = S[i];
            char use = 0;
            bool won = false;

            // First try to win with available moves
            for (char c : next) {
                if ((c == 'R' && opponent == 'S') ||
                    (c == 'P' && opponent == 'R') ||
                    (c == 'S' && opponent == 'P')) {
                    won = true;
                    use = c;
                    break;
                }
            }

            if (!won) {
                // Pick any available move (even if it draws or loses)
                use = next[0];
            }

            score += won ? 1 : 0;

            // Update available moves for next round
            vector<char> new_next;
            for (char c : hand) {
                if (c != use) {
                    new_next.push_back(c);
                }
            }
            next = new_next;
        }
        max_score = max(max_score, score);
    }

    cout << max_score << endl;
    return 0;
}
