#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    char s[n + 1];
    cin.getline(s, n + 1);  // Read the string including spaces
    char c[n + 1];
    char mapping[256];  // To store the mapping between characters and their transformed characters

    // Initialize the mapping for 'P', 'R', and 'S'
    mapping['P'] = 'S';
    mapping['R'] = 'P';
    mapping['S'] = 'R';

    int sum = n;
    for (int i = 0; i < n; i++) {
        c[i] = mapping[s[i]];
    }

    // Calculate the maximum sum by trying to keep the sequence different
    int max_sum = 0;
    int count = 1;
    max_sum = count;

    for (int i = 1; i < n; i++) {
        if (c[i] != c[i - 1]) {
            count++;
        } else {
            max_sum = max(max_sum, count);
            count = 1;
        }
    }
    max_sum = max(max_sum, count);  // Check the last segment

    printf("%d", max_sum);
    return 0;
}
