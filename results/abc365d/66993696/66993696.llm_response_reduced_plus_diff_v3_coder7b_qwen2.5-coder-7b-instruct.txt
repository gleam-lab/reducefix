```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s(n + 1), t(n + 1);
    cin >> s;
    
    // Create a copy of the original string
    t = s;
    
    // Transform all characters according to the given rules
    for(int i = 1; i <= n; ++i) {
        if(s[i] == 'P') t[i] = 'S';
        else if(s[i] == 'R') t[i] = 'P';
        else t[i] = 'R';
    }
    
    // Count the number of changes needed to make the two strings identical
    int changes = 0;
    for(int i = 1; i <= n; ++i) {
        if(s[i] != t[i]) ++changes;
    }
    
    // The minimum number of operations is half of the changes
    cout << changes / 2 << endl;
    
    return 0;
}
```
