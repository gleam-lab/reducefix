#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    string s(n, ' ');
    string c(n, ' ');
    cin >> s;
    
    // First pass to transform 'P' to 'S' and 'R' to 'P'
    for (int i = 0; i < n; ++i) {
        c[i] = (s[i] == 'P') ? 'S' : 'R';
    }
    
    // Calculate the size of the transformed segment
    int size = 0;
    int last = c[0];
    for (int i = 1; i < n; ++i) {
        if (c[i] == last) {
            size++;
        } else {
            size = 1;
            last = c[i];
        }
    }
    
    // Update the maximum size found so far
    int max_size = max(size, n - size);
    
    // Second pass to transform 'S' to 'R' and 'P' to 'S'
    for (int i = 0; i < n; ++i) {
        c[i] = (s[i] == 'P') ? 'S' : 'R';
    }
    
    // Calculate the size of the transformed segment
    size = 0;
    last = c[0];
    for (int i = 1; i < n; ++i) {
        if (c[i] == last) {
            size++;
        } else {
            size = 1;
            last = c[i];
        }
    }
    
    // Update the maximum size found so far
    max_size = max(max_size, size);
    
    // Output the result
    printf("%d\n", max_size);
    
    return 0;
}
