#include <bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int n;

// Function to compute the length of a valid sequence
// by greedily changing characters that would create duplicates
int computeMaxSequence() {
    int changed = 0;
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            // Try to change current character to original
            char original = s[i];
            char prev = c[i - 1];
            if (original != prev) {
                c[i] = original;
            } else {
                // If original is also same, we have to leave it as is => decrease count
                return -1;
            }
        }
    }

    return n;
}

int main() {
    scanf("%d", &n);
    scanf("%s", s + 1); // Input string is 1-indexed

    // Try one greedy approach: always replace with counter move and fix conflicts
    int max_ans = 0;
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int count = n;
    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];  // Try setting to original
            count--;
        }
    }
    max_ans = count;

    // Try another variation where we start by keeping first character unchanged
    c[1] = s[1];
    count = n - 1;
    for (int i = 2; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';

        if (c[i] == c[i - 1]) {
            c[i] = s[i];  // Try original
            count--;
        }
    }
    max_ans = max(max_ans, count);

    printf("%d\n", max_ans);
    return 0;
}
