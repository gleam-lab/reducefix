#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int solve(int n) {
    // Try all possible starting characters for the transformed string
    int best = 0;
    
    // We'll try to assign each position a character different from original
    // and maximize the number of distinct adjacent pairs
    for (char start : {'P', 'R', 'S'}) {
        if (start == s[1]) continue; // must be different
        
        c[1] = start;
        int count = 1;
        
        for (int i = 2; i <= n; i++) {
            // Choose c[i] to be different from both s[i] and c[i-1]
            for (char ch : {'P', 'R', 'S'}) {
                if (ch != s[i]) {
                    c[i] = ch;
                    break;
                }
            }
            
            // If current is same as previous, we can potentially fix it by using original
            // But only if original is different from previous
            if (c[i] == c[i-1]) {
                bool can_use_original = true;
                for (char ch : {'P', 'R', 'S'}) {
                    if (ch != s[i] && ch != c[i-1]) {
                        c[i] = ch;
                        can_use_original = false;
                        break;
                    }
                }
                if (can_use_original) {
                    // Only use original if no other option exists
                    c[i] = s[i];
                } else {
                    count++;
                }
            } else {
                count++;
            }
        }
        best = max(best, count);
    }
    
    return best;
}

// Greedy approach: build sequence where each char is not equal to original and minimize consecutive duplicates
int solve_greedy(int n) {
    if (n == 0) return 0;
    if (n == 1) return 0; // because we must choose different from original
    
    // For each position, we have two choices (not equal to s[i])
    // We want to maximize number of positions where c[i] != c[i-1]
    vector<vector<char>> options(n+1);
    for (int i = 1; i <= n; i++) {
        for (char ch : {'P', 'R', 'S'}) {
            if (ch != s[i]) {
                options[i].push_back(ch);
            }
        }
    }
    
    // dp[i][j] = maximum score up to position i, ending with options[i][j]
    vector<vector<int>> dp(n+1, vector<int>(2, 0));
    
    // Base case
    dp[1][0] = 1;
    dp[1][1] = 1;
    
    for (int i = 2; i <= n; i++) {
        for (int j = 0; j < 2; j++) {
            char cur = options[i][j];
            int val = 0;
            for (int k = 0; k < 2; k++) {
                char prev = options[i-1][k];
                int add = (cur != prev) ? 1 : 0;
                val = max(val, dp[i-1][k] + add);
            }
            dp[i][j] = val;
        }
    }
    
    return max(dp[n][0], dp[n][1]);
}

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s+1);
    
    if (n == 1) {
        printf("0");
        return 0;
    }
    
    // Use greedy DP solution
    vector<string> opts(n+1);
    for (int i = 1; i <= n; i++) {
        string av = "";
        for (char ch : {'P', 'R', 'S'}) {
            if (ch != s[i]) {
                av += ch;
            }
        }
        opts[i] = av;
    }
    
    // dp[i][0] = max count ending at i with first option
    // dp[i][1] = max count ending at i with second option
    vector<vector<int>> dp(n+1, vector<int>(2, 0));
    
    // Base: first position contributes 1 to count (no adjacency)
    dp[1][0] = 1;
    dp[1][1] = 1;
    
    for (int i = 2; i <= n; i++) {
        char opt0 = opts[i][0];
        char opt1 = opts[i][1];
        
        char prev0 = opts[i-1][0];
        char prev1 = opts[i-1][1];
        
        // Current opt0
        int trans0 = max(dp[i-1][0] + (opt0 != prev0), dp[i-1][1] + (opt0 != prev1));
        dp[i][0] = trans0;
        
        // Current opt1
        int trans1 = max(dp[i-1][0] + (opt1 != prev0), dp[i-1][1] + (opt1 != prev1));
        dp[i][1] = trans1;
    }
    
    int ans = max(dp[n][0], dp[n][1]);
    printf("%d", ans);
    
    return 0;
}
