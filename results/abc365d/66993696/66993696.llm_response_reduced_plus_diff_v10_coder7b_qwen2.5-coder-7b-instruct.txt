```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s(n + 1);
    cin >> s;
    
    vector<int> count(3, 0);
    for (int i = 1; i <= n; ++i) {
        count[s[i] - 'P']++;
    }
    
    int p = count[0], r = count[1], s_ = count[2];
    
    int res = min({p, r, s_});
    cout << n - res << endl;
    
    return 0;
}
```
