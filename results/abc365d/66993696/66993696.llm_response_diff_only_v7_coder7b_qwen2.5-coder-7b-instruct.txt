#include<bits/stdc++.h>
using namespace std;

const int MAXN = 200011;
char s[MAXN], c[MAXN];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Initialize c array with initial values based on s array
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int ans = 0, sum = n;

    // First pass: try to maximize score without changing the first character
    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];
            --sum;
        }
    }
    ans = max(ans, sum);

    // Reset c array for the second pass
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Second pass: try to maximize score with the first character changed
    if (s[1] != s[2]) {
        c[1] = s[1];
        --sum;
        for (int i = 2; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                --sum;
            }
        }
        ans = max(ans, sum);
    } else {
        // If the first two characters are the same, we need to handle it differently
        for (int i = 2; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                --sum;
            }
        }
        ans = max(ans, sum);
    }

    printf("%d\n", ans);
    return 0;
}
