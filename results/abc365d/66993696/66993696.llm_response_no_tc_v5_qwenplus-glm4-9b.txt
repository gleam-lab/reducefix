#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    int sum = n;
    for (int i = 1; i <= n; i++) {
        c[i] = (s[i] == 'P') ? 'S' : 'R';
    }

    // First pass to handle consecutive characters
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];
            sum--;
        }
    }

    // Check if the first two characters are the same
    if (c[1] == c[2]) {
        c[1] = s[1];
        sum--;
        for (int i = 3; i <= n; i++) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                sum--;
            }
        }
    }

    int ans = max(sum, 1); // Ensure at least 1 if sum <= 0
    printf("%d", ans);
    return 0;
}
