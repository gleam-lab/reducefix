#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Define the winning move for each gesture
    auto win = [](char ch) {
        if (ch == 'P') return 'S';
        if (ch == 'R') return 'P';
        return 'R'; // 'S' -> 'R'
    };

    // Try both strategies: not changing first element and changing first element
    int ans = 0;

    // Strategy 1: do not change the first character
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    int sum = n;
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];  // change to original gesture to break consecutive
            sum--;
        }
    }
    ans = max(ans, sum);

    // Strategy 2: change the first character (try to avoid conflict at start)
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    sum = n;
    c[1] = s[1]; // change first to original
    sum--;

    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];
            sum--;
        }
    }
    ans = max(ans, sum);

    // Strategy 3: try changing second if first two would conflict in default strategy
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    sum = n;
    if (n >= 2 && c[1] == c[2]) {
        c[2] = s[2];
        sum--;
    }
    for (int i = 3; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];
            sum--;
        }
    }
    ans = max(ans, sum);

    printf("%d", ans);
    return 0;
}
