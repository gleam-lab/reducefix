```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    string s;
    cin >> n >> s;

    // c[i] is the move chosen by the second player to potentially beat s[i]
    vector<char> c(n);
    for (int i = 0; i < n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Try all possible changes of one position and keep track of the maximum wins
    int max_wins = 0;
    for (int k = 0; k < n; ++k) {
        vector<char> modified_c = c;
        int total_wins = n;

        // Change c[k] to s[k]
        modified_c[k] = s[k];
        total_wins--;

        // Traverse left to right and eliminate consecutive same moves
        for (int i = 1; i < n; ++i) {
            if (modified_c[i] == modified_c[i - 1]) {
                modified_c[i] = s[i];
                total_wins--;
            }
        }

        max_wins = max(max_wins, total_wins);
    }

    cout << max_wins << endl;
    return 0;
}
```
