#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    string s;
    cin >> n >> s;
    
    // Count frequency of each move
    map<char, int> cnt;
    for (char c : s) {
        cnt[c]++;
    }
    
    // Beat opponent's move: P->S, R->P, S->R
    auto beat = [](char c) {
        if (c == 'P') return 'S';
        if (c == 'R') return 'P';
        return 'R';
    };
    
    // Try all 3 possible starting moves
    int ans = 0;
    for (char start : {'P', 'R', 'S'}) {
        vector<char> t(n);
        t[0] = start;
        
        // Greedily choose moves
        for (int i = 1; i < n; i++) {
            char prev = t[i-1];
            char opp = s[i];
            
            // If we can avoid repeating our own move, choose the winning move
            if (beat(opp) != prev) {
                t[i] = beat(opp);
            } else {
                // Otherwise, choose any move that isn't the same as previous
                for (char c : {'P', 'R', 'S'}) {
                    if (c != prev) {
                        t[i] = c;
                        break;
                    }
                }
            }
        }
        
        // Count wins
        int wins = 0;
        for (int i = 0; i < n; i++) {
            char my_move = t[i];
            char opp_move = s[i];
            if ((my_move == 'P' && opp_move == 'R') ||
                (my_move == 'R' && opp_move == 'S') ||
                (my_move == 'S' && opp_move == 'P')) {
                wins++;
            }
        }
        ans = max(ans, wins);
    }
    
    cout << ans << endl;
    return 0;
}
