```cpp
#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200011;
char s[MAXN], c[MAXN];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Initialize the result array with the initial move sequence
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int ans = 0, sum = n;

    // Try to minimize the number of adjacent same moves
    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];
            --sum;
        }
    }
    ans = max(ans, sum);

    // Reset the result array to the initial move sequence
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Special case: handle the first two moves separately
    if (s[1] == s[2]) {
        c[1] = s[1];
        --sum;
        for (int i = 3; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                --sum;
            }
        }
    }
    ans = max(ans, sum);

    printf("%d\n", ans);
    return 0;
}
```

### Explanation of Fixes
1. **Initialization**: The `c` array is initialized with the first move sequence based on the rules provided.
2. **Minimizing Adjacent Same Moves**: The loop iterates through the array starting from the second element to ensure that no two adjacent elements are the same. If they are, it changes the current element to match the next one, thus minimizing the number of such occurrences.
3. **Edge Case Handling**: The special case where the first two moves are the same is handled separately by setting the first element to match the second and then proceeding similarly as before.
4. **Output**: The maximum possible number of distinct moves is printed as the answer.
