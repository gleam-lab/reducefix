#include<bits/stdc++.h>
using namespace std;
char s[200011];
char c[200011];
int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    int win = 0;
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }
    int lose = 0;
    for (int i = 1; i < n; ++i) {
        if (c[i] == s[i + 1]) ++lose;
    }
    win = n - lose;
    printf("%d\n", win);
}
