#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    char s[200011], c[200011];
    scanf("%s", s + 1);
    
    // Initial transformation of s to c based on given rules
    for (int i = 1; i <= n; i++) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }
    
    // Count the number of unique characters in the transformed array
    int unique_count = 0;
    for (int i = 1; i <= n; i++) {
        if (i == 1 || c[i] != c[i - 1]) {
            unique_count++;
        }
    }
    
    // First max() call
    int max_unique = unique_count;
    
    // Reset c to the initial transformation of s
    for (int i = 1; i <= n; i++) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }
    
    // Modify c to minimize the number of unique elements
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i - 1]) {
            continue; // Same as the previous
        }
        // Find the next character that is the same as the previous character of s
        int j = i + 1;
        while (j <= n && c[j] != c[i - 1]) j++;
        if (j <= n) {
            // Swap all in the range from i to j-1 to make them all the same as s[i-1]
            for (int k = i; k < j; k++) {
                swap(c[k], s[k]);
            }
            i = j - 1; // Skip checking the swapped elements
        }
    }
    
    // Count the number of unique characters in the modified array
    unique_count = 0;
    for (int i = 1; i <= n; i++) {
        if (i == 1 || c[i] != c[i - 1]) {
            unique_count++;
        }
    }
    
    // Second max() call
    max_unique = max(max_unique, unique_count);
    
    // Output the result
    printf("%d", max_unique);
    return 0;
}
