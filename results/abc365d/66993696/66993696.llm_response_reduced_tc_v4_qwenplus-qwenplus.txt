```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 200010;
char s[MAXN], c[MAXN];

int solve(int n) {
    // Apply the transformation: P->S, R->P, S->R
    for (int i = 0; i < n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Eliminate adjacent duplicates
    int count = n;
    for (int i = 1; i < n; ++i) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];  // Revert to original character
            count--;
        }
    }
    return count;
}

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s);

    // Try from beginning
    int ans = solve(n);

    // Try from end (equivalent to reversing string and trying from beginning)
    reverse(s, s + n);
    ans = max(ans, solve(n));

    printf("%d\n", ans);
    return 0;
}
```
