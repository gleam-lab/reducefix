```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    string s;
    cin >> n >> s;

    // Strategy mapping: P -> S, R -> P, S -> R
    string strategy = "S";  // index 0 unused
    strategy += 'S';        // P -> S
    strategy += 'P';        // R -> P
    strategy += 'R';        // S -> R

    auto counter = [&](char c) { return strategy[c - 'A']; };

    int max_safe = 0;

    // Try all possible modifications for the first character (3 options)
    for (char first_char : {'R', 'P', 'S'}) {
        if (first_char != s[0]) continue;  // Only allow original character for first position

        char* modified = new char[n];
        modified[0] = counter(s[0]);

        int safe_count = 1;

        for (int i = 1; i < n; ++i) {
            char next_counter = counter(s[i]);
            if (next_counter == modified[i - 1]) {
                // Conflict, cannot be safe
                modified[i] = s[i]; // Revert to original
            } else {
                modified[i] = next_counter;
                ++safe_count;
            }
        }

        max_safe = max(max_safe, safe_count);
        delete[] modified;
    }

    cout << max_safe << endl;
    return 0;
}
```
