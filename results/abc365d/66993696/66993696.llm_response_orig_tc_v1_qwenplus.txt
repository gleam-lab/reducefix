        vector<char> t(n + 1);
        t[1] = start;
        int score = (t[1] == 'P' || t[1] == 'R' || t[1] == 'S') ? 1 : 0;
        
        for (int i = 2; i <= n; i++) {
            // Determine what opponent played at position i-1
            char opp_prev = s[i-1];
            char beat_prev;
            if (opp_prev == 'P') beat_prev = 'S';
            else if (opp_prev == 'R') beat_prev = 'P';
            else beat_prev = 'R';
            
            // We cannot repeat our previous move
            if (t[i-1] == beat_prev) {
                // If the winning move is same as last move, we must play something else
                // Try to play the opponent's move (safe, won't create duplicate)
                t[i] = s[i];
            } else {
                // Otherwise, we can play the winning move
                char opp_curr = s[i];
                t[i] = (opp_curr == 'P') ? 'S' : (opp_curr == 'R') ? 'P' : 'R';
            }
            score++;
        }
        return score;
    };

    int ans = 0;

    {
        vector<char> t(n + 1);

            score++;
        }
