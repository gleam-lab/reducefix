#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200011;
char s[MAXN], c[MAXN];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    // Initialize count of each character
    int count[3] = {0};
    for(int i = 1; i <= n; ++i) {
        count[s[i] - 'P']++;
    }
    
    // Determine the most frequent character
    int max_count = *max_element(count, count + 3);
    
    // If there is only one type of character, return n / 2
    if(max_count == n) {
        printf("%d\n", n / 2);
        return 0;
    }
    
    // Otherwise, calculate the maximum number of valid moves
    int result = 0;
    for(int i = 1; i <= n; ++i) {
        c[i] = s[i];
        if(i > 1 && c[i] == c[i - 1]) {
            if(s[i] == 'P') {
                if(count['S' - 'P']) {
                    --count['S' - 'P'];
                    c[i] = 'S';
                } else if(count['R' - 'P']) {
                    --count['R' - 'P'];
                    c[i] = 'R';
                }
            } else if(s[i] == 'R') {
                if(count['P' - 'R']) {
                    --count['P' - 'R'];
                    c[i] = 'P';
                } else if(count['S' - 'R']) {
                    --count['S' - 'R'];
                    c[i] = 'S';
                }
            } else {
                if(count['R' - 'S']) {
                    --count['R' - 'S'];
                    c[i] = 'R';
                } else if(count['P' - 'S']) {
                    --count['P' - 'S'];
                    c[i] = 'P';
                }
            }
        }
    }
    
    // Count the number of changes made
    for(int i = 1; i <= n; ++i) {
        if(c[i] != s[i]) {
            ++result;
        }
    }
    
    printf("%d\n", result / 2);
    return 0;
}
