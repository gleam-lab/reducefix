#include<bits/stdc++.h>
using namespace std;

int main() {
    char s[200011];
    char c[200011];
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);  // Note: Use s+1 to avoid the null terminator issue
    for (int i = 1; i <= n; i++) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int ans = 1;  // Initialize to 1 to account for all possible scenarios at least once
    int sum = n;
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];
            sum--;
        }
    }
    ans = max(ans, sum);

    // Reset c to new values after the first transformation
    for (int i = 1; i <= n; i++) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Consider the case where the first two elements are the same
    if (s[1] == s[2]) {
        c[1] = s[1];
        sum--;
        for (int i = 3; i <= n; i++) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                sum--;
            }
        }
    }
    ans = max(ans, sum);

    printf("%d\n", ans);
    return 0;
}
