#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200011;
char s[MAXN], c[MAXN];

void solve() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Initialize c array with transformed values of s
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int ans = 0, sum = n;

    // First transformation
    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            c[i] = s[i];
            --sum;
        }
    }
    ans = max(ans, sum);

    // Second transformation
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    if (s[1] == s[2]) {
        c[1] = s[1];
        --sum;
        for (int i = 3; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                --sum;
            }
        }
    }
    ans = max(ans, sum);

    printf("%d\n", ans);
}

int main() {
    solve();
    return 0;
}
