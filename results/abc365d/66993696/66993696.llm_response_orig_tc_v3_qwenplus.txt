#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    // Define the winning move
    auto win = [](char x) {
        if (x == 'P') return 'S';
        if (x == 'R') return 'P';
        return 'R';
    };
    
    // Try both strategies: either change all or allow one adjacent same
    int ans = 0;
    
    // Strategy 1: Change every character to its winning move
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    
    int cnt = n;
    // Remove consecutive duplicates greedily from left to right
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];  // Revert back to original move
            cnt--;
        }
    }
    ans = max(ans, cnt);
    
    // Strategy 2: Try to keep first two same if possible
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    
    if (n >= 2 && s[1] == s[2]) {
        // Force first move to be same as original
        c[1] = s[1];
        cnt = n - 1;
        for (int i = 2; i <= n; i++) {
            if (c[i] == c[i-1]) {
                c[i] = s[i];
                cnt--;
            }
        }
        ans = max(ans, cnt);
    }
    
    // Strategy 3: Try breaking at different positions
    // Instead of just first position, try to simulate optimal placement
    // We realize that the greedy from left may not be optimal.
    // Let's use a better strategy: dynamic programming isn't needed.
    // Actually, the key is that we can choose any subset of positions to revert.
    // But constraint is no two adjacent in final sequence are equal.
    // So problem reduces to: choose maximum subset of indices where we use win(s[i]),
    // such that no two adjacent have same win(s[i]), and others we use s[i] (which breaks chain).
    
    // Alternate approach: simulate greedy but allow skipping more intelligently.
    // But note: previous method already tries two starts.
    // However, the real solution is simpler: we can always achieve a high score by careful assignment.
    
    // After reading known similar problems, the correct idea:
    // For each position, we have two choices: play optimally (winning move) or play original.
    // We want to maximize number of winning moves, with constraint that no two adjacent final moves are same.
    
    // Use greedy: traverse and whenever possible, use winning move unless it matches previous.
    cnt = 0;
    char prev = 0;
    for (int i = 1; i <= n; i++) {
        char w = win(s[i]);
        if (w != prev) {
            cnt++;
            prev = w;
        } else {
            prev = s[i];
        }
    }
    ans = max(ans, cnt);
    
    // Also try starting with original move
    if (n > 0) {
        cnt = 1;
        prev = s[1];
        for (int i = 2; i <= n; i++) {
            char w = win(s[i]);
            if (w != prev) {
                cnt++;
                prev = w;
            } else {
                prev = s[i];
            }
        }
        ans = max(ans, cnt);
    }
    
    printf("%d\n", ans);
    return 0;
}
