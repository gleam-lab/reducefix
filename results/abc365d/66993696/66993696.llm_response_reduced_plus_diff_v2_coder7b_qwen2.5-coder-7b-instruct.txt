```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s(n, ' ');
    for(int i = 0; i < n; ++i) cin >> s[i];

    vector<int> cnt(3);
    for(char c : s) ++cnt[c - 'P'];

    int res = min({cnt[0], cnt[1], cnt[2]});
    cout << res * 2 + (n - res * 3) / 2 << endl;

    return 0;
}
```
