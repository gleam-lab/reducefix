#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int main()
{
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    // Function to compute the maximum wins for a given starting choice
    auto solve = [&](char start) -> int {
        vector<char> choices(n + 1);
        int wins = 0;
        
        choices[1] = start;
        if (s[1] == 'R' && choices[1] == 'P') wins++;
        else if (s[1] == 'P' && choices[1] == 'S') wins++;
        else if (s[1] == 'S' && choices[1] == 'R') wins++;
        
        for (int i = 2; i <= n; i++) {
            // Choose optimally: beat opponent or maintain continuity if possible
            char beat = (s[i] == 'R') ? 'P' : (s[i] == 'P') ? 'S' : 'R';
            
            if (beat == choices[i-1]) {
                choices[i] = choices[i-1];
            } else {
                choices[i] = beat;
            }
            
            if (s[i] == 'R' && choices[i] == 'P') wins++;
            else if (s[i] == 'P' && choices[i] == 'S') wins++;
            else if (s[i] == 'S' && choices[i] == 'R') wins++;
        }
        
        return wins;
    };
    
    // Try all three possible starting moves and take maximum
    int ans = 0;
    ans = max(ans, solve('R'));
    ans = max(ans, solve('P'));
    ans = max(ans, solve('S'));
    
    printf("%d", ans);
    return 0;
}
