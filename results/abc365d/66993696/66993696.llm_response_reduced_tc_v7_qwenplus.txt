#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int solve(int n) {
    // Try all possible starting characters for the transformed string
    int max_score = 0;
    
    // We'll try three possibilities: start with 'P', 'R', or 'S'
    vector<char> options = {'P', 'R', 'S'};
    for (char start : options) {
        c[1] = start;
        int score = 1; // first position always counts
        
        for (int i = 2; i <= n; i++) {
            // Determine what character would beat s[i]
            char beat = (s[i] == 'P') ? 'S' : (s[i] == 'R') ? 'P' : 'R';
            
            // If we can continue without conflict, use the beating character
            if (beat != c[i-1]) {
                c[i] = beat;
                score++;
            } else {
                // Must choose a different character, which might not beat s[i]
                // Try to pick one that doesn't conflict and preferably beats
                if (beat == 'P') {
                    c[i] = (c[i-1] != 'R') ? 'R' : 'S';
                } else if (beat == 'R') {
                    c[i] = (c[i-1] != 'S') ? 'S' : 'P';
                } else { // beat == 'S'
                    c[i] = (c[i-1] != 'P') ? 'P' : 'R';
                }
                // Only score if we actually beat s[i]
                if (c[i] == beat) {
                    score++;
                }
            }
        }
        max_score = max(max_score, score);
    }
    
    return max_score;
}

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s+1);
    
    if (n == 0) {
        printf("0");
        return 0;
    }
    
    if (n == 1) {
        // For single character, we can always choose the move that beats it
        printf("1");
        return 0;
    }
    
    int ans = solve(n);
    printf("%d", ans);
    return 0;
}
