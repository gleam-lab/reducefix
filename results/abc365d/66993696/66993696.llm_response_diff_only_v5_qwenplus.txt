#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int solve(int n) {
    // Try all possible starting choices for the first character
    int max_score = 0;
    
    // We'll try three strategies: change first char to each of 'P', 'R', 'S'
    for (char first : {'P', 'R', 'S'}) {
        c[1] = first;
        int score = 1; // Count first position
        
        for (int i = 2; i <= n; i++) {
            // Choose c[i] optimally based on previous choice
            if (s[i] == 'P') c[i] = 'S';
            else if (s[i] == 'R') c[i] = 'P';
            else c[i] = 'R';
            
            // If there's a conflict with previous character, we must use s[i] instead
            if (c[i] == c[i-1]) {
                c[i] = s[i];
            } else {
                score++;
            }
        }
        
        max_score = max(max_score, score);
    }
    
    return max_score;
}

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    printf("%d", solve(n));
    return 0;
}
