#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    // Function to get the winning move
    auto win = [](char x) {
        if (x == 'P') return 'S';
        if (x == 'R') return 'P';
        if (x == 'S') return 'R';
        return x;
    };
    
    // Try both possibilities for the first element: either change it or not
    int ans = 0;
    
    // Case 1: Do not force s[1] to be unchanged
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    
    int sum = n;
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];
            sum--;
        }
    }
    ans = max(ans, sum);
    
    // Case 2: Force c[1] = s[1] (i.e., no win at position 1)
    for (int i = 1; i <= n; i++) {
        c[i] = win(s[i]);
    }
    c[1] = s[1];
    sum = n - 1;
    
    for (int i = 2; i <= n; i++) {
        if (c[i] == c[i-1]) {
            c[i] = s[i];
            sum--;
        }
    }
    ans = max(ans, sum);
    
    printf("%d", ans);
    return 0;
}
