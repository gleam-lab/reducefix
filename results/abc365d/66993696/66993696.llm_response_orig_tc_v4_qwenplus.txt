#include<bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);
    
    // Define what each move beats
    auto beats = [](char a, char b) {
        if (a == 'R') return b == 'S';
        if (a == 'P') return b == 'R';
        if (a == 'S') return b == 'P';
        return false;
    };
    
    // Try all 3 possible starting moves for the first character
    int ans = 0;
    for (char first : {'R', 'P', 'S'}) {
        c[1] = first;
        int sum = 0;
        
        // If first move beats opponent's first move, we win this round
        if (beats(c[1], s[1])) sum++;
        
        // For remaining positions, choose optimally
        for (int i = 2; i <= n; i++) {
            // Try all three moves and pick the best one
            char best_move = c[i-1]; // Start with previous move
            int best_wins = -1;
            
            for (char move : {'R', 'P', 'S'}) {
                // We can't repeat moves
                if (move == c[i-1]) continue;
                
                int wins = 0;
                if (beats(move, s[i])) wins++;
                
                if (wins > best_wins) {
                    best_wins = wins;
                    best_move = move;
                }
            }
            
            c[i] = best_move;
            if (beats(c[i], s[i])) sum++;
        }
        
        ans = max(ans, sum);
    }
    
    printf("%d", ans);
    return 0;
}
