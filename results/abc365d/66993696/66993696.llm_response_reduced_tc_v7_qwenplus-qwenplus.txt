#include <bits/stdc++.h>
using namespace std;

char s[200011];
char c[200011];

// Function to compute the length of the longest run with no two adjacent characters being equal
int computeMaxLength(int n) {
    int len = 0;
    for (int i = 1; i <= n; i++) {
        if (i == 1 || c[i] != c[i - 1]) {
            len++;
        } else {
            // Try to change current character to original one and check if it helps
            c[i] = s[i];
            if (i == 1 || c[i] != c[i - 1]) {
                len++;
            }
        }
    }
    return len;
}

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Try all possible transformations: P->S, R->P, S->R
    // We'll try each starting transformation and pick the best result

    int max_len = 0;

    // There are 3 possible initial transformations
    char trans[] = {'P', 'R', 'S'};
    for (int t = 0; t < 3; t++) {
        c[1] = (s[1] == 'P' ? (t == 0 ? 'S' : (t == 1 ? 'P' : 'R')) :
                (s[1] == 'R' ? (t == 0 ? 'P' : (t == 1 ? 'S' : 'P')) : (t == 0 ? 'R' : (t == 1 ? 'P' : 'S'))));

        for (int i = 2; i <= n; i++) {
            // Default transformation
            if (s[i] == 'P') c[i] = 'S';
            else if (s[i] == 'R') c[i] = 'P';
            else c[i] = 'R';

            // Avoid duplicate with previous
            if (c[i] == c[i - 1]) {
                c[i] = s[i]; // Revert to original
            }
        }

        // Compute length of longest non-repeating sequence
        int curr_len = 0;
        for (int i = 1; i <= n; i++) {
            if (i == 1 || c[i] != c[i - 1]) {
                curr_len++;
            }
        }
        max_len = max(max_len, curr_len);
    }

    printf("%d\n", max_len);
}
