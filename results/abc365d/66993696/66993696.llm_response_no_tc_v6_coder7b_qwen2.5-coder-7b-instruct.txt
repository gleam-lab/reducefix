```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 200011;
char s[MAXN], c[MAXN];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);

    // Initialize counter for changes needed
    int changes = 0;
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';

        // Check if consecutive elements need to be changed
        if (i > 1 && c[i] == c[i - 1]) {
            c[i] = s[i]; // Change to match the previous element
            changes++;
        }
    }

    // Calculate maximum possible pairs
    int maxPairs = n - changes;

    // Special case: check if the first two elements need to be changed
    if (s[1] == s[2]) {
        c[1] = s[1];
        changes++;
        for (int i = 3; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];
                changes++;
            }
        }
        maxPairs = max(maxPairs, n - changes);
    }

    printf("%d\n", maxPairs);
    return 0;
}
```
