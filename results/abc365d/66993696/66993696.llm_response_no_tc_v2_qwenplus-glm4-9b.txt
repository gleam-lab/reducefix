#include<bits/stdc++.h>
using namespace std;
char s[200011];
char c[200011];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",s+1);  // Use s+1 to skip the initial placeholder character
    for(int i = 1; i <= n; i++)
    {
        if(s[i] == 'P') c[i] = 'S';
        else if(s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }
    
    int ans = n;  // Start with the full length as we can have all distinct characters at first
    int sum = 2;  // At least the first two characters are distinct

    for(int i = 2; i <= n; i++)
    {
        if(c[i] == c[i-1])
        {
            c[i] = s[i];  // Change to original character to maintain distinctness
            sum--;         // Decrease sum as we can't have two same characters in a row
        }
    }
    
    ans = max(ans, sum);  // Update the maximum distinct characters count

    // Reset characters to original mapping for second pass
    for(int i = 1; i <= n; i++)
    {
        if(s[i] == 'P') c[i] = 'S';
        else if(s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Check if the first two characters can be the same
    if(s[1] == s[2])
    {
        sum = 1;  // At most only the first character can be distinct
        for(int i = 3; i <= n; i++)
        {
            if(c[i] == c[i-1])
            {
                c[i] = s[i];  // Change to original character to maintain distinctness
                sum--;         // Decrease sum as we can't have two same characters in a row
            }
        }
    }
    else
    {
        sum = 2;  // The first two characters can't be the same, so they are distinct
    }
    
    ans = max(ans, sum);  // Update the maximum distinct characters count based on both passes

    printf("%d", ans);
}
