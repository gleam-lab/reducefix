#include<bits/stdc++.h>
using namespace std;
char s[200011];
char c[200011];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",s+1); // Start reading from index 1 to avoid null character issue
    c[1] = s[1]; // Initialize the first character of c to be the same as s[1]
    int sum = n; // Initialize sum of valid characters as n

    for(int i = 2; i <= n; i++) { // Start from 2 because we already handled the first character
        // Determine the next character in c based on the rules
        if(s[i] == 'P') c[i] = 'S';
        else if(s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';

        // If the next character in c is the same as the previous one, revert to s[i] and decrement sum
        if(c[i] == c[i-1]) {
            c[i] = s[i];
            sum--;
        }
    }

    // Store the maximum sum encountered
    int max_sum = sum;

    // Reset c to original transformation from s, but without any reversion
    for(int i = 1; i <= n; i++) {
        if(s[i] == 'P') c[i] = 'S';
        else if(s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // If the first two characters in the original string are the same, revert the first character only
    if(s[1] == s[2]) {
        c[1] = s[1];
        sum--;
        for(int i = 3; i <= n; i++) {
            if(c[i] == c[i-1]) {
                c[i] = s[i];
                sum--;
            }
        }
    }

    // Update the maximum sum
    max_sum = max(max_sum, sum);

    // Output the maximum sum
    printf("%d", max_sum);
}
