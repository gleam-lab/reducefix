#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    string s(n + 1);
    cin >> s;
    
    // Transform all characters as per the rules
    for(int i = 1; i <= n; ++i) {
        if(s[i] == 'P') s[i] = 'S';
        else if(s[i] == 'R') s[i] = 'P';
        else s[i] = 'R';
    }
    
    // Initialize answer and count of mismatches
    int ans = 0, mismatch = 0;
    
    // Count mismatches between consecutive characters
    for(int i = 2; i <= n; ++i) {
        if(s[i] != s[i - 1]) ++mismatch;
    }
    
    // If first character is different from second
    if(s[1] != s[2]) ++mismatch;
    
    // The minimum number of changes required is the number of mismatches
    ans = mismatch;
    
    // Try changing the first character
    if(s[1] != s[2]) {
        char original = s[1];
        s[1] = s[2];
        mismatch = 0;
        for(int i = 2; i <= n; ++i) {
            if(s[i] != s[i - 1]) ++mismatch;
        }
        ans = min(ans, mismatch);
        s[1] = original; // Revert back
    }
    
    printf("%d\n", ans);
    return 0;
}
