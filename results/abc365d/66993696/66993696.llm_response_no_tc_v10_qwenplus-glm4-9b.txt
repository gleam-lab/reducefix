#include<bits/stdc++.h>
using namespace std;
char s[200011];
char c[200011];

int main() {
    int n;
    scanf("%d", &n);
    scanf("%s", s + 1);  // Read the string starting from index 1 as the first character is not used
    for (int i = 1; i <= n; i++) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // First pass to minimize changes
    int sum = n;
    int i = 2;
    while (i <= n) {
        if (c[i] == c[i - 1]) {
            // Swap to minimize changes
            char temp = c[i - 1];
            c[i - 1] = c[i];
            c[i] = temp;
            sum--;
            i += 2;  // Move two steps forward as we've effectively handled two characters
        } else {
            i++;
        }
    }

    // Check the first two characters separately to avoid index out of bounds
    if (s[1] == s[2]) {
        c[1] = c[2] = s[1];  // Set both to s[1] or s[2]
        sum -= 2;
        i = 3;
        while (i <= n) {
            if (c[i] == c[i - 1]) {
                c[i] = s[i];  // Restore to original
                sum--;
                i += 2;
            } else {
                i++;
            }
        }
    }

    printf("%d", sum);
    return 0;
}
