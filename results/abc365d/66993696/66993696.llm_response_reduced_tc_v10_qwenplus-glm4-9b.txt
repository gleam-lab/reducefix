#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    char s[200011], c[200011];
    scanf("%s", s + 1);

    // Mapping characters according to the given rules
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    int ans = 1; // Start with the minimum possible value
    int sum = n; // Initial sum of elements in c

    // Loop through the array to apply the second rule
    for (int i = 2; i <= n; ++i) {
        if (c[i] == c[i - 1]) {
            sum--; // Decrease sum because the current element is the same as the previous
        }
    }

    ans = max(ans, sum); // Check if the sum is a better solution

    // Reset c array to apply the first rule
    for (int i = 1; i <= n; ++i) {
        if (s[i] == 'P') c[i] = 'S';
        else if (s[i] == 'R') c[i] = 'P';
        else c[i] = 'R';
    }

    // Check if the first two characters are the same
    if (s[1] == s[2]) {
        sum--; // Decrease sum because the first two characters are the same
        // Apply the second rule starting from the third element
        for (int i = 3; i <= n; ++i) {
            if (c[i] == c[i - 1]) {
                sum--; // Decrease sum because the current element is the same as the previous
            }
        }
        ans = max(ans, sum); // Check if the updated sum is a better solution
    }

    printf("%d", ans);
    return 0;
}
