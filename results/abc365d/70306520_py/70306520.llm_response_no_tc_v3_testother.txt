```python
N = int(input())
S = input().strip()

# Map Aoki's move to the winning move for Takahashi
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] will store the maximum wins up to game i, where j represents Takahashi's move:
# 0: Rock, 1: Paper, 2: Scissors
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for move_idx, move_char in enumerate("RPS"):
    if move_char != lose_move[S[0]]:  # Takahashi doesn't lose
        dp[0][move_idx] = 1 if move_char == win_move[S[0]] else 0

# Fill DP table
for i in range(1, N):
    for curr in range(3):
        curr_char = "RPS"[curr]
        # If this move makes Takahashi lose, skip
        if curr_char == lose_move[S[i]]:
            continue
        
        # Check all possible previous moves
        for prev in range(3):
            if prev == curr:  # Adjacent moves must be different
                continue
            if dp[i-1][prev] == -1:  # Invalid state
                continue
                
            wins = dp[i-1][prev] + (1 if curr_char == win_move[S[i]] else 0)
            if wins > dp[i][curr]:
                dp[i][curr] = wins

print(max(dp[N-1]))
```
