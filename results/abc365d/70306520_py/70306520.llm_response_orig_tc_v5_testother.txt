```python
N = int(input())
S = input().strip()

# Map opponent move to winning move
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] = maximum wins up to game i, where j represents last move (0=R, 1=P, 2=S)
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for j in range(3):
    move = "RPS"[j]
    if move != win_move[S[0]] and move != lose_move[S[0]]:  # Draw
        dp[0][j] = 0
    elif move == win_move[S[0]]:  # Win
        dp[0][j] = 1
    # Cannot lose, so no else case

# Fill DP table
for i in range(1, N):
    for last in range(3):
        if dp[i-1][last] == -1:
            continue
        for curr in range(3):
            if curr == last:  # Consecutive moves must be different
                continue
            move = "RPS"[curr]
            if move == win_move[S[i]]:
                dp[i][curr] = max(dp[i][curr], dp[i-1][last] + 1)
            elif move == lose_move[S[i]]:
                continue  # Cannot lose
            else:  # Draw
                dp[i][curr] = max(dp[i][curr], dp[i-1][last])

print(max(dp[N-1]))
```
