```python
N = int(input())
S = input().strip()

# Define winning moves: what move beats each of Aoki's moves
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}

# dp[i][j] will store the maximum wins up to game i, where j is the move used in game i (0=R, 1=P, 2=S)
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for move_idx, move_char in enumerate(['R', 'P', 'S']):
    if move_char == win_move[S[0]]:
        dp[0][move_idx] = 1  # Win
    elif move_char == S[0]:
        dp[0][move_idx] = 0  # Draw
    # Cannot lose, so no other options

# Fill DP table
for i in range(1, N):
    current_aoki = S[i]
    for curr in range(3):  # Current move (0=R, 1=P, 2=S)
        curr_char = ['R', 'P', 'S'][curr]
        
        # Check if this move is valid against Aoki's move (no loss)
        if (curr_char == 'R' and current_aoki == 'P') or \
           (curr_char == 'P' and current_aoki == 'S') or \
           (curr_char == 'S' and current_aoki == 'R'):
            continue  # This would be a loss, not allowed
        
        max_prev = -1
        for prev in range(3):  # Previous move
            if prev != curr and dp[i-1][prev] != -1:  # Different from previous and valid
                max_prev = max(max_prev, dp[i-1][prev])
        
        if max_prev == -1:
            continue  # No valid previous sequence
        
        # Calculate result of current game
        if curr_char == win_move[current_aoki]:
            dp[i][curr] = max_prev + 1  # Win
        else:
            dp[i][curr] = max_prev  # Draw

print(max(dp[N-1]))
```
