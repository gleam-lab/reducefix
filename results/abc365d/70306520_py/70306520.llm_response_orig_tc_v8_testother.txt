```python
N = int(input())
S = input().strip()

# Define winning moves: what move beats each of Aoki's moves
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] will store the maximum wins up to game i, where j represents the move Takahashi made in game i
# 0 = Rock (R), 1 = Paper (P), 2 = Scissors (S)
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for j, move in enumerate(['R', 'P', 'S']):
    if move == win_move[S[0]]:
        dp[0][j] = 1  # Win
    elif move == S[0]:
        dp[0][j] = 0  # Draw
    # If it's a losing move, we don't allow it (remains -1)

# Fill DP table
for i in range(1, N):
    current_opponent = S[i]
    for curr in range(3):
        curr_move = ['R', 'P', 'S'][curr]
        
        # Skip invalid moves (Takahashi loses)
        if curr_move == lose_move[current_opponent]:
            continue
            
        # Determine score for this move
        if curr_move == win_move[current_opponent]:
            win_bonus = 1
        else:  # draw
            win_bonus = 0
            
        # Check all valid previous moves (different from current)
        best_prev = -1
        for prev in range(3):
            if prev != curr and dp[i-1][prev] != -1:
                best_prev = max(best_prev, dp[i-1][prev])
                
        if best_prev != -1:
            dp[i][curr] = best_prev + win_bonus

print(max(dp[N-1]))
```
