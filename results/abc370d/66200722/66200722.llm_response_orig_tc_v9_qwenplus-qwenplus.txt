```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9 + 7;

int H, W, Q;
set<int> row_has_bomb[400010];
set<int> col_has_bomb[400010];

bool visited[400010];

inline int get_index(int r, int c) {
    return (r - 1) * W + (c - 1);
}

void place_bomb(int r, int c) {
    int idx = get_index(r, c);
    if (visited[idx]) return;
    visited[idx] = true;

    // Remove from row and column sets
    row_has_bomb[r].erase(c);
    col_has_bomb[c].erase(r);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> H >> W >> Q;

    // Initialize all walls present
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row_has_bomb[i].insert(j);
            col_has_bomb[j].insert(i);
        }
    }

    memset(visited, 0, sizeof(visited));

    while (Q--) {
        int r, c;
        cin >> r >> c;

        int idx = get_index(r, c);

        if (!visited[idx]) {
            // Bomb at (r,c) has not been placed yet
            place_bomb(r, c);
        } else {
            // Bomb at (r,c) already exploded, now explode in four directions

            // Up
            auto it = col_has_bomb[c].lower_bound(r);
            if (it != col_has_bomb[c].begin()) {
                --it;
                place_bomb(*it, c);
            }

            // Down
            it = col_has_bomb[c].upper_bound(r);
            if (it != col_has_bomb[c].end()) {
                place_bomb(*it, c);
            }

            // Left
            it = row_has_bomb[r].lower_bound(c);
            if (it != row_has_bomb[r].begin()) {
                --it;
                place_bomb(r, *it);
            }

            // Right
            it = row_has_bomb[r].upper_bound(c);
            if (it != row_has_bomb[r].end()) {
                place_bomb(r, *it);
            }
        }
    }

    long long remaining = 0;
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            int idx = get_index(i, j);
            if (!visited[idx]) remaining++;
        }
    }

    cout << remaining << endl;

    return 0;
}
```
