#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll INF = 1e18;

struct DSU {
    vector<ll> par, sz;
    DSU(ll n) : par(n), sz(n, 1) {
        iota(par.begin(), par.end(), 0);
    }

    ll find(ll u) {
        if(u == par[u]) return u;
        return par[u] = find(par[u]);
    }

    void unite(ll u, ll v) {
        u = find(u), v = find(v);
        if(u == v) return;
        if(sz[u] < sz[v]) swap(u, v);
        par[v] = u;
        sz[u] += sz[v];
    }

    bool same(ll u, ll v) {
        return find(u) == find(v);
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    ll h, w, q;
    cin >> h >> w >> q;

    vector<vector<pair<ll, ll>>> bombs(h, vector<pair<ll, ll>>(w));

    DSU dsu(h * w);

    while(q--) {
        ll r, c;
        cin >> r >> c;
        r--; c--;

        if(bombs[r][c].first == 0 && bombs[r][c].second == 0) {
            for(ll dr = -1; dr <= 1; dr++) {
                for(ll dc = -1; dc <= 1; dc++) {
                    if(dr == 0 && dc == 0) continue;
                    ll nr = r + dr, nc = c + dc;
                    if(nr < 0 || nr >= h || nc < 0 || nc >= w) continue;
                    if(dsu.find(nr * w + nc) != dsu.find(r * w + c)) {
                        dsu.unite(nr * w + nc, r * w + c);
                    }
                }
            }
            bombs[r][c] = {1, dsu.find(r * w + c)};
        } else {
            bombs[r][c] = {2, dsu.find(r * w + c)};
        }
    }

    ll ans = 0;
    for(ll i = 0; i < h; i++) {
        for(ll j = 0; j < w; j++) {
            if(bombs[i][j].first == 1 && dsu.find(i * w + j) == dsu.find(bombs[i][j].second)) {
                ans++;
            }
        }
    }

    cout << ans << endl;

    return 0;
}
