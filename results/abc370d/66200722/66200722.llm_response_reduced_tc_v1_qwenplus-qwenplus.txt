#include <bits/stdc++.h>
using namespace std;

void solve();
int main() {
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    solve();
}

void solve() {
    int h, w, q;
    cin >> h >> w >> q;

    // For each row and column, maintain a set of bomb positions
    vector<set<int>> row_bombs(h), col_bombs(w);

    // Initially all cells have walls
    long long remaining_walls = 1LL * h * w;

    // Track if a cell has been bombed (to avoid double counting)
    vector<vector<bool>> bombed(h, vector<bool>(w, false));

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (bombed[r][c]) {
            // Wall already destroyed, look in all four directions
            // Up
            auto it = row_bombs[r].lower_bound(c);
            if (it != row_bombs[r].begin()) {
                it--;
                int destroy_col = *it;
                if (!bombed[r][destroy_col]) {
                    bombed[r][destroy_col] = true;
                    remaining_walls--;
                }
            }

            // Down
            it = row_bombs[r].upper_bound(c);
            if (it != row_bombs[r].end()) {
                int destroy_col = *it;
                if (!bombed[r][destroy_col]) {
                    bombed[r][destroy_col] = true;
                    remaining_walls--;
                }
            }

            // Left
            auto cit = col_bombs[c].lower_bound(r);
            if (cit != col_bombs[c].begin()) {
                cit--;
                int destroy_row = *cit;
                if (!bombed[destroy_row][c]) {
                    bombed[destroy_row][c] = true;
                    remaining_walls--;
                }
            }

            // Right
            cit = col_bombs[c].upper_bound(r);
            if (cit != col_bombs[c].end()) {
                int destroy_row = *cit;
                if (!bombed[destroy_row][c]) {
                    bombed[destroy_row][c] = true;
                    remaining_walls--;
                }
            }
        } else {
            // First time bombing this cell
            bombed[r][c] = true;
            remaining_walls--;
            row_bombs[r].insert(c);
            col_bombs[c].insert(r);
        }
    }

    cout << remaining_walls << endl;
}
