int main() {

        row_walls[i].insert(W + 1);

            remaining--;

            auto it = col_walls[C].lower_bound(R);
            --it;
            if (*it > 0 && !row_walls[*it].count(C)) {
                row_walls[*it].insert(C);
                col_walls[C].insert(*it);
                remaining--;
            }
            
            // Look down
            it = col_walls[C].upper_bound(R);
            if (*it <= H && !row_walls[*it].count(C)) {
                row_walls[*it].insert(C);

            }
