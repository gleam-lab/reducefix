#include <iostream>
#include <vector>
#include <set>

using namespace std;

const int MAX_H = 4e5 + 10;
const int MAX_W = 4e5 + 10;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> rows(h);
    vector<set<int>> cols(w);
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        
        if (rows[r].find(c) != rows[r].end()) {
            rows[r].erase(rows[r].find(c));
            cols[c].erase(cols[c].find(r));
        } else {
            rows[r].insert(c);
            cols[c].insert(r);
        }
    }
    
    long long remaining_walls = 0LL;
    for (int i = 0; i < h; ++i) {
        remaining_walls += (w - rows[i].size());
    }
    for (int j = 0; j < w; ++j) {
        remaining_walls += (h - cols[j].size());
    }
    
    cout << remaining_walls << endl;
    
    return 0;
}
