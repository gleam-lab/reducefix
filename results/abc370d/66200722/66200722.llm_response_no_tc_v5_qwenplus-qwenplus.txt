#include <bits/stdc++.h>
using namespace std;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Grid to track presence of a bomb at each cell
    vector<unordered_set<int>> row_bombs(H), col_bombs(W);
    vector<vector<bool>> has_bomb(H, vector<bool>(W, false));
    
    auto add_bomb = [&](int r, int c) {
        has_bomb[r][c] = true;
        row_bombs[r].insert(c);
        col_bombs[c].insert(r);
    };

    auto remove_bomb = [&](int r, int c) {
        has_bomb[r][c] = false;
        row_bombs[r].erase(c);
        col_bombs[c].erase(r);
    };

    // Add initial bombs (none)
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            add_bomb(i, j); // Initially all cells have bombs
        }
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (!has_bomb[R][C]) {
            // Bomb already exists here - destroy walls in four directions

            // Up
            auto it = row_bombs[R].lower_bound(C);
            if (it != row_bombs[R].begin()) {
                --it;
                remove_bomb(R, *it);
            }

            // Down
            it = row_bombs[R].upper_bound(C);
            if (it != row_bombs[R].end()) {
                remove_bomb(R, *it);
            }

            // Left
            auto cit = col_bombs[C].lower_bound(R);
            if (cit != col_bombs[C].begin()) {
                --cit;
                remove_bomb(*cit, C);
            }

            // Right
            cit = col_bombs[C].upper_bound(R);
            if (cit != col_bombs[C].end()) {
                remove_bomb(*cit, C);
            }

        } else {
            // No bomb yet - place one
            remove_bomb(R, C);
        }
    }

    long long remaining = 0;
    for (int i = 0; i < H; ++i) {
        remaining += row_bombs[i].size();
    }
    cout << remaining << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
