#include <bits/stdc++.h>
using namespace std;

using pii = pair<int, int>;
using vi = vector<int>;
using vpii = vector<pii>;

template <class T> using vec = vector<T>;
template <class T> using vvec = vector<vector<T>>;

#define all(v) (v).begin(), (v).end()
#define rep(i, n) for (int i = 0; i < (int)(n); ++i)
#define debug(...) ;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // We'll use a grid to track if a wall exists at each cell
    unordered_set<long long> has_wall;
    auto pos = [&](int x, int y) { return (1LL * x << 30) | y; };

    // Initialize: every cell has a wall
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            has_wall.insert(pos(i, j));
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        long long cur = pos(r, c);

        if (has_wall.count(cur)) {
            // If wall exists, destroy it
            has_wall.erase(cur);
        } else {
            // Otherwise, destroy first walls in each direction

            // Up
            for (int i = r - 1; i >= 0; --i) {
                long long p = pos(i, c);
                if (has_wall.count(p)) {
                    has_wall.erase(p);
                    break;
                }
            }

            // Down
            for (int i = r + 1; i < H; ++i) {
                long long p = pos(i, c);
                if (has_wall.count(p)) {
                    has_wall.erase(p);
                    break;
                }
            }

            // Left
            for (int j = c - 1; j >= 0; --j) {
                long long p = pos(r, j);
                if (has_wall.count(p)) {
                    has_wall.erase(p);
                    break;
                }
            }

            // Right
            for (int j = c + 1; j < W; ++j) {
                long long p = pos(r, j);
                if (has_wall.count(p)) {
                    has_wall.erase(p);
                    break;
                }
            }
        }
    }

    cout << has_wall.size() << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
