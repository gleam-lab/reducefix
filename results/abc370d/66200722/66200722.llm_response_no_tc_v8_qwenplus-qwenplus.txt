#include <bits/stdc++.h>
using namespace std;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Use 2D grid to track presence of wall in each cell
    vector<vector<bool>> has_wall(H, vector<bool>(W, true));

    // For each row and column, maintain ordered sets of remaining walls
    vector<set<int>> row_walls(H), col_walls(W);

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            row_walls[i].insert(j);
            col_walls[j].insert(i);
        }
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (has_wall[R][C]) {
            // Bomb destroys the wall at its position
            has_wall[R][C] = false;
            row_walls[R].erase(C);
            col_walls[C].erase(R);
        } else {
            // Bomb searches in four directions for nearest walls
            vector<pair<int, int>> to_remove;

            // Up
            auto up_it = col_walls[C].lower_bound(R);
            if (up_it != col_walls[C].begin()) {
                --up_it;
                to_remove.emplace_back(*up_it, C);
            }

            // Down
            auto down_it = col_walls[C].upper_bound(R);
            if (down_it != col_walls[C].end()) {
                to_remove.emplace_back(*down_it, C);
            }

            // Left
            auto left_it = row_walls[R].lower_bound(C);
            if (left_it != row_walls[R].begin()) {
                --left_it;
                to_remove.emplace_back(R, *left_it);
            }

            // Right
            auto right_it = row_walls[R].upper_bound(C);
            if (right_it != row_walls[R].end()) {
                to_remove.emplace_back(R, *right_it);
            }

            // Remove found walls
            for (auto [r, c] : to_remove) {
                if (has_wall[r][c]) {
                    has_wall[r][c] = false;
                    row_walls[r].erase(c);
                    col_walls[c].erase(r);
                }
            }
        }
    }

    // Count remaining walls
    int result = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (has_wall[i][j]) result++;
        }
    }

    cout << result << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
