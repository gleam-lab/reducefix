#include <iostream>
#include <vector>
#include <set>
#include <cassert>

using namespace std;

struct UF {
    vector<int> id;
    vector<int> sz;
    
    UF(int n): id(n), sz(n, 1) {
        iota(id.begin(), id.end(), 0);
    }

    int find(int u) {
        while (u != id[u]) {
            id[u] = id[id[u]];
            u = id[u];
        }
        return u;
    }

    void unite(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (pu == pv) return;
        
        if (sz[pu] > sz[pv]) swap(pu, pv);
        id[pu] = pv;
        sz[pv] += sz[pu];
    }
};

void solve() {
    int h, w, q;
    cin >> h >> w >> q;
    
    UF uf(h*w+2);

    vector<pair<int, int>> bombs;
    for(int i=0; i<h; ++i){
        for(int j=0; j<w; ++j){
            bombs.emplace_back(i*w+j, (i+1)*w+(j+1));
        }
    }
    
    vector<bool> removed(h*w, false);
    
    for(int qi=0; qi<q; ++qi){
        int ri, ci;
        cin >> ri >> ci;
        --ri, --ci;
        int bi = ri*w + ci;
        
        if(removed[bi]){
            continue;
        }
        
        int ni = bombs[bi].first, nj = bombs[bi].second;
        removed[bi] = true;
        
        if(ni < h*w){
            int pi = uf.find(ni), pj = uf.find(nj);
            
            if(pi != nj && !removed[ni]){ // check right neighbor
                uf.unite(pi, pj);
            }
        }
        
        if(ni > 0){
            int pi = uf.find(ni), pj = uf.find(nj);
            
            if(pi != nj && !removed[ni-1]){ // check left neighbor
                uf.unite(pi, pj);
            }
        }
        
        if(nj < h*w){
            int pi = uf.find(ni), pj = uf.find(nj);
            
            if(pi != nj && !removed[nj]){ // check bottom neighbor
                uf.unite(pi, pj);
            }
        }
        
        if(nj > 0){
            int pi = uf.find(ni), pj = uf.find(nj);
            
            if(pi != nj && !removed[nj-1]){ // check top neighbor
                uf.unite(pi, pj);
            }
        }
    }
    
    int res = 0;
    for(bool r: removed){
        if(!r){
            ++res;
        }
    }
    
    cout << res << endl;
}

    cin.tie(NULL);
    cout.tie(NULL);
    
    int tc = 1;
    // cin >> tc;
