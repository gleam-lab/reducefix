
```cpp
#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <cstring>

using namespace std;

const int MAXN = 400005; // Maximum possible cells in the grid

vector<int> parent(MAXN);
vector<int> rank(MAXN);

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]); // Path compression
    }
    return parent[x];
}

void union_sets(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);

    if (rootX != rootY) {
        if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
    }
}

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    memset(parent, -1, sizeof(parent));
    memset(rank, 0, sizeof(rank));
    int totalWalls = h * w;

    for (int i = 0; i < q; ++i) {
        int r, c;
        cin >> r >> c;

        r--; c--; // Convert to 0-based index

        // Perform the union operation in all four directions
        if (r > 0) union_sets(r, c); // Upward
        if (r < h - 1) union_sets(r + 1, c); // Downward
        if (c > 0) union_sets(r, c); // Leftward
        if (c < w - 1) union_sets(r, c + 1); // Rightward
    }

    // Find the size of each component and calculate the number of remaining walls
    map<int, int> componentSize;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            int root = find(i * w + j);
            componentSize[root]++;
        }
    }

    int remainingWalls = totalWalls;
    for (auto& p : componentSize) {
        remainingWalls -= (p.second * 4 - 3); // Each component has 4 walls, minus 3 as they share walls
    }

    cout << remainingWalls << endl;

    return 0;
}
```

This fixed version uses a Union-Find data structure to efficiently manage the merging of cells and walls. After processing all queries, it calculates the number of remaining walls by subtracting the total number of walls in the grid from the sum of walls within each component, adjusted for shared walls.
