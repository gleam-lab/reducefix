#include <bits/stdc++.h>
using namespace std;
#define int long long
#define INFLL 1e18
#define endl '\n'

struct UnionFind {
    vector<int> parent;
    vector<int> rank;

    UnionFind(int n) {
        parent.resize(n);
        rank.resize(n);
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    bool merge(int x, int y) {
        int px = find(x);
        int py = find(y);
        if (px != py) {
            if (rank[px] < rank[py]) swap(px, py);
            parent[py] = px;
            if (rank[px] == rank[py]) rank[px]++;
            return true;
        }
        return false;
    }
};

int32_t main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<int>> grid(H, vector<int>(W, 0));
    vector<vector<int>> up(H + 1, vector<int>(W, -1));
    vector<vector<int>> down(H + 1, vector<int>(W, -1));
    vector<vector<int>> left(H, vector<int>(W + 1, -1));
    vector<vector<int>> right(H, vector<int>(W + 1, -1));

    UnionFind uf(H * W);

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i > 0) down[i][j] = up[i - 1][j];
            if (j > 0) right[i][j] = left[i][j - 1];
            up[i][j] = left[i][j];
        }
    }

    for (int i = H - 1; i >= 0; --i) {
        for (int j = 0; j < W; ++j) {
            down[i][j] = -1;
        }
    }

    for (int j = W - 1; j >= 0; --j) {
        for (int i = 0; i < H; ++i) {
            right[i][j] = -1;
        }
    }

    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        if (grid[r][c]) continue;

        int idx = r * W + c;
        grid[r][c] = 1;

        if (up[r][c] != -1) {
            uf.merge(idx, up[r][c]);
        }
        if (down[r][c] != -1) {
            uf.merge(idx, down[r][c]);
        }
        if (left[r][c] != -1) {
            uf.merge(idx, left[r][c]);
        }
        if (right[r][c] != -1) {
            uf.merge(idx, right[r][c]);
        }
    }

    int wall_count = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!grid[i][j]) {
                wall_count += 4;
            } else {
                int idx = i * W + j;
                if (up[i][j] != -1) {
                    int up_idx = up[i][j] * W + j;
                    if (uf.find(idx) != uf.find(up_idx)) {
                        wall_count -= 1;
                    }
                }
                if (down[i][j] != -1) {
                    int down_idx = down[i][j] * W + j;
                    if (uf.find(idx) != uf.find(down_idx)) {
                        wall_count -= 1;
                    }
                }
                if (left[i][j] != -1) {
                    int left_idx = left[i][j] * W + j;
                    if (uf.find(idx) != uf.find(left_idx)) {
                        wall_count -= 1;
                    }
                }
                if (right[i][j] != -1) {
                    int right_idx = right[i][j] * W + j;
                    if (uf.find(idx) != uf.find(right_idx)) {
                        wall_count -= 1;
                    }
                }
            }
        }
    }

    cout << wall_count << endl;

    return 0;
}
