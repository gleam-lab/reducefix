#include<bits/stdc++.h>
using namespace std;

#define pb push_back
#define fi first
#define se second
#define mp make_pair
#define sz(x) ((int)(x).size())

typedef long long ll;
const int mod=1e9+7;
const double PI=acos(-1.0);

vector<bool> grid;

bool check(int i,int j){
    return i>=0&&i<sz(grid)&&j>=0&&j<grid[i].size();
}

void solve(){
    int h,w,q;
    cin>>h>>w>>q;
    grid.resize(h,vector<bool>(w,true));

    while(q--){
        int r,c;
        cin>>r>>c;
        --r,--c;
        if(grid[r][c]){
            grid[r][c]=false;
            q--;
        }
        else{
            int dr[]={0,0,1,-1},dc[]={1,-1,0,0};
            for(int i=0;i<4;i++){
                int nr=r+dr[i],nc=c+dc[i];
                while(check(nr,nc)&&!grid[nr][nc]){
                    nr+=dr[i];
                    nc+=dc[i];
                }
                if(check(nr,nc)){
                    grid[nr][nc]=false;
                    q--;
                }
            }
        }
    }

    int ans=0;
    for(int i=0;i<h;i++){
        for(int j=0;j<w;j++){
            if(grid[i][j])ans++;
        }
    }
    cout<<ans<<"\n";
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int t=1;
    // cin>>t;
    while(t--){
        solve();
    }
    return 0;
}

It uses a boolean grid to represent the presence or absence of walls, checks each query according to the rules provided, and updates the grid accordingly. Finally, it counts and prints the remaining walls. The solution is efficient and handles the constraints appropriately.
