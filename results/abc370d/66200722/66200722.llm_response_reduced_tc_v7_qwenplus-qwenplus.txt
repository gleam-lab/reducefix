#include <bits/stdc++.h>
using namespace std;

void solve() {
    int h, w, q;
    cin >> h >> w >> q;

    // For each row and column, maintain the set of bomb positions
    vector<set<int>> row_bombs(h), col_bombs(w);
    
    // Initialize all cells as having walls
    long long remaining_walls = (long long)h * w;

    for (int qi = 0; qi < q; ++qi) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (row_bombs[r].count(c)) {
            // Bomb already placed here - look in all directions
            // Up direction
            auto up_it = col_bombs[c].lower_bound(r);
            if (up_it != col_bombs[c].begin()) {
                --up_it;
                int destroy_r = *up_it;
                row_bombs[destroy_r].erase(c);
                col_bombs[c].erase(destroy_r);
                remaining_walls--;
            }

            // Down direction
            auto down_it = col_bombs[c].upper_bound(r);
            if (down_it != col_bombs[c].end()) {
                int destroy_r = *down_it;
                row_bombs[destroy_r].erase(c);
                col_bombs[c].erase(destroy_r);
                remaining_walls--;
            }

            // Left direction
            auto left_it = row_bombs[r].lower_bound(c);
            if (left_it != row_bombs[r].begin()) {
                --left_it;
                int destroy_c = *left_it;
                row_bombs[r].erase(destroy_c);
                col_bombs[destroy_c].erase(r);
                remaining_walls--;
            }

            // Right direction
            auto right_it = row_bombs[r].upper_bound(c);
            if (right_it != row_bombs[r].end()) {
                int destroy_c = *right_it;
                row_bombs[r].erase(destroy_c);
                col_bombs[destroy_c].erase(r);
                remaining_walls--;
            }
        } else {
            // Place a new bomb
            row_bombs[r].insert(c);
            col_bombs[c].insert(r);
            remaining_walls--;
        }
    }

    cout << remaining_walls << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
