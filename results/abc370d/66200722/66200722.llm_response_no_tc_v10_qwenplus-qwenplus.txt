#include <bits/stdc++.h>
using namespace std;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    vector<map<int, bool>> row(H), col(W);
    long long remaining = 1LL * H * W;

    auto place_bomb = [&](int r, int c) {
        if (row[r].find(c) != row[r].end()) {
            // Bomb destroys the wall at its position
            row[r][c] = false;
            col[c][r] = false;
            remaining--;
        } else {
            // Look up
            auto it_up = col[c].lower_bound(r);
            if (it_up != col[c].begin()) {
                --it_up;
                int pos = it_up->first;
                row[pos][c] = false;
                col[c][pos] = false;
                remaining--;
            }

            // Look down
            auto it_down = col[c].upper_bound(r);
            if (it_down != col[c].end()) {
                int pos = it_down->first;
                row[pos][c] = false;
                col[c][pos] = false;
                remaining--;
            }

            // Look left
            auto it_left = row[r].lower_bound(c);
            if (it_left != row[r].begin()) {
                --it_left;
                int pos = it_left->first;
                row[r][pos] = false;
                col[pos][r] = false;
                remaining--;
            }

            // Look right
            auto it_right = row[r].upper_bound(c);
            if (it_right != row[r].end()) {
                int pos = it_right->first;
                row[r][pos] = false;
                col[pos][r] = false;
                remaining--;
            }
        }

        // Mark bomb position
        row[r][c] = true;
        col[c][r] = true;
    };

    for (int i = 0; i < Q; ++i) {
        int R, C;
        cin >> R >> C;
        R--; C--;
        place_bomb(R, C);
    }

    cout << remaining << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
