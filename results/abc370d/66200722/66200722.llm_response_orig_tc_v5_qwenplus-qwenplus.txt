#include <bits/stdc++.h>
using namespace std;

const int INF = 1 << 30;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<bool>> wall(H, vector<bool>(W, true));

    auto valid = [&](int r, int c) {
        return 0 <= r && r < H && 0 <= c && c < W;
    };

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        R--, C--;

        if (wall[R][C]) {
            wall[R][C] = false;
        } else {
            // up
            for (int r = R - 1; r >= 0; --r) {
                if (wall[r][C]) {
                    wall[r][C] = false;
                    break;
                }
            }

            // down
            for (int r = R + 1; r < H; ++r) {
                if (wall[r][C]) {
                    wall[r][C] = false;
                    break;
                }
            }

            // left
            for (int c = C - 1; c >= 0; --c) {
                if (wall[R][c]) {
                    wall[R][c] = false;
                    break;
                }
            }

            // right
            for (int c = C + 1; c < W; ++c) {
                if (wall[R][c]) {
                    wall[R][c] = false;
                    break;
                }
            }
        }
    }

    int count = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (wall[i][j]) count++;
        }
    }

    cout << count << endl;

    return 0;
}
