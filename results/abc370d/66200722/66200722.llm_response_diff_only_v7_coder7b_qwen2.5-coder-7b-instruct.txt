```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 400005;
bool grid[MAXN][MAXN];

int main(){
    int H, W, Q;
    scanf("%d%d%d", &H, &W, &Q);

    memset(grid, true, sizeof(grid));

    while(Q--){
        int R, C;
        scanf("%d%d", &R, &C);
        R--, C--;

        if(grid[R][C]){
            grid[R][C] = false;
        } else {
            for(int i=R-1; i>=0 && grid[i][C]; i--) grid[i][C] = false;
            for(int i=R+1; i<H && grid[i][C]; i++) grid[i][C] = false;
            for(int j=C-1; j>=0 && grid[R][j]; j--) grid[R][j] = false;
            for(int j=C+1; j<W && grid[R][j]; j++) grid[R][j] = false;
        }
    }

    int count = 0;
    for(int i=0; i<H; i++){
        for(int j=0; j<W; j++){
            if(grid[i][j]) count++;
        }
    }

    printf("%d\n", count);

    return 0;
}
```
This solution fixes several issues in the original code:

1. It simplifies the logic for placing bombs by directly destroying walls in the specified direction until a non-wall or the boundary is reached.
2. It uses a simple array-based grid to represent the state of the cells instead of complex data structures like maps and union-find.
3. It counts the remaining walls by iterating through the entire grid once, which is efficient given the problem constraints.
