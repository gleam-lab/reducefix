```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>

using namespace std;

const int MAXN = 400005;

struct Query {
    int type, r, c;
};

struct Cell {
    set<int> rows, cols;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<Cell> rows(H), cols(W);
    vector<Query> queries(Q);

    for(int i = 0; i < Q; i++) {
        cin >> queries[i].type >> queries[i].r >> queries[i].c;
        queries[i].r--; queries[i].c--;
    }

    vector<bool> destroyed(H * W, false);

    for(int i = 0; i < Q; i++) {
        int type = queries[i].type;
        int r = queries[i].r;
        int c = queries[i].c;

        if(type == 1) {
            if(destroyed[r * W + c]) continue;

            destroyed[r * W + c] = true;
            rows[r].cols.erase(c);
            cols[c].rows.erase(r);
        } else {
            if(!destroyed[r * W + c]) continue;

            destroyed[r * W + c] = false;
            rows[r].cols.insert(c);
            cols[c].rows.insert(r);

            int left = c;
            while(left > 0 && !destroyed[r * W + left - 1]) {
                rows[r].cols.insert(left - 1);
                cols[left - 1].rows.insert(r);
                left--;
            }

            int right = c;
            while(right < W - 1 && !destroyed[r * W + right + 1]) {
                rows[r].cols.insert(right + 1);
                cols[right + 1].rows.insert(r);
                right++;
            }

            int up = r;
            while(up > 0 && !destroyed[(up - 1) * W + c]) {
                cols[c].rows.insert(up - 1);
                rows[up - 1].cols.insert(c);
                up--;
            }

            int down = r;
            while(down < H - 1 && !destroyed[(down + 1) * W + c]) {
                cols[c].rows.insert(down + 1);
                rows[down + 1].cols.insert(c);
                down++;
            }
        }
    }

    int count = 0;
    for(int i = 0; i < H; i++) {
        count += rows[i].cols.size();
    }
    for(int i = 0; i < W; i++) {
        count += cols[i].rows.size();
    }
    cout << count << endl;

    return 0;
}
```
