#include <bits/stdc++.h>
using namespace std;

using vi = vector<int>;
using pii = pair<int, int>;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row and column, track the positions that have been bombed
    vector<set<int>> row_bombs(H), col_bombs(W);

    // Track destroyed cells
    set<pair<int, int>> destroyed;

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (destroyed.count({R, C})) {
            // Bomb in four directions
            // Up
            auto it = row_bombs[R].lower_bound(C);
            if (it != row_bombs[R].begin()) {
                --it;
                int left = *it;
                if (left >= 0 && !destroyed.count({R, left})) {
                    destroyed.insert({R, left});
                }
            }

            // Down
            it = row_bombs[R].upper_bound(C);
            if (it != row_bombs[R].end()) {
                int right = *it;
                if (right < W && !destroyed.count({R, right})) {
                    destroyed.insert({R, right});
                }
            }

            // Left
            auto cit = col_bombs[C].lower_bound(R);
            if (cit != col_bombs[C].begin()) {
                --cit;
                int up = *cit;
                if (up >= 0 && !destroyed.count({up, C})) {
                    destroyed.insert({up, C});
                }
            }

            // Right
            cit = col_bombs[C].upper_bound(R);
            if (cit != col_bombs[C].end()) {
                int down = *cit;
                if (down < H && !destroyed.count({down, C})) {
                    destroyed.insert({down, C});
                }
            }
        } else {
            // Place bomb at this position
            destroyed.insert({R, C});
            row_bombs[R].insert(C);
            col_bombs[C].insert(R);
        }
    }

    // Total remaining walls = total cells - number of destroyed cells
    cout << 1LL * H * W - destroyed.size() << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
