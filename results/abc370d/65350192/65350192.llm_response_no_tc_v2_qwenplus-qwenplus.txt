#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> row_sets(h), col_sets(w);

    // Initialize all cells to have walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row_sets[i].insert(j);
            col_sets[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there is a wall at (r, c)
        if (row_sets[r].count(c)) {
            // Destroy the wall at (r, c)
            row_sets[r].erase(c);
            col_sets[c].erase(r);
        } else {
            // Up direction
            auto it = col_sets[c].lower_bound(r);
            if (it != col_sets[c].begin()) {
                it--;
                int destroy_r = *it;
                row_sets[destroy_r].erase(c);
                col_sets[c].erase(it);
            }

            // Down direction
            it = col_sets[c].lower_bound(r);
            if (it != col_sets[c].end()) {
                int destroy_r = *it;
                row_sets[destroy_r].erase(c);
                col_sets[c].erase(it);
            }

            // Left direction
            it = row_sets[r].lower_bound(c);
            if (it != row_sets[r].begin()) {
                it--;
                int destroy_c = *it;
                col_sets[destroy_c].erase(r);
                row_sets[r].erase(it);
            }

            // Right direction
            it = row_sets[r].lower_bound(c);
            if (it != row_sets[r].end()) {
                int destroy_c = *it;
                col_sets[destroy_c].erase(r);
                row_sets[r].erase(it);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += row_sets[i].size();
    }
    cout << ans << "\n";

    return 0;
}
