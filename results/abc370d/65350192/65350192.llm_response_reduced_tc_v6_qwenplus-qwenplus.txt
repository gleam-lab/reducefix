```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> rows(h), cols(w);
    
    // Initialize sets with all columns/rows having walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (rows[r].count(c)) {
            // Cell has a wall - remove it
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Look in each direction and remove first wall found
            
            // Down
            auto down = cols[c].lower_bound(r);
            if (down != cols[c].end()) {
                int row = *down;
                cols[c].erase(row);
                rows[row].erase(c);
            }
            
            // Up
            auto up = cols[c].lower_bound(r);
            if (up != cols[c].begin()) {
                --up;
                int row = *up;
                cols[c].erase(row);
                rows[row].erase(c);
            }
            
            // Right
            auto right = rows[r].lower_bound(c);
            if (right != rows[r].end()) {
                int col = *right;
                rows[r].erase(col);
                cols[col].erase(r);
            }
            
            // Left
            auto left = rows[r].lower_bound(c);
            if (left != rows[r].begin()) {
                --left;
                int col = *left;
                rows[r].erase(col);
                cols[col].erase(r);
            }
        }
    }
    
    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i)
        ans += rows[i].size();
    
    cout << ans << "\n";
    return 0;
}
```
