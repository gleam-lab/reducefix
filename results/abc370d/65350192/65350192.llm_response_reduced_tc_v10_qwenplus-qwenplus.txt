#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // For each column, keep track of remaining walls in that column (rows)
    vector<set<int>> colWalls(w);
    // For each row, keep track of remaining walls in that row (columns)
    vector<set<int>> rowWalls(h);
    
    // Initialize all walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            colWalls[j].insert(i);
            rowWalls[i].insert(j);
        }
    }
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        // Check if there's a wall at the bomb location
        if (colWalls[c].count(r)) {
            // Directly destroy the wall at this position
            colWalls[c].erase(r);
            rowWalls[r].erase(c);
        } else {
            // Look up: find first wall above (smaller row index)
            auto itUp = colWalls[c].lower_bound(r);
            if (itUp != colWalls[c].end()) {
                int row = *itUp;
                colWalls[c].erase(row);
                rowWalls[row].erase(c);
            }
            
            // Look down: find first wall below (larger row index)
            itUp = colWalls[c].lower_bound(r);
            if (itUp != colWalls[c].begin()) {
                --itUp;
                int row = *itUp;
                colWalls[c].erase(row);
                rowWalls[row].erase(c);
            }
            
            // Look right: find first wall to the right
            auto itRight = rowWalls[r].lower_bound(c);
            if (itRight != rowWalls[r].end()) {
                int col = *itRight;
                rowWalls[r].erase(col);
                colWalls[col].erase(r);
            }
            
            // Look left: find first wall to the left
            itRight = rowWalls[r].lower_bound(c);
            if (itRight != rowWalls[r].begin()) {
                --itRight;
                int col = *itRight;
                rowWalls[r].erase(col);
                colWalls[col].erase(r);
            }
        }
    }
    
    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += colWalls[i].size();
    }
    cout << ans << "\n";
    
    return 0;
}
