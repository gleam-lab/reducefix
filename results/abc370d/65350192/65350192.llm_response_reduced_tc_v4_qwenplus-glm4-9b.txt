#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // Using set to store the walls at each cell
    vector<vector<set<int>>> grid(h, vector<set<int>>(w));
    
    // Initial population of walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            grid[i][j].insert(i);
            grid[i][j].insert(j);
        }
    }
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;  // Convert to 0-based index
        
        // Check if the bomb is placed on a cell with no walls left
        if (grid[r][c].size() == 0) continue;
        
        // Remove the wall at (r, c)
        grid[r][c].erase(c);
        grid[r][c].erase(r);
        
        // Find the highest wall to the left of (r, c)
        auto leftIt = lower_bound(grid[r][c].begin(), grid[r][c].end(), c - 1);
        if (leftIt != grid[r][c].end()) {
            // Remove the wall above the highest wall to the left
            grid[r][c].erase(*leftIt);
            grid[*leftIt][c].erase(r);
        }
        
        // Find the highest wall to the right of (r, c)
        auto rightIt = upper_bound(grid[r][c].begin(), grid[r][c].end(), c);
        if (rightIt != grid[r][c].begin()) {
            --rightIt; // Get the last wall to the right
            // Remove the wall above the highest wall to the right
            grid[r][c].erase(*rightIt);
            grid[*rightIt][c].erase(r);
        }
        
        // Find the lowest wall above (r, c)
        leftIt = lower_bound(grid[r][c].begin(), grid[r][c].end(), r + 1);
        if (leftIt != grid[r][c].end()) {
            // Remove the wall to the left of the lowest wall above
            grid[r][c].erase(*leftIt);
            grid[r][c - 1].erase(*leftIt);
        }
        
        // Find the lowest wall below (r, c)
        rightIt = upper_bound(grid[r][c].begin(), grid[r][c].end(), r);
        if (rightIt != grid[r][c].begin()) {
            --rightIt; // Get the last wall below
            // Remove the wall to the left of the lowest wall below
            grid[r][c].erase(*rightIt);
            grid[r][c - 1].erase(*rightIt);
        }
    }
    
    // Count the remaining walls
    long long remainingWalls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            remainingWalls += grid[i][j].size();
        }
    }
    
    cout << remainingWalls << endl;
    
    return 0;
}
