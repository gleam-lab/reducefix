#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true)); // Using bool to track walls

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--; c--; // Convert to 0-based index

        if (!grid[r][c]) continue; // No wall to destroy

        // Mark the wall as destroyed
        grid[r][c] = false;

        // Check and destroy horizontal walls
        for (int j = 0; j < w; ++j) {
            if (j != c && grid[r][j]) {
                grid[r][j] = false;
            }
        }

        // Check and destroy vertical walls
        for (int i = 0; i < h; ++i) {
            if (i != r && grid[i][c]) {
                grid[i][c] = false;
            }
        }
    }

    // Count remaining walls
    int remainingWalls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (grid[i][j]) {
                remainingWalls++;
            }
        }
    }

    cout << remainingWalls << "\n";
    return 0;
}
