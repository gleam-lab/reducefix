#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
typedef long long ll;

struct UnionFind{
    vector<ll> par, siz;
    UnionFind(ll n):par(n,-1),siz(n,1){}
    void init(ll n){
        fill(par.begin(),par.end(),-1);
        fill(siz.begin(),siz.end(),1);
    }
    ll root(ll x){
        if(par[x] == -1)return x;
        return par[x] = root(par[x]);
    }
    bool same(ll x,ll y){return root(x) == root(y);}
    bool merge(ll x,ll y){
        x = root(x); y = root(y);
        if(x==y)return false;
        if(siz[x]<siz[y])swap(x,y);
        par[y] = x;
        siz[x] += siz[y];
        return true;
    }
    ll size(ll x){
        return siz[root(x)];
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    ll h,w,q;
    cin >> h >> w >> q;
    vector<set<pair<ll,ll>>> hor(w),ver(h);
    rep(i,h)rep(j,w){
        ver[i].emplace(j,i);
        hor[j].emplace(i,j);
    }
    UnionFind uf_h(w),uf_v(h);
    rep(qi,q){
        ll r,c;
        cin >> r >> c;
        r--,c--;
        if(!hor[c].empty()){
            if(*hor[c].begin() == r)hor[c].erase(hor[c].begin());
            else if(*hor[c].rbegin() == r)hor[c].erase(prev(hor[c].rend()));
            else{
                auto itr = hor[c].upper_bound({r,r});
                if(itr != hor[c].begin())itr--;
                uf_h.merge(*itr,*(next(itr)));
                hor[c].erase(itr);
                hor[c].erase(next(itr));
            }
        }
        if(!ver[r].empty()){
            if(*ver[r].begin() == c)ver[r].erase(ver[r].begin());
            else if(*ver[r].rbegin() == c)ver[r].erase(prev(ver[r].rend()));
            else{
                auto itr = ver[r].upper_bound({c,c});
                if(itr != ver[r].begin())itr--;
                uf_v.merge(*itr,*(next(itr)));
                ver[r].erase(itr);
                ver[r].erase(next(itr));
            }
        }
    }
    ll ans = 0;
    set<ll> st;
    rep(i,w)if(!hor[i].empty())st.insert(uf_h.root(i));
    ans += st.size();
    st.clear();
    rep(i,h)if(!ver[i].empty())st.insert(uf_v.root(i));
    ans += st.size();
    cout << ans << endl;
    return 0;
}
