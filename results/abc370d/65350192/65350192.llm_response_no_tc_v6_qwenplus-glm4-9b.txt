#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

using ll = long long;

int main(void) {
    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<int>> grid(H, vector<int>(W, -1));
    vector<vector<bool>> destroyed(H, vector<bool>(W, false));

    rep(q, Q) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // If the cell is already destroyed or no wall can be destroyed
        if (destroyed[r][c] || grid[r][c] == 0) continue;

        // Destroy the wall at (r, c)
        grid[r][c] = 0;
        destroyed[r][c] = true;

        // Destroy walls upwards
        for (int i = r - 1; i >= 0 && grid[i][c] == -1; i--) {
            grid[i][c] = 0;
            destroyed[i][c] = true;
        }

        // Destroy walls downwards
        for (int i = r + 1; i < H && grid[i][c] == -1; i++) {
            grid[i][c] = 0;
            destroyed[i][c] = true;
        }

        // Destroy walls to the left
        for (int j = c - 1; j >= 0 && grid[r][j] == -1; j--) {
            grid[r][j] = 0;
            destroyed[r][j] = true;
        }

        // Destroy walls to the right
        for (int j = c + 1; j < W && grid[r][j] == -1; j++) {
            grid[r][j] = 0;
            destroyed[r][j] = true;
        }
    }

    // Count the remaining walls
    int remainingWalls = 0;
    rep(i, H) rep(j, W) {
        if (!destroyed[i][j]) {
            remainingWalls++;
        }
    }

    cout << remainingWalls << "\n";

    return 0;
}
