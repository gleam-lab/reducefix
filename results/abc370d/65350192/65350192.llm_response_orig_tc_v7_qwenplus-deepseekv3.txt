#include <bits/stdc++.h>

int main() {

    vector<set<int>> row(h), col(w);
    rep(i,h) rep(j,w) {
        row[i].insert(j);
        col[j].insert(i);
    }
    
    int ans = h * w;
    rep(qi,q) {

        r--; c--;
        
        if (row[r].count(c)) {
            row[r].erase(c);
            col[c].erase(r);
            ans--;
        } else {
            // Up
            auto it = col[c].lower_bound(r);
            if (it != col[c].begin()) {
                --it;
                int tmp = *it;
                row[tmp].erase(c);
                col[c].erase(it);
                ans--;
            }
            
            // Down
            it = col[c].lower_bound(r);
            if (it != col[c].end()) {
                int tmp = *it;
                row[tmp].erase(c);
                col[c].erase(it);
                ans--;
            }
            
            // Left
            it = row[r].lower_bound(c);
            if (it != row[r].begin()) {
                --it;
                int tmp = *it;
                col[tmp].erase(r);
                row[r].erase(it);
                ans--;
            }
            
            // Right
            it = row[r].lower_bound(c);
            if (it != row[r].end()) {

            }
        }

}
