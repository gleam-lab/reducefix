#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> grid(h, vector<int>(w));
    vector<set<pair<int, int>>> rows(w), cols(h);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            grid[i][j] = 1;
            rows[j].insert({i, j});
            cols[i].insert({i, j});
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        --r; --c;
        if (rows[c].count({r, c})) {
            rows[c].erase({r, c});
            cols[r].erase({r, c});
        } else {
            set<pair<int, int>> row_to_remove, col_to_remove;
            for (auto &p : rows[c]) {
                if (p.first == r) {
                    row_to_remove.insert(p);
                } else if (p.first > r) {
                    col_to_remove.insert(p);
                }
            }
            for (auto &p : cols[r]) {
                if (p.second == c) {
                    col_to_remove.insert(p);
                } else if (p.second > c) {
                    row_to_remove.insert(p);
                }
            }
            for (auto &p : row_to_remove) {
                rows[c].erase(p);
                cols[p.first].erase(p);
            }
            for (auto &p : col_to_remove) {
                cols[r].erase(p);
                rows[p.first].erase(p);
            }
        }
    }

    ll remaining_walls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            remaining_walls += grid[i][j];
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
