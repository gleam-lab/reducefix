```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    
    // Initialize sets with all wall positions
    rep(i,h) rep(j,w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        // Check if there's a wall at the bomb position
        if (tate[c].count(r)) {
            // Bomb hits a wall - just remove it
            tate[c].erase(r);
            yoko[r].erase(c);
        } else {
            // Bomb hits empty cell - destroy first walls in each direction
            
            // Down direction
            auto it = tate[c].lower_bound(r);
            if (it != tate[c].end()) {
                int tmp = *it;
                tate[c].erase(tmp);
                yoko[tmp].erase(c);
            }
            
            // Up direction
            it = tate[c].lower_bound(r);
            if (it != tate[c].begin()) {
                --it;
                int tmp = *it;
                tate[c].erase(tmp);
                yoko[tmp].erase(c);
            }
            
            // Right direction
            auto jt = yoko[r].lower_bound(c);
            if (jt != yoko[r].end()) {
                int tmp = *jt;
                yoko[r].erase(tmp);
                tate[tmp].erase(r);
            }
            
            // Left direction
            jt = yoko[r].lower_bound(c);
            if (jt != yoko[r].begin()) {
                --jt;
                int tmp = *jt;
                yoko[r].erase(tmp);
                tate[tmp].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    rep(i,w) ans += tate[i].size();
    cout << ans << endl;
    
    return 0;
}
```
