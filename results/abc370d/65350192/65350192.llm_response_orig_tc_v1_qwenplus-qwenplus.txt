```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> row(w), col(h);
    
    // Initialize sets with all wall positions
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (col[r].count(c)) {
            // Wall exists, destroy it
            col[r].erase(c);
            row[c].erase(r);
        } else {
            // Wall doesn't exist, look in four directions
            
            // Up
            auto it = row[c].lower_bound(r);
            if (it != row[c].begin()) {
                --it;
                int u = *it;
                col[u].erase(c);
                row[c].erase(it);
            }
            
            // Down
            it = row[c].lower_bound(r);
            if (it != row[c].end()) {
                int d = *it;
                col[d].erase(c);
                row[c].erase(it);
            }
            
            // Left
            auto jt = col[r].lower_bound(c);
            if (jt != col[r].begin()) {
                --jt;
                int l = *jt;
                row[l].erase(r);
                col[r].erase(jt);
            }
            
            // Right
            jt = col[r].lower_bound(c);
            if (jt != col[r].end()) {
                int ri = *jt;
                row[ri].erase(r);
                col[r].erase(jt);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    
    cout << ans << endl;
    return 0;
}
```
