#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)

using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> walls(h, vector<int>(w));
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            walls[i][j] = 1; // Initialize all cells with one wall
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if (walls[r][c] == 0) continue; // If there's no wall, do nothing

        walls[r][c] = 0; // Destroy the wall at (r, c)
        // Check and destroy the top wall if it exists
        if (r > 0 && walls[r - 1][c] == 1) {
            walls[r - 1][c] = 0;
        }
        // Check and destroy the bottom wall if it exists
        if (r < h - 1 && walls[r + 1][c] == 1) {
            walls[r + 1][c] = 0;
        }
        // Check and destroy the left wall if it exists
        if (c > 0 && walls[r][c - 1] == 1) {
            walls[r][c - 1] = 0;
        }
        // Check and destroy the right wall if it exists
        if (c < w - 1 && walls[r][c + 1] == 1) {
            walls[r][c + 1] = 0;
        }
    }

    int remaining_walls = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            remaining_walls += walls[i][j];
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
