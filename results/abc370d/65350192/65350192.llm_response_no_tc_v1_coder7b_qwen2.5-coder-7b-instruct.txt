```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (grid[r][c]) {
            grid[r][c] = false;
        } else {
            for (int i = r - 1; i >= 0; --i) {
                if (grid[i][c]) {
                    grid[i][c] = false;
                    break;
                }
            }
            for (int i = r + 1; i < h; ++i) {
                if (grid[i][c]) {
                    grid[i][c] = false;
                    break;
                }
            }
            for (int j = c - 1; j >= 0; --j) {
                if (grid[r][j]) {
                    grid[r][j] = false;
                    break;
                }
            }
            for (int j = c + 1; j < w; ++j) {
                if (grid[r][j]) {
                    grid[r][j] = false;
                    break;
                }
            }
        }
    }

    int ans = 0;
    for (const auto& row : grid) {
        ans += count(row.begin(), row.end(), true);
    }
    cout << ans << "\n";
    return 0;
}
```

This solution uses a 2D boolean array `grid` to represent the state of each cell in the grid. Initially, all cells are marked as having walls (`true`). For each query, it either removes a wall or destroys the nearest wall in each direction. Finally, it counts the number of remaining walls and prints the result. This approach ensures both correctness and efficiency within the given constraints.
