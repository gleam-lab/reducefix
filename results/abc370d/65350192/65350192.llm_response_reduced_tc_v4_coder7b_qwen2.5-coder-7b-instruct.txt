#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    while(q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        if(grid[r][c]) {
            grid[r][c] = false;
            for(int i = 0; i < h; ++i) if(grid[i][c]) break;
            else grid[i][c] = false;
            for(int i = h-1; i >= 0; --i) if(grid[i][c]) break;
            else grid[i][c] = false;
            for(int j = 0; j < w; ++j) if(grid[r][j]) break;
            else grid[r][j] = false;
            for(int j = w-1; j >= 0; --j) if(grid[r][j]) break;
            else grid[r][j] = false;
        }
    }
    int ans = 0;
    for(auto &row: grid) for(bool b: row) ans += !b;
    cout << ans << '\n';
    return 0;
}
