#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<pair<int, int>>> grid(h, set<pair<int, int>>, greater<pair<int, int>>());
    rep(i, h) rep(j, w) {
        grid[i].insert({i, j});
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        auto& row = grid[r];
        auto& col = grid[c];

        // Check if there's a wall at the exact position
        if (row.count({r, c})) {
            row.erase({r, c});
        } else {
            // If no wall at exact position, remove walls from the row and column
            // Remove walls above
            for (auto it = row.lower_bound({r, c}); it != row.end(); ++it) {
                col.erase({it->first, c});
                row.erase(it);
            }
            // Remove walls below
            for (auto it = row.lower_bound({r, c}); it != row.end(); ++it) {
                col.erase({it->first, c});
                row.erase(it);
            }
            // Remove walls to the left
            for (auto it = col.lower_bound({r, c}); it != col.end(); ++it) {
                row.erase({r, it->second});
                col.erase(it);
            }
            // Remove walls to the right
            for (auto it = col.lower_bound({r, c}); it != col.end(); ++it) {
                row.erase({r, it->second});
                col.erase(it);
            }
        }
    }

    int remaining_walls = 0;
    for (auto& row : grid) {
        for (auto& p : row) {
            remaining_walls++;
        }
    }
    cout << remaining_walls << '\n';
    return 0;
}
