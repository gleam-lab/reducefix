```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    
    // Initialize sets with all positions
    rep(i, h) rep(j, w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (tate[c].find(r) != tate[c].end()) {
            // Wall exists at the position - remove it
            tate[c].erase(r);
            yoko[r].erase(c);
        } else {
            // No wall here - destroy first walls in each direction

            // Down
            auto it = tate[c].lower_bound(r);
            if (it != tate[c].end()) {
                int tmp = *it;
                tate[c].erase(tmp);
                yoko[tmp].erase(c);
            }

            // Up
            it = tate[c].lower_bound(r);
            if (it != tate[c].begin()) {
                --it;
                int tmp = *it;
                tate[c].erase(tmp);
                yoko[tmp].erase(c);
            }

            // Right
            auto it2 = yoko[r].lower_bound(c);
            if (it2 != yoko[r].end()) {
                int tmp = *it2;
                yoko[r].erase(tmp);
                tate[tmp].erase(r);
            }

            // Left
            it2 = yoko[r].lower_bound(c);
            if (it2 != yoko[r].begin()) {
                --it2;
                int tmp = *it2;
                yoko[r].erase(tmp);
                tate[tmp].erase(r);
            }
        }
    }

    int ans = 0;
    rep(i, w) ans += tate[i].size();
    cout << ans << "\n";
    
    return 0;
}
```
