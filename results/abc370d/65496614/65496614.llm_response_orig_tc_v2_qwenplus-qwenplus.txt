#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i=0;i<(int)(n);++i)
#define rep1(i,n) for(int i=1;i<=(int)(n);++i)

using ll = long long;
using pii = pair<int, int>;

const int INF = 1 << 30;

int H, W, Q;

const int MAXN = 4e5 + 5;

set<int> rows[MAXN], cols[MAXN];

bool exists(int r, int c) {
    return rows[r].count(c);
}

void remove(int r, int c) {
    if (rows[r].count(c)) {
        rows[r].erase(c);
        cols[c].erase(r);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> H >> W >> Q;

    // Initialize rows and cols
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        if (exists(r, c)) {
            // Directly destroy the bomb
            remove(r, c);
        } else {
            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                --it;
                remove(*it, c);
            }

            // Down
            it = cols[c].upper_bound(r);
            if (it != cols[c].end()) {
                remove(*it, c);
            }

            // Left
            auto& row_set = rows[r];
            it = row_set.lower_bound(c);
            if (it != row_set.begin()) {
                --it;
                remove(r, *it);
            }

            // Right
            it = row_set.upper_bound(c);
            if (it != row_set.end()) {
                remove(r, *it);
            }
        }
    }

    int ans = 0;
    for (int i = 1; i <= H; ++i) {
        ans += rows[i].size();
    }

    cout << ans << "\n";

    return 0;
}
