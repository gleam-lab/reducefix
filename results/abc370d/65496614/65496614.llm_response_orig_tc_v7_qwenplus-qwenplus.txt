#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rrep(i, n) for(int i = (int)(n)-1; i >= 0; --i)

using ll = long long;
using vi = vector<int>;
using vvi = vector<vector<int>>;
using vb = vector<bool>;

const int dx[4] = {-1, 1, 0, 0};
const int dy[4] = {0, 0, -1, 1};

int H, W, Q;

// Data structures to store the available cells in each row and column using ordered sets
vector<set<int>> row_sets, col_sets;

// Grid to check if a wall is present at a cell
vb grid;

// Helper function to convert 2D coordinates to linear index
int idx(int r, int c) {
    return (r - 1) * W + (c - 1);
}

void destroy(int r, int c) {
    int id = idx(r, c);
    if (!grid[id]) return;

    grid[id] = false;

    // Remove from row and column sets
    row_sets[r].erase(c);
    col_sets[c].erase(r);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> H >> W >> Q;

    int total_cells = H * W;
    grid.assign(total_cells + 5, true); // 1-based indexing

    // Initialize row-wise and column-wise sets
    row_sets.resize(H + 2);
    col_sets.resize(W + 2);

    for (int i = 1; i <= H; ++i) {
        row_sets[i].clear();
        for (int j = 1; j <= W; ++j)
            row_sets[i].insert(j);
    }

    for (int j = 1; j <= W; ++j) {
        col_sets[j].clear();
        for (int i = 1; i <= H; ++i)
            col_sets[j].insert(i);
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        int id = idx(r, c);

        if (grid[id]) {
            // If there's a wall, destroy it
            destroy(r, c);
        } else {
            // Else, find the nearest walls in each direction
            for (int d = 0; d < 4; ++d) {
                int x = r, y = c;

                if (d == 0) { // Up
                    auto it = col_sets[c].lower_bound(r);
                    if (it != col_sets[c].begin()) {
                        --it;
                        destroy(*it, c);
                    }
                } else if (d == 1) { // Down
                    auto it = col_sets[c].upper_bound(r);
                    if (it != col_sets[c].end()) {
                        destroy(*it, c);
                    }
                } else if (d == 2) { // Left
                    auto it = row_sets[r].lower_bound(c);
                    if (it != row_sets[r].begin()) {
                        --it;
                        destroy(r, *it);
                    }
                } else { // Right
                    auto it = row_sets[r].upper_bound(c);
                    if (it != row_sets[r].end()) {
                        destroy(r, *it);
                    }
                }
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < total_cells; ++i)
        ans += grid[i];

    cout << ans << "\n";
    return 0;
}
