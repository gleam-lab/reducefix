#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i, n) for(int i = 1; i <= (int)(n); ++i)

using ll = long long;
using pii = pair<int, int>;

const int dx[4] = {-1, 1, 0, 0};
const int dy[4] = {0, 0, -1, 1};

int H, W, Q;

// For each row and column, keep track of wall positions using ordered sets
vector<set<int>> rows, cols;

// Function to check if a cell is valid
bool in_bounds(int x, int y) {
    return 1 <= x && x <= H && 1 <= y && y <= W;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> H >> W >> Q;

    // Initialize: each cell has a wall
    // We will use sets to maintain walls per row and column
    rows.resize(H + 2); // rows[0] unused
    cols.resize(W + 2); // cols[0] unused

    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // If there's a wall at this position
        if (rows[r].count(c)) {
            // Destroy the wall at (r, c)
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Otherwise, destroy first walls in four directions
            int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
            
            for (auto [dr, dc] : dirs) {
                int nr = r, nc = c;
                while (in_bounds(nr + dr, nc + dc)) {
                    nr += dr;
                    nc += dc;
                    if (rows[nr].count(nc)) {
                        rows[nr].erase(nc);
                        cols[nc].erase(nr);
                        break;
                    }
                }
            }
        }
    }

    // Count remaining walls
    long long ans = 0;
    for (int i = 1; i <= H; ++i) {
        ans += rows[i].size();
    }

    cout << ans << endl;
    return 0;
}
