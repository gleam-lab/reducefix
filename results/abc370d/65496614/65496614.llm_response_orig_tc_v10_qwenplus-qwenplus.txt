#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i = 0; i < (int)(n) ; ++i)
#define rep1(i,n) for(int i = 1; i < (int)(n) ; ++i)

using ll = long long;
using vi = vector<int>;
using vvi = vector<vector<int>>;
using vb = vector<bool>;

const int dx[4] = {-1, 1, 0, 0};
const int dy[4] = {0, 0, -1, 1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int H, W, Q;
    cin >> H >> W >> Q;

    // Initialize grid and sets
    vector<set<int>> rows(H + 2), cols(W + 2);
    vb row_has_bomb(H + 2, false), col_has_bomb(W + 2, false);
    vector<vector<bool>> has_wall(H + 2, vector<bool>(W + 2, true));

    auto erase = [&](int r, int c) {
        if (!has_wall[r][c]) return;
        has_wall[r][c] = false;
        rows[r].erase(c);
        cols[c].erase(r);
    };

    // Initialize rows and cols with all columns/rows
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        if (has_wall[r][c]) {
            // Bomb placed on a wall: destroy it and stop
            erase(r, c);
        } else {
            // Bomb placed on empty cell: destroy first walls in four directions
            for (int d = 0; d < 4; ++d) {
                int nr = r, nc = c;
                while (true) {
                    nr += dx[d];
                    nc += dy[d];
                    if (nr < 1 || nr > H || nc < 1 || nc > W) break;
                    if (has_wall[nr][nc]) {
                        erase(nr, nc);
                        break;
                    }
                }
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            ans += has_wall[i][j];
        }
    }

    cout << ans << endl;
    return 0;
}
