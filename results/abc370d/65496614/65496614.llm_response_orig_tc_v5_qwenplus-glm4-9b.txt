### 盲禄拢莽聽聛氓庐聻莽聨掳
盲禄楼盲赂聥忙聵炉盲赂聙盲赂陋莽庐聙氓聦聳莽職聞莽陇潞盲戮聥茂录聦氓聫陋猫聙聝猫聶聭忙掳麓氓鹿鲁氓聮聦氓聻聜莽聸麓忙聳鹿氓聬聭莽職聞氓陇聞莽聬聠茂录職

```cpp
#include <iostream>
#include <vector>

using namespace std;

int H, W, Q;
vector<int> parent;

int find(int x) {
    if (parent[x] != x) parent[x] = find(parent[x]);
    return parent[x];
}

void unionSets(int x, int y) {
    parent[find(x)] = find(y);
}

int main() {
    cin >> H >> W >> Q;
    parent.resize(W + 1);
    for (int i = 1; i <= W; ++i) parent[i] = i;

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (find(r) == find(c)) {
            // 忙聴垄茫聛芦茅聙拢莽碌聬茫聛聲茫聜聦茫聛娄茫聛聞茫聜聥
            continue;
        } else {
            // 茅聙拢莽碌聬茫聛聶茫聜聥
            unionSets(r, c);
            // 氓聭篓氓聸虏茫聛庐氓拢聛茫聜聮茅聶陇氓聨禄
            for (int i = 1; i <= H && find(i) == find(c); ++i) {
                unionSets(i, r);
            }
            for (int j = 1; j <= W && find(j) == find(r); ++j) {
                unionSets(j, c);
            }
        }
    }

    // 氓聣漏盲陆聶莽職聞氓垄聶莽職聞忙聲掳茅聡聫
    int remainingWalls = 0;
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            if (find(i) != find(j)) {
                ++remainingWalls;
            }
        }
    }

    cout << remainingWalls << endl;

    return 0;
}
```
