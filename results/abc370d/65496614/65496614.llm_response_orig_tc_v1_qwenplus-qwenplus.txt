#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rrep(i, n) for(int i = (int)(n)-1; i >= 0; --i)

using ll = long long;
using pii = pair<int, int>;

const int INF = 1 << 30;

int H, W, Q;

struct UnionFind {
    vector<int> par;
    UnionFind(int n) : par(n) {
        for (int i = 0; i < n; ++i) par[i] = i;
    }
    int find(int x) {
        return x == par[x] ? x : par[x] = find(par[x]);
    }
    void unite(int x, int y) {
        par[find(y)] = find(x);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> H >> W >> Q;

    int total = H * W;
    vector<vector<int>> grid(H, vector<int>(W, 1)); // 1: wall exists

    // For each row, keep track of the next available cell to destroy in left/right directions
    vector<set<int>> rows(H);
    // For each column, keep track of the next available cell to destroy in up/down directions
    vector<set<int>> cols(W);

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    auto remove = [&](int r, int c) {
        if (!grid[r][c]) return false;
        grid[r][c] = 0;
        rows[r].erase(c);
        cols[c].erase(r);
        return true;
    };

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (grid[r][c]) {
            remove(r, c);
        } else {
            // Look up
            auto& col_up = cols[c];
            auto it = col_up.lower_bound(r);
            if (it != col_up.begin()) {
                it--;
                remove(*it, c);
            }

            // Look down
            it = col_up.lower_bound(r);
            if (it != col_up.end()) {
                remove(*it, c);
            }

            // Look left
            auto& row_left = rows[r];
            auto jt = row_left.lower_bound(c);
            if (jt != row_left.begin()) {
                jt--;
                remove(r, *jt);
            }

            // Look right
            jt = row_left.lower_bound(c);
            if (jt != row_left.end()) {
                remove(r, *jt);
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            ans += grid[i][j];
        }
    }

    cout << ans << "\n";

    return 0;
}
