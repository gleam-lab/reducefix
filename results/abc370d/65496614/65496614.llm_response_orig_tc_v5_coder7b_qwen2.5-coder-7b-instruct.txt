void update(vector<vector<int>>& v, int x, int y) {
    if (v[x][y]) return;
    v[x][y] = true;
    for (int i = x + 1; i < MAXN && !v[i][y]; ++i) v[i][y] = true;
    for (int i = x - 1; i >= 0 && !v[i][y]; --i) v[i][y] = true;

    for (int j = y - 1; j >= 0 && !v[x][j]; --j) v[x][j] = true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, q;
    cin >> h >> w >> q;

    rep(i, h) rep(j, w) grid[i][j] = true;

    int r, c;
    while (q--) {

        --r, --c; // Convert to zero-based index
        if (!grid[r][c]) {
