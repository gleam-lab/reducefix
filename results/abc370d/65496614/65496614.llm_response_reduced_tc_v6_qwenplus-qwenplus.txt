#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i, n) for(int i = 1; i <= (int)(n); ++i)

using vi = vector<int>;
using vvi = vector<vi>;
using vb = vector<bool>;
using vvb = vector<vb>;
using pii = pair<int, int>;
using vpii = vector<pii>;

template<typename T>
using vec = vector<T>;

template<typename T>
using vvec = vector<vector<T>>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    const int MAX = max(H, W) + 2;

    // Store whether a wall exists at (r, c)
    vvb hasWall(H + 2, vb(W + 2, true));

    // For each row, keep track of the nearest walls to the left and right
    vector<set<int>> rowLeft(H + 2), rowRight(H + 2);
    // For each column, keep track of the nearest walls above and below
    vector<set<int>> colAbove(H + 2), colBelow(H + 2);

    // Initialize sets
    for (int r = 1; r <= H; ++r) {
        for (int c = 1; c <= W; ++c) {
            rowLeft[r].insert(c);
            rowRight[r].insert(c);
            colAbove[c].insert(r);
            colBelow[c].insert(r);
        }
    }

    auto removeWall = [&](int r, int c) {
        if (!hasWall[r][c]) return;

        hasWall[r][c] = false;

        // Remove from row sets
        rowLeft[r].erase(c);
        rowRight[r].erase(c);

        // Remove from column sets
        colAbove[c].erase(r);
        colBelow[c].erase(r);
    };

    auto processBomb = [&](int r, int c) {
        if (hasWall[r][c]) {
            removeWall(r, c);
            return;
        }

        // Up, Down, Left, Right directions
        // Up
        if (!colAbove[c].empty()) {
            auto it = colAbove[c].lower_bound(r);
            if (it != colAbove[c].begin()) {
                --it;
                int ur = *it;
                removeWall(ur, c);
            }
        }

        // Down
        if (!colBelow[c].empty()) {
            auto it = colBelow[c].lower_bound(r);
            if (it != colBelow[c].end()) {
                int dr = *it;
                removeWall(dr, c);
            }
        }

        // Left
        if (!rowLeft[r].empty()) {
            auto it = rowLeft[r].lower_bound(c);
            if (it != rowLeft[r].begin()) {
                --it;
                int lc = *it;
                removeWall(r, lc);
            }
        }

        // Right
        if (!rowRight[r].empty()) {
            auto it = rowRight[r].lower_bound(c);
            if (it != rowRight[r].end()) {
                int rc = *it;
                removeWall(r, rc);
            }
        }
    };

    while (Q--) {
        int r, c;
        cin >> r >> c;
        processBomb(r, c);
    }

    int remaining = 0;
    for (int r = 1; r <= H; ++r) {
        for (int c = 1; c <= W; ++c) {
            remaining += hasWall[r][c];
        }
    }

    cout << remaining << "\n";

    return 0;
}
