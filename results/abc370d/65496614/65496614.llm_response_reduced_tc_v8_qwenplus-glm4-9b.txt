#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    vector<vector<set<int>>> row(h + 2, vector<set<int>>(w + 2));
    vector<vector<set<int>>> col(w + 2, vector<set<int>>(h + 2));

    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            row[i][j] = {i};
            col[j][i] = {j};
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;

        auto row_set = row[r][c];
        auto col_set = col[c][r];

        auto erase_from_row = [&](int k) {
            auto it = row_set.lower_bound(k);
            if (it != row_set.end()) {
                row[r][c].erase(it);
                col[k][r].erase(col_set.lower_bound(r));
            }
        };

        auto erase_from_col = [&](int k) {
            auto it = col_set.lower_bound(k);
            if (it != col_set.end()) {
                col[c][r].erase(it);
                row[k][c].erase(row_set.lower_bound(r));
            }
        };

        // Erase walls in the specified row and column
        erase_from_row(r);
        erase_from_col(c);

        // Clear the set as it's no longer needed
        row_set.clear();
        col_set.clear();
    }

    // Count the remaining walls
    int remaining_walls = 0;
    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            remaining_walls += row[i][j].size() + col[j][i].size();
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
