#include <iostream>
#include <vector>
#include <set>

using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); ++i)
#define all(x) (x).begin(), (x).end()

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    vector<vector<set<int>>> col(h + 1, vector<set<int>>(w + 1));
    vector<vector<set<int>>> row(h + 1, vector<set<int>>(w + 1));

    // Initialize columns and rows
    rep(i, h + 1) {
        rep(j, w + 1) {
            col[i][j] = set<int>{j + 1, j + 2};
            row[i][j] = set<int>{i + 1, i + 2};
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        --r;
        --c;

        // Check if there is a wall at (r, c)
        if (col[r][c].size() > 1) {
            // Remove a wall at (r, c)
            col[r][c].erase(col[r][c].begin());
            row[r][c].erase(row[r][c].begin());
            continue;
        }

        // Find the first wall to the left and right
        auto left = col[r][c].lower_bound(c + 1);
        auto right = prev(left);
        auto up = row[r][c].lower_bound(r + 1);
        auto down = prev(up);

        // Remove walls directly above and below
        row[r][c].erase(up);
        row[r][c].erase(down);

        // Remove walls to the left and right
        if (left != col[r][c].end()) {
            col[r][c].erase(left);
        }
        if (right != col[r][c].begin()) {
            col[r][c].erase(right);
        }

        // Update columns and rows
        if (left != col[r][c].end()) {
            col[*left][c].insert(r + 1);
        }
        if (right != col[r][c].begin()) {
            col[*right][c].insert(r + 1);
        }
        if (up != row[r][c].end()) {
            row[r][c].insert(*up);
        }
        if (down != row[r][c].begin()) {
            row[r][c].insert(*down);
        }
    }

    // Count remaining walls
    int remaining_walls = 0;
    rep(i, h + 1) {
        rep(j, w + 1) {
            remaining_walls += col[i][j].size() - 1;
            remaining_walls += row[i][j].size() - 1;
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
