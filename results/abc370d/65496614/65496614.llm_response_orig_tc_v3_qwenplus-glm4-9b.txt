#include <vector>
#include <set>

using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    vector<vector<bool>> grid(h + 2, vector<bool>(w + 2, true));

    while (q--) {
        int r, c;
        cin >> r >> c;
        if (grid[r][c]) {
            grid[r][c] = false; // 茅聰聙忙炉聛氓垄聶氓拢聛
        } else {
            // 茅聰聙忙炉聛盲赂聤忙聳鹿茫聙聛盲赂聥忙聳鹿茫聙聛氓路娄忙聳鹿茫聙聛氓聫鲁忙聳鹿莽職聞氓垄聶氓拢聛
            for (int i = 1; i < r; ++i) {
                if (grid[i][c]) {
                    grid[i][c] = false;
                }
            }
            for (int i = r + 1; i <= h; ++i) {
                if (grid[i][c]) {
                    grid[i][c] = false;
                }
            }
            for (int j = 1; j < c; ++j) {
                if (grid[r][j]) {
                    grid[r][j] = false;
                }
            }
            for (int j = c + 1; j <= w; ++j) {
                if (grid[r][j]) {
                    grid[r][j] = false;
                }
            }
        }
    }

    int remainingWalls = 0;
    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            if (grid[i][j]) {
                ++remainingWalls;
            }
        }
    }

    cout << remainingWalls << endl;
    return 0;
}
