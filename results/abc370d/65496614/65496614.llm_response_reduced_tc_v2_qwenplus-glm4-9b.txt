
```cpp
#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); ++i)
#define rrep(i, n) for (int i = (int)(n) - 1; i >= 0; --i)

using ll = long long;

const int MAX_H = 4e5 + 5;
const int MAX_W = 4e5 + 5;

vector<vector<set<int>>> row_walls(MAX_H, vector<set<int>>(MAX_W));
vector<vector<set<int>>> col_walls(MAX_H, vector<set<int>>(MAX_W));

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    rep(i, MAX_H) {
        rep(j, MAX_W) {
            row_walls[i][j].insert(j);
            col_walls[i][j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--; // Convert to 0-based index

        if (row_walls[r][c].size() == 1 && col_walls[r][c].size() == 1) {
            // Destroy the first wall in row and column
            int first_row_wall = *row_walls[r][c].begin();
            int first_col_wall = *col_walls[r][c].begin();
            row_walls[r][c].erase(first_row_wall);
            col_walls[r][c].erase(first_col_wall);
        }

        // Remove the wall from the current cell
        row_walls[r][c].erase(c);
        col_walls[r][c].erase(r);
    }

    int remaining_walls = 0;
    rep(i, h) rep(j, w) {
        if (row_walls[i][j].size() == 0 || col_walls[i][j].size() == 0) {
            remaining_walls++;
        }
    }

    cout << remaining_walls << endl;

    return 0;
}
```

This code initializes the wall structures for rows and columns. For each query, it checks the wall existence at the given cell and removes the wall if it exists. If the cell is empty, it removes the first available wall from the row and column. After processing all queries, it counts the remaining walls by checking the existence of walls in each cell. The result is then printed.
