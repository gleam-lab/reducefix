#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i,n) for(int i = 1; i <= (int)(n); ++i)

using vi = vector<int>;
using vvi = vector<vi>;
using vb = vector<bool>;
using vvb = vector<vb>;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // We use H+2 and W+2 to handle borders easily
    vvb has_wall(H + 2, vb(W + 2, true));

    // For each row, keep track of intact columns
    vector<set<int>> row(H + 2);
    // For each column, keep track of intact rows
    vector<set<int>> col(W + 2);

    // Initialize sets with all columns/rows
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row[i].insert(j);
            col[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        if (!has_wall[r][c]) {
            // Bomb at (r,c) was already placed
            // Try to destroy the nearest walls in four directions

            // Up
            auto it = col[c].lower_bound(r);
            if (it != col[c].begin()) {
                --it;
                int x = *it;
                has_wall[x][c] = false;
                row[x].erase(c);
                col[c].erase(it);
            }

            // Down
            it = col[c].lower_bound(r);
            if (it != col[c].end()) {
                int x = *it;
                has_wall[x][c] = false;
                row[x].erase(c);
                col[c].erase(it);
            }

            // Left
            auto it2 = row[r].lower_bound(c);
            if (it2 != row[r].begin()) {
                --it2;
                int y = *it2;
                has_wall[r][y] = false;
                col[y].erase(r);
                row[r].erase(it2);
            }

            // Right
            it2 = row[r].lower_bound(c);
            if (it2 != row[r].end()) {
                int y = *it2;
                has_wall[r][y] = false;
                col[y].erase(r);
                row[r].erase(it2);
            }
        } else {
            // Destroy wall at (r,c)
            has_wall[r][c] = false;
            row[r].erase(c);
            col[c].erase(r);
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            if (has_wall[i][j]) ans++;
        }
    }

    cout << ans << endl;
    return 0;
}
