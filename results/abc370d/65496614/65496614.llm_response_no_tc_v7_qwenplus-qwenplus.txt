#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i, n) for(int i = 1; i <= (int)(n); ++i)

using ll = long long;
using pii = pair<int, int>;

const int dx[] = {-1, 1, 0, 0};
const int dy[] = {0, 0, -1, 1};

int H, W, Q;
bool wall[400010]; // flattened grid

// Convert 2D coordinates to 1D index
int idx(int r, int c) {
    return (r - 1) * W + (c - 1);
}

// Get the first wall in direction (dir) from (r,c)
int get_next(int r, int c, int dir) {
    int nr = r, nc = c;
    while (true) {
        nr += dx[dir];
        nc += dy[dir];
        if (nr < 1 || nr > H || nc < 1 || nc > W) return -1;
        int pos = idx(nr, nc);
        if (wall[pos]) return pos;
    }
    return -1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> H >> W >> Q;
    fill(wall, wall + H * W, true);
    
    vector<int> active(H * W, 1); // 1 if wall is present

    auto destroy = [&](int r, int c) {
        int pos = idx(r, c);
        if (!active[pos]) return;
        active[pos] = 0;
    };

    while (Q--) {
        int r, c;
        cin >> r >> c;
        int pos = idx(r, c);

        if (active[pos]) {
            // There's a wall at (r,c), destroy it
            destroy(r, c);
        } else {
            // Look in all four directions and destroy the first wall found in each
            vector<int> to_destroy;

            // Up
            int nr = r - 1;
            while (nr >= 1) {
                int pos2 = idx(nr, c);
                if (active[pos2]) {
                    to_destroy.push_back(pos2);
                    break;
                }
                nr--;
            }

            // Down
            nr = r + 1;
            while (nr <= H) {
                int pos2 = idx(nr, c);
                if (active[pos2]) {
                    to_destroy.push_back(pos2);
                    break;
                }
                nr++;
            }

            // Left
            int nc = c - 1;
            while (nc >= 1) {
                int pos2 = idx(r, nc);
                if (active[pos2]) {
                    to_destroy.push_back(pos2);
                    break;
                }
                nc--;
            }

            // Right
            nc = c + 1;
            while (nc <= W) {
                int pos2 = idx(r, nc);
                if (active[pos2]) {
                    to_destroy.push_back(pos2);
                    break;
                }
                nc++;
            }

            // Destroy the walls
            for (int p : to_destroy) {
                active[p] = 0;
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    rep(i, H * W) ans += active[i];

    cout << ans << endl;
    return 0;
}
