```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i,n) for(int i = 1; i <= (int)(n); ++i)

using vi = vector<int>;
using vvi = vector<vi>;
using vb = vector<bool>;
using vvb = vector<vb>;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // We'll use 1-based indexing for rows and columns
    vvb has_wall(H + 2, vb(W + 2, true));  // Add padding for easier boundary handling

    // For each row, we maintain a set of columns that have walls
    vector<set<int>> row_walls(H + 2);
    // For each column, we maintain a set of rows that have walls
    vector<set<int>> col_walls(W + 2);

    // Initialize the sets with all positions
    rep1(i, H) rep1(j, W) {
        row_walls[i].insert(j);
        col_walls[j].insert(i);
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;

        if (has_wall[R][C]) {
            // Case 1: There's a wall at (R, C), destroy it
            has_wall[R][C] = false;
            
            row_walls[R].erase(C);
            col_walls[C].erase(R);
        } else {
            // Case 2: No wall at (R, C), destroy first walls in each direction
            
            // Left
            if (!row_walls[R].empty() && *row_walls[R].begin() < C) {
                auto it = row_walls[R].lower_bound(C);
                if (it != row_walls[R].begin()) {
                    --it;
                    int c = *it;
                    has_wall[R][c] = false;
                    row_walls[R].erase(it);
                    col_walls[c].erase(R);
                }
            }

            // Right
            if (!row_walls[R].empty() && *row_walls[R].rbegin() > C) {
                auto it = row_walls[R].upper_bound(C);
                if (it != row_walls[R].end()) {
                    int c = *it;
                    has_wall[R][c] = false;
                    row_walls[R].erase(it);
                    col_walls[c].erase(R);
                }
            }

            // Up
            if (!col_walls[C].empty() && *col_walls[C].begin() < R) {
                auto it = col_walls[C].lower_bound(R);
                if (it != col_walls[C].begin()) {
                    --it;
                    int r = *it;
                    has_wall[r][C] = false;
                    col_walls[C].erase(it);
                    row_walls[r].erase(C);
                }
            }

            // Down
            if (!col_walls[C].empty() && *col_walls[C].rbegin() > R) {
                auto it = col_walls[C].upper_bound(R);
                if (it != col_walls[C].end()) {
                    int r = *it;
                    has_wall[r][C] = false;
                    col_walls[C].erase(it);
                    row_walls[r].erase(C);
                }
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    rep1(i, H) rep1(j, W) {
        if (has_wall[i][j]) ans++;
    }

    cout << ans << endl;
    return 0;
}
```
