#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (int)(n); ++i)
#define rep1(i, n) for(int i = 1; i <= (int)(n); ++i)

using ll = long long;
using pii = pair<int, int>;

const int dx[] = { -1, 1, 0, 0 };
const int dy[] = { 0, 0, -1, 1 };

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Grid to track presence of wall
    vector<vector<bool>> has_wall(H + 2, vector<bool>(W + 2, true));

    // For each row, keep a set of columns with walls
    vector<set<int>> row(W + 2);
    // For each column, keep a set of rows with walls
    vector<set<int>> col(H + 2);

    // Initialize sets
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row[i].insert(j);
            col[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        if (has_wall[r][c]) {
            // Bomb placed on a wall -> destroy it
            has_wall[r][c] = false;
            row[r].erase(c);
            col[c].erase(r);
        } else {
            // Bomb placed on empty cell -> find nearest walls in 4 directions

            // Up and Down
            auto it = col[c].lower_bound(r);
            if (it != col[c].end()) { // Down
                int y = *it;
                if (has_wall[y][c]) {
                    has_wall[y][c] = false;
                    row[y].erase(c);
                    col[c].erase(y);
                }
            }
            if (it != col[c].begin()) { // Up
                --it;
                int y = *it;
                if (has_wall[y][c]) {
                    has_wall[y][c] = false;
                    row[y].erase(c);
                    col[c].erase(y);
                }
            }

            // Left and Right
            it = row[r].lower_bound(c);
            if (it != row[r].end()) { // Right
                int x = *it;
                if (has_wall[r][x]) {
                    has_wall[r][x] = false;
                    row[r].erase(x);
                    col[x].erase(r);
                }
            }
            if (it != row[r].begin()) { // Left
                --it;
                int x = *it;
                if (has_wall[r][x]) {
                    has_wall[r][x] = false;
                    row[r].erase(x);
                    col[x].erase(r);
                }
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    rep1(i, H) rep1(j, W) ans += has_wall[i][j];
    cout << ans << endl;

    return 0;
}
