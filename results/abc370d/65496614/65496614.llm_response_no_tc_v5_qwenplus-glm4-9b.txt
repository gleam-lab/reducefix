#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    vector<vector<bool>> grid(h + 2, vector<bool>(w + 2, true));

    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            grid[i][j] = false;
        }
    }

    int r, c;
    vector<pair<int, int>> queries(q);

    for (int i = 0; i < q; ++i) {
        cin >> r >> c;
        queries[i] = {r, c};
    }

    // Sort queries by column and then by row
    sort(queries.begin(), queries.end());

    vector<int> column챈혰혰(h + 2, 1); // 챈혰혰찾혗혪챌짬혢찾혖짝찾혗혢책혞혰찾혖짬책짙혖찾혖혣책짯혱책혵짢찾혖혲찾혗혢찾혖혢찾혖짤찾혖혛찾혖혢
    vector<int> row챈혰혰(w + 2, 1);  // 챈혰혰찾혗혪챌짬혢찾혖짝찾혗혢챔징혣찾혖짬책짙혖찾혖혣책짯혱책혵짢찾혖혲찾혗혢찾혖혢찾혖짤찾혖혛찾혖혢

    // Process queries in order
    for (auto& p : queries) {
        r = p.first;
        c = p.second;

        // If the query is at a cell with a wall, destroy the wall
        if (grid[r][c]) {
            grid[r][c] = false;
            // Remove the column flag if this is the first wall in this column
            if (row챈혰혰[c] == 1) {
                column챈혰혰[r] = 0;
            }
            // Remove the row flag if this is the first wall in this row
            if (column챈혰혰[r] == 1) {
                row챈혰혰[c] = 0;
            }
            continue;
        }

        // If the query is at a cell without a wall, destroy walls as described
        // Check upwards
        for (int i = r - 1; i > 0 && !grid[i][c]; --i) {
            grid[i][c] = false;
            if (column챈혰혰[i] == 1) {
                row챈혰혰[c] = 0;
            }
        }
        // Check downwards
        for (int i = r + 1; i <= h && !grid[i][c]; ++i) {
            grid[i][c] = false;
            if (column챈혰혰[i] == 1) {
                row챈혰혰[c] = 0;
            }
        }
        // Check leftwards
        for (int j = c - 1; j > 0 && !grid[r][j]; --j) {
            grid[r][j] = false;
            if (row챈혰혰[j] == 1) {
                column챈혰혰[r] = 0;
            }
        }
        // Check rightwards
        for (int j = c + 1; j <= w && !grid[r][j]; ++j) {
            grid[r][j] = false;
            if (row챈혰혰[j] == 1) {
                column챈혰혰[r] = 0;
            }
        }
    }

    // Count remaining walls
    int remainingWalls = 0;
    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            if (grid[i][j]) {
                remainingWalls++;
            }
        }
    }

    cout << remainingWalls << endl;

    return 0;
}
