#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> rows(h), cols(w);

    // Initialize each row and column with all positions having a wall
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there is a wall at the bomb position
        if (rows[r].count(c)) {
            // Remove the wall at (r, c)
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                int u = *--it;
                rows[u].erase(c);
                cols[c].erase(u);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int d = *it;
                rows[d].erase(c);
                cols[c].erase(d);
            }

            // Left
            auto it2 = rows[r].lower_bound(c);
            if (it2 != rows[r].begin()) {
                int l = *--it2;
                cols[l].erase(r);
                rows[r].erase(l);
            }

            // Right
            it2 = rows[r].lower_bound(c);
            if (it2 != rows[r].end()) {
                int ri = *it2;
                cols[ri].erase(r);
                rows[r].erase(ri);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << '\n';

    return 0;
}
