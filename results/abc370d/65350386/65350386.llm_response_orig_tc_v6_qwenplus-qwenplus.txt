#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> row(w), col(h);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (col[r].count(c)) {
            // cell has wall, remove it
            col[r].erase(c);
            row[c].erase(r);
        } else {
            // look up
            auto it = row[c].lower_bound(r);
            if (it != row[c].begin()) {
                it--;
                int x = *it;
                col[x].erase(c);
                row[c].erase(it);
            }

            // look down
            it = row[c].lower_bound(r);
            if (it != row[c].end()) {
                int x = *it;
                col[x].erase(c);
                row[c].erase(it);
            }

            // look left
            auto jt = col[r].lower_bound(c);
            if (jt != col[r].begin()) {
                jt--;
                int y = *jt;
                row[y].erase(r);
                col[r].erase(y);
            }

            // look right
            jt = col[r].lower_bound(c);
            if (jt != col[r].end()) {
                int y = *jt;
                row[y].erase(r);
                col[r].erase(jt);
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    cout << ans << endl;
    return 0;
}
