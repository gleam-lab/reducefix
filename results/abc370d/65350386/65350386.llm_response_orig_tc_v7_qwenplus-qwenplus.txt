#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long hang;

// Union-Find Tree
struct UnionFind {
    vector<int> parent, size;
    UnionFind(int n) : parent(n), size(n,1) {
        iota(parent.begin(), parent.end(), 0);
    }
    int root(int x) {
        while (x != parent[x]) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }
    void unite(int x, int y) {
        x = root(x);
        y = root(y);
        if (x == y) return;
        if (size[x] < size[y]) swap(x, y);
        parent[y] = x;
        size[x] += size[y];
    }
    bool same(int x, int y) { return root(x) == root(y); }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int H, W, Q;
    cin >> H >> W >> Q;

    int N = H * W;
    vector<vector<set<int>>> grid(H, vector<set<int>>(W));
    UnionFind uf(N);

    // Initialize walls in each cell
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            grid[i][j].insert(i * W + j);

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        R--;
        C--;

        int idx = R * W + C;

        if (grid[R][C].count(idx)) {
            // Already has a wall, remove it
            grid[R][C].erase(idx);
        } else {
            // No wall at current position, destroy first visible walls in all directions

            // Up
            for (int r = R - 1; r >= 0; --r) {
                if (!grid[r][C].empty()) {
                    int w = *grid[r][C].begin();
                    grid[r][C].erase(w);
                    break;
                }
            }

            // Down
            for (int r = R + 1; r < H; ++r) {
                if (!grid[r][C].empty()) {
                    int w = *grid[r][C].begin();
                    grid[r][C].erase(w);
                    break;
                }
            }

            // Left
            for (int c = C - 1; c >= 0; --c) {
                if (!grid[R][c].empty()) {
                    int w = *grid[R][c].begin();
                    grid[R][c].erase(w);
                    break;
                }
            }

            // Right
            for (int c = C + 1; c < W; ++c) {
                if (!grid[R][c].empty()) {
                    int w = *grid[R][c].begin();
                    grid[R][c].erase(w);
                    break;
                }
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            ans += grid[i][j].size();

    cout << ans << endl;

    return 0;
}
