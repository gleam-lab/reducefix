Here's the corrected version of your code. The main issue was with how the walls were being destroyed when placing a new bomb. Instead of just destroying the first wall it encounters in any direction, we should continue until we hit another wall or reach the edge of the grid.

```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    while (q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        if (!grid[r][c]) {
            for (int i = r-1; i >= 0 && !grid[i][c]; --i) {
                grid[i][c] = true;
            }
            for (int i = r+1; i < h && !grid[i][c]; ++i) {
                grid[i][c] = true;
            }
            for (int j = c-1; j >= 0 && !grid[r][j]; --j) {
                grid[r][j] = true;
            }
            for (int j = c+1; j < w && !grid[r][j]; ++j) {
                grid[r][j] = true;
            }
        } else {
            grid[r][c] = false;
        }
    }
    int ans = 0;
    for (const auto& row : grid) {
        ans += count(row.begin(), row.end(), false);
    }
    cout << ans << "\n";
    return 0;
}
```
