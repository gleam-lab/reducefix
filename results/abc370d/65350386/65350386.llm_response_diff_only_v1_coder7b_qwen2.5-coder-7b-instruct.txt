#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<map<int,bool>> tate(w),yoko(h);
    rep(i,h) rep(j,w) {
        tate[j][i] = true;
        yoko[i][j] = true;
    }
    rep(qi,q) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if (tate[c][r]) {
            tate[c][r] = false;
            yoko[r][c] = false;
        } else {
            auto tate_iter_l = tate[c].lower_bound(r);
            if (tate_iter_l != tate[c].begin() && prev(tate_iter_l)->second) {
                int tmp = prev(tate_iter_l)->first;
                tate[c][tmp] = false;
                yoko[tmp][c] = false;
            }

            auto tate_iter_r = tate[c].upper_bound(r);
            if (tate_iter_r != tate[c].end() && next(tate_iter_r)->second) {
                int tmp = next(tate_iter_r)->first;
                tate[c][tmp] = false;
                yoko[tmp][c] = false;
            }
    
            auto yoko_iter_u = yoko[r].lower_bound(c);
            if (yoko_iter_u != yoko[r].begin() && prev(yoko_iter_u)->second) {
                int tmp = prev(yoko_iter_u)->first;
                yoko[r][tmp] = false;
                tate[tmp][r] = false;
            }
    
            auto yoko_iter_d = yoko[r].upper_bound(c);
            if (yoko_iter_d != yoko[r].end() && next(yoko_iter_d)->second) {
                int tmp = next(yoko_iter_d)->first;
                yoko[r][tmp] = false;
                tate[tmp][r] = false;
            }
    
        }
    }
    int ans = 0;
    rep(i,w) {
        ans += count_if(begin(tate[i]), end(tate[i]), [](const pair<int,bool>& p){return p.second;});
    }
    cout << ans << "\n";
    return 0;
}
