#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> rows(h), cols(w);

    // Initialize each row and column with all indices (representing walls)
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if the current cell has a wall
        if (rows[r].count(c)) {
            // Remove the wall at (r, c)
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Find and remove the nearest walls in the four directions

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                it--;
                int up_row = *it;
                rows[up_row].erase(c);
                cols[c].erase(it);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int down_row = *it;
                rows[down_row].erase(c);
                cols[c].erase(it);
            }

            // Left
            it = rows[r].lower_bound(c);
            if (it != rows[r].begin()) {
                it--;
                int left_col = *it;
                cols[left_col].erase(r);
                rows[r].erase(it);
            }

            // Right
            it = rows[r].lower_bound(c);
            if (it != rows[r].end()) {
                int right_col = *it;
                cols[right_col].erase(r);
                rows[r].erase(it);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";

    return 0;
}
