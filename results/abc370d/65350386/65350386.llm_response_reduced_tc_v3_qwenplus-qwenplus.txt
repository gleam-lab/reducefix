#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (n); i++)
using ll = long long;

const int INF = 1 << 30;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row and column, track available walls in sorted order
    vector<set<int>> rows(H), cols(W);
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // If there's a wall at (r,c)
        if (cols[c].count(r)) {
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Otherwise, destroy the first walls in all four directions

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                --it;
                int y = *it;
                rows[y].erase(c);
                cols[c].erase(y);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int y = *it;
                rows[y].erase(c);
                cols[c].erase(y);
            }

            // Left
            auto jt = rows[r].lower_bound(c);
            if (jt != rows[r].begin()) {
                --jt;
                int x = *jt;
                cols[x].erase(r);
                rows[r].erase(x);
            }

            // Right
            jt = rows[r].lower_bound(c);
            if (jt != rows[r].end()) {
                int x = *jt;
                cols[x].erase(r);
                rows[r].erase(x);
            }
        }
    }

    int ans = 0;
    for (auto &row : rows)
        ans += row.size();

    cout << ans << endl;
    return 0;
}
