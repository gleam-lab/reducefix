#include <bits/stdc++.h>
using namespace std;

const int MAXN = 4e5+5;

struct Point{
    int x,y;
    bool operator<(const Point &p) const{ return x==p.x?y<p.y:x<p.x;}
};

vector<Point> grid[MAXN];
bool vis[MAXN][MAXN];

void clear(){
    for(auto& v:grid) v.clear();
    memset(vis,false,sizeof(vis));
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int h, w, q;
    cin >> h >> w >> q;

    while(q--){
        int r,c;
        cin >> r >> c;
        --r,--c;

        clear();

        queue<Point> q1;
        set<int> s1,s2,s3,s4;
        s1.insert(r);
        s2.insert(c);
        q1.push({r,c});

        while(!q1.empty()){
            Point p = q1.front(); 
            q1.pop();
            int x=p.x,y=p.y;

            if(vis[x][y]) continue;
            vis[x][y]=true;
            grid[y].push_back(p);

            if(x-1>=0 && !vis[x-1][y] && s1.find(x-1)==s1.end()){
                s1.insert(x-1);
                q1.push({x-1,y});
            }
            if(x+1<h && !vis[x+1][y] && s1.find(x+1)==s1.end()){
                s1.insert(x+1);
                q1.push({x+1,y});
            }
            if(y-1>=0 && !vis[x][y-1] && s2.find(y-1)==s2.end()){
                s2.insert(y-1);
                q1.push({x,y-1});
            }
            if(y+1<w && !vis[x][y+1] && s2.find(y+1)==s2.end()){
                s2.insert(y+1);
                q1.push({x,y+1});
            }
        }
    }

    int cnt=0;
    for(int i=0;i<w;++i){
        for(auto &p:grid[i]){
            cnt++;
        }
    }

    cout<<cnt<<"\n";

    return 0;
}
