#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    vector<queue<pair<int, int>>> vertical(w);
    vector<queue<pair<int, int>>> horizontal(h);

    for(int qi = 0; qi < q; qi++) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        if(grid[r][c]) {
            grid[r][c] = false;
            vertical[c].push({r, c});
            horizontal[r].push({r, c});
        } else {
            while(!vertical[c].empty() && !grid[vertical[c].front().first][c]) {
                vertical[c].pop();
            }
            if(!vertical[c].empty()) {
                auto [rr, cc] = vertical[c].front();
                grid[rr][cc] = false;
                vertical[c].pop();
                horizontal[rr].push({rr, cc});
            }

            while(!horizontal[r].empty() && !grid[r][horizontal[r].front().second]) {
                horizontal[r].pop();
            }
            if(!horizontal[r].empty()) {
                auto [rr, cc] = horizontal[r].front();
                grid[rr][cc] = false;
                horizontal[r].pop();
                vertical[cc].push({rr, cc});
            }
        }
    }

    int count = 0;
    for(int i = 0; i < h; ++i) {
        for(int j = 0; j < w; ++j) {
            if(grid[i][j]) count++;
        }
    }

    cout << count << endl;
    return 0;
}
