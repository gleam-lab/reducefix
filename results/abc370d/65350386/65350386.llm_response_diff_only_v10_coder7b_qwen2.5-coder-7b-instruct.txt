```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (!grid[r][c]) continue;

        grid[r][c] = false;

        for (int i = r - 1; i >= 0 && grid[i][c]; --i)
            grid[i][c] = false;

        for (int i = r + 1; i < h && grid[i][c]; ++i)
            grid[i][c] = false;

        for (int j = c - 1; j >= 0 && grid[r][j]; --j)
            grid[r][j] = false;

        for (int j = c + 1; j < w && grid[r][j]; ++j)
            grid[r][j] = false;
    }

    int ans = 0;
    for (const auto& row : grid)
        ans += count(row.begin(), row.end(), true);

    cout << ans << '\n';
    return 0;
}
```

This solution uses a 2D boolean array `grid` to represent whether each cell contains a wall or not. When a bomb is dropped, it destroys the wall at the specified position and all consecutive walls in the same row and column above/below and left/right until it hits another wall or reaches the boundary. The final answer is the total number of remaining walls, which is calculated by counting the number of `true` entries in the `grid`. This approach ensures that the solution is both correct and efficient, adhering to the problem constraints.
