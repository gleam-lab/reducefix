```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long hang;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // For each row and column, we keep track of available positions using sets
    vector<set<int>> row(h), col(w);
    
    // Initialize all walls are present
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[i].insert(j);
            col[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (row[r].count(c)) {
            // There's a wall at (r,c). Remove it.
            row[r].erase(c);
            col[c].erase(r);
        } else {
            // No wall at (r,c). Need to destroy the first walls in each direction.

            // Up
            auto it = col[c].lower_bound(r);
            if (it != col[c].begin()) {
                --it;
                int u = *it;
                row[u].erase(c);
                col[c].erase(it);
            }

            // Down
            it = col[c].lower_bound(r);
            if (it != col[c].end()) {
                int d = *it;
                row[d].erase(c);
                col[c].erase(it);
            }

            // Left
            auto jt = row[r].lower_bound(c);
            if (jt != row[r].begin()) {
                --jt;
                int l = *jt;
                col[l].erase(r);
                row[r].erase(jt);
            }

            // Right
            jt = row[r].lower_bound(c);
            if (jt != row[r].end()) {
                int ri = *jt;
                col[ri].erase(r);
                row[r].erase(jt);
            }
        }
    }

    // Count remaining walls
    int total = 0;
    for (int i = 0; i < h; ++i)
        total += row[i].size();
    
    cout << total << endl;
    return 0;
}
```
