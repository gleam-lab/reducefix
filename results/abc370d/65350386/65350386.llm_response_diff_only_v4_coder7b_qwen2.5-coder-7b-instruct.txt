```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    int count = h * w;
    
    while(q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if(grid[r][c]) {
            grid[r][c] = false;
            count--;
        } else {
            // Destroy walls upwards
            for(int i = r - 1; i >= 0 && !grid[i][c]; --i) {
                grid[i][c] = false;
                count--;
            }
            
            // Destroy walls downwards
            for(int i = r + 1; i < h && !grid[i][c]; ++i) {
                grid[i][c] = false;
                count--;
            }
            
            // Destroy walls to the left
            for(int j = c - 1; j >= 0 && !grid[r][j]; --j) {
                grid[r][j] = false;
                count--;
            }
            
            // Destroy walls to the right
            for(int j = c + 1; j < w && !grid[r][j]; ++j) {
                grid[r][j] = false;
                count--;
            }
        }
    }
    
    cout << count << "\n";
    return 0;
}
```

This solution uses a boolean matrix `grid` to represent whether each cell has a wall or not. The total number of walls is initialized as `h * w`. For each query, it checks if the cell has a wall. If it does, it removes the wall and decreases the count. If it doesn't, it iterates upwards, downwards, left, and right to remove consecutive empty cells, updating both the `grid` and the `count` accordingly. Finally, it outputs the number of remaining walls. This approach ensures correctness and efficiency within the given constraints.
