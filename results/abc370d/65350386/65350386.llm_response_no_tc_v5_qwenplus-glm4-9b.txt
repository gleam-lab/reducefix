```cpp
#include <iostream>
#include <vector>
#include <set>
using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<set<int>>> tate(h, vector<set<int>>(w)), yoko(h, vector<set<int>>(w));

    // Initialize the grid with walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            tate[i][j].insert(i);
            tate[i][j].insert(i + h);
            yoko[i][j].insert(j);
            yoko[i][j].insert(j + w);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Place a bomb and destroy walls
        if (tate[r][c].count(c)) {
            tate[r][c].erase(c);
            yoko[r][c].erase(c);
        } else {
            // Find the highest wall below
            auto tate_iter = tate[r][c].upper_bound(c);
            if (tate_iter != tate[r][c].end()) {
                tate[r][c].erase(*tate_iter);
                yoko[*tate_iter][c].erase(r);
            }

            // Find the lowest wall above
            tate_iter = tate[r][c].lower_bound(c);
            if (tate_iter != tate[r][c].begin()) {
                tate[r][c].erase(*prev(tate_iter));
                yoko[*prev(tate_iter)][c].erase(r);
            }

            // Find the rightmost wall to the right
            tate_iter = yoko[r][c].upper_bound(c);
            if (tate_iter != yoko[r][c].end()) {
                tate[r][c].erase(*tate_iter);
                tate[*tate_iter][r].erase(c);
            }

            // Find the leftmost wall to the left
            tate_iter = yoko[r][c].lower_bound(c);
            if (tate_iter != yoko[r][c].begin()) {
                tate[r][c].erase(*prev(tate_iter));
                tate[*prev(tate_iter)][r].erase(c);
            }
        }
    }

    // Calculate remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            remaining_walls += tate[i][j].size() / 2 + yoko[i][j].size() / 2;
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
```

This code initializes each cell with two walls (one for the row and one for the column), processes each query to destroy walls as described, and finally counts the remaining walls. The use of sets ensures that each wall is uniquely represented and can be efficiently removed when a bomb is placed.
