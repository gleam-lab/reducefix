#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (int)(n); i++)
using ll = long long;

// Define sets for each row and column to track remaining walls
vector<set<int>> row_walls, col_walls;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // Initialize row and column sets with all positions having walls
    row_walls.resize(H);
    col_walls.resize(W);
    for (int r = 0; r < H; ++r) {
        for (int c = 0; c < W; ++c) {
            row_walls[r].insert(c);
            col_walls[c].insert(r);
        }
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        // If there's a wall at (R,C), destroy it
        if (col_walls[C].count(R)) {
            row_walls[R].erase(C);
            col_walls[C].erase(R);
        } else {
            // Look up
            auto it = col_walls[C].lower_bound(R);
            if (it != col_walls[C].begin()) {
                --it;
                int r = *it;
                row_walls[r].erase(C);
                col_walls[C].erase(it);
            }

            // Look down
            it = col_walls[C].lower_bound(R);
            if (it != col_walls[C].end()) {
                int r = *it;
                row_walls[r].erase(C);
                col_walls[C].erase(it);
            }

            // Look left
            auto it2 = row_walls[R].lower_bound(C);
            if (it2 != row_walls[R].begin()) {
                --it2;
                int c = *it2;
                col_walls[c].erase(R);
                row_walls[R].erase(it2);
            }

            // Look right
            it2 = row_walls[R].lower_bound(C);
            if (it2 != row_walls[R].end()) {
                int c = *it2;
                col_walls[c].erase(R);
                row_walls[R].erase(it2);
            }
        }
    }

    int ans = 0;
    for (int r = 0; r < H; ++r)
        ans += row_walls[r].size();
        
    cout << ans << '\n';
    return 0;
}
