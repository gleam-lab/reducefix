#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)
#define _rep(i, n) for (int i = (n) - 1; i >= 0; i--)

using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> grid(h, vector<int>(w));
    vector<vector<bool>> rowWalls(h, vector<bool>(w, true));
    vector<vector<bool>> colWalls(w, vector<bool>(w, true));

    for (int i = 0; i < h; i++) {
        rep(j, w) {
            grid[i][j] = 1;  // 1 wall in each cell initially
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check and destroy the wall at (r, c) if it exists
        if (rowWalls[r][c]) {
            grid[r][c] = 0;  // Remove the wall
            rowWalls[r][c] = false;
            colWalls[c][r] = false;

            // Remove walls that are adjacent to (r, c) if they are the only wall left
            _rep(i, r + 1) {
                if (colWalls[c][i]) {
                    grid[i][c] = 0;
                    rowWalls[i][c] = false;
                    colWalls[c][i] = false;
                }
            }
            _rep(i, c + 1) {
                if (rowWalls[r][i]) {
                    grid[r][i] = 0;
                    rowWalls[r][i] = false;
                    colWalls[i][r] = false;
                }
            }
        }
    }

    // Count the remaining walls
    ll remainingWalls = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            remainingWalls += grid[i][j];
        }
    }

    cout << remainingWalls << endl;
    return 0;
}
