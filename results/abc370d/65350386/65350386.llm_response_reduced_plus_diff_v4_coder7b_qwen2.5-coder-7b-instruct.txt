#include <bits/stdc++.h>
using namespace std;

#define ll long long

struct BIT {
    int n;
    vector<ll> tree;
    BIT(int _n) : n(_n), tree(_n+1, 0) {}
    void update(int idx, ll val) {
        while(idx <= n) {
            tree[idx] += val;
            idx += idx & (-idx);
        }
    }
    ll sum(int idx) {
        ll ret = 0;
        while(idx > 0) {
            ret += tree[idx];
            idx -= idx & (-idx);
        }
        return ret;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int h, w, q;
    cin >> h >> w >> q;
    
    BIT bit_h(h), bit_w(w);
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    for(int i=0; i<h; ++i) bit_h.update(i+1, w);
    for(int j=0; j<w; ++j) bit_w.update(j+1, h);

    for(int qi=0; qi<q; ++qi) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        
        if(grid[r][c]) {
            grid[r][c] = false;
            bit_h.update(r+1, -1);
            bit_w.update(c+1, -1);
        } else {
            int cnt_r = bit_h.sum(r+1);
            int cnt_c = bit_w.sum(c+1);
            if(cnt_r == 0 || cnt_c == 0) continue;

            int start_r = bit_h.sum(r) == cnt_r ? r : r+1;
            int end_r = bit_h.sum(r+2) == cnt_r ? r+1 : r;
            int start_c = bit_w.sum(c) == cnt_c ? c : c+1;
            int end_c = bit_w.sum(c+2) == cnt_c ? c+1 : c;

            for(int i=start_r; i<=end_r; ++i) bit_h.update(i, -1);
            for(int j=start_c; j<=end_c; ++j) bit_w.update(j, -1);
        }
    }

    ll ans = 0;
    for(int i=0; i<h; ++i) {
        ans += bit_h.sum(i+1);
    }
    cout << ans << '\n';

    return 0;
}
