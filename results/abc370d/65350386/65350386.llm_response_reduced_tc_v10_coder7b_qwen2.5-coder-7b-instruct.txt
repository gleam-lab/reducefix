#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    rep(i, h) rep(j, w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--, c--;

        if (!tate[c].empty() && *(tate[c].begin()) == r) {
            tate[c].erase(tate[c].begin());
            yoko[r].erase(c);
        } else {
            auto it = tate[c].lower_bound(r);
            if (it != tate[c].begin()) {
                it--;
                tate[c].erase(it);
                yoko[*it].erase(c);
            }

            it = tate[c].upper_bound(r);
            if (it != tate[c].end()) {
                tate[c].erase(it);
                yoko[*it].erase(c);
            }

            it = yoko[r].lower_bound(c);
            if (it != yoko[r].begin()) {
                it--;
                yoko[r].erase(it);
                tate[*it].erase(r);
            }

            it = yoko[r].upper_bound(c);
            if (it != yoko[r].end()) {
                yoko[r].erase(it);
                tate[*it].erase(r);
            }
        }
    }

    int ans = 0;
    rep(i, w) {
        ans += tate[i].size();
    }
    cout << ans << "\n";

    return 0;
}
