#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (int)(n); i++)
using ll = long long;

// Data structures to represent rows and columns
vector<set<int>> row_walls, col_walls;

// Remove wall from a cell
void removeWall(int r, int c) {
    row_walls[r].erase(c);
    col_walls[c].erase(r);
}

// Place bomb at (r,c)
void placeBomb(int r, int c) {
    // If there's a wall at the position, destroy it
    if (row_walls[r].count(c)) {
        removeWall(r, c);
        return;
    }

    // Otherwise, destroy first walls in each direction
    auto& row = row_walls[r];
    auto& col = col_walls[c];

    // Left
    auto it = row.lower_bound(c);
    if (it != row.begin()) {
        int x = *prev(it);
        removeWall(r, x);
    }

    // Right
    it = row.lower_bound(c);
    if (it != row.end()) {
        int x = *it;
        removeWall(r, x);
    }

    // Up
    auto jt = col.lower_bound(r);
    if (jt != col.begin()) {
        int y = *prev(jt);
        removeWall(y, c);
    }

    // Down
    jt = col.lower_bound(r);
    if (jt != col.end()) {
        int y = *jt;
        removeWall(y, c);
    }
}

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Initialize walls
    row_walls.resize(H);
    col_walls.resize(W);

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            row_walls[i].insert(j);
            col_walls[j].insert(i);
        }
    }

    // Process queries
    while (Q--) {
        int r, c;
        cin >> r >> c;
        placeBomb(r - 1, c - 1);
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < H; ++i) {
        ans += row_walls[i].size();
    }
    cout << ans << endl;

    return 0;
}
