#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<n;i++)
#define vi vector<int>
#define vl vector<long long>
#define vb vector<bool>
#define vs vector<string>
#define vvi vector<vector<int>>
#define pb push_back
#define mp make_pair
const int MOD = 1e9 + 7;

void solve() {
    int n,m,q; cin >> n >> m >> q;
    vvi grid(n, vi(m, 1));
    while(q--) {
        int r,c; cin >> r >> c; r--,c--;
        if(grid[r][c] == 1) {
            grid[r][c] = 0;
        } else {
            int dr[] = {-1,1,0,0};
            int dc[] = {0,0,-1,1};
            for(int i=0;i<4;i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                while(nr>=0 && nr<n && nc>=0 && nc<m && grid[nr][nc]==0) {
                    nr += dr[i];
                    nc += dc[i];
                }
                if(nr>=0 && nr<n && nc>=0 && nc<m && grid[nr][nc]==1) {
                    grid[nr][nc] = 0;
                }
            }
        }
    }
    int ans = 0;
    rep(i,n) rep(j,m) if(grid[i][j]) ans++;
    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
