#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>

int main(){
    ll h,w,q;cin >> h >> w >> q;
    vector<vector<bool>> grid(h,vector<bool>(w,true));
    
    while(q--){
        ll r,c;cin >> r >> c;
        r--;c--;
        
        if(grid[r][c]){
            grid[r][c] = false;
            continue;
        }
        
        bool found = false;
        // Check up
        for(int i=r-1; i>=0; --i){
            if(grid[i][c]){
                grid[i][c] = false;
                found = true;
                break;
            }
        }
        
        if(found) continue;
        
        // Check down
        for(int i=r+1; i<h; ++i){
            if(grid[i][c]){
                grid[i][c] = false;
                found = true;
                break;
            }
        }
        
        if(found) continue;
        
        // Check left
        for(int j=c-1; j>=0; --j){
            if(grid[r][j]){
                grid[r][j] = false;
                found = true;
                break;
            }
        }
        
        if(found) continue;
        
        // Check right
        for(int j=c+1; j<w; ++j){
            if(grid[r][j]){
                grid[r][j] = false;
                found = true;
                break;
            }
        }
    }
    
    ll ans = 0;
    for(auto& row : grid){
        for(bool cell : row){
            if(cell) ans++;
        }
    }
    cout << ans << endl;
}
