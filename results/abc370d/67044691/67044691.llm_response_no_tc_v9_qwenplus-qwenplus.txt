#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for(int i = 0; i < (int)n; i++)

int main() {
    ll h, w, q;
    cin >> h >> w >> q;

    vector<set<ll>> rows(h), cols(w);

    // Initially all cells have walls
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        ll r, c;
        cin >> r >> c;
        r--; c--;

        if (rows[r].count(c)) {
            // Wall exists at (r,c), remove it
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Find the nearest walls in all four directions
            ll up = -1, down = -1, left = -1, right = -1;

            // Up: find first wall above (r, c)
            if (!cols[c].empty()) {
                auto it = cols[c].lower_bound(r);
                if (it != cols[c].begin()) {
                    --it;
                    up = *it;
                }
            }

            // Down: find first wall below (r, c)
            if (!cols[c].empty()) {
                auto it = cols[c].upper_bound(r);
                if (it != cols[c].end()) {
                    down = *it;
                }
            }

            // Left: find first wall to the left on row r
            if (!rows[r].empty()) {
                auto it = rows[r].lower_bound(c);
                if (it != rows[r].begin()) {
                    --it;
                    left = *it;
                }
            }

            // Right: find first wall to the right on row r
            if (!rows[r].empty()) {
                auto it = rows[r].upper_bound(c);
                if (it != rows[r].end()) {
                    right = *it;
                }
            }

            // Remove those walls
            if (up != -1) {
                rows[up].erase(c);
                cols[c].erase(up);
            }
            if (down != -1) {
                rows[down].erase(c);
                cols[c].erase(down);
            }
            if (left != -1) {
                cols[left].erase(r);
                rows[r].erase(left);
            }
            if (right != -1) {
                cols[right].erase(r);
                rows[r].erase(right);
            }
        }
    }

    // Count remaining walls
    ll ans = 0;
    for (ll i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << endl;

    return 0;
}
