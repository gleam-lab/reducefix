#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<set<int>> rows(H), cols(W);
    // Initialize the rows and columns with wall positions
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }
    
    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        R--; C--; // Convert to zero-based index
        
        // Check if there's a wall at the given position
        if (rows[R].find(C) != rows[R].end()) {
            rows[R].erase(C);
            cols[C].erase(R);
        } else {
            // No wall at (R, C), clear the walls in the four directions
            // Clear vertical walls to the left
            while (!cols[C].empty() && cols[C].find(R) != cols[C].end()) {
                rows[*(cols[C].begin())].erase(*(cols[C].begin()));
                cols[C].erase(*(cols[C].begin()));
            }
            // Clear vertical walls to the right
            while (!cols[C].empty() && cols[C].rbegin()->first != R) {
                rows[*(cols[C].rbegin()->first)].erase(*(cols[C].rbegin()->first));
                cols[C].erase(*(cols[C].rbegin()->first));
            }
            // Clear horizontal walls upwards
            while (!rows[R].empty() && rows[R].find(C) != rows[R].end()) {
                cols[*(rows[R].begin())].erase(*(rows[R].begin()));
                rows[R].erase(*(rows[R].begin()));
            }
            // Clear horizontal walls downwards
            while (!rows[R].empty() && rows[R].rbegin()->first != C) {
                cols[*(rows[R].rbegin()->first)].erase(*(rows[R].rbegin()->first));
                rows[R].erase(*(rows[R].rbegin()->first));
            }
        }
    }
    
    // Calculate the number of remaining walls
    int remainingWalls = 0;
    for (const auto& row : rows) {
        remainingWalls += row.size();
    }
    
    cout << remainingWalls << endl;
    
    return 0;
}
