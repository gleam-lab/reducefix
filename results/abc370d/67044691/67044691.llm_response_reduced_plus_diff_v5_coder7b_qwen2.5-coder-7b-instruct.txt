                ri--;
                if (cols[c].count(*ri)) {
                    cols[c].erase(ri);
                    rows[*ri].erase(c);
                }
            }
            if (ri != rows[r].end()) {

                    cols[c].erase(ri);

                if (rows[*ci].count(c)) {
