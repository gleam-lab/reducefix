#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<n;i++)
#define repi(i,a,b) for(int i=a;i<b;i++)
#define pb push_back
#define mp make_pair
#define F first
#define S second
#define INF 1e18
#define MOD 1000000007
typedef pair<ll,ll> pi;
typedef vector<ll> vi;
typedef vector<pi> vpi;
typedef set<ll> si;
typedef map<ll,ll> mli;

void solve() {
    ll h,w,q;
    cin >> h >> w >> q;
    vi dr={0,0,1,-1};
    vi dc={1,-1,0,0};
    vector<vector<bool>> g(h,vector<bool>(w,true));
    while(q--) {
        ll r,c;
        cin >> r >> c;
        r--;c--;
        if(g[r][c]) {
            g[r][c]=false;
        } else {
            for(int i=0;i<4;i++) {
                ll nr=r+dr[i],nc=c+dc[i];
                if(nr>=0&&nr<h&&nc>=0&&nc<w&&!g[nr][nc]) {
                    g[nr][nc]=true;
                }
            }
        }
    }
    ll ans=0;
    rep(i,h) rep(j,w) ans+=(int)g[i][j];
    cout << ans << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll t;
    cin >> t;
    while(t--) solve();
    return 0;
}
