#include <iostream>
#include <vector>
#include <set>
using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<set<int>>> grid(h, vector<set<int>>(w));
    
    // Initialize the grid with walls
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            grid[i][j].insert(j);
        }
    }

    for (int i = 0; i < q; i++) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (grid[r][c].find(c) != grid[r][c].end()) {
            // If there's a wall at (r, c), remove it
            grid[r][c].erase(c);
        } else {
            // Find the first wall in the row above and below
            int up = -1, down = -1;
            for (int pos : grid[r]) {
                if (pos > c) break;
                down = pos; // First wall below (r, c)
            }
            for (int pos : grid[r]) {
                if (pos < c) {
                    up = pos; // First wall above (r, c)
                    break;
                }
            }

            // Find the first wall in the column to the left and right
            int left = -1, right = -1;
            for (int i = 0; i < h; i++) {
                if (grid[i][c].find(c) == grid[i][c].end()) continue; // No wall at (i, c)
                if (left == -1) left = i; // First wall to the left
                right = i; // First wall to the right
            }

            // Remove the walls
            if (up != -1) grid[up][c].erase(c);
            if (down != -1) grid[down][c].erase(c);
            if (left != -1) {
                for (int i = 0; i < h; i++) {
                    grid[i][left].erase(c);
                }
            }
            if (right != -1) {
                for (int i = 0; i < h; i++) {
                    grid[i][right].erase(c);
                }
            }
        }
    }

    // Calculate the number of remaining walls
    long long remaining_walls = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            remaining_walls += grid[i][j].size();
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
