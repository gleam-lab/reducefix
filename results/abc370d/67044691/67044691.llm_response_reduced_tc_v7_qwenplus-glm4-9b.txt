#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    long long H, W, Q;
    cin >> H >> W >> Q;

    vector<set<long long>> row(H), col(W);

    // Initialize rows and columns with all walls
    for (long long i = 0; i < H; ++i) {
        for (long long j = 0; j < W; ++j) {
            row[i].insert(j);
            col[j].insert(i);
        }
    }

    while (Q--) {
        long long R, C;
        cin >> R >> C;
        R--; C--;

        // Check if a wall exists at (R, C)
        if (!row[R].erase(C)) {
            // If there is no wall at (R, C), destroy the first wall in each direction
            for (long long i = 0; i < 2; ++i) {
                long long dr = i == 0 ? -1 : 1;
                for (long long j = 0; j < 2; ++j) {
                    long long dc = j == 0 ? -1 : 1;
                    long long r = R + dr * dr;
                    long long c = C + dc * dc;
                    // Destroy walls in the specified direction until another wall is found
                    while (row[r].find(c) != row[r].end()) {
                        row[r].erase(c);
                        col[c].erase(r);
                        r += dr;
                        c += dc;
                    }
                }
            }
        }
    }

    // Calculate the number of remaining walls
    long long remainingWalls = 0;
    for (const auto& r : row) {
        remainingWalls += r.size();
    }

    cout << remainingWalls << endl;

    return 0;
}
