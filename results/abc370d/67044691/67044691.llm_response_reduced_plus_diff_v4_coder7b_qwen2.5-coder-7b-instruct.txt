#include <bits/stdc++.h>
using namespace std;

struct DisjointSet {
    vector<int> parent, rank;
    
    DisjointSet(int n) {
        parent.resize(n);
        rank.resize(n, 0);
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int u) {
        if (parent[u] != u) {
            parent[u] = find(parent[u]);
        }
        return parent[u];
    }

    void unionSets(int u, int v) {
        int pu = find(u);
        int pv = find(v);

        if (pu == pv) return;

        if (rank[pu] > rank[pv]) {
            parent[pv] = pu;
        } else if (rank[pu] < rank[pv]) {
            parent[pu] = pv;
        } else {
            parent[pu] = pv;
            rank[pv]++;
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, q;
    cin >> h >> w >> q;

    DisjointSet ds(h * w + 1);
    set<pair<int, pair<int, int>>> bombs;

    for (int i = 0; i < q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        bombs.insert({r * w + c, {r, c}});
    }

    while (!bombs.empty()) {
        auto [idx, coord] = *bombs.rbegin();
        bombs.erase(prev(bombs.end()));

        int r = coord.first, c = coord.second;
        int id = r * w + c;

        vector<pair<int, int>> directions{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        for (auto& dir : directions) {
            int nr = r + dir.first, nc = c + dir.second;
            if (nr >= 0 && nr < h && nc >= 0 && nc < w) {
                int nid = nr * w + nc;
                if (ds.find(id) != ds.find(nid)) {
                    ds.unionSets(id, nid);
                }
            }
        }
    }

    unordered_set<int> uniqueParents;
    for (int i = 0; i < h * w; ++i) {
        uniqueParents.insert(ds.find(i));
    }

    cout << uniqueParents.size() << endl;

    return 0;
}
