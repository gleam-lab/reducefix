#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define vi vector<int>
#define vl vector<vector<int>>

int main() {
    ll h, w, q;
    cin >> h >> w >> q;
    vl grid(h, vi(w, 1)); // Initialize all cells with 1 wall
    ll remainingWalls = h * w; // Start with all walls

    while (q--) {
        ll r, c;
        cin >> r >> c;
        r--; c--; // Convert to zero-indexed

        if (grid[r][c] == 1) {
            // If there's a wall at the given cell, destroy it
            grid[r][c]--;
            remainingWalls--;
        } else {
            // No wall at the given cell, destroy the first walls in each direction
            // Find the first wall above and below
            ll firstWallAbove = -1, firstWallBelow = -1;
            for (ll i = r - 1; i >= 0 && firstWallAbove == -1; i--) {
                if (grid[i][c] > 0) firstWallAbove = i;
            }
            for (ll i = r + 1; i < h && firstWallBelow == -1; i++) {
                if (grid[i][c] > 0) firstWallBelow = i;
            }
            // Find the first wall to the left and right
            ll firstWallLeft = -1, firstWallRight = -1;
            for (ll j = c - 1; j >= 0 && firstWallLeft == -1; j--) {
                if (grid[r][j] > 0) firstWallLeft = j;
            }
            for (ll j = c + 1; j < w && firstWallRight == -1; j++) {
                if (grid[r][j] > 0) firstWallRight = j;
            }

            // Destroy the walls in each direction
            if (firstWallAbove != -1) {
                grid[firstWallAbove][c]--;
                remainingWalls--;
            }
            if (firstWallBelow != -1) {
                grid[firstWallBelow][c]--;
                remainingWalls--;
            }
            if (firstWallLeft != -1) {
                grid[r][firstWallLeft]--;
                remainingWalls--;
            }
            if (firstWallRight != -1) {
                grid[r][firstWallRight]--;
                remainingWalls--;
            }
        }
    }

    cout << remainingWalls << endl;
    return 0;
}
