#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define repi(i,a,b) for(int i=a;i<b;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    ll h,w,q;cin >> h >> w >> q;
    vector<vector<bool>> grid(h,vector<bool>(w,true));
    set<pair<ll,ll>> s;
    rep(i,h)rep(j,w)s.insert({i,j});

    while(q--){
        ll r,c;cin >> r >> c;
        r--;c--;

        if(grid[r][c]){
            grid[r][c]=false;
            s.erase(s.find({r,c}));

            set<ll> delrows,delcols;
            delrows.insert(r);delcols.insert(c);

            auto it=s.lower_bound({r,c});
            while(it!=s.end()&&it->first==r){
                delcols.insert((*it)[1]);
                it++;
            }
            it=s.upper_bound({r,c});
            while(it!=s.begin()){
                it--;
                if((*it)[0]==r){
                    delcols.insert((*it)[1]);
                }
                else break;
            }
            s.erase(delrows.begin(),delrows.end());
            s.erase(delcols.begin(),delcols.end());

            for(auto &p:s){
                if(p.first==r||p.second==c){
                    s.erase(p);
                }
            }
        }
    }
    
    cout << s.size() << '\n';
}
