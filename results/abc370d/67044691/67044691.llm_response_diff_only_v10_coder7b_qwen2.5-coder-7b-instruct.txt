#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(int i=0;i<(int)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

int main(){
    ll h,w,q;cin >> h >> w >> q;
    vector<vector<bool>> grid(h,vector<bool>(w,true));
    rep(i,q){
        ll r,c;cin >> r >> c;
        r--;c--;
        if(grid[r][c]){
            grid[r][c]=false;
            // Check up
            for(ll rr=r-1;rr>=0;rr--){
                if(grid[rr][c])break;
                grid[rr][c]=false;
            }
            // Check down
            for(ll rr=r+1;rr<h;rr++){
                if(grid[rr][c])break;
                grid[rr][c]=false;
            }
            // Check left
            for(ll cc=c-1;cc>=0;cc--){
                if(grid[r][cc])break;
                grid[r][cc]=false;
            }
            // Check right
            for(ll cc=c+1;cc<w;cc++){
                if(grid[r][cc])break;
                grid[r][cc]=false;
            }
        }
    }
    ll ans=0;
    rep(i,h)rep(j,w)ans+=(grid[i][j]?1:0);
    cout << ans << endl;
}
