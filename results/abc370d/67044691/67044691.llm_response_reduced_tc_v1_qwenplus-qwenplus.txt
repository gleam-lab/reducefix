#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,r) for(ll i=0;i<(ll)r;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

int main(){
    ll h, w, q;
    cin >> h >> w >> q;
    vector<set<ll>> row(h), col(w);
    
    // Initialize all walls present
    rep(i, h) rep(j, w) {
        row[i].insert(j);
        col[j].insert(i);
    }

    rep(_, q) {
        ll r, c;
        cin >> r >> c;
        r--; c--;

        // If there's a wall at the current position
        if (row[r].count(c)) {
            row[r].erase(c);
            col[c].erase(r);
        } else {
            // Look in four directions for nearest walls
            // Right (east)
            auto right = row[r].upper_bound(c);
            if (right != row[r].end()) {
                ll j = *right;
                row[r].erase(j);
                col[j].erase(r);
            }

            // Left (west)
            if (!row[r].empty() && *row[r].rbegin() < c) {
                auto it = row[r].lower_bound(c);
                if (it != row[r].begin()) {
                    --it;
                    ll j = *it;
                    row[r].erase(j);
                    col[j].erase(r);
                }
            }

            // Down (south)
            auto down = col[c].upper_bound(r);
            if (down != col[c].end()) {
                ll i = *down;
                col[c].erase(i);
                row[i].erase(c);
            }

            // Up (north)
            if (!col[c].empty() && *col[c].rbegin() < r) {
                auto it = col[c].lower_bound(r);
                if (it != col[c].begin()) {
                    --it;
                    ll i = *it;
                    col[c].erase(i);
                    row[i].erase(c);
                }
            }
        }
    }

    ll ans = 0;
    rep(i, h) ans += row[i].size();
    cout << ans << endl;
}
