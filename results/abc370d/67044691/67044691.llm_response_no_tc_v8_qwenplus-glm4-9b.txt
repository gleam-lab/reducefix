#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    long long H, W, Q;
    cin >> H >> W >> Q;
    
    vector<vector<set<long long>>> row(H, vector<set<long long>>(W));
    vector<vector<set<long long>>> col(W, vector<set<long long>>(H));
    
    // Initialize walls in row and column sets
    for (long long i = 0; i < H; ++i) {
        for (long long j = 0; j < W; ++j) {
            row[i][j] = {i, j};
            col[j][i] = {i, j};
        }
    }
    
    for (long long q = 0; q < Q; ++q) {
        long long R, C;
        cin >> R >> C;
        --R; --C; // Convert to zero-indexed
        
        // Check if there's a wall at the specified position
        if (!row[R][C].empty()) {
            row[R][C].erase(row[R][C].begin()); // Remove the first wall
            col[C][R].erase(col[C][R].begin()); // Remove the first wall
        } else {
            // Remove the first wall from each adjacent row and column
            for (auto &p : row[R]) {
                auto it = p.upper_bound(C);
                if (it != p.end()) p.erase(it);
            }
            for (auto &p : col[C]) {
                auto it = p.upper_bound(R);
                if (it != p.end()) p.erase(it);
            }
        }
    }
    
    long long remainingWalls = 0;
    // Count the remaining walls in each row and column
    for (long long i = 0; i < H; ++i) {
        for (const auto &wall : row[i]) {
            remainingWalls++;
        }
    }
    for (long long j = 0; j < W; ++j) {
        for (const auto &wall : col[j]) {
            remainingWalls++;
        }
    }
    
    cout << remainingWalls << endl;
    
    return 0;
}
