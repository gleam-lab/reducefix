    vector<vector<int>> grid(h, vector<int>(w, 1));
    set<pair<ll, ll>> row_set[h], col_set[w];

    rep(i, h) rep(j, w) {
        row_set[i].insert({j, j});
        col_set[j].insert({i, i});
    }

        --r, --c;
        
        if(grid[r][c] == 1) {
            grid[r][c] = 0;
            row_set[r].erase(row_set[r].find({c, c}));
            col_set[c].erase(col_set[c].find({r, r}));
        } else {
            vector<pair<ll, ll>> to_destroy;
            auto it = row_set[r].upper_bound({c, c});
            if(it != row_set[r].end()) to_destroy.push_back(*it);
            if(it != row_set[r].begin()) to_destroy.push_back(*prev(it));

            it = col_set[c].upper_bound({r, r});
            if(it != col_set[c].end()) to_destroy.push_back(*it);
            if(it != col_set[c].begin()) to_destroy.push_back(*prev(it));

            for(auto [x, y] : to_destroy) {
                grid[x][y] = 0;
                row_set[x].erase(row_set[x].find({y, y}));
                col_set[y].erase(col_set[y].find({x, x}));
            }
        }
    }

    ll count = 0;
    rep(i, h) count += accumulate(grid[i].begin(), grid[i].end(), 0);
    cout << count << '\n';

    return 0;
}
