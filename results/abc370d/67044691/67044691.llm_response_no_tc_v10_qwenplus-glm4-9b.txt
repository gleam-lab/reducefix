#include <bits/stdc++.h>
using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> grid(h, vector<int>(w, 1)); // 1 wall in each cell initially

    for (int i = 0; i < q; i++) {
        int r, c;
        cin >> r >> c;
        r--; c--; // Convert to 0-indexed

        if (grid[r][c] == 1) { // If there's a wall at (r, c)
            grid[r][c] = 0; // Destroy it
        } else { // If no wall at (r, c), destroy walls around it
            // Check above and destroy if wall exists
            for (int i = r - 1; i >= 0 && grid[i][c] == 1; i--) {
                grid[i][c] = 0;
            }
            // Check below and destroy if wall exists
            for (int i = r + 1; i < h && grid[i][c] == 1; i++) {
                grid[i][c] = 0;
            }
            // Check left and destroy if wall exists
            for (int j = c - 1; j >= 0 && grid[r][j] == 1; j--) {
                grid[r][j] = 0;
            }
            // Check right and destroy if wall exists
            for (int j = c + 1; j < w && grid[r][j] == 1; j++) {
                grid[r][j] = 0;
            }
        }
    }

    int remaining_walls = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            remaining_walls += grid[i][j];
        }
    }

    cout << remaining_walls << endl;
}
