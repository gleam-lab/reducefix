int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true));
    ll ans = H * W;
    ll cnt = 0;
    auto erase  = [&](int r, int c){
        cnt++;
        grid[r][c] = false;
    };
    rep(itr, Q){
        int R, C;
        cin >> R >> C;
        R--; C--;
        if(grid[R][C]){
            erase(R, C);
            continue;
        }
        // Check up
        for(int i = R-1; i >= 0; --i){
            if(grid[i][C]) erase(i, C);
            else break;
        }
        // Check down
        for(int i = R+1; i < H; ++i){
            if(grid[i][C]) erase(i, C);
            else break;
        }
        // Check left
        for(int j = C-1; j >= 0; --j){
            if(grid[R][j]) erase(R, j);
            else break;
        }
        // Check right
        for(int j = C+1; j < W; ++j){
            if(grid[R][j]) erase(R, j);
            else break;
        }
    }
    cout << ans - cnt << endl;
    return 0;
}
