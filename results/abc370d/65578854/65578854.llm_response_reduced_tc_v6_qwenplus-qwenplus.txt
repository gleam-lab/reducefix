#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, s, n) for (ll i = (s); i < (ll)(n); i++)
#define all(x) x.begin(), x.end()

template<typename T> inline bool chmax(T &a, T b) { return ((a < b) ? (a = b, true) : (false)); }
template<typename T> inline bool chmin(T &a, T b) { return ((a > b) ? (a = b, true) : (false)); }

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<set<int>> row(W), col(H);
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    ll remaining = H * W;

    auto destroy = [&](int r, int c) {
        if (row[c].count(r)) {
            remaining--;
            row[c].erase(r);
            col[r].erase(c);
            return true;
        }
        return false;
    };

    rep(_, Q) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (destroy(R, C)) continue;

        // Up
        auto it = row[C].lower_bound(R);
        if (it != row[C].begin()) {
            --it;
            destroy(*it, C);
        }

        // Down
        it = row[C].lower_bound(R);
        if (it != row[C].end()) {
            destroy(*it, C);
        }

        // Left
        auto& cur_row = col[R];
        auto jt = cur_row.lower_bound(C);
        if (jt != cur_row.begin()) {
            --jt;
            destroy(R, *jt);
        }

        // Right
        jt = cur_row.lower_bound(C);
        if (jt != cur_row.end()) {
            destroy(R, *jt);
        }
    }

    cout << remaining << endl;
    return 0;
}
