#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // For each row, keep a set of columns that still have walls
    vector<set<int>> rows(H);
    // For each column, keep a set of rows that still have walls
    vector<set<int>> cols(W);

    // Initialize all walls present
    for (int r = 0; r < H; ++r) {
        for (int c = 0; c < W; ++c) {
            rows[r].insert(c);
            cols[c].insert(r);
        }
    }

    ll total_destroyed = 0;

    auto destroy = [&](int r, int c) {
        if (rows[r].count(c)) {
            rows[r].erase(c);
            cols[c].erase(r);
            total_destroyed++;
        }
    };

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        // If there is a wall at (R, C), destroy it
        if (rows[R].count(C)) {
            destroy(R, C);
            continue;
        }

        // Look up in the column C for the first wall above R
        auto it = cols[C].lower_bound(R);
        if (it != cols[C].begin()) {
            --it;
            int r = *it;
            destroy(r, C);
        }

        // Look down in the column C for the first wall below R
        it = cols[C].lower_bound(R);
        if (++it != cols[C].end()) {
            int r = *it;
            destroy(r, C);
        }

        // Look left in the row R for the first wall left of C
        it = rows[R].lower_bound(C);
        if (it != rows[R].begin()) {
            --it;
            int c = *it;
            destroy(R, c);
        }

        // Look right in the row R for the first wall right of C
        it = rows[R].lower_bound(C);
        if (++it != rows[R].end()) {
            int c = *it;
            destroy(R, c);
        }
    }

    ll total_remaining = 1LL * H * W - total_destroyed;
    cout << total_remaining << endl;

    return 0;
}
