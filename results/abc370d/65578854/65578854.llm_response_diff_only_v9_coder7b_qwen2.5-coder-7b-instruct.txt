#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, s, n) for (ll i = (s); i < (ll)(n); i++)
#define all(x) x.begin(), x.end()
template<typename T> inline bool chmax(T &a, T b) { return ((a < b) ? (a = b, true) : (false)); }
template<typename T> inline bool chmin(T &a, T b) { return ((a > b) ? (a = b, true) : (false)); }

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true));
    ll ans = H * W;

    while(Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if(grid[R][C]) {
            grid[R][C] = false;
            ans--;
            continue;
        }

        // Check and remove walls in the same row
        for(int j = 0; j < W; ++j) {
            if(grid[R][j]) {
                grid[R][j] = false;
                ans--;
                break;
            }
        }

        // Check and remove walls in the same column
        for(int i = 0; i < H; ++i) {
            if(grid[i][C]) {
                grid[i][C] = false;
                ans--;
                break;
            }
        }
    }

    cout << ans << endl;
    return 0;
}
