#include <bits/stdc++.h>
using namespace std;

struct Node {
    int value;
    set<Node*> prev;
    set<Node*> next;
};

void removeNode(Node* node) {
    if (!node->prev.empty()) {
        node->prev.rbegin()->next.erase(node->next.begin());
    }
    if (!node->next.empty()) {
        node->next.begin()->prev.erase(node->prev.begin());
    }
    node->prev.clear();
    node->next.clear();
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<vector<Node>> grid(H, vector<Node>(W));
    vector<Node*> horizontals(H);
    vector<Node*> verticals(W);

    for (int i = 0; i < H; ++i) {
        horizontals[i] = new Node{.value=i};
        horizontals[i]->next.insert(&grid[i][W-1]);
        grid[i][0].prev.insert(horizontals[i]);
    }

    for (int j = 0; j < W; ++j) {
        verticals[j] = new Node{.value=j};
        verticals[j]->next.insert(&grid[H-1][j]);
        grid[0][j].prev.insert(verticals[j]);
    }

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            grid[i][j].prev.insert(&grid[i][j-1]);
            grid[i][j].next.insert(&grid[i][j+1]);
            grid[i][j].prev.insert(&grid[i-1][j]);
            grid[i][j].next.insert(&grid[i+1][j]);
        }
    }

    int walls = H * W;
    while (Q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        Node* current = &grid[r][c];
        if (current->prev.size() == 0 && current->next.size() == 0) {
            walls--;
            removeNode(current);
        } else {
            auto it = current->prev.lower_bound(nullptr);
            if (it != current->prev.end()) {
                Node* top = *it;
                removeNode(top);
                walls--;
            }
            it = current->next.lower_bound(nullptr);
            if (it != current->next.end()) {
                Node* bottom = *it;
                removeNode(bottom);
                walls--;
            }
            it = current->prev.lower_bound(nullptr);
            if (it != current->prev.begin()) {
                --it;
                Node* left = *it;
                removeNode(left);
                walls--;
            }
            it = current->next.lower_bound(nullptr);
            if (it != current->next.begin()) {
                --it;
                Node* right = *it;
                removeNode(right);
                walls--;
            }
        }
    }

    cout << walls << '\n';
    return 0;
}
