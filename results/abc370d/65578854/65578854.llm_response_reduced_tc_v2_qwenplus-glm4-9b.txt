#include <iostream>
#include <vector>
#include <set>
#include <map>

using namespace std;

bool isWall(map<int, multiset<int>>& grid, int row, int col) {
    if (grid[row].count(col))
        return true;
    multiset<int> left, right, up, down;
    if (row > 0) up = grid[row - 1];
    if (row < grid.size() - 1) down = grid[row + 1];
    if (col > 0) left = grid[row].upper_bound(0);
    if (col < grid[row].size() - 1) right = grid[row].lower_bound(col + 1);
    
    multiset<int> allWalls;
    allWalls.insert(all(left));
    allWalls.insert(all(up));
    allWalls.insert(all(right));
    allWalls.insert(all(down));
    
    auto it = allWalls.lower_bound(row);
    if (it != allWalls.begin()) {
        auto prevIt = it - 1;
        grid[prevIt->first].erase(prevIt->second);
        allWalls.erase(prevIt);
    }
    it = allWalls.lower_bound(row);
    if (it != allWalls.end()) {
        grid[it->first].erase(it->second);
        allWalls.erase(it);
    }
    it = allWalls.lower_bound(col);
    if (it != allWalls.begin()) {
        auto prevIt = it - 1;
        grid[prevIt->first].erase(prevIt->second);
        allWalls.erase(prevIt);
    }
    it = allWalls.lower_bound(col);
    if (it != allWalls.end()) {
        grid[it->first].erase(it->second);
        allWalls.erase(it);
    }
    
    return false;
}

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    map<int, multiset<int>> grid(H);
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            grid[i].insert(j);
        }
    }
    
    for (int i = 0; i < Q; ++i) {
        int R, C;
        cin >> R >> C;
        R--; C--;
        if (isWall(grid, R, C)) {
            grid[R].erase(C);
        }
    }
    
    int remainingWalls = 0;
    for (auto& row : grid) {
        remainingWalls += row.second.size();
    }
    
    cout << remainingWalls << endl;
    
    return 0;
}

This code uses a `map` to store the columns of each row, which allows us to quickly access and modify the walls. The `multiset` data structure is used to keep the columns sorted, which is helpful for efficiently finding the walls to remove. The `isWall` function is used to determine if there is a wall at the specified position and to remove walls according to the problem statement. The main loop processes each query and updates the grid accordingly. After processing all queries, the code calculates the total number of remaining walls and outputs this value.
