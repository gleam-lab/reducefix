#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<set<int>> row_walls(H + 2); // 1-based indexing
    vector<set<int>> col_walls(W + 2); // 1-based indexing
    
    // Initialize all walls
    for (int i = 1; i <= H; i++) {
        for (int j = 1; j <= W; j++) {
            row_walls[i].insert(j);
            col_walls[j].insert(i);
        }
    }
    
    ll remaining = (ll)H * W;
    
    while (Q--) {
        int R, C;
        cin >> R >> C;
        
        if (row_walls[R].count(C)) {
            // Direct hit - remove this wall
            row_walls[R].erase(C);
            col_walls[C].erase(R);
            remaining--;
        } else {
            // Find adjacent walls in four directions
            // Left
            auto left_it = row_walls[R].lower_bound(C);
            if (left_it != row_walls[R].begin()) {
                --left_it;
                int c = *left_it;
                row_walls[R].erase(c);
                col_walls[c].erase(R);
                remaining--;
            }
            
            // Right
            auto right_it = row_walls[R].upper_bound(C);
            if (right_it != row_walls[R].end()) {
                int c = *right_it;
                row_walls[R].erase(c);
                col_walls[c].erase(R);
                remaining--;
            }
            
            // Up
            auto up_it = col_walls[C].lower_bound(R);
            if (up_it != col_walls[C].begin()) {
                --up_it;
                int r = *up_it;
                row_walls[r].erase(C);
                col_walls[C].erase(r);
                remaining--;
            }
            
            // Down
            auto down_it = col_walls[C].upper_bound(R);
            if (down_it != col_walls[C].end()) {
                int r = *down_it;
                row_walls[r].erase(C);
                col_walls[C].erase(r);
                remaining--;
            }
        }
    }
    
    cout << remaining << endl;
    
    return 0;
}
