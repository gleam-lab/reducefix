#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<set<int>> row(W), col(H);
    
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    ll remaining = H * W;

    auto destroy = [&](int r, int c) {
        if (col[r].count(c)) {
            remaining--;
            col[r].erase(c);
            row[c].erase(r);
            return true;
        }
        return false;
    };

    rep(_, Q) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (destroy(R, C)) continue;

        // Up
        auto it = row[C].lower_bound(R);
        if (it != row[C].begin()) {
            it--;
            destroy(*it, C);
        }

        // Down
        it = row[C].lower_bound(R);
        if (it != row[C].end()) {
            destroy(*it, C);
        }

        // Left
        it = col[R].lower_bound(C);
        if (it != col[R].begin()) {
            it--;
            destroy(R, *it);
        }

        // Right
        it = col[R].lower_bound(C);
        if (it != col[R].end()) {
            destroy(R, *it);
        }
    }

    cout << remaining << endl;
    return 0;
}
