#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, s, n) for (ll i = (s); i < (ll)(n); i++)

template<typename T> inline bool chmax(T &a, T b) { return ((a < b) ? (a = b, true) : (false)); }
template<typename T> inline bool chmin(T &a, T b) { return ((a > b) ? (a = b, true) : (false)); }

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<set<int>> row(W), col(H);
    
    rep(i, H) rep(j, W) {
        row[j].insert(i);
        col[i].insert(j);
    }

    ll ans = H * W;

    auto destroy = [&](int r, int c) {
        if (row[c].count(r)) {
            ans--;
            row[c].erase(r);
            col[r].erase(c);
            return true;
        }
        return false;
    };

    rep(_, Q) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (destroy(R, C)) continue;

        // Up
        auto it = row[C].lower_bound(R);
        if (it != row[C].begin()) {
            it--;
            destroy(*it, C);
        }

        // Down
        it = row[C].lower_bound(R);
        if (it != row[C].end()) {
            destroy(*it, C);
        }

        // Left
        auto jt = col[R].lower_bound(C);
        if (jt != col[R].begin()) {
            jt--;
            destroy(R, *jt);
        }

        // Right
        jt = col[R].lower_bound(C);
        if (jt != col[R].end()) {
            destroy(R, *jt);
        }
    }

    cout << ans << endl;
    return 0;
}
