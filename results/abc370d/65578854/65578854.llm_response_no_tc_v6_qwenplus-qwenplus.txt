#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, s, n) for (ll i = (s); i < (ll)(n); i++)

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, keep a sorted set of columns that still have walls
    vector<set<int>> row(H);
    // For each column, keep a sorted set of rows that still have walls
    vector<set<int>> col(W);

    // Initialize all walls present
    for (int r = 0; r < H; ++r) {
        for (int c = 0; c < W; ++c) {
            row[r].insert(c);
            col[c].insert(r);
        }
    }

    ll remaining_walls = (ll)H * W;

    auto erase = [&](int r, int c) {
        if (row[r].count(c)) {
            row[r].erase(c);
            col[c].erase(r);
            remaining_walls--;
        }
    };

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        // If there's a wall at the current position, destroy it and nothing more
        if (row[R].count(C)) {
            erase(R, C);
            continue;
        }

        // Up direction
        auto it = col[C].lower_bound(R);
        if (it != col[C].begin()) {
            --it;
            erase(*it, C);
        }

        // Down direction
        it = col[C].lower_bound(R);
        if (it != col[C].end() && *it != R) { // Skip if equal to R since we're looking for beyond R
            erase(*it, C);
        }

        // Left direction
        auto it2 = row[R].lower_bound(C);
        if (it2 != row[R].begin()) {
            --it2;
            erase(R, *it2);
        }

        // Right direction
        it2 = row[R].lower_bound(C);
        if (it2 != row[R].end() && *it2 != C) { // Skip if equal to C
            erase(R, *it2);
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
