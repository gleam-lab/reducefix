```cpp
#include <bits/stdc++.h>
using namespace std;

// Define constants
const int MAX_H = 4e5 + 5;
const int MAX_W = 4e5 + 5;

// Global variables
int H, W, Q;
vector<vector<bool>> grid(MAX_H, vector<bool>(MAX_W, true));

// Function to handle bomb placement
void placeBomb(int R, int C) {
    if (grid[R][C]) {
        grid[R][C] = false;
        return;
    }
    
    // Find the nearest walls in each direction
    for (int i = R - 1; i >= 0 && !grid[i][C]; --i) grid[i][C] = false;
    for (int i = R + 1; i < H && !grid[i][C]; ++i) grid[i][C] = false;
    for (int j = C - 1; j >= 0 && !grid[R][j]; --j) grid[R][j] = false;
    for (int j = C + 1; j < W && !grid[R][j]; ++j) grid[R][j] = false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    // Read input
    cin >> H >> W >> Q;
    for (int i = 0; i < Q; ++i) {
        int R, C;
        cin >> R >> C;
        placeBomb(R - 1, C - 1);
    }

    // Calculate and output the number of remaining walls
    int remainingWalls = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (grid[i][j]) ++remainingWalls;
        }
    }
    cout << remainingWalls << endl;

    return 0;
}
```
