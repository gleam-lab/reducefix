
```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true));

    auto destroy_walls = [&](int r, int c) {
        grid[r][c] = false;
        for (int i = 0; i < H; ++i) {
            if (grid[i][c]) break;
            grid[i][c] = false;
        }
        for (int i = 0; i < W; ++i) {
            if (grid[r][i]) break;
            grid[r][i] = false;
        }
    };

    ll remaining_walls = H * W;
    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        --R; --C;
        if (grid[R][C]) {
            destroy_walls(R, C);
        }
    }

    for (auto &row : grid) {
        for (bool wall : row) {
            if (wall) {
                remaining_walls--;
            }
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
```
