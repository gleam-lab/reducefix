#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<set<int>>> rows(H + 1, vector<set<int>>(W + 1));
    vector<vector<bool>> hasWall(H + 1, vector<bool>(W + 1, true));

    // Initialize the walls
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i][j] = set<int>{j};
            rows[i][j].insert(i);
        }
    }

    int remainingWalls = H * W; // Initially, all walls exist

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        --R; --C; // Convert to 0-based index

        if (rows[R + 1][C + 1].size() == 1) {
            // There is a wall at (R+1, C+1), just destroy it
            hasWall[R + 1][C + 1] = false;
            rows[R + 1][C + 1].erase(C + 1);
            rows[R + 1][C + 1].erase(R + 1);
            remainingWalls--;
        } else {
            // No wall at (R+1, C+1), destroy walls around it
            set<int> toDestroy;

            // Destroy up
            for (int i : rows[R + 1][C + 1]) {
                toDestroy.insert(i);
            }
            rows[R + 1][C + 1].clear();

            // Destroy down
            for (int i : rows[R][C + 1]) {
                toDestroy.insert(i);
            }
            rows[R][C + 1].clear();

            // Destroy left
            for (int i : rows[R + 1][C]) {
                toDestroy.insert(i);
            }
            rows[R + 1][C].clear();

            // Destroy right
            for (int i : rows[R + 1][C + 2]) {
                toDestroy.insert(i);
            }
            rows[R + 1][C + 2].clear();

            // Remove destroyed walls from the grid
            for (int i : toDestroy) {
                if (hasWall[i][C + 1]) {
                    remainingWalls--;
                }
                hasWall[i][C + 1] = false;
                rows[i][C + 1].clear();
            }
        }
    }

    cout << remainingWalls << '\n';
    return 0;
}
