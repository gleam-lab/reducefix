#include <iostream>
#include <vector>
#include <set>

using namespace std;

struct PairHash {
    template <class T1, class T2>
    size_t operator () (const pair<T1,T2> &pair) const {
        auto hash1 = hash<T1>{}(pair.first);
        auto hash2 = hash<T2>{}(pair.second);
        return hash1 ^ hash2;
    }
};

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> row comienzos(h + 1, vector<int>(w + 1, 0));

    vector<pair<int, int>> queries;

    // Initialize walls
    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            walls.insert({i, j});
        }
    }

    // Read queries
    for (int i = 0; i < q; ++i) {
        int r, c;
        cin >> r >> c;
        queries.push_back({r, c});
    }

    // Process queries
    for (auto &query : queries) {
        int r = query.first, c = query.second;
        auto it = walls.lower_bound({r, c});
        if (it != walls.end() && (*it).first == r && (*it).second == c) {
            walls.erase(it);
        } else {
            if (it != walls.begin() && (*prev(it)).first == r && (*prev(it)).second == c) {
                walls.erase(prev(it));
            }
            if (it != walls.end() && (*next(it)).first == r && (*next(it)).second == c) {
                walls.erase(next(it));
            }
            // Erase walls from top
            while (it != walls.begin() && (*prev(it)).second == c) {
                walls.erase(prev(it));
                --it;
            }
            // Erase walls from bottom
            while (it != walls.end() && (*next(it)).second == c) {
                walls.erase(next(it));
                // Move to the next element
                it++;
            }
        }
    }

    // Calculate the remaining number of walls
    int remaining_walls = 0;
    for (int i = 1; i <= h; ++i) {
        for (int j = 1; j <= w; ++j) {
            remaining_walls += walls.count({i, j});
        }
    }

    cout << remaining_walls << endl;

    return 0;
}
