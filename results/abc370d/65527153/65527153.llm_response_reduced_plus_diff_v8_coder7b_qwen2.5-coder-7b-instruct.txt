#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define vi(x) vector<x>

const int MAXN = 4e5 + 5;
vi(int) g[MAXN];
bool vis[MAXN];

void dfs(int u){
    vis[u] = true;
    for(auto &v : g[u]){
        if(!vis[v]) dfs(v);
    }
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);

    int n, m, q;
    cin >> n >> m >> q;

    vi(set<int>) st(n + 1), st2(m + 1);

    for(int i = 1; i <= n; ++i){
        for(int j = 1; j <= m; ++j){
            st[i].insert(j);
            st2[j].insert(i);
        }
    }

    while(q--){
        int r, c;
        cin >> r >> c;

        if(st[r].find(c) != st[r].end()){
            st[r].erase(c);
            st2[c].erase(r);
        }else{
            auto it = st[r].lower_bound(c);
            if(it != st[r].end()){
                st[r].erase(it);
                st2[*it].erase(r);
            }

            it = st[r].upper_bound(c);
            if(it != st[r].begin()){
                --it;
                st[r].erase(it);
                st2[*it].erase(r);
            }

            it = st2[c].lower_bound(r);
            if(it != st2[c].end()){
                st2[c].erase(it);
                st[*it].erase(c);
            }

            it = st2[c].upper_bound(r);
            if(it != st2[c].begin()){
                --it;
                st2[c].erase(it);
                st[*it].erase(c);
            }
        }
    }

    int res = 0;
    for(int i = 1; i <= n; ++i){
        if(!st[i].empty()){
            res += st[i].size();
            dfs(i);
        }
    }

    cout << res << '\n';

    return 0;
}
