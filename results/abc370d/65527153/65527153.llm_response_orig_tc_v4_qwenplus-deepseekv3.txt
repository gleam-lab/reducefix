### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<set<int>> row_walls(H + 1);
    vector<set<int>> col_walls(W + 1);
    
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row_walls[i].insert(j);
            col_walls[j].insert(i);
        }
    }
    
    while (Q--) {
        int R, C;
        cin >> R >> C;
        
        if (row_walls[R].count(C)) {
            row_walls[R].erase(C);
            col_walls[C].erase(R);
            continue;
        }
        
        // Check up
        auto it_up = col_walls[C].lower_bound(R);
        if (it_up != col_walls[C].begin()) {
            --it_up;
            int r = *it_up;
            row_walls[r].erase(C);
            col_walls[C].erase(r);
        }
        
        // Check down
        auto it_down = col_walls[C].upper_bound(R);
        if (it_down != col_walls[C].end()) {
            int r = *it_down;
            row_walls[r].erase(C);
            col_walls[C].erase(r);
        }
        
        // Check left
        auto it_left = row_walls[R].lower_bound(C);
        if (it_left != row_walls[R].begin()) {
            --it_left;
            int c = *it_left;
            row_walls[R].erase(c);
            col_walls[c].erase(R);
        }
        
        // Check right
        auto it_right = row_walls[R].upper_bound(C);
        if (it_right != row_walls[R].end()) {
            int c = *it_right;
            row_walls[R].erase(c);
            col_walls[c].erase(R);
        }
    }
    
    long long ans = 0;
    for (int i = 1; i <= H; ++i) {
        ans += row_walls[i].size();
    }
    
    cout << ans << "\n";
    
    return 0;
}
```

### Explanation
- **Initialization**: We initialize two vectors of sets, `row_walls` and `col_walls`, to keep track of walls in each row and column. Each cell (i, j) is added to both `row_walls[i]` and `col_walls[j]`.
- **Processing Queries**: For each query (R, C):
  - If the cell (R, C) has a wall, it is removed from both `row_walls[R]` and `col_walls[C]`.
  - If the cell (R, C) has no wall, we search for the nearest walls in each direction:
    - **Up**: The first wall above (R, C) in the same column.
    - **Down**: The first wall below (R, C) in the same column.
    - **Left**: The first wall to the left of (R, C) in the same row.
    - **Right**: The first wall to the right of (R, C) in the same row.
  - Each found wall is removed from both sets.
- **Counting Remaining Walls**: After processing all queries, we sum the sizes of all sets in `row_walls` to get the total number of remaining walls.

This approach efficiently handles each query using binary search operations on sets, ensuring optimal performance even for large input sizes.
