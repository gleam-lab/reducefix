```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, maintain a set of columns that still have walls
    vector<set<int>> rows(H + 1);
    // For each column, maintain a set of rows that still have walls
    vector<set<int>> cols(W + 1);

    // Initialize all walls
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // Check if the current cell has a wall
        auto it = rows[r].find(c);
        if (it != rows[r].end()) {
            // Wall exists here, destroy it
            rows[r].erase(it);
            cols[c].erase(r);
        } else {
            // Destroy in four directions

            // Right (increasing column)
            auto right = rows[r].lower_bound(c);
            if (right != rows[r].end()) {
                int col = *right;
                rows[r].erase(col);
                cols[col].erase(r);
            }

            // Left (decreasing column)
            right = rows[r].lower_bound(c);
            if (right != rows[r].begin()) {
                --right;
                int col = *right;
                rows[r].erase(col);
                cols[col].erase(r);
            }

            // Down (increasing row)
            auto down = cols[c].lower_bound(r);
            if (down != cols[c].end()) {
                int row = *down;
                cols[c].erase(down);
                rows[row].erase(c);
            }

            // Up (decreasing row)
            down = cols[c].lower_bound(r);
            if (down != cols[c].begin()) {
                --down;
                int row = *down;
                cols[c].erase(down);
                rows[row].erase(c);
            }
        }
    }

    // Count remaining walls
    ll remaining = 0;
    for (int i = 1; i <= H; ++i) {
        remaining += rows[i].size();
    }
    cout << remaining << "\n";

    return 0;
}
```
