#include <bits/stdc++.h>
using namespace std;
using ll = long long;

template<typename T>
using grid = vector<vector<T>>;

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, keep a sorted list of columns that still have walls
    vector<set<int>> row_walls(H + 1);
    // For each column, keep a sorted list of rows that still have walls
    vector<set<int>> col_walls(W + 1);

    // Initialize all walls present
    for (int r = 1; r <= H; ++r) {
        for (int c = 1; c <= W; ++c) {
            row_walls[r].insert(c);
            col_walls[c].insert(r);
        }
    }

    auto destroy_wall = [&](int r, int c) {
        if (row_walls[r].erase(c)) {
            col_walls[c].erase(r);
        }
    };

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // If there's a wall at the bomb position, just destroy it
        if (row_walls[r].count(c)) {
            destroy_wall(r, c);
            continue;
        }

        // Look right in the same row
        auto it = row_walls[r].lower_bound(c);
        if (it != row_walls[r].end()) {
            destroy_wall(r, *it);
        }

        // Look left in the same row
        if (it != row_walls[r].begin()) {
            if (it == row_walls[r].end()) it = row_walls[r].end();
            --it;
            if (*it < c) {
                destroy_wall(r, *it);
            }
        }

        // Look down in the same column
        auto jt = col_walls[c].lower_bound(r);
        if (jt != col_walls[c].end()) {
            destroy_wall(*jt, c);
        }

        // Look up in the same column
        if (jt != col_walls[c].begin()) {
            if (jt == col_walls[c].end()) jt = col_walls[c].end();
            --jt;
            if (*jt < r) {
                destroy_wall(*jt, c);
            }
        }
    }

    ll remaining = 0;
    for (int i = 1; i <= H; ++i) {
        remaining += row_walls[i].size();
    }
    cout << remaining << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
