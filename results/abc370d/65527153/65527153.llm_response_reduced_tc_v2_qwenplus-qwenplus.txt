#include <bits/stdc++.h>
using namespace std;
using ll = long long;

template<typename T>
using grid = vector<vector<T>>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, keep track of walls in that row
    vector<set<int>> rows(H + 1);
    // For each column, keep track of walls in that column
    vector<set<int>> cols(W + 1);

    // Initialize all walls
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // If there's a wall at (r, c), remove it
        if (rows[r].count(c)) {
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Look right
            auto it = rows[r].lower_bound(c);
            if (it != rows[r].end()) {
                int y = *it;
                rows[r].erase(y);
                cols[y].erase(r);
            }

            // Look left
            it = rows[r].lower_bound(c);
            if (it != rows[r].begin()) {
                --it;
                int y = *it;
                rows[r].erase(y);
                cols[y].erase(r);
            }

            // Look down
            auto cit = cols[c].lower_bound(r);
            if (cit != cols[c].end()) {
                int x = *cit;
                cols[c].erase(x);
                rows[x].erase(c);
            }

            // Look up
            cit = cols[c].lower_bound(r);
            if (cit != cols[c].begin()) {
                --cit;
                int x = *cit;
                cols[c].erase(x);
                rows[x].erase(c);
            }
        }
    }

    // Count remaining walls
    ll ans = 0;
    for (int i = 1; i <= H; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";

    return 0;
}
