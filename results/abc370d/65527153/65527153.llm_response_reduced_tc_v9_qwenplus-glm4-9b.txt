#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

using ll = long long;

const int N = 4e5 + 10;

int h, w, q;
vector<vector<set<int>>> row_walls(N), col_walls(N);

void destroy_wall(int r, int c) {
    // Destroy the wall directly at the given position
    row_walls[r][c].erase(c);
    col_walls[c][r].erase(r);
    
    // Destroy the walls above and below the given position if needed
    auto it = row_walls[r].lower_bound(c);
    if (it != row_walls[r].end()) {
        row_walls[r][*it].erase(c);
    }
    it = row_walls[r].upper_bound(c);
    if (it != row_walls[r].begin()) {
        it--;
        row_walls[r][*it].erase(c);
    }
    
    // Destroy the walls to the left and right of the given position if needed
    it = col_walls[c].lower_bound(r);
    if (it != col_walls[c].end()) {
        col_walls[c][*it].erase(r);
    }
    it = col_walls[c].upper_bound(r);
    if (it != col_walls[c].begin()) {
        it--;
        col_walls[c][*it].erase(r);
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> h >> w >> q;

    for (int i = 1; i <= h; i++) {
        for (int j = 1; j <= w; j++) {
            row_walls[i].push_back({j});
            col_walls[j].push_back({i});
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        destroy_wall(r, c);
    }

    ll remaining_walls = 0;
    for (int i = 1; i <= h; i++) {
        for (int j = 1; j <= w; j++) {
            remaining_walls += row_walls[i][j].size();
        }
    }

    cout << remaining_walls / 2 << '\n'; // Each destroyed pair contributes 2 to the total count

    return 0;
}
