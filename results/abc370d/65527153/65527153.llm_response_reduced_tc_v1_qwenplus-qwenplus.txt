#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using PII = pair<int, int>;

#define x first
#define y second

void solve()
{
    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, keep a set of columns that still have walls
    vector<set<int>> rowWalls(H + 1);
    // For each column, keep a set of rows that still have walls
    vector<set<int>> colWalls(W + 1);

    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rowWalls[i].insert(j);
            colWalls[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // If the cell has a wall, remove it
        if (rowWalls[r].count(c)) {
            rowWalls[r].erase(c);
            colWalls[c].erase(r);
        } else {
            // Otherwise, destroy the nearest walls in all four directions

            // Right in row
            auto itR = rowWalls[r].lower_bound(c);
            if (itR != rowWalls[r].end()) {
                int toErase = *itR;
                rowWalls[r].erase(toErase);
                colWalls[toErase].erase(r);
            }

            // Left in row
            itR = rowWalls[r].lower_bound(c);
            if (itR != rowWalls[r].begin()) {
                --itR;
                int toErase = *itR;
                rowWalls[r].erase(toErase);
                colWalls[toErase].erase(r);
            }

            // Down in column
            auto itC = colWalls[c].lower_bound(r);
            if (itC != colWalls[c].end()) {
                int toErase = *itC;
                colWalls[c].erase(toErase);
                rowWalls[toErase].erase(c);
            }

            // Up in column
            itC = colWalls[c].lower_bound(r);
            if (itC != colWalls[c].begin()) {
                --itC;
                int toErase = *itC;
                colWalls[c].erase(toErase);
                rowWalls[toErase].erase(c);
            }
        }
    }

    ll ans = 0;
    for (int i = 1; i <= H; ++i) {
        ans += rowWalls[i].size();
    }
    cout << ans << "\n";
}

int main()
{
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    solve();
    return 0;
}
