#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using PII = pair<int, int>;
#define x first
#define y second
#define all(x) x.begin(),x.end()

const int N = 400020, INF = 0x3f3f3f3f;
int n, m;

void solve()
{
    int q;
    cin >> n >> m >> q;
    vector<vector<bool>> grid(n + 1, vector<bool>(m + 1, true));

    while (q--) {
        int r, c; cin >> r >> c;
        if (grid[r][c]) {
            grid[r][c] = false;
            // Check up
            for (int i = r - 1; i >= 1; --i) {
                if (!grid[i][c]) break;
                grid[i][c] = false;
            }
            // Check down
            for (int i = r + 1; i <= n; ++i) {
                if (!grid[i][c]) break;
                grid[i][c] = false;
            }
            // Check left
            for (int j = c - 1; j >= 1; --j) {
                if (!grid[r][j]) break;
                grid[r][j] = false;
            }
            // Check right
            for (int j = c + 1; j <= m; ++j) {
                if (!grid[r][j]) break;
                grid[r][j] = false;
            }
        }
    }

    ll ans = 0;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            if (grid[i][j])
                ans++;

    cout << ans << "\n";
}

signed main()
{
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int T = 1;
    while (T--) solve();
    return 0;
}

This solution uses a straightforward approach without utilizing any data structures like sets. It iterates through the queries and checks adjacent cells to determine which walls should be destroyed. The time complexity is O(Q * max(H, W)), which is efficient enough given the problem constraints.
