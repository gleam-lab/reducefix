int n, m;

void solve()
{
    int q;
    cin >> n >> m >> q;
    vector<vector<bool>> grid(n, vector<bool>(m, true));
    
    auto destroyWalls = [&](int r, int c) -> void {
        if (grid[r][c]) {
            grid[r][c] = false;
            return;
        }

        // Destroy walls above
        for (int i = r - 1; i >= 0; --i) {
            if (grid[i][c]) {
                grid[i][c] = false;
                break;
            }
        }

        // Destroy walls below
        for (int i = r + 1; i < n; ++i) {
            if (grid[i][c]) {
                grid[i][c] = false;
                break;
            }
        }

        // Destroy walls to the left
        for (int j = c - 1; j >= 0; --j) {
            if (grid[r][j]) {
                grid[r][j] = false;
                break;
            }
        }

        // Destroy walls to the right
        for (int j = c + 1; j < m; ++j) {
            if (grid[r][j]) {
                grid[r][j] = false;
                break;
            }
        }
    };

    while (q--) {
        int r, c; cin >> r >> c;
        destroyWalls(r - 1, c - 1); // Adjusting for zero-based indexing
    }

    ll ans = 0;
    for (auto &row : grid) {
        for (bool cell : row) {
            if (cell) ++ans;
        }
    }
    cout << ans << "\n";
}
