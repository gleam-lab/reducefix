#include <bits/stdc++.h>
using namespace std;
using ll = long long;

template<typename T>
void print(T val) {
    cout << val << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<set<int>> row(W + 1), col(H + 1);
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        auto it = col[r].find(c);
        if (it != col[r].end()) {
            // Bomb destroys itself
            col[r].erase(it);
            row[c].erase(r);
            continue;
        }

        // Up direction
        it = col[r].lower_bound(c);
        if (it != col[r].begin()) {
            --it;
            int left_c = *it;
            row[left_c].erase(r);
            col[r].erase(left_c);
        }

        // Down direction
        it = col[r].lower_bound(c);
        if (it != col[r].end()) {
            int right_c = *it;
            row[right_c].erase(r);
            col[r].erase(right_c);
        }

        // Left direction
        auto rit = row[c].lower_bound(r);
        if (rit != row[c].begin()) {
            --rit;
            int up_r = *rit;
            col[up_r].erase(c);
            row[c].erase(up_r);
        }

        // Right direction
        rit = row[c].lower_bound(r);
        if (rit != row[c].end()) {
            int down_r = *rit;
            col[down_r].erase(c);
            row[c].erase(down_r);
        }
    }

    ll remaining = 0;
    for (int i = 1; i <= H; ++i)
        remaining += col[i].size();
    print(remaining);

    return 0;
}
