
```cpp
#include <iostream>
#include <set>
#include <vector>
#include <algorithm>

using namespace std;

int H, W, Q;
int grid[4010][4010]; // Using a larger size to avoid boundary issues

void process(int r, int c) {
    // Check and remove the wall at (r, c) if it exists
    if (grid[r][c] == 1) {
        grid[r][c] = 0;
    } else {
        // Remove walls to the left, right, up, and down
        for (int i = c; i > 0; i--) {
            if (grid[r][i] == 1) {
                grid[r][i] = 0;
            } else {
                break;
            }
        }
        for (int i = c; i < W; i++) {
            if (grid[r][i] == 1) {
                grid[r][i] = 0;
            } else {
                break;
            }
        }
        for (int i = r; i > 0; i--) {
            if (grid[i][c] == 1) {
                grid[i][c] = 0;
            } else {
                break;
            }
        }
        for (int i = r; i < H; i++) {
            if (grid[i][c] == 1) {
                grid[i][c] = 0;
            } else {
                break;
            }
        }
    }
}

int main() {
    cin >> H >> W >> Q;
    for (int i = 1; i <= H; i++) {
        for (int j = 1; j <= W; j++) {
            grid[i][j] = 1; // Initialize all cells with a wall
        }
    }

    for (int i = 0; i < Q; i++) {
        int r, c;
        cin >> r >> c;
        process(r, c);
    }

    // Count the number of remaining walls
    int remainingWalls = 0;
    for (int i = 1; i <= H; i++) {
        for (int j = 1; j <= W; j++) {
            remainingWalls += grid[i][j];
        }
    }
    cout << remainingWalls << endl;

    return 0;
}
```

This code initializes a grid and processes each query by either removing the wall at the specified cell or removing walls to the left, right, up, and down if no wall exists at the specified cell. Finally, it counts and prints the number of remaining walls. The code uses a 2D array `grid` to represent the grid, with `grid[i][j]` indicating whether there is a wall at cell (i, j).
