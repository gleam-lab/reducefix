const int N = 200020, INF = 0x3f3f3f3f;
int n, m;

void solve()
{
    int q;
    cin >> n >> m >> q;
    vector<set<int>> st1(n + 1), st2(m + 1);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            st1[i].emplace(j);
            st2[j].emplace(i);
        }
    }

    auto erase = [&](int r, int c) -> void {
        st1[r].erase(c);
        st2[c].erase(r);
    };

    while (q--) {
        int r, c; cin >> r >> c;
        if (st1[r].count(c)) {
            erase(r, c);
            continue;
        }

        // Correctly identify the closest wall to the right
        auto it = st1[r].upper_bound(c);
        if (it != st1[r].end()) {
            erase(r, *it);
        } else {
            // If no wall to the right, check above
            it = st2[c].upper_bound(r);
            if (it != st2[c].end()) {
                erase(*it, c);
            }
        }
    }

    ll ans = 0;
    for (int i = 1; i <= n; i++) ans += st1[i].size();
    cout << ans << "\n";
}

    int T = 1;
    while (T--) solve();

}

### Explanation of Fixes
1. **Corrected the logic for destroying walls**:
   - The original code was incorrectly checking for walls to the left and right without considering the upper bounds correctly.
   - I updated the code to use `upper_bound` which returns the first element greater than the specified value. This ensures we get the correct wall to destroy if any exists.

2. **Edge Case Handling**:
   - If no wall is found to the right, the code now checks above to ensure that the correct wall is destroyed or no wall is destroyed if none exist.

3. **Efficiency**:
   - The use of `std::set` allows for efficient insertion and deletion operations (`O(log n)` time complexity), making the solution feasible within the constraints.
