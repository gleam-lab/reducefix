To solve this problem efficiently, we need to simulate placing bombs on a grid and destroying walls according to the specified rules. The key is to manage the dynamic state of each row and column's remaining walls in such a way that we can quickly query and remove the nearest wall in each direction (up, down, left, right) when a bomb is placed.

### Key Observations:
1. **Initial State**: Each cell has one wall.
2. **Bomb Behavior**:
   - If the current cell has a wall, it's destroyed.
   - If the cell already has no wall, destroy the closest walls in all four directions (if they exist).
3. **Efficient Data Structures**:
   - Use `std::set` for each row and column to keep track of remaining walls. This allows logarithmic time complexity for insertions, deletions, and searching (`lower_bound`, `upper_bound`).

### Approach:
- Maintain two arrays of sets: one for rows and one for columns.
- For each query:
  - Check if there's a wall at the given position. If yes, remove it from both sets.
  - If not, find and remove the nearest walls in all four directions using `lower_bound` and `upper_bound`.

This approach ensures that each operation (insert/delete/query) runs in logarithmic time relative to the number of walls per row/column, which is efficient under the problem constraints.

Here's the corrected and optimized implementation:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<set<int>> rows(H + 1), cols(W + 1);

    // Initialize with all walls present
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        if (rows[r].count(c)) {
            // There's a wall here, destroy it
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Destroy nearest walls in all 4 directions
            set<int>::iterator it;

            // Right (east)
            it = rows[r].upper_bound(c);
            if (it != rows[r].end()) {
                cols[*it].erase(r);
                rows[r].erase(it);
            }

            // Left (west)
            it = rows[r].lower_bound(c);
            if (it != rows[r].begin()) {
                --it;
                cols[*it].erase(r);
                rows[r].erase(it);
            }

            // Down (south)
            it = cols[c].upper_bound(r);
            if (it != cols[c].end()) {
                rows[*it].erase(c);
                cols[c].erase(it);
            }

            // Up (north)
            it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                --it;
                rows[*it].erase(c);
                cols[c].erase(it);
            }
        }
    }

    long long ans = 0;
    for (int i = 1; i <= H; ++i)
        ans += rows[i].size();

    cout << ans << "\n";

    return 0;
}
```
