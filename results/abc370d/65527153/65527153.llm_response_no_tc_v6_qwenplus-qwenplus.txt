#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pii = pair<int, int>;
#define x first
#define y second
#define all(x) (x).begin(), (x).end()
#define rep(i, a, b) for (int i = (a); i < (b); ++i)

const int INF = 0x3f3f3f3f;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int H, W, Q;
    cin >> H >> W >> Q;

    // For each row, maintain a set of columns that still have walls
    vector<set<int>> rowWalls(H + 1);
    // For each column, maintain a set of rows that still have walls
    vector<set<int>> colWalls(W + 1);

    // Initialize all walls
    for (int r = 1; r <= H; ++r) {
        for (int c = 1; c <= W; ++c) {
            rowWalls[r].insert(c);
            colWalls[c].insert(r);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;

        // If there's a wall at (r,c), destroy it
        if (rowWalls[r].count(c)) {
            rowWalls[r].erase(c);
            colWalls[c].erase(r);
        } else {
            // Else, destroy the nearest walls in each direction

            // Right (next greater column)
            auto it = rowWalls[r].lower_bound(c);
            if (it != rowWalls[r].end()) {
                int c2 = *it;
                rowWalls[r].erase(c2);
                colWalls[c2].erase(r);
            }

            // Left (next smaller column)
            it = rowWalls[r].lower_bound(c);
            if (it != rowWalls[r].begin()) {
                --it;
                int c2 = *it;
                rowWalls[r].erase(c2);
                colWalls[c2].erase(r);
            }

            // Down (next greater row)
            auto jt = colWalls[c].lower_bound(r);
            if (jt != colWalls[c].end()) {
                int r2 = *jt;
                colWalls[c].erase(r2);
                rowWalls[r2].erase(c);
            }

            // Up (next smaller row)
            jt = colWalls[c].lower_bound(r);
            if (jt != colWalls[c].begin()) {
                --jt;
                int r2 = *jt;
                colWalls[c].erase(r2);
                rowWalls[r2].erase(c);
            }
        }
    }

    // Count remaining walls
    ll ans = 0;
    for (int r = 1; r <= H; ++r) {
        ans += rowWalls[r].size();
    }
    cout << ans << "\n";

    return 0;
}
