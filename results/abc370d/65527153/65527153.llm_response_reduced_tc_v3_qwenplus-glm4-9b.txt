#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

struct SegmentTree {
    vector<vector<int>> tree;
    vector<vector<int>> lazy;
    int n;

    SegmentTree(int size) : n(size), tree(4 * size), lazy(4 * size) {}

    void updateRange(int l, int r, int start, int end, int node, int val) {
        if (lazy[node] != 0) {
            tree[node] += lazy[node];
            if (start != end) {
                lazy[node * 2] += lazy[node];
                lazy[node * 2 + 1] += lazy[node];
            }
            lazy[node] = 0;
        }
        if (start > r || end < l) return;
        if (start >= l && end <= r) {
            tree[node] += val;
            if (start != end) {
                lazy[node * 2] += val;
                lazy[node * 2 + 1] += val;
            }
            return;
        }
        int mid = (start + end) / 2;
        updateRange(l, r, start, mid, node * 2, val);
        updateRange(l, r, mid + 1, end, node * 2 + 1, val);
        tree[node] = tree[node * 2] + tree[node * 2 + 1];
    }

    void updateRange(int l, int r, int node, int start, int end) {
        if (lazy[node] != 0) {
            tree[node] += lazy[node];
            if (start != end) {
                lazy[node * 2] += lazy[node];
                lazy[node * 2 + 1] += lazy[node];
            }
            lazy[node] = 0;
        }
        if (start > r || end < l) return;
        if (start >= l && end <= r) {
            if (tree[node] == 0) return; // No wall to destroy
            tree[node] = 0; // Destroy all walls in this segment
            if (start != end) {
                lazy[node * 2] = 1;
                lazy[node * 2 + 1] = 1;
            }
            return;
        }
        int mid = (start + end) / 2;
        updateRange(l, r, node * 2, start, mid);
        updateRange(l, r, node * 2 + 1, mid + 1, end);
        tree[node] = tree[node * 2] + tree[node * 2 + 1];
    }

    int query(int l, int r, int node, int start, int end) {
        if (start > r || end < l) return 0;
        if (lazy[node] != 0) {
            tree[node] += lazy[node];
            if (start != end) {
                lazy[node * 2] += lazy[node];
                lazy[node * 2 + 1] += lazy[node];
            }
            lazy[node] = 0;
        }
        if (start >= l && end <= r) return tree[node];
        int mid = (start + end) / 2;
        int p = query(l, r, node * 2, start, mid);
        int q = query(l, r, node * 2 + 1, mid + 1, end);
        return p + q;
    }
};

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    SegmentTree tree(W);
    for (int i = 0; i < H; ++i) {
        int numWalls;
        cin >> numWalls;
        for (int j = 0; j < numWalls; ++j) {
            int r, c;
            cin >> r >> c;
            tree.updateRange(1, W, c, r, H);
        }
    }
    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        if (tree.query(1, W, c, r, H) > 0) {
            tree.updateRange(1, W, c, r, H);
        }
    }
    cout << tree.query(1, W, 1, 1, W) << endl;
    return 0;
}
