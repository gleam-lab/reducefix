#include <iostream>
#include <vector>
#include <set>
using namespace std;

const int MAXN = 400010;
int N, M, Q;
vector<vector<set<int>>> rowWalls(MAXN), colWalls(MAXN);
vector<pair<int, int>> queries;

int remainingWalls() {
    int remaining = 0;
    for (int i = 1; i <= N; ++i) {
        remaining += rowWalls[i].size();
        for (auto w : colWalls[i]) {
            remaining += rowWalls[w].size();
        }
    }
    return remaining;
}

void destroyWalls(int r, int c) {
    // Remove wall at (r, c) if it exists
    if (!rowWalls[r].erase(c)) {
        return; // Wall did not exist
    }

    // Find and remove all walls in the row
    for (auto w : colWalls[c]) {
        rowWalls[w].erase(r);
    }

    // Find and remove all walls in the column
    for (auto w : rowWalls[r]) {
        colWalls[w].erase(c);
    }
}

void adjustForDeletion(int r, int c) {
    // Remove wall at (r, c) from row and column
    rowWalls[r].erase(c);
    colWalls[c].erase(r);

    // Check for and remove walls above and below in the column
    int above = r - 1, below = r + 1;
    while (above > 0 || below <= N) {
        if (above > 0 && !rowWalls[above].count(c)) {
            rowWalls[above].insert(c);
            colWalls[c].insert(above);
        }
        if (below <= N && !rowWalls[below].count(c)) {
            rowWalls[below].insert(c);
            colWalls[c].insert(below);
        }
        above--;
        below++;
    }
}

int main() {
    cin >> N >> M >> Q;
    for (int i = 1; i <= Q; ++i) {
        int r, c;
        cin >> r >> c;
        queries.emplace_back(r, c);
    }
    // Process queries in reverse so we can easily adjust for walls being destroyed
    for (auto it = queries.rbegin(); it != queries.rend(); ++it) {
        int r = it->first;
        int c = it->second;
        destroyWalls(r, c);
        // Adjust for walls that should be removed
        adjustForDeletion(r, c);
    }
    // Output the number of remaining walls
    cout << remainingWalls() << endl;
    return 0;
}

The `destroyWalls` function removes the wall at a specific cell and checks if there are any walls above and below that should be removed. The `adjustForDeletion` function ensures that walls above and below the removed wall are adjusted to maintain the integrity of the grid's wall structure. The final count of remaining walls is correctly computed by the `remainingWalls` function.
