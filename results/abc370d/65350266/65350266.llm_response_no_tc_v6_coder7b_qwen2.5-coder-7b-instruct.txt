```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    
    rep(qi,q) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if(grid[r][c]){
            grid[r][c] = false;
        }else{
            // Destroy walls above
            for(int i=r-1; i>=0 && !grid[i][c]; --i){
                grid[i][c] = false;
            }
            
            // Destroy walls below
            for(int i=r+1; i<h && !grid[i][c]; ++i){
                grid[i][c] = false;
            }
            
            // Destroy walls left
            for(int j=c-1; j>=0 && !grid[r][j]; --j){
                grid[r][j] = false;
            }
            
            // Destroy walls right
            for(int j=c+1; j<w && !grid[r][j]; ++j){
                grid[r][j] = false;
            }
        }
    }

    int ans = 0;
    rep(i,h){
        rep(j,w){
            if(grid[i][j]) ++ans;
        }
    }
    cout << ans << "\n";
    return 0;
}
```
This solution uses a boolean matrix to represent whether each cell contains a wall or not. It iterates over the queries and destroys the appropriate walls based on the rules provided. Finally, it counts the remaining walls and outputs this count as the answer. This approach is efficient and straightforward, adhering to the problem constraints.
