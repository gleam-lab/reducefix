```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    
    rep(i,h) rep(j,w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }
    
    rep(qi,q) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (tate[c].count(r)) {
            // Wall exists at the cell: destroy it
            tate[c].erase(r);
            yoko[r].erase(c);
        } else {
            // Look up direction
            auto tate_up = tate[c].lower_bound(r);
            if (tate_up != tate[c].begin()) {
                tate_up--;
                int u = *tate_up;
                tate[c].erase(u);
                yoko[u].erase(c);
            }

            // Look down direction
            auto tate_dn = tate[c].lower_bound(r);
            if (tate_dn != tate[c].end()) {
                int d = *tate_dn;
                tate[c].erase(d);
                yoko[d].erase(c);
            }

            // Look left direction
            auto yoko_lt = yoko[r].lower_bound(c);
            if (yoko_lt != yoko[r].begin()) {
                yoko_lt--;
                int l = *yoko_lt;
                yoko[r].erase(l);
                tate[l].erase(r);
            }

            // Look right direction
            auto yoko_rt = yoko[r].lower_bound(c);
            if (yoko_rt != yoko[r].end()) {
                int ri = *yoko_rt;
                yoko[r].erase(ri);
                tate[ri].erase(r);
            }
        }
    }

    int ans = 0;
    rep(i, w) {
        ans += tate[i].size();
    }
    cout << ans << "\n";
    return 0;
}
```
