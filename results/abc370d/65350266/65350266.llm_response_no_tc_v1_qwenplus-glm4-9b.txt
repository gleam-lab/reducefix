#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> yoko(h), tate(w);
    vector<vector<bool>> removed(h, vector<bool>(w, false));

    rep(i, h) {
        rep(j, w) {
            yoko[i].insert(j);
            tate[j].insert(i);
        }
    }

    rep(q) {
        int r, c;
        cin >> r >> c;
        --r; --c;
        if (!removed[r][c]) {
            removed[r][c] = true;
            // Remove wall at (r, c)
            auto it_yoko = yoko[r].lower_bound(c);
            while (it_yoko != yoko[r].end() && removed[*it_yoko][r]) it_yoko++;
            if (it_yoko != yoko[r].end()) {
                yoko[r].erase(it_yoko);
                tate[*it_yoko].erase(r);
            }
            auto it_tate = tate[c].lower_bound(r);
            while (it_tate != tate[c].end() && removed[r][*it_tate]) it_tate++;
            if (it_tate != tate[c].end()) {
                tate[c].erase(it_tate);
                yoko[*it_tate].erase(c);
            }
        }
    }

    int remaining_walls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            remaining_walls += !removed[i][j] + (yoko[i][j] > j) + (tate[c][i] > i);
        }
    }

    cout << remaining_walls << "\n";
    return 0;
}
