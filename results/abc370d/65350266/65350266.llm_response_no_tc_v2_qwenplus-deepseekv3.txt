using namespace std;

int main() {

    int h, w, q;
    cin >> h >> w >> q;

    vector<set<int>> rows(h), cols(w);
    rep(i, h) rep(j, w) {
        rows[i].insert(j);
        cols[j].insert(i);
    }

    int ans = h * w;

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (rows[r].count(c)) {
            rows[r].erase(c);
            cols[c].erase(r);
            ans--;
        } else {
            // Check up
            auto it_up = cols[c].lower_bound(r);
            if (it_up != cols[c].begin()) {
                it_up--;
                int i = *it_up;
                rows[i].erase(c);
                cols[c].erase(i);

            }

            // Check down
            auto it_down = cols[c].upper_bound(r);
            if (it_down != cols[c].end()) {
                int i = *it_down;
                rows[i].erase(c);

                ans--;
            }

            // Check left
            auto it_left = rows[r].lower_bound(c);
            if (it_left != rows[r].begin()) {
                it_left--;
                int j = *it_left;
                rows[r].erase(j);

### Explanation
1. **Initialization**: We initialize two sets for each row and column to keep track of the walls. Each cell (i, j) is initially present in `rows[i]` and `cols[j]`.
