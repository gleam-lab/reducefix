#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<int> walls(h * w, 1);
    
    for (int i = 0; i < q; ++i) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        int index = r * w + c;
        
        // Check if there's a wall at (r, c)
        if (walls[index] == 1) {
            walls[index] = 0; // Destroy the wall
        } else {
            // Find the first wall to destroy when no wall exists at (r, c)
            // Check up
            for (int i = r - 1; i >= 0; --i) {
                if (walls[i * w + c] == 1) {
                    walls[i * w + c] = 0;
                    break;
                }
            }
            // Check down
            for (int i = r + 1; i < h; ++i) {
                if (walls[i * w + c] == 1) {
                    walls[i * w + c] = 0;
                    break;
                }
            }
            // Check left
            for (int j = c - 1; j >= 0; --j) {
                if (walls[r * w + j] == 1) {
                    walls[r * w + j] = 0;
                    break;
                }
            }
            // Check right
            for (int j = c + 1; j < w; ++j) {
                if (walls[r * w + j] == 1) {
                    walls[r * w + j] = 0;
                    break;
                }
            }
        }
    }
    
    // Count the remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < h * w; ++i) {
        if (walls[i] == 1) {
            remaining_walls++;
        }
    }
    
    cout << remaining_walls << endl;
    return 0;
}
