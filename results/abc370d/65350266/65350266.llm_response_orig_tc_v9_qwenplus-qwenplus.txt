#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (n); i++)
using ll = long long;
using pii = pair<int, int>;

const int MAX_H = 400010;
const int MAX_W = 400010;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Use sets to keep track of wall positions in each row and column
    vector<set<int>> rows(H), cols(W);
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there's a wall at the current position
        if (rows[r].count(c)) {
            // Destroy this wall
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Try to destroy walls in all four directions

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                it--;
                int pos = *it;
                rows[pos].erase(c);
                cols[c].erase(pos);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int pos = *it;
                rows[pos].erase(c);
                cols[c].erase(pos);
            }

            // Left
            auto jt = rows[r].lower_bound(c);
            if (jt != rows[r].begin()) {
                jt--;
                int pos = *jt;
                cols[pos].erase(r);
                rows[r].erase(pos);
            }

            // Right
            jt = rows[r].lower_bound(c);
            if (jt != rows[r].end()) {
                int pos = *jt;
                cols[pos].erase(r);
                rows[r].erase(pos);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < H; ++i)
        ans += rows[i].size();
    cout << ans << endl;

    return 0;
}
