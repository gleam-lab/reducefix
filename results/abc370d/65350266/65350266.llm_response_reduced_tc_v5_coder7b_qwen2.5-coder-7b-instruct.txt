#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    rep(i, h) rep(j, w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--, c--;

        // Check if there's already a wall at (r, c)
        if (tate[c].count(r)) {
            tate[c].erase(r);
            yoko[r].erase(c);
        } else {
            // Find the nearest wall above
            auto it_up = tate[c].upper_bound(r);
            if (it_up != tate[c].begin()) {
                --it_up;
                tate[c].erase(it_up);
                yoko[*it_up].erase(c);
            }

            // Find the nearest wall below
            auto it_down = tate[c].lower_bound(r);
            if (it_down != tate[c].end()) {
                tate[c].erase(it_down);
                yoko[*it_down].erase(c);
            }

            // Find the nearest wall to the left
            auto it_left = yoko[r].upper_bound(c);
            if (it_left != yoko[r].begin()) {
                --it_left;
                yoko[r].erase(it_left);
                tate[*it_left].erase(r);
            }

            // Find the nearest wall to the right
            auto it_right = yoko[r].lower_bound(c);
            if (it_right != yoko[r].end()) {
                yoko[r].erase(it_right);
                tate[*it_right].erase(r);
            }
        }
    }

    int ans = 0;
    rep(i, w) {
        ans += tate[i].size();
    }
    cout << ans << "\n";
    return 0;
}
