#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> PII;
const int MAXN=1e5+7;

struct Node{
    int val,idx;
    bool operator<(const Node &other) const {return idx<other.idx;}
}a[MAXN];

int n,m,q,cnt=0;
int tr[2*MAXN],sz[2*MAXN];
vector<int> G[MAXN];

void add(int u){
    sz[u]+=tr[u]=1;
    while(u/=2)sz[u]+=tr[u]=max(tr[u<<1],tr[u<<1|1]);
}

int ask(int l,int r){
    int res=0;
    for(l+=cnt,r+=cnt;l<=r;l>>=1,r>>=1){
        if(l&1)res=max(res,tr[l++]);
        if(!(r&1))res=max(res,tr[r--]);
    }
    return res;
}

int find(int x){
    int L=1,R=cnt,mid,ans=-1;
    while(L<=R){
        mid=(L+R)>>1;
        if(ask(L,mid)>=x)L=mid+1,ans=mid;
        else R=mid-1;
    }
    return ans;
}

int main(){
    ios::sync_with_stdio(false);cin.tie(nullptr);
    cin>>n>>m>>q;
    cnt=n+m+2;
    for(int i=1;i<=n;i++)G[i].push_back(m+i);
    for(int i=1;i<=m;i++)G[m+i].push_back(n+i);
    for(int i=1;i<=n+m;i++){
        cin>>a[i].val,a[i].idx=i;
        add(a[i].idx);
    }
    sort(a+1,a+n+m+1);
    for(int i=1;i<=n+m;i++){
        int pos=find(a[i].val);
        for(auto x:G[pos])add(x);
        for(auto x:G[a[i].idx])if(x!=pos)add(x);
        cout<<(cnt-sz[n+i]-sz[n+i+1])<<'\n';
    }
    return 0;
}
