#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> row(w), col(h);

    // Initialize all walls present
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if the cell has a wall
        if (row[c].count(r)) {
            // Wall exists, destroy it
            row[c].erase(r);
            col[r].erase(c);
        } else {
            // Wall doesn't exist, destroy in four directions

            // Up and Down (check column c)
            auto& vertical = row[c];

            // Down: first wall below (r)
            auto it = vertical.upper_bound(r);
            if (it != vertical.end()) {
                int idx = *it;
                vertical.erase(idx);
                col[idx].erase(c);
            }

            // Up: first wall above (r)
            if (!vertical.empty()) {
                it = vertical.lower_bound(r);
                if (it != vertical.begin()) {
                    --it;
                    int idx = *it;
                    vertical.erase(idx);
                    col[idx].erase(c);
                }
            }

            // Left and Right (check row r)
            auto& horizontal = col[r];

            // Right: first wall to the right of (c)
            it = horizontal.upper_bound(c);
            if (it != horizontal.end()) {
                int idx = *it;
                horizontal.erase(idx);
                row[idx].erase(r);
            }

            // Left: first wall to the left of (c)
            if (!horizontal.empty()) {
                it = horizontal.lower_bound(c);
                if (it != horizontal.begin()) {
                    --it;
                    int idx = *it;
                    horizontal.erase(idx);
                    row[idx].erase(r);
                }
            }
        }
    }

    // Count remaining walls
    ll ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    cout << ans << endl;

    return 0;
}
