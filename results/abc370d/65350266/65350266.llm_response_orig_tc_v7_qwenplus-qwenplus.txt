#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;

    // For each row and column, keep track of remaining walls using sets
    vector<set<int>> rows(h), cols(w);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there's a wall at (r,c)
        if (rows[r].count(c)) {
            // Destroy the wall directly
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int x = *it;
                cols[c].erase(x);
                rows[x].erase(c);
            }

            // Down
            if (it != cols[c].begin()) {
                --it;
                int x = *it;
                cols[c].erase(x);
                rows[x].erase(c);
            }

            // Left
            it = rows[r].lower_bound(c);
            if (it != rows[r].end()) {
                int y = *it;
                rows[r].erase(y);
                cols[y].erase(r);
            }

            // Right
            if (it != rows[r].begin()) {
                --it;
                int y = *it;
                rows[r].erase(y);
                cols[y].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";
}
