#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> rows(h), cols(w);
    
    // Initialize each row and column with all positions having a wall
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if the current cell has a wall
        auto it = rows[r].find(c);
        if (it != rows[r].end()) {
            // Wall exists here - destroy it
            rows[r].erase(it);
            cols[c].erase(r);
        } else {
            // Wall does not exist here - destroy first wall in each direction

            // Up
            auto up = cols[c].lower_bound(r);
            if (up != cols[c].begin()) {
                --up;
                int row = *up;
                cols[c].erase(up);
                rows[row].erase(c);
            }

            // Down
            auto down = cols[c].lower_bound(r);
            if (down != cols[c].end()) {
                int row = *down;
                cols[c].erase(down);
                rows[row].erase(c);
            }

            // Left
            auto left = rows[r].lower_bound(c);
            if (left != rows[r].begin()) {
                --left;
                int col = *left;
                rows[r].erase(left);
                cols[col].erase(r);
            }

            // Right
            auto right = rows[r].lower_bound(c);
            if (right != rows[r].end()) {
                int col = *right;
                rows[r].erase(right);
                cols[col].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";

    return 0;
}
