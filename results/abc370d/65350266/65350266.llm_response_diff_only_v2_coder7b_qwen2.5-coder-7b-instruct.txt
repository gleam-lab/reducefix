#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    while(q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if(grid[r][c]) {
            grid[r][c] = false;
            // Check up
            for(int i = r - 1; i >= 0 && !grid[i][c]; --i)
                grid[i][c] = true;
            // Check down
            for(int i = r + 1; i < h && !grid[i][c]; ++i)
                grid[i][c] = true;
            // Check left
            for(int j = c - 1; j >= 0 && !grid[r][j]; --j)
                grid[r][j] = true;
            // Check right
            for(int j = c + 1; j < w && !grid[r][j]; ++j)
                grid[r][j] = true;
        }
    }
    int ans = 0;
    for(auto &row : grid)
        ans += count(row.begin(), row.end(), true);
    cout << ans << "\n";
    return 0;
}
