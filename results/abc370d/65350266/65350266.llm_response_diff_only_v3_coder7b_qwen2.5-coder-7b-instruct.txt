#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    
    while (q--) {
        int r, c;
        cin >> r >> c;
        --r; --c;
        
        if (grid[r][c]) {
            grid[r][c] = false;
            continue;
        }

        for (int i = r - 1; i >= 0; --i) {
            if (grid[i][c]) {
                grid[i][c] = false;
                break;
            }
        }
        for (int i = r + 1; i < h; ++i) {
            if (grid[i][c]) {
                grid[i][c] = false;
                break;
            }
        }
        for (int j = c - 1; j >= 0; --j) {
            if (grid[r][j]) {
                grid[r][j] = false;
                break;
            }
        }
        for (int j = c + 1; j < w; ++j) {
            if (grid[r][j]) {
                grid[r][j] = false;
                break;
            }
        }
    }
    
    int ans = 0;
    for (const auto& row : grid) {
        ans += count(row.begin(), row.end(), true);
    }
    
    cout << ans << "\n";
    return 0;
}

This solution maintains a grid where each cell represents whether it has a wall or not. When a bomb is dropped, it checks and destroys walls according to the problem's rules. The final answer is calculated by counting the remaining walls in the grid. This approach ensures that the solution is both correct and efficient, handling the constraints provided.
