#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (n); ++i)
#define forn(i, a, b) for(int i = (a); i <= (b); ++i)
typedef long long ll;
typedef pair<int, int> P;

int main() {
    cin.tie(nullptr);
    ios::sync_with_stdio(false);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<set<int>> row(W), col(H);
    rep(i, H) {
        rep(j, W) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there's a wall at (r,c)
        if (row[c].count(r)) {
            row[c].erase(r);
            col[r].erase(c);
        } else {
            // Up
            auto it = row[c].lower_bound(r);
            if (it != row[c].begin()) {
                --it;
                int x = *it;
                row[c].erase(it);
                col[x].erase(c);
            }

            // Down
            it = row[c].lower_bound(r);
            if (it != row[c].end()) {
                int x = *it;
                row[c].erase(it);
                col[x].erase(c);
            }

            // Left
            it = col[r].lower_bound(c);
            if (it != col[r].begin()) {
                --it;
                int y = *it;
                col[r].erase(it);
                row[y].erase(r);
            }

            // Right
            it = col[r].lower_bound(c);
            if (it != col[r].end()) {
                int y = *it;
                col[r].erase(it);
                row[y].erase(r);
            }
        }
    }

    int ans = 0;
    rep(i, W) ans += row[i].size();
    cout << ans << endl;

    return 0;
}
