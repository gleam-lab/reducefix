#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> tate(w), yoko(h);
    rep(i, h) rep(j, w) {
        tate[j].insert(i);
        yoko[i].insert(j);
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        if (!tate[c].empty() && *tate[c].begin() == r) { // Destroy top wall
            tate[c].erase(tate[c].begin());
            yoko[r].erase(yoko[r].begin());
        } else if (!tate[c].empty() && *tate[c].rbegin() == r) { // Destroy bottom wall
            tate[c].erase(prev(tate[c].rend()));
            yoko[r].erase(prev(yoko[r].rend()));
        } else if (!yoko[r].empty() && *yoko[r].begin() == c) { // Destroy left wall
            yoko[r].erase(yoko[r].begin());
            tate[c].erase(tate[c].begin());
        } else if (!yoko[r].empty() && *yoko[r].rbegin() == c) { // Destroy right wall
            yoko[r].erase(prev(yoko[r].rend()));
            tate[c].erase(prev(tate[c].rend()));
        }
    }

    int ans = 0;
    for (auto& st : tate) {
        ans += st.size();
    }
    for (auto& st : yoko) {
        ans += st.size();
    }
    cout << ans / 2 << '\n'; // Each wall is counted twice
    return 0;
}
