#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> rows(h), cols(w);
    
    // Initialize each row and column with all positions
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there's a wall at the bomb position
        if (rows[r].count(c)) {
            // Destroy wall at (r, c)
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Up
            auto up = cols[c].lower_bound(r);
            if (up != cols[c].begin()) {
                --up;
                int u = *up;
                cols[c].erase(u);
                rows[u].erase(c);
            }

            // Down
            auto down = cols[c].lower_bound(r);
            if (down != cols[c].end()) {
                int d = *down;
                cols[c].erase(d);
                rows[d].erase(c);
            }

            // Left
            auto left = rows[r].lower_bound(c);
            if (left != rows[r].begin()) {
                --left;
                int l = *left;
                rows[r].erase(l);
                cols[l].erase(r);
            }

            // Right
            auto right = rows[r].lower_bound(c);
            if (right != rows[r].end()) {
                int ri = *right;
                rows[r].erase(ri);
                cols[ri].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";

    return 0;
}
