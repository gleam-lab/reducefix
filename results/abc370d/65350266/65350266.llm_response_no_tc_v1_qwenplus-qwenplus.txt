#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // For each column, keep track of remaining walls in that column (sorted)
    vector<set<int>> col_walls(W);
    // For each row, keep track of remaining walls in that row (sorted)
    vector<set<int>> row_walls(H);

    // Initialize all walls
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            col_walls[j].insert(i);
            row_walls[i].insert(j);
        }
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;
        
        // Check if there is a wall at the current cell
        auto it = row_walls[R].find(C);
        if (it != row_walls[R].end()) {
            // Wall exists at (R, C), destroy it
            row_walls[R].erase(it);
            col_walls[C].erase(col_walls[C].find(R));
        } else {
            // No wall here, look in four directions
            // Up (in column C, above R)
            auto up_it = col_walls[C].lower_bound(R);
            if (up_it != col_walls[C].begin()) {
                up_it--;
                int r = *up_it;
                row_walls[r].erase(row_walls[r].find(C));
                col_walls[C].erase(up_it);
            }

            // Down (in column C, below R)
            auto down_it = col_walls[C].lower_bound(R);
            if (down_it != col_walls[C].end()) {
                int r = *down_it;
                row_walls[r].erase(row_walls[r].find(C));
                col_walls[C].erase(down_it);
            }

            // Left (in row R, left of C)
            auto left_it = row_walls[R].lower_bound(C);
            if (left_it != row_walls[R].begin()) {
                left_it--;
                int c = *left_it;
                col_walls[c].erase(col_walls[c].find(R));
                row_walls[R].erase(left_it);
            }

            // Right (in row R, right of C)
            auto right_it = row_walls[R].lower_bound(C);
            if (right_it != row_walls[R].end()) {
                int c = *right_it;
                col_walls[c].erase(col_walls[c].find(R));
                row_walls[R].erase(right_it);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < W; ++i) {
        ans += col_walls[i].size();
    }
    cout << ans << endl;

    return 0;
}
