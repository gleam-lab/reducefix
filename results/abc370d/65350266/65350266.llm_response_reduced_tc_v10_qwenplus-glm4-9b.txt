#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<vector<set<int>>> walls(h, vector<set<int>>(w));
    
    // Initialize walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            walls[i][j] = {i, j};
        }
    }

    for (int i = 0; i < h; ++i) {
        sort(walls[i].begin(), walls[i].end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return a.second < b.second;
        });
    }
    for (int j = 0; j < w; ++j) {
        sort(walls[0][j].begin(), walls[0][j].end());
        walls[0][j].erase(walls[0][j].begin());
    }
    for (int i = 1; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            auto it = lower_bound(walls[i - 1][j].begin(), walls[i - 1][j].end(), i);
            if (it != walls[i - 1][j].end()) {
                walls[i][j].insert(*it);
            }
        }
    }

    for (int i = 0; i < h; ++i) {
        sort(walls[i].begin(), walls[i].end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return a.first < b.first;
        });
    }
    for (int j = 0; j < w; ++j) {
        sort(walls[j].begin(), walls[j].end());
        walls[j].erase(walls[j].begin());
    }
    for (int i = 1; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            auto it = lower_bound(walls[j][i - 1].begin(), walls[j][i - 1].end(), i);
            if (it != walls[j][i - 1].end()) {
                walls[i][j].insert(*it);
            }
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        auto it = walls[r][c].find(c);
        if (it == walls[r][c].end()) {
            cout << 0 << '\n';
        } else {
            cout << walls[r][c].size() << '\n';
        }
    }

    return 0;
}
