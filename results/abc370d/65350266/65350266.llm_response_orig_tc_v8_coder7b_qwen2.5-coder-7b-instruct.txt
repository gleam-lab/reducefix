#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    while(q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if(grid[r][c]) {
            grid[r][c] = false;
        } else {
            for(int i = 0; i < h; i++) {
                if(grid[i][c] && i < r) {
                    grid[i][c] = false;
                } else if(grid[i][c]) break;
            }
            for(int i = h-1; i >= 0; i--) {
                if(grid[i][c] && i > r) {
                    grid[i][c] = false;
                } else if(grid[i][c]) break;
            }
            for(int j = 0; j < w; j++) {
                if(grid[r][j] && j < c) {
                    grid[r][j] = false;
                } else if(grid[r][j]) break;
            }
            for(int j = w-1; j >= 0; j--) {
                if(grid[r][j] && j > c) {
                    grid[r][j] = false;
                } else if(grid[r][j]) break;
            }
        }
    }
    int ans = 0;
    rep(i,h) rep(j,w) if(grid[i][j]) ans++;
    cout << ans << "\n";
    return 0;
}
