#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<pair<int, int>>> grid(h, set<pair<int, int>>());
    vector<set<int>> row_walls(w), col_walls(h);
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            grid[i].insert({i, j});
            row_walls[j].insert(i);
            col_walls[i].insert(j);
        }
    }
    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;  // Convert to 0-based index
        if (grid[r].count({r, c})) {
            grid[r].erase({r, c});
            row_walls[c].erase(r);
            col_walls[r].erase(c);
        } else {
            auto down_iter = grid[r].lower_bound({r, c});
            if (down_iter != grid[r].end()) {
                int tmp = down_iter->first;
                grid[r].erase({r, c});
                row_walls[c].erase(r);
                col_walls[r].erase(c);
                grid[tmp].erase({tmp, c});
                row_walls[c].erase(tmp);
                col_walls[tmp].erase(c);
            }
            auto up_iter = grid[r].lower_bound({r, c});
            if (up_iter != grid[r].begin()) {
                up_iter--;
                int tmp = up_iter->first;
                grid[r].erase({r, c});
                row_walls[c].erase(r);
                col_walls[r].erase(c);
                grid[tmp].erase({tmp, c});
                row_walls[c].erase(tmp);
                col_walls[tmp].erase(c);
            }
            auto right_iter = grid[r].lower_bound({r, c});
            if (right_iter != grid[r].end()) {
                int tmp = right_iter->second;
                grid[r].erase({r, c});
                row_walls[c].erase(r);
                col_walls[r].erase(c);
                grid[r].erase({r, tmp});
                row_walls[tmp].erase(r);
                col_walls[r].erase(tmp);
            }
            auto left_iter = grid[r].lower_bound({r, c});
            if (left_iter != grid[r].begin()) {
                left_iter--;
                int tmp = left_iter->second;
                grid[r].erase({r, c});
                row_walls[c].erase(r);
                col_walls[r].erase(c);
                grid[r].erase({r, tmp});
                row_walls[tmp].erase(r);
                col_walls[r].erase(tmp);
            }
        }
    }
    int remaining_walls = 0;
    for (int i = 0; i < w; i++) {
        remaining_walls += row_walls[i].size();
    }
    for (int i = 0; i < h; i++) {
        remaining_walls += col_walls[i].size();
    }
    cout << remaining_walls << endl;
    return 0;
}
