#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // Initialize a vector of sets for each row and column
    vector<set<int>> row_wall(H, set<int>(W));
    vector<set<int>> col_wall(W, set<int>(H));
    
    // Mark all cells as having one wall each
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            row_wall[i].insert(j);
            col_wall[j].insert(i);
        }
    }

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        R--; C--;
        
        if (row_wall[R][C]) {
            // If the wall exists at (R, C), destroy it
            row_wall[R].erase(C);
            col_wall[C].erase(R);
        } else {
            // Find the first wall in each direction
            // Up
            auto it = row_wall[R].lower_bound(C);
            while (it != row_wall[R].end() && !col_wall[*it][R]) {
                it++;
            }
            if (it != row_wall[R].end()) {
                row_wall[R].erase(*it);
                col_wall[*it].erase(R);
            }
            // Down
            it = row_wall[R].lower_bound(C);
            if (it == row_wall[R].end()) {
                it = prev(row_wall[R].end());
            }
            while (it != row_wall[R].begin() && !col_wall[*it][R]) {
                it--;
            }
            if (it != row_wall[R].begin()) {
                row_wall[R].erase(*it);
                col_wall[*it].erase(R);
            }
            // Left
            it = col_wall[C].lower_bound(R);
            while (it != col_wall[C].end() && !row_wall[*it][C]) {
                it++;
            }
            if (it != col_wall[C].end()) {
                col_wall[C].erase(*it);
                row_wall[*it].erase(C);
            }
            // Right
            it = col_wall[C].lower_bound(R);
            if (it == col_wall[C].end()) {
                it = prev(col_wall[C].end());
            }
            while (it != col_wall[C].begin() && !row_wall[*it][C]) {
                it--;
            }
            if (it != col_wall[C].begin()) {
                col_wall[C].erase(*it);
                row_wall[*it].erase(C);
            }
        }
    }
    
    // Calculate the number of remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < H; ++i) {
        remaining_walls += row_wall[i].size();
    }
    for (int j = 0; j < W; ++j) {
        remaining_walls += col_wall[j].size();
    }
    
    cout << remaining_walls << endl;
    return 0;
}
