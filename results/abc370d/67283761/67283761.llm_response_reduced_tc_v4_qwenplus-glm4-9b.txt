#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> walls(H, vector<bool>(W, true));
    int totalWalls = H * W;

    while (Q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        if (walls[r][c]) {
            walls[r][c] = false;
        } else {
            for (int dr = -1; dr <= 1; ++dr) {
                for (int dc = -1; dc <= 1; ++dc) {
                    if (dr == 0 && dc == 0) continue;
                    int nr = r, nc = c;
                    while (true) {
                        nr += dr;
                        nc += dc;
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W || !walls[nr][nc]) break;
                        walls[nr][nc] = false;
                    }
                }
            }
        }
    }

    cout << totalWalls - count(walls.begin(), walls.end(), false) << endl;

    return 0;
}
