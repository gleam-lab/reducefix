#include <vector>
#include <stack>
#include <queue>
#include <tuple>
#include <iostream>
#include <fstream>
#include <cmath>
#include <cstring>
#include <climits>
#include <set>
#include <map>
#include <algorithm>
#include <iomanip>
#include <numeric>
using namespace std;
#pragma warning(disable: 4189)
#pragma warning(disable: 4267)

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> map(H, vector<bool>(W, true));

    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        if (map[r][c]) {
            map[r][c] = false;
        } else {
            // Find the nearest wall and break it
            queue<pair<int, int>> q;
            vector<vector<bool>> visited(H, vector<bool>(W, false));
            q.push({r, c});
            visited[r][c] = true;
            while (!q.empty()) {
                auto [x, y] = q.front();
                q.pop();
                for (auto [dx, dy] : {{-1, 0}, {1, 0}, {0, -1}, {0, 1}}) {
                    int nx = x + dx, ny = y + dy;
                    if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                        visited[nx][ny] = true;
                        if (map[nx][ny]) {
                            map[nx][ny] = false;
                            break;
                        }
                        q.push({nx, ny});
                    }
                }
            }
        }
    }

    int remaining_walls = 0;
    for (const auto& row : map) {
        for (bool cell : row) {
            if (cell) {
                ++remaining_walls;
            }
        }
    }
    cout << remaining_walls << endl;

    return 0;
}
