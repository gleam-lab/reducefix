
```cpp
#include <vector>
#include <iostream>
using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<vector<int>> grid(H, vector<int>(W, H * W)); // Each cell initially has H * W walls
    
    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        
        // Check if the bomb destroys a wall
        if (--grid[r][c] == 0) {
            for (int i = 0; i <= H; ++i) {
                if (i != r && grid[i][c] > 0) grid[i][c] -= 1;
            }
            for (int j = 0; j <= W; ++j) {
                if (j != c && grid[r][j] > 0) grid[r][j] -= 1;
            }
        }
    }
    
    // Calculate the remaining number of walls
    int remainingWalls = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            remainingWalls += max(grid[i][j] - 1, 0);
        }
    }
    
    cout << remainingWalls << endl;
    
    return 0;
}
```

Explanation:
- The grid is initialized to have `H * W` walls in each cell.
- For each bomb placement, if there are walls in the cell, decrement the wall count for that cell and all adjacent cells.
- If there are no walls in the cell, decrement the wall count only for that cell.
- The remaining walls are then calculated by summing up the walls left in each cell and subtracting 1 as each cell starts with `H * W` walls but the final count should represent walls remaining, not total walls.
- This solution handles the edge case where the grid has no walls left after the queries by simply returning 0.
