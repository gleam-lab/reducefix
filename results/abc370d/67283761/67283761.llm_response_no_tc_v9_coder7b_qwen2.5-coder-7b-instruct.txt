```cpp
#include <vector>
#include <queue>
#include <utility>
#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    // Initialize the grid with walls
    vector<vector<bool>> grid(H, vector<bool>(W, true));

    while(Q--) {
        int r, c;
        cin >> r >> c;
        --r, --c; // Convert to zero-based index

        if(grid[r][c]) {
            grid[r][c] = false; // Destroy the wall
        } else {
            queue<pair<int, int>> q;
            vector<vector<bool>> visited(H, vector<bool>(W, false));
            q.push({r, c});
            visited[r][c] = true;

            while(!q.empty()) {
                auto [x, y] = q.front();
                q.pop();

                for(int dx = -1; dx <= 1; ++dx) {
                    for(int dy = -1; dy <= 1; ++dy) {
                        if(dx == 0 && dy == 0) continue;
                        int nx = x + dx, ny = y + dy;
                        if(nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && grid[nx][ny]) {
                            grid[nx][ny] = false; // Destroy the wall
                            visited[nx][ny] = true;
                            q.push({nx, ny});
                        }
                    }
                }
            }
        }
    }

    // Count the remaining walls
    int remaining_walls = 0;
    for(auto &row : grid) {
        for(bool cell : row) {
            if(cell) ++remaining_walls;
        }
    }

    cout << remaining_walls << endl;

    return 0;
}
```

### Explanation
1. **Initialization**: The grid is initialized with walls (`true`).
2. **Processing Queries**:
   - For each query, check if the cell has a wall.
   - If it does, destroy the wall.
   - If it doesn't, perform a breadth-first search (BFS) starting from the cell to destroy all adjacent walls in all directions until no more walls are found.
3. **Counting Remaining Walls**: After processing all queries, count the number of walls still present in the grid.
4. **Output**: Print the number of remaining walls.

This solution ensures correctness and efficiency within the constraints provided.
