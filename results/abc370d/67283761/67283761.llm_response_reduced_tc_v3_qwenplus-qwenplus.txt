#include <vector>
#include <queue>
#include <tuple>
#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<bool>> has_wall(H, vector<bool>(W, true));
    
    // For each direction: up, down, left, right
    // We use maps to find nearest wall in each direction efficiently
    vector<vector<int>> up(W, -1), down(W, -1), left(H, -1), right(H, -1);

    auto get_up = [&](int r, int c) -> int {
        for (int i = r - 1; i >= 0; --i) {
            if (has_wall[i][c]) return i;
        }
        return -1;
    };

    auto get_down = [&](int r, int c) -> int {
        for (int i = r + 1; i < H; ++i) {
            if (has_wall[i][c]) return i;
        }
        return -1;
    };

    auto get_left = [&](int r, int c) -> int {
        for (int j = c - 1; j >= 0; --j) {
            if (has_wall[r][j]) return j;
        }
        return -1;
    };

    auto get_right = [&](int r, int c) -> int {
        for (int j = c + 1; j < W; ++j) {
            if (has_wall[r][j]) return j;
        }
        return -1;
    };

    int total_destroyed = 0;

    while (Q--) {
        int R, C;
        cin >> R >> C;
        --R; --C;

        if (has_wall[R][C]) {
            has_wall[R][C] = false;
            ++total_destroyed;
        } else {
            // Try to destroy walls in all four directions
            int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}}; // up, down, left, right
            int (*get_dir[])(int, int) = {get_up, get_down, get_left, get_right};

            for (int i = 0; i < 4; ++i) {
                int nr = R + dirs[i][0];
                int nc = C + dirs[i][1];

                while (nr >= 0 && nr < H && nc >= 0 && nc < W && !has_wall[nr][nc]) {
                    nr += dirs[i][0];
                    nc += dirs[i][1];
                }

                if (nr >= 0 && nr < H && nc >= 0 && nc < W && has_wall[nr][nc]) {
                    has_wall[nr][nc] = false;
                    ++total_destroyed;
                }
            }
        }
    }

    cout << (H * W - total_destroyed) << endl;
    return 0;
}
