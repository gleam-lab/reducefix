#include <vector>
#include <iostream>
using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true)); // grid[i][j] = true if wall exists, false otherwise
    int remainingWalls = H * W; // initial number of walls

    for (int q = 0; q < Q; ++q) {
        int R, C;
        cin >> R >> C;
        --R; --C; // convert to 0-based index

        if (!grid[R][C]) { // if no wall exists at (R, C)
            // Look for the first wall to destroy in all four directions
            int destroyU = R - 1;
            int destroyD = R + 1;
            int destroyL = C - 1;
            int destroyR = C + 1;

            // Up
            while (destroyU >= 0 && grid[destroyU][C]) {
                grid[destroyU][C] = false;
                --destroyU;
            }
            // Down
            while (destroyD < H && grid[destroyD][C]) {
                grid[destroyD][C] = false;
                ++destroyD;
            }
            // Left
            while (destroyL >= 0 && grid[R][destroyL]) {
                grid[R][destroyL] = false;
                --destroyL;
            }
            // Right
            while (destroyR < W && grid[R][destroyR]) {
                grid[R][destroyR] = false;
                ++destroyR;
            }
        } else { // if wall exists at (R, C), destroy it
            grid[R][C] = false;
            --remainingWalls;
        }
    }

    cout << remainingWalls << endl;

    return 0;
}
