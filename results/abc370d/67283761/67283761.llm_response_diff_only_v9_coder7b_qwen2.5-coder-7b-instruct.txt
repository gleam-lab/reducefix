```cpp
#include <vector>
#include <queue>
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<bool>> map(H, vector<bool>(W, true));
    queue<pair<int, pair<int, int>>> q;

    while(Q--) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        if(!map[r][c]) continue;

        map[r][c] = false;
        q.push({r, {c, 0}});
        q.push({r, {c, 1}});
        q.push({r, {c, 2}});
        q.push({r, {c, 3}});

        while(!q.empty()) {
            auto [x, y] = q.front().second;
            int cx = q.front().first, cy = q.second;
            q.pop();

            int dx[4] = {-1, 1, 0, 0}, dy[4] = {0, 0, -1, 1};

            for(int d = 0; d < 4; ++d) {
                int nx = cx + dx[d], ny = cy + dy[d];

                if(nx < 0 || nx >= H || ny < 0 || ny >= W) continue;
                if(!map[nx][ny]) continue;

                map[nx][ny] = false;
                q.push({nx, {ny, d}});
            }
        }
    }

    int ans = 0;
    for(int i = 0; i < H; ++i) 
        for(int j = 0; j < W; ++j)
            if(map[i][j]) ans++;

    cout << ans << '\n';
    return 0;
}
```

This solution uses BFS to traverse through the graph. For every cell, it checks its adjacent cells (up, down, left, right) and removes them if they contain a wall. The complexity of this approach is O(H*W+Q), which is efficient enough given the problem constraints.
