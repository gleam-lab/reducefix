#include <iostream>
#include <vector>
using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> map(H, vector<bool>(W, true));

    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r; --c;
        
        if (map[r][c]) {
            // Bomb explodes and destroys the wall at (r, c)
            map[r][c] = false;
        } else {
            // Bomb explodes and destroys walls around (r, c)
            // We need to check and destroy walls to the top, bottom, left, and right
            // We will use a vector to keep track of the walls to be destroyed
            vector<pair<int, int>> walls_to_destroy;
            
            // Top walls
            for (int j = c - 1; j >= 0; --j) {
                if (map[r][j]) {
                    walls_to_destroy.push_back({r, j});
                    map[r][j] = false;
                }
            }
            
            // Bottom walls
            for (int j = c + 1; j < W; ++j) {
                if (map[r][j]) {
                    walls_to_destroy.push_back({r, j});
                    map[r][j] = false;
                }
            }
            
            // Left walls
            for (int i = r - 1; i >= 0; --i) {
                if (map[i][c]) {
                    walls_to_destroy.push_back({i, c});
                    map[i][c] = false;
                }
            }
            
            // Right walls
            for (int i = r + 1; i < H; ++i) {
                if (map[i][c]) {
                    walls_to_destroy.push_back({i, c});
                    map[i][c] = false;
                }
            }
        }
    }
    
    // Count the remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (map[i][j]) {
                remaining_walls++;
            }
        }
    }
    
    // Output the result
    cout << remaining_walls << endl;
    
    return 0;
}
