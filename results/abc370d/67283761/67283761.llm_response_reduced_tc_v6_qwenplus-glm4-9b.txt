#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true));

    for (int q = 0; q < Q; q++) {
        int r, c;
        cin >> r >> c;
        --r; --c;

        if (grid[r][c]) {
            grid[r][c] = false;
        } else {
            // Determine the minimum distance to a wall in each direction
            int up = 0, down = 0, left = 0, right = 0;
            for (up = 0; r - up >= 0 && !grid[r - up][c]; ++up);
            for (down = 0; r + down < H && !grid[r + down][c]; ++down);
            for (left = 0; c - left >= 0 && !grid[r][c - left]; ++left);
            for (right = 0; c + right < W && !grid[r][c + right]; ++right);

            // Bomb at (r, c) destroys all walls in the four directions up to the nearest wall
            grid[r][c] = false;
            grid[r - up][c] = false;
            grid[r + down][c] = false;
            grid[r][c - left] = false;
            grid[r][c + right] = false;
        }
    }

    // Count the remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (grid[i][j]) {
                remaining_walls++;
            }
        }
    }

    cout << remaining_walls << endl;

    return 0;
}
