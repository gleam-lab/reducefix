
```cpp
#include <vector>
#include <iostream>

using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<vector<int>> rows(H + 1, vector<int>(W + 1, 0));
    vector<vector<int>> cols(H + 1, vector<int>(W + 1, 0));
    vector<vector<bool>> destroyed(H + 1, vector<bool>(W + 1, false));
    
    int totalWalls = H * W;
    
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            if (i != 1) rows[i][j] = rows[i - 1][j];
            else rows[i][j] = i;
            
            if (j != 1) cols[i][j] = cols[i][j - 1];
            else cols[i][j] = j;
        }
    }
    
    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        
        if (!destroyed[r][c]) {
            totalWalls--;
            destroyed[r][c] = true;
            
            // Destroy wall above
            int row = rows[r][c];
            while (row > 1 && !destroyed[row - 1][c]) {
                destroyed[row - 1][c] = true;
                totalWalls--;
                row--;
            }
            
            // Destroy wall below
            row = rows[r][c];
            while (row < H && !destroyed[row + 1][c]) {
                destroyed[row + 1][c] = true;
                totalWalls--;
                row++;
            }
            
            // Destroy wall left
            int col = cols[r][c];
            while (col > 1 && !destroyed[r][col - 1]) {
                destroyed[r][col - 1] = true;
                totalWalls--;
                col--;
            }
            
            // Destroy wall right
            col = cols[r][c];
            while (col < W && !destroyed[r][col + 1]) {
                destroyed[r][col + 1] = true;
                totalWalls--;
                col++;
            }
        }
    }
    
    cout << totalWalls << endl;
    
    return 0;
}
```
