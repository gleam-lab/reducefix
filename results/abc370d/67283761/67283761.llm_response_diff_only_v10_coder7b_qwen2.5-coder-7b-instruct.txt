#include <vector>
#include <queue>
#include <tuple>
#include <iostream>
#include <cmath>
#include <cstring>
#include <climits>
#include <set>
#include <map>
#include <algorithm>
#include <iomanip>
#include <numeric>
using namespace std;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> grid(H, vector<bool>(W, true));

    queue<array<int, 3>> q; // {x, y, direction} where direction is 0 (up), 1 (right), 2 (down), 3 (left)
    vector<vector<vector<int>>> cache(4, vector<vector<int>>(H, vector<int>(W, 0)));

    for(int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        if(grid[r][c]) {
            grid[r][c] = false;
        } else {
            for(int d = 0; d < 4; ++d) {
                int x = r, y = c;
                while(true) {
                    x += d == 0 ? -1 : d == 2 ? 1 : 0;
                    y += d == 1 ? 1 : d == 3 ? -1 : 0;
                    if(x < 0 || x >= H || y < 0 || y >= W || !grid[x][y]) break;
                    grid[x][y] = false;
                    cache[d][r][c] = max(abs(r - x), abs(c - y));
                    q.push({x, y, d});
                }
            }
        }
    }

    int ans = H * W;
    for(auto& row : grid) {
        for(bool val : row) {
            if(val) --ans;
        }
    }

    cout << ans << '\n';

    return 0;
}
