### Problem Analysis

Given a grid with walls, the task is to process a series of queries where each query places a bomb at a specified cell. If there's a wall at the cell, it's destroyed. If there are no walls, the bomb destroys walls in all four cardinal directions until it reaches another wall or the edge of the grid.

### Key Observations

1. When a bomb destroys walls, it only affects walls that are directly adjacent to it (up, down, left, right).
2. To efficiently handle multiple queries, we need to avoid recalculating the destruction range for each query from scratch.

### Solution Strategy

1. Use dictionaries to keep track of Wall Ends (WE) in each direction for each cell. This allows us to quickly find the next wall in any direction starting from a given cell.
2. When placing a bomb, we start from the center and move in all four cardinal directions, marking the end of the wall destruction process at each step.
3. After all queries are processed, the number of remaining walls is simply the total number of cells minus the number of walls that were destroyed.

### Implementation

```python
from collections import defaultdict
import sys
input = sys.stdin.read

def solve():
    data = input().split()
    h = int(data[0])
    w = int(data[1])
    q = int(data[2])
    
    queries = []
    index = 3
    for _ in range(q):
        r = int(data[index]) - 1
        c = int(data[index + 1]) - 1
        queries.append((r, c))
        index += 2
    
    # Initialize the grid (not needed for this solution)
    grid = [[True] * w for _ in range(h)]
    
    # Dictionaries to keep track of wall ends in each direction
    up = defaultdict(list)  # Wall ends in the up direction (row index)
    right = defaultdict(list)  # Wall ends in the right direction (col index)
    down = defaultdict(list)  # Wall ends in the down direction (row index)
    left = defaultdict(list)  # Wall ends in the left direction (col index)
    
    total_walls = h * w
    remaining_walls = total_walls
    
    for r, c in queries:
        # Place bomb and destroy walls in all directions
        if not grid[r][c]:  # If there's no wall, we need to destroy walls
            # Check and destroy walls in all four directions
            remaining_walls -= 1
            for dir in [up, right, down, left]:
                queue = [(r, c, dir)]  # Start from the center
                while queue:
                    i, j, dir = queue.pop(0)
                    if dir == 'up':
                        if i > 0:
                            if not grid[i-1][j]:
                                remaining_walls -= 1
                                queue.append((i-1, j, 'up'))
                    elif dir == 'right':
                        if j < w-1:
                            if not grid[i][j+1]:
                                remaining_walls -= 1
                                queue.append((i, j+1, 'right'))
                    elif dir == 'down':
                        if i < h-1:
                            if not grid[i+1][j]:
                                remaining_walls -= 1
                                queue.append((i+1, j, 'down'))
                    elif dir == 'left':
                        if j > 0:
                            if not grid[i][j-1]:
                                remaining_walls -= 1
                                queue.append((i, j-1, 'left'))
        else:
            # If there's a wall, it's destroyed by the bomb
            remaining_walls -= 1
    
    return remaining_walls

# Read input from standard input and solve the problem
if __name__ == "__main__":
    print(solve())
```
