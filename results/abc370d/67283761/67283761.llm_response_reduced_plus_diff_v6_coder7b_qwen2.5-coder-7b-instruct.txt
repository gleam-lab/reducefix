#include <bits/stdc++.h>

using namespace std;

struct Cell {
    bool has_wall;
    int nearest_wall[4]; // Up, Down, Left, Right
};

void solve() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    vector<vector<Cell>> grid(H, vector<Cell>(W));
    for (auto &row : grid) {
        for (auto &cell : row) {
            cell.has_wall = true;
            memset(cell.nearest_wall, -1, sizeof(cell.nearest_wall));
        }
    }

    queue<array<int, 2>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (grid[i][j].has_wall) {
                q.push({i, j});
                grid[i][j].nearest_wall[0] = i == 0 ? -1 : i;
                grid[i][j].nearest_wall[1] = i == H-1 ? -1 : i;
                grid[i][j].nearest_wall[2] = j == 0 ? -1 : j;
                grid[i][j].nearest_wall[3] = j == W-1 ? -1 : j;
            }
        }
    }

    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx < 0 || nx >= H || ny < 0 || ny >= W || !grid[nx][ny].has_wall) continue;
            
            int dist = abs(x - nx) + abs(y - ny);
            if (dist < grid[x][y].nearest_wall[d]) {
                grid[x][y].nearest_wall[d] = dist;
                q.push({nx, ny});
            }
        }
    }

    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;
        
        if (grid[r][c].has_wall) {
            grid[r][c].has_wall = false;
        } else {
            for (int d = 0; d < 4; ++d) {
                if (grid[r][c].nearest_wall[d] != -1) {
                    int nx = r + dx[d] * grid[r][c].nearest_wall[d], ny = c + dy[d] * grid[r][c].nearest_wall[d];
                    grid[nx][ny].has_wall = false;
                }
            }
        }
    }

    int ans = 0;
    for (const auto &row : grid) {
        for (const auto &cell : row) {
            if (cell.has_wall) ++ans;
        }
    }

    cout << ans << '\n';
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        solve();
    }

    return 0;
}
