```cpp
#include <iostream>
#include <vector>
#include <set>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    // Each cell starts with a wall (true means wall exists)
    vector<vector<bool>> has_wall(H + 2, vector<bool>(W + 2, true));

    // For each row and column, maintain sets of positions with walls
    vector<set<int>> row_walls(H + 1);
    vector<set<int>> col_walls(W + 1);

    // Initialize the sets
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            row_walls[i].insert(j);
            col_walls[j].insert(i);
        }
    }

    for (int _ = 0; _ < Q; ++_) {
        int r, c;
        cin >> r >> c;

        if (has_wall[r][c]) {
            // If there's a wall at (r,c), just remove it
            has_wall[r][c] = false;
            row_walls[r].erase(c);
            col_walls[c].erase(r);
        } else {
            // Otherwise, look in all four directions
            int dirs[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
            for (auto [dr, dc] : dirs) {
                int nr = r + dr;
                int nc = c + dc;
                while (nr >= 1 && nr <= H && nc >= 1 && nc <= W) {
                    if (has_wall[nr][nc]) {
                        // Found the first wall in this direction, destroy it
                        has_wall[nr][nc] = false;
                        row_walls[nr].erase(nc);
                        col_walls[nc].erase(nr);
                        break;
                    }
                    nr += dr;
                    nc += dc;
                }
            }
        }
    }

    // Count remaining walls
    int result = 0;
    for (int i = 1; i <= H; ++i) {
        for (int j = 1; j <= W; ++j) {
            if (has_wall[i][j]) ++result;
        }
    }

    cout << result << endl;

    return 0;
}
```
