Here's the corrected version of your C++ code. The issue was that the logic for breaking walls wasn't correctly implemented. I've updated it to ensure that only the first wall encountered in the specified directions is destroyed.

```cpp
#include <vector>
#include <iostream>
#include <unordered_set>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;
    vector<vector<bool>> map(H, vector<bool>(W, true));
    unordered_set<int> visited;

    for(int i = 0; i < Q; ++i){
        int r, c;
        cin >> r >> c;
        --r, --c;
        
        if(!visited.count(r * W + c)){
            visited.insert(r * W + c);
            continue;
        }

        // Directions: Up, Down, Left, Right
        int dr[] = {-1, 1, 0, 0};
        int dc[] = {0, 0, -1, 1};

        for(int d = 0; d < 4; ++d){
            int nr = r + dr[d], nc = c + dc[d];
            while(nr >= 0 && nr < H && nc >= 0 && nc < W){
                if(!map[nr][nc]) break;
                visited.erase(nr * W + nc);
                map[nr][nc] = false;
                nr += dr[d];
                nc += dc[d];
            }
        }
    }

    cout << H * W - visited.size() << '\n';
    
    return 0;
}
```

This solution uses a set to keep track of visited cells, which ensures that each cell is processed exactly once. It iterates through each query, and if the cell has not been visited before, it simply marks it as visited. If the cell has already been visited, it breaks the walls in the four cardinal directions until it finds a non-wall cell or reaches the boundary. This approach ensures that the problem is solved efficiently even for large inputs.
