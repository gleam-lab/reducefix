#include <vector>
#include <queue>
#include <tuple>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<bool>> map(H, vector<bool>(W, true));
    vector<vector<vector<int>>> cache(4, vector<vector<int>>(H, vector<int>(W, 0)));

    vector<pair<int, int>> directions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    int broken = 0;

    for (int i = 0; i < Q; ++i) {
        int r, c;
        cin >> r >> c;
        --r, --c;

        if (map[r][c]) {
            ++broken;
            map[r][c] = false;
        } else {
            for (int d = 0; d < 4; ++d) {
                int x = r, y = c;
                while (true) {
                    x += directions[d].first;
                    y += directions[d].second;

                    if (x >= 0 && x < H && y >= 0 && y < W) {
                        if (map[x][y]) {
                            ++broken;
                            map[x][y] = false;
                            cache[d][r][c] = max(abs(r - x), abs(c - y));
                            break;
                        }
                        cache[d][x][y] = max(cache[d][x][y], 1);
                    } else {
                        break;
                    }
                }
            }
        }
    }

    cout << (H * W - broken) << '\n';

    return 0;
}
