    for (int q = 0; q < Q; ++q) {

        R--; C--;

            grid[R][C] = false;
            remaining--;
        } else {

            for (int i = R - 1; i >= 0; --i) {

                    remaining--;
                    break;

            // Check down
            for (int i = R + 1; i < H; ++i) {

                    break;
                }

            // Check left
            for (int j = C - 1; j >= 0; --j) {
