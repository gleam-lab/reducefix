#include <iostream>
#include <vector>
#include <set>
#include <queue>
#include <tuple>
#include <algorithm>
using namespace std;

// Direction vectors for up, down, left, right
const int dx[] = {-1, 1, 0, 0};
const int dy[] = {0, 0, -1, 1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Q;
    cin >> H >> W >> Q;

    vector<vector<bool>> has_wall(H, vector<bool>(W, true));

    // For each row and column, we'll keep sorted sets of wall positions
    vector<set<int>> rows(W), cols(H);

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            rows[j].insert(i);
            cols[i].insert(j);
        }
    }

    auto remove_wall = [&](int r, int c) {
        if (!has_wall[r][c]) return;
        has_wall[r][c] = false;
        rows[c].erase(r);
        cols[r].erase(c);
    };

    for (int q = 0; q < Q; ++q) {
        int r, c;
        cin >> r >> c;
        --r; --c;

        if (has_wall[r][c]) {
            remove_wall(r, c);
        } else {
            // Look in all four directions
            for (int d = 0; d < 4; ++d) {
                int x = r, y = c;
                if (d == 0) { // Up
                    auto it = rows[c].lower_bound(r);
                    if (it != rows[c].begin()) {
                        --it;
                        remove_wall(*it, c);
                    }
                } else if (d == 1) { // Down
                    auto it = rows[c].lower_bound(r);
                    if (it != rows[c].end() && *it == r) ++it;
                    if (it != rows[c].end()) {
                        remove_wall(*it, c);
                    }
                } else if (d == 2) { // Left
                    auto it = cols[r].lower_bound(c);
                    if (it != cols[r].begin()) {
                        --it;
                        remove_wall(r, *it);
                    }
                } else { // Right
                    auto it = cols[r].lower_bound(c);
                    if (it != cols[r].end() && *it == c) ++it;
                    if (it != cols[r].end()) {
                        remove_wall(r, *it);
                    }
                }
            }
        }
    }

    int remaining = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (has_wall[i][j]) ++remaining;
        }
    }

    cout << remaining << endl;

    return 0;
}
