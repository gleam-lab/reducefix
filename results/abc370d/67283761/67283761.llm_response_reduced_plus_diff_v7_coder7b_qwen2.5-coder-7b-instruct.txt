```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 4e5 + 10;

int n, m, q, arr[MAXN][MAXN];

struct Node {
    int x, y, id;
};

bool cmp(Node &a, Node &b) {
    if(a.x != b.x) return a.x < b.x;
    return a.y < b.y;
}

void solve() {
    cin >> n >> m >> q;
    memset(arr, 1, sizeof(arr));
    vector<Node> v(q);
    for(int i = 0; i < q; ++i) {
        cin >> v[i].x >> v[i].y;
        v[i].x--, v[i].y--;
        v[i].id = i;
    }
    sort(v.begin(), v.end(), cmp);

    vector<int> ans(q);
    set<int> s;
    int lastX = -1;
    for(int i = 0; i < q; ++i) {
        if(v[i].x != lastX) {
            s.clear();
            lastX = v[i].x;
        }
        while(!s.empty() && arr[v[i].x][*(s.rbegin())] == 0) s.erase(s.rbegin());
        if(arr[v[i].x][v[i].y]) {
            arr[v[i].x][v[i].y] = 0;
        } else {
            auto it = s.lower_bound(v[i].y);
            if(it != s.begin()) {
                --it;
                int prevY = *it;
                arr[v[i].x][prevY] = 0;
                s.erase(prevY);
            }
            if(it != s.end()) {
                int nextY = *it;
                arr[v[i].x][nextY] = 0;
                s.erase(nextY);
            }
        }
        s.insert(v[i].y);
        ans[v[i].id] = n * m - accumulate(&arr[0][0], &arr[n][m], 0);
    }

    for(auto a : ans) cout << a << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    solve();

    return 0;
}
```
