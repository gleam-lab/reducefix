#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> grid(h, vector<int>(w, 1)); // 1 represents a wall, 0 represents no wall
    vector<vector<int>> up(h, vector<int>(w, 0)); // Track the closest up wall
    vector<vector<int>> down(h, vector<int>(w, 0)); // Track the closest down wall
    vector<vector<int>> left(h, vector<int>(w, 0)); // Track the closest left wall
    vector<vector<int>> right(h, vector<int>(w, 0)); // Track the closest right wall

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (i > 0) up[i][j] = up[i-1][j] + 1;
            if (i < h-1) down[i][j] = down[i+1][j] + 1;
            if (j > 0) left[i][j] = left[i][j-1] + 1;
            if (j < w-1) right[i][j] = right[i][j+1] + 1;
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if (grid[r][c] == 1) {
            grid[r][c] = 0; // Destroy the wall at the bomb location
            up[r][c] = down[r][c] = left[r][c] = right[r][c] = 0; // Reset the counts for the neighboring walls
        } else {
            // Find the nearest walls around the bomb location
            int min_up = (r > 0) ? min(up[r-1][c], down[0][c] - r) : 0;
            int min_down = (r < h-1) ? min(down[r+1][c], up[h-1][c] - (h-1-r)) : 0;
            int min_left = (c > 0) ? min(left[r][c-1], right[r][w-1] - (w-1-c)) : 0;
            int min_right = (c < w-1) ? min(right[r][c+1], left[r][0] - c) : 0;

            // Update the grid and the closest wall counts
            grid[r][c] = 0;
            up[r][c] = min_up;
            down[r][c] = min_down;
            left[r][c] = min_left;
            right[r][c] = min_right;

            // Update the neighboring walls
            if (r > 0) {
                up[r-1][c] = min(min_up + 1, up[r-1][c]);
            }
            if (r < h-1) {
                down[r+1][c] = min(min_down + 1, down[r+1][c]);
            }
            if (c > 0) {
                left[r][c-1] = min(min_left + 1, left[r][c-1]);
            }
            if (c < w-1) {
                right[r][c+1] = min(min_right + 1, right[r][c+1]);
            }
        }
    }

    // Count the remaining walls
    int remaining_walls = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            remaining_walls += grid[i][j];
        }
    }

    cout << remaining_walls << endl;
    return 0;
}
