```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // Each row and column's available columns/rows using sets
    vector<set<int>> rows(h), cols(w);
    
    // Initialize all cells have walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    for (int qi = 0; qi < q; ++qi) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (rows[r].count(c)) {
            // If wall exists at the position, destroy it
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Otherwise, destroy first walls in each direction

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                --it;
                int u = *it;
                rows[u].erase(c);
                cols[c].erase(u);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int d = *it;
                rows[d].erase(c);
                cols[c].erase(d);
            }

            // Left
            auto jt = rows[r].lower_bound(c);
            if (jt != rows[r].begin()) {
                --jt;
                int l = *jt;
                cols[l].erase(r);
                rows[r].erase(jt);
            }

            // Right
            jt = rows[r].lower_bound(c);
            if (jt != rows[r].end()) {
                int ri = *jt;
                cols[ri].erase(r);
                rows[r].erase(jt);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += rows[i].size();
    }
    cout << ans << "\n";

    return 0;
}
```
