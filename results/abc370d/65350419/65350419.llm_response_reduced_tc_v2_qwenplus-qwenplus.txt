#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> row(w), col(h);
    
    // Initialize sets with all indices (empty cells are not present)
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    for (int qi = 0; qi < q; ++qi) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        
        if (col[r].count(c)) {
            // Cell has a wall, destroy it
            col[r].erase(c);
            row[c].erase(r);
        } else {
            // Destroy the first walls in each direction
            auto& ri = row[c];
            auto it = ri.lower_bound(r);
            if (it != ri.begin()) {
                it--;
                int up = *it;
                col[up].erase(c);
                ri.erase(it);
            }

            it = ri.lower_bound(r);
            if (it != ri.end()) {
                int down = *it;
                col[down].erase(c);
                ri.erase(it);
            }

            auto& ci = col[r];
            it = ci.lower_bound(c);
            if (it != ci.begin()) {
                it--;
                int left = *it;
                row[left].erase(r);
                ci.erase(it);
            }

            it = ci.lower_bound(c);
            if (it != ci.end()) {
                int right = *it;
                row[right].erase(r);
                ci.erase(it);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    cout << ans << endl;
    return 0;
}
