#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // For each row and column, keep track of available walls using ordered sets
    vector<set<int>> rows(h), cols(w);
    
    // Initialize all cells with their positions
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // If there is a wall at (r,c), destroy it
        if (rows[r].count(c)) {
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Otherwise, look in four directions and destroy first walls found

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                it--;
                int up = *it;
                rows[up].erase(c);
                cols[c].erase(up);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int down = *it;
                rows[down].erase(c);
                cols[c].erase(down);
            }

            // Left
            auto rit = rows[r].lower_bound(c);
            if (rit != rows[r].begin()) {
                rit--;
                int left = *rit;
                cols[left].erase(r);
                rows[r].erase(left);
            }

            // Right
            rit = rows[r].lower_bound(c);
            if (rit != rows[r].end()) {
                int right = *rit;
                cols[right].erase(r);
                rows[r].erase(right);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < h; ++i)
        ans += rows[i].size();

    cout << ans << endl;
    return 0;
}
