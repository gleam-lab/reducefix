#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // For each column, keep track of remaining walls in that column (vertical)
    vector<set<int>> colWalls(W);
    // For each row, keep track of remaining walls in that row (horizontal)
    vector<set<int>> rowWalls(H);

    // Initialize all walls present
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            colWalls[j].insert(i);
            rowWalls[i].insert(j);
        }
    }

    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;

        if (colWalls[C].count(R)) {
            // Wall exists at the position, destroy it
            colWalls[C].erase(R);
            rowWalls[R].erase(C);
        } else {
            // Look up
            auto it = colWalls[C].lower_bound(R);
            if (it != colWalls[C].begin()) {
                it--;
                int r = *it;
                colWalls[C].erase(r);
                rowWalls[r].erase(C);
            }

            // Look down
            it = colWalls[C].lower_bound(R);
            if (it != colWalls[C].end()) {
                int r = *it;
                colWalls[C].erase(r);
                rowWalls[r].erase(C);
            }

            // Look left
            it = rowWalls[R].lower_bound(C);
            if (it != rowWalls[R].begin()) {
                it--;
                int c = *it;
                rowWalls[R].erase(c);
                colWalls[c].erase(R);
            }

            // Look right
            it = rowWalls[R].lower_bound(C);
            if (it != rowWalls[R].end()) {
                int c = *it;
                rowWalls[R].erase(c);
                colWalls[c].erase(R);
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < W; ++i) {
        ans += colWalls[i].size();
    }
    cout << ans << endl;

    return 0;
}
