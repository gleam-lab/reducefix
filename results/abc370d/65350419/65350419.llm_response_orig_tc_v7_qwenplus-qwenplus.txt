```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // tate[j] stores the row indices where walls exist in column j
    // yoko[i] stores the column indices where walls exist in row i
    vector<set<int>> tate(w), yoko(h);
    
    // Initialize all cells have a wall
    for(int i = 0; i < h; ++i) {
        for(int j = 0; j < w; ++j) {
            tate[j].insert(i);
            yoko[i].insert(j);
        }
    }

    while(q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        auto& row_set = yoko[r];
        auto& col_set = tate[c];

        // Check if there's a wall at current cell
        if(row_set.count(c)) {
            // Destroy wall at (r,c)
            row_set.erase(c);
            col_set.erase(r);
        } else {
            // Up
            auto it = col_set.lower_bound(r);
            if(it != col_set.begin()) {
                it--;
                int destroy_row = *it;
                col_set.erase(destroy_row);
                yoko[destroy_row].erase(c);
            }

            // Down
            it = col_set.lower_bound(r);
            if(it != col_set.end()) {
                int destroy_row = *it;
                col_set.erase(destroy_row);
                yoko[destroy_row].erase(c);
            }

            // Left
            auto& curr_col_set = row_set;
            auto jt = curr_col_set.lower_bound(c);
            if(jt != curr_col_set.begin()) {
                jt--;
                int destroy_col = *jt;
                curr_col_set.erase(destroy_col);
                tate[destroy_col].erase(r);
            }

            // Right
            jt = curr_col_set.lower_bound(c);
            if(jt != curr_col_set.end()) {
                int destroy_col = *jt;
                curr_col_set.erase(destroy_col);
                tate[destroy_col].erase(r);
            }
        }
    }

    int ans = 0;
    for(int i = 0; i < w; ++i) {
        ans += tate[i].size();
    }

    cout << ans << "\n";
    return 0;
}
```
