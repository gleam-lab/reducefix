#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<int>> tate(h, vector<int>(w, 0));
    vector<vector<int>> yoko(h, vector<int>(w, 0));
    
    // Initialize the grid with walls counted in all directions
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            tate[i][j] = i;    // Wall count vertically
            yoko[i][j] = j;    // Wall count horizontally
        }
    }

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        // Check if there is a wall at the given position
        if (tate[r][c] != 0 && yoko[r][c] != 0) {
            // Destroy the wall at the given position
            tate[r][c] = 0;
            yoko[r][c] = 0;
        } else {
            // Find the first wall in each direction to destroy
            int min_r = 0, max_r = h - 1, min_c = 0, max_c = w - 1;
            while (min_r < h && tate[min_r][c] == 0) min_r++; // Up
            while (max_r >= 0 && tate[max_r][c] == 0) max_r--; // Down
            while (min_c < w && yoko[r][min_c] == 0) min_c++; // Left
            while (max_c >= 0 && yoko[r][max_c] == 0) max_c--; // Right

            // Destroy walls in each direction
            for (int i = min_r; i <= max_r; ++i) tate[i][c] = 0;
            for (int j = min_c; j <= max_c; ++j) yoko[r][j] = 0;
        }
    }

    // Calculate the remaining number of walls
    int ans = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            ans += tate[i][j] + yoko[i][j] - 2; // Subtract 2 because each cell has two walls initially
        }
    }

    cout << ans << "\n";
    return 0;
}
