```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;

    // Each row and column's walls are stored as sets for efficient nearest neighbor queries
    vector<set<int>> rows(H), cols(W);

    // Initialize all walls
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            rows[i].insert(j);
            cols[j].insert(i);
        }
    }

    while (Q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if the current cell has a wall
        if (rows[r].count(c)) {
            // Wall exists: remove it
            rows[r].erase(c);
            cols[c].erase(r);
        } else {
            // Wall doesn't exist: remove first wall in each direction

            // Up
            auto it = cols[c].lower_bound(r);
            if (it != cols[c].begin()) {
                it--;
                int row = *it;
                rows[row].erase(c);
                cols[c].erase(it);
            }

            // Down
            it = cols[c].lower_bound(r);
            if (it != cols[c].end()) {
                int row = *it;
                rows[row].erase(c);
                cols[c].erase(it);
            }

            // Left
            auto jt = rows[r].lower_bound(c);
            if (jt != rows[r].begin()) {
                jt--;
                int col = *jt;
                cols[col].erase(r);
                rows[r].erase(jt);
            }

            // Right
            jt = rows[r].lower_bound(c);
            if (jt != rows[r].end()) {
                int col = *jt;
                cols[col].erase(r);
                rows[r].erase(jt);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < H; ++i) {
        ans += rows[i].size();
    }
    cout << ans << '\n';

    return 0;
}
```
