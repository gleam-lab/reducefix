#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<vector<bool>> grid(h, vector<bool>(w, true));
    while (q--) {
        int r, c;
        cin >> r >> c;
        --r; --c; // Adjusting indices as input is 1-based
        if (!grid[r][c]) continue; // Skip if already destroyed
        grid[r][c] = false; // Destroy the wall at (r, c)
        
        // Check and destroy walls above and below
        for (int i = r - 1; i >= 0 && !grid[i][c]; --i) grid[i][c] = false;
        for (int i = r + 1; i < h && !grid[i][c]; ++i) grid[i][c] = false;
        
        // Check and destroy walls to the left and right
        for (int j = c - 1; j >= 0 && !grid[r][j]; --j) grid[r][j] = false;
        for (int j = c + 1; j < w && !grid[r][j]; ++j) grid[r][j] = false;
    }
    
    // Count remaining walls
    int ans = 0;
    for (const auto &row : grid) ans += count(row.begin(), row.end(), true);
    cout << ans << '\n';
    return 0;
}
