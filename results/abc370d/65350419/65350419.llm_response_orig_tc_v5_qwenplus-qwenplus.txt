```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    // For each column, keep track of walls in sets for binary search
    vector<set<int>> col(w);
    // For each row, same
    vector<set<int>> row(h);

    // Initialize all cells with walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            col[j].insert(i);
            row[i].insert(j);
        }
    }

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--;
        c--;

        if (row[r].count(c)) {
            // If there's a wall at the bomb location, remove it
            row[r].erase(c);
            col[c].erase(r);
        } else {
            // Otherwise, destroy first walls in four directions

            // Up
            auto it = col[c].lower_bound(r);
            if (it != col[c].begin()) {
                --it;
                int u = *it;
                col[c].erase(it);
                row[u].erase(c);
            }

            // Down
            it = col[c].lower_bound(r);
            if (it != col[c].end()) {
                int d = *it;
                col[c].erase(it);
                row[d].erase(c);
            }

            // Left
            it = row[r].lower_bound(c);
            if (it != row[r].begin()) {
                --it;
                int l = *it;
                row[r].erase(it);
                col[l].erase(r);
            }

            // Right
            it = row[r].lower_bound(c);
            if (it != row[r].end()) {
                int rgt = *it;
                row[r].erase(it);
                col[rgt].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += col[i].size();
    }
    cout << ans << "\n";

    return 0;
}
```
