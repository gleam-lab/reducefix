#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main(void) {
    int h, w, q;
    cin >> h >> w >> q;
    vector<set<int>> yoko(h), tate(w);
    rep(i, h) {
        rep(j, w) {
            yoko[i].insert(j);
            tate[j].insert(i);
        }
    }
    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--; c--;
        if (tate[c].count(r)) {
            tate[c].erase(r);
            yoko[r].erase(c);
        } else {
            // Erase vertical walls
            auto it = tate[c].lower_bound(r);
            if (it != tate[c].begin()) {
                it--;
                yoko[*it].erase(c);
                tate[c].erase(*it);
            }
            it = tate[c].lower_bound(r);
            if (it != tate[c].end()) {
                yoko[*it].erase(c);
                tate[c].erase(*it);
            }
            // Erase horizontal walls
            it = yoko[r].lower_bound(c);
            if (it != yoko[r].begin()) {
                it--;
                tate[*it].erase(r);
                yoko[r].erase(*it);
            }
            it = yoko[r].lower_bound(c);
            if (it != yoko[r].end()) {
                tate[*it].erase(r);
                yoko[r].erase(*it);
            }
        }
    }
    int ans = 0;
    rep(i, w) {
        ans += tate[i].size();
    }
    cout << ans << "\n";
    return 0;
}
