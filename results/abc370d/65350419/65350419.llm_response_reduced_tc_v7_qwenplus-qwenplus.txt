#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> row(w), col(h);
    
    // Initialize sets with all column indices for each row and vice versa
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    auto safe_erase = [&](set<int> &s, int val) {
        auto it = s.find(val);
        if (it != s.end()) {
            s.erase(it);
        }
    };

    while (q--) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if there is a wall at this cell
        if (row[c].count(r)) {
            safe_erase(row[c], r);
            safe_erase(col[r], c);
        } else {
            // Up
            auto rit = row[c].lower_bound(r);
            if (rit != row[c].begin()) {
                rit--;
                safe_erase(row[c], *rit);
                safe_erase(col[*rit], c);
            }

            // Down
            rit = row[c].lower_bound(r);
            if (rit != row[c].end()) {
                safe_erase(row[c], *rit);
                safe_erase(col[*rit], c);
            }

            // Left
            auto cit = col[r].lower_bound(c);
            if (cit != col[r].begin()) {
                cit--;
                safe_erase(col[r], *cit);
                safe_erase(row[*cit], r);
            }

            // Right
            cit = col[r].lower_bound(c);
            if (cit != col[r].end()) {
                safe_erase(col[r], *cit);
                safe_erase(row[*cit], r);
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    cout << ans << endl;

    return 0;
}
