```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int h, w, q;
    cin >> h >> w >> q;
    
    // Each row and column's walls are represented by a set of indices
    vector<set<int>> row(w), col(h);
    
    // Initialize all walls
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row[j].insert(i);
            col[i].insert(j);
        }
    }

    for (int qi = 0; qi < q; ++qi) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        if (row[c].count(r)) {
            // Wall exists at (r,c): destroy it
            row[c].erase(r);
            col[r].erase(c);
        } else {
            // No wall here: destroy the first walls in each direction

            // Up
            auto rit = row[c].lower_bound(r);
            if (rit != row[c].begin()) {
                --rit;
                row[c].erase(rit);
                col[*rit].erase(c);
            }

            // Down
            rit = row[c].lower_bound(r);
            if (rit != row[c].end()) {
                row[c].erase(rit);
                col[*rit].erase(c);
            }

            // Left
            auto cit = col[r].lower_bound(c);
            if (cit != col[r].begin()) {
                --cit;
                col[r].erase(cit);
                row[*cit].erase(r);
            }

            // Right
            cit = col[r].lower_bound(c);
            if (cit != col[r].end()) {
                col[r].erase(cit);
                row[*cit].erase(r);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < w; ++i) {
        ans += row[i].size();
    }
    cout << ans << "\n";
    return 0;
}
```
