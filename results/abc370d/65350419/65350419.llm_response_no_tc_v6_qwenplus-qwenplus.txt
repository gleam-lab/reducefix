#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int H, W, Q;
    cin >> H >> W >> Q;
    
    // For each column, keep track of remaining walls in that column (sorted)
    vector<set<int>> colWalls(W);
    // For each row, keep track of remaining walls in that row (sorted)
    vector<set<int>> rowWalls(H);
    
    // Initialize: all cells have walls
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            colWalls[j].insert(i);
            rowWalls[i].insert(j);
        }
    }
    
    while (Q--) {
        int R, C;
        cin >> R >> C;
        R--; C--;
        
        // Check if there's a wall at the bomb position
        if (rowWalls[R].count(C)) {
            // Destroy wall at the bomb position
            rowWalls[R].erase(C);
            colWalls[C].erase(R);
        } else {
            // Up
            auto it = colWalls[C].lower_bound(R);
            if (it != colWalls[C].begin()) {
                it--;
                int r = *it;
                rowWalls[r].erase(C);
                colWalls[C].erase(r);
            }

            // Down
            it = colWalls[C].lower_bound(R);
            if (it != colWalls[C].end()) {
                int r = *it;
                rowWalls[r].erase(C);
                colWalls[C].erase(r);
            }

            // Left
            auto jt = rowWalls[R].lower_bound(C);
            if (jt != rowWalls[R].begin()) {
                jt--;
                int c = *jt;
                colWalls[c].erase(R);
                rowWalls[R].erase(c);
            }

            // Right
            jt = rowWalls[R].lower_bound(C);
            if (jt != rowWalls[R].end()) {
                int c = *jt;
                colWalls[c].erase(R);
                rowWalls[R].erase(c);
            }
        }
    }

    // Count remaining walls
    int ans = 0;
    for (int i = 0; i < H; ++i) {
        ans += rowWalls[i].size();
    }
    cout << ans << endl;

    return 0;
}
