struct SegTree{
    int n;
    vector<int> tree;
    SegTree(vector<int>& arr){
        n = arr.size();
        tree.resize(4*n);
        build(arr, 0, n-1, 1);
    }
    void build(vector<int>& arr, int tl, int tr, int pos){
        if(tl == tr){
            tree[pos] = arr[tl];
            return;
        }
        int tm = (tl+tr)/2;
        build(arr, tl, tm, 2*pos);
        build(arr, tm+1, tr, 2*pos+1);
        tree[pos] = min(tree[2*pos], tree[2*pos+1]);
    }
    int getMin(int l, int r, int tl, int tr, int pos){
        if(l <= tl && r >= tr) return tree[pos];
        if(l > tr || r < tl) return INF;
        int tm = (tl+tr)/2;
        return min(getMin(l, r, tl, tm, 2*pos), getMin(l, r, tm+1, tr, 2*pos+1));
    }
};
