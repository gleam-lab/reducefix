```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main() {
    int h, w, q;
    cin >> h >> w >> q;
    
    vector<set<int>> row_left(h), row_right(h), col_up(w), col_down(w);

    // Initialize sets for each direction
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            row_left[i].insert(j);
            row_right[i].insert(j);
            col_up[j].insert(i);
            col_down[j].insert(i);
        }
    }

    auto erase_in_all = [&](int r, int c) {
        row_left[r].erase(c);
        row_right[r].erase(c);
        col_up[c].erase(r);
        col_down[c].erase(r);
    };

    auto get_next_or_end = [](const set<int>& s, int val) -> int {
        auto it = s.lower_bound(val);
        if (it == s.end()) return -1;
        return *it;
    };

    auto get_prev_or_begin = [](const set<int>& s, int val) -> int {
        auto it = s.lower_bound(val);
        if (it == s.begin()) return -1;
        --it;
        if (it == s.end()) return -1;
        return *it;
    };

    rep(qi, q) {
        int r, c;
        cin >> r >> c;
        r--; c--;

        // Check if the cell has a wall
        if (col_down[c].count(r)) {
            erase_in_all(r, c);
        } else {
            // Up
            auto up = get_prev_or_begin(col_up[c], r);
            if (up != -1) {
                erase_in_all(up, c);
            }

            // Down
            auto down = get_next_or_end(col_down[c], r);
            if (down != -1) {
                erase_in_all(down, c);
            }

            // Left
            auto left = get_prev_or_begin(row_left[r], c);
            if (left != -1) {
                erase_in_all(r, left);
            }

            // Right
            auto right = get_next_or_end(row_right[r], c);
            if (right != -1) {
                erase_in_all(r, right);
            }
        }
    }

    int ans = 0;
    for (int i = 0; i < h; ++i) {
        ans += row_left[i].size();
    }

    cout << ans << endl;
    return 0;
}
```
