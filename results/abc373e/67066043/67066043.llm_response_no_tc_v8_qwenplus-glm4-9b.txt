#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n,m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    vector<ll> votes_needed(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        votes_needed[i] = max(0LL, m - 1 - a[i]);
    }
    vector<pair<ll, int>> candidates(a.begin(), a.end());

    // Sort candidates based on their current votes, then by their index
    sort(candidates.begin(), candidates.end());

    vector<ll> prefix_sum(n);
    prefix_sum[0] = 0;
    for(int i = 1; i < n; ++i) {
        prefix_sum[i] = prefix_sum[i-1] + votes_needed[candidates[i].second];
    }

    // Binary search to find the minimum votes needed for the candidates
    ll low = 0, high = k;
    while(low <= high) {
        ll mid = (low + high) / 2;
        ll current_k = 0;
        bool valid = true;
        for(int i = n - 1; i >= 0 && current_k <= mid; --i) {
            current_k += votes_needed[candidates[i].second];
            if(current_k > mid) {
                valid = false;
                break;
            }
        }
        if(valid && prefix_sum[candidates[n-1].second] <= mid) {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    // Output the result
    for(int i = 0; i < n; ++i) {
        if(votes_needed[i] > 0 && prefix_sum[i] <= low) {
            cout << low - prefix_sum[i] << ' ';
        } else {
            cout << (votes_needed[i] > 0 ? -1 : 0) << ' ';
        }
    }

    return 0;
}
