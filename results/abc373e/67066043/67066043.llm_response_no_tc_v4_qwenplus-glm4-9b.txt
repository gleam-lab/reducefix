#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    // Calculate the number of votes still uncounted
    ll remaining_votes = k - accumulate(a.begin(), a.end(), 0LL);

    // Sort candidates based on their current votes in descending order
    sort(a.begin(), a.end(), greater<ll>());

    // Initialize the answer vector
    vector<ll> ans(n, -1);

    // Iterate over each candidate
    for (int i = 0; i < n; i++) {
        // Calculate votes needed to guarantee victory
        ll votes_needed = 0;
        for (int j = 0; j < m && j < n; j++) {
            if (j == i) {
                votes_needed = j * (a[j] + 1);
            } else {
                votes_needed += a[j] + 1;
            }
        }

        // Check if the candidate already has enough votes or if it's impossible
        if (a[i] >= votes_needed) {
            ans[i] = 0;
        } else if (remaining_votes + a[i] < votes_needed) {
            ans[i] = -1;
        } else {
            // Calculate minimum additional votes needed
            ans[i] = votes_needed - a[i] - remaining_votes;
        }
    }

    // Output the results
    for (ll i : ans) {
        cout << i << ' ';
    }
    cout << '\n';

    return 0;
}
