```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    // Calculate the current maximum votes any candidate can get without winning
    ll max_votes_without_winning = *max_element(a.begin(), a.end()) + k / (n - m);

    // Determine the minimum additional votes needed for each candidate
    vector<ll> ans(n, -1);
    for(int i = 0; i < n; ++i) {
        if(a[i] > max_votes_without_winning) continue; // Already won or impossible to win
        ll additional_votes_needed = max(max_votes_without_winning - a[i], 0LL);
        ans[i] = additional_votes_needed;
    }

    // Output the results
    for(ll vote : ans) {
        cout << vote << " ";
    }
    return 0;
}
```
