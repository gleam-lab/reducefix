### Approaches

2. **Binary Search for Minimum Votes**: For each candidate, we can use binary search to find the minimum number of additional votes they need such that even if all other candidates receive all remaining votes, the candidate i still has enough votes to be elected.
3. **Prefix Sums and Sorting**: Sort the votes to efficiently compute how many candidates would have more votes than candidate i after adding X votes. This helps in determining if the candidate meets the election condition.

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    ll total = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        total += A[i];
    }
    ll remaining = K - total;

    if (M == N) {
        for (int i = 0; i < N; ++i) {
            cout << "0 ";
        }
        cout << endl;
        return 0;
    }

    vector<ll> sortedA = A;
    sort(sortedA.begin(), sortedA.end());
    vector<ll> prefix(N + 1);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = prefix[i] + sortedA[i];
    }

    vector<ll> ans(N, -1);
    for (int i = 0; i < N; ++i) {
        ll low = 0, high = remaining;
        ll best = -1;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll newVote = A[i] + mid;
            int pos = upper_bound(sortedA.begin(), sortedA.end(), newVote) - sortedA.begin();
            int needed = N - M;
            if (pos > needed) {
                best = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        ans[i] = best;
    }

    for (int i = 0; i < N; ++i) {
        if (ans[i] != -1 && ans[i] <= remaining) {
            cout << ans[i] << " ";
        } else {
            cout << "-1 ";
        }
    }
    cout << endl;

    return 0;
}
```

4. **Binary Search for Each Candidate**: For each candidate, binary search is used to find the minimum additional votes (X) such that, even if all other candidates receive all remaining votes, the candidate's total votes (A[i] + X) ensure that no more than M-1 candidates have more votes.
5. **Output**: The results for each candidate are printed, ensuring that the solution is within the remaining votes (K - total votes). If no solution exists within the constraints, -1 is printed for that candidate.

This approach efficiently checks each candidate's condition using binary search and sorting, ensuring optimal performance even for large input sizes.
