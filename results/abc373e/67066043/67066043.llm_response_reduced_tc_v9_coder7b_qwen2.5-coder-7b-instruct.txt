#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i=0; i<n; i++) {
        cin >> a[i];
        k -= a[i];
    }

    vector<ll> b(a);
    sort(b.begin(), b.end());
    vector<ll> prefix(n+1);
    for(int i=0; i<n; i++) {
        prefix[i+1] = b[i] + prefix[i];
    }

    vector<ll> ans(n, -1);
    for(int i=0; i<n; i++) {
        if(m == n || i == n-m) {
            ans[i] = 0;
            continue;
        }
        
        ll target = k / (n-m-1);
        if(target < a[i]) {
            ans[i] = -1;
            continue;
        }

        ll need = target - a[i];
        if(need == 0) {
            ans[i] = 0;
            continue;
        }

        ll low = 0, high = k;
        while(low < high) {
            ll mid = (low + high) / 2;
            ll sum = mid * (m + 1) - prefix[m+1] + prefix[i+1];
            if(sum >= need) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        ans[i] = low;
    }

    for(ll x : ans) cout << x << " ";
    return 0;
}
