#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

int main() {
    int n, m, k;
    cin >> n >> m >> k;
    vector<long long> A(n);
    long long total_votes = 0;
    
    // Read votes and calculate total votes
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
        total_votes += A[i];
    }
    
    // Sort candidates by their votes in descending order
    vector<pair<long long, int>> votes(A.size());
    for (int i = 0; i < n; ++i) {
        votes[i] = {A[i], i};
    }
    sort(votes.rbegin(), votes.rend());
    
    // Calculate votes needed for the last candidate in the sorted list to win
    long long votes_needed_to_win = n - m;
    if (votes[0].first >= votes_needed_to_win) {
        // If the top candidate already has enough votes, everyone else needs 0 votes
        for (int i = 1; i < n; ++i) {
            cout << 0 << ' ';
        }
        cout << endl;
        return 0;
    }
    
    // Calculate remaining votes after the top candidate has votes_needed_to_win
    long long remaining_votes = k - total_votes + votes_needed_to_win;
    
    // Calculate votes each candidate needs to guarantee they are in the top m
    vector<long long> remaining_votes_needed(n, -1);
    for (int i = 0; i < n; ++i) {
        if (votes[i].first >= votes_needed_to_win) {
            remaining_votes_needed[i] = 0;
        } else {
            // The votes needed is the votes to reach the minimum required plus any remaining votes
            long long votes_to_match = votes_needed_to_win - votes[i].first + 1;
            if (i < n - m) {
                // If the candidate is not in the top m, they need to catch up the top m minus current index
                votes_to_match += votes[0].first - votes[i].first;
            }
            if (votes_to_match <= remaining_votes) {
                remaining_votes_needed[i] = votes_to_match;
            } else {
                remaining_votes_needed[i] = -1;
            }
        }
    }
    
    // Output the answer
    for (int i = 0; i < n; ++i) {
        cout << remaining_votes_needed[i] << ' ';
    }
    cout << endl;
    return 0;
}
