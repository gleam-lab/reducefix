#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    vector<ll> res(N, -1);

    // For each candidate, determine the minimum number of additional votes needed to guarantee election.
    for (int i = 0; i < N; ++i) {
        ll my_votes = A[i];

        // Make a copy and simulate adding enough votes to this candidate to try.
        vector<ll> temp = A;
        temp[i] = my_votes;

        // We will binary search on X: how many additional votes to give to candidate i
        ll low = 0, high = K + 1;
        while (low <= high) {
            ll mid = (low + high) / 2;
            temp[i] = A[i] + mid;

            // Sort in descending order
            vector<ll> sorted = temp;
            sort(sorted.rbegin(), sorted.rend());

            // Count how many candidates have strictly more votes than temp[i]
            int count = 0;
            for (int j = 0; j < N; ++j) {
                if (sorted[j] > temp[i]) ++count;
            }

            // Candidate i is elected if fewer than M candidates have strictly more votes
            if (count < M) {
                res[i] = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        if (res[i] > K) res[i] = -1;
    }

    for (auto x : res) cout << x << " ";
    cout << "\n";
}
