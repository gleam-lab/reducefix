#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // Calculate the votes each candidate needs to guarantee their victory
    vector<ll> votes_needed(n, llONG_MAX);
    vector<ll> current_votes(0);
    for(int i = 0; i < n; i++) {
        // Add the current votes to the sum
        current_votes.push_back(a[i]);
        ll sum_votes = accumulate(current_votes.begin(), current_votes.end(), 0LL);
        
        // If this candidate is the i-th from the top, it needs m votes to guarantee victory
        if(i + 1 >= m) {
            // Calculate the votes needed for this candidate to be the top remaining candidate
            votes_needed[i] = (m - 1) * (sum_votes - current_votes[i]) + current_votes[i];
        }
    }
    
    // Sort the votes needed to find the minimum votes needed to guarantee victory for any candidate
    sort(votes_needed.begin(), votes_needed.end());

    // Subtract the votes already received by the top candidates
    for(int i = 0; i < n; i++) {
        votes_needed[i] -= a[i];
    }
    
    // Calculate the votes left to be distributed
    ll votes_left = k - accumulate(a.begin(), a.end(), 0LL);
    
    // Determine the minimum votes each candidate needs to guarantee victory
    vector<ll> result(n, -1);
    for(int i = 0; i < n; i++) {
        if(votes_needed[i] <= votes_left) {
            result[i] = votes_needed[i] - a[i];
        } else {
            result[i] = -1;
        }
    }
    
    // If the minimum votes for any candidate to guarantee victory is greater than votes left, print -1 for all candidates
    if(*min_element(result.begin(), result.end()) == llONG_MAX) {
        for(int i = 0; i < n; i++) {
            result[i] = -1;
        }
    }
    
    for(ll x : result) cout << x << ' ';
    return 0;
}
