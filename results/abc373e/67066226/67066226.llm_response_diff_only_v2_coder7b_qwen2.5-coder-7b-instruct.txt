#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m;
    ll k;
    cin >> n >> m >> k;
    
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }
    
    vector<int> ord(n);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return a[i] < a[j]; });
    
    vector<ll> pref(n + 1);
    for(int i = 0; i < n; ++i) {
        pref[i + 1] = a[i] + pref[i];
    }
    
    vector<ll> ans(n, -1);
    for(int i = 0; i < n; ++i) {
        ll l = 0, r = k + 1;
        while(l <= r) {
            ll mid = (l + r) / 2;
            ll rid = upper_bound(all(a), a[i] + mid) - a.begin();
            ll lid = max(i - (m - 1), 0);
            ll cnt = 0;
            
            if(rid > lid) {
                cnt += (rid - lid) * (a[i] + mid) - (pref[rid] - pref[lid]);
            }
            
            if(lid <= i && i < rid) {
                cnt--;
            } else {
                cnt += mid;
            }
            
            if(cnt > k) {
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        
        if(l == k + 1) {
            ans[ord[i]] = -1;
        } else {
            ans[ord[i]] = l;
        }
    }
    
    for(auto i : ans) cout << i << ' ';
    
    return 0;
}
