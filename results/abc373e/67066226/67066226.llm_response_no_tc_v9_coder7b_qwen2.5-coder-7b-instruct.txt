#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }

    // Sort indices based on the number of votes in descending order
    vector<int> ord(n);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return a[i] > a[j]; });

    // Calculate prefix sums of sorted vote counts
    vector<ll> pref(n + 1);
    for(int i = 0; i < n; i++) {
        pref[i + 1] = a[ord[i]] + pref[i];
    }

    vector<ll> ans(n, -1);

    for(int i = 0; i < n; i++) {
        ll needed_votes = 0;
        for(int j = 0; j < m; j++) {
            if(a[ord[j]] > a[ord[i]]) {
                needed_votes++;
            }
        }
        
        if(needed_votes == 0) {
            ans[ord[i]] = 0;
            continue;
        }

        ll left = 0, right = k + 1;
        while(left < right) {
            ll mid = (left + right) / 2;
            ll count = 0;

            for(int j = 0; j < n; j++) {
                if(j != i && a[ord[j]] + mid > a[ord[i]]) {
                    count++;
                }
            }

            if(count < m) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }

        if(left <= k) {
            ans[ord[i]] = left;
        }
    }

    for(auto i : ans) cout << i << ' ';
    return 0;
}
