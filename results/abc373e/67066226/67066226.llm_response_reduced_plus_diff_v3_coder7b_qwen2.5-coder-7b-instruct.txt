To solve this problem, we need to determine the minimum number of additional votes each candidate needs to ensure they will win regardless of how the remaining votes are distributed among the other candidates. Here's a step-by-step approach to achieve this:

1. **Calculate Current Votes**: First, calculate the current total votes received by each candidate.
2. **Sort Candidates**: Sort the candidates based on their current vote counts.
3. **Determine Thresholds**: For each candidate, determine the threshold above which they can guarantee a win. This involves checking how many votes other candidates could potentially receive and ensuring that the current candidate's votes plus additional votes exceed the maximum votes any other candidate could receive.
4. **Binary Search**: Use binary search to efficiently find the minimum number of additional votes needed for each candidate to guarantee a win.

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    ll k;
    cin >> n >> m >> k;

    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    vector<pair<ll, int>> candidates(n);
    for (int i = 0; i < n; ++i) {
        candidates[i].first = a[i];
        candidates[i].second = i;
    }

    sort(candidates.begin(), candidates.end());

    vector<ll> ans(n, -1);
    for (int i = 0; i < n; ++i) {
        ll target = candidates[i].first + 1;
        int idx = i;
        while (idx < n && candidates[idx].first == target) {
            ++idx;
        }
        if (idx < n) {
            ll need = target - candidates[i].first;
            if (need <= k) {
                ans[candidates[i].second] = need;
                k -= need;
            } else {
                break;
            }
        } else {
            ans[candidates[i].second] = 0;
        }
    }

    for (auto x : ans) {
        cout << x << " ";
    }
    cout << endl;

    return 0;
}
