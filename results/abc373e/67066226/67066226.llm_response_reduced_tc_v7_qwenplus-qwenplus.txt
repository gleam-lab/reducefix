#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    // Create list of candidate indices sorted by current votes
    vector<int> ord(n);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return a[i] < a[j]; });

    // Sorted list of current votes
    vector<ll> b = a;
    sort(all(b));

    // Prefix sum array
    vector<ll> pref(n + 1);
    for (int i = 0; i < n; ++i) {
        pref[i + 1] = pref[i] + b[i];
    }

    vector<ll> ans(n, -1);

    for (int idx = 0; idx < n; ++idx) {
        int i = ord[idx]; // Current candidate index
        ll low = 0, high = k + 1;
        ll res = -1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll target = b[idx] + mid;

            // Binary search to find how many candidates have votes > target
            ll cnt = 0;
            ll pos = lower_bound(b.begin(), b.end(), target + 1) - b.begin();
            ll topCandidates = n - pos;

            if (topCandidates < m) {
                // We already are in the top M candidates
                ans[i] = 0;
                break;
            }

            // Need to make sure we can limit number of candidates > us to < M
            // So we need to overtake enough candidates
            ll requiredTops = m;
            ll start = max(0LL, n - requiredTops);
            
            // We need to raise our vote count so that at most M-1 candidates have more votes
            // Take the M-1th candidate from the top and ensure we're at least equal or higher
            if (start >= n) {
                ans[i] = 0;
                break;
            }
            
            ll neededVotes = b[n - m] + 1 - a[i];
            neededVotes = max(0LL, neededVotes);
            if (neededVotes <= k) {
                ans[i] = neededVotes;
            } else {
                ans[i] = -1;
            }
            break;
        }
    }

    for (auto x : ans) {
        cout << x << ' ';
    }
    cout << '\n';
}
