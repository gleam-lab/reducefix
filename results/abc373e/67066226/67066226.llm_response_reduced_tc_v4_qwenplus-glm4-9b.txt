#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }
    vector<ll> sorted_a(a);
    sort(sorted_a.begin(), sorted_a.end(), greater<ll>());
    vector<ll> prefix(n + 1);
    prefix[0] = 0;
    for (int i = 0; i < n; i++) {
        prefix[i + 1] = prefix[i] + sorted_a[i];
    }
    
    vector<ll> ans(n, LLONG_MAX);
    for (int i = 0; i < n; i++) {
        ll required_votes = k + 1 - a[i];
        if (required_votes < 0) required_votes = 0;
        ll max_votes_needed = sorted_a[i] * (n - m);
        if (required_votes >= max_votes_needed) {
            ans[i] = 0;
        } else {
            ll total_votes = k;
            ll remaining_candidates = n - i - 1;
            ll votes_needed = max_votes_needed - required_votes;
            while (total_votes < votes_needed && remaining_candidates > 0) {
                votes_needed -= sorted_a[i];
                total_votes += sorted_a[i];
                remaining_candidates--;
            }
            if (total_votes < votes_needed) {
                ans[i] = -1;
            } else {
                ans[i] = votes_needed + required_votes;
            }
        }
    }
    
    for (ll val : ans) cout << val << ' ';
    return 0;
}
