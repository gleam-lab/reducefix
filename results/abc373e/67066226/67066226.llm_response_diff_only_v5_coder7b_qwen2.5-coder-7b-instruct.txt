        ll min_votes_needed = 0;

        while(l <= r) {

            ll count = 0;
            for(int j = 0; j < n; ++j) {

            }

            // If the count is less than m, the candidate wins
            if(count < m) {
                r = mid - 1;
                min_votes_needed = mid;

                l = mid + 1;
