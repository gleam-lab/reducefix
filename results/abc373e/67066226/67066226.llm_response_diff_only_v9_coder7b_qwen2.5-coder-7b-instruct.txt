#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }

    // Calculate the current total number of votes needed to win
    ll total_votes_needed = 0;
    for(int i = 0; i < n - m; i++) {
        total_votes_needed += a[i];
    }

    // Sort the votes in descending order
    sort(a.rbegin(), a.rend());

    vector<ll> ans(n, -1);
    for(int i = 0; i < n; i++) {
        // If the current candidate can already win, set the answer to 0
        if(i >= n - m) {
            ans[i] = 0;
        } else {
            // Calculate the number of votes needed for the current candidate to win
            ll votes_needed = total_votes_needed - a[i] + 1;

            // If the remaining votes are not enough, set the answer to -1
            if(votes_needed > k) {
                ans[i] = -1;
            } else {
                ans[i] = votes_needed;
            }
        }
    }

    for(auto i : ans) cout << i << ' ';
    return 0;
}
