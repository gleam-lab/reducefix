```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        K -= A[i];
    }

    // ord[i] = original index of the candidate with i-th smallest current votes
    vector<int> ord(N);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return A[i] < A[j]; });

    // sorted copy of A
    vector<ll> B = A;
    sort(all(B));

    // prefix sums of B
    vector<ll> pref(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        pref[i + 1] = pref[i] + B[i];
    }

    vector<ll> res(N, -1);

    for (int idx = 0; idx < N; ++idx) {
        ll low = 0, high = K + 1;
        ll answer = -1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll target = B[idx] + mid;

            // Find how many candidates have strictly more votes than target in worst-case
            // Worst case: remaining votes go to top M candidates who are below target
            int left = 0, right = N;
            while (left < right) {
                int m = (left + right) / 2;
                if (B[m] > target)
                    right = m;
                else
                    left = m + 1;
            }
            int cnt_better = N - left;

            // We need at most M-1 candidates to have more votes than us
            if (cnt_better < M) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        res[ord[idx]] = (answer <= K ? answer : -1);
    }

    for (ll x : res) cout << x << ' ';
    cout << '\n';

    return 0;
}
```
