```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (ll& a : A) {
        cin >> a;
        K -= a;
    }

    // ord[i] = original index of the candidate whose current votes are the i-th smallest
    vector<int> ord(N);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return A[i] < A[j]; });

    // sortedA[i] = number of votes of the candidate with i-th smallest vote count
    vector<ll> sortedA = A;
    sort(all(sortedA));

    vector<ll> prefix(N + 1);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = prefix[i] + sortedA[i];
    }

    vector<ll> result(N, -1);

    for (int idx = 0; idx < N; ++idx) {
        int orig_idx = ord[idx];
        ll low = 0;
        ll high = K + 1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll new_score = A[orig_idx] + mid;

            // Binary search to find how many candidates have strictly more votes than new_score
            int cnt_more = N - (ll)(upper_bound(all(sortedA), new_score) - sortedA.begin());

            if (cnt_more < M) {
                // This many additional votes is sufficient
                high = mid - 1;
            } else {
                // Need more votes
                low = mid + 1;
            }
        }

        if (low > K)
            result[orig_idx] = -1;
        else
            result[orig_idx] = low;
    }

    for (ll x : result) {
        cout << x << " ";
    }
    cout << "\n";

    return 0;
}
```
