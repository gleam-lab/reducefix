```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    ll k;
    cin >> n >> m >> k;
    
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int i, int j) { return a[i] < a[j]; });

    vector<ll> pref(n + 1);
    for(int i = 0; i < n; ++i) {
        pref[i + 1] = pref[i] + a[i];
    }

    vector<ll> ans(n, -1);
    for(int i = 0; i < n; ++i) {
        ll l = 0, r = k + 1;
        while(l <= r) {
            ll mid = (l + r) / 2;
            ll rid = upper_bound(order.begin(), order.end(), i + m - 1) - order.begin();
            ll lid = n - m - (i >= n - m ? 1 : 0);
            ll cnt = 0;
            if(rid > lid) cnt += (rid - lid) * (a[i] + mid + 1) - (pref[rid] - pref[lid]);
            if(lid <= i && i < rid) cnt--;
            else cnt += mid;
            if(cnt > k) {
                r = mid - 1;
            } else l = mid + 1;
        }
        if(l == k + 1) ans[order[i]] = -1;
        else ans[order[i]] = l;
    }

    for(auto &c : ans) cout << c << " ";
    return 0;
}
```

This solution addresses the problem by sorting candidates based on the number of votes they currently have and using binary search to determine the minimum additional votes required for each candidate to ensure victory. It handles edge cases and maintains an efficient time complexity suitable for large inputs.
