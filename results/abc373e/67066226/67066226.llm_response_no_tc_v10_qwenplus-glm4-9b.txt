#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

int main() {
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    ll sum_votes = 0;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        sum_votes += a[i];
    }
    k -= sum_votes;

    vector<pair<ll, int>> candidates(n);
    for (int i = 0; i < n; ++i) {
        candidates[i] = {a[i], i};
    }
    sort(candidates.begin(), candidates.end());

    vector<ll> prefix(n);
    prefix[0] = candidates[0].first;
    for (int i = 1; i < n; ++i) {
        prefix[i] = prefix[i - 1] + candidates[i].first;
    }

    vector<ll> ans(n, LLONG_MAX);

    // Calculate minimum votes needed for each candidate to win
    for (int i = 0; i < n; ++i) {
        ll current_votes = candidates[i].first;
        ll required_votes = m - 1;
        ll votes_needed = 0;

        if (i + 1 < n && prefix[i + 1] - prefix[i] == required_votes) {
            votes_needed = k - (prefix[i + 1] - prefix[i]);
        } else {
            votes_needed = k - (prefix[i + required_votes] - prefix[i]);
        }

        if (!votes_needed || votes_needed > k) {
            ans[candidates[i].second] = -1;
        } else {
            ans[candidates[i].second] = votes_needed;
        }
    }

    for (ll vote : ans) {
        cout << (vote == LLONG_MAX ? -1 : vote) << " ";
    }
    cout << endl;

    return 0;
}
