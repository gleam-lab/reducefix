#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

vector<int> minVotesToWin(int n, int m, int k, vector<int>& a) {
    vector<int> res(n, -1);
    vector<int> prefix(n + 1);
    
    // Calculate prefix sum array for the votes received by each candidate
    for(int i = 1; i <= n; ++i) {
        prefix[i] = prefix[i - 1] + a[i - 1];
    }

    // Iterate over each candidate
    for(int i = 0; i < n; ++i) {
        int votes_needed = 0;

        // Check if the current candidate can win without additional votes
        bool can_win_without_votes = true;
        for(int j = 0; j < n; ++j) {
            if(j != i) {
                if(a[j] > a[i]) {
                    can_win_without_votes = false;
                    break;
                }
            }
        }
        
        if(can_win_without_votes) {
            res[i] = 0;
            continue;
        }

        // Binary search for the minimum number of additional votes needed
        int low = 0, high = k - prefix[n];
        while(low <= high) {
            int mid = low + (high - low) / 2;
            
            int count = 0;
            for(int j = 0; j < n; ++j) {
                if(j != i) {
                    if(a[j] + mid > a[i]) {
                        count++;
                    }
                }
            }

            if(count < m) {
                high = mid - 1;
                votes_needed = mid;
            } else {
                low = mid + 1;
            }
        }

        res[i] = votes_needed;
    }

    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    cin >> n >> m >> k;
    vector<int> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    vector<int> result = minVotesToWin(n, m, k, a);

    for(int i = 0; i < n; ++i) {
        cout << result[i] << " ";
    }

    return 0;
}
