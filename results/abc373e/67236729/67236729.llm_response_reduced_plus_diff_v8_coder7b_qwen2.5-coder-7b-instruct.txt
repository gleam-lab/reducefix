#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using pll = pair<ll, ll>;

constexpr int N = 2e5 + 10;
int a[N], b[N], sm[N];

bool check(ll o, ll d, int m, int n, int k) {
    int p = upper_bound(b + 1, b + n + 1, o + d) - b;
    int up = n - p + 1;
    if (up >= m) return false;
    ll S = (sm[p - (m - up)] - sm[p]);
    if (b[p - (m - up)] <= o && o <= b[p - 1]) {
        S -= o;
        S += b[p - (m - up) - 1];
    }
    return (o + d + 1) * (m - up) - S > k - d;
}

void solve() {
    int n, m, k;
    cin >> n >> m >> k;
    if (n == m) {
        rep0(i, n) cout << 0 << " ";
        cout << "\n";
        return;
    }
    rep1(i, n) cin >> a[i];
    rep1(i, n) {
        b[i] = a[i];
        k -= a[i];
    }
    sort(b + 1, b + n + 1);
    for (int i = n; i >= 1; i--) sm[i] = sm[i + 1] + b[i];
    rep1(i, n) {
        ll l = 0, r = k, ans = -1;
        while (l <= r) {
            ll mid = l + r >> 1;
            if (check(a[i], mid, m, n, k)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        cout << ans << " ";
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(0);
    int T = 1;
    // cin >> T;
    while (T--) solve();
    return 0;
}
