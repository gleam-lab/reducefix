#include <iostream>
#include <vector>
#include <algorithm>
#include <cassert>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    i64 k;
    cin >> n >> m >> k;

    vector<i64> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    vector<i64> b = a;
    sort(b.begin(), b.end());

    vector<i64> sm(n + 1, 0);
    for (int i = n - 1; i >= 0; --i) {
        sm[i] = sm[i + 1] + b[i];
    }

    auto check = [&](i64 o, i64 d) -> bool {
        int p = upper_bound(b.begin(), b.end(), o + d) - b.begin();
        int up = n - p + 1;
        if (up >= m) return true;
        i64 S = sm[p] - sm[max(p - (m - up), 0)];
        return (o + d + 1) * (m - up) <= S;
    };

    for (int i = 0; i < n; ++i) {
        i64 l = 0, r = k, ans = -1;
        while (l <= r) {
            i64 mid = l + (r - l) / 2;
            if (check(a[i], mid)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        cout << ans << " ";
    }
    cout << "\n";

    return 0;
}
