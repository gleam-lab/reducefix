        suffix_sum[i] = suffix_sum[i+1] + a[i];

        if(m == 1){
            result[i] = 0;
            continue;
        }

        ll need_votes = 0;
        ll higher_votes = 0;
        
        for(ll j = i+1; j < n; j++){
            if(a[j] > a[i])
                higher_votes++;
        }
        
        if(higher_votes < m-1){
            result[i] = 0;
            continue;
        }

        for(ll j = i+1; j < n; j++){
            if(a[j] > a[i]){
                need_votes = max(need_votes, a[j] - a[i]);
            }
        }

        need_votes++;
        result[i] = need_votes;
    }
