#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

constexpr int N = 200000 + 10;

i64 a[N], sorted[N], prefix_sum[N];

// Function to determine if candidate with current votes `base` can be guaranteed victory with `extra` additional votes
bool is_valid(i64 base, i64 extra, i64 n, i64 m, i64 remaining_votes) {
    i64 total = base + extra;

    // Binary search to find how many candidates have votes > total
    i64 left = 0, right = n;
    while (left < right) {
        i64 mid = (left + right) / 2;
        if (sorted[mid] > total)
            left = mid + 1;
        else
            right = mid;
    }
    i64 greater_count = n - left;

    if (greater_count < m)
        return true;

    // Need to make sure we can raise enough of the lower candidates to not exceed M
    i64 needed = greater_count - m + 1;
    i64 min_increase = (total + 1) * needed;
    for (i64 i = n - needed; i < n; ++i)
        min_increase -= sorted[i];
    return min_increase <= remaining_votes;
}

        k -= a[i];
    }

    for (i64 i = 0; i < n; ++i)
        sorted[i] = a[i];
    sort(sorted, sorted + n, greater<i64>());

    prefix_sum[n] = 0;
    for (i64 i = n - 1; i >= 0; --i)
        prefix_sum[i] = prefix_sum[i + 1] + sorted[i];

        i64 low = 0, high = k, ans = -1;

            i64 mid = (low + high) / 2;

            } else {
                low = mid + 1;
            }

}
