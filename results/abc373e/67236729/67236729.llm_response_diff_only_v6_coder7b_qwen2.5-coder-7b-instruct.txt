#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve() {
    ll n, m, k;
    cin >> n >> m >> k;
    
    vector<ll> a(n + 1);
    for(int i = 1; i <= n; ++i) cin >> a[i];

    // Sort the vote counts
    sort(a.begin() + 1, a.end());

    // Calculate the prefix sums
    vector<ll> pre(n + 1);
    pre[0] = 0;
    for(int i = 1; i <= n; ++i) pre[i] = pre[i - 1] + a[i];

    // Function to check if candidate i can win with at least x more votes
    auto can_win = [&](int i, ll x) {
        ll needed = 0;
        for(int j = i + 1; j <= n; ++j) {
            needed += max(0LL, a[j] - a[i] + x);
        }
        return needed <= k - (pre[n] - pre[i]);
    };

    // Binary search to find the minimum number of votes needed for candidate i
    for(int i = 1; i <= n; ++i) {
        ll left = 0, right = k, ans = -1;
        while(left <= right) {
            ll mid = (left + right) / 2;
            if(can_win(i, mid)) {
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        cout << ans << " ";
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int t = 1;
    cin >> t;
    while(t--) {
        solve();
    }

    return 0;
}
