2. **Calculate Needed Votes**: For each candidate, calculate the minimum number of additional votes needed based on the difference between the current vote count and the votes of other candidates.
3. **Binary Search**: Use binary search to find the exact number of additional votes needed without exceeding the remaining votes.

    }

    sort(a.rbegin(), a.rend());

    vector<int> prefix_sum(n + 1);
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + a[i - 1];
    }

    vector<int> results(n, 0);

        int low = 0, high = k;
        while (low < high) {
            int mid = (low + high) / 2;
            int remaining_votes = k - mid;
            int votes_needed = prefix_sum[min(m - 1, i + 1)] - prefix_sum[i + 1];

            if (votes_needed <= remaining_votes) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        results[i] = low - a[i];
        if (results[i] < 0) results[i] = 0;
    }

    for (int i = 0; i < n; ++i) {

    }

    return 0;
}
