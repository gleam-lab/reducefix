#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

const int N = 2e5 + 10;

void solve() {
    int n, m, k;
    cin >> n >> m >> k;
    vector<i64> a(n);
    for (auto &x : a) cin >> x;

    i64 total_remaining = k;
    vector<int> ids(n);
    iota(ids.begin(), ids.end(), 0);
    sort(ids.begin(), ids.end(), [&](int i, int j) { return a[i] > a[j]; });

    vector<i64> res(n, -1);
    vector<i64> sorted_a = a;
    sort(sorted_a.rbegin(), sorted_a.rend());

    // prefix sum of top M-1 candidates' votes
    vector<i64> pre_sum(n + 1);
    for (int i = 1; i <= n; ++i)
        pre_sum[i] = pre_sum[i - 1] + sorted_a[i - 1];

    for (int i = 0; i < n; ++i) {
        int rank = lower_bound(sorted_a.begin(), sorted_a.end(), a[i], greater<>()) - sorted_a.begin();
        if (rank < m) {
            res[i] = 0;
            continue;
        }

        i64 need_votes = max(0LL, sorted_a[m - 1] - a[i] + 1);
        if (need_votes <= total_remaining) {
            res[i] = need_votes;
        }
    }

    for (int i = 0; i < n; ++i) {
        if (res[i] == -1) {
            cout << -1 << " ";
        } else {
            cout << res[i] << " ";
        }
    }
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
