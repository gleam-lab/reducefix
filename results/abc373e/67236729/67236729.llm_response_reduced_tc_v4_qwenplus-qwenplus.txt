#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

constexpr int N = 2e5 + 10;

void solve() {
    i64 n, m, k;
    cin >> n >> m >> k;
    
    vector<i64> a(n);
    for (auto &x : a) {
        cin >> x;
    }
    
    i64 remaining = k;
    for (auto x : a) {
        remaining -= x;
    }
    
    if (m >= n) {
        for (int i = 0; i < n; ++i) {
            cout << 0 << " ";
        }
        cout << "\n";
        return;
    }
    
    vector<i64> sorted_a = a;
    sort(sorted_a.begin(), sorted_a.end());
    
    // Prefix sum from the end (largest elements)
    vector<i64> prefix_sum(n+1);
    for (int i = n-1; i >= 0; --i) {
        prefix_sum[i] = prefix_sum[i+1] + sorted_a[i];
    }
    
    for (int i = 0; i < n; ++i) {
        i64 candidate_votes = a[i];
        
        // Binary search for the minimum X
        i64 left = 0, right = remaining, ans = -1;
        
        while (left <= right) {
            i64 mid = (left + right) / 2;
            i64 total = candidate_votes + mid;
            
            // Find how many candidates have votes > total
            int idx = upper_bound(sorted_a.begin(), sorted_a.end(), total) - sorted_a.begin();
            int count = n - idx;
            
            // We need less than M candidates with more votes
            if (count < m) {
                // Even if all remaining votes go to the top M-1 candidates
                int need_top = m - count - 1;
                if (idx <= need_top) {
                    ans = mid;
                    right = mid - 1;
                } else {
                    i64 required_extra = 0;
                    // Calculate how many votes are needed to prevent the next M-count-1 candidates from beating us
                    for (int j = idx; j < idx + (need_top); ++j) {
                        required_extra += max(0LL, sorted_a[j] - total + 1);
                    }
                    if (required_extra <= mid) {
                        ans = mid;
                        right = mid - 1;
                    } else {
                        left = mid + 1;
                    }
                }
            } else {
                left = mid + 1;
            }
        }
        
        if (ans == -1) {
            cout << -1 << " ";
        } else {
            cout << ans << " ";
        }
    }
    
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    
    return 0;
}
