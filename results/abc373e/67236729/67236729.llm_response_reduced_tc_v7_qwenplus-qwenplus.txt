#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

constexpr int N = 2e5 + 10;

void solve() {
    int n, m;
    i64 k;
    cin >> n >> m >> k;
    
    vector<i64> a(n);
    for (auto &x : a) {
        cin >> x;
    }
    
    if (m >= n) {
        for (int i = 0; i < n; ++i) {
            cout << 0 << " ";
        }
        cout << "\n";
        return;
    }
    
    i64 remaining = k;
    for (auto x : a) {
        remaining -= x;
    }
    
    vector<i64> sorted_a = a;
    sort(sorted_a.begin(), sorted_a.end());
    
    vector<i64> prefix_sum(n + 1);
    for (int i = 0; i < n; ++i) {
        prefix_sum[i + 1] = prefix_sum[i] + sorted_a[i];
    }
    
    auto check = [&](int idx, i64 add) -> bool {
        i64 x = a[idx] + add;
        
        // Binary search for the first value > x
        int low = 0, high = n;
        while (low < high) {
            int mid = (low + high) / 2;
            if (sorted_a[mid] > x) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        int cnt = n - low;
        
        if (cnt < m) {
            return false;
        }
        
        // Need to take m smallest values larger than x
        i64 sum_needed = prefix_sum[low] - prefix_sum[low - m];
        return x * m > sum_needed;
    };
    
    vector<i64> res(n);
    for (int i = 0; i < n; ++i) {
        if (check(i, 0)) {
            res[i] = 0;
            continue;
        }
        
        i64 l = 0, r = remaining;
        while (l < r) {
            i64 mid = (l + r) / 2;
            if (check(i, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        
        if (check(i, l)) {
            res[i] = l;
            remaining -= l;
        } else {
            res[i] = -1;
        }
    }
    
    for (auto x : res) {
        cout << x << " ";
    }
    cout << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(0);
    solve();
    return 0;
}
