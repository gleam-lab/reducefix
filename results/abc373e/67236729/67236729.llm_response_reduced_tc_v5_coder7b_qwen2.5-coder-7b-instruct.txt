const int MAXN = 2e5 + 5;

bool can_win(int N, int M, int K, int A[], int i, int X) {
    int max_votes = 0;
    for (int j = 1; j <= N; ++j) {
        if (j == i) {

        } else {

        }
    }

}

int binary_search(int N, int M, int K, int A[]) {
    int left = 0, right = K;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (can_win(N, M, K, A, 1, mid)) {
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    return left;
}
