```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;
typedef vector<ll> vl;
typedef vector<pll> vpl;
typedef vector<vl> vvl;
typedef vector<vpl> vvpl;

const ll INF = 1e18;

ll binsearch(ll left, ll right, function<bool(ll)> f) {
    while(left < right) {
        ll mid = left + (right - left) / 2;
        if(f(mid)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    return left;
}

void solve() {
    ll n, m, k;
    cin >> n >> m >> k;
    vl a(n);
    for(auto &ai : a) cin >> ai;

    vl suff(n + 1);
    for(ll i = n - 1; i >= 0; i--) {
        suff[i] = suff[i + 1] + a[i];
    }

    vl pref(n + 1);
    for(ll i = 0; i < n; i++) {
        pref[i + 1] = pref[i] + a[i];
    }

    vl res(n);
    for(ll i = 0; i < n; i++) {
        ll need = 0;
        if(m - 1 > i) {
            need = max(need, suff[m] - suff[i + 1] - a[i]);
        }
        if(m <= i) {
            need = max(need, pref[i] - pref[m - 1]);
        }
        res[i] = need;
    }

    for(ll i = 0; i < n; i++) {
        if(res[i] == 0) continue;
        ll l = 0, r = k;
        while(l < r) {
            ll mid = (l + r) / 2;
            if(binsearch(0, n, [&](ll x){
                    return pref[x] + mid >= pref[i + 1] + res[i];
                }) <= m - 1 &&
               binsearch(0, n, [&](ll x){
                    return suff[x] + mid >= suff[i + 1] + res[i];
                }) >= m) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        res[i] = l;
    }

    for(ll i = 0; i < n; i++) {
        if(res[i] > k) res[i] = -1;
        cout << res[i] << " ";
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    ll t = 1;
    cin >> t;
    while(t--) solve();

    return 0;
}
```

This solution uses binary search to determine the minimum number of votes needed for each candidate to guarantee their victory. It calculates the suffix and prefix sums to efficiently handle the range queries required. The time complexity is O(N log K), making it suitable for large inputs within the given constraints.
