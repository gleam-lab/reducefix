#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

void solve() {
    i64 N, M, K;
    cin >> N >> M >> K;
    
    vector<i64> A(N);
    for (auto &x : A) cin >> x;
    i64 sum_A = 0;
    for (auto x : A) sum_A += x;
    i64 rem = K - sum_A;
    
    if (M == N) {
        // All candidates will be elected
        for (int i = 0; i < N; ++i) cout << 0 << " ";
        cout << "\n";
        return;
    }
    
    vector<i64> sorted_A = A;
    sort(sorted_A.begin(), sorted_A.end());
    
    // Precompute prefix sums from the end
    vector<i64> suffix_sum(N + 1, 0);
    for (int i = N - 1; i >= 0; --i) {
        suffix_sum[i] = suffix_sum[i + 1] + sorted_A[i];
    }
    
    auto compute_min_votes = [&](i64 target, i64 rem) -> i64 {
        // Binary search for minimum X such that candidate with `target` votes can win
        i64 low = 0, high = rem;
        i64 res = -1;
        
        while (low <= high) {
            i64 mid = (low + high) / 2;
            i64 new_vote = target + mid;
            
            // Use binary search to find how many candidates have more than new_vote
            int cnt = N - (upper_bound(sorted_A.begin(), sorted_A.end(), new_vote) - sorted_A.begin());
            
            if (cnt < M) {
                // Not enough candidates ahead, need to increase votes
                low = mid + 1;
            } else {
                // Enough candidates are behind, mid might be a valid solution
                res = mid;
                high = mid - 1;
            }
        }
        return res;
    };
    
    vector<i64> result;
    for (int i = 0; i < N; ++i) {
        i64 current = A[i];
        i64 min_needed = compute_min_votes(current, rem);
        result.push_back(min_needed);
    }
    
    for (auto x : result) cout << x << " ";
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
