```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

using ll = long long;

struct Candidate {
    ll votes;
    int idx;
};

bool compare(const Candidate& a, const Candidate& b) {
    return a.votes < b.votes;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    ll K;
    cin >> N >> M >> K;

    vector<Candidate> candidates(N);
    ll total_votes = 0;

    for (int i = 0; i < N; ++i) {
        cin >> candidates[i].votes;
        total_votes += candidates[i].votes;
        candidates[i].idx = i;
    }

    K -= total_votes;

    vector<ll> results(N, -1);

    // Sort candidates by number of votes
    sort(candidates.begin(), candidates.end(), compare);

    // Prefix sum array
    vector<ll> prefix_sum(N + 1);
    for (int i = 0; i < N; ++i)
        prefix_sum[i + 1] = prefix_sum[i] + candidates[i].votes;

    for (int i = 0; i < N; ++i) {
        int current_rank = lower_bound(candidates.begin(), candidates.end(), candidates[i], compare) - candidates.begin();
        
        // Binary search to find minimum X
        ll left = 0, right = K;
        ll answer = -1;

        while (left <= right) {
            ll mid = (left + right) / 2;
            ll required_votes = candidates[i].votes + mid;

            // Find how many candidates have more than this candidate after adding mid votes
            int higher_count = N - (upper_bound(candidates.begin(), candidates.end(), Candidate{required_votes, -1}, compare) - candidates.begin);

            if (higher_count < M) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        if (answer != -1 && answer <= K)
            results[candidates[i].idx] = answer;
        else
            results[candidates[i].idx] = 0;
    }

    for (int i = 0; i < N; ++i)
        cout << results[i] << " ";
    cout << "\n";

    return 0;
}
```
