```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

struct Candidate {
    ll votes;
    int index;
};

bool compare(const Candidate &a, const Candidate &b) {
    return a.votes < b.votes;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    ll K;
    cin >> N >> M >> K;

    vector<Candidate> candidates(N);
    ll total = 0;

    for (int i = 0; i < N; ++i) {
        cin >> candidates[i].votes;
        total += candidates[i].votes;
        candidates[i].index = i;
    }

    K -= total; // Remaining votes

    // Sort candidates by current votes
    sort(candidates.begin(), candidates.end(), compare);

    vector<ll> prefix_sum(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        prefix_sum[i + 1] = prefix_sum[i] + candidates[i].votes;
    }

    vector<ll> result(N);

    for (int i = 0; i < N; ++i) {
        int pos = i; // position in sorted array

        // Binary search on required additional votes
        ll left = 0, right = K;
        ll answer = -1;

        while (left <= right) {
            ll mid = (left + right) / 2;
            ll target_votes = candidates[i].votes + mid;

            // We want to find how many candidates have votes > target_votes
            // This is equivalent to finding first index where votes > target_votes
            int idx = upper_bound(candidates.begin(), candidates.end(), Candidate{target_votes, -1}, compare) - candidates.begin();

            // We need at least (N - M + 1) candidates with votes >= target_votes including this one
            // So we need at most (M - 1) candidates with votes > this candidate's new total
            if (N - idx >= M) {
                // Even with maximum possible votes, not enough
                left = mid + 1;
                continue;
            }

            // Calculate minimum votes needed to raise top M-1 candidates above us
            int top = N - M;
            if (pos <= top) top++; // Skip our own candidate

            ll needed = 0;
            if (top > 0) {
                needed = max(0LL, (target_votes + 1) * top - (prefix_sum[N] - prefix_sum[N - top]));
            }

            if (needed <= K - mid) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        result[candidates[i].index] = (answer == -1 ? -1 : answer);
    }

    for (int i = 0; i < N; ++i) {
        cout << result[i] << " ";
    }
    cout << "\n";

    return 0;
}
```
