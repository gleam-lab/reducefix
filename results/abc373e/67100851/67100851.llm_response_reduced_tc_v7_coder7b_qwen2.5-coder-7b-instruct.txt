bool canWin(vector<Candidate>& candidates, int M, long long K, int targetIndex, long long x) {

    vector<long long> prefixSum(n + 1, 0);

            count++;
        }
    }
    
    // Restore original vote counts
    candidates[targetIndex].votes -= x;
    
    // Check if the number of candidates beating the target candidate is less than M
    return count < M;
}
