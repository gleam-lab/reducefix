#include <iostream>
#include <algorithm>
using namespace std;

typedef struct {
    long long a;
    int idx;
} Candidate;

int main() {
    int N, M;
    long long K;
    cin >> N >> M >> K;
    
    Candidate candidates[N];
    long long sum_votes = 0;
    for (int i = 0; i < N; ++i) {
        cin >> candidates[i].a;
        sum_votes += candidates[i].a;
        candidates[i].idx = i;
    }
    
    sort(candidates, candidates + N, [](const Candidate &a, const Candidate &b) {
        return a.a < b.a;
    });

    long long remaining_votes = K - sum_votes;
    long long votes_needed[N];
    for (int i = 0; i < N; ++i) {
        votes_needed[i] = -1; // Default value: cannot win
    }

    // Calculate minimum votes needed for each candidate to guarantee victory
    for (int i = 0; i < N; ++i) {
        long long needed = M; // A[i] needs at least M votes to be in the top M
        long long votes_up_to_i = 0;
        for (int j = 0; j <= i; ++j) {
            votes_up_to_i += candidates[j].a;
        }
        long long votes_from_i_to_N = votes_up_to_i; // Assume all votes from i to N go to i+1 to N
        long long votes_needed_for_remaining = votes_from_i_to_N - votes_up_to_i;
        
        for (int j = i + 1; j < N; ++j) {
            votes_needed_for_remaining -= candidates[j].a;
        }
        
        // Calculate the minimum votes needed for i to guarantee at least M votes
        if (votes_needed_for_remaining < needed) {
            votes_needed[i] = needed - votes_needed_for_remaining;
        } else {
            votes_needed[i] = 0; // Already enough votes to guarantee victory
        }
        
        // Ensure we do not exceed the remaining votes
        if (votes_needed[i] > remaining_votes) {
            votes_needed[i] = -1;
        }
    }

    // Output the result
    for (int i = 0; i < N; ++i) {
        if (votes_needed[i] != -1) {
            cout << votes_needed[i] << " ";
        } else {
            cout << "0 "; // If it cannot win, the output is 0 as per problem statement
        }
    }
    cout << endl;

    return 0;
}
