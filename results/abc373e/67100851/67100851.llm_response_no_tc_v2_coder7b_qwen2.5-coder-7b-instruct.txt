#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Candidate {
    long long votes;
    int index;
};

bool canBeElected(vector<Candidate>& candidates, int candidateIndex, long long additionalVotes, int M) {
    // Calculate the total votes needed for the candidate to be in the top N-M positions
    long long requiredVotes = 0;
    for (int i = 0; i < candidates.size() - M; ++i) {
        requiredVotes += candidates[i].votes + additionalVotes;
    }
    
    // Check if the current candidate's total votes plus additional votes exceed the required votes
    return candidates[candidateIndex].votes + additionalVotes > requiredVotes;
}

vector<long long> calculateMinimumVotes(vector<long long>& votes, int M, long long K) {
    vector<Candidate> candidates(votes.size());
    for (int i = 0; i < votes.size(); ++i) {
        candidates[i] = {votes[i], i};
    }
    
    // Sort candidates based on votes
    sort(candidates.begin(), candidates.end(), [](const Candidate& a, const Candidate& b) {
        return a.votes < b.votes;
    });
    
    vector<long long> minVotes(votes.size(), -1); // Initialize with -1 indicating failure
    
    for (int i = 0; i < votes.size(); ++i) {
        if (canBeElected(candidates, i, 0, M)) {
            minVotes[i] = 0; // Already elected
            continue;
        }
        
        // Binary search for the minimum additional votes needed
        long long left = 0, right = K;
        while (left <= right) {
            long long mid = left + (right - left) / 2;
            if (canBeElected(candidates, i, mid, M)) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        minVotes[i] = left;
    }
    
    return minVotes;
}

int main() {
    int N, M;
    long long K;
    cin >> N >> M >> K;
    
    vector<long long> votes(N);
    for (int i = 0; i < N; ++i) {
        cin >> votes[i];
    }
    
    vector<long long> result = calculateMinimumVotes(votes, M, K);
    
    for (int i = 0; i < N; ++i) {
        cout << result[i] << " ";
    }
    cout << endl;
    
    return 0;
}

This solution includes:
1. A `Candidate` struct to store the votes and original index of each candidate.
2. A function `canBeElected` to check if a candidate can be elected given additional votes.
3. The main function reads input, calculates the minimum votes needed for each candidate using binary search, and outputs the results.
