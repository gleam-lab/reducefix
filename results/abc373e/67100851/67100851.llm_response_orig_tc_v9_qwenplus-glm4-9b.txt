    long long remaining_votes = K - total_votes;

    // Calculate the minimum votes needed for each candidate to secure a victory
    for (int i = 0; i < N; ++i) {
        if (i < M) {
            // If there are less than M candidates, they are all guaranteed to win
            votes_needed[i] = 0;
