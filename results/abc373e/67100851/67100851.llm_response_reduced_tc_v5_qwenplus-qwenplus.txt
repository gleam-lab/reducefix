```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

struct Candidate {
    ll votes;
    int index;
};

bool compare(const Candidate &a, const Candidate &b) {
    return a.votes < b.votes;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M, K;
    cin >> N >> M >> K;
    vector<Candidate> candidates(N);
    ll total_votes = 0;

    for (ll i = 0; i < N; ++i) {
        cin >> candidates[i].votes;
        total_votes += candidates[i].votes;
        candidates[i].index = i;
    }

    // Remaining votes to distribute
    ll remaining = K - total_votes;

    // Sort candidates by current votes
    sort(candidates.begin(), candidates.end(), compare);

    // Prefix sum of current votes
    vector<ll> prefix_sum(N + 1, 0);
    for (ll i = 0; i < N; ++i) {
        prefix_sum[i + 1] = prefix_sum[i] + candidates[i].votes;
    }

    vector<ll> result(N);

    // For each candidate, binary search on minimum required additional votes
    for (ll i = 0; i < N; ++i) {
        ll left = 0, right = remaining;
        ll needed = -1;

        while (left <= right) {
            ll mid = (left + right) / 2;
            ll target_votes = candidates[i].votes + mid;

            // Find how many candidates have strictly more votes in the current scenario
            // Binary search to find first candidate with votes > target_votes
            ll low = 0, high = N;
            while (low < high) {
                ll m = (low + high) / 2;
                if (candidates[m].votes > target_votes)
                    high = m;
                else
                    low = m + 1;
            }
            ll num_strictly_higher = N - low;

            // To be elected, fewer than M candidates should have more votes than this one
            if (num_strictly_higher < M) {
                needed = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        result[candidates[i].index] = (needed == -1) ? -1 : needed;
    }

    for (ll i = 0; i < N; ++i) {
        cout << result[i] << " ";
    }
    cout << endl;

    return 0;
}
```
