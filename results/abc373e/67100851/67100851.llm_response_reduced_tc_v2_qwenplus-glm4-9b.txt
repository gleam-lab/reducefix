typedef struct {

bool canWin(const vector<Candidate>& candidates, long long votes, int candidate_index, int M) {
    long long sum = 0;
    for (int i = 0; i < candidates.size(); ++i) {

            if (candidates[i].votes + votes >= candidates[candidate_index].votes) {
                sum++;
            }
        }
    }
    return sum < M;
}
