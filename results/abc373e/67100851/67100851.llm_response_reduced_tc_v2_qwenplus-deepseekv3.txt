### Approaches
1. **Sorting and Binary Search**: For each candidate, we need to determine the minimum number of additional votes (X) such that when they receive X votes, no more than M-1 candidates can surpass their total votes. We can use binary search to efficiently find this X for each candidate.
2. **Prefix Sum**: Compute prefix sums to quickly calculate the sum of votes needed for candidates to be adjusted to a certain threshold.
3. **Binary Search Condition**: For each candidate, use binary search to find the smallest X such that the condition is met. The condition involves ensuring that after allocating X votes to the candidate, the sum of votes needed to make M-1 candidates surpass their new vote count is not feasible.

#include <iostream>
#include <algorithm>
using namespace std;

typedef struct {
    long long a;
    int idx;
} node;

node A[200000];
long long psum[200001], K;
int N, M;

bool is_possible(long long X, int i) {
    long long new_vote = A[i].a + X;
    int cnt = 0;
    long long total = 0;

    // Binary search to find the number of candidates with a > new_vote - 1
    auto it = upper_bound(A, A + N, new_vote, [](long long val, const node& b) { return val < b.a; });
    int higher = N - (it - A);
    
    if (higher >= M) {
        return false;
    }

    // The remaining candidates can be adjusted to not exceed new_vote
    long long remaining_votes = K - X;
    long long sum_adjust = 0;
    int adjust_candidates = M - 1 - higher;

    if (adjust_candidates > 0) {
        // Sort the candidates to find the top M-1 candidates
        int start = it - A;
        int end = start + adjust_candidates;
        if (end > N) end = N;
        for (int j = start; j < end; ++j) {
            if (A[j].a > new_vote) continue;
            sum_adjust += new_vote - A[j].a;
        }
    }

    if (sum_adjust <= remaining_votes) {
        return true;
    } else {
        return false;
    }
}

int main() {
    scanf("%d %d %lld", &N, &M, &K);
    long long sum_A = 0;
    for (int i = 0; i < N; ++i) {
        scanf("%lld", &A[i].a);
        A[i].idx = i;
        sum_A += A[i].a;
    }
    K -= sum_A;

    sort(A, A + N, [](const node& a, const node& b) { return a.a < b.a; });

    for (int i = 0; i < N; ++i) {
        long long lo = 0, hi = K;
        long long ans = -1;
        while (lo <= hi) {
            long long mid = (lo + hi) / 2;
            if (is_possible(mid, i)) {
                ans = mid;
                hi = mid - 1;
            } else {
                lo = mid + 1;

### Explanation
1. **Input Handling**: The input is read, and the total votes already counted (`sum_A`) is computed. The remaining votes (`K`) is adjusted by subtracting `sum_A`.
2. **Sorting**: The candidates are sorted based on their current votes to facilitate binary search operations.
3. **Binary Search for Each Candidate**: For each candidate, we perform a binary search to find the minimum additional votes (`X`) required. The `is_possible` function checks if the current `X` ensures that the candidate will be elected by verifying that no more than M-1 candidates can have more votes than them after the allocation.
4. **Output**: The results are sorted back to the original order of candidates and printed.
