#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

int main() {
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (ll &x : A) cin >> x;

    // Sort candidates by their current votes
    sort(A.begin(), A.end());

    // Prefix sum of votes
    vector<ll> prefix(N);
    prefix[0] = A[0];
    for (int i = 1; i < N; ++i) {
        prefix[i] = prefix[i - 1] + A[i];
    }

    // Calculate votes remaining
    ll remaining_votes = K - prefix[N - 1];
    
    vector<ll> votes_needed(N);
    for (int i = 0; i < N; ++i) {
        // Number of candidates who have more or equal votes than candidate i
        ll competitors = i + 1;

        // Number of votes that need to be distributed to make candidate i the winner
        ll votes_for_i_to_win = min(N - M, competitors) * A[i] - (prefix[i] - (competitors - 1) * A[i]);
        
        if (votes_for_i_to_win <= 0) {
            votes_needed[i] = 0;
        } else {
            votes_needed[i] = votes_for_i_to_win;
        }
    }

    // Check if the votes needed are less than or equal to the remaining votes
    for (ll &vote : votes_needed) {
        if (vote > remaining_votes) {
            vote = -1; // Candidate cannot win
        }
    }

    // Output the result
    for (ll vote : votes_needed) {
        cout << vote << " ";
    }
    cout << endl;

    return 0;
}
