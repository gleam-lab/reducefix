#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;
vector<ll> A;

ll solve(int i) {
    ll remaining_votes = K - accumulate(A.begin(), A.end(), 0LL);
    ll low = 0, high = remaining_votes;
    ll min_X = -1;

    while (low <= high) {
        ll mid = low + (high - low) / 2;
        ll current_votes = A[i] + mid;
        
        vector<ll> temp = A;
        temp[i] += mid;
        sort(temp.begin(), temp.end(), greater<ll>());

        int position = 0;
        while (position < N && temp[position] > current_votes) position++;
        if (position < M) {
            min_X = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    return min_X;
}
