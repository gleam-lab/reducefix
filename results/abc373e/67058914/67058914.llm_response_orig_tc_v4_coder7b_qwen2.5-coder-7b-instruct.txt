#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Candidate {
    int index;
    ll votes;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int N, M;
    ll K;
    cin >> N >> M >> K;
    
    vector<Candidate> candidates(N);
    for (int i = 0; i < N; ++i) {
        cin >> candidates[i].votes;
        candidates[i].index = i;
    }
    
    sort(candidates.begin(), candidates.end(), [](const Candidate &a, const Candidate &b) {
        return a.votes < b.votes;
    });
    
    vector<ll> prefix(N + 1);
    for (int i = 1; i <= N; ++i) {
        prefix[i] = prefix[i - 1] + candidates[i - 1].votes;
    }
    
    vector<int> results(N, -1);
    
    for (int i = 0; i < N; ++i) {
        if (candidates[i].votes >= prefix[N] - prefix[i] - candidates[i].votes + 1) {
            results[candidates[i].index] = 0;
            continue;
        }
        
        int left = 0, right = K - prefix[i];
        while (left < right) {
            int mid = (left + right) / 2;
            if (candidates[i].votes + mid >= prefix[N] - prefix[i] - candidates[i].votes + 1) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        
        results[candidates[i].index] = left;
    }
    
    for (int i = 0; i < N; ++i) {
        cout << results[i] << " ";
    }
    
    return 0;
}
