#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<pair<ll, int>> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i].first;
    }

    vector<ll> res(N);

    // Prepare sorted list of other candidates' votes
    vector<ll> sorted_votes;
    for (auto &p : A) sorted_votes.push_back(p.first);
    sort(sorted_votes.begin(), sorted_votes.end());

    // For each candidate, binary search on required additional votes
    for (int i = 0; i < N; ++i) {
        ll a_i = A[i].first;
        ll remaining = K - (sorted_votes[N - M] - a_i);
        
        // Binary search to find minimum X
        ll low = 0, high = K;
        while (low < high) {
            ll mid = (low + high) / 2;
            ll total = a_i + mid;
            
            // We need to ensure at most M-1 candidates have more votes than total
            // So we use upper_bound to find how many candidates have > total votes
            ll count = sorted_votes.end() - upper_bound(sorted_votes.begin(), sorted_votes.end(), total);
            
            if (count >= M) {
                // Not enough, need more votes
                low = mid + 1;
            } else {
                high = mid;
            }
        }

        // After binary search, check if it's possible
        ll final_total = a_i + low;
        ll count = sorted_votes.end() - upper_bound(sorted_votes.begin(), sorted_votes.end(), final_total);
        
        if (count < M) {
            res[i] = low;
        } else {
            res[i] = -1;  // Cannot guarantee victory
        }
    }

    // Print result
    for (ll x : res) {
        cout << x << " ";
    }
    cout << endl;

    return 0;
}
