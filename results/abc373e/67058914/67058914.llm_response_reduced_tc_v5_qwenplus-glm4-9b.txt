#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

int main() {
  cin >> N >> M >> K;
  vector<ll> A(N);
  for (int i = 0; i < N; ++i) {
    cin >> A[i];
  }

  sort(A.begin(), A.end());

  vector<ll> votes_needed(N);
  for (int i = 0; i < N; ++i) {
    if (i < M - 1) {
      // If there are fewer than M-1 candidates ahead, they need to catch up to the M-th
      votes_needed[i] = (M - 1 - i) * (A[M - 1] - A[i]);
    } else {
      // Otherwise, they need to be ahead of the M-th candidate
      votes_needed[i] = (A[i] - A[M - 1]) + (M - 1) * (A[N - 1] - A[i]);
    }
  }

  sort(votes_needed.begin(), votes_needed.end());

  ll remaining_votes = K;
  for (int i = 0; i < N; ++i) {
    ll votes_for_i = votes_needed[i];
    if (remaining_votes < votes_for_i) {
      votes_needed[i] = votes_for_i - remaining_votes;
      remaining_votes = 0;
    } else {
      votes_needed[i] = 0;
      remaining_votes -= votes_for_i;
    }
  }

  for (int i = 0; i < N; ++i) {
    cout << votes_needed[i] << (i == N - 1 ? '\n' : ' ');
  }

  return 0;
}
