#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

struct Node {
  ll val;
  int idx;
};

int main() {
    cin >> N >> M >> K;

    vector<Node> a(N + 5);
    vector<ll> votes(N + 5, 0);
    vector<ll> max_votes(N + 5, 0);

    for (int i = 1; i <= N; ++i) {
        cin >> a[i].val;
        a[i].idx = i;
        votes[i] = a[i].val;
    }

    // Sort candidates by votes received so far (descending order)
    sort(a.rbegin(), a.rend(), [](const Node &A, const Node &B) {
        return A.val > B.val;
    });

    // Calculate the maximum votes any candidate can get
    ll current_max = -1;
    for (int i = 1; i <= N; ++i) {
        if (a[i].val > current_max) {
            current_max = a[i].val;
        }
        max_votes[i] = current_max;
    }

    // Calculate the remaining votes after counting current votes
    ll remaining_votes = K - accumulate(votes.begin() + 1, votes.end(), 0LL);

    // Initialize the additional votes needed for each candidate
    vector<ll> additional_votes_needed(N + 1, -1);

    // Iterate over each candidate to determine the minimum additional votes needed
    for (int i = 1; i <= N; ++i) {
        int more_votes_than_i = 0;
        for (int j = 1; j <= i - 1; ++j) {
            if (a[j].val > a[i].val) {
                more_votes_than_i++;
            }
        }
        
        for (int j = i + 1; j <= N; ++j) {
            if (a[j].val > a[i].val) {
                more_votes_than_i++;
            }
        }

        if (more_votes_than_i + 1 < M) {
            // Calculate the number of votes needed to surpass all other candidates
            ll votes_needed_to_succeed = max_votes[i] + remaining_votes + 1 - votes[i];
            if (votes_needed_to_succeed > 0) {
                additional_votes_needed[i] = votes_needed_to_succeed;
            }
        }
    }

    // Output the result
    for (int i = 1; i <= N; ++i) {
        cout << additional_votes_needed[i] << ' ';
    }
    cout << '\n';

    return 0;
}
