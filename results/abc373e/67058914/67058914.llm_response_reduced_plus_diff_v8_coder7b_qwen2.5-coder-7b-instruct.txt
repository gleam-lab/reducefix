    return A.val < B.val;
  };

  sort(a.begin() + 1, a.begin() + N + 1, cmp);

  vector<ll> pre(N + 5);
  pre[0] = 0LL;
  for (int i = 1; i <= N; ++i) {
    pre[i] = pre[i - 1] + a[i].val;
  }

  K -= pre[N];

  vector<ll> ans(N + 5, -1);
  for (int i = 1; i <= N; ++i) {
    if (K >= a[i].val) {
      K -= a[i].val;
      ans[a[i].idx] = 0;
      continue;
    }
    
    int l = 0, r = K;
    while (l < r) {
      int mid = (l + r + 1) / 2;
      int cnt = 0;
      for (int j = i + 1; j <= N && cnt < M; ++j) {
        if (a[j].val + mid > a[i].val) ++cnt;
      }
      
      if (cnt == M) r = mid - 1;
      else l = mid;
    }
    
    if (l != K) ans[a[i].idx] = l;
  }

}
