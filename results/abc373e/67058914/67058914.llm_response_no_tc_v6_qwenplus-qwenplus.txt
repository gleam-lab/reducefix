#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    ll total_current = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        total_current += A[i];
    }
    ll remaining_votes = K - total_current;

    // Create a list of (value, index) pairs
    vector<pair<ll, int>> sorted_A(N);
    for (int i = 0; i < N; ++i) {
        sorted_A[i] = {A[i], i};
    }

    sort(sorted_A.begin(), sorted_A.end());

    // Prefix sum array
    vector<ll> prefix(N + 1, 0);
    for (int i = 1; i <= N; ++i) {
        prefix[i] = prefix[i - 1] + sorted_A[i - 1].first;
    }

    vector<ll> result(N);

    // For each candidate, binary search on minimum required additional votes
    for (int i = 0; i < N; ++i) {
        ll base = A[i];

        // Binary search over X: extra votes needed
        ll left = 0, right = remaining_votes;
        ll answer = -1;

        while (left <= right) {
            ll mid = (left + right) / 2;
            ll current_total = base + mid;

            // Find how many candidates are strictly greater than current_total
            // among the ones not being considered (i.e., other than candidate i)
            int low = 0, high = N;
            while (low < high) {
                int m = (low + high) / 2;
                if (sorted_A[m].first > current_total) high = m;
                else low = m + 1;
            }

            // `cnt` is the number of candidates with vote count > current_total
            int cnt = N - low;

            // Check if candidate i can still be in top M
            if (cnt < M) {
                answer = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        result[i] = (answer == -1 ? -1 : answer);
    }

    // Print results in original order
    for (int i = 0; i < N; ++i) {
        cout << result[i] << " ";
    }
    cout << '\n';

    return 0;
}
