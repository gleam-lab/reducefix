```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

struct Node {
  ll val;
  int idx;
};

int main() {
  cin >> N >> M >> K;

  vector<Node> a(N + 5);

  for (int i = 1; i <= N; ++i) {
    cin >> a[i].val;
    a[i].idx = i;
  }

  auto cmp = [](const auto &A, const auto &B)->bool {
    return A.val < B.val;
  };

  sort(a.begin() + 1, a.begin() + N + 1, cmp) ;

  vector<ll> pre(N + 5);
  pre[0] = 0LL;
  for (int i = 1; i <= N; ++i) {
    pre[i] = pre[i - 1] + a[i].val;
  }

  K -= pre[N];

  vector<ll> ans(N + 5);
  for (int i = 1; i <= N; ++i) {
    ans[i] = 0;
  }

  for (int i = 1; i <= N; ++i) {
    ll cur = a[i].val + K;
    int idx = upper_bound(a.begin() + i + 1, a.begin() + N + 1, (Node) {cur, 0}, cmp) - a.begin();
    int cnt = (N - idx + 1);
    if (cnt >= M) {
      ans[a[i].idx] = -1;
      continue;
    }
    int l = 0, r = K;
    while (l < r) {
      int mid = (l + r) >> 1;
      cur = a[i].val + mid;
      idx = upper_bound(a.begin() + i + 1, a.begin() + N + 1, (Node) {cur, 0}, cmp) - a.begin();
      ll need = (mid + 1) * (N - idx + 1) - (pre[N] - pre[idx - 1]);
      if (need > K) r = mid;
      else l = mid + 1;
    }
    ans[a[i].idx] = max(0LL, l - 1);
  }

  for (int i = 1; i <= N; ++i) {
    cout << ans[i] << ' ';
  }

  return 0;
}
```

2. It optimizes the binary search for the minimum number of additional votes needed.
3. It handles edge cases correctly, especially when a candidate's current vote count plus the maximum possible additional votes does not allow them to win even with the best-case scenario for other candidates.
