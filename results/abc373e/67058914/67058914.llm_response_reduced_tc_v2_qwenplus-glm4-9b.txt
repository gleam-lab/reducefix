#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ll = long long;

int main() {
    int N, M, K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }

    // Pair each candidate's votes with their index and sort by votes
    vector<pair<ll, int>> candidates(N);
    for (int i = 0; i < N; ++i) {
        candidates[i] = {A[i], i + 1};
    }
    sort(candidates.begin(), candidates.end());

    // Precompute cumulative sums of votes
    vector<ll> prefix_sums(N + 1, 0);
    for (int i = 1; i <= N; ++i) {
        prefix_sums[i] = prefix_sums[i - 1] + candidates[i - 1].first;
    }

    // Calculate the minimum additional votes needed for each candidate
    vector<ll> additional_votes(N);
    for (int i = 0; i < N; ++i) {
        // Number of votes remaining
        ll remaining_votes = K - prefix_sums[N];
        // Maximum votes any candidate can get after all votes are counted
        ll max_votes = prefix_sums[N] + remaining_votes;
        // Index of the candidate with the same number of votes as the i-th candidate
        int same_votes_index = lower_bound(candidates.begin(), candidates.end(), make_pair(candidates[i].first, 0), [](const pair<ll, int>& a, const pair<ll, int>& b) {
            return a.first < b.first;
        }) - candidates.begin();

        // Total votes ahead of the i-th candidate
        ll votes_ahead = prefix_sums[same_votes_index - 1];
        // Minimum votes needed to guarantee i-th candidate is among the top M
        ll min_votes_needed = max_votes - votes_ahead - (N - same_votes_index);

        // Additional votes needed for the i-th candidate
        additional_votes[i] = max(0LL, min_votes_needed - candidates[i].first);
    }

    // Output the result
    for (const auto& vote : additional_votes) {
        cout << vote << " ";
    }
    cout << endl;

    return 0;
}
