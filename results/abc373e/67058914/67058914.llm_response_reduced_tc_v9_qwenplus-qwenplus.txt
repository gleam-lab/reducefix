#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<pair<ll, int>> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i].first;
        A[i].second = i;
    }

    ll total = 0;
    for (auto &p : A)
        total += p.first;
    ll remaining = K - total;

    vector<ll> res(N);

    // Create a sorted list of votes to help determine how many can be above us
    vector<ll> sorted_votes;
    for (auto &p : A)
        sorted_votes.push_back(p.first);
    sort(sorted_votes.begin(), sorted_votes.end());

    for (int i = 0; i < N; ++i) {
        ll my_vote = A[i].first;
        ll low = 0, high = remaining;
        ll answer = -1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll target = my_vote + mid;
            // Find how many candidates have strictly more than target
            ll count_above = N - (upper_bound(sorted_votes.begin(), sorted_votes.end(), target) - sorted_votes.begin());

            if (count_above >= M) {
                // Still not enough, need more votes
                low = mid + 1;
            } else {
                answer = mid;
                high = mid - 1;
            }
        }

        res[i] = (answer == -1 ? -1 : answer);
    }

    for (auto &x : res)
        cout << x << " ";
    cout << endl;

    return 0;
}
