```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Node {
    ll val;
    int idx;
};

int N, M;
ll K;

vector<Node> a(N + 5);

auto cmp = [](const auto &A, const auto &B)->bool {
    return A.val < B.val;
};

void solve() {
    cin >> N >> M >> K;
    
    for (int i = 1; i <= N; ++i) {
        cin >> a[i].val;
        a[i].idx = i;
    }

    sort(a.begin() + 1, a.begin() + N + 1, cmp);

    vector<ll> pre(N + 5);
    pre[0] = 0LL;
    for (int i = 1; i <= N; ++i) {
        pre[i] = pre[i - 1] + a[i].val;
    }

    K -= pre[N];

    vector<ll> ans(N + 5, 0);
    for (int i = 1; i <= N; ++i) {
        ans[i] = a[i].val + K;
        int idx = upper_bound(a.begin() + i + 1, a.begin() + N + 1, (Node){ans[i], 0}, cmp) - a.begin();
        int cnt = (N - idx + 1);
        if (cnt >= M) {
            ans[i] = -1;
            continue;
        }
        int l = 0, r = K;
        while (l < r) {
            int mid = (l + r + 1) / 2;
            ans[i] = a[i].val + mid;
            idx = upper_bound(a.begin() + i + 1, a.begin() + N + 1, (Node){ans[i], 0}, cmp) - a.begin();
            ll need = (ans[i] + 1) * (N - idx + 1) - pre[N];
            if (need > K) r = mid - 1;
            else l = mid;
        }
        ans[i] -= K;
    }

    for (int i = 1; i <= N; ++i) {
        cout << ans[i] << ' ';
    }
    cout << '\n';
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);
    int T;
    cin >> T;
    while (T--) solve();
    return 0;
}
```
