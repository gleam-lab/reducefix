    }

    ll remaining = K - accumulate(A.begin(), A.end(), 0LL);

    vector<ll> res(N);

        ll low = 0, high = remaining;
        ll ans = -1;

        while (low <= high) {
            ll mid = low + (high - low) / 2;
            ll new_votes = current + mid;

            auto it = upper_bound(sortedA.begin(), sortedA.end(), new_votes);

            int initial_count = N - (k + 1);

                continue;

                high = mid - 1;
