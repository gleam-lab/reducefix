```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int N, M;
  ll K;
  cin >> N >> M >> K;

  vector<ll> A(N);
  for (int i = 0; i < N; ++i) {
    cin >> A[i];
  }

  // Remaining votes
  ll remaining = K;
  for (ll a : A) {
    remaining -= a;
  }

  vector<int> idx(N);
  iota(idx.begin(), idx.end(), 0);

  // Sort the candidates by current votes
  sort(idx.begin(), idx.end(), [&](int i, int j) { return A[i] < A[j]; });
  vector<ll> sortedA(N);
  for (int i = 0; i < N; ++i) {
    sortedA[i] = A[idx[i]];
  }

  vector<ll> prefix(N + 1);
  for (int i = 1; i <= N; ++i) {
    prefix[i] = prefix[i - 1] + sortedA[i - 1];
  }

  vector<ll> res(N);

  // For each candidate, binary search on minimum required votes
  for (int i = 0; i < N; ++i) {
    int pos = lower_bound(sortedA.begin(), sortedA.end(), A[i]) - sortedA.begin();
    while (pos < N && sortedA[pos] != A[i]) ++pos;

    // Binary search over additional votes needed
    ll low = 0, high = remaining;
    ll ans = -1;

    while (low <= high) {
      ll mid = (low + high) / 2;
      ll cur_total = A[i] + mid;

      // We now check if with cur_total votes, candidate i is elected
      // i.e., fewer than M candidates can have more votes than them

      // Find how many candidates already have >= cur_total
      int cnt = N - (upper_bound(sortedA.begin(), sortedA.end(), cur_total) - sortedA.begin());

      if (cnt < M) {
        ans = mid;
        high = mid - 1;
      } else {
        low = mid + 1;
      }
    }

    res[i] = ans;
  }

  for (ll x : res) {
    cout << x << " ";
  }
  cout << "\n";

  return 0;
}
```
