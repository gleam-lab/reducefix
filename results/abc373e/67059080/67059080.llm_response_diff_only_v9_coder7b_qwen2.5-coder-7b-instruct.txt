```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

struct Node {
  ll val;
  int idx;
};

int main() {
  ios_base::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr);

  cin >> N >> M >> K;

  vector<Node> a(N + 5);
  for (int i = 1; i <= N; ++i) {
    cin >> a[i].val;
    a[i].idx = i;
  }

  sort(a.rbegin(), a.rend());

  vector<ll> pre(N + 5);
  pre[0] = 0LL;
  for (int i = 1; i <= N; ++i) {
    pre[i] = pre[i - 1] + a[i].val;
  }

  K -= pre[N];

  vector<ll> ans(N + 5, -1);

  for (int i = 1; i <= N; ++i) {
    if (a[i].val > pre[N] - pre[i]) {
      ans[a[i].idx] = 0;
    } else {
      ll need = (M - 1) * (pre[N] - pre[i]) - a[i].val + 1;
      if (need <= K) {
        ans[a[i].idx] = need;
      }
    }
  }

  for (int i = 1; i <= N; ++i) {
    cout << ans[i] << ' ';
  }

  return 0;
}
```
