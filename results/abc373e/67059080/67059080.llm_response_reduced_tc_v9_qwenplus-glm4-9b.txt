#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M, K;

int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    cin >> N >> M >> K;

    vector<ll> A(N);
    ll sum = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        sum += A[i];
    }

    // Sort candidates by their current votes
    vector<pair<ll, int>> votes(N);
    for (int i = 0; i < N; ++i) {
        votes[i] = {A[i], i};
    }
    sort(votes.begin(), votes.end());

    vector<ll> extra_votes(N);
    for (int i = 0; i < M - 1; ++i) {
        extra_votes[votes[i].second] = -1; // Candidates who cannot win
    }

    // Calculate the minimum votes needed for each candidate
    for (int i = 0; i < M - 1; ++i) {
        if (extra_votes[votes[i].second] == -1) {
            continue;
        }
        ll needed = votes[i].first + (M - 1 - i) * (N - M + i);
        extra_votes[votes[i].second] = needed;
    }

    // If all top M-1 candidates cannot win, assign -1 to all others
    for (int i = M - 1; i < N; ++i) {
        if (extra_votes[votes[i].second] == -1) {
            extra_votes[votes[i].second] = -1;
        } else {
            // Reassign needed votes to ensure they are unique and non-decreasing
            extra_votes[votes[i].second] = extra_votes[votes[i].second] + 1;
        }
    }

    // Output the results
    for (int i = 0; i < N; ++i) {
        if (extra_votes[i] == -1) {
            cout << -1 << ' ';
        } else {
            cout << extra_votes[i] << ' ';
        }
    }
    cout << '\n';

    return 0;
}
