#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Node {
    long long val;
    int idx;
};

int main() {
    int N, M, K;
    cin >> N >> M >> K;

    vector<Node> candidates(N);
    vector<long long> votes(N);

    for (int i = 0; i < N; ++i) {
        cin >> votes[i];
        candidates[i].val = votes[i];
        candidates[i].idx = i + 1;
    }

    // Sort candidates by votes received
    sort(candidates.begin(), candidates.end(), [](const Node &a, const Node &b) {
        return a.val < b.val;
    });

    // Calculate the number of votes needed to guarantee victory for each candidate
    vector<long long> needed_votes(N);
    for (int i = 0; i < N; ++i) {
        long long votes_needed = 0;
        if (i + 1 >= M) {
            // Calculate votes needed to win against the top M-1 candidates
            votes_needed = candidates[i + M - 1].val - candidates[i].val + 1;
            needed_votes[i] = votes_needed - votes[i] + K;
        } else {
            // If there are fewer than M candidates, all candidates need exactly votes_needed - votes[i] + K votes
            votes_needed = candidates[i + 1].val - candidates[i].val + 1;
            needed_votes[i] = votes_needed - votes[i] + K;
        }
    }

    // Sort needed votes to find the minimum number of additional votes required for each candidate
    sort(needed_votes.begin(), needed_votes.end());

    // Output the minimum number of additional votes required for each candidate
    for (long long &vote : needed_votes) {
        cout << vote << " ";
    }
    cout << endl;

    return 0;
}
