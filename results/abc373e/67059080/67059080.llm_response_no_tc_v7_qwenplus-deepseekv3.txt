### Problem Analysis
The problem requires us to determine, for each candidate, the minimum number of additional votes needed to ensure their election, given that a candidate is elected if fewer than M candidates have more votes than them after all votes are counted. The solution involves calculating the votes each candidate needs to secure their position, considering the worst-case scenario where other candidates receive as many votes as possible to prevent the current candidate's election.

### Approach

#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    ll K;
    cin >> N >> M >> K;

    vector<pair<ll, int>> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i].first;
        A[i].second = i;
    }

    sort(A.begin(), A.end());

    vector<ll> prefix(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = prefix[i] + A[i].first;
    }

    ll remaining = K - prefix[N];
    vector<ll> ans(N, -1);

    for (int i = 0; i < N; ++i) {
        ll low = 0, high = remaining;
        ll best = -1;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll new_vote = A[i].first + mid;
            auto it = upper_bound(A.begin(), A.end(), make_pair(new_vote, N));
            int cnt = A.end() - it;
            if (cnt < M) {
                best = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        ans[A[i].second] = best;
    }

    for (int i = 0; i < N; ++i) {
        cout << ans[i] << ' ';
    }
    cout << '\n';

    return 0;
}
