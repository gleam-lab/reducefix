
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

struct Node {
  ll val;
  int idx;
};

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);

  int N, M, K;
  cin >> N >> M >> K;

  vector<Node> A(N + 1);
  ll sum = 0;
  for (int i = 1; i <= N; ++i) {
    cin >> A[i].val;
    sum += A[i].val;
    A[i].idx = i;
  }

  sort(A.begin() + 1, A.begin() + N + 1, [](const Node& a, const Node& b) {
    return a.val < b.val;
  });

  vector<ll> remaining_votes(N + 1);
  remaining_votes[0] = 0;
  for (int i = 1; i <= N; ++i) {
    remaining_votes[i] = K - remaining_votes[i - 1];
  }

  vector<ll> votes_needed(N + 1);
  for (int i = N; i >= 1; --i) {
    ll votes_to_win = sum + i - A[i].val;
    ll remaining_votes_suffix = min(remaining_votes[i - 1], votes_to_win - A[i].val);

    if (remaining_votes_suffix >= (i - 1) - A[i].idx + M) {
      votes_needed[A[i].idx] = votes_to_win - A[i].val - remaining_votes_suffix;
    } else {
      votes_needed[A[i].idx] = -1;
    }

    sum -= A[i].val;
  }

  for (int i = 1; i <= N; ++i) {
    if (votes_needed[i] == -1) {
      cout << -1 << ' ';
    } else {
      cout << votes_needed[i] << ' ';
    }
  }
  cout << '\n';

  return 0;
}
```

### Explanation
1. **Input Parsing**: The input is parsed to read the number of candidates `N`, the number of candidates that can win `M`, and the total number of votes `K`. The array `A` is filled with the current number of votes each candidate has.

2. **Sorting Candidates**: The candidates are sorted by the number of votes they currently have in ascending order. This allows us to process them from the one with the least votes upwards.

3. **Calculating Remaining Votes**: The array `remaining_votes` is filled from right to left, indicating the number of votes available after the votes of the current candidate have been subtracted.

4. **Calculating Votes Needed**: Starting from the last candidate, we calculate the number of votes they need to guarantee they will win. This is done by adjusting the sum of votes up to the `i`-th candidate and comparing it with the votes available after the votes of the `i`-th candidate.

5. **Output**: The votes needed for each candidate are printed. If a candidate cannot guarantee a win, `-1` is printed.
