#include <bits/stdc++.h>

    a[i].idx = i;

  sort(a.begin() + 1, a.begin() + N + 1, [](const Node &A, const Node &B) {

  });

  vector<ll> prefix(N + 1);
  prefix[0] = 0;
  for (int i = 1; i <= N; ++i) {
    prefix[i] = prefix[i - 1] + a[i].val;
  }

  K -= prefix[N];

  // Calculate the minimum additional votes needed for each candidate.
  for (int i = 1; i <= N; ++i) {
    ll current_votes = prefix[i - 1] + a[i].val;
    ll additional_votes_needed = 0;

    // The current candidate needs to be in the top M candidates in the final count.
    int remaining_candidates = N - i + 1;

    // Find the number of candidates that need to be surpassed.
    int candidates_to_surpass = max(0, M - remaining_candidates);

    if (candidates_to_surpass > 0) {
      ll votes_to_surpass = prefix[N] - prefix[N - candidates_to_surpass];
      additional_votes_needed = votes_to_surpass - (prefix[i - 1] - prefix[N]);

      if (additional_votes_needed > K - current_votes) {

      ans[i].val = 0;
