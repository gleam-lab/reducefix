```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int N, M;
ll K;

struct Node {
  ll val;
  int idx;
} ;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);
  
  cin >> N >> M >> K;

  vector<Node> a(N + 5);

  for (int i = 1; i <= N; ++i) {
    cin >> a[i].val;
    a[i].idx = i;
  }

  sort(a.begin() + 1, a.begin() + N + 1, [](const auto &A, const auto &B){
    return A.val < B.val;
  });

  vector<ll> pre(N + 5);
  pre[0] = 0LL;
  for (int i = 1; i <= N; ++i) {
    pre[i] = pre[i - 1] + a[i].val;
  }

  K -= pre[N];

  vector<ll> ans(N + 5, -1);

  for (int i = 1; i <= N; ++i) {
    ll left = 0, right = K;
    while (left < right) {
      ll mid = (left + right) / 2;
      ll totalVotesNeeded = a[i].val + mid + (M - 1) * (a[i].val + mid + 1 - a[i].val);
      ll votesLeftForOthers = K - mid;
      if (totalVotesNeeded <= votesLeftForOthers) {
        right = mid;
      } else {
        left = mid + 1;
      }
    }
    if (a[i].val + left + (M - 1) * (a[i].val + left + 1 - a[i].val) <= K) {
      ans[a[i].idx] = left;
    }
  }

  for (int i = 1; i <= N; ++i) {
    cout << ans[i] << " ";
  }

  return 0;
}
```
