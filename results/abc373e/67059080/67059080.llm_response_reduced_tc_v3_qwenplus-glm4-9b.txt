
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

using ll = long long;

int main() {
    int N, M;
    ll K;
    cin >> N >> M >> K;

    vector<ll> A(N);
    ll sum_existing_votes = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        sum_existing_votes += A[i];
    }

    ll remaining_votes = K - sum_existing_votes;

    // Create a vector to store the votes and the index of the candidates
    vector<pair<ll, int>> votes(N);
    for (int i = 0; i < N; ++i) {
        votes[i] = {A[i], i + 1}; // +1 because candidates are 1-indexed
    }

    // Sort the votes to use binary search on them
    sort(votes.begin(), votes.end());

    vector<ll> min_votes_needed(N, -1);

    // Iterate over each candidate in descending order of current votes
    for (int i = 0; i < N; ++i) {
        ll current_votes = votes[i].first;
        ll index = votes[i].second;
        ll votes_needed_to_win = M - 1;

        // Binary search for the number of votes after which the candidate can win
        int l = i - 1, r = i + votes_needed_to_win;
        if (r >= N) {
            min_votes_needed[index - 1] = 0; // They can already win
            continue;
        }
        ll votes_to_win = votes[r].first - current_votes + 1;

        // If they need more than the remaining votes, they can't win
        if (votes_to_win > remaining_votes) {
            min_votes_needed[index - 1] = -1;
            continue;
        }

        // Calculate the minimum votes needed
        while (l < r) {
            int mid = (l + r + 1) / 2;
            votes_to_win = votes[mid].first - current_votes + 1;
            if (votes_to_win <= remaining_votes) {
                r = mid;
            } else {
                l = mid - 1;
            }
        }
        min_votes_needed[index - 1] = votes[l].first - current_votes + 1;
    }

    // Output the results
    for (ll needed : min_votes_needed) {
        cout << needed << " ";
    }
    cout << endl;

    return 0;
}
```

This code sorts the candidates by their current votes and then uses binary search to find the minimum number of additional votes each candidate needs to guarantee their victory. If a candidate cannot guarantee victory, the code sets the value to -1.
