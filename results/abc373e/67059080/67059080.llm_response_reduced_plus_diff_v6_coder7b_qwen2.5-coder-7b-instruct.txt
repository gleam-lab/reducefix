};

    a[i].idx = i;
  }

    return A.val < B.val;
  };

  sort(a.begin() + 1, a.begin() + N + 1, cmp);

  vector<ll> pre(N + 5);
  pre[0] = 0LL;
  for (int i = 1; i <= N; ++i) {
    pre[i] = pre[i - 1] + a[i].val;
  }

  K -= pre[N];
  if (K < 0) {
    cout << "-1";
    return 0;
  }

  vector<int> ans(N + 5, -1);

    ll cur = a[i].val + K;
    int idx = upper_bound(a.begin() + i + 1, a.begin() + N + 1, (Node) {cur, 0}, cmp) - a.begin();
    int cnt = (N - idx + 1);
    if (cnt >= M) {
      ans[a[i].idx] = 0;
      continue;
    }
    ll need = (cur + 1) * (M - cnt) - pre[idx - 1];
    if (need <= K) {
      ans[a[i].idx] = need;
    }

}
