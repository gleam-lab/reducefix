#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;
using ll = long long;

int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    int N, M, K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) cin >> A[i];
    
    // Pair of votes received and index, then sort by votes received
    vector<pair<ll, int>> votes(N);
    for (int i = 0; i < N; ++i) {
        votes[i] = {A[i], i};
    }
    sort(votes.begin(), votes.end());

    // Calculate the votes that need to be distributed
    ll votes_needed = 0;
    for (int i = 0; i < N; ++i) {
        votes_needed += votes[i].first;
    }
    K -= votes_needed;

    // Calculate the minimum additional votes needed for each candidate
    vector<ll> additional_votes(N);
    for (int i = 0; i < N; ++i) {
        if (M - 1 > i) { // If the number of votes needed is more than candidates
            additional_votes[votes[i].second] = 0;
        } else {
            // Calculate additional votes for the candidate to win
            ll votes_beaten = votes[i].first + votes_needed + (M - 1 - i) * K;
            additional_votes[votes[i].second] = votes_beaten - votes[i].first;
        }
    }

    // Output the results
    for (ll vote : additional_votes) {
        cout << vote << " ";
    }
    cout << endl;

    return 0;
}
