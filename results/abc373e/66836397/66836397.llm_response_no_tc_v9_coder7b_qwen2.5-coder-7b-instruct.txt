ll n, m, K;

bool check(ll i, ll x) {
    ll need = max(0LL, m - (n - i));
    if (need * 2 > x) return false;
    ll sum = accumulate(all(a), 0LL);
    if (sum + x < K) return false;
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);

    vector<ll> ans(n, -1);
    forn(i, n) {
        ll lo = 0, hi = K - accumulate(all(a), 0LL);
        while (lo <= hi) {
            ll mid = (lo + hi) / 2;
            if (check(i, mid)) {
                ans[i] = mid;
                hi = mid - 1;
            } else {
                lo = mid + 1;
            }
        }
    }

}

This solution uses binary search to determine the minimum number of votes required for each candidate to ensure their victory. It checks whether adding `x` votes to candidate `i` would result in them having enough votes to win, considering that up to `m` other candidates could also win. The binary search narrows down the range of possible values for `x` efficiently.
