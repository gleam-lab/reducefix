#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

ll binary_search(vector<ll>& A, vector<ll>& ac, ll K, ll m, ll i) {
    ll n = A.size();
    ll left = 0, right = K;
    while (left < right) {
        ll mid = (left + right) / 2;
        ll na = A[i] + mid;
        ll r = upper_bound(A.begin(), A.end(), na) - A.begin();
        ll uppercnt = n - r;
        ll rem = K - mid;
        ll M = m - uppercnt;
        if (M <= 0) {
            left = mid + 1;
        } else {
            ll l = max(r - M, 0LL);
            ll len = r - l;
            bool flag = A[l] <= A[i];
            if (flag) l--;
            ll cnt = (na + 1) * len - (ac[r] - ac[l] - (flag ? A[i] : 0));
            if (rem >= cnt) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
    }
    return left;
}

int main() {
    ll n, m, K;
    cin >> n >> m >> K;
    vector<ll> A(n);
    for (ll i = 0; i < n; i++) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    vector<ll> ac(n + 1, 0);
    for (ll i = 0; i < n; i++) {
        ac[i + 1] = ac[i] + A[i];
    }
    for (ll i = 0; i < n; i++) {
        ll result = binary_search(A, ac, K, m, i);
        if (result == K) {
            cout << -1 << " ";
        } else {
            cout << result << " ";
        }
    }
    cout << endl;
    return 0;
}

### Explanation of Changes
1. **Binary Search Function**: Created a helper function `binary_search` to encapsulate the logic for determining the minimum number of additional votes needed for each candidate.
2. **Efficiency**: The original code had a nested loop that was inefficient. The binary search reduces the complexity significantly.
3. **Edge Cases**: Handled edge cases where a candidate might already have enough votes or might not be able to win under any circumstances.
4. **Code Structure**: Improved readability by breaking down the logic into smaller functions and using meaningful variable names.
