#include <bits/stdc++.h>
using namespace std;

using ll = long long;

#define rep(i, x, limit) for (int i = (x); i < (limit); i++)

const ll infl = 1LL << 60;

int main() {
    ll N, M, K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    ll total = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        total += A[i];
    }
    ll rem = K - total;

    vector<pair<ll, int>> sortedA(N);
    for (int i = 0; i < N; ++i) {
        sortedA[i] = {A[i], i};
    }
    sort(sortedA.begin(), sortedA.end());

    vector<ll> prefixSum(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        prefixSum[i + 1] = prefixSum[i] + sortedA[i].first;
    }

    vector<ll> result(N);

    for (int i = 0; i < N; ++i) {
        ll orig = A[i];

        // Binary search on additional votes needed
        ll low = 0;
        ll high = rem;
        ll ans = -1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll newScore = orig + mid;

            // Find how many candidates have strictly more votes than newScore
            ll l = 0, r = N;
            while (l < r) {
                ll m = (l + r);
                if (m >= N || sortedA[m].first > newScore)
                    r = m - 1;
                else
                    l = m + 1;
            }

            // After loop, l is the index of last element <= newScore
            ll countGreater = N - (l + 1);

            if (countGreater >= M) {
                // Too many people above us, cannot guarantee win
                low = mid + 1;
                continue;
            }

            // Need to block top (M - 1) candidates from staying above us
            ll needBlock = M - 1 - countGreater;

            if (needBlock <= 0) {
                // Already in top M
                ans = mid;
                high = mid - 1;
                continue;
            }

            // Try to cap those ahead to at most newScore
            ll idx = l;
            while (idx >= 0 && sortedA[idx].first > orig) idx--;
            idx++;

            if (idx + needBlock > N) {
                low = mid + 1;
                continue;
            }

            ll required = (newScore + 1) * needBlock - (prefixSum[idx + needBlock] - prefixSum[idx]);

            if (required <= rem - mid) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        result[i] = (ans == -1 ? -1 : ans);
    }

    for (int i = 0; i < N; ++i) {
        cout << result[i] << " ";
    }
    cout << endl;

    return 0;
}
