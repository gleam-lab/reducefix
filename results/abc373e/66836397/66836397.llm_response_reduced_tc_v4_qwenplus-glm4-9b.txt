#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

int main() {
    long long n, m, K;
    cin >> n >> m >> K;
    vector<long long> a(n);
    for (long long &ai : a) {
        cin >> ai;
    }

    // Sort the votes received by each candidate
    sort(a.begin(), a.end());

    // Calculate the votes needed for a candidate to become the top M candidates
    vector<long long> needed(n);
    long long sum_votes = 0;
    for (int i = n - 1; i >= n - m; --i) {
        sum_votes += a[i];
    }
    for (int i = 0; i < n - m; ++i) {
        needed[i] = sum_votes - a[i];
    }

    // Calculate the remaining votes and the number of votes needed for the minimum candidate in top M
    long long remaining_votes = K - sum_votes;
    long long min_needed_for_m = needed[n - m];

    // Output the result for each candidate
    for (int i = 0; i < n; ++i) {
        if (i < n - m) {
            needed[i] = max(0LL, needed[i] - remaining_votes);
        } else {
            needed[i] = min_needed_for_m - remaining_votes;
        }
        cout << needed[i] << " ";
    }
    cout << endl;

    return 0;
}
