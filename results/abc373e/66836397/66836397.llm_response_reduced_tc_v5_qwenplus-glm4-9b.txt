#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

int main() {
    long long n, m, k;
    cin >> n >> m >> k;
    vector<long long> a(n);
    long long votes_sum = 0;
    for (long long &ai : a) {
        cin >> ai;
        votes_sum += ai;
    }

    // Sort candidates by number of votes
    sort(a.begin(), a.end());

    vector<long long> additional_votes(n, 0);

    // Iterate from the most voted to the least voted candidate
    for (int i = n - 1; i >= 0; --i) {
        long long votes_needed = votes_sum + a[i] - (m - 1) * a[i];
        if (votes_needed > k) { // More votes needed than remaining votes
            additional_votes[i] = -1;
        } else if (votes_needed <= votes_sum) { // Already have enough votes
            additional_votes[i] = 0;
        } else { // Need additional votes
            additional_votes[i] = votes_needed - votes_sum;
            votes_sum += additional_votes[i];
        }
    }

    // Output the result
    for (int i = 0; i < n; ++i) {
        cout << additional_votes[i] << (i < n - 1 ? " " : "\n");
    }

    return 0;
}
