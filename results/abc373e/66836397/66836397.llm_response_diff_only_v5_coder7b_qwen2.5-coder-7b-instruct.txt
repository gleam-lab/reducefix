#include<bits/stdc++.h>
using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long
#define vi vector<int>
#define vl vector<long long>
#define pii pair<int,int>
#define pll pair<ll,ll>
#define fi first
#define se second
#define endl "\n"

const int mod = 1e9 + 7;

bool check(int idx, int m, int k, vi &a, int rem) {
    int need = 0;
    int max_votes = a[idx];
    int votes_needed = 0;
    int num_wins = 0;
    for(int i = 0; i < a.size(); i++) {
        if(i != idx && a[i] > max_votes) {
            num_wins++;
            votes_needed += a[i] - max_votes;
        }
    }
    if(num_wins >= m) return true;
    need = m - num_wins;
    if(need * (max_votes + 1) - votes_needed <= rem) return true;
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n, m, k;
    cin >> n >> m >> k;
    vi a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a.begin(), a.end());
    vi ans(n, -1);
    int rem = k - accumulate(a.begin(), a.end(), 0LL);
    for(int i = 0; i < n; i++) {
        if(check(i, m, k, a, rem)) {
            ans[i] = 0;
        } else {
            int left = 0, right = rem;
            while(left <= right) {
                int mid = (left + right) / 2;
                if(check(i, m, k, a, rem - mid)) {
                    ans[i] = mid;
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            }
        }
    }
    for(auto x : ans) {
        cout << x << " ";
    }
    cout << endl;
    return 0;
}
