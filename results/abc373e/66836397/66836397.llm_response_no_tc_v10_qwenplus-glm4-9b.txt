#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

long long calculate_votes_needed(long long current_votes, long long remaining_votes, long long rank, long long m, long long n) {
    // Maximum votes that can be given to candidates ahead of rank if they win
    long long max_ahead = (n - rank) * (current_votes + 1);
    // Maximum votes that can be given to candidates after rank if they win
    long long max_behind = rank * current_votes;
    // Votes to distribute
    long long votes_to_distribute = remaining_votes - max_ahead - max_behind;

    // Calculate the minimum votes needed to be at least m-th rank
    if (m <= rank) {
        return 0; // Already in the m-th rank or better
    } else if (m - rank <= votes_to_distribute) {
        return 0; // Can reach m-th rank with remaining votes only
    } else {
        return m - rank - votes_to_distribute; // Need extra votes
    }
}

int main() {
    long long n, m, k;
    cin >> n >> m >> k;

    vector<long long> a(n);
    for (long long &ai : a) {
        cin >> ai;
    }

    // Sort the votes for binary search
    sort(a.begin(), a.end());

    vector<long long> results(n, -1);

    // Calculate the votes needed for each candidate
    for (int i = 0; i < n; ++i) {
        if (i < n - m) {
            // Calculate votes needed for this candidate to be at least m-th rank
            long long votes_needed = calculate_votes_needed(a[i], k - a[i], i, m, n);
            if (votes_needed != -1) {
                results[i] = votes_needed;
            }
        }
    }

    // Output the results
    for (int i = 0; i < n; ++i) {
        cout << results[i] << " ";
    }
    cout << endl;

    return 0;
}
