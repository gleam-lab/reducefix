#include <bits/stdc++.h>
using namespace std;

#define rep(i, x, limit) for (int i = (x); i < (limit); ++i)
using ll = long long;
using pll = pair<ll, ll>;

ll N, M, K;
vector<ll> A;
ll remaining;

// Binary search to find the minimum number of votes needed for candidate i to guarantee victory
bool is_possible(ll candidate_idx, ll X) {
    ll current = A[candidate_idx] + X;
    
    // We need to count how many candidates have more votes than current after this addition
    // Since other candidates can take at most remaining votes in total,
    // we need to ensure that even if all remaining votes are distributed optimally to others,
    // candidate still has enough votes to be in top M.

    vector<ll> temp = A;
    temp[candidate_idx] = current;

    sort(temp.begin(), temp.end(), greater<ll>());
    
    // The candidate must be in the top M positions
    return temp[M - 1] >= current ? false : true;
}

ll min_votes_needed(ll candidate_idx) {
    if (is_possible(candidate_idx, 0)) {
        return 0;
    }
    
    ll low = 0, high = remaining;
    ll result = -1;
    
    while (low <= high) {
        ll mid = (low + high) / 2;
        if (is_possible(candidate_idx, mid)) {
            result = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    
    return result;
}

int main() {
    cin >> N >> M >> K;
    A.resize(N);
    remaining = K;
    
    for (ll i = 0; i < N; ++i) {
        cin >> A[i];
        remaining -= A[i];
    }

    vector<pll> results;
    for (ll i = 0; i < N; ++i) {
        if (is_possible(i, 0)) {
            cout << "0 ";
        } else {
            ll res = min_votes_needed(i);
            cout << (res == -1 ? -1 : res) << " ";
        }
    }

    return 0;
}
