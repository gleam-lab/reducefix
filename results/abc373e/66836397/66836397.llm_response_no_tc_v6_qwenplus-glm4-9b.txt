#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>

using namespace std;

int main() {
    long long n, m, K;
    cin >> n >> m >> K;
    
    vector<long long> A(n);
    long long votes_given = 0;
    for (int i = 0; i < n; ++i) {
        cin >> A[i];
        votes_given += A[i];
    }
    
    // Sort the candidates' current votes in descending order
    sort(A.begin(), A.end(), greater<long long>());
    
    // Vector to store the minimum votes needed for each candidate to guarantee victory
    vector<long long> needed_votes(n, -1);
    
    // Iterate over each candidate
    for (int i = 0; i < n; ++i) {
        long long max_rival_votes = (i < m - 1) ? A[i + 1] : 0;
        
        // Calculate the votes needed to guarantee victory
        long long additional_votes_needed = m - 1 - (A[i] - max_rival_votes) + votes_given + 1 - K;
        if (additional_votes_needed <= 0) {
            needed_votes[i] = 0;
        } else if (additional_votes_needed <= K - votes_given) {
            needed_votes[i] = additional_votes_needed;
        } else {
            needed_votes[i] = -1;
        }
    }
    
    // Output the result
    for (int i = 0; i < n; ++i) {
        cout << needed_votes[i] << (i < n - 1 ? " " : "");
    }
    cout << endl;
    
    return 0;
}
