        ac[i + 1] = ac[i] + a[i];
    }

    vector<ll> ans(n, -1);

        ll na = a[i];
        ll r = lower_bound(a.begin(), a.end(), na + 1) - a.begin();

        ll cnt = na * (r - l) - (ac[r] - ac[l]);

        if (rem >= cnt) {
            ans[i] = 0;
        } else {
            ll x = (cnt - rem + na - 1) / na;
            ans[i] = max(x, 0LL);
        }
    }

    for (ll i = 0; i < n; ++i) {

    return 0;
}
