#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    ll total_votes = 0;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        total_votes += a[i];
    }
    k -= total_votes; // Update remaining votes after what we already have

    // Sort candidates by votes received (descending)
    sort(a.rbegin(), a.rend());

    vector<ll> votes_needed(n);
    for (int i = 0; i < n; ++i) {
        // Maximum votes this candidate needs to guarantee election
        ll max_votes = a[i] + (n - m - 1) * (a[i] + 1) / 2;
        if (a[i] + (n - m) * (a[i] + 1) / 2 <= k) {
            // If candidate i is already guaranteed to win
            votes_needed[i] = 0;
        } else if (max_votes > k) {
            // If candidate i cannot ever win
            votes_needed[i] = -1;
        } else {
            // Calculate the minimum votes needed for candidate i to guarantee victory
            ll required_votes = max_votes - (a[i] + (n - m) * (a[i] + 1) / 2);
            votes_needed[i] = required_votes;
        }
    }

    for (ll i = 0; i < n; ++i) {
        cout << votes_needed[i] << ' ';
    }
    cout << endl;

    return 0;
}
