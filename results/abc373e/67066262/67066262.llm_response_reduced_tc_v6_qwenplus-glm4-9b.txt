#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long ll;

int main() {
    int n, m;
    ll k;
    cin >> n >> m >> k;

    vector<ll> a(n);
    ll total_votes = 0;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        total_votes += a[i];
    }

    // Sort candidates by the number of votes they have received so far
    vector<pair<ll, int>> sorted_candidates(a.begin(), a.end());
    sort(sorted_candidates.begin(), sorted_candidates.end());

    // Calculate the votes needed to be the m-th most voted
    ll votes_needed_to_be_mth = sorted_candidates[n - m].first + m;

    // Find the candidate who is currently the m-th most voted
    int mth_index = -1;
    for (int i = 0; i < n; ++i) {
        if (sorted_candidates[i].first >= votes_needed_to_be_mth) {
            mth_index = i;
            break;
        }
    }

    // Calculate the votes needed for the remaining candidates to be less than or equal to the m-th most voted
    ll votes_needed_to_secure_victory = 0;
    if (mth_index != -1) {
        for (int i = mth_index + 1; i < n; ++i) {
            votes_needed_to_secure_victory += votes_needed_to_be_mth - sorted_candidates[i].first;
        }
    }

    // Calculate the number of additional votes needed for each candidate
    vector<ll> additional_votes_needed(n);
    for (int i = 0; i < n; ++i) {
        if (mth_index == -1 || sorted_candidates[i].first < votes_needed_to_be_mth) {
            additional_votes_needed[i] = votes_needed_to_secure_victory;
        } else {
            additional_votes_needed[i] = votes_needed_to_be_mth - sorted_candidates[i].first;
        }
    }

    // If there are not enough votes left to secure the victory for any candidate, set the result to -1
    if (votes_needed_to_secure_victory > k - total_votes) {
        memset(additional_votes_needed.data(), -1, sizeof(ll) * n);
    }

    // Output the result
    for (ll vote : additional_votes_needed) {
        cout << vote << " ";
    }
    cout << endl;

    return 0;
}
