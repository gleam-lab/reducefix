#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

typedef long long ll;

int main() {
    int n, m;
    ll k;
    cin >> n >> m >> k;

    vector<ll> a(n);
    ll sum_votes = 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        sum_votes += a[i];
    }

    // Remaining votes to be distributed
    k -= sum_votes;

    // Sort the candidates by current votes in descending order
    vector<pair<ll, int>> candidates(n);
    for (int i = 0; i < n; i++) {
        candidates[i] = {a[i], i};
    }
    sort(candidates.rbegin(), candidates.rend());

    // If m is 1, only the candidate with the most votes can win
    if (m == 1) {
        vector<ll> votes_needed(n, -1);
        for (int i = 1; i < n && candidates[i].first != candidates[0].first; i++) {
            votes_needed[candidates[i].second] = k;
        }
        votes_needed[candidates[0].second] = 0;
        for (ll i = 0; i < n; i++) {
            cout << (votes_needed[i] == -1 ? -1 : votes_needed[i]) << (i < n - 1 ? ' ' : '\n');
        }
        return 0;
    }

    // Calculate the minimum votes needed for each candidate to guarantee victory
    vector<ll> votes_needed(n, -1);
    ll max_votes = candidates[0].first;
    ll votes_needed_to_win = max_votes + m - 1;
    ll remaining_votes = k - (n - m) * votes_needed_to_win;
    vector<ll> sorted_votes(n);

    for (int i = 0; i < n; i++) {
        sorted_votes[i] = votes_needed_to_win - (max_votes - candidates[i].first);
        remaining_votes -= sorted_votes[i];
    }

    // Ensure that we distribute any remaining votes first to the candidates closest to victory
    sort(sorted_votes.begin(), sorted_votes.begin() + n - m, greater<ll>());
    for (ll i = 0; i < n - m; i++) {
        sorted_votes[i] += remaining_votes;
    }

    // Fill the votes_needed array
    for (int i = 0; i < n; i++) {
        votes_needed[candidates[i].second] = sorted_votes[i];
    }

    for (ll i = 0; i < n; i++) {
        cout << (votes_needed[i] == -1 ? -1 : votes_needed[i]) << (i < n - 1 ? ' ' : '\n');
    }

    return 0;
}
