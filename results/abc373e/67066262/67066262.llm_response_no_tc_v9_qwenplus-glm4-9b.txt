#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> votes(N);
    ll total_votes = 0;
    for (int i = 0; i < N; ++i) {
        cin >> votes[i];
        total_votes += votes[i];
    }

    // Sort candidates by the number of votes they currently have in descending order
    sort(votes.rbegin(), votes.rend());

    // Array to store the minimum additional votes needed for each candidate
    vector<ll> additional_votes(N);

    // Calculate the minimum votes needed for the M-th place candidate to secure victory
    ll votes_needed_for_Mth_place = votes[M-1] + (M < N ? M : 0) - votes[M];

    // Iterate over the candidates from the top down
    for (int i = 0; i < N; ++i) {
        // Calculate the maximum votes a candidate can get if all remaining votes go to them
        ll max_votes_without_remaining = votes[i] + K - total_votes;

        // If the candidate is already the M-th place candidate or better,
        // or if they can reach the M-th place by getting all remaining votes, they secure victory with 0 additional votes
        if (votes[i] >= votes_needed_for_Mth_place || max_votes_without_remaining >= votes_needed_for_Mth_place) {
            additional_votes[i] = 0;
        } else {
            // Calculate the number of votes needed to reach the M-th place candidate's votes
            additional_votes[i] = votes_needed_for_Mth_place - votes[i];
        }
    }

    // Output the results for each candidate
    for (ll &vote : additional_votes) {
        if (vote == 0) {
            cout << 0 << ' ';
        } else if (vote == -1) {
            cout << -1 << ' '; // This case will not occur due to the logic above
        } else {
            cout << vote << ' ';
        }
    }
    cout << '\n';

    return 0;
}
