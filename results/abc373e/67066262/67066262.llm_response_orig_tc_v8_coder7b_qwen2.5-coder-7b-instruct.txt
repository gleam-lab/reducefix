#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

bool check(vector<ll>& a, int n, int m, ll k, vector<int>& res) {
    vector<pair<ll, int>> v;
    for(int i = 0; i < n; i++) {
        v.push_back({a[i], i});
    }
    sort(v.rbegin(), v.rend());
    priority_queue<int> pq;
    for(int i = 0; i < m; i++) {
        pq.push(v[i].second);
    }
    int c = 0;
    for(int i = m; i < n; i++) {
        pq.push(v[i].second);
        pq.pop();
        c++;
    }
    unordered_set<int> s;
    while(!pq.empty()) {
        s.insert(pq.top());
        pq.pop();
    }
    for(int i = 0; i < n; i++) {
        if(s.count(i)) continue;
        if(a[i] + k < v[c].first) return false;
        res[i] = v[c].first - a[i];
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }
    vector<int> res(n, -1);
    if(check(a, n, m, k, res)) {
        for(int i = 0; i < n; i++) {
            if(res[i] == -1) res[i] = 0;
        }
    } else {
        for(int i = 0; i < n; i++) {
            res[i] = -1;
        }
    }
    for(int i = 0; i < n; i++) {
        cout << res[i] << " ";
    }
    cout << endl;
    return 0;
}
