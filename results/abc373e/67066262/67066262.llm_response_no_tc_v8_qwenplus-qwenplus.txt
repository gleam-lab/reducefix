#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        K -= A[i];
    }

    // Create sorted copy of A to use
    vector<ll> sortedA = A;
    sort(all(sortedA));

    // Prepare prefix sums of the sorted array
    vector<ll> prefix(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = prefix[i] + sortedA[i];
    }

    vector<ll> result(N);

    for (int i = 0; i < N; ++i) {
        ll low = 0, high = K + 1;
        ll current = A[i];

        while (low < high) {
            ll mid = (low + high) / 2;
            ll new_vote = current + mid;

            // Binary search for first candidate with votes > new_vote
            int idx = lower_bound(sortedA.begin(), sortedA.end(), new_vote + 1) - sortedA.begin();

            // We want to ensure that at most M-1 candidates have more votes than i
            // So we must increase those with votes <= new_vote among the top M candidates
            int left = max(0, N - M);
            if (idx <= left) {
                // Enough candidates already have <= new_vote
                high = mid;
            } else {
                // Need to raise i's votes higher
                low = mid + 1;
            }
        }

        // After binary search, check validity
        if (low > K) {
            result[i] = -1;
        } else {
            result[i] = low;
        }
    }

    for (ll x : result) {
        cout << x << " ";
    }
    cout << "\n";
}
