#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }
    auto comp = [&](int l, int r) { return a[l] < a[r]; };
    sort(a.begin(), a.end(), comp);
    vector<ll> prefix(n+1);
    for (int i = 0; i < n; ++i) {
        prefix[i+1] = prefix[i] + a[i];
    }
    vector<ll> ans(n, -1);
    for (int i = n - 1; i >= 0; --i) {
        if (i >= n - m) {
            ans[i] = 0;
        } else {
            ll votes_needed = m - 1 - (prefix[n] - prefix[i+1]) / n;
            if (k + votes_needed >= votes_needed) {
                ans[i] = votes_needed;
                k -= votes_needed;
            } else {
                ans[i] = -1;
            }
        }
    }
    for (auto &ai : ans) cout << ai << ' ';
    return 0;
}
