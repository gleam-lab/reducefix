```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        k -= a[i];
    }

    vector<int> ord(n);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return a[i] < a[j]; });

    vector<ll> ans(n, -1);
    for (int i = 0; i < n; ++i) {
        ll target = a[ord[i]] + 1;
        ll needed_votes = 0;
        int count_greater = 0;
        for (int j = i + 1; j < n; ++j) {
            if (a[ord[j]] > target) {
                ++count_greater;
            } else {
                needed_votes += max(target - a[ord[j]], 0LL);
            }
        }
        if (count_greater < m) {
            ans[ord[i]] = needed_votes;
        }
    }

    for (auto x : ans) {
        cout << x << " ";
    }
    cout << endl;

    return 0;
}
```
