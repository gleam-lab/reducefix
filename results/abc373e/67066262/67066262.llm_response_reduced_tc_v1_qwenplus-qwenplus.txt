#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        K -= A[i];
    }

    // Create sorted version of A
    vector<pair<ll, int>> sorted_A(N);
    for (int i = 0; i < N; ++i) {
        sorted_A[i] = {A[i], i};
    }
    sort(all(sorted_A));

    vector<ll> ans(N, -1);

    // Prefix sum array of sorted votes
    vector<ll> prefix(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = prefix[i] + sorted_A[i].first;
    }

    // For each candidate, binary search on minimum required additional votes
    for (int i = 0; i < N; ++i) {
        ll low = 0, high = K + 1;
        ll base = A[i];

        while (low < high) {
            ll mid = (low + high) / 2;
            ll target = base + mid;

            // Binary search to find how many candidates have strictly more votes than target in the sorted list
            int idx = upper_bound(sorted_A.begin(), sorted_A.end(), make_pair(target, N)) - sorted_A.begin();

            // If there are at least M candidates with more votes, this is not sufficient
            if (N - idx >= M) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }

        if (low <= K) {
            ans[i] = low;
        } else {
            ans[i] = -1;
        }
    }

    for (auto x : ans) {
        cout << x << ' ';
    }
    cout << '\n';

    return 0;
}
