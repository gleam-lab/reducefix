#include <bits/stdc++.h>

typedef long long ll;

int main() {

    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }

    // Calculate the prefix sum of sorted votes
    vector<ll> sorted_a(a);
    sort(sorted_a.begin(), sorted_a.end());
    vector<ll> pref(n + 1);
    for (int i = 0; i < n; i++) {
        pref[i + 1] = sorted_a[i] + pref[i];
    }

    vector<ll> ans(n, -1);

    for (int i = 0; i < n; i++) {
        // Calculate the maximum number of votes that can be distributed to other candidates without losing
        ll max_votes_for_others = 0;
        for (int j = 0; j < n; j++) {
            if (j != i) {
                max_votes_for_others += min(m - 1, sorted_a[j] - a[j]);
            }
        }

        // The remaining votes after accounting for the maximum votes for others
        ll remaining_votes = k - max_votes_for_others;

        // Check if we can still win even if we get all remaining votes
        if (a[i] + remaining_votes >= sorted_a[n - m]) {
            ans[i] = 0;
        } else {
            // Binary search for the minimum additional votes needed
            ll l = 0, r = remaining_votes;
            while (r - l > 1) {
                ll mid = (l + r) / 2;
                if (a[i] + mid >= sorted_a[n - m]) {
                    r = mid;
                } else {
                    l = mid;
                }
            }
            ans[i] = r;
        }
    }

    for (auto i : ans) cout << i << ' ';
    return 0;
}
