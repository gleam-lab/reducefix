```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        k -= a[i];
    }
    
    // Calculate the total votes needed to ensure a candidate wins
    ll total_votes_needed = 0;
    for(int i = 0; i < n; i++) {
        total_votes_needed += max(0LL, a[i] - (k / (n - i - 1)));
    }
    
    // Check if it's even possible for any candidate to win
    if(total_votes_needed > k) {
        for(int i = 0; i < n; i++) cout << -1 << " ";
        return 0;
    }
    
    vector<ll> ans(n, -1);
    for(int i = 0; i < n; i++) {
        if(a[i] + k / (n - i - 1) >= a[m]) {
            ans[i] = 0;
        } else {
            ans[i] = a[m] - (a[i] + k / (n - i - 1)) - 1;
        }
    }
    
    for(auto i : ans) cout << i << ' ';
    return 0;
}
```

### Explanation:
1. **Input Reading**: The code reads the number of candidates `n`, the threshold `m`, and the total number of votes `k`. It also reads the initial vote counts for each candidate into the vector `a`.

2. **Total Votes Needed Calculation**:
   - We calculate the total votes needed for each candidate to potentially win. This is done by iterating through each candidate and calculating the minimum votes required to surpass the candidate at position `m` (the `m-th` most voted candidate).

3. **Check Feasibility**:
   - If the total votes needed exceed the remaining votes `k`, it means no candidate can win, and we print `-1` for all candidates.

4. **Calculate Minimum Additional Votes**:
   - For each candidate, if they can win without needing additional votes (`a[i] + k / (n - i - 1) >= a[m]`), set their additional votes needed to `0`.
   - Otherwise, compute the minimum additional votes needed to ensure they surpass the `m-th` most voted candidate.

5. **Output**:
   - Finally, print the results for each candidate.

This approach ensures that the solution is efficient and handles all constraints and edge cases correctly.
