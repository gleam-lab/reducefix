#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // Find the candidate with the maximum votes to calculate the maximum votes needed
    ll max_votes = *max_element(a.begin(), a.end());
    ll votes_given = accumulate(a.begin(), a.end(), 0LL);

    // Vector to store the additional votes needed for each candidate
    vector<ll> additional_votes_needed(n, -1);

    // Iterate over the candidates in descending order of their current votes
    for (int i = 0; i < n; ++i) {
        // Calculate the number of candidates who have more votes than the current candidate
        ll candidates_with_more_votes = count_if(a.begin(), a.end(), [i](ll votes) { return votes > a[i]; });

        // If there are already M or fewer candidates with more votes, this candidate is already elected
        if (candidates_with_more_votes <= m) {
            additional_votes_needed[i] = 0;
            continue;
        }

        // Calculate the minimum votes needed to guarantee that no other candidate can beat this candidate
        ll votes_needed = m - candidates_with_more_votes + 1;
        votes_needed *= (max_votes - a[i] + 1); // Maximum votes this candidate can potentially have

        // Check if it is possible to achieve this number of votes with the remaining votes
        if (votes_needed - votes_given <= k) {
            additional_votes_needed[i] = votes_needed - votes_given;
        } else {
            additional_votes_needed[i] = -1; // Not possible to guarantee victory
        }
    }

    // Output the result
    for (ll vote_needed : additional_votes_needed) {
        cout << vote_needed << ' ';
    }
    cout << '\n';

    return 0;
}
