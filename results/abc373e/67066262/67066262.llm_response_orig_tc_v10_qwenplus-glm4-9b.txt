2. **Prefix Sum Calculation**: We compute the prefix sum of the sorted votes to quickly determine the total votes of the top M candidates.
3. **Binary Search for Minimum Votes Required**: For each candidate, we calculate the minimum additional votes required for them to secure their victory by checking how many votes they need to surpass the sum of votes of the top M candidates excluding themselves. We use binary search to find the minimum votes required in an efficient way.
4. **Edge Cases**: We need to handle the cases where a candidate is already guaranteed to win or cannot win under any circumstances.

    sort(votes.begin(), votes.end(), greater<long long>());

    vector<long long> additionalVotes(n);
    for (int i = 0; i < n; ++i) {

            additionalVotes[i] = 0;
        } else if (m <= n - votes[i]) {
            // Candidate can win if the remaining votes are distributed such that they are the top m candidates
            long long remainingVotes = k - prefixSum[n - m - 1];
            long long targetVotes = prefixSum[n - m - 1] + votes[i];

                long long left = 0, right = remainingVotes;

                    } else {
                        left = mid + 1;
                    }
                }
                additionalVotes[i] = left;
            }
        } else {
            // Candidate cannot win
            additionalVotes[i] = -1;
        }
    }
