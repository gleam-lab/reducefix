#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N = 2e5 + 5;
int n, m, k, sum, cnt, a[N], b[N], c[N];
bool check(int x, int y)
{
    int required_votes = x + y; // x is votes needed for a[i], y is remaining votes
    int z = upper_bound(b + 1, b + 1 + n - m + 1, required_votes + 1) - b - 1;
    if (z < n - m + 1)
        return 0; // More than m-1 people have more votes than a[i]
    // Calculate the total votes needed to secure the victory
    long long votes_needed = (x + 1) * (z - n + m) - (c[z - 1] - c[n - m]);
    // Check if the remaining votes are enough
    return required_votes * (m - 1) - votes_needed <= k - y;
}

int main()
{
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int i, j, x, y, z, t;
    cin >> n >> m >> k;
    for (i = 1; i <= n; i++)
        cin >> a[i], k -= a[i], b[i] = a[i];
    sort(b + 1, b + n + 1);
    for (i = 1; i <= n; i++)
        c[i] = c[i - 1] + b[i];
    for (i = 1; i <= n; i++)
    {
        int l = 0, r = k + 2; // r is set to k+2 to handle edge cases
        while (l < r - 1)
        {
            int mid = (l + r) / 2;
            if (check(a[i], mid))
                r = mid;
            else
                l = mid;
        }
        if (r <= k && check(a[i], r))
            cout << r << " ";
        else
            cout << "-1 ";
    }
    return 0;
}
