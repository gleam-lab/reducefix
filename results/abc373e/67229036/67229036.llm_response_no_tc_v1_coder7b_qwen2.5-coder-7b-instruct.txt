#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=2e5+5;
int n,m,k,sum,cnt,a[N],b[N],c[N];

bool check(ll x,ll y){
    int z=upper_bound(b+1,b+n,x+y+1)-b-1;
    if(z<m)
        return false;
    ll need = (x+y+1)*(z-m)-(c[z]-c[m-1]-max(x,b[m]));
    return need <= k-y;
}

signed main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    }
    sort(b+1,b+n+1);
    for(i=1;i<=n;i++)
        c[i]=c[i-1]+b[i];
    
    for(i=1;i<=n;i++){
        int l=0,r=k+1;
        while(l<r){
            int mid=(l+r)>>1;
            if(check(a[i],mid))
                r=mid;
            else
                l=mid+1;
        }
        if(l>k)
            cout<<"-1 ";
        else
            cout<<l<<" ";
    }
    return 0;
}

   The corrected function uses `upper_bound` to find the first position where the cumulative sum exceeds `x + y + 1`. Then, it checks if this position is greater than or equal to `m`. If not, it returns `false`.

2. **Binary Search Loop**: The loop in the `main` function ensures that we find the smallest `X` such that the candidate will always win. This is done correctly using binary search within the range `[0, k]`.
