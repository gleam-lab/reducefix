#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 2e5 + 5;

int n, m, k;
int a[N];
int sorted_votes[N];
int prefix_sum[N];

// Function to compute how many candidates can have votes >= threshold
int count_ge(const int threshold) {
    return n - (lower_bound(sorted_votes + 1, sorted_votes + n + 1, threshold) - sorted_votes);
}

bool is_victory_possible(int base, int additional) {
    int total = base + additional;
    // We want to assume that other candidates get as many votes as possible without beating us
    // So we pick top M-1 candidates from remaining votes who can beat us
    // If we can prevent at least (M) candidates from getting more than us, we win

    // We want to find the maximum number of candidates that can have votes > total
    // This is equivalent to how many have current votes > total

    int available = k - additional;
    if (available < 0) return false;

    // Binary search for how many candidates can potentially have more than total
    int low = 0, high = n, res = -1;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (mid >= n) break;

        // Take the top 'mid' candidates after excluding current candidate
        int required = 0;
        for (int i = n; i >= max(1LL, n - mid + 1); --i) {
            int cand = sorted_votes[i];
            if (cand > total) {
                required += (cand - total);
                if (required > available) break;
            }
        }

        if (required <= available) {
            res = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    // If at least M candidates can have votes > total, then we don't win
    return res < m;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m >> k;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        sorted_votes[i] = a[i];
    }
    sort(sorted_votes + 1, sorted_votes + n + 1);

    for (int i = 1; i <= n; ++i) {
        int l = 0, r = k + 1;
        int ans = -1;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (is_victory_possible(a[i], mid)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        if (ans == -1) {
            cout << "-1 ";
        } else {
            cout << ans << " ";
        }
    }
    cout << endl;

    return 0;
}
