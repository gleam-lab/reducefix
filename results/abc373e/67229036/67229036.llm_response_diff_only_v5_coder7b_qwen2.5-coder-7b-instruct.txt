#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=2e5+5;
int n,m,k,suffix_sum[N];

// Function to check if candidate i can win with x more votes
bool can_win(int i, int x){
    int remaining_votes = k - s[i] - x; // Remaining votes after giving x to candidate i
    int count = 0; // Count of candidates having more votes than candidate i
    for(int j=1; j<=n; ++j){
        if(j != i && suffix_sum[j] + x > s[i]){
            ++count;
        }
    }
    return count < m;
}

signed main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> n >> m >> k;
    vector<ll> a(n+1), b(n+1), c(n+1);

    for(int i=1; i<=n; ++i){
        cin >> a[i];
        s[i] = a[i]; // Sum of votes till candidate i
    }

    sort(s.begin()+1, s.end()); // Sorting the vote counts
    suffix_sum[n] = s[n]; // Suffix sum array initialization

    for(int i=n-1; i>=1; --i){
        suffix_sum[i] = suffix_sum[i+1] + s[i]; // Filling the suffix sum array
    }

    for(int i=1; i<=n; ++i){
        if(can_win(i, 0)){
            cout << "0 "; // Candidate i already wins without any extra votes
        } else {
            int low = 0, high = k - s[i], ans = -1;
            while(low <= high){
                int mid = low + (high-low)/2;
                if(can_win(i, mid)){
                    ans = mid;
                    high = mid-1;
                } else {
                    low = mid+1;
                }
            }
            cout << ans << " "; // Minimum extra votes required for candidate i to win
        }
    }
    return 0;
}
