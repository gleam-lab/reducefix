    }
    
    vector<int> sorted_A = A;
    sort(sorted_A.begin(), sorted_A.end(), greater<int>());
    int threshold = sorted_A[M - 1];
    
    for (int i = 0; i < N; ++i) {
        if (A[i] >= threshold) {
            cout << "0 ";
            continue;
        }
        int required = threshold - A[i] + 1;
        if (required <= K) {

        }
    }

    return 0;
}
