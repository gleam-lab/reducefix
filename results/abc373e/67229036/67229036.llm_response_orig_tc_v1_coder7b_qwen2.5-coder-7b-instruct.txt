#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);

    ll n, m, k; cin >> n >> m >> k;
    vector<ll> a(n), b(n);
    for (ll &x : a) cin >> x, k -= x;

    sort(a.begin(), a.end());
    partial_sum(a.begin(), a.end(), b.begin());

    vector<ll> c(n);
    for (ll i = 0; i < n; ++i) {
        ll need = 0;
        if (i + 1 > m) {
            need = max(0LL, a[i] - (b.back() - b[i]));
        }
        c[i] = max(0LL, need + k / (n - i));
    }

    for (ll x : c) cout << x << " ";

    return 0;
}
