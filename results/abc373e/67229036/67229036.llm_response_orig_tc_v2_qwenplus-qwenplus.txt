```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 2e5 + 5;

int n, m, k;
int a[N], sorted_a[N];

bool is_possible(int idx, int x) {
    // Candidate idx gets x more votes -> total becomes a[idx] + x
    int total = a[idx] + x;

    int remaining = k - x;
    if (remaining < 0) return false;

    // Try to assign as many votes as possible to other candidates to beat this one
    int required = 0;
    int extra = remaining;

    for (int i = n; i >= 1; --i) {
        if (i == idx) continue;
        int cap = max(total - 1, (int)0);
        int add = min(cap - sorted_a[i], remaining);
        if (add < 0) continue;
        required += add;
        if (required > extra) return false;
    }

    return required <= extra;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m >> k;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        sorted_a[i] = a[i];
    }

    sort(sorted_a + 1, sorted_a + n + 1);

    vector<int> result(n + 1);

    for (int i = 1; i <= n; ++i) {
        int l = 0, r = k;
        int ans = -1;

        while (l <= r) {
            int mid = (l + r) / 2;
            if (is_possible(i, mid)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }

        result[i] = (ans != -1 ? ans : -1);
    }

    for (int i = 1; i <= n; ++i) {
        cout << result[i];
        if (i != n) cout << " ";
    }
    cout << endl;

    return 0;
}
```
