#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 2e5 + 5;
int n, m, k, sum, cnt, a[N], b[N], c[N];
int result[N];

bool check(int x, int y) {
    int z = upper_bound(b, b + n, x + y) - b - 1;
    if (z < n - m) return 0;
    return (x + y) * (z - n + m) - (c[z] - c[n - m] - max(x, b[n - m])) <= k - y;
}

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        sum += a[i];
    }
    for (int i = 0; i < n; i++) b[i] = a[i];
    sort(b, b + n);
    for (int i = 0; i < n; i++) c[i] = c[i - 1] + b[i];
    for (int i = 0; i < n; i++) {
        int l = 0, r = k + 1;
        while (l + 1 < r) {
            int mid = (l + r) / 2;
            if (check(a[i], mid)) {
                r = mid;
            } else {
                l = mid;
            }
        }
        result[i] = (l < k && check(a[i], l)) ? l : -1;
    }
    for (int i = 0; i < n; i++) {
        if (result[i] == -1 && (i == 0 || result[i - 1] == -1)) {
            int required_votes = m * (sum + 1) - c[n - 1] + a[i];
            if (required_votes <= k) {
                result[i] = required_votes - sum;
            }
        }
    }
    for (int i = 0; i < n; i++) {
        if (result[i] == -1 && (i == n - 1 || result[i + 1] == -1)) {
            int required_votes = n * (sum + 1) - c[n - 1] - a[i];
            if (required_votes <= k) {
                result[i] = required_votes - sum;
            }
        }
    }
    for (int i = 0; i < n; i++) cout << result[i] << (i < n - 1 ? " " : "");
    return 0;
}
