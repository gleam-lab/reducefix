#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N = 2e5 + 5;
int n, m, k, sum, cnt, a[N], b[N], c[N];
int solve() {
    // Sort candidates based on votes they have already received in descending order
    sort(a + 1, a + n + 1, greater<int>());
    
    // c[i] will store the cumulative votes of the top i candidates
    for (int i = 1; i <= n; ++i) {
        c[i] = c[i - 1] + a[i];
    }
    
    // b[i] will store the total votes the top i candidates need to guarantee victory
    for (int i = 1; i <= n; ++i) {
        // Calculate the number of votes needed to exceed the m-th candidate
        int votes_needed = m * (c[m] / m + 1) - c[m];
        b[i] = votes_needed - a[i];
    }

    // Sort the votes needed in descending order
    sort(b + 1, b + n + 1, greater<int>());

    // Calculate the minimum votes needed for each candidate
    for (int i = 1; i <= n; ++i) {
        // Check if the i-th candidate can guarantee victory
        if (a[i] + b[i] >= m * (c[m] / m + 1)) {
            // If a[i] + b[i] is greater than or equal to the votes of the m-th candidate
            // and the sum of votes of the top m candidates is greater than or equal to m times the
            // votes of the i-th candidate, they can guarantee victory
            if (c[m] >= m * (a[i] + b[i] / m)) {
                cout << b[i] << " ";
            } else {
                cout << "-1 ";
            }
        } else {
            cout << "-1 ";
        }
    }
    return 0;
}

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    solve();
    return 0;
}
