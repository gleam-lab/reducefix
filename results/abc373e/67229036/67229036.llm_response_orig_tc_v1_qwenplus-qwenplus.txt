#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 2e5 + 5;

int n, m, k;
int a[N];
int sorted_a[N];

bool is_possible(int idx, int X) {
    // Candidate at idx gets X more votes: a[idx] + X
    int total = a[idx] + X;

    // We need at least (n - m) candidates with votes >= total
    // because then less than M candidates can have strictly more than total
    int cnt = 0;
    for (int i = n - 1; i >= 0; --i) {
        if (i == idx) continue;
        if (a[i] > total) cnt++;
    }

    return cnt < m;
}

int compute_min_votes(int idx) {
    if (is_possible(idx, 0)) return 0;

    int lo = 0, hi = k + 1;
    while (lo < hi) {
        int mid = (lo + hi) / 2;
        if (is_possible(idx, mid)) {
            hi = mid;
        } else {
            lo = mid + 1;
        }
    }

    if (lo > k) return -1;
    return lo;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    vector<int> res(n);
    for (int i = 0; i < n; ++i) {
        res[i] = compute_min_votes(i);
    }

    for (int i = 0; i < n; ++i) {
        cout << res[i] << " ";
    }
    cout << endl;

    return 0;
}
