To solve this problem, we need to determine the minimum number of additional votes each candidate needs to guarantee a win regardless of how the remaining votes are distributed among other candidates.

### Key Observations:
1. **Candidate Victory Condition**:
   A candidate `i` is elected if fewer than `M` other candidates have strictly more votes than them after all votes are counted.
   
2. **Binary Search Strategy**:
   For each candidate, we perform a binary search over the possible number of additional votes they could receive (`X`) from the remaining `K - sum(A)` votes. We simulate the worst-case distribution of these votes to verify whether `X` is sufficient for the candidate to still be elected.

3. **Sorting and Prefix Sum**:
   To simulate the worst-case scenario efficiently:
   - Sort the current vote counts.
   - Maintain prefix sums to allow fast computation of total votes in ranges.

4. **Worst Case Simulation**:
   When checking if `X` votes added to candidate `i` guarantees victory:
   - Assume all other candidates (except the top `n - M` candidates) take as many of the remaining votes as possible to beat candidate `i`.
   - Use binary search to find the smallest X such that no matter how the rest of the votes are allocated, candidate `i` remains in the top `M` vote-getters.

#include <bits/stdc++.h>
using namespace std;
#define int long long

const int N = 2e5 + 5;
int n, m, k;
int a[N], b[N], prefix[N];

// Check if candidate i with x votes added can be elected
bool check(int i, int x) {
    int total_used = 0;
    // Assign initial votes: candidate i gets x extra
    int my_votes = a[i] + x;

    // Try to assign up to x votes to other candidates to maximize those above me
    // We pick top (n - M) candidates who can get boosted to be >= my_votes

    // Binary search the first index where b[j] >= my_votes
    int low = 0, high = n, pos = n;
    while (low < high) {
        int mid = (low + high) / 2;
        if (b[mid] >= my_votes)
            high = mid;
        else
            low = mid + 1;
    }
    pos = low;

    // Now we try to increase (n - M) other candidates to at least my_votes
    int needed = max(0LL, (n - m) - (n - pos));
    int left = x;

    // Greedily boost top (n - M) candidates to just exceed or equal my_votes
    for (int j = 0; j < needed; ++j) {
        int idx = n - 1 - j;
        if (idx >= pos) continue;
        int delta = my_votes - b[idx];
        if (delta > 0) {
            if (left < delta) return false;
            left -= delta;
        }
    }

    return true;
}

    for (int i = 0; i < n; ++i)
        b[i] = a[i];
    sort(b, b + n);

    // Compute prefix sums
    prefix[0] = 0;
    for (int i = 1; i <= n; ++i)
        prefix[i] = prefix[i - 1] + b[i - 1];

    // For each candidate, binary search on minimum required additional votes
    for (int i = 0; i < n; ++i) {
        int lo = 0, hi = k + 1, ans = -1;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (check(i, mid)) {
                ans = mid;
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }

        // Output result
        if (ans <= k)
            cout << ans << " ";

    }

    return 0;
}
