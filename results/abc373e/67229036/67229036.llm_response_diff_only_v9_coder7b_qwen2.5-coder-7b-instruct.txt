#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=2e5+5;
ll n,m,k,sum,cnt,a[N],b[N],c[N];

bool check(ll x,ll y){
    int z=upper_bound(b+1,b+1+n,x+y+1)-b-1;
    if(z<n-m)
        return false;
    ll left_votes_needed=max((x+y+1)*(z-n+m)-(c[z]-c[n-m-1]-max(x,b[n-m])),0LL);
    return left_votes_needed<=y;
}

signed main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int i,j,x,y,z,t;
    cin>>n>>m>>k;
    sum=0;
    for(i=1;i<=n;i++){
        cin>>a[i]; 
        k-=a[i];
        sum+=a[i];
        b[i]=a[i];
    }
    sort(b+1,b+n+1);
    c[0]=0;
    for(i=1;i<=n;i++)
        c[i]=c[i-1]+b[i];
    
    for(i=1;i<=n;i++){
        int l=0,r=k+1;
        while(l<r){
            int mid=(l+r)>>1;
            if(check(a[i],mid))
                r=mid;
            else
                l=mid+1;
        }
        if(l>k)
            cout<<"-1 ";
        else
            cout<<l<<" ";
    }
    return 0;
}
