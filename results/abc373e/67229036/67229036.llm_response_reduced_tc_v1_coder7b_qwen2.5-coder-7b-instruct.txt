#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=2e5+5;
int n,m,k,sum,cnt,a[N],b[N],c[N];

bool check(int x,int y){
    int z=lower_bound(b+1,b+1+n,x+y+1)-b-1;
    if(z<m)
        return false;
    return (x+y+1)*(z-m)-(c[z]-c[m-1]-max(x,b[m]))<=k-y;
}

signed main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int i,j,x,y,z,t;
    cin>>n>>m>>k;
    sum=0;
    for(i=1;i<=n;i++){
        cin>>a[i]; sum+=a[i];
    }
    k-=sum;
    for(i=1;i<=n;i++) b[i]=a[i];
    sort(b+1,b+n+1);
    c[0]=0;
    for(i=1;i<=n;i++) c[i]=c[i-1]+b[i];
    vector<ll> res(n,-1);
    for(i=1;i<=n;i++){
        int l=0,r=k+1;
        while(l<r){
            int mid=(l+r)>>1;
            if(check(a[i],mid)) r=mid;
            else l=mid+1;
        }
        if(l<=k) res[i-1]=l;
    }
    for(auto x:res) cout<<x<<" ";
    return 0;
}
