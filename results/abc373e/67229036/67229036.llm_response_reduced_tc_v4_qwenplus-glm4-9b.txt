#include<bits/stdc++.h>
using namespace std;
#define int long long

const int N=2e5+5;
int n, m, k, sum;
vector<pair<int, int>> candidates;

bool check(int x) {
    // x is the number of votes we need to guarantee the victory for the candidate at index i
    int extra_votes = x - candidates[i-1].first;
    int remaining_votes = k - sum;
    int more_votes_needed = 0;

    // Count how many candidates can potentially get more votes than the current candidate
    for (int j = i + 1; j < n && candidates[j].first <= candidates[i].first + extra_votes; ++j) {
        more_votes_needed += candidates[j].second;
    }

    // If the remaining votes are still less than the votes needed to guarantee the victory
    // and the number of candidates that can potentially get more votes is less than m, we fail
    if ((remaining_votes - more_votes_needed) < (n - i - m - 1)) {
        return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    cin >> n >> m >> k;
    sum = 0;
    for (int i = 0; i < n; ++i) {
        int a;
        cin >> a;
        sum += a;
        candidates.emplace_back(a, 1);
    }
    // Sort candidates by their number of votes received so far
    sort(candidates.begin(), candidates.end());

    // Initialize the votes needed array with -1
    vector<int> votes_needed(n, -1);

    // Iterate through each candidate
    for (int i = 0; i < n; ++i) {
        // If the candidate has already won, continue to the next
        if (i >= m - 1) {
            if (check(candidates[i].first)) {
                votes_needed[i] = 0;
                continue;
            }
        }
        // Binary search to find the minimum votes needed to guarantee victory
        int l = 0, r = k - sum - candidates[i].first;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (check(candidates[i].first + mid)) {
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        // If we have found a valid number of votes, store it
        if (l <= k - sum) {
            votes_needed[i] = l;
        }
    }

    // Output the results
    for (int i : votes_needed) {
        cout << i << " ";
    }
    cout << endl;
    return 0;
}
