#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    for (ll &a : A) {
        cin >> a;
        K -= a;
    }

    // We'll process each candidate independently
    vector<ll> res(N, -1);

    // Create a sorted copy of A to use in binary search
    vector<ll> sorted_A = A;
    sort(all(sorted_A));

    for (int i = 0; i < N; ++i) {
        ll target_votes = A[i];
        
        // Binary search on the number of additional votes needed
        ll low = 0, high = K;
        ll answer = -1;

        while (low <= high) {
            ll mid = (low + high) / 2;
            ll my_total = target_votes + mid;

            // We need to ensure that at least (N - M) candidates have <= my_total votes
            // Find how many candidates already have > my_total votes without additional votes
            ll count = 0;
            for (int j = 0; j < N; ++j) {
                if (j == i) continue;
                if (A[j] > my_total) count += (A[j] - my_total);
            }

            if (count <= K - mid) {
                answer = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        // If we found a valid X, assign it. Otherwise -1
        res[i] = (answer != -1) ? answer : -1;
    }

    // Print result
    for (ll x : res) cout << x << " ";
    cout << "\n";

    return 0;
}
