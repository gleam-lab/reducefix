3. **Check Feasibility**: Ensure that the total votes required (including X) do not exceed the remaining votes (K).

### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int N, M;
    ll K;
    cin >> N >> M >> K;
    vector<ll> A(N);
    ll total = 0;
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        total += A[i];
    }
    ll remaining = K - total;
    vector<int> ord(N);
    iota(all(ord), 0);
    sort(all(ord), [&](int i, int j) { return A[i] < A[j]; });
    vector<ll> sortedA = A;
    sort(all(sortedA));
    vector<ll> prefix(N + 1);
    for (int i = 0; i < N; ++i) {
        prefix[i + 1] = sortedA[i] + prefix[i];
    }
    vector<ll> ans(N, -1);
    for (int idx = 0; idx < N; ++idx) {
        int i = ord[idx];
        ll low = 0, high = remaining, best = -1;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll newVotes = A[i] + mid;
            int pos = upper_bound(all(sortedA), newVotes) - sortedA.begin();
            int needed = N - pos;
            if (needed < M) {
                best = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        if (best != -1) {
            ans[i] = best;
        }
    }
    for (int i = 0; i < N; ++i) {
        cout << ans[i] << " ";
    }
    cout << endl;
    return 0;
}
```

### Explanation
1. **Sorting and Prefix Sum**: The candidates are sorted by their current votes, and a prefix sum array is constructed to facilitate quick sum calculations.
2. **Binary Search for Each Candidate**: For each candidate, a binary search is conducted to find the minimum additional votes (X) needed. The search checks if adding X votes to the candidate ensures that fewer than M candidates have more votes.
3. **Feasibility Check**: The binary search adjusts the range based on whether the current mid value of X allows the candidate to be elected. If feasible, it tries to find a smaller X; otherwise, it increases X.
4. **Output**: The results for all candidates are printed, indicating the minimum votes needed or -1 if it's impossible.

This approach ensures correctness and efficiency, handling the large constraints effectively.
