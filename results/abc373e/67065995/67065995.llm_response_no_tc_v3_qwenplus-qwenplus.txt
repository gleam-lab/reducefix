```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // Create sorted version of current votes
    vector<ll> sorted_a = a;
    sort(all(sorted_a));

    vector<ll> prefix(n + 1);
    for (int i = 0; i < n; ++i) {
        prefix[i + 1] = prefix[i] + sorted_a[i];
    }

    vector<ll> res(n, -1);

    for (int i = 0; i < n; ++i) {
        ll low = 0, high = k;
        ll my_vote = a[i];

        // Binary search on required additional votes
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll target = my_vote + mid;

            // Find how many candidates have vote > target in the worst case
            // We use binary search to find first candidate with votes > target
            int idx = upper_bound(all(sorted_a), target) - sorted_a.begin();

            // Total number of candidates who can possibly have more votes than target
            int count = n - idx;

            // To prevent being beaten by more than M-1 others
            if (count < m) {
                // This many votes is sufficient
                res[i] = mid;
                high = mid - 1;
            } else {
                // Not enough votes, try higher
                low = mid + 1;
            }
        }

        if (res[i] > k) {
            res[i] = -1; // Impossible to guarantee win
        }
    }

    for (auto x : res) {
        cout << x << " ";
    }
    cout << "\n";
}
```
