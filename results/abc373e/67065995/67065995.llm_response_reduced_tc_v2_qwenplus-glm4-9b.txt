#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ll = long long;

int main() {
    int n, m, k;
    cin >> n >> m >> k;

    vector<ll> votes(n);
    for (int i = 0; i < n; ++i) {
        cin >> votes[i];
    }

    // Sort the votes to determine the current ranking
    sort(votes.begin(), votes.end());

    // Calculate the votes left to distribute
    ll remaining_votes = k - accumulate(votes.begin(), votes.end(), 0LL);

    // Prepare the answer array with default values
    vector<ll> ans(n, -1);

    // Iterate over candidates in decreasing order of votes they have
    for (int i = n - 1; i >= 0; --i) {
        // If the current candidate already has more votes than the maximum allowed losers (m), they are safe
        if (i < m) {
            ans[i] = 0;
            continue;
        }

        // Calculate the minimum votes needed to guarantee a win
        ll votes_needed = votes[i] + remaining_votes + (m - 1) - votes[i - m];

        // If the votes needed exceed the remaining votes, it's impossible for this candidate to win
        if (votes_needed > remaining_votes) {
            continue;
        }

        // Check if the candidate can secure a win with the calculated votes needed
        bool can_win = true;
        ll current_votes = votes[i] + votes_needed;
        ll current_rank = upper_bound(votes.begin(), votes.end(), current_votes) - votes.begin();

        // If the current rank is less than m, the candidate can win
        if (current_rank >= m) {
            ans[i] = votes_needed;
        }
    }

    // Output the results
    for (int i = 0; i < n; ++i) {
        cout << (ans[i] == -1 ? -1 : ans[i]) << (i < n - 1 ? ' ' : '\n');
    }

    return 0;
}
