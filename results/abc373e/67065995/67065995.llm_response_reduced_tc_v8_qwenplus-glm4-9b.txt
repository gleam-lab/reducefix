#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    ll total_votes = 0;
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
        total_votes += a[i];
    }
    k -= total_votes;

    vector<pair<ll, int>> candidates(n);
    for(int i = 0; i < n; ++i) {
        candidates[i].first = a[i];
        candidates[i].second = i;
    }
    sort(candidates.begin(), candidates.end());

    vector<ll> current_votes(n, 0);
    vector<ll> prefix_sum(n + 1, 0);
    vector<ll> max_votes(n, -1);
    ll current_max = 0;

    for(int i = n - 1; i >= 0; --i) {
        ll votes_needed = m - (n - i - 1);
        if (current_votes[i] + votes_needed <= k) {
            max_votes[candidates[i].second] = 0; // Already guaranteed
        } else {
            max_votes[candidates[i].second] = votes_needed - (k - current_votes[i]);
        }
        current_votes[i] += max_votes[candidates[i].second];
        prefix_sum[i] = prefix_sum[i + 1] + max_votes[candidates[i].second];
        current_max = max(current_max, max_votes[candidates[i].second]);
    }

    for(int i = 0; i < n; ++i) {
        if (max_votes[i] == 0) {
            continue;
        }
        ll votes_needed = m - (n - i - 1);
        ll max_possible_votes = current_votes[i] + (a[i] - candidates[i].first + current_max);
        ll min_votes_needed = votes_needed - (max_possible_votes - a[i]);
        if (min_votes_needed <= k) {
            cout << min_votes_needed << ' ';
        } else {
            cout << -1 << ' ';
        }
    }
}
