#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, m;
    ll k;
    cin >> n >> m >> k;
    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    vector<pair<ll, int>> sorted_a(n);
    for (int i = 0; i < n; ++i) {
        sorted_a[i] = {a[i], i};
    }
    sort(sorted_a.begin(), sorted_a.end());

    vector<ll> remaining_votes(k);
    for (ll i = 0; i < k; ++i) {
        remaining_votes[i] = i;
    }

    vector<ll> ans(n, -1);
    for (int i = 0; i < n; ++i) {
        ll current_votes = sorted_a[i].first;
        ll rank = i + 1; // Rank starting from 1
        ll required_votes = m * (current_votes + 1) - rank;
        if (required_votes <= 0) {
            ans[sorted_a[i].second] = 0;
            continue;
        }

        ll min_votes_needed = required_votes;
        if (min_votes_needed > k) {
            ans[sorted_a[i].second] = -1;
            continue;
        }

        // Find the minimum votes needed from the remaining ballots
        ll j = lower_bound(remaining_votes.begin(), remaining_votes.end(), min_votes_needed) - remaining_votes.begin();
        ll votes_needed = remaining_votes[j];

        ans[sorted_a[i].second] = votes_needed - min_votes_needed;
        // Update the remaining votes
        for (int k = j; k < remaining_votes.size() && remaining_votes[k] >= votes_needed; ++k) {
            remaining_votes[k] -= votes_needed;
        }
    }

    for (int i = 0; i < n; ++i) {
        cout << ans[i] << " ";
    }
    cout << endl;
    return 0;
}
