    vector<ll> pref(n + 1);

    vector<ll> res(n, -1);

        ll need = 0;

        // Binary search to find the minimum additional votes needed
        ll lo = 0, hi = k + 1;
        while (lo < hi) {
            ll mid = (lo + hi) / 2;
            ll extra_votes = mid * target - (pref[min(n, i + target)] - pref[i]);

            if (extra_votes > k) {
                hi = mid;
            } else {
                need = mid;
                lo = mid + 1;
            }
        }

        // Check if the calculated need is valid
        if (need <= k) {
            res[idx[i]] = need;
        }
    }

    // Output the results
    for (int i = 0; i < n; ++i) {

}
