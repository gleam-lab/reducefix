To solve this problem, we need to simulate the rising sea levels and keep track of the areas that remain above water. The key steps are as follows:

1. Parse the input to get the dimensions of the island and the elevation matrix.
2. For each year, determine which sections of the island sink based on the current sea level.
3. Update the elevation matrix to reflect the new state after each year.
4. Calculate the remaining area above water and output it.

Here's the corrected and optimized C++ code to achieve this:

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef pair<int, pair<int, int>> pii;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, y;
    cin >> h >> w >> y;

    vector<vector<int>> height(h, vector<int>(w));
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> height[i][j];
        }
    }

    vector<vector<bool>> visited(h, vector<bool>(w, false));
    queue<pii> q;

    // Directions: up, down, left, right
    const int di[] = {-1, 1, 0, 0};
    const int dj[] = {0, 0, -1, 1};

    for (int year = 1; year <= y; ++year) {
        // Initialize the queue with cells that will sink in this year
        for (int i = 0; i < h; ++i) {
            for (int j = 0; j < w; ++j) {
                if (!visited[i][j] && height[i][j] <= year) {
                    q.push({height[i][j], {i, j}});
                    visited[i][j] = true;
                }
            }
        }

        // Process the queue to sink all connected cells
        while (!q.empty()) {
            auto [h, pos] = q.front();
            q.pop();
            for (int k = 0; k < 4; ++k) {
                int ni = pos.first + di[k];
                int nj = pos.second + dj[k];
                if (ni >= 0 && ni < h && nj >= 0 && nj < w && !visited[ni][nj] && height[ni][nj] <= year) {
                    visited[ni][nj] = true;
                    q.push({height[ni][nj], {ni, nj}});
                }
            }
        }

        // Count the remaining area above water
        int remaining_area = 0;
        for (int i = 0; i < h; ++i) {
            for (int j = 0; j < w; ++j) {
                if (!visited[i][j]) {
                    remaining_area++;
                }
            }
        }

        // Reset the visited array for the next year
        fill(visited.begin(), visited.end(), vector<bool>(w, false));

        // Print the result for the current year
        cout << remaining_area << '\n';
    }

    return 0;
}
```

### Explanation:
1. **Initialization**: We read the dimensions `h` and `w` of the island and the number of years `y`. We then read the elevation matrix into a 2D vector `height`.
2. **Queue Initialization**: For each year, we initialize a queue with all cells that will sink due to the current sea level (`height[i][j] <= year`). These cells are marked as visited.
3. **BFS to Sink Cells**: We use BFS to mark all connected cells that will sink during the current year.
4. **Count Remaining Area**: After processing all sinking cells for the current year, we count the remaining cells that are still above water.
5. **Reset Visited Array**: We reset the visited array for the next year.
6. **Output**: We print the remaining area for each year.

This approach ensures that we correctly simulate the rising sea levels and calculate the remaining area above water efficiently.
