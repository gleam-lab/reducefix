#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

#define maxn 1001
#define INF 0x3f3f3f3f

int h, w, y;
int m[maxn][maxn];
int s[maxn][maxn]; // s[i][j] will store the minimum sea level at which the cell (i, j) sinks
int visited[maxn][maxn];
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

void bfs(int r, int c, int level) {
    queue<pair<int, int>> q;
    q.push({r, c});
    visited[r][c] = 1;
    s[r][c] = level;

    while (!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();

        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k];
            int ny = y + dy[k];
            if (nx < 0 || nx >= h || ny < 0 || ny >= w) continue;
            if (visited[nx][ny] || m[x][y] >= level) continue;

            visited[nx][ny] = 1;
            q.push({nx, ny});
            s[nx][ny] = level;
        }
    }
}

        s[0][i] = s[h - 1][i] = m[0][i] = m[h - 1][i];
    }

    // Initialize the BFS queue with the border cells
    queue<pair<int, int>> q;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {

            }
        }
    }

    // Perform BFS
    while (!q.empty()) {
        int r = q.front().first;
        int c = q.front().second;
        q.pop();

        for (int k = 0; k < 4; ++k) {
            int nx = r + dx[k];
            int ny = c + dy[k];
            if (nx < 0 || nx >= h || ny < 0 || ny >= w) continue;
            if (s[r][c] <= m[nx][ny]) {
                q.push({nx, ny});
                s[r][c] = min(s[r][c], m[nx][ny]);
            }
        }
    }

    // Count the cells that remain above sea level for each year
    vector<int> ans(y + 1, 0);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            ans[min(s[i][j], y)]++;
        }
    }

    // Calculate the remaining area for each year
    for (int i = 1; i <= y; ++i) {
        cout << (h * w - ans[i]) << '\n';
