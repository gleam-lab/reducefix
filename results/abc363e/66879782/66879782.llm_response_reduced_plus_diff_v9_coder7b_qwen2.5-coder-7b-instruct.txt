#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> height(h, vector<int>(w));
    vector<vector<bool>> seen(h, vector<bool>(w, false));
    vector<vector<int>> q;
    
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> height[i][j];
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                q.push_back({height[i][j], i, j});
                seen[i][j] = true;
            }
        }
    }
    
    sort(q.begin(), q.end());
    
    vector<int> res(y + 1, h * w);
    
    for (auto [h_val, x, y] : q) {
        res[h_val]++;
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx, ny = y + dy;
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !seen[nx][ny] && height[nx][ny] <= h_val) {
                    q.push_back({height[nx][ny], nx, ny});
                    seen[nx][ny] = true;
                }
            }
        }
    }
    
    for (int i = y; i > 0; --i) {
        res[i] -= res[i + 1];
    }
    
    for (int i = 1; i <= y; ++i) {
        cout << res[i] << '\n';
    }
    
    return 0;
}
