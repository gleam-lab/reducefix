#include <bits/stdc++.h>

                sink[i][j] = heights[i][j];

            }

        if (visited[i][j]) continue;
        visited[i][j] = true;

            if (heights[ni][nj] <= height && sink[ni][nj] > height) {
                sink[ni][nj] = height;
                pq.push_back({height, {ni, nj}});
            }
        }

    for (int i = 0; i < h; ++i)
        for (int j = 0; j < w; ++j)

    partial_sum(areas.begin(), areas.end(), areas.begin());

    for (int i = 1; i <= y; ++i)
