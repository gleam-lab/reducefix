#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> height(h, vector<int>(w));
    vector<vector<bool>> visited(h, vector<bool>(w, false));
    vector<vector<int>> sink(h, vector<int>(w, INT_MAX));
    queue<pair<int, pair<int, int>>> q;
    int di[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> height[i][j];
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                sink[i][j] = height[i][j];
                q.push({height[i][j], {i, j}});
            }
        }
    }

    while (!q.empty()) {
        auto [H, p] = q.front();
        auto [r, c] = p;
        q.pop();
        if (visited[r][c]) continue;
        visited[r][c] = true;
        for (int i = 0; i < 4; i++) {
            int nr = r + di[i][0];
            int nc = c + di[i][1];
            if (nr < 0 || nr >= h || nc < 0 || nc >= w) continue;
            if (height[nr][nc] <= H && sink[nr][nc] > H) {
                sink[nr][nc] = H;
                q.push({H, {nr, nc}});
            }
        }
    }

    vector<int> pref(y + 1, 0);
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (sink[i][j] <= y) {
                pref[sink[i][j]]++;
            }
        }
    }

    for (int i = 1; i <= y; i++) {
        pref[i] += pref[i - 1];
        cout << h * w - pref[i] << '\n';
    }

    return 0;
}
