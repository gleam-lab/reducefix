#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1005;
int h, w, y;
int A[MAXN][MAXN];

void bfs(vector<vector<bool>>& visited, vector<vector<int>>& grid, int sea_level) {
    queue<pair<int, int>> q;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (!visited[i][j] && grid[i][j] > sea_level) {
                q.push({i, j});
                visited[i][j] = true;
            }
        }
    }
    
    int dr[] = {0, 0, 1, -1};
    int dc[] = {1, -1, 0, 0};
    while (!q.empty()) {
        auto [r, c] = q.front(); q.pop();
        for (int k = 0; k < 4; ++k) {
            int nr = r + dr[k], nc = c + dc[k];
            if (nr >= 0 && nr < h && nc >= 0 && nc < w && !visited[nr][nc] && grid[nr][nc] > sea_level) {
                q.push({nr, nc});
                visited[nr][nc] = true;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> h >> w >> y;
    for (int i = 0; i < h; ++i)
        for (int j = 0; j < w; ++j)
            cin >> A[i][j];

    vector<long long> areas(y + 1, 0);
    for (int s = 1; s <= y; ++s) {
        vector<vector<bool>> visited(h, vector<bool>(w, false));
        bfs(visited, A, s);
        long long count = 0;
        for (int i = 0; i < h; ++i)
            for (int j = 0; j < w; ++j)
                if (!visited[i][j])
                    ++count;
        areas[s] = count;
    }

    for (int s = 1; s <= y; ++s)
        cout << h * w - areas[s] << '\n';

    return 0;
}
