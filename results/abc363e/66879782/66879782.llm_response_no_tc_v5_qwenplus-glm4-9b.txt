#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int h, w, y;
    cin >> h >> w >> y;
    int height[h][w];
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> height[i][j];
        }
    }

    // Initialize the islands that will be submerged at sea level
    vector<vector<int>> island(h, vector<int>(w));
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            bool isEdge = (i == 0 || i == h - 1 || j == 0 || j == w - 1);
            bool isLowest = true;
            for (int di = -1; di <= 1; ++di) {
                for (int dj = -1; dj <= 1; ++dj) {
                    int ni = i + di, nj = j + dj;
                    if (ni < 0 || ni >= h || nj < 0 || nj >= w || di == 0 && dj == 0) continue;
                    if (height[ni][nj] <= height[i][j]) {
                        isLowest = false;
                        break;
                    }
                }
                if (!isLowest) break;
            }
            if (isEdge || isLowest) {
                island[i][j] = 1;
            }
        }
    }

    // Use BFS to propagate the sea level
    queue<pair<int, int>> q;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (island[i][j] == 1) {
                island[i][j] = 0; // mark as visited/sunk
                q.push({i, j}); // add to queue
            }
        }
    }
    while (!q.empty()) {
        auto [i, j] = q.front();
        q.pop();
        for (int di = -1; di <= 1; ++di) {
            for (int dj = -1; dj <= 1; ++dj) {
                int ni = i + di, nj = j + dj;
                if (ni < 0 || ni >= h || nj < 0 || nj >= w || di == 0 && dj == 0) continue;
                if (island[ni][nj] == 1) {
                    island[ni][nj] = 0;
                    q.push({ni, nj});
                }
            }
        }
    }

    // Calculate the area above sea level after Y years
    ll remainingArea = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (height[i][j] > y) {
                remainingArea++;
            }
        }
    }

    // Output the results for each year
    for (int y = 0; y < Y; ++y) {
        cout << remainingArea << '\n';
    }

    return 0;
}
