#include <iostream>
#include <vector>
#include <queue>
using namespace std;

const int INF = 1000000000; // represents the initial sink height if it's never set by the sea level

int main() {
    int h, w, y;
    cin >> h >> w >> y;

    vector<vector<int>> elev(h, vector<int>(w));
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> elev[i][j];
        }
    }

    // Create a 2D vector to keep track of the first year a cell will sink
    vector<vector<int>> sink_year(h, vector<int>(w, y + 1)); // Initialize to y+1 which is out of bounds

    // Find the cells that will sink first by checking the borders
    vector<pair<int, int>> borders; // To store the sink points and their heights
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if ((i == 0 || i == h - 1) || (j == 0 || j == w - 1)) {
                borders.emplace_back(elev[i][j], i * w + j);
            }
        }
    }

    // Sort borders to process from lowest to highest elevation point
    sort(borders.begin(), borders.end());

    // Process each border point and propagate the sinking
    for (const auto& [height, index] : borders) {
        int r = index / w;
        int c = index % w;

        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                int nr = r + i;
                int nc = c + j;
                if (nr >= 0 && nr < h && nc >= 0 && nc < w && elev[nr][nc] > height) {
                    sink_year[nr][nc] = min(sink_year[nr][nc], sink_year[r][c] + 1);
                }
            }
        }
    }

    // Compute the initial area above sea level
    long long above_sea_level_area = 0;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (sink_year[i][j] > y) {
                above_sea_level_area++;
            }
        }
    }

    // Output the area above sea level for each year
    for (int i = 0; i < y; ++i) {
        cout << above_sea_level_area << '\n';
    }

    return 0;
}
