```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MAX_N = 1000;

int h, w, y;
int height[MAX_N][MAX_N];
int flooded[MAX_N][MAX_N];

struct Point {
    int x, y;
    Point(int x, int y) : x(x), y(y) {}
};

vector<Point> directions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

void simulateFlood(int year) {
    vector<Point> toVisit;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (height[i][j] <= year) {
                flooded[i][j] = 1; // This cell will be flooded after the year
            } else {
                for (auto dir : directions) {
                    int ni = i + dir[0], nj = j + dir[1];
                    if (ni >= 0 && ni < h && nj >= 0 && nj < w && height[ni][nj] <= year && !flooded[ni][nj]) {
                        toVisit.push_back(Point(ni, nj));
                    }
                }
            }
        }
    }
    
    while (!toVisit.empty()) {
        Point current = toVisit.front();
        toVisit.erase(toVisit.begin());
        
        for (auto dir : directions) {
            int ni = current.x + dir[0], nj = current.y + dir[1];
            if (ni >= 0 && ni < h && nj >= 0 && nj < w && !flooded[ni][nj] && height[ni][nj] <= year) {
                flooded[ni][nj] = 1;
                toVisit.push_back(Point(ni, nj));
            }
        }
    }
}

int calculateUnfloodedArea(int year) {
    int count = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (!flooded[i][j] && height[i][j] > year) {
                count++;
            }
        }
    }
    return count;
}

int main() {
    cin >> h >> w >> y;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> height[i][j];
        }
    }
    
    // Initialize flooded array
    memset(flooded, 0, sizeof(flooded));
    
    for (int year = 1; year <= y; year++) {
        simulateFlood(year);
        cout << calculateUnfloodedArea(year) << '\n';
    }

    return 0;
}
```

This C++ code initializes the `height` array with the given elevation values and the `flooded` array to keep track of which cells will be flooded. The `simulateFlood` function implements the flooding process for a given year, marking cells that will be flooded as well as spreading the flood to adjacent cells. The `calculateUnfloodedArea` function calculates the number of cells that remain unflooded after a given year. The main function reads the input, initializes the arrays, and iterates over each year, printing the number of unflooded cells for each year.
