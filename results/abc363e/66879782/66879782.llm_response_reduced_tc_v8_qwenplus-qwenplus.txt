#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(), (x).end()

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> height(h, vector<int>(w));
    vector<vector<bool>> sink(h, vector<bool>(w, false));
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> height[i][j];
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                pq.emplace(height[i][j], i, j);
                sink[i][j] = true;
            }
        }
    }

    // Directions: up, down, left, right
    vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    while (!pq.empty()) {
        auto [H, r, c] = pq.top();
        pq.pop();

        for (auto [dr, dc] : dirs) {
            int nr = r + dr;
            int nc = c + dc;
            if (nr >= 0 && nr < h && nc >= 0 && nc < w && !sink[nr][nc]) {
                if (height[nr][nc] <= H) {
                    sink[nr][nc] = true;
                    pq.emplace(H, nr, nc);
                }
            }
        }
    }

    // Count remaining area each year
    vector<int> count(y + 2, 0); // 1-based index
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (!sink[i][j]) {
                if (height[i][j] > y)
                    count[y]++;
                else
                    count[height[i][j]]++;
            }
        }
    }

    // Prefix sum to get cumulative area sunk by year i
    vector<int> result(y + 1);
    int total_area = h * w;
    int cum_sunk = 0;
    for (int i = 1; i <= y; ++i) {
        cum_sunk += count[i];
        result[i] = total_area - cum_sunk;
    }

    for (int i = 1; i <= y; ++i) {
        cout << result[i] << '\n';
    }

    return 0;
}
