```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> heights(h, vector<int>(w));
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> heights[i][j];
        }
    }

    vector<vector<bool>> visited(h, vector<bool>(w, false));
    vector<vector<int>> sink(h, vector<int>(w, INT_MAX));

    queue<pair<int, pair<int, int>>> q;
    vector<pair<int, int>> directions{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

    // Initialize the boundary cells
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                sink[i][j] = heights[i][j];
                q.push({heights[i][j], {i, j}});
            }
        }
    }

    while (!q.empty()) {
        auto [height, pos] = q.front();
        q.pop();

        int r = pos.first, c = pos.second;
        visited[r][c] = true;

        for (const auto& dir : directions) {
            int nr = r + dir.first, nc = c + dir.second;
            if (nr < 0 || nr >= h || nc < 0 || nc >= w || visited[nr][nc]) continue;
            if (heights[nr][nc] <= height) {
                sink[nr][nc] = min(height, sink[nr][nc]);
                q.push({sink[nr][nc], {nr, nc}});
            }
        }
    }

    vector<int> prefix(y + 1, 0);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (sink[i][j] <= y) {
                prefix[sink[i][j]]++;
            }
        }
    }

    for (int i = 1; i <= y; ++i) {
        prefix[i] += prefix[i - 1];
        cout << h * w - prefix[i] << '\n';
    }

    return 0;
}
```
