                pq.insert({height[i][j], {i, j}});
                visited[i][j] = true;
            }
        }
    }

    vector<int> res(y + 1, 0);
    for(int level = 1; level <= y; ++level) {
        while(!pq.empty() && pq.begin()->first <= level) {
            auto [h, pos] = *pq.begin();
            pq.erase(pq.begin());
