
The problem involves simulating the process of the sea level rising on an island represented by a grid. Each cell in the grid has a certain elevation, and the grid's borders are considered to be sea level. As the sea level rises each year, some cells will sink below the water level and make the island smaller. We are asked to calculate the area of the island that remains above sea level for each year up to the given year \( Y \).

Below is a detailed approach to solve the problem:

1. **Initialization**:
    - Read the dimensions \( H \) and \( W \) of the island, and the number of years \( Y \) for which we need to find the area above the sea level.
    - Read the elevation grid \( A \).

2. **Mapping the Sea and Border**:
    - Create a grid \( S \) to mark the sea and the border of the island.
    - For marine mammals, set sea level at 0 and the borders at -1. All other cells are initialized to -2 since they are not yet marked as sea or land.

3. **Flood Fill for Sea-Level Marking**:
    - Perform a flood fill (similar to the BFS algorithm) starting from all 0s (sea level) and mark the entire area connected to them as sea level in \( S \).
    - If a cell during the flood fill is not marked as sea or border, mark it as border (-1).

4. **Simulating the Sink**:
    - For each year, calculate how many cells will sink by comparing their elevation with the sea level.
    - For each cell that will sink, update its elevation to sea level and perform a flood fill from it to mark the connected cells that will sink in that year.

5. **Calculating the Remaining Area**:
    - After marking the sinking cells for each year, calculate the area of the island that remains above sea level by subtracting the number of sinking cells from the total area of the island.

6. **Output**:
    - Print the remaining area for each year.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void flood_fill(vector<vector<int>>& grid, int r, int c, int h, int w) {
    if (r < 0 || r >= h || c < 0 || c >= w) return;
    if (grid[r][c] == 0) return; // already visited
    grid[r][c] = 0; // mark as sunk
    flood_fill(grid, r - 1, c, h, w);
    flood_fill(grid, r + 1, c, h, w);
    flood_fill(grid, r, c - 1, h, w);
    flood_fill(grid, r, c + 1, h, w);
}

int main() {
    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> A(h, vector<int>(w));
    vector<vector<int>> S(h, vector<int>(w, -2)); // sea or border
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> A[i][j];
            if (i == 0 || j == 0 || i == h - 1 || j == w - 1 || A[i][j] == 0)
                S[i][j] = 0; // sea
        }
    }

    vector<int> years(y + 1, 0);
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (S[i][j] == 0) continue; // sea
            if (A[i][j] > 0) {
                if (A[i][j] <= y) {
                    flood_fill(S, i, j, h, w);
                    years[A[i][j]]++;
                }
            }
        }
    }

    for (int i = 1; i <= y; ++i) {
        years[i] += years[i - 1];
        cout << h * w - years[i] << endl;
    }

    return 0;
}
