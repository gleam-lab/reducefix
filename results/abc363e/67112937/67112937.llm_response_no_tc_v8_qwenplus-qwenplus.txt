```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Priority queue to process cells in increasing order of elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the border cells (sea level at year 0)
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    int totalCells = H * W;
    int remaining = totalCells;

    // To store results for each year
    vector<int> result(Y + 1, 0);
    for (int y = 1; y <= Y; ++y) {
        result[y] = remaining;
    }

    while (!pq.empty()) {
        auto [height, x, y] = pq.top();
        pq.pop();

        // If height is greater than current year, skip
        if (height > Y) continue;

        // Subtract only if this cell was counted as remaining
        if (result[height] == remaining) {
            remaining--;
            result[height]--;
        }

        for (int dir = 0; dir < 4; ++dir) {
            int nx = x + d[dir][0];
            int ny = y + d[dir][1];

            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }
    }

    // Propagate remaining area across years
    for (int i = Y; i >= 1; --i) {
        if (result[i] < result[i - 1])
            result[i - 1] = result[i];
    }

    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << "\n";
    }

    return 0;
}
```
