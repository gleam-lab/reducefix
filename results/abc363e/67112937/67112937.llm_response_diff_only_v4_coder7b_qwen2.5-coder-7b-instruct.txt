```cpp
// Copyright (C) 2025 hanziwei

#include <bits/stdc++.h>

using u32 = unsigned;
using i64 = long long;
using u64 = unsigned long long;

using i128 = __int128;
using u128 = unsigned __int128;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int32_t main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int H, W, Y;
    std::cin >> H >> W >> Y;

    std::vector<std::vector<int>> A(H, std::vector<int>(W));
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            std::cin >> A[i][j];
        }
    }

    std::vector<std::vector<bool>> visited(H, std::vector<bool>(W, false));

    auto isValid = [&](int x, int y) {
        return x >= 0 && x < H && y >= 0 && y < W && !visited[x][y] && A[x][y] > 0;
    };

    auto bfs = [&](int seaLevel) {
        int count = 0;
        std::queue<std::pair<int, int>> q;
        for (int i = 0; i < H; ++i) {
            for (int j = 0; j < W; ++j) {
                if (!visited[i][j] && A[i][j] > seaLevel) {
                    visited[i][j] = true;
                    q.emplace(i, j);
                    ++count;
                    while (!q.empty()) {
                        auto [x, y] = q.front();
                        q.pop();
                        for (const auto& dir : d) {
                            int nx = x + dir[0], ny = y + dir[1];
                            if (isValid(nx, ny)) {
                                visited[nx][ny] = true;
                                q.emplace(nx, ny);
                                ++count;
                            }
                        }
                    }
                }
            }
        }
        return count;
    };

    for (int i = 0; i <= Y; ++i) {
        std::cout << bfs(i) << '\n';
    }

    return 0;
}
```

This solution uses BFS to efficiently calculate the number of land cells remaining above a certain sea level. The `isValid` function checks if a cell is within bounds, not yet visited, and its elevation is greater than the current sea level. The `bfs` function iterates over all cells, starting a new BFS from unvisited land cells above the current sea level, counting the total number of such cells. This approach ensures that we correctly account for the entire island's state after each year's rise in sea level.
