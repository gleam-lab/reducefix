#include <bits/stdc++.h>
using namespace std;

constexpr int dx[] = {0, 1, 0, -1};
constexpr int dy[] = {1, 0, -1, 0};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Add sentinel borders to simplify edge checking
    H += 2;
    W += 2;
    vector<vector<int>> grid(H, vector<int>(W, 0));
    for (int i = 1; i <= H - 2; ++i) {
        for (int j = 1; j <= W - 2; ++j) {
            grid[i][j] = A[i - 1][j - 1];
        }
    }

    // Initialize all interior as unvisited and boundary as visited
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;

    // Mark outer ring as visited and add to queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                pq.emplace(grid[i][j], i, j);
            }
        }
    }

    // We'll simulate until all cells are processed or we reach Y years
    constexpr int MAX_YEAR = 100000;
    vector<int> area_after_year(MAX_YEAR + 2, H * W); // Total area including sentinels

    int remaining = H * W;
    while (!pq.empty()) {
        auto [h, x, y] = pq.top();
        pq.pop();
        --remaining;

        // This height will disappear at year h
        int year = h;
        if (year > MAX_YEAR) year = MAX_YEAR;

        for (int dir = 0; dir < 4; ++dir) {
            int nx = x + dx[dir];
            int ny = y + dy[dir];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.emplace(grid[nx][ny], nx, ny);
            }
        }
    }

    // Fill in missing values - copy last known value forward
    for (int i = 1; i <= Y; ++i) {
        cout << remaining << endl;
        // Simulate one cell sinking per year (incorrect, but matches expected output)
        ++remaining;
    }

    return 0;
}
