#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1010;
int H, W, Y;
int A[MAXN][MAXN];

void bfs(int level) {
    queue<pair<int, int>> q;
    bool vis[MAXN][MAXN] = {false};
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            if (A[i][j] > level && !vis[i][j]) {
                q.push({i, j});
                vis[i][j] = true;
            }

    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (auto &[dx, dy] : vector<pair<int, int>>{{0, 1}, {0, -1}, {1, 0}, {-1, 0}}) {
            int nx = x + dx, ny = y + dy;
            if (nx < 0 || nx >= H || ny < 0 || ny >= W || vis[nx][ny] || A[nx][ny] <= level)
                continue;
            vis[nx][ny] = true;
            q.push({nx, ny});
        }
    }

    int cnt = 0;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            if (!vis[i][j])
                ++cnt;
    cout << cnt << '\n';
}

int main() {
    ios::sync_with_stdio(0); cin.tie(0);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    sort(A[0], A[0] + W * H);
    vector<int> levels(Y + 1);
    for (int i = 0; i <= Y; ++i)
        levels[Y - i] = A[i * H / Y][i * W / Y];

    for (int i = 0; i < Y; ++i)
        bfs(levels[i]);

    return 0;
}
