To solve this problem efficiently, we need to simulate the rising sea levels and determine how much of the island remains above water each year. The key steps involve:

1. Reading the initial height map of the island.
2. Using a priority queue to manage the sections of the island that will be affected first by the rising sea level.
3. Iteratively sinking sections of the island as the sea level rises and updating the priority queue accordingly.
4. Calculating the remaining area above sea level after each year.

Here's the corrected and optimized C++ code:

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Cell {
    int h, x, y;
    bool operator<(const Cell& other) const {
        return h > other.h;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> vis(H, vector<bool>(W, false));
    vector<vector<int>> dirs{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    priority_queue<Cell> pq;

    // Initialize the priority queue with the boundary cells
    for (int j = 0; j < W; ++j) {
        pq.emplace(A[0][j], 0, j);
        pq.emplace(A[H-1][j], H-1, j);
        vis[0][j] = true;
        vis[H-1][j] = true;
    }
    for (int i = 1; i < H-1; ++i) {
        pq.emplace(A[i][0], i, 0);
        pq.emplace(A[i][W-1], i, W-1);
        vis[i][0] = true;
        vis[i][W-1] = true;
    }

    while (!pq.empty()) {
        auto [h, x, y] = pq.top(); pq.pop();

        if (h <= 0) continue;

        for (auto [dx, dy] : dirs) {
            int nx = x + dx, ny = y + dy;
            if (nx < 0 || nx >= H || ny < 0 || ny >= W || vis[nx][ny]) continue;
            vis[nx][ny] = true;
            pq.emplace(max(0, h - 1), nx, ny);
        }
    }

    int ans = H * W;
    for (int i = 0; i < Y; ++i) {
        ans -= pq.size();
        cout << ans << '\n';
    }

    return 0;
}
```

### Explanation:
1. **Initialization**: We initialize a priority queue with all boundary cells since they will be the first to be affected by rising sea levels.
2. **Sinking Process**: As the sea level rises, we sink sections of the island and update the priority queue with new sections that become exposed.
3. **Output Calculation**: For each year, we calculate the remaining area above sea level by subtracting the number of remaining sections from the total area.

This approach ensures that we efficiently handle the rising sea levels and accurately compute the remaining island area.
