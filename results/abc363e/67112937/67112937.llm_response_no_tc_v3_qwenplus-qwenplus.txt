#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Create a priority queue to simulate the rising sea level flooding process
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the border cells as initially flooded (connected to the sea)
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    // We will record for each year how many cells are destroyed by that year
    vector<int> destroyed_by(Y + 2, 0); // 1-based index up to Y
    int remaining = H * W;

    while (!pq.empty()) {
        auto [h, x, y] = pq.top();
        pq.pop();

        // This cell gets submerged at sea level h -> at year h
        if (h <= Y) {
            destroyed_by[h]++;
        }

        for (auto [dx, dy] : array<array<int, 2>, 4>{{{0,1},{0,-1},{1,0},{-1,0}}}) {
            int nx = x + dx[0], ny = y + dx[1];
            if (nx < 0 || nx >= H || ny < 0 || ny >= W || visited[nx][ny]) {
                continue;
            }
            visited[nx][ny] = true;
            pq.emplace(A[nx][ny], nx, ny);
        }
    }

    // For each year, accumulate the destroyed cells and subtract from total
    for (int year = 1; year <= Y; ++year) {
        destroyed_by[year] += destroyed_by[year - 1];
        cout << (remaining - destroyed_by[year]) << '\n';
    }

    return 0;
}
