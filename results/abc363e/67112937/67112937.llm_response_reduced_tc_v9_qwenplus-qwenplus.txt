#include <bits/stdc++.h>

using i64 = long long;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int H, W, Y;
    std::cin >> H >> W >> Y;

    std::vector<std::vector<int>> A(H, std::vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            std::cin >> A[i][j];
        }
    }

    // Initialize boundary cells as visited and add them to the priority queue
    std::priority_queue<std::tuple<int, int, int>, std::vector<std::tuple<int, int, int>>, std::greater<>> pq;
    std::vector<std::vector<bool>> vis(H, std::vector<bool>(W, false));

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                vis[i][j] = true;
                pq.emplace(A[i][j], i, j);
            }
        }
    }

    int remaining = H * W;
    std::vector<int> ans(Y);

    for (int year = 0; year < Y; ++year) {
        int current_sea_level = year + 1;

        // Process all cells in the priority queue whose elevation is <= current sea level
        while (!pq.empty()) {
            auto [height, x, y] = pq.top();
            if (height > current_sea_level) break;

            pq.pop();
            --remaining;

            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];

                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.emplace(A[nx][ny], nx, ny);
                }
            }
        }

        ans[year] = remaining;
    }

    for (int a : ans) {
        std::cout << a << '\n';
    }

    return 0;
}
