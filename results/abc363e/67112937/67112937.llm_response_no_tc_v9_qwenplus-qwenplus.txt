#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Priority queue to simulate water rising using Dijkstra-like approach
    priority_queue<array<int, 3>, vector<array<int, 3>>, greater<array<int, 3>>> pq;
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Initialize the border cells as visited and add them to the queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.push({A[i][j], i, j});
                vis[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    vector<int> result(Y + 1, 0);  // result[i] stores area after i years

    // We will process up to maximum possible elevation or Y years
    int year = 0;
    while (!pq.empty() && year <= Y) {
        // Handle all cells that flood at current year
        vector<array<int, 3>> batch;

        while (!pq.empty() && pq.top()[0] == year) {
            batch.push_back(pq.top());
            pq.pop();
        }

        for (auto [h, x, y] : batch) {
            --remaining;
            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.push({A[nx][ny], nx, ny});
                }
            }
        }

        result[year] = remaining;
        ++year;
    }

    // In case we haven't processed all years, fill in the remaining with last computed value
    while (year <= Y) {
        result[year++] = remaining;
    }

    // Output results
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << "\n";
    }

    return 0;
}
