#include <bits/stdc++.h>

using namespace std;

const int N = 1005;
int h[N], w[N], y;
bool vis[N][N];

void bfs(int x, int y, int sea_level) {
    queue<pair<int, int>> q;
    q.emplace(x, y);
    vis[x][y] = true;
    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        for (auto &[dx, dy] : vector<pair<int, int>>{{0, 1}, {0, -1}, {1, 0}, {-1, 0}}) {
            int nx = x + dx, ny = y + dy;
            if (nx < 0 || nx >= h[y] || ny < 0 || ny >= w[y]) continue;
            if (vis[nx][ny] || A[nx][ny] <= sea_level) continue;
            vis[nx][ny] = true;
            q.emplace(nx, ny);
        }
    }
}

int main() {
    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    cin >> h[0] >> w[0] >> y;
    vector<vector<int>> A(h[0], vector<int>(w[0]));
    for (int i = 0; i < h[0]; ++i)
        for (int j = 0; j < w[0]; ++j)
            cin >> A[i][j];

    for (int i = 0; i <= y; ++i) {
        memset(vis, false, sizeof(vis));
        for (int j = 0; j < h[0]; ++j)
            for (int k = 0; k < w[0]; ++k)
                if (A[j][k] > i && !vis[j][k])
                    bfs(j, k, i);
        cout << h[0] * w[0] - count(vis[0], vis[h[0]], false) << '\n';
    }
    return 0;
}
