#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    priority_queue<pair<int, pair<int, int>>, vector<pair<int, pair<int, int>>>, greater<>> pq;
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Initialize the queue with all boundary cells
    for (int i = 0; i < H; ++i) {
        if (!vis[i][0]) pq.emplace(A[i][0], make_pair(i, 0)), vis[i][0] = true;
        if (!vis[i][W-1]) pq.emplace(A[i][W-1], make_pair(i, W-1)), vis[i][W-1] = true;
    }
    for (int j = 1; j < W-1; ++j) {
        if (!vis[0][j]) pq.emplace(A[0][j], make_pair(0, j)), vis[0][j] = true;
        if (!vis[H-1][j]) pq.emplace(A[H-1][j], make_pair(H-1, j)), vis[H-1][j] = true;
    }

    int ans = H * W;
    while (Y--) {
        while (!pq.empty() && pq.top().first <= Y) {
            auto [height, pos] = pq.top();
            pq.pop();
            --ans;
            int x = pos.first, y = pos.second;
            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0], ny = y + d[k][1];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.emplace(max(height, A[nx][ny]), make_pair(nx, ny));
                }
            }
        }
        cout << ans << '\n';
    }

    return 0;
}
