#include <bits/stdc++.h>

using i64 = long long;
using u32 = unsigned int;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int H, W, Y;
    std::cin >> H >> W >> Y;

    std::vector<std::vector<int>> A(H, std::vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            std::cin >> A[i][j];
        }
    }

    // Create a priority queue that stores (elevation, x, y)
    std::priority_queue<std::tuple<int, int, int>, std::vector<std::tuple<int, int, int>>, std::greater<>> pq;
    std::vector<std::vector<bool>> visited(H, std::vector<bool>(W, false));

    // Initialize the perimeter of the grid
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    std::vector<int> result(Y + 2);  // Store results to avoid printing during processing

    for (int year = 1; !pq.empty() && year <= Y; ++year) {
        while (!pq.empty() && std::get<0>(pq.top()) <= year) {
            auto [h, x, y] = pq.top();
            pq.pop();

            if (!visited[x][y]) {
                --remaining;
                visited[x][y] = true;
            }

            for (int k = 0; k < 4; ++k) {
                int nx = x + d[k][0];
                int ny = y + d[k][1];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    pq.emplace(A[nx][ny], nx, ny);
                    visited[nx][ny] = true;
                }
            }
        }
        result[year] = remaining;
    }

    // Fill in missing years if necessary
    for (int year = 1; year <= Y; ++year) {
        if (!pq.empty()) {
            result[year] = remaining;
        } else {
            break;
        }
    }

    for (int year = 1; year <= Y; ++year) {
        std::cout << result[year] << "\n";
    }

    return 0;
}
