```cpp
#include <bits/stdc++.h>
using namespace std;

constexpr int dx[] = {0, 1, 0, -1};
constexpr int dy[] = {1, 0, -1, 0};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Initialize visited matrix
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Priority queue to process cells by elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;

    // Push all boundary cells into the queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                vis[i][j] = true;
                pq.emplace(A[i][j], i, j);
            }
        }
    }

    // Total land area
    int remaining = H * W;

    // Answer array for each year
    vector<int> ans(Y);

    // Process each year
    for (int year = 1; year <= Y; ++year) {
        // Process all cells in the priority queue with elevation <= current year
        while (!pq.empty()) {
            auto [elevation, x, y] = pq.top();
            if (elevation > year) break;
            pq.pop();

            // If already processed, skip
            if (!vis[x][y]) {
                remaining--;
                vis[x][y] = true;
            }

            // Explore neighbors
            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    pq.emplace(A[nx][ny], nx, ny);
                }
            }
        }

        ans[year - 1] = remaining;
    }

    // Print answers
    for (int a : ans) {
        cout << a << '\n';
    }

    return 0;
}
```
