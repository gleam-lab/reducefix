```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // We will process cells in order of increasing height using a min-heap (priority queue)
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the perimeter of the island
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    // To store the number of land cells remaining at each year
    vector<int> result(Y + 1, 0);
    int totalLand = H * W;
    int sunkCount = 0;

    while (!pq.empty()) {
        auto [height, x, y] = pq.top();
        pq.pop();

        // If this cell is processed after the sea level rise of `year`
        if (height > Y) continue;

        // This cell sinks at `height` years
        sunkCount++;

        for (int k = 0; k < 4; ++k) {
            int nx = x + d[k][0];
            int ny = y + d[k][1];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }

        // Update all years from current height up to next height or Y
        int nextHeight = pq.empty() ? Y + 1 : get<0>(pq.top());
        int endYear = min(Y, nextHeight - 1);
        for (int year = height; year <= endYear; ++year) {
            result[year] = totalLand - sunkCount;
        }

        if (nextHeight > Y) break;
    }

    // For any years not covered yet, they have same result as last computed
    for (int year = 1; year <= Y; ++year) {
        if (result[year] == 0) {
            result[year] = totalLand - sunkCount;
        }
        cout << result[year] << '\n';
    }

    return 0;
}
```
