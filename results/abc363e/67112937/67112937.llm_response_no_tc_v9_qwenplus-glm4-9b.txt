#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Point {
    int x, y, height;
    bool operator>(const Point& other) const {
        return height > other.height;
    }
};

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    
    vector<vector<int>> grid(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> grid[i][j];
        }
    }
    
    priority_queue<Point, vector<Point>, greater<Point>> pq;
    
    // Initialize the priority queue with boundaries
    for (int j = 0; j < W; ++j) {
        if (j == 0) pq.push({0, j, grid[0][j]}); // left boundary
        if (j == W - 1) pq.push({0, j, grid[0][j]}); // right boundary
    }
    for (int i = 0; i < H; ++i) {
        if (i == 0) pq.push({i, 0, grid[i][0]}); // top boundary
        if (i == H - 1) pq.push({i, 0, grid[i][0]}); // bottom boundary
    }
    
    vector<int> areas(Y);
    areas[0] = H * W; // Initial area is the whole grid
    
    bool firstYear = true;
    while (!pq.empty()) {
        Point p = pq.top();
        pq.pop();
        
        if (p.height > Y || (firstYear && p.height <= Y)) {
            break;
        }
        
        if (!firstYear && p.height <= Y) {
            areas[Y - 1 - (p.height - Y)] -= 1;
        }
        
        // Check adjacent cells
        int x = p.x, y = p.y;
        for (int dx = -1; dx <= 1; ++dx) {
            int nx = x + dx;
            if (nx < 0 || nx >= H) continue;
            for (int dy = -1; dy <= 1; ++dy) {
                int ny = y + dy;
                if (ny < 0 || ny >= W) continue;
                if (grid[nx][ny] > p.height) {
                    pq.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        
        if (!firstYear && p.height <= Y) {
            areas[Y - 1 - (p.height - Y)] += 1;
        }
    }
    
    for (int area : areas) {
        cout << area << endl;
    }
    
    return 0;
}
