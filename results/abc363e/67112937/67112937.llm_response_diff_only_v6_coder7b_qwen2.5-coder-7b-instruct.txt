#include <vector>
#include <queue>
#include <algorithm>

const int MAXN = 1005;
int H, W, Y;

struct Node {

        return h > other.h;
    }
};

void bfs(int sea_level) {
    priority_queue<Node> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {

                pq.push({A[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }

}
