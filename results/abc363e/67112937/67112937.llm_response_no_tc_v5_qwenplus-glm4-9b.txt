#include <iostream>
#include <vector>
#include <queue>

using namespace std;

struct Cell {
    int x, y, height;
};

struct Compare {
    bool operator()(const Cell& a, const Cell& b) {
        return a.height < b.height;
    }
};

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> A[i][j];
        }
    }

    priority_queue<Cell, vector<Cell>, Compare> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the priority queue with the edges
    for (int i = 0; i < W; i++) {
        pq.push({0, i, A[0][i]});
        visited[0][i] = true;
    }
    for (int i = 0; i < H; i++) {
        pq.push({i, 0, A[i][0]});
        visited[i][0] = true;
    }
    for (int i = 0; i < H; i++) {
        pq.push({i, W - 1, A[i][W - 1]});
        visited[i][W - 1] = true;
    }
    for (int i = 0; i < W; i++) {
        pq.push({H - 1, i, A[H - 1][i]});
        visited[H - 1][i] = true;
    }

    while (!pq.empty() && pq.top().height >= 0) {
        Cell current = pq.top();
        pq.pop();

        int x = current.x;
        int y = current.y;
        int height = current.height;

        // Check neighbors
        for (int i = -1; i <= 1; i++) {
            for (int j = -1; j <= 1; j++) {
                if (i == 0 && j == 0) continue;

                int nx = x + i;
                int ny = y + j;

                if (nx < 0 || ny < 0 || nx >= H || ny >= W || visited[nx][ny]) continue;

                if (A[nx][ny] <= height + 1) {
                    visited[nx][ny] = true;
                    pq.push({nx, ny, A[nx][ny]});
                }
            }
        }
    }

    // Count the number of visited cells that are still above the sea level
    int ans = 0;
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            if (!visited[i][j]) {
                ans++;
            }
        }
    }

    for (int i = 0; i < Y; i++) {
        cout << ans << endl;
    }

    return 0;
}
