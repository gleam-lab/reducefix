#include <bits/stdc++.h>
using namespace std;

#define int long long
#define close ios_base::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);
#define MAXN 110
#define LIMIT 22
#define inf 0x3f3f3f3f3f3f
#define endl '\n'
const double eps = 1e-7;
vector<int> dx = {0, 0, 1, -1};
vector<int> dy = {1, -1, 0, 0};

struct Node {
    int x, y, val;
    bool operator<(const Node& a) const {
        return val > a.val;
    }
};

signed main() {
    close;
    int h, w, y;
    cin >> h >> w >> y;

    vector<vector<int>> grid(h, vector<int>(w));
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<bool>> vis(h, vector<bool>(w, false));
    priority_queue<Node, vector<Node>, greater<Node>> pq;

    // Initialize the queue with the edges of the grid and mark them as visited
    for (int i = 0; i < h; i++) {
        pq.push({i, 0, grid[i][0]});
        pq.push({i, w - 1, grid[i][w - 1]});
        vis[i][0] = vis[i][w - 1] = true;
    }
    for (int i = 1; i < w - 1; i++) {
        pq.push({0, i, grid[0][i]});
        pq.push({h - 1, i, grid[h - 1][i]});
        vis[0][i] = vis[h - 1][i] = true;
    }

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();

        // Process neighbors
        for (int i = 0; i < 4; i++) {
            int nx = cur.x + dx[i];
            int ny = cur.y + dy[i];
            if (nx >= 0 && nx < h && ny >= 0 && ny < w && !vis[nx][ny]) {
                vis[nx][ny] = true;
                pq.push({nx, ny, grid[nx][ny]});
            }
        }
    }

    // Calculate the initial area above the water level
    long long remainingArea = 0;
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (!vis[i][j]) {
                remainingArea++;
            }
        }
    }

    // Output the area of the island above sea level each year
    for (int i = 0; i < y; i++) {
        cout << remainingArea << endl;
    }

    return 0;
}
