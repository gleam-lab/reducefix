#include <bits/stdc++.h>

const int MAXH = 1005;
const int MAXW = 1005;

int H, W, Y;
int grid[MAXH][MAXW];
bool visited[MAXH][MAXW];

struct Cell {
    int x, y, val;
    bool operator<(const Cell& other) const {
        return val > other.val;
    }
};

signed main() {
    close;
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> grid[i][j];

    priority_queue<Cell> pq;
    
    // Mark and add all border cells to the priority queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.push({i, j, grid[i][j]});
                visited[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    int result[100005] = {0};  // To store results for each year

    // Process levels in increasing order
    while (!pq.empty()) {
        Cell cur = pq.top();
        pq.pop();

        if (cur.val <= Y) {
            result[cur.val]++;
        }

        for (int dir = 0; dir < 4; ++dir) {
            int nx = cur.x + dx[dir];
            int ny = cur.y + dy[dir];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                // The new cell's "threshold" is max of its own value and current cell's value
                // because it will be eroded when either its own level or its neighbor's erosion reaches it
                pq.push({nx, ny, max(cur.val, grid[nx][ny])});
            }
        }
    }

    // Accumulate and output
    int ans = H * W;
    for (int i = 1; i <= Y; ++i) {
        ans -= result[i];
        cout << ans << endl;
    }

    return 0;
}
