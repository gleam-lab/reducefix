#include <bits/stdc++.h>
using namespace std;
#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
const int MAXN = 1005;
int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};

struct Node {
    int x, y, val;
    bool operator<(const Node& a) const {
        return val > a.val; // Min-heap: smaller elevation comes first
    }
};

signed main() {
    close;
    int H, W, Y;
    cin >> H >> W >> Y;
    
    int grid[H][W];
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> grid[i][j];
        }
    }

    // Visited array to track which cells have been added to the priority queue
    bool vis[H][W];
    memset(vis, false, sizeof(vis));
    
    priority_queue<Node> pq;
    int remaining = H * W;

    // Add all border cells to the priority queue since they are adjacent to the sea
    for (int i = 0; i < H; i++) {
        pq.push({i, 0, grid[i][0]});
        vis[i][0] = true;
        
        if (W > 1) {
            pq.push({i, W-1, grid[i][W-1]});
            vis[i][W-1] = true;
        }
    }
    
    for (int j = 1; j < W-1; j++) {
        pq.push({0, j, grid[0][j]});
        vis[0][j] = true;
        
        if (H > 1) {
            pq.push({H-1, j, grid[H-1][j]});
            vis[H-1][j] = true;
        }
    }

    // Process each year from 1 to Y
    for (int year = 1; year <= Y; year++) {
        // Remove all cells that sink at this sea level (elevation <= year)
        while (!pq.empty() && pq.top().val <= year) {
            Node cur = pq.top();
            pq.pop();
            remaining--;

            // Check neighbors and add them to the priority queue if not visited
            for (int d = 0; d < 4; d++) {
                int nx = cur.x + dx[d];
                int ny = cur.y + dy[d];
                
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        
        cout << remaining << endl;
    }

    return 0;
}
