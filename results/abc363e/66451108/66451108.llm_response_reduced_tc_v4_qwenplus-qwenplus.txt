#include <bits/stdc++.h>

#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

const int MAXN = 1005;
const int dx[] = {0, 0, 1, -1};
const int dy[] = {1, -1, 0, 0};

struct Node {
    int x, y, val;
    bool operator<(const Node &a) const {
        return val > a.val;
    }
};

signed main() {
    close;
    int h, w; cin >> h >> w;
    int y; cin >> y;
    int grid[MAXN][MAXN];
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> grid[i][j];
        }
    }

    bool vis[MAXN][MAXN];
    memset(vis, false, sizeof(vis));

    priority_queue<Node> pq;
    
    // Push all boundary cells
    for (int i = 0; i < h; ++i) {
        if (!vis[i][0]) {
            vis[i][0] = true;
            pq.push({i, 0, grid[i][0]});
        }
        if (!vis[i][w-1]) {
            vis[i][w-1] = true;
            pq.push({i, w-1, grid[i][w-1]});
        }
    }
    for (int j = 1; j < w - 1; ++j) {
        if (!vis[0][j]) {
            vis[0][j] = true;
            pq.push({0, j, grid[0][j]});
        }
        if (!vis[h-1][j]) {
            vis[h-1][j] = true;
            pq.push({h-1, j, grid[h-1][j]});
        }
    }

    for (int i = 1; i <= y; ++i) {
