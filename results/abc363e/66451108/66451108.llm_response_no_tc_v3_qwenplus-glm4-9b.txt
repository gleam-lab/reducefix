#include <bits/stdc++.h>
using namespace std;

#define MAXN 1001
#define MOD 1000000007
int n, m, y;
int grid[MAXN][MAXN];
int dp[MAXN][MAXN][21]; // dp[i][j][level] indicates the highest level that can be reached by the sea level at (i, j)

int main() {
    cin >> n >> m >> y;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    // Initialize dp array
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            dp[i][j][0] = grid[i][j];
        }
    }

    // Fill dp table
    for (int level = 1; level <= y; level++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                // Find the minimum level that can be reached by all adjacent cells
                for (int dx = -1; dx <= 1; dx++) {
                    for (int dy = -1; dy <= 1; dy++) {
                        int ni = i + dx, nj = j + dy;
                        if (ni >= 0 && ni < n && nj >= 0 && nj < m) {
                            dp[i][j][level] = min(dp[i][j][level], dp[ni][nj][level - 1]);
                        }
                    }
                }
                // If the current cell can be submerged at this level
                if (grid[i][j] <= dp[i][j][level]) {
                    dp[i][j][level] = grid[i][j]; // The sea level can reach this cell
                } else {
                    dp[i][j][level] = -1; // The cell cannot be submerged at this level
                }
            }
        }
    }

    // Calculate the remaining area above sea level after y years
    int remainingArea = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (dp[i][j][y] == -1 || dp[i][j][y] > grid[i][j]) {
                remainingArea++;
            }
        }
    }

    // Output the result for each year
    for (int year = 1; year <= y; year++) {
        cout << n * m - remainingArea + (year == y ? 0 : 1) << endl;
    }

    return 0;
}
