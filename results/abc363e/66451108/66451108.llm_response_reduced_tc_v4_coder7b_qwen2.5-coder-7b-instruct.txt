4. **Count Remaining Area**: Calculate the area of the island that remains above water after each year.

```cpp
#include <bits/stdc++.h>
using namespace std;

    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<bool>> vis(h, vector<bool>(w, false));
    priority_queue<Node> q;
    for (int i = 0; i < h; i++) {
        q.push({i, 0, grid[i][0]});
        q.push({i, w - 1, grid[i][w - 1]});
        vis[i][0] = vis[i][w - 1] = true;
    }
    for (int i = 1; i < w - 1; i++) {
        q.push({0, i, grid[0][i]});
        q.push({h - 1, i, grid[h - 1][i]});
        vis[0][i] = vis[h - 1][i] = true;
    }

    int res = h * w;
    for (int level = 1; level <= y; level++) {
        while (!q.empty() && q.top().val <= level) {
            Node cur = q.top(); q.pop();
            res--;
            for (int i = 0; i < 4; i++) {
                int nx = cur.x + dx[i];
                int ny = cur.y + dy[i];
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    q.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        cout << res << endl;
    }

}
```
