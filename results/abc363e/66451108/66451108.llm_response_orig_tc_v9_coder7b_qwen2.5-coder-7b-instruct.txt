void bfs(int h, int w, vector<vector<int>>& grid, int level, vector<vector<bool>>& visited) {
    queue<pair<int, int>> q;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (!visited[i][j] && grid[i][j] <= level) {

                q.push({nx, ny});
                visited[nx][ny] = true;
            }
        }
    }
}
