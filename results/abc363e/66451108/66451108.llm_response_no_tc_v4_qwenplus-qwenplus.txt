```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define pii pair<int, int>
#define x first
#define y second

const int MAXH = 1005;
const int MAXW = 1005;
const int MAXY = 100005;

int H, W, Y;
int grid[MAXH][MAXW];
bool vis[MAXH][MAXW];
int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};

// We will store the results for each year to output at the end
int result[MAXY];

struct Node {
    int x, y, val;
    bool operator<(const Node& other) const {
        return val > other.val;
    }
};

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> grid[i][j];

    priority_queue<Node> pq;
    memset(vis, false, sizeof(vis));

    // Initialize the border cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H-1 || j == 0 || j == W-1) {
                if (!vis[i][j]) {
                    vis[i][j] = true;
                    pq.push({i, j, grid[i][j]});
                }
            }
        }
    }

    int remaining = H * W;

    // To handle years efficiently
    int last_level = 0;
    vector<int> updates_at[MAXY + 2];  // Store how many islands sink at each year

    while (!pq.empty()) {
        Node cur = pq.top(); pq.pop();
        int current_elevation = cur.val;
        int current_year = current_elevation;

        if (current_year > Y)
            current_year = Y + 1;  // These cells won't be submerged within Y years

        // Count this cell as submerged
        updates_at[current_year]++;

        for (int dir = 0; dir < 4; ++dir) {
            int nx = cur.x + dx[dir];
            int ny = cur.y + dy[dir];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                vis[nx][ny] = true;
                pq.push({nx, ny, grid[nx][ny]});
            }
        }
    }

    // Build result from cumulative submersion
    for (int year = 1; year <= Y; ++year) {
        remaining -= updates_at[year];
        result[year] = remaining;
    }

    for (int year = 1; year <= Y; ++year)
        cout << result[year] << "\n";

    return 0;
}
```
