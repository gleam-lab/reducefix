    // Push all border cells into the priority queue
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {

            }

    int res = h * w;
    vector<int> ans(Y + 1, res); // ans[0] is initial state (sea level 0)
