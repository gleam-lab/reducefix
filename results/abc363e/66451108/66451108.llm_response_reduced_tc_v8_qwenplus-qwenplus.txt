```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
const int MAXN = 1005;
const int MOD = 1e9 + 7;

int H, W, Y;
int grid[MAXN][MAXN];
bool visited[MAXN][MAXN];

// Direction vectors for BFS (up, down, left, right)
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

// For storing cells at a certain elevation level
vector<vector<pair<int, int>>> level_cells(MAXN * MAXN); // max elevation is 1e5

// BFS queue
queue<pair<int, int>> q;

signed main() {
    close;
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> grid[i][j];
            level_cells[grid[i][j]].push_back({i, j});
        }
    }

    // Initialize the border as visited (initial sea)
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H-1 || j == 0 || j == W-1) {
                visited[i][j] = true;
                q.push({i, j});
            }
        }
    }

    // Total area of the island
    int total_area = H * W;

    // We'll store the number of remaining land cells after each year
    vector<int> ans(Y+2, total_area);

    // Start from year 1 to Y
    for (int year = 1; year <= Y; ++year) {
        // Add all cells with elevation == year that are now connected to the sea
        for (auto &cell : level_cells[year]) {
            int x = cell.first;
            int y = cell.second;
            if (!visited[x][y]) {
                visited[x][y] = true;
                q.push(cell);
            }
        }

        // Perform BFS to flood connected land areas not higher than current sea level
        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && grid[nx][ny] <= year) {
                    visited[nx][ny] = true;
                    q.push({nx, ny});
                    ans[year]--;
                }
            }
        }

        // Carry forward the answer for this year
        if (year < Y) {
            ans[year+1] = ans[year];
        }
    }

    // Output the results
    for (int year = 1; year <= Y; ++year) {
        cout << ans[year] << "\n";
    }

    return 0;
}
```
