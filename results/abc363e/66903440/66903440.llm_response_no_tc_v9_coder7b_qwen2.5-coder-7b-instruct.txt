#include<bits/stdc++.h>
using namespace std;
using ll = long long;
const int INF = 1e9;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    vector<vector<bool>> seen(H, vector<bool>(W));
    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                seen[i][j] = true;
                q.push({i, j});
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        while (!q.empty()) {
            auto [i, j] = q.front(); q.pop();
            for (int di = -1; di <= 1; ++di) {
                for (int dj = -1; dj <= 1; ++dj) {
                    if (abs(di) + abs(dj) != 1) continue;
                    int ni = i + di, nj = j + dj;
                    if (ni >= 0 && ni < H && nj >= 0 && nj < W && !seen[ni][nj] && A[ni][nj] <= y) {
                        seen[ni][nj] = true;
                        q.push({ni, nj});
                    }
                }
            }
        }

        int remaining_area = 0;
        for (int i = 0; i < H; ++i)
            for (int j = 0; j < W; ++j)
                if (!seen[i][j])
                    ++remaining_area;

        cout << remaining_area << '\n';
    }

    return 0;
}
