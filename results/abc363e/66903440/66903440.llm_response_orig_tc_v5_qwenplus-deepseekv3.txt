After analyzing the code and the failing case, I noticed that the main issue is in the boundary condition checking. In the code, there's a typo in the boundary check where `ni` is checked twice instead of checking `ni` and `nj`. Here's the fixed code:

    rep(i,H)rep(j,W){
        if(i==0||i==H-1||j==0||j==W-1){

            Q[A[i][j]].push({i,j});
        }

        while(!Q[y].empty()){
            auto [i,j]=Q[y].front();
            Q[y].pop();
            ans--;
            rep(k,4){
                int ni=i+dx[k];
                int nj=j+dy[k];
                if(ni>=0&&ni<H&&nj>=0&&nj<W){
                    if(F[ni][nj]){
                        F[ni][nj]=false;
                        Q[max(A[ni][nj],y)].push({ni,nj});
                    }
                }
            }
        }
        cout<<ans<<'\n';
    }
}
