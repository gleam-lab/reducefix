```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = int64_t;

const int INF = 1e9;

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    int total = H * W;
    vector<vector<bool>> is_land(H, vector<bool>(W, true));
    vector<queue<pair<int, int>>> year_queues(Y + 2); // Years 1 to Y

    // Initialize border cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                is_land[i][j] = false;
                int level = min(Y + 1, A[i][j]);
                year_queues[level].push({i, j});
            }
        }
    }

    // Process each year's changes
    for (int year = 1; year <= Y; ++year) {
        queue<pair<int, int>> q = year_queues[year];

        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();

            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && is_land[nx][ny]) {
                    is_land[nx][ny] = false;
                    int next_level = min(Y + 1, A[nx][ny]);
                    if (next_level > year) {
                        year_queues[next_level].push({nx, ny});
                    } else {
                        // If the adjacent cell also sinks this year, process it now
                        q.push({nx, ny});
                    }
                }
            }
            total--;
        }

        cout << total << "\n";
    }

    return 0;
}
```
