#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);

    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Initialize answer and visited array
    int ans = H * W;
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    
    // We'll process cells in increasing order of elevation using events per year
    vector<vector<pair<int,int>>> add_cells(Y+1); // cells that become exposed to flooding at sea level = y
    
    // Add boundary cells to be processed at their elevation or when sea level reaches them
    rep(i,H) rep(j,W) {
        if(i == 0 || i == H-1 || j == 0 || j == W-1) {
            if(A[i][j] <= Y) {
                add_cells[A[i][j]].push_back({i,j});
            } else {
                add_cells[Y].push_back({i,j}); // if elevation > Y, we still need to consider it by year Y
            }
        }
    }
    
    // Use BFS to simulate flooding over years
    queue<pair<int,int>> q;
    
    rep(y, Y) {
        int sea_level = y + 1; // sea level after y+1 years
        
        // Add new boundary cells that start flooding at this sea level
        for(auto [i,j] : add_cells[sea_level]) {
            if(!visited[i][j] && A[i][j] <= sea_level) {
                visited[i][j] = true;
                q.push({i,j});
                ans--;
            }
        }
        
        // Flood propagation from all currently queued cells
        while(!q.empty()) {
            auto [i,j] = q.front(); q.pop();
            
            rep(k,4) {
                int ni = i + dx[k];
                int nj = j + dy[k];
                if(ni >= 0 && ni < H && nj >= 0 && nj < W && !visited[ni][nj] && A[ni][nj] <= sea_level) {
                    visited[ni][nj] = true;
                    q.push({ni,nj});
                    ans--;
                }
            }
        }
        
        cout << ans << endl;
    }
}
