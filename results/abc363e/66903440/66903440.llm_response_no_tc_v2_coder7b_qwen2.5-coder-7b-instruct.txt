#include<bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); 
    cin.tie(NULL); 

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    int ans = H * W;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                q.push({i, j});
                visited[i][j] = true;
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        int new_ans = ans;
        while (!q.empty()) {
            auto [i, j] = q.front(); q.pop();
            new_ans--;
            for (int k = 0; k < 4; ++k) {
                int ni = i + dx[k], nj = j + dy[k];
                if (ni >= 0 && ni < H && nj >= 0 && nj < W && !visited[ni][nj]) {
                    if (A[ni][nj] <= y) {
                        visited[ni][nj] = true;
                        q.push({ni, nj});
                    }
                }
            }
        }
        ans = new_ans;
        cout << ans << '\n';
    }

    return 0;
}
