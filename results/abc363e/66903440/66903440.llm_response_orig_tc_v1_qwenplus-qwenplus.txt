```cpp
#include <bits/stdc++.h>
using namespace std;
#include <atcoder/all>
using namespace atcoder;
using ll = int64_t;

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Initialize answer
    ll ans = 1LL * H * W;
    
    // visited matrix
    vector<vector<bool>> sink(H, vector<bool>(W, false));

    // For each cell on the border, add to the queue of its height
    vector<queue<pair<int, int>>> q(200001); // max possible height is up to 1e5 + some buffer

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                sink[i][j] = true;
                q[A[i][j]].push({i, j});
                ans--;
            }
        }
    }

    // Directions: up, right, down, left
    const int dx[] = {-1, 0, 1, 0};
    const int dy[] = {0, 1, 0, -1};

    for (int year = 1; year <= Y; ++year) {
        while (!q[year].empty()) {
            auto [x, y] = q[year].front();
            q[year].pop();

            for (int k = 0; k < 4; ++k) {
                int nx = x + dx[k];
                int ny = y + dy[k];

                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !sink[nx][ny]) {
                    sink[nx][ny] = true;
                    int h = A[nx][ny];
                    q[max(h, year)].push({nx, ny});
                    ans--;
                }
            }
        }
        cout << ans << '\n';
    }

    return 0;
}
```
