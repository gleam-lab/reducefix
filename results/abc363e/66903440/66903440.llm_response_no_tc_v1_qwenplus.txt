#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Total area initially
    int ans = H * W;
    vector<vector<bool>> sunk(H, vector<bool>(W, false));
    
    // Instead of pre-allocating huge queue array, use events per year
    vector<vector<pair<int,int>>> events(Y+1); // events[year] = list of cells that sink at this sea level
    
    // Initially, all border cells will eventually sink when sea level reaches their elevation
    rep(i,H) rep(j,W) {
        if(i == 0 || i == H-1 || j == 0 || j == W-1) {
            if(A[i][j] <= Y) {
                events[A[i][j]].push_back({i,j});
            }
        }
    }
    
    // Process each year from 1 to Y
    for(int y = 1; y <= Y; y++) {
        queue<pair<int,int>> q;
        
        // Add all cells that start sinking at sea level y
        for(auto [i,j] : events[y]) {
            if(!sunk[i][j]) {
                sunk[i][j] = true;
                q.push({i,j});
                ans--;
            }
        }
        
        // Flood fill: any adjacent cell with elevation <= current sea level also sinks
        while(!q.empty()) {
            auto [i,j] = q.front(); q.pop();
            
            rep(k,4) {
                int ni = i + dx[k];
                int nj = j + dy[k];
                if(ni >= 0 && ni < H && nj >= 0 && nj < W && !sunk[ni][nj] && A[ni][nj] <= y) {
                    sunk[ni][nj] = true;
                    q.push({ni,nj});
                    ans--;
                }
            }
        }
        
        cout << ans << endl;
    }
}
