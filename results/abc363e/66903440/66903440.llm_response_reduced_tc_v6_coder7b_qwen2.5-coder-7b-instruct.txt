```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }
    
    vector<vector<int>> grid(H, vector<int>(W, 0));
    queue<pair<int, int>> q;
    
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            if(i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                grid[i][j] = 1;
                q.push({i, j});
            }
        }
    }
    
    vector<int> ans(Y + 1, H * W);
    
    for(int k = 1; k <= Y; ++k) {
        int size = q.size();
        while(size--) {
            auto [i, j] = q.front(); q.pop();
            --ans[k];
            for(int di = -1; di <= 1; ++di) {
                for(int dj = -1; dj <= 1; ++dj) {
                    if(abs(di) + abs(dj) != 1) continue;
                    int ni = i + di, nj = j + dj;
                    if(ni >= 0 && ni < H && nj >= 0 && nj < W && !grid[ni][nj] && A[ni][nj] <= k) {
                        grid[ni][nj] = 1;
                        q.push({ni, nj});
                        --ans[k];
                    }
                }
            }
        }
    }
    
    for(int i = 1; i <= Y; ++i) {
        cout << ans[i] << '\n';
    }
    
    return 0;
}
```
