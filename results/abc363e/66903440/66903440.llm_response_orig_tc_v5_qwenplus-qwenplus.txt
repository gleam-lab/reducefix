```cpp
#include <bits/stdc++.h>
using namespace std;
using atcoder::queue;

using ll = int64_t;

const int dx[] = {1, 0, -1, 0};
const int dy[] = {0, 1, 0, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    int total = H * W;
    vector<vector<bool>> is_submerged(H, vector<bool>(W, false));
    vector<queue<pair<int, int>>> year_queue(200000); // Max possible Y

    // Initialize borders
    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                year_queue[A[i][j]].push({i, j});
                is_submerged[i][j] = true;
                --total;
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        while (!year_queue[y].empty()) {
            auto [x, y_coord] = year_queue[y].front();
            year_queue[y].pop();

            for (int k = 0; k < 4; ++k) {
                int nx = x + dx[k];
                int ny = y_coord + dy[k];

                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !is_submerged[nx][ny]) {
                    int h = A[nx][ny];
                    if (h <= y) {
                        is_submerged[nx][ny] = true;
                        --total;
                        year_queue[max(h, y)].push({nx, ny});
                    }
                }
            }
        }
        cout << total << "\n";
    }

    return 0;
}
```
