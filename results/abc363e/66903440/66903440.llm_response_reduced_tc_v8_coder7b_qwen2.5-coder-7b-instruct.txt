#include <bits/stdc++.h>

using namespace std;

vector<vector<int>> grid;
vector<vector<bool>> visited;
int n, m, y;

// Directions array for moving up, down, left, right
const vector<pair<int, int>> directions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

// Function to perform BFS to mark all reachable cells from a given cell
void bfs(int x, int y, int level) {
    queue<pair<int, int>> q;
    q.push({x, y});
    visited[x][y] = true;

    while (!q.empty()) {
        pair<int, int> current = q.front();
        q.pop();

        // Check all four possible directions
        for (auto &dir : directions) {
            int nx = current.first + dir.first;
            int ny = current.second + dir.second;

            // If the new position is within bounds and not yet visited and its height is less than or equal to the current water level
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && !visited[nx][ny] && grid[nx][ny] <= level) {
                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> m >> y;
    grid.resize(n, vector<int>(m));
    visited.resize(n, vector<bool>(m));

    // Read the grid
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> grid[i][j];
        }
    }

    // For each water level from 1 to y
    for (int level = 1; level <= y; ++level) {
        // Reset the visited matrix
        fill(visited.begin(), visited.end(), vector<bool>(m, false));

        // Perform BFS from every unvisited cell whose height is less than or equal to the current water level
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (!visited[i][j] && grid[i][j] <= level) {
                    bfs(i, j, level);
                }
            }
        }

        // The number of remaining cells is the total cells minus the visited cells
        int remaining_cells = n * m;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (visited[i][j]) {
                    --remaining_cells;
                }
            }
        }

        // Print the result
        cout << remaining_cells << endl;
    }

    return 0;
}
