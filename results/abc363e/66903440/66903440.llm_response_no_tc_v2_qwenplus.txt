#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Current remaining area
    int ans = H*W;
    // Visited marker - true means still above water
    vector<vector<bool>> visited(H, vector<bool>(W, true));
    
    // We'll use a priority queue or event-based simulation
    // Instead: group cells by their elevation and process in increasing order of sea level
    vector<vector<pair<int,int>>> events(Y+1); // events[sea_level] = list of border cells to trigger
    
    // Initialize: all border cells will sink when sea level reaches their elevation
    rep(i,H)rep(j,W){
        if(i==0 || i==H-1 || j==0 || j==W-1){
            if(A[i][j] <= Y){
                events[A[i][j]].push_back({i,j});
            }
        }
    }
    
    // Use BFS from borders as sea level rises
    queue<pair<int,int>> q;
    
    for(int y = 1; y <= Y; y++){
        // Add all new border cells that become submerged at sea level y
        for(auto [i,j] : events[y]){
            if(visited[i][j]){
                visited[i][j] = false;
                q.push({i,j});
                ans--;
            }
        }
        
        // Propagate flooding: any adjacent cell with elevation <= y gets flooded
        while(!q.empty()){
            auto [i,j] = q.front(); q.pop();
            
            rep(k,4){
                int ni = i + dx[k];
                int nj = j + dy[k];
                if(ni >= 0 && ni < H && nj >= 0 && nj < W){
                    if(visited[ni][nj] && A[ni][nj] <= y){
                        visited[ni][nj] = false;
                        q.push({ni,nj});
                        ans--;
                    }
                }
            }
        }
        
        cout << ans << endl;
    }
}
