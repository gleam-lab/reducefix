#include<bits/stdc++.h>
using namespace std;
using ll = int64_t;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }
    
    // Initial area calculation
    ll total_area = H * W;
    
    // Visited matrix to keep track of whether a cell has sunk or not
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    
    // Queue for BFS
    queue<pair<int, int>> q;
    
    // Initialize the queue with the boundary cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                total_area -= A[i][j]; // Subtract the sinking area from total area
                q.push({i, j});
            }
        }
    }
    
    // Perform BFS to simulate the sinking process
    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        
        for (int k = 0; k < 4; ++k) { // Check the 4 adjacent cells
            int nx = x + (k & 1 ? -1 : 1);
            int ny = y + (k & 1 ? 1 : -1);
            
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && A[nx][ny] <= A[x][y] + 1) {
                visited[nx][ny] = true;
                q.push({nx, ny});
                total_area -= A[nx][ny]; // Subtract the sinking area from total area
            }
        }
    }
    
    // Output the remaining area for each year
    for (int y = 1; y <= Y; ++y) {
        cout << total_area << endl;
    }
    
    return 0;
}
