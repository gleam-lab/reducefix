```cpp
#include <bits/stdc++.h>
using namespace std;
#include <atcoder/all>
using namespace atcoder;

using ll = int64_t;
using pii = pair<int, int>;

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    const int dx[4] = {1, 0, -1, 0};
    const int dy[4] = {0, 1, 0, -1};

    // Total area of the island
    int total_area = H * W;

    // Visited matrix to mark if a cell is submerged
    vector<vector<bool>> submerged(H, vector<bool>(W, false));

    // Queue for BFS processing by year
    vector<queue<pii>> yearly_queue(Y + 2);

    // Prepare initial boundary cells (edges) for processing
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                submerged[i][j] = true;
                yearly_queue[A[i][j]].push({i, j});
            }
        }
    }

    // Start from the first year and process each year up to Y
    int years_passed = 0;
    while (years_passed < Y) {
        ++years_passed;

        queue<pii> q = yearly_queue[years_passed];
        yearly_queue[years_passed].empty(); // Clear the queue

        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();

            --total_area;

            for (int k = 0; k < 4; ++k) {
                int nx = x + dx[k];
                int ny = y + dy[k];

                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !submerged[nx][ny]) {
                    if (A[nx][ny] <= years_passed) {
                        submerged[nx][ny] = true;
                        yearly_queue[max(A[nx][ny], years_passed)].push({nx, ny});
                    }
                }
            }
        }

        cout << total_area << "\n";
    }
}
```
