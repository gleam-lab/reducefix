#include<bits/stdc++.h>
using namespace std;
using ll=int64_t;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Use union-find or rather: process by elevation threshold
    // Key insight: at year y, any cell with elevation <= y that is connected to the sea (border) via cells of elevation <= y will be submerged
    
    // Instead: simulate backwards or use event processing
    // Better: use priority queue starting from borders, like Dijkstra
    
    priority_queue<tuple<int,int,int>, vector<tuple<int,int,int>>, greater<tuple<int,int,int>>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    
    // Start from all border cells
    rep(i,H) {
        rep(j,W) {
            if(i == 0 || i == H-1 || j == 0 || j == W-1) {
                pq.push({A[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }
    
    vector<int> result(Y+1, H*W);
    int count = H*W;
    
    while(!pq.empty()) {
        auto [elev, i, j] = pq.top();
        pq.pop();
        
        count--;
        // This cell gets submerged when sea level reaches 'elev'
        if(elev <= Y) {
            result[elev] = min(result[elev], count);
        }
        
        rep(d,4) {

                visited[ni][nj] = true;
                pq.push({A[ni][nj], ni, nj});
            }
        }
    }
    
    // Propagate the results forward - area can't increase
    for(int y = 1; y <= Y; y++) {
        result[y] = min(result[y], result[y-1]);

    }
