void bfs(int y) {
    while (!Q[y].empty()) {
        auto [i, j] = Q[y].front();
        Q[y].pop();
        F[i][j] = false;

                F[ni][nj] = false;
                Q[max(y, A[ni][nj])].push({ni, nj});
            }
        }
    }
}
