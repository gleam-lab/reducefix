#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Current area above sea level
    int ans = H * W;
    // Visited/flooded marker
    vector<vector<bool>> flooded(H, vector<bool>(W, false));
    
    // We'll process flooding in increasing order of elevation
    // Instead of having a huge array of queues, use events per year
    vector<vector<pair<int,int>>> add_to_queue(Y+1); // For each year, cells that get added
    
    // Initialize: border cells will flood when sea level >= their elevation
    rep(i,H) rep(j,W) {
        if(i == 0 || i == H-1 || j == 0 || j == W-1) {
            int sea_level_needed = A[i][j]; // This cell floods when sea level reaches A[i][j]
            if(sea_level_needed <= Y) {
                add_to_queue[sea_level_needed].push_back({i,j});
            }
        }
    }
    
    // BFS-like propagation using a single queue for current processing
    queue<pair<int,int>> q;
    
    for(int y = 1; y <= Y; y++) {
        // Add all cells that become exposed to flooding at sea level y
        for(auto &cell : add_to_queue[y]) {
            int i = cell.first, j = cell.second;
            if(!flooded[i][j]) {
                flooded[i][j] = true;
                q.push({i,j});
                ans--;
            }
        }
        
        // Propagate flooding from the queue: any adjacent cell with elevation <= y
        // will also flood and can trigger further flooding
        while(!q.empty()) {
            auto [i,j] = q.front(); q.pop();
            
            rep(k,4) {
                int ni = i + dx[k];
                int nj = j + dy[k];
                if(ni >= 0 && ni < H && nj >= 0 && nj < W) {
                    if(!flooded[ni][nj] && A[ni][nj] <= y) {
                        flooded[ni][nj] = true;
                        q.push({ni,nj});
                        ans--;
                    }
                }
            }
        }
        
        cout << ans << '\n';
    }
}
