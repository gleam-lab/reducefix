```cpp
#include <bits/stdc++.h>
using namespace std;
using namespace atcoder;
using ll = int64_t;

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);
    
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    int ans = H * W;
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    vector<queue<pair<int, int>>> q(Y + 2);

    // Initialize the perimeter (edges) as visited and add to queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                q[A[i][j]].push({i, j});
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        while (!q[y].empty()) {
            auto [x, y_coord] = q[y].front();
            q[y].pop();
            ans--;

            for (int dir = 0; dir < 4; ++dir) {
                int nx = x + dx[dir];
                int ny = y_coord + dy[dir];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    visited[nx][ny] = true;
                    // If the cell's elevation is less than or equal to current sea level,
                    // it will be submerged now. Otherwise, defer it.
                    if (A[nx][ny] <= y) {
                        q[y].push({nx, ny});
                    } else {
                        q[A[nx][ny]].push({nx, ny});
                    }
                }
            }
        }
        cout << ans << "\n";
    }

    return 0;
}
```
