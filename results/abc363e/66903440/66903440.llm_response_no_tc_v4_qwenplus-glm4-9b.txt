#include<bits/stdc++.h>
using namespace std;
using ll = long long;

const int MAX_H = 1000;
const int MAX_W = 1000;
const int MAX_Y = 100000;
const int MAX_A = 100000;

int H, W, Y;
int A[MAX_H][MAX_W];
ll ans[MAX_Y];
bool visited[MAX_H][MAX_W];

void floodFill(int x, int y) {
    visited[x][y] = true;
    for (int i = 0; i < 4; ++i) {
        int nx = x + (i & 1 ? -1 : 1);
        int ny = y + (i & 2 ? -1 : 1);
        if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && A[nx][ny] > A[x][y]) {
            floodFill(nx, ny);
        }
    }
}

int main() {
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Initialize the visited array to false
    memset(visited, false, sizeof(visited));

    // Find the initial area that does not sink
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!visited[i][j]) {
                floodFill(i, j);
            }
        }
    }

    // Calculate the area of the island that remains above sea level for each year
    for (int year = 0; year < Y; ++year) {
        int remaining_area = 0;
        for (int i = 0; i < H; ++i) {
            for (int j = 0; j < W; ++j) {
                if (!visited[i][j]) {
                    remaining_area++;
                }
            }
        }
        ans[year] = remaining_area;
    }

    // Output the results
    for (int i = 0; i < Y; ++i) {
        cout << ans[i] << endl;
    }

    return 0;
}
