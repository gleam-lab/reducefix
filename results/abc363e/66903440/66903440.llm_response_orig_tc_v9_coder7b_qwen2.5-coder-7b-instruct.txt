```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1010;
int H, W, Y;
int A[MAXN][MAXN];
bool vis[MAXN][MAXN];

vector<int> dx = {0, 0, 1, -1};
vector<int> dy = {1, -1, 0, 0};

void bfs(int h) {
    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!vis[i][j] && A[i][j] > h) {
                q.push({i, j});
                vis[i][j] = true;
            }
        }
    }

    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny] && A[nx][ny] > h) {
                q.push({nx, ny});
                vis[nx][ny] = true;
            }
        }
    }
}

int count_area() {
    int cnt = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!vis[i][j]) {
                ++cnt;
                bfs(A[i][j]);
            }
        }
    }
    return cnt;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    for (int h = 1; h <= Y; ++h) {
        memset(vis, false, sizeof(vis));
        cout << count_area() << '\n';
    }

    return 0;
}
```
