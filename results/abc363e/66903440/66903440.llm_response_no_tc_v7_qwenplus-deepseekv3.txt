            max_A = max(max_A, A[i][j]);
        }

            if(i==0 || i==H-1 || j==0 || j==W-1){

                Q[A[i][j]].push({i,j});
                ans--;
            }
        }
    }
    
    for(int y=1;y<=Y;y++){
