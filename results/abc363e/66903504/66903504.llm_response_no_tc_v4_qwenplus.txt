    vector<vector<pair<int,int>>> cells(100002);
    rep(i,H)rep(j,W){
        cells[A[i][j]].push_back({i,j});
    }

    int ans = H*W;
    vector<vector<bool>> submerged(H, vector<bool>(W, false));
    queue<pair<int,int>> q;

    // Initialize: add all border cells that are not yet submerged
    auto add_if_border = [&](int i, int j) {
        if ((i == 0 || i == H-1 || j == 0 || j == W-1) && !submerged[i][j] && A[i][j] <= Y) {

            q.push({i,j});
            ans--;

    };

    // Process each year

        // Add all cells with elevation y that are on border
        for(auto [i,j] : cells[y]) {
            add_if_border(i, j);
