    queue<pair<int,int>> q;

    // Precompute: for each sea level y, which boundary cells get submerged?
    vector<vector<pair<int,int>>> add_at_level(Y+1);
    
    // Mark initial boundary cells
    rep(i,H) {
        rep(j,W) {
            if (i == 0 || i == H-1 || j == 0 || j == W-1) {
                if (A[i][j] <= Y) {

            }

        // Add all cells on border with height <= y that haven't been processed
        for (auto [i,j] : add_at_level[y]) {
