#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Create events: when each cell gets submerged
    vector<vector<int>> sink_year(H, vector<int>(W, Y+1));
    queue<pair<int,int>> q;
    
    // Initialize: border cells sink at year = A[i][j]
    rep(i,H){
        rep(j,W){
            if(i==0 || i==H-1 || j==0 || j==W-1){
                sink_year[i][j] = A[i][j];
                q.push({i,j});
            }
        }
    }
    
    // Multi-source BFS to propagate sinking
    while(!q.empty()){
        auto [i,j] = q.front(); q.pop();
        int current_year = sink_year[i][j];
        
        rep(k,4){
            int ni = i + dx[k];
            int nj = j + dy[k];
            if(ni>=0 && ni<H && nj>=0 && nj<W){
                // The neighbor sinks at max(current_year, A[ni][nj])
                // but only if this would make it sink earlier
                int new_year = max(current_year, A[ni][nj]);
                if(new_year < sink_year[ni][nj]){
                    sink_year[ni][nj] = new_year;
                    q.push({ni,nj});
                }
            }
        }
    }
    
    // Count how many cells survive each year
    vector<int> surviving_count(Y+1, H*W);
    vector<int> events(Y+2, 0); // events[y] = number of cells that sink at year y
    
    rep(i,H)rep(j,W){
        if(sink_year[i][j] <= Y){
            events[sink_year[i][j]]++;
        }
    }
    
    // Calculate surviving area for each year
    int current_area = H*W;
    for(int y=1; y<=Y; y++){
        current_area -= events[y];
        cout << current_area << '\n';
    }
}
