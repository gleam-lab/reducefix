```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1005;

int h, w, y;
vector<vector<int>> a(MAXN, vector<int>(MAXN));
vector<vector<bool>> visited(MAXN, vector<bool>(MAXN, false));
vector<queue<pair<int, int>>> q(202020);

int dx[] = {1, 0, -1, 0, 1, 1, -1, -1};
int dy[] = {0, 1, 0, -1, 1, -1, 1, -1};

void bfs(int level) {
    queue<pair<int, int>> &current_q = q[level];
    while (!current_q.empty()) {
        auto [i, j] = current_q.front(); current_q.pop();
        visited[i][j] = true;
        for (int k = 0; k < 8; ++k) {
            int ni = i + dx[k], nj = j + dy[k];
            if (ni >= 0 && ni < h && nj >= 0 && nj < w && !visited[ni][nj] && a[ni][nj] <= level) {
                visited[ni][nj] = true;
                q[level + 1].push({ni, nj});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    cin >> h >> w >> y;
    for (int i = 0; i < h; ++i) for (int j = 0; j < w; ++j) cin >> a[i][j];

    int total = h * w;
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                visited[i][j] = true;
                q[a[i][j]].push({i, j});
            }
        }
    }

    for (int level = 1; level <= y; ++level) {
        bfs(level);
        total -= q[level].size();
        cout << total << endl;
    }

    return 0;
}
```
