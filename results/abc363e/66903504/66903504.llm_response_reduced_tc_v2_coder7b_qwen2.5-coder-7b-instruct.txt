#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1005;
const int MAXY = 100005;

int H, W, Y;
int A[MAXN][MAXN];
bool F[MAXN][MAXN];
queue<pair<int, int>> Q[MAXY];

void bfs(int sea_level) {
    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!F[i][j] && A[i][j] <= sea_level) {
                q.push({i, j});
                F[i][j] = true;
            }
        }
    }
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !F[nx][ny] && A[nx][ny] <= sea_level) {
                F[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    cin >> H >> W >> Y;
    memset(F, 0, sizeof(F));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                F[i][j] = true;
                Q[A[i][j]].push({i, j});
            }
        }
    }
    
    for (int sea_level = 1; sea_level <= Y; ++sea_level) {
        while (!Q[sea_level].empty()) {
            auto [x, y] = Q[sea_level].front(); Q[sea_level].pop();
            F[x][y] = true;
            for (int k = 0; k < 4; ++k) {
                int nx = x + dx[k], ny = y + dy[k];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !F[nx][ny] && A[nx][ny] <= sea_level) {
                    F[nx][ny] = true;
                    Q[max(A[nx][ny], sea_level)].push({nx, ny});
                }
            }
        }
        int count = 0;
        for (int i = 0; i < H; ++i) {
            for (int j = 0; j < W; ++j) {
                if (!F[i][j]) count++;
            }
        }
        cout << count << '\n';
    }
    
    return 0;
}
