#include <bits/stdc++.h>
using namespace std;

using ll = int64_t;

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    // Initialize a visited matrix to track if a cell is submerged
    vector<vector<bool>> submerged(H, vector<bool>(W, false));

    // Queue for BFS per elevation level
    vector<queue<pair<int, int>>> level_queues(Y + 2); // Up to Y+1

    // Initialize border cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                submerged[i][j] = true;
                if (A[i][j] <= Y)
                    level_queues[A[i][j]].push({i, j});
            }
        }
    }

    int remaining = H * W;
    queue<pair<int, int>> q;

    for (int year = 1; year <= Y; ++year) {
        // Add all cells with elevation == year to the BFS queue
        while (!level_queues[year].empty()) {
            auto [x, y] = level_queues[year].front();
            level_queues[year].pop();

            if (!submerged[x][y]) {
                submerged[x][y] = true;
                remaining--;
                q.push({x, y});
            }
        }

        // Perform BFS for this year's submersion cascade
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();

            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !submerged[nx][ny] &&
                    A[nx][ny] <= year) {
                    submerged[nx][ny] = true;
                    remaining--;
                    q.push({nx, ny});
                }
            }
        }

        cout << remaining << "\n";
    }
}
