```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    vector<vector<int>> vis(H, vector<int>(W, 0));
    vector<vector<int>> q(Y + 1);

    // Initialize the queue with boundary cells
    for(int i = 0; i < H; ++i) {
        if(A[i][0] <= Y || A[i][W - 1] <= Y) q[A[i][0]].emplace_back(i * W), q[A[i][W - 1]].emplace_back(i * W + W - 1);
        if(i == 0 || i == H - 1) {
            for(int j = 1; j < W - 1; ++j) {
                if(A[i][j] <= Y) q[A[i][j]].emplace_back(i * W + j);
            }
        }
    }

    // Process each level of sea rise
    for(int h = 1; h <= Y; ++h) {
        int cnt = 0;
        for(auto u : q[h]) {
            int i = u / W, j = u % W;
            if(vis[i][j]) continue;
            vis[i][j] = 1;
            ++cnt;
            for(int d = 0; d < 4; ++d) {
                int ii = i + d, jj = j + (d & 1 ? -1 : 1);
                if(ii >= 0 && ii < H && jj >= 0 && jj < W && !vis[ii][jj] && A[ii][jj] <= h) {
                    q[h].emplace_back(ii * W + jj);
                }
            }
        }
        cout << H * W - cnt << '\n';
    }

    return 0;
}
```
