#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Store cells by their elevation
    vector<vector<pair<int,int>>> cells_at_elevation(100001);
    rep(i,H)rep(j,W){
        cells_at_elevation[A[i][j]].push_back({i,j});
    }
    
    vector<vector<bool>> submerged(H, vector<bool>(W, false));
    int remaining = H * W;
    
    // Process years from 1 to Y
    for(int year = 1; year <= Y; year++){
        // Add all cells with elevation <= current sea level (year)
        // that haven't been processed yet
        for(auto [i, j] : cells_at_elevation[year]){
            if(!submerged[i][j]){
                queue<pair<int,int>> q;
                q.push({i,j});
                submerged[i][j] = true;
                remaining--;
                
                while(!q.empty()){
                    auto [r, c] = q.front(); q.pop();
                    
                    rep(d,4){
                        int nr = r + dx[d];
                        int nc = c + dy[d];
                        if(nr >= 0 && nr < H && nc >= 0 && nc < W && !submerged[nr][nc]){
                            // If adjacent cell has elevation <= current sea level, it also sinks
                            if(A[nr][nc] <= year){
                                submerged[nr][nc] = true;
                                remaining--;
                                q.push({nr, nc});
                            }
                        }
                    }
                }
            }
        }
        
        cout << remaining << '\n';
    }
}
