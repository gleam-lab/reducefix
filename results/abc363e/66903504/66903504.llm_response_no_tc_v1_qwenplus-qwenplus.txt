#include <bits/stdc++.h>
using namespace std;

using ll = int64_t;
const int INF = 1000000001;

#define rep(i, n) for (int i = 0; i < (n); ++i)
#define all(x) (x).begin(), (x).end()

int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    rep(i, H) rep(j, W) cin >> A[i][j];

    int total = H * W;
    vector<vector<bool>> alive(H, vector<bool>(W, true));

    // Queue for each elevation level
    vector<queue<pair<int, int>>> q_by_level(Y + 2);

    // Initialize border cells
    rep(i, H) {
        for (int j : {0, W - 1}) {
            if (alive[i][j]) {
                alive[i][j] = false;
                q_by_level[A[i][j]].push({i, j});
            }
        }
    }

    rep(j, W) {
        for (int i : {0, H - 1}) {
            if (j == 0 || j == W - 1) continue;
            if (alive[i][j]) {
                alive[i][j] = false;
                q_by_level[A[i][j]].push({i, j});
            }
        }
    }

    auto process_queue = [&](int sea_level) {
        queue<pair<int, int>> q;
        while (!q_by_level[sea_level].empty()) {
            q.push(q_by_level[sea_level].front());
            q_by_level[sea_level].pop();
        }

        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            total--;

            rep(k, 4) {
                int nx = x + dx[k];
                int ny = y + dy[k];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && alive[nx][ny]) {
                    if (A[nx][ny] <= sea_level) {
                        alive[nx][ny] = false;
                        q.push({nx, ny});
                    } else {
                        // This cell will be processed in the future when sea_level reaches A[nx][ny]
                        alive[nx][ny] = false;
                        q_by_level[A[nx][ny]].push({nx, ny});
                    }
                }
            }
        }
    };

    // For output
    vector<int> result(Y + 1);
    result[0] = total;

    for (int year = 1; year <= Y; ++year) {
        process_queue(year);
        result[year] = total;
    }

    for (int year = 1; year <= Y; ++year) {
        cout << result[year] << "\n";
    }
}
