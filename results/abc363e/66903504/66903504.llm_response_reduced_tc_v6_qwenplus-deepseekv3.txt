    // Process boundary cells
    rep(i,H)rep(j,W){
        if(i==0||i==H-1||j==0||j==W-1){
            pq.push({A[i][j],{i,j}});
            sunk[i][j]=true;
        }
    }
    
    for(int y=1;y<=Y;y++){
        while(!pq.empty() && pq.top().first<=y){
            auto cell=pq.top();
            pq.pop();
            ans--;
            rep(k,4){
                int ni=cell.second.first+dx[k];
                int nj=cell.second.second+dy[k];
                if(ni>=0&&ni<H&&nj>=0&&nj<W){
                    if(!sunk[ni][nj]){
                        sunk[ni][nj]=true;
                        pq.push({A[ni][nj],{ni,nj}});
                    }
                }
            }
        }
        cout<<ans<<endl;
    }
}

1. Replaced the array of queues with a priority queue to handle cells in elevation order
2. Fixed the boundary condition checking to properly handle 1x1 islands
3. Simplified the neighbor checking to only 4 directions (up/down/left/right)
4. Added proper tracking of sunk cells to avoid duplicate processing
5. The priority queue ensures we process cells in the correct order as the sea level rises
