#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Create events: when each cell gets flooded
    vector<vector<int>> flood_year(H, vector<int>(W, Y+1));
    queue<pair<int,int>> q;
    
    // Initialize: all border cells will flood at year A[i][j]
    rep(i,H){
        rep(j,W){
            if(i==0||i==H-1||j==0||j==W-1){
                flood_year[i][j] = A[i][j];
                q.push({i,j});
            }
        }
    }
    
    // BFS to propagate flooding - a cell floods when its elevation <= current sea level
    // and it's connected to already flooded area
    while(!q.empty()){
        auto [i,j] = q.front(); q.pop();
        int current_flood_year = flood_year[i][j];
        
        rep(k,4){
            int ni = i + dx[k];
            int nj = j + dy[k];
            if(ni>=0 && ni<H && nj>=0 && nj<W){
                // The neighbor will flood at max(A[ni][nj], current_flood_year)
                // because it needs sea level to reach at least its own elevation
                int new_flood_year = max(A[ni][nj], current_flood_year);
                if(new_flood_year < flood_year[ni][nj]){
                    flood_year[ni][nj] = new_flood_year;
                    q.push({ni,nj});
                }
            }
        }
    }
    
    // Count how many cells survive each year
    vector<int> surviving_count(Y+1, H*W);
    vector<int> flood_events(Y+2, 0); // flood_events[t] = number of cells that flood at year t
    
    rep(i,H){
        rep(j,W){
            int t = flood_year[i][j];
            if(t <= Y){
                flood_events[t]++;
            }
        }
    }
    
    // Subtract cells that flood at each year
    for(int y=1; y<=Y; y++){
        surviving_count[y] = surviving_count[y-1] - flood_events[y];
        cout << surviving_count[y] << '\n';
    }
}
