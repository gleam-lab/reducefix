```cpp
#include <bits/stdc++.h>
using namespace std;
#include <atcoder/all>
using namespace atcoder;
using ll = int64_t;

int main() {
    cin.tie(0)->sync_with_stdio(0);
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) for (int j = 0; j < W; ++j) cin >> A[i][j];

    const int MAX_A = 200000;
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    vector<queue<pair<int, int>>> que(MAX_A + 2);
    int total_area = H * W;

    // Initialize borders as visited and add to queues
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                que[A[i][j]].push({i, j});
                --total_area;
            }
        }
    }

    int dx[] = {1, 0, -1, 0};
    int dy[] = {0, 1, 0, -1};

    for (int y = 1; y <= Y; ++y) {
        while (!que[y].empty()) {
            auto [x, y_coord] = que[y].front();
            que[y].pop();
            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y_coord + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    visited[nx][ny] = true;
                    int next_threshold = A[nx][ny];
                    if (next_threshold <= y) {
                        que[y].push({nx, ny});
                    } else {
                        que[next_threshold].push({nx, ny});
                    }
                    --total_area;
                }
            }
        }
        cout << total_area << '\n';
    }

    return 0;
}
```
