#include<bits/stdc++.h>
using namespace std;
#include<atcoder/all>
using namespace atcoder;
using ll=int64_t;
using ld=long double;
int inf=1000000001;
ll INF=1e18;
#define rep(i,n) for(int i=0;i<n;i++)
#define all(x) x.begin(),x.end()
#define pb push_back
#define sz(x) (ll)x.size()
template<typename T>bool chmin(T& a,T b){if(a>b){a=b;return true;}return false;}
template<typename T>bool chmax(T& a,T b){if(a<b){a=b;return true;}return false;}

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};

int main(){
    cin.tie(0)->sync_with_stdio(0);
    
    int H,W,Y;
    cin>>H>>W>>Y;
    vector<vector<int>>A(H,vector<int>(W));
    rep(i,H)rep(j,W)cin>>A[i][j];
    
    // Create events: when each cell gets submerged
    vector<vector<bool>>sunk(H,vector<bool>(W,false));
    vector<vector<int>>q(Y+1); // q[y] = list of cells that get processed at year y
    
    // Initially, border cells with A[i][j] <= y will be sunk at year A[i][j]
    rep(i,H){
        rep(j,W){
            if(i==0||i==H-1||j==0||j==W-1){
                if(A[i][j] <= Y){
                    q[A[i][j]].pb(i*W+j);
                }
                sunk[i][j] = true;
            }
        }
    }
    
    int remaining = H*W;
    // Count already sunk border cells that are below sea level from start
    rep(i,H) rep(j,W) {
        if((i==0||i==H-1||j==0||j==W-1) && A[i][j] <= 0) {
            remaining--;
        }
    }
    
    // Process each year
    for(int y=1; y<=Y; y++){
        queue<pair<int,int>> Q;
        // Add all cells that have elevation exactly y and are on the "frontier"
        for(int pos : q[y]){
            int i = pos / W;
            int j = pos % W;
            if(!sunk[i][j] && A[i][j] <= y){
                sunk[i][j] = true;
                Q.push({i,j});
                remaining--;
            }
        }
        
        // Flood fill from newly submerged cells
        while(!Q.empty()){
            auto [i,j] = Q.front(); Q.pop();
            rep(k,4){
                int ni = i + dx[k];
                int nj = j + dy[k];
                if(ni>=0 && ni<H && nj>=0 && nj<W && !sunk[ni][nj] && A[ni][nj] <= y){
                    sunk[ni][nj] = true;
                    Q.push({ni,nj});
                    remaining--;
                }
            }
        }
        
        cout << remaining << '\n';
    }
}
