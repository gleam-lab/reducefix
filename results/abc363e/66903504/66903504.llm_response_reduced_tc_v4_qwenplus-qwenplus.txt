```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Binary matrix to mark if a cell is still above sea level
    vector<vector<bool>> alive(H, vector<bool>(W, true));

    // Queue for BFS from the borders
    queue<pair<int, int>> q;

    // First, mark all border cells and add them to queue if they sink
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                alive[i][j] = false;
                q.emplace(i, j);
            }
        }
    }

    // Prepare event queues: year -> list of cells whose threshold is that year
    vector<vector<pair<int, int>>> events(Y + 2);  // up to Y+1

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            int threshold = A[i][j];
            if (threshold <= Y) {
                events[threshold].push_back({i, j});
            }
        }
    }

    // Main simulation
    int remaining = H * W;
    vector<int> result(Y + 1);

    // Process each year
    for (int year = 1; year <= Y; ++year) {
        // Add all cells with this elevation to the sinking queue
        for (auto [i, j] : events[year]) {
            if (alive[i][j]) {
                alive[i][j] = false;
                q.push({i, j});
            }
        }

        // Standard BFS to propagate sinking
        while (!q.empty()) {
            auto [i, j] = q.front();
            q.pop();
            remaining--;

            static const int dx[] = {0, 1, 0, -1};
            const int dy[] = {1, 0, -1, 0};

            for (int d = 0; d < 4; ++d) {
                int ni = i + dx[d];
                int nj = j + dy[d];
                if (ni >= 0 && ni < H && nj >= 0 && nj < W && alive[ni][nj]) {
                    alive[ni][nj] = false;
                    q.push({ni, nj});
                }
            }
        }

        result[year] = remaining;
    }

    // Output results for years 1 to Y
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << '\n';
    }

    return 0;
}
```
