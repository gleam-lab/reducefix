```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false); cin.tie(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> seen(H, vector<bool>(W, false));
    vector<vector<int>> q(Y + 1);
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                seen[i][j] = true;
                q[A[i][j]].push_back(i * W + j);
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        while (!q[y].empty()) {
            int idx = q[y].back(); q[y].pop_back();
            int i = idx / W, j = idx % W;
            for (int di = -1; di <= 1; ++di) {
                for (int dj = -1; dj <= 1; ++dj) {
                    if (abs(di) + abs(dj) != 1) continue;
                    int ni = i + di, nj = j + dj;
                    if (ni >= 0 && ni < H && nj >= 0 && nj < W && !seen[ni][nj]) {
                        seen[ni][nj] = true;
                        q[max(y, A[ni][nj])].push_back(ni * W + nj);
                    }
                }
            }
        }
        int cnt = 0;
        for (int i = 0; i < H; ++i) {
            for (int j = 0; j < W; ++j) {
                if (!seen[i][j]) ++cnt;
            }
        }
        cout << cnt << '\n';
    }

    return 0;
}
```
