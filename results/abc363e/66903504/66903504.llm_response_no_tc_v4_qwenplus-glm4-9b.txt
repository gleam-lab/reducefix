#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    
    queue<pair<int, int>> q;
    
    // Initialize the queue with the edges of the island and mark them as visited
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if ((i == 0 || j == 0 || i == H - 1 || j == W - 1) || (i > 0 && j > 0 && A[i - 1][j - 1] + 1 >= A[i][j])) {
                q.push({i, j});
                visited[i][j] = true;
            }
        }
    }
    
    int initial_area = 0;
    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        initial_area++;
        for (int k = 0; k < 4; ++k) {
            int nx = x + k == 0 ? W - 1 : k == 1 ? 0 : x - 1;
            int ny = y + k == 2 ? H - 1 : k == 3 ? 0 : y - 1;
            if (!visited[nx][ny] && (i == 0 || i == H - 1 || j == 0 || j == W - 1) || (nx > 0 && nx < H && ny > 0 && ny < W && A[nx][ny] <= A[i][j] + 1)) {
                q.push({nx, ny});
                visited[nx][ny] = true;
            }
        }
    }
    
    // Calculate the remaining area after Y years
    for (int year = 1; year <= Y; year++) {
        queue<pair<int, int>> current_queue;
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();
            for (int k = 0; k < 4; ++k) {
                int nx = x + k == 0 ? W - 1 : k == 1 ? 0 : x - 1;
                int ny = y + k == 2 ? H - 1 : k == 3 ? 0 : y - 1;
                if (!visited[nx][ny] && (nx >= 0 && nx < W && ny >= 0 && ny < H)) {
                    if (A[nx][ny] <= A[x][y] + year) {
                        current_queue.push({nx, ny});
                        visited[nx][ny] = true;
                        if (nx == 0 || nx == W - 1 || ny == 0 || ny == H - 1 || (nx > 0 && nx < W && ny > 0 && ny < H && A[nx][ny] <= A[nx - 1][ny - 1] + year + 1)) {
                            current_queue.push({nx, ny});
                            visited[nx][ny] = true;
                        }
                    }
                }
            }
        }
        q = current_queue;
    }
    
    // Output the remaining area after Y years
    cout << initial_area - q.size() << endl;
    return 0;
}
