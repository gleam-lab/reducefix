```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1005;
int H, W, Y;
int A[MAXN][MAXN];
bool F[MAXN][MAXN];

int dx[] = {1, 0, -1, 0, 1, 1, -1, -1};
int dy[] = {0, 1, 0, -1, 1, -1, 1, -1};

void bfs(int sea_level) {
    queue<pair<int, int>> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!F[i][j] && A[i][j] <= sea_level) {
                q.push({i, j});
                F[i][j] = true;
            }
        }
    }
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int k = 0; k < 8; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !F[nx][ny] && A[nx][ny] <= sea_level) {
                q.push({nx, ny});
                F[nx][ny] = true;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    for (int sea_level = 1; sea_level <= Y; ++sea_level) {
        memset(F, 0, sizeof(F));
        bfs(sea_level);
        int remaining_area = 0;
        for (int i = 0; i < H; ++i)
            for (int j = 0; j < W; ++j)
                if (!F[i][j])
                    ++remaining_area;
        cout << remaining_area << endl;
    }

    return 0;
}
```

### Explanation:
1. **Initialization**: The `F` array is initialized to keep track of which sections have been submerged.
2. **BFS Function**: The `bfs` function performs a breadth-first search starting from sections that are submerged at the current sea level. It marks all reachable sections as submerged.
3. **Main Loop**: For each sea level from 1 to Y, the `bfs` function is called to determine which sections are submerged. The remaining area is calculated by counting unmarked sections in the `F` array.
4. **Output**: The remaining area is printed for each sea level.

This approach ensures that the algorithm handles the constraints efficiently and correctly processes the changes in the island's shape over time.
