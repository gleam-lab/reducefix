```cpp
#include <bits/stdc++.h>
using namespace std;
#include <atcoder/all>
using namespace atcoder;
using ll = int64_t;

int main() {
    cin.tie(0)->sync_with_stdio(0);
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    int total_cells = H * W;
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    vector<queue<pair<int, int>>> year_queue(Y + 2);

    // Initialize the edge cells into their respective year queues
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                year_queue[A[i][j]].push({i, j});
                visited[i][j] = true;
                --total_cells; // Mark as visited and reduce total land area
            }
        }
    }

    // Directions: up, right, down, left
    vector<pair<int, int>> dirs = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};

    // Process each year
    for (int y = 1; y <= Y; ++y) {
        queue<pair<int, int>> q = year_queue[y];

        while (!q.empty()) {
            auto [x, y_coord] = q.front();
            q.pop();

            for (auto& d : dirs) {
                int nx = x + d.first;
                int ny = y_coord + d.second;
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    if (A[nx][ny] <= y) {
                        visited[nx][ny] = true;
                        --total_cells;
                        year_queue[max(y, A[nx][ny])].push({nx, ny});
                    }
                }
            }
        }

        cout << total_cells << '\n';
    }

    return 0;
}
```
