#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int H, W, Y;
const int MAX = 1005;
vector<vector<int>> A(MAX, vector<int>(MAX));
vector<vector<bool>> visited(MAX, vector<bool>(MAX, false));
vector<vector<bool>> borderCheck(int H, int W, vector<vector<int>>& A, int level) {
    vector<vector<bool>> isBorder(H, vector<bool>(W, false));
    queue<pair<int, int>> q;
    
    // Initialize the border cells
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            if ((i == 0 || i == H - 1 || j == 0 || j == W - 1) && A[i][j] > level && !visited[i][j]) {
                visited[i][j] = true;
                q.push({i, j});
            }

    int dx[] = {-1, 1, 0, 0};
    int dy[] = {0, 0, -1, 1};

    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && A[nx][ny] > level) {
                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }

    return visited;
}

int main() {
    cin.tie(0)->sync_with_stdio(0);
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    vector<int> queries(Y + 2, 0);
    vector<int> heightCount(100005, 0);

    // Count how many cells have each elevation
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            heightCount[A[i][j]]++;

    // Prefix sum to get total cells above sea level up to each height
    for (int h = 1; h <= 100000; ++h)
        heightCount[h] += heightCount[h - 1];

    vector<int> ans(Y + 2, 0);
    for (int year = 1; year <= Y; ++year) {
        // Mark all unvisited cells as potentially still land
        for (int i = 0; i < H; ++i)
            fill(visited[i].begin(), visited[i].end(), false);

        // Perform BFS from borders and mark cells above current sea level
        auto res = borderCheck(H, W, A, year);

        int cnt = 0;
        for (int i = 0; i < H; ++i)
            for (int j = 0; j < W; ++j)
                if (!res[i][j] && A[i][j] > year)
                    cnt++;

        cout << cnt << '\n';
    }

    return 0;
}
