#include <bits/stdc++.h>
using namespace std;

using ll = int64_t;
#define rep(i, n) for(int i = 0; i < (n); ++i)
#define all(x) (x).begin(), (x).end()

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    rep(i, H) rep(j, W) cin >> A[i][j];

    // Initialize visited and current land count
    vector<vector<bool>> is_submerged(H, vector<bool>(W, false));
    int total_area = H * W;

    // Prepare queues for multi-level BFS based on elevation
    vector<queue<pair<int, int>>> level_queue(Y + 2);  // up to Y+1 years

    // Mark borders as submerged initially and add to respective queues
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            if(i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                if(A[i][j] <= Y) {
                    level_queue[A[i][j]].push({i, j});
                }
                is_submerged[i][j] = true;
                total_area--;
            }
        }
    }

    // Process each year with BFS
    for(int year = 1; year <= Y; ++year) {
        queue<pair<int, int>> q = level_queue[year];

        while(!q.empty()) {
            auto [x, y] = q.front(); q.pop();

            for(int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if(nx >= 0 && nx < H && ny >= 0 && ny < W && !is_submerged[nx][ny]) {
                    int h = A[nx][ny];
                    if(h <= Y) {
                        if(h > year) {
                            // Delay submerging until correct year
                            level_queue[h].push({nx, ny});
                        } else {
                            q.push({nx, ny});
                        }
                    }
                    is_submerged[nx][ny] = true;
                    total_area--;
                }
            }
        }
        cout << total_area << '\n';
    }
}
