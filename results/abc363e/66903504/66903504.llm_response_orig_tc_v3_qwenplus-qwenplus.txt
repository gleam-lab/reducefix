#include <bits/stdc++.h>
using namespace std;
using atcoder::minstd_rand0;

typedef long long ll;

const int dx[] = {0, 1, 0, -1};
const int dy[] = {1, 0, -1, 0};

int main() {
    cin.tie(0)->sync_with_stdio(0);

    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    int total_area = H * W;

    // Mark borders as already flooded
    vector<vector<bool>> flooded(H, vector<bool>(W, false));
    queue<pair<int, int>> q;

    // Use a map to group cells by their elevation
    map<int, vector<pair<int, int>>> elevation_map;

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                elevation_map[A[i][j]].emplace_back(i, j);
                flooded[i][j] = true;
                --total_area;
            }
        }
    }

    for (int y = 1; y <= Y; ++y) {
        // Process all cells with elevation equal to current year
        for (auto& p : elevation_map[y]) {
            q.push(p);
        }

        while (!q.empty()) {
            auto [x, y_coord] = q.front();
            q.pop();

            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y_coord + dy[d];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !flooded[nx][ny]) {
                    if (A[nx][ny] <= y) {
                        flooded[nx][ny] = true;
                        --total_area;
                        // Flood neighbors only if elevation is <= current year
                        elevation_map[A[nx][ny]].erase(
                            remove(elevation_map[A[nx][ny]].begin(), elevation_map[A[nx][ny]].end(), make_pair(nx, ny)),
                            elevation_map[A[nx][ny]].end());
                        elevation_map[y].push_back({nx, ny});
                    }
                }
            }
        }

        cout << total_area << "\n";
    }

    return 0;
}
