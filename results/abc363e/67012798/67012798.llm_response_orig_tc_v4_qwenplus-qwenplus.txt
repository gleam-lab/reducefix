```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            cin >> A[i][j];
        }
    }

    // Initialize island as all land
    vector<vector<bool>> island(H, vector<bool>(W, true));

    // Prepare queues for each year
    vector<queue<pair<int, int>>> q(Y + 2); // up to Y+1

    // Mark borders and enqueue accordingly
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1)
            {
                q[A[i][j]].push({i, j});
                island[i][j] = false;
            }
        }
    }

    int totalArea = H * W;

    // Process each year
    for (int year = 1; year <= Y; ++year)
    {
        queue<pair<int, int>> currentQ = move(q[year]);
        while (!currentQ.empty())
        {
            auto [x, y] = currentQ.front();
            currentQ.pop();
            totalArea--;

            for (int d = 0; d < 4; ++d)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && ny >= 0 && nx < H && ny < W && island[nx][ny])
                {
                    if (A[nx][ny] <= year)
                    {
                        island[nx][ny] = false;
                        currentQ.push({nx, ny}); // Continue BFS from this new sinking cell
                    }
                    else
                    {
                        if (A[nx][ny] <= Y)
                            q[A[nx][ny]].push({nx, ny});
                        island[nx][ny] = false; // This cell will sink in the future
                    }
                }
            }
        }

        cout << totalArea << "\n";
    }

    return 0;
}
```
