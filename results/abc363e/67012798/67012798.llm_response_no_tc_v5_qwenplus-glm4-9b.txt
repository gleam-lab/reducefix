#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> A[i][j];
        }
    }

    // Initialize the queue with the boundary cells that are at sea level
    queue<pair<int, int>> q;
    for (int i = 0; i < H; i++) {
        q.push({i, 0});
        q.push({i, W - 1});
        if (A[i][0] == A[i][W - 1]) {
            q.push({0, i});
            q.push({H - 1, i});
        }
    }

    // Initial land area
    int land_area = H * W;

    for (int year = 0; year < Y; year++) {
        int next_level = A[q.front().first][q.front().second] + 1;
        vector<vector<bool>> visited(H, vector<bool>(W, false));
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();

            // Mark this cell as visited
            visited[x][y] = true;

            // Check all adjacent cells
            for (int d = 0; d < 4; d++) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx < 0 || nx >= H || ny < 0 || ny >= W) continue;
                if (A[nx][ny] <= next_level) {
                    // If the cell is below or at sea level, it will sink
                    if (!visited[nx][ny]) {
                        q.push({nx, ny});
                        visited[nx][ny] = true;
                    }
                    land_area--;
                }
            }
        }
        cout << land_area << endl;
    }

    return 0;
}

This corrected code uses a queue to simulate the process of the sea level rising year by year. It starts by initializing the queue with all boundary cells and cells that are at the same level as the boundary. For each year, it checks all cells that have sunk into the sea and updates the land area accordingly. It also keeps track of visited cells to avoid processing the same cell multiple times.
