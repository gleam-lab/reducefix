```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> submerged(H, vector<bool>(W, false));
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<tuple<int, int, int>>> pq;

    // Add all border cells to the priority queue
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (i == 0 || i == H-1 || j == 0 || j == W-1)
            {
                pq.push({A[i][j], i, j});
                submerged[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    vector<int> result(Y + 1, remaining);

    while (!pq.empty())
    {
        auto [elevation, x, y] = pq.top();
        pq.pop();

        // If this cell would be submerged at year = elevation
        if (elevation <= Y)
        {
            // Only count the first time we process a cell at or below sea level
            if (elevation > 0)
            {
                result[elevation] = --remaining;
            }
        }

        // Check neighbors
        for (int d = 0; d < 4; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx >= 0 && ny >= 0 && nx < H && ny < W && !submerged[nx][ny])
            {
                submerged[nx][ny] = true;
                pq.push({A[nx][ny], nx, ny});
            }
        }
    }

    // Fill in results for years where no change occurred
    for (int i = 1; i <= Y; i++)
    {
        if (i > 1 && result[i] == remaining) // This means no update happened at year i
        {
            result[i] = result[i-1];
        }
        cout << result[i] << '\n';
    }

    return 0;
}
```
