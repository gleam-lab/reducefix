```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    // Initialize island map and queue for each possible height
    vector<vector<bool>> island(H, vector<bool>(W, true));
    vector<queue<pair<int, int>>> q(Y + 2); // We use up to Y+1

    // Mark border cells as submerged and add them to the queue corresponding to their height
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1)
            {
                if (A[i][j] <= Y)
                {
                    q[A[i][j]].push({i, j});
                }
                island[i][j] = false;
            }
        }
    }

    int ans = H * W;

    // Keep track of visited coordinates to avoid processing same cell multiple times
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    for (int year = 1; year <= Y; year++)
    {
        // Process all cells that got added to this year's queue
        while (!q[year].empty())
        {
            auto [x, y] = q[year].front();
            q[year].pop();

            if (!visited[x][y])
            {
                visited[x][y] = true;
                ans--;

                for (int d = 0; d < 4; d++)
                {
                    int nx = x + dx[d];
                    int ny = y + dy[d];

                    if (nx >= 0 && ny >= 0 && nx < H && ny < W && !visited[nx][ny] && island[nx][ny])
                    {
                        if (A[nx][ny] <= year)
                        {
                            q[year].push({nx, ny});
                        }
                        else
                        {
                            q[A[nx][ny]].push({nx, ny});
                        }
                        island[nx][ny] = false;
                    }
                }
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
```
