```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            cin >> A[i][j];
        }
    }

    // Initialize island as all land
    vector<vector<bool>> island(H, vector<bool>(W, true));

    // Queue for each year's initial sinking
    vector<queue<pair<int, int>>> q(Y + 2); // up to Y+1

    // Mark borders and add corresponding cells to queue
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1)
            {
                int level = A[i][j];
                if (level <= Y)
                {
                    q[level].push({i, j});
                    island[i][j] = false;
                }
            }
        }
    }

    int ans = H * W;

    // Process each year
    for (int year = 1; year <= Y; ++year)
    {
        // BFS for this year's erosion
        queue<pair<int, int>> bfsQueue;

        while (!q[year].empty())
        {
            auto [x, y] = q[year].front();
            q[year].pop();
            bfsQueue.push({x, y});
            ans--;
        }

        // Expand from the newly added cells
        while (!bfsQueue.empty())
        {
            auto [x, y] = bfsQueue.front();
            bfsQueue.pop();

            for (int d = 0; d < 4; ++d)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx >= 0 && ny >= 0 && nx < H && ny < W && island[nx][ny])
                {
                    int next_level = A[nx][ny];
                    if (next_level <= Y && next_level > year)
                    {
                        q[next_level].push({nx, ny});
                    }
                    else if (next_level <= year)
                    {
                        island[nx][ny] = false;
                        ans--;
                        bfsQueue.push({nx, ny});
                    }
                }
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
```
