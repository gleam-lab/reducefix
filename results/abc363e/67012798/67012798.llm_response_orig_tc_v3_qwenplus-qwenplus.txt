```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            cin >> A[i][j];
        }
    }

    // Initialize the island map
    vector<vector<bool>> island(H, vector<bool>(W, true));

    // Create a queue for each possible year + buffer
    const int MAX_ELEVATION = 1e5 + 5;
    vector<queue<pair<int, int>>> q(MAX_ELEVATION); // up to 1e5

    // Process borders
    for (int i = 0; i < H; ++i)
    {
        for (int j = 0; j < W; ++j)
        {
            if ((i == 0 || i == H - 1 || j == 0 || j == W - 1))
            {
                int level = A[i][j];
                q[level].push({i, j});
                island[i][j] = false;
            }
        }
    }

    int totalArea = H * W;

    // Prepare an array to store area at each year
    vector<int> yearly_area(Y + 2, totalArea);

    // Process from lowest elevation to highest
    for (int year = 1; year <= Y; ++year)
    {
        // We process all cells with elevation <= year that have been added to the queue
        while (!q[year].empty())
        {
            totalArea--;
            auto [x, y] = q[year].front();
            q[year].pop();

            for (int d = 0; d < 4; ++d)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= H || ny >= W)
                    continue;

                if (island[nx][ny])
                {
                    int nextYear = A[nx][ny];
                    if (nextYear <= year)
                    {
                        q[year].push({nx, ny}); // flood now
                    }
                    else
                    {
                        q[nextYear].push({nx, ny}); // flood later
                    }
                    island[nx][ny] = false;
                }
            }
        }

        yearly_area[year] = totalArea;
    }

    // Output results
    for (int i = 1; i <= Y; ++i)
    {
        cout << yearly_area[i] << "\n";
    }

    return 0;
}
```
