```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> island(H, vector<bool>(W, true));
    // Use a single queue and process by sea level
    vector<vector<pair<int, int>>> events(100001); // events[sea_level] = list of cells to sink at that level
    
    // Add all boundary cells to their respective elevation events
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (i == 0 || i == H-1 || j == 0 || j == W-1)
            {
                events[A[i][j]].push_back({i, j});
                island[i][j] = false;
            }
        }
    }

    int ans = H * W;
    queue<pair<int, int>> q;

    for (int year = 1; year <= Y; year++)
    {
        // Add all cells with elevation <= current sea level that are on the boundary
        for (auto& cell : events[year])
        {
            if (island[cell.first][cell.second])
            {
                island[cell.first][cell.second] = false;
                q.push(cell);
                ans--;
            }
        }

        // Flood fill from all cells that sank at this sea level
        while (!q.empty())
        {
            auto [x, y] = q.front();
            q.pop();

            for (int d = 0; d < 4; d++)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= H || ny >= W)
                    continue;

                if (island[nx][ny] && A[nx][ny] <= year)
                {
                    island[nx][ny] = false;
                    q.push({nx, ny});
                    ans--;
                }
            }
        }

        cout << ans << '\n';
    }

    return 0;
}
```
