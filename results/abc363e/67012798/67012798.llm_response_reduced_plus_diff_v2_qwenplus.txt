#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    // Instead of processing from edges, we process by elevation in reverse order
    vector<tuple<int, int, int>> cells;
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cells.emplace_back(A[i][j], i, j);
        }
    }
    
    // Sort by elevation in descending order
    sort(cells.rbegin(), cells.rend());
    
    vector<vector<bool>> island(H, vector<bool>(W, true));
    int remaining = H * W;
    
    // Process each year
    int idx = 0;
    for (int year = 1; year <= Y; year++)
    {
        // Add all cells with elevation <= year that haven't been processed yet
        while (idx < cells.size() && get<0>(cells[idx]) <= year)
        {
            auto [elev, x, y] = cells[idx];
            if (island[x][y])
            {
                queue<pair<int, int>> q;
                q.push({x, y});
                island[x][y] = false;
                remaining--;
                
                // Flood fill to connect adjacent cells with elevation <= year
                while (!q.empty())
                {
                    auto [cx, cy] = q.front();
                    q.pop();
                    
                    for (int d = 0; d < 4; d++)
                    {
                        int nx = cx + dx[d];
                        int ny = cy + dy[d];
                        
                        if (nx >= 0 && ny >= 0 && nx < H && ny < W && island[nx][ny] && A[nx][ny] <= year)
                        {
                            island[nx][ny] = false;
                            q.push({nx, ny});
                            remaining--;
                        }
                    }
                }
            }
            idx++;
        }
        
        cout << remaining << '\n';
    }

    return 0;
}
