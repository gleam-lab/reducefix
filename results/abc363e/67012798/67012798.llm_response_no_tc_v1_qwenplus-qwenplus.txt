```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    // This will store the year when each cell gets submerged
    vector<vector<int>> submerge_year(H, vector<int>(W, -1));

    queue<pair<int, int>> q;

    // Initialize border cells
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1)
            {
                submerge_year[i][j] = A[i][j];
                q.emplace(i, j);
            }
        }
    }

    // BFS to simulate flooding process
    while (!q.empty())
    {
        auto [x, y] = q.front();
        q.pop();

        for (int d = 0; d < 4; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx < 0 || ny < 0 || nx >= H || ny >= W)
                continue;

            if (submerge_year[nx][ny] == -1)
            {
                // The cell gets submerged at the earliest level it becomes reachable
                submerge_year[nx][ny] = max(A[nx][ny], submerge_year[x][y]);
                q.emplace(nx, ny);
            }
        }
    }

    // Count how many cells remain unsubmerged at each year
    vector<int> remaining(Y + 2, H * W); // Start with all cells above water

    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (submerge_year[i][j] != -1 && submerge_year[i][j] <= Y)
            {
                remaining[submerge_year[i][j]]--;
            }
        }
    }

    // Accumulate the decrements to get the remaining area at each year
    for (int i = 1; i <= Y; i++)
    {
        remaining[i] = min(remaining[i], remaining[i - 1]);
        cout << remaining[i] << "\n";
    }

    return 0;
}
```
