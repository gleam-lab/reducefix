```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> island(H, vector<bool>(W, true));
    vector<queue<pair<int, int>>> q(100005); // Maximum sea level rise is 10^5

    // Initialize with boundary cells that will sink at their respective elevations
    for (int i = 0; i < H; i++)
    {
        for (int j = 0; j < W; j++)
        {
            if (i == 0 || i == H-1 || j == 0 || j == W-1)
            {
                q[A[i][j]].push({i, j});
                island[i][j] = false;
            }
        }
    }

    int ans = H * W;

    for (int year = 1; year <= Y; year++)
    {
        // Process all cells that sink at this sea level
        while (!q[year].empty())
        {
            auto [x, y] = q[year].front();
            q[year].pop();
            ans--;

            // Check neighbors
            for (int d = 0; d < 4; d++)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= H || ny >= W)
                    continue;

                // If neighbor is still above water and its elevation is <= current sea level
                if (island[nx][ny] && A[nx][ny] <= year)
                {
                    island[nx][ny] = false;
                    q[year].push({nx, ny});
                }
            }
        }

        cout << ans << '\n';
    }

    return 0;
}
```
