#include <bits/stdc++.h>

using namespace std;

constexpr int dx[] = {-1, 1, 0, 0};
constexpr int dy[] = {0, 0, -1, 1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> A[i][j];
        }
    }

    // We use a map to keep track of the elevation and the corresponding 2D index in the queue
    map<int, vector<pair<int, int>>> elevationMap;
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            if ((i == 0 || i == H - 1 || j == 0 || j == W - 1) || A[i][j] == 0) {
                elevationMap[0].push_back({i, j});
            } else {
                elevationMap[A[i][j]].push_back({i, j});
            }
        }
    }

    vector<queue<pair<int, int>>> q(Y + 1, queue<pair<int, int>>());
    for (auto& p : elevationMap) {
        for (auto& idx : p.second) {
            q[p.first].push({idx.first, idx.second});
        }
    }

    vector<int> remainingArea(Y + 1, H * W);
    for (int year = 1; year <= Y; year++) {
        int nextYear = min(year + 1, Y + 1);
        while (!q[year].empty()) {
            auto [x, y] = q[year].front();
            q[year].pop();
            for (int d = 0; d < 4; d++) {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx < 0 || ny < 0 || nx >= H || ny >= W || A[nx][ny] <= year) {
                    continue;
                }
                A[nx][ny]--;
                if (A[nx][ny] > 0) {
                    q[nextYear].push({nx, ny});
                }
            }
        }
        int remaining = 0;
        for (auto& p : elevationMap) {
            for (auto& idx : p.second) {
                remaining++;
            }
        }
        remainingArea[year] = remaining;
    }

    for (int i = 1; i <= Y; i++) {
        cout << remainingArea[i] << '\n';
    }

    return 0;
}
