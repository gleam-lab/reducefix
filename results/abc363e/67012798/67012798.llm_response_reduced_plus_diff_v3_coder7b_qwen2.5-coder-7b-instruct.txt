```cpp
#include <bits/stdc++.h>
using namespace std;

const int dx[] = {-1, 1, 0, 0};
const int dy[] = {0, 0, -1, 1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> visited(H, vector<bool>(W, false));
    vector<queue<pair<int, int>>> q(Y + 1);

    // Initialize queues based on initial elevations
    for(int i = 0; i < H; ++i) {
        for(int j = 0; j < W; ++j) {
            if((i == 0 || i == H - 1 || j == 0 || j == W - 1)) {
                q[A[i][j]].push({i, j});
                visited[i][j] = true;
            }
        }
    }

    long long total_area = H * W;

    for(int t = 1; t <= Y; ++t) {
        int sz = q[t].size();
        for(int k = 0; k < sz; ++k) {
            auto [x, y] = q[t].front();
            q[t].pop();

            --total_area;

            for(int d = 0; d < 4; ++d) {
                int nx = x + dx[d], ny = y + dy[d];
                if(nx < 0 || ny < 0 || nx >= H || ny >= W || visited[nx][ny]) continue;
                if(A[nx][ny] <= t) {
                    q[t].push({nx, ny});
                    visited[nx][ny] = true;
                }
            }
        }
        cout << total_area << '\n';
    }

    return 0;
}
```
