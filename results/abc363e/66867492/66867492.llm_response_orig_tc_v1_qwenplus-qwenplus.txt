#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using pii = pair<int, int>;
#define pb push_back
#define eb emplace_back
#define el '\n'

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // We will use a min-heap priority queue to process cells in order of increasing height
    using T = tuple<int, int, int>;  // (height, row, col)
    priority_queue<T, vector<T>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the border cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    int total_cells = H * W;
    vector<int> result(Y + 2);  // result[i] is the area after i years

    // Process from year 0 to Y
    int year = 0;
    while (!pq.empty()) {
        auto [height, r, c] = pq.top();
        pq.pop();

        // If the current height can be submerged this year
        while (year < height) {
            result[year] = total_cells;
            ++year;
            if (year > Y) break;
        }
        if (year > Y) break;

        --total_cells;

        const int dr[] = {-1, 0, 1, 0};
        const int dc[] = {0, 1, 0, -1};

        for (int k = 0; k < 4; ++k) {
            int nr = r + dr[k];
            int nc = c + dc[k];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W && !visited[nr][nc]) {
                visited[nr][nc] = true;
                pq.emplace(A[nr][nc], nr, nc);
            }
        }
    }

    // Fill remaining years with whatever remains
    while (year <= Y) {
        result[year] = total_cells;
        ++year;
    }

    // Output results for each year
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << el;
    }

    return 0;
}
