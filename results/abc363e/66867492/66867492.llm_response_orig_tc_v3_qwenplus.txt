#include <bits/stdc++.h>

using vi = vc<int>; using vvi = vc<vi>;
using Pi = pair<int, int>;

template<typename T> struct cmp { 
    bool operator()(const T& l, const T& r) { 
        return l[0] > r[0]; 
    } 
};
template<typename T> using pq_c = priority_queue<T, vc<T>, cmp<T>>;

#define pb push_back
#define em emplace
#define eb emplace_back
#define pob pop_back

template<typename T> inline bool chmin(T &a, T b){if(a > b){a = b; return true;}return false;}
template<typename T> istream& operator>>(istream& i, vc<T>& v) { rep(j, size(v))i >> v[j]; return i; }

int main(){
    int H,W,Y; 
    cin >> H >> W >> Y;
    vvi A(H, vi(W)), used(H, vi(W, 0)); 
    cin >> A;
    
    pq_c<vi> pq;
    
    // Add all border cells to the priority queue
    rep(i, H) {
        pq.push({A[i][0], i, 0});
        pq.push({A[i][W-1], i, W-1});
        used[i][0] = 1;
        used[i][W-1] = 1;
    }
    rep3(i, 1, W-1) {
        pq.push({A[0][i], 0, i});
        pq.push({A[H-1][i], H-1, i});
        used[0][i] = 1;
        used[H-1][i] = 1;
    }
    
    int ans = H * W;
    vi dx = {1,0,-1,0}, dy = {0,1,0,-1};
    
    // Process each year
    rep3(i, 1, Y+1) {
        // Remove all cells with elevation <= current sea level
        while(!pq.empty() && pq.top()[0] <= i) {
            auto cell = pq.top(); 
            pq.pop();
            
            // This cell is submerged
            --ans;
            int y = cell[1], x = cell[2];
            
            // Check all 4 adjacent cells
            rep(d, 4) {
                int ny = y + dy[d], nx = x + dx[d];
                if(ny < 0 || nx < 0 || ny >= H || nx >= W) continue;
                
                // If not visited, mark it and add to queue
                if(!used[ny][nx]) {
                    used[ny][nx] = 1;
                    pq.push({A[ny][nx], ny, nx});
                }
            }
        }
        cout << ans << el;
    }
}
