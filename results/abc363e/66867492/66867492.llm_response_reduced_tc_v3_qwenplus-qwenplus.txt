#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (n); ++i)
#define all(a) a.begin(), a.end()
typedef long long ll;
typedef pair<int, int> Pii;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    rep(i, H) rep(j, W) cin >> A[i][j];

    // Initialize used and add boundary cells to the priority queue
    vector<vector<bool>> used(H, vector<bool>(W, false));
    priority_queue<Pii, vector<Pii>, greater<Pii>> pq;

    // Add all border cells to the priority queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i * W + j);
                used[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    vector<int> dx = {0, 1, 0, -1};
    vector<int> dy = {1, 0, -1, 0};

    // Process each year
    for (int year = 1; year <= Y; ++year) {
        while (!pq.empty() && pq.top().first <= year) {
            int val = pq.top().first;
            int idx = pq.top().second;
            pq.pop();

            int i = idx / W;
            int j = idx % W;

            if (used[i][j]) {
                remaining--;
                for (int d = 0; d < 4; ++d) {
                    int ni = i + dx[d];
                    int nj = j + dy[d];
                    if (ni >= 0 && ni < H && nj >= 0 && nj < W && !used[ni][nj]) {
                        used[ni][nj] = true;
                        pq.emplace(A[ni][nj], ni * W + nj);
                    }
                }
            }
        }
        cout << remaining << endl;
    }
}
