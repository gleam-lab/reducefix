int main(){
    int H,W,Y; cin>>H>>W>>Y;
    vvi A(H, vi(W)), used(H, vi(W));
    pq_c<vi> pq;

    // Initialize the border cells (connected to sea)
    rep(i, H) {
        if (!used[i][0]) {
            used[i][0] = 1;
            pq.push({A[i][0], i, 0});
        }
        if (!used[i][W-1]) {
            used[i][W-1] = 1;
            pq.push({A[i][W-1], i, W-1});
        }
    }
    rep(j, W) {
        if (!used[0][j]) {
            used[0][j] = 1;
            pq.push({A[0][j], 0, j});
        }
        if (!used[H-1][j]) {
            used[H-1][j] = 1;
            pq.push({A[H-1][j], H-1, j});
        }
    }

    int total = H * W;
    vi dx = {-1, 0, 1, 0};
    vi dy = {0, 1, 0, -1};
    vi result(Y + 2); // To store results for year 1 to Y

    // Process up to Y years
    vi current(Y + 2, 0);
    int cnt = 0;

    while (!pq.empty()) {
        auto top = pq.top();
        int height = top[0];
        int x = top[1];
        int y = top[2];
        pq.pop();

        // If this cell submerges within the Y years
        if (height <= Y) {
            cnt++;
            current[height]++;
        }

        // Flood fill neighbors
        rep(d, 4) {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx < 0 || ny < 0 || nx >= H || ny >= W) continue;
            if (!used[nx][ny]) {
                used[nx][ny] = 1;
                pq.push({A[nx][ny], nx, ny});
            }
        }
    }

    // Compute prefix sum of submerged cells
    vi submerged(Y + 2, 0);
    for (int i = 1; i <= Y; ++i) {
        submerged[i] = submerged[i - 1] + current[i];
        result[i] = total - submerged[i];
        cout << result[i] << endl;
    }
}
