#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using P = pair<int, int>;

struct Init {
    Init() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        cout << fixed << setprecision(12);
    }
} init;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Mark boundary cells
    vector<vector<bool>> is_boundary(H, vector<bool>(W, false));
    for (int i = 0; i < H; ++i) {
        is_boundary[i][0] = true;
        is_boundary[i][W - 1] = true;
    }
    for (int j = 0; j < W; ++j) {
        is_boundary[0][j] = true;
        is_boundary[H - 1][j] = true;
    }

    // Prepare data structure for processing
    using Event = tuple<int, int, int>;
    priority_queue<Event, vector<Event>, greater<Event>> pq;

    vector<vector<bool>> visited(H, vector<bool>(W, false));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (is_boundary[i][j]) {
                pq.emplace(A[i][j], i, j);
                visited[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    int dx[] = {0, 1, 0, -1};
    int dy[] = {-1, 0, 1, 0};

    vector<int> result(Y + 2);

    // Process events in order of elevation
    while (!pq.empty()) {
        auto [elevation, x, y] = pq.top();
        pq.pop();

        // If we haven't already processed this year
        if (elevation <= Y && result[elevation] == 0) {
            result[elevation] = remaining;
        }

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
                remaining--;
            }
        }
    }

    // Fill in any missing values
    for (int i = 1; i <= Y; ++i) {
        if (result[i] == 0) {
            result[i] = result[i - 1];
        }
        cout << result[i] << "\n";
    }

    return 0;
}
