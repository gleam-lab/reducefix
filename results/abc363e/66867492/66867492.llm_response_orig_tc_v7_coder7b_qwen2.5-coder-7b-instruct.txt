To solve this problem, we need to simulate the rising sea levels and determine the area of the island that remains above sea level after each year. The key steps involve:

1. Reading the initial elevation of the island.
2. Using a priority queue to manage the sections of the island that are currently above the sea level.
3. For each year, processing the sections that will sink due to the rising sea level and updating the remaining area accordingly.

Here's the corrected and optimized version of your code:

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Init { Init() { ios::sync_with_stdio(false); cin.tie(nullptr); cout << fixed << setprecision(12); } } init;

int main() {
    int H, W, Y; cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    vector<vector<bool>> used(H, vector<bool>(W));

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
            if (A[i][j] > 0) {
                used[i][j] = true;
            }
        }
    }

    priority_queue<pair<int, pair<int, int>>, vector<pair<int, pair<int, int>>>, greater<>> pq;

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (used[i][j]) {
                pq.emplace(A[i][j], make_pair(i, j));
            }
        }
    }

    int ans = H * W;
    vector<int> dx = {1, 0, -1, 0}, dy = {0, 1, 0, -1};

    for (int y = 1; y <= Y; ++y) {
        while (!pq.empty() && pq.top().first <= y) {
            auto [height, pos] = pq.top(); pq.pop();
            --ans;
            for (int d = 0; d < 4; ++d) {
                int ny = pos.first + dy[d], nx = pos.second + dx[d];
                if (ny < 0 || nx < 0 || ny >= H || nx >= W) continue;
                if (!used[ny][nx] && A[ny][nx] <= y) {
                    used[ny][nx] = true;
                    pq.emplace(y, make_pair(ny, nx));
                }
            }
        }
        cout << ans << endl;
    }

    return 0;
}
```

### Explanation of Changes:

1. **Initialization**: We initialize the `used` array to keep track of which sections are currently above the sea level.
2. **Priority Queue**: We use a min-priority queue to always process the lowest elevation first, ensuring that sections that will sink due to the rising sea level are processed correctly.
3. **Processing Each Year**: For each year, we process all sections that will sink due to the current sea level and update the remaining area accordingly.
4. **Output**: We print the remaining area after each year.
