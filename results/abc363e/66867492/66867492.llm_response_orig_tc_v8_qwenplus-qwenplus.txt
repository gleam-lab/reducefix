#include <bits/stdc++.h>
using namespace std;

struct Init {
    Init() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        cout << fixed << setprecision(12);
    }
} init;

using i64 = int64_t;
using vi = vector<int>;
using vvi = vector<vi>;
using Pi = pair<int, int>;

template<typename T> inline bool chmin(T &a, T b) { if (a > b) { a = b; return true; } return false; }
template<typename T> inline bool chmax(T &a, T b) { if (a < b) { a = b; return true; } return false; }

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vvi A(H, vi(W));
    for (auto &row : A) {
        for (auto &x : row) {
            cin >> x;
        }
    }

    // Initialize used matrix and priority queue
    vvi used(H, vi(W, 0));
    priority_queue<Pi, vector<Pi>, greater<Pi>> pq;

    // Enqueue all border cells
    for (int i = 0; i < H; ++i) {
        for (int j : {0, W - 1}) {
            if (!used[i][j]) {
                used[i][j] = 1;
                pq.emplace(A[i][j], i * W + j);
            }
        }
    }

    for (int j = 1; j < W - 1; ++j) {
        for (int i : {0, H - 1}) {
            if (!used[i][j]) {
                used[i][j] = 1;
                pq.emplace(A[i][j], i * W + j);
            }
        }
    }

    int ans = H * W;
    vi dx = {1, 0, -1, 0}, dy = {0, 1, 0, -1};

    vi results(Y);

    for (int year = 1; year <= Y; ++year) {
        while (!pq.empty() && pq.top().first <= year) {
            auto [elevation, code] = pq.top();
            pq.pop();
            --ans;

            int r = code / W;
            int c = code % W;

            for (int d = 0; d < 4; ++d) {
                int ny = r + dy[d];
                int nx = c + dx[d];

                if (ny >= 0 && nx >= 0 && ny < H && nx < W && !used[ny][nx]) {
                    used[ny][nx] = 1;
                    pq.emplace(A[ny][nx], ny * W + nx);
                }
            }
        }
        results[year - 1] = ans;
    }

    for (int res : results) {
        cout << res << '\n';
    }
}
