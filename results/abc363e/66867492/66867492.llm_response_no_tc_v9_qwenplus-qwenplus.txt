#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;
using pii = pair<int, int>;

struct Init {
    Init() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        cout << fixed << setprecision(12);
    }
} init;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (auto& row : A) {
        for (auto& x : row) {
            cin >> x;
        }
    }

    // Min-heap priority queue to process cells in increasing order of elevation
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the border cells (first and last rows/columns)
    for (int i = 0; i < H; ++i) {
        for (int j : {0, W - 1}) {
            if (!visited[i][j]) {
                pq.emplace(A[i][j], i * W + j);
                visited[i][j] = true;
            }
        }
    }

    for (int j = 1; j < W - 1; ++j) {
        for (int i : {0, H - 1}) {
            if (!visited[i][j]) {
                pq.emplace(A[i][j], i * W + j);
                visited[i][j] = true;
            }
        }
    }

    int remaining = H * W;
    vector<int> dx = {1, 0, -1, 0};
    vector<int> dy = {0, 1, 0, -1};

    // We'll store the year each cell gets submerged
    vector<vector<int>> submerge_year(H, vector<int>(W, -1));
    int current_year = 0;

    while (!pq.empty()) {
        auto [elevation, pos] = pq.top();
        pq.pop();

        int i = pos / W;
        int j = pos % W;

        current_year = elevation;

        for (int d = 0; d < 4; ++d) {
            int ni = i + dx[d];
            int nj = j + dy[d];

            if (ni >= 0 && ni < H && nj >= 0 && nj < W && !visited[ni][nj]) {
                visited[ni][nj] = true;
                pq.emplace(A[ni][nj], ni * W + nj);
            }
        }

        // This cell is now submerged
        --remaining;

        // Store how many years it takes for this cell to be submerged
        submerge_year[i][j] = elevation;
    }

    // For each year from 1 to Y, count how many cells are still above water
    vector<int> counts(Y + 2);  // counts[y] = number of cells still above sea level at year y

    // Start with all cells above sea level
    counts[0] = H * W;

    // For each cell, subtract 1 from counts[submerge_year[i][j] + 1]
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (submerge_year[i][j] != -1 && submerge_year[i][j] < Y) {
                counts[submerge_year[i][j] + 1]--;
            }
        }
    }

    // Build cumulative answer
    for (int i = 1; i <= Y; ++i) {
        counts[i] += counts[i - 1];
        cout << counts[i] << '\n';
    }

    return 0;
}
