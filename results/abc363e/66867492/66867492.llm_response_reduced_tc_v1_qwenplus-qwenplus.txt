#include <bits/stdc++.h>
using namespace std;
using i64 = int64_t;

struct Init {
    Init() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        cout.precision(12);
    }
} init;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // We will use a visited matrix to track if a cell has sunk
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Priority queue to process cells in order of their elevation
    priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;

    // Push all the border cells into the priority queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.push({A[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }

    // Total land area
    int remaining = H * W;

    // Directions for BFS
    vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    // To store result for each year
    vector<int> res(Y);

    for (int year = 1; year <= Y; ++year) {
        // Process all cells in the queue with elevation <= current sea level
        while (!pq.empty() && pq.top()[0] <= year) {
            auto top = pq.top();
            pq.pop();
            --remaining;

            int x = top[1], y = top[2];

            for (auto [dx, dy] : dirs) {
                int nx = x + dx, ny = y + dy;
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    visited[nx][ny] = true;
                    pq.push({A[nx][ny], nx, ny});
                }
            }
        }

        res[year - 1] = remaining;
    }

    for (int r : res)
        cout << r << '\n';

    return 0;
}
