        if (!visited[i][0]) {
            visited[i][0] = 1;
            pq.push({A[i][0], i, 0});
        }

            visited[i][W-1] = 1;
            pq.push({A[i][W-1], i, W-1});
        }

            visited[0][j] = 1;
            pq.push({A[0][j], 0, j});

            visited[H-1][j] = 1;
            pq.push({A[H-1][j], H-1, j});
        }
    }

    vi result(Y + 1);
    vi dx = {-1, 0, 1, 0};
    vi dy = {0, 1, 0, -1};

    for (int t = 0; !pq.empty();) {
        vi curr = pq.top(); pq.pop();
        int h = curr[1], w = curr[2];
        
        while (t < curr[0] && t < Y) {
            result[++t] = remaining;
        }

        remaining--;

        rep(d, 4) {
            int nh = h + dx[d];
            int nw = w + dy[d];
            if (nh >= 0 && nh < H && nw >= 0 && nw < W && !visited[nh][nw]) {
                visited[nh][nw] = 1;
                pq.push({A[nh][nw], nh, nw});
            }
        }
    }

    // Fill in any missing results at the end
    while (++t <= Y) {
        result[t] = remaining;
    }

    // Output results
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << '\n';
    }
}
