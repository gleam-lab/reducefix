import sys
import heapq

def main():
    H, W, Y = map(int, sys.stdin.readline().split())
    
    # Read elevation data
    A = []
    for _ in range(H):
        row = list(map(int, sys.stdin.readline().split()))
        A.append(row)
    
    # Initialize result array
    result = [H * W] * (Y + 1)  # result[i] = area after i years
    
    # Use Dijkstra-like algorithm with min-heap
    # We'll process cells in order of increasing elevation threshold
    heap = []
    visited = [[False] * W for _ in range(H)]
    
    # Add all border cells to the heap
    for i in range(H):
        for j in range(W):
            if i == 0 or i == H - 1 or j == 0 or j == W - 1:
                heapq.heappush(heap, (A[i][j], i, j))
                visited[i][j] = True
    
    # Process cells in order of elevation
    while heap:
        elev, i, j = heapq.heappop(heap)
        
        # This cell gets submerged at year = elev (if elev <= Y)
        if elev <= Y:
            result[elev] -= 1
        
        # Check neighbors
        for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            ni, nj = i + di, j + dj
            if 0 <= ni < H and 0 <= nj < W and not visited[ni][nj]:
                visited[ni][nj] = True
                # The water level needed to reach this cell is max(current_path_threshold, cell_elevation)
                # But since we're using a min-heap, we use the maximum of the two
                new_threshold = max(elev, A[ni][nj])
                heapq.heappush(heap, (new_threshold, ni, nj))
    
    # Convert from "remaining at exactly year i" to "remaining after year i"
    # We need cumulative counts: what remains after year i is total minus those submerged by year i
    remaining = H * W
    for year in range(1, Y + 1):
        # Subtract cells that get submerged at this year
        if year < len(result):
            remaining -= (H * W - result[year]) - (H * W - result[year-1]) if year > 0 else (H * W - result[year])
        
        # Recompute using prefix sum of submersions
        pass
    
    # Actually, let's rebuild result correctly
    # First, create an array of how many cells get submerged at each year
    submerged_at_year = [0] * (Y + 2)
    
    # Reset and recompute
    heap = []
    visited = [[False] * W for _ in range(H)]
    
    for i in range(H):
        for j in range(W):
            if i == 0 or i == H - 1 or j == 0 or j == W - 1:
                heapq.heappush(heap, (A[i][j], i, j))
                visited[i][j] = True
    
    while heap:
        elev, i, j = heapq.heappop(heap)
        year_submerged = min(elev, Y + 1)
        submerged_at_year[year_submerged] += 1
        
        for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            ni, nj = i + di, j + dj
            if 0 <= ni < H and 0 <= nj < W and not visited[ni][nj]:
                visited[ni][nj] = True
                new_threshold = max(elev, A[ni][nj])
                heapq.heappush(heap, (new_threshold, ni, nj))
    
    # Calculate remaining area after each year
    remaining = H * W
    for year in range(1, Y + 1):
        remaining -= submerged_at_year[year]
        print(remaining)

if __name__ == "__main__":
    main()
