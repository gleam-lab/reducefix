import sys
import heapq

def main():
    H, W, Y = map(int, sys.stdin.readline().split())
    
    # Read elevation data
    A = []
    for _ in range(H):
        row = list(map(int, sys.stdin.readline().split()))
        A.append(row)
    
    # Initialize result array for each year
    result = [0] * (Y + 1)
    total_cells = H * W
    
    # Use a priority queue to simulate rising sea level
    # Start from the boundary cells as they are first to be submerged
    heap = []
    visited = [[False] * W for _ in range(H)]
    
    # Add all boundary cells to the heap
    for i in range(H):
        for j in range(W):
            if i == 0 or i == H - 1 or j == 0 or j == W - 1:
                heapq.heappush(heap, (A[i][j], i, j))
                visited[i][j] = True
    
    # Count of remaining cells above water
    remaining = total_cells
    
    # Process cells in order of increasing elevation
    for year in range(1, Y + 1):
        # While there are cells with elevation <= current sea level (year)
        while heap and heap[0][0] <= year:
            elev, i, j = heapq.heappop(heap)
            remaining -= 1
            
            # Check all 4 adjacent cells
            for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                ni, nj = i + di, j + dj
                if 0 <= ni < H and 0 <= nj < W and not visited[ni][nj]:
                    visited[ni][nj] = True
                    heapq.heappush(heap, (A[ni][nj], ni, nj))
        
        result[year] = remaining
    
    # Output results for each year
    for i in range(1, Y + 1):
        print(result[i])

if __name__ == "__main__":
    main()
