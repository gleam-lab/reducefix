#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define all(x) begin(x), end(x)
using ll = long long;
using P = pair<int, int>;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int h, w, y;
    cin >> h >> w >> y;
    vector<vector<int>> grid(h, vector<int>(w));
    rep(i, h) rep(j, w) cin >> grid[i][j];

    vector<vector<bool>> visited(h, vector<bool>(w, false));

    auto bfs = [&](int level) {
        queue<P> q;
        rep(i, h) rep(j, w) if (!visited[i][j] && grid[i][j] <= level) {
            visited[i][j] = true;
            q.emplace(i, j);
        }

        int count = 0;
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();
            count++;

            static constexpr array<pair<int, int>, 4> dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
            for (auto& [dx, dy] : dir) {
                int nx = x + dx, ny = y + dy;
                if (nx < 0 || nx >= h || ny < 0 || ny >= w) continue;
                if (visited[nx][ny]) continue;
                if (grid[nx][ny] <= level) {
                    visited[nx][ny] = true;
                    q.emplace(nx, ny);
                }
            }
        }

        return count;
    };

    vector<int> areas(y);
    rep(level, y) areas[level] = bfs(level + 1);

    rep(i, y) cout << areas[i] << '\n';

    return 0;
}
