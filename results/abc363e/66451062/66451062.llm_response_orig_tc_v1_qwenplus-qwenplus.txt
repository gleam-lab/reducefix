#include <bits/stdc++.h>
using namespace std;

#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
#define inf 0x3f3f3f3f3f3f
const int MOD = 1e9 + 7;
const int MAXN = 1005;

int H, W, Y;
int grid[MAXN][MAXN];
bool vis[MAXN][MAXN];

struct Node {
    int x, y, val;
    bool operator<(const Node &a) const {
        return val > a.val;
    }
};

priority_queue<Node> pq;
int totalCells;
int res[MAXN];

signed main() {
    close;
    cin >> H >> W >> Y;
    totalCells = H * W;
    
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> grid[i][j];
        }
    }

    memset(vis, false, sizeof(vis));

    // Initialize border cells
    for (int i = 0; i < H; i++) {
        if (!vis[i][0]) {
            vis[i][0] = true;
            pq.push({i, 0, grid[i][0]});
        }
        if (!vis[i][W - 1]) {
            vis[i][W - 1] = true;
            pq.push({i, W - 1, grid[i][W - 1]});
        }
    }

    for (int j = 1; j < W - 1; j++) {
        if (!vis[0][j]) {
            vis[0][j] = true;
            pq.push({0, j, grid[0][j]});
        }
        if (!vis[H - 1][j]) {
            vis[H - 1][j] = true;
            pq.push({H - 1, j, grid[H - 1][j]});
        }
    }

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};

    int remaining = totalCells;
    int currentYear = 1;

    while (currentYear <= Y && !pq.empty()) {
        while (!pq.empty() && pq.top().val <= currentYear) {
            Node cur = pq.top(); pq.pop();
            remaining--;

            for (int k = 0; k < 4; k++) {
                int nx = cur.x + dx[k];
                int ny = cur.y + dy[k];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        res[currentYear++] = remaining;
    }

    // In case some years are left with no change
    while (currentYear <= Y) {
        res[currentYear++] = remaining;
    }

    for (int i = 1; i <= Y; i++) {
        cout << res[i] << endl;
    }

    return 0;
}
