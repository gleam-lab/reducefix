#include <bits/stdc++.h>
using namespace std;
#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
#define MAXN 110
#define LIMIT 22
#define inf 0x3f3f3f3f3f3f
#define endl '\n'
const double eps = 1e-7;
int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};
const int MOD = 1e9 + 7;

struct Node {
    int x, y, val;
    bool operator<(const Node& a) const {
        return val > a.val; // Use '>' to prioritize higher elevations
    }
};

signed main() {
    close;
    int h, w; cin >> h >> w;
    int y; cin >> y;
    vector<vector<int>> grid(h, vector<int>(w));
    
    // Read the grid
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> grid[i][j];
        }
    }
    
    // Initialize visited array
    vector<vector<bool>> vis(h, vector<bool>(w, false));
    
    // Initialize priority queue (min-heap)
    priority_queue<Node, vector<Node>, greater<Node>> q;
    
    // Push edges of the grid into the queue
    for (int i = 0; i < h; i++) {
        q.push({i, 0, grid[i][0]});
        q.push({i, w - 1, grid[i][w - 1]});
        q.push({i, w - 2, grid[i][w - 2]});
        q.push({i, 1, grid[i][1]});
        vis[i][0] = vis[i][w - 1] = true;
        vis[i][w - 2] = vis[i][1] = true;
    }
    
    for (int j = 1; j < w - 1; j++) {
        q.push({0, j, grid[0][j]});
        q.push({h - 1, j, grid[h - 1][j]});
        q.push({0, j + 1, grid[0][j + 1]});
        q.push({h - 1, j + 1, grid[h - 1][j + 1]});
        vis[0][j] = vis[0][j + 1] = true;
        vis[h - 1][j] = vis[h - 1][j + 1] = true;
    }
    
    // Process the years
    for (int level = 1; level <= y; level++) {
        int remainingArea = 0;
        while (!q.empty() && q.top().val <= level) {
            Node cur = q.top(); q.pop();
            for (int i = 0; i < 4; i++) {
                int nx = cur.x + dx[i];
                int ny = cur.y + dy[i];
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !vis[nx][ny] && grid[nx][ny] > level) {
                    vis[nx][ny] = true;
                    remainingArea++;
                    q.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        cout << remainingArea << endl;
    }
    
    return 0;
}
