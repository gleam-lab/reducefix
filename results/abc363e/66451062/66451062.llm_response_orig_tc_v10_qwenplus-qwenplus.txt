#include <bits/stdc++.h>
using namespace std;

#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
#define MAXN 1005
#define dx array<int,4>{-1,1,0,0}
#define dy array<int,4>{0,0,-1,1}
const int MOD = 1e9 + 7;

struct Node {
    int x, y, val;
    bool operator < (const Node &a) const {
        return val > a.val;
    }
};

signed main() {
    close;
    int h, w; cin >> h >> w;
    int Y; cin >> Y;
    int grid[MAXN][MAXN];
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            cin >> grid[i][j];
        }
    }

    bool vis[MAXN][MAXN] = {};
    priority_queue<Node> q;

    // Push all boundary cells into the priority queue
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                q.push({i, j, grid[i][j]});
                vis[i][j] = true;
            }
        }
    }

    int res = h * w;
    int year = 1;
    vector<int> ans(Y + 2, res);

    while (!q.empty() && year <= Y) {
        while (!q.empty() && q.top().val <= year) {
            Node cur = q.top(); q.pop();
            if (ans[year] == res) ans[year] = --res;
            else res--, ans[year] = res;
            for (int d = 0; d < 4; d++) {
                int nx = cur.x + dx[d];
                int ny = cur.y + dy[d];
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    q.push({nx, ny, grid[nx][ny]});
                }
            }
        }
        year++;
    }

    // Fill in any remaining years with the last computed value
    for (int i = 1; i <= Y; i++) {
        if (ans[i] == 0) ans[i] = ans[i - 1];
        cout << ans[i] << "\n";
    }

    return 0;
}
