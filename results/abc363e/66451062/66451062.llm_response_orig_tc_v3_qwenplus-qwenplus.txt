```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
int H, W, Y;

const int MAXN = 1005;
int grid[MAXN][MAXN];
bool visited[MAXN][MAXN];

struct Cell {
    int x, y, val;
    bool operator<(const Cell& other) const {
        return val > other.val;
    }
};

signed main() {
    close;
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> grid[i][j];
        }
    }

    priority_queue<Cell> pq;
    memset(visited, false, sizeof(visited));

    // Initialize the queue with all boundary cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                pq.push({i, j, grid[i][j]});
            }
        }
    }

    int totalArea = H * W;
    vector<int> result(Y + 2, 0);

    int year = 1;
    while (!pq.empty() && year <= Y) {
        while (!pq.empty() && pq.top().val <= year) {
            Cell cur = pq.top(); pq.pop();
            if (grid[cur.x][cur.y] > year) continue;
            if (!visited[cur.x][cur.y]) {
                visited[cur.x][cur.y] = true;
                totalArea--;
            }
            for (int d = 0; d < 4; ++d) {
                int nx = cur.x + (d == 1 ? 1 : (d == 2 ? -1 : 0));
                int ny = cur.y + (d == 3 ? 1 : (d == 0 ? -1 : 0));
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                    if (grid[nx][ny] <= year) {
                        visited[nx][ny] = true;
                        totalArea--;
                        pq.push({nx, ny, grid[nx][ny]});
                    }
                }
            }
        }
        result[year] = totalArea;
        year++;
    }

    // In case we didn't process all years due to early queue exhaustion
    while (year <= Y) {
        result[year] = totalArea;
        year++;
    }

    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << endl;
    }

    return 0;
}
```
