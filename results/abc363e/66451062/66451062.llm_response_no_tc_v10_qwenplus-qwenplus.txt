```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define pii pair<int, int>
#define fi first
#define se second
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define close ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
const int MOD = 1e9 + 7;
const int MAXN = 1005;
const int dx[] = {0, 0, 1, -1};
const int dy[] = {1, -1, 0, 0};

int H, W, Y;
int grid[MAXN][MAXN];
bool sunk[MAXN][MAXN];

// To store cells by elevation for processing in order
vector<vector<pii>> elevation_cells(100001);

signed main() {
    close;
    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> grid[i][j];
            elevation_cells[grid[i][j]].pb({i, j});
        }
    }

    // Initialize border as sunk
    priority_queue<pii, vector<pii>, function<bool(pii, pii)>> pq([](pii a, pii b) {
        return grid[a.fi][a.se] > grid[b.fi][b.se];
    });

    auto isBorder = [&](int x, int y) {
        return x == 0 || x == H - 1 || y == 0 || y == W - 1;
    };

    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (isBorder(i, j)) {
                sunk[i][j] = true;
                pq.push({i, j});
            }
        }
    }

    // Prepare array to store the number of cells submerged each year
    vector<int> destroyed(Y + 2, 0);

    while (!pq.empty()) {
        auto cur = pq.top(); pq.pop();
        int ci = cur.fi, cj = cur.se;
        int current_elevation = grid[ci][cj];

        if (current_elevation > Y)
            continue;

        for (int d = 0; d < 4; ++d) {
            int ni = ci + dx[d];
            int nj = cj + dy[d];
            if (ni >= 0 && ni < H && nj >= 0 && nj < W && !sunk[ni][nj]) {
                sunk[ni][nj] = true;
                if (grid[ni][nj] <= Y) {
                    destroyed[grid[ni][nj]]++;
                }
                pq.push({ni, nj});
            }
        }
    }

    // Calculate remaining area for each year
    int total = H * W;
    vector<int> result(Y + 1);
    for (int i = 1; i <= Y; ++i) {
        destroyed[i] += destroyed[i - 1];
        result[i] = total - destroyed[i];
    }

    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << endl;
    }

    return 0;
}
```
