```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Priority queue to process cells in increasing order of elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> sunk(H, vector<bool>(W, false));

    // Mark and push all boundary cells (connected to sea)
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                sunk[i][j] = true;
            }
        }
    }

    // To store the number of remaining land cells at each year
    vector<int> result(Y + 1, H * W);  // Initially full area

    // BFS with priority queue to process lowest elevations first
    while (!pq.empty()) {
        auto [height, x, y] = pq.top();
        pq.pop();

        // This cell sinks when sea level reaches 'height'
        if (height <= Y) {
            result[height]--;
        }

        for (auto [dx, dy] : d) {
            int nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !sunk[nx][ny]) {
                sunk[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }
    }

    // Cumulative reduction: fill in gaps and maintain non-increasing sequence
    int current = H * W;
    for (int i = 1; i <= Y; ++i) {
        current = min(current, result[i]);
        cout << current << "\n";
    }

    return 0;
}
```
