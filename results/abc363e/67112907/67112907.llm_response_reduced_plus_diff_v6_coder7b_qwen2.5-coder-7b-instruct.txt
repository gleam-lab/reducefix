```cpp
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>

using namespace std;

const int dx[] = {0, 0, 1, -1};
const int dy[] = {1, -1, 0, 0};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for(int i = 0; i < H; ++i){
        for(int j = 0; j < W; ++j){
            cin >> A[i][j];
        }
    }

    priority_queue<pair<int, pair<int, int>>, vector<pair<int, pair<int, int>>>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    for(int i = 0; i < H; ++i){
        pq.emplace(A[i][0], make_pair(i, 0)), visited[i][0] = true;
        pq.emplace(A[0][i], make_pair(0, i)), visited[0][i] = true;
    }
    for(int i = 1; i < W; ++i){
        pq.emplace(A[H-1][i], make_pair(H-1, i)), visited[H-1][i] = true;
    }
    for(int i = 1; i < H-1; ++i){
        pq.emplace(A[i][W-1], make_pair(i, W-1)), visited[i][W-1] = true;
    }

    int total_area = H * W;
    while(Y--){
        auto [height, pos] = pq.top();
        pq.pop();

        total_area -= 1;
        for(int k = 0; k < 4; ++k){
            int nx = pos.first + dx[k], ny = pos.second + dy[k];
            if(nx < 0 || nx >= H || ny < 0 || ny >= W || visited[nx][ny]){
                continue;
            }
            visited[nx][ny] = true;
            if(A[nx][ny] <= height){
                pq.emplace(A[nx][ny], make_pair(nx, ny));
            }
        }
        cout << total_area << '\n';
    }

    return 0;
}
```
