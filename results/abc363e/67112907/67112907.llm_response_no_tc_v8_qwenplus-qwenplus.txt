```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Priority queue to simulate water rising: min-heap by elevation
    using T = tuple<int, int, int>;
    priority_queue<T, vector<T>, greater<>> pq;
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Initialize the border (sea level starts at 0)
    auto add_to_pq = [&](int x, int y) {
        if (!vis[x][y]) {
            vis[x][y] = true;
            pq.emplace(A[x][y], x, y);
        }
    };

    // Add all edge cells
    for (int i = 0; i < H; ++i) {
        add_to_pq(i, 0);
        add_to_pq(i, W - 1);
    }
    for (int j = 1; j < W - 1; ++j) {
        add_to_pq(0, j);
        add_to_pq(H - 1, j);
    }

    int remaining = H * W;
    vector<int> result(Y + 1); // Store results for years 1..Y

    int current_year = 0;
    while (current_year < Y && !pq.empty()) {
        int h, x, y;
        tie(h, x, y) = pq.top();
        pq.pop();

        if (h <= current_year) {
            --remaining;
            for (int i = 0; i < 4; ++i) {
                int nx = x + d[i][0];
                int ny = y + d[i][1];
                if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                    vis[nx][ny] = true;
                    pq.emplace(A[nx][ny], nx, ny);
                }
            }
        } else {
            // We can't process this yet, wait for later year
            ++current_year;
            result[current_year] = remaining;
        }
    }

    // Fill in any missing years due to early termination
    while (current_year < Y) {
        ++current_year;
        result[current_year] = remaining;
    }

    // Output results
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << "\n";
    }

    return 0;
}
```
