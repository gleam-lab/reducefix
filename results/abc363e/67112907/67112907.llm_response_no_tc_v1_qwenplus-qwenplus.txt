```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // We'll process cells in order of their elevation using a min-heap (priority queue)
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Mark and add all border cells to the priority queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                vis[i][j] = true;
                pq.emplace(A[i][j], i, j);
            }
        }
    }

    // We'll track the number of destroyed cells per year
    vector<int> destroyed(Y + 2, 0);  // Using Y+2 to avoid index issues
    int remaining = H * W;

    while (!pq.empty()) {
        auto [height, x, y] = pq.top();
        pq.pop();

        // This cell gets submerged at 'height' years
        if (height <= Y) {
            destroyed[height]++;
        }

        for (auto [dx, dy] : d) {
            int nx = x + dx;
            int ny = y + dy;
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                vis[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }
    }

    // For each year, accumulate the destroyed cells and output the remaining area
    for (int i = 1; i <= Y; ++i) {
        destroyed[i] += destroyed[i - 1];
        cout << (H * W - destroyed[i]) << "\n";
    }

    return 0;
}
```
