#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Cell {
    int height, x, y;
};

struct Compare {
    bool operator()(const Cell& a, const Cell& b) const {
        return a.height < b.height;
    }
};

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    
    vector<vector<int>> sea(H, vector<int>(W));
    vector<vector<bool>> visited(H, vector<bool>(W, false));
    
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> sea[i][j];
        }
    }
    
    priority_queue<Cell, vector<Cell>, Compare> pq;
    
    // Initialize the priority queue and visited vector
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            if ((i == 0 || i == H - 1 || j == 0 || j == W - 1) || sea[i][j] == 0) {
                pq.push({sea[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }

    vector<int> remainingArea(Y);
    remainingArea[0] = H * W;

    for (int year = 1; year <= Y; year++) {
        int sinked = 0;
        while (!pq.empty()) {
            Cell top = pq.top();
            pq.pop();
            int h = top.height - year;
            if (h < 0) {
                sinked++;
                for (int dx = -1; dx <= 1; dx++) {
                    for (int dy = -1; dy <= 1; dy++) {
                        int nx = top.x + dx;
                        int ny = top.y + dy;
                        if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                            visited[nx][ny] = true;
                            pq.push({sea[nx][ny], nx, ny});
                        }
                    }
                }
            }
        }
        remainingArea[year] = remainingArea[year - 1] - sinked;
    }

    for (int i = 0; i < Y; i++) {
        cout << remainingArea[i] << '\n';
    }

    return 0;
}
