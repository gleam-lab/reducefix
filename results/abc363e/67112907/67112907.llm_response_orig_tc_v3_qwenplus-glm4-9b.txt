
This problem is a simulation problem that involves simulating the rising sea level on an island and calculating the area of the island that remains above sea level.

### Problem Analysis
The island is represented as a grid, where each cell has an elevation. Each year, the sea level rises by 1 unit. If a cell is at or below the sea level, it will be submerged. If a cell becomes submerged due to the rising sea level, all adjacent cells (up, down, left, right) that are at or below the current sea level will also be submerged. This process is recursive as each submerged cell can cause its adjacent cells to become submerged, which in turn can cause further cells to become submerged.

### Solution Strategy
1. **Initialization**: Read the input values and initialize the necessary data structures.
2. **Simulation**: For each year:
   - Determine which cells will be submerged based on the current sea level.
   - Recursively mark all cells that become submerged due to adjacent submerged cells.
   - Calculate the area of the island that remains above sea level.
3. **Output**: Output the remaining area for each year.

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

using P = pair<int, int>;
using Q = queue<P>;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;
    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    vector<vector<bool>> visited(H, vector<bool>(W, false));
    queue<P> q;

    // Initialize the queue with cells that are initially submerged
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (A[i][j] == 0) {
                visited[i][j] = true;
                q.push({i, j});
            }
        }
    }

    vector<int> ans(Y, 0);

    // Process the queue and simulate the rising sea level
    for (int year = 0; year < Y; ++year) {
        int remaining_area = 0;
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();

            // Check all four adjacent cells
            for (int dx = -1; dx <= 1; ++dx) {
                for (int dy = -1; dy <= 1; ++dy) {
                    int nx = x + dx, ny = y + dy;
                    if (nx < 0 || nx >= H || ny < 0 || ny >= W) continue;
                    if (visited[nx][ny]) continue;

                    // If the adjacent cell is submerged, mark it and add it to the queue
                    if (A[nx][ny] <= year) {
                        visited[nx][ny] = true;
                        q.push({nx, ny});
                    } else {
                        // Otherwise, it's part of the remaining area
                        remaining_area++;
                    }
                }
            }
        }
        ans[year] = remaining_area;
    }

    // Output the results for each year
    for (int i = 0; i < Y; ++i) {
        cout << ans[i] << '\n';
    }

    return 0;
}

### Explanation
1. **Initialization**: We read the input and initialize the `visited` matrix to keep track of submerged cells.
2. **Simulation**: We use a queue to process cells that are submerged. For each submerged cell, we check its neighbors. If a neighbor is not yet visited and its elevation is less than or equal to the current sea level (year), we mark it as visited and add it to the queue.
3. **Remaining Area Calculation**: For each year, we calculate the number of cells that are not submerged.
