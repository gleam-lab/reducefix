```cpp
#include <bits/stdc++.h>
using namespace std;

constexpr int dx[] = {0, 1, 0, -1};
constexpr int dy[] = {1, 0, -1, 0};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Initialize visited matrix
    vector<vector<bool>> vis(H, vector<bool>(W, false));

    // Priority queue to process cells by elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;

    // Mark and add boundary cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                vis[i][j] = true;
                pq.emplace(A[i][j], i, j);
            }
        }
    }

    int remaining = H * W;
    vector<int> result(Y);

    // Process each year
    for (int year = 1; year <= Y; ++year) {
        while (!pq.empty() && get<0>(pq.top()) <= year) {
            auto [h, x, y] = pq.top();
            pq.pop();
            if (!vis[x][y]) {
                --remaining;
                vis[x][y] = true;
                for (int d = 0; d < 4; ++d) {
                    int nx = x + dx[d];
                    int ny = y + dy[d];
                    if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                        pq.emplace(A[nx][ny], nx, ny);
                    }
                }
            }
        }
        result[year - 1] = remaining;
    }

    for (int ans : result) {
        cout << ans << '\n';
    }

    return 0;
}
```
