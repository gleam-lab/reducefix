```cpp
#include <bits/stdc++.h>
using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            cin >> A[i][j];
        }
    }

    // Visited matrix to mark cells submerged
    vector<vector<char>> vis(H, vector<char>(W, 0));

    // Priority queue to process cells in increasing order of elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;

    // Mark and enqueue all border cells
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                vis[i][j] = true;
            }
        }
    }

    int remaining = H * W;

    // Array to store the results for each year
    vector<int> result(Y + 1, 0);

    // We'll process the events in order of rising sea level
    int curr_year = 0;

    while (!pq.empty()) {
        auto [h, x, y] = pq.top();
        pq.pop();

        // If the current height exceeds the current year, advance the year
        while (curr_year < h) {
            result[++curr_year] = remaining;
        }

        if (A[x][y] <= curr_year && !vis[x][y]) {
            vis[x][y] = true;
            --remaining;
        }

        // Flood fill from this cell
        for (auto& [dx, dy] : d) {
            int nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !vis[nx][ny]) {
                pq.emplace(A[nx][ny], nx, ny);
                vis[nx][ny] = true;
                --remaining;
            }
        }
    }

    // Fill any remaining years with the final count
    while (curr_year < Y) {
        result[++curr_year] = remaining;
    }

    // Output the results for each year
    for (int i = 1; i <= Y; ++i) {
        cout << result[i] << '\n';
    }

    return 0;
}
```
