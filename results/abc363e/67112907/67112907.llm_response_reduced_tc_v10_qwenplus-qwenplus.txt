```cpp
#include <bits/stdc++.h>
using namespace std;

constexpr int dx[] = {0, 1, 0, -1};
constexpr int dy[] = {1, 0, -1, 0};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Add all border cells to the priority queue
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Mark borders and add to PQ
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                visited[i][j] = true;
                pq.emplace(A[i][j], i, j);
            }
        }
    }

    int remaining = H * W;
    vector<int> ans(Y + 1); // ans[0] unused or for initial state

    // Process in BFS-style with increasing elevation
    while (!pq.empty()) {
        auto [h, x, y] = pq.top();
        pq.pop();

        // This handles years up to h
        while (ans.size() <= h) {
            ans.push_back(remaining);
        }

        remaining--;

        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }
    }

    // Fill any missing years with previous value (if whole island still exists)
    while (ans.size() <= Y) {
        ans.push_back(ans.back());
    }

    // Output answers for each year
    for (int i = 1; i <= Y; ++i) {
        cout << ans[i] << '\n';
    }

    return 0;
}
```
