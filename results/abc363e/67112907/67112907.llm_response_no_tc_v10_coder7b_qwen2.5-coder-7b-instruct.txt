```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i)
        for (int j = 0; j < W; ++j)
            cin >> A[i][j];

    vector<vector<bool>> visited(H, vector<bool>(W, false));

    auto valid = [&](int x, int y) {
        return x >= 0 && x < H && y >= 0 && y < W;
    };

    auto bfs = [&](int level) {
        queue<pair<int, int>> q;
        for (int i = 0; i < H; ++i)
            for (int j = 0; j < W; ++j)
                if (!visited[i][j] && A[i][j] > level) {
                    q.emplace(i, j);
                    visited[i][j] = true;
                }

        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            for (auto [dx, dy] : vector<pair<int, int>>{{0, 1}, {0, -1}, {1, 0}, {-1, 0}}) {
                int nx = x + dx, ny = y + dy;
                if (valid(nx, ny) && !visited[nx][ny] && A[nx][ny] > level) {
                    q.emplace(nx, ny);
                    visited[nx][ny] = true;
                }
            }
        }
    };

    vector<int> result(Y);
    for (int level = 0; level <= 1e5; ++level) {
        bfs(level);
        int count = 0;
        for (int i = 0; i < H; ++i)
            for (int j = 0; j < W; ++j)
                if (!visited[i][j])
                    ++count;
        for (int i = 0; i < min(Y, level + 1); ++i)
            result[i] = count;
    }

    for (int i = 0; i < Y; ++i)
        cout << result[i] << '\n';

    return 0;
}
```

### Explanation:
1. **Input Handling**: The code reads the dimensions `H` and `W` along with the number of queries `Y`. It then reads the elevation matrix `A`.
2. **Breadth-First Search (BFS)**: A BFS function is defined to mark all reachable land cells above the current sea level `level`.
3. **Result Calculation**: For each possible sea level from 0 to 100000, the BFS function marks all reachable land cells. The total number of unvisited cells (land) is counted and stored in the result array.
4. **Output**: Finally, the results for each query are printed.

This approach ensures that the algorithm efficiently calculates the area of the island above the sea level for each query within the given constraints.
