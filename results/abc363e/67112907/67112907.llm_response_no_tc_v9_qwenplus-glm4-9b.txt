#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int main() {
    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // Priority queue to store the sections of land with their heights and coordinates
    // and a map to check if a cell has been visited
    priority_queue<pair<int, pair<int, int>>, vector<pair<int, pair<int, int>>>, greater<pair<int, pair<int, int>>>> pq;
    vector<vector<bool>> visited(H, vector<bool>(W, false));

    // Initialize the priority queue with land that is adjacent to the sea level
    for (int i = 0; i < H; ++i) {
        if (!visited[i][0]) pq.push({A[i][0], {i, 0}});
        if (!visited[i][W - 1]) pq.push({A[i][W - 1], {i, W - 1}});
    }
    for (int j = 0; j < W; ++j) {
        if (!visited[0][j]) pq.push({A[0][j], {0, j}});
        if (!visited[H - 1][j]) pq.push({A[H - 1][j], {H - 1, j}});
    }

    // Iterate over the years
    for (int year = 0; year < Y; ++year) {
        int remaining_area = 0;
        // Process the cells that are at the current sea level or above
        while (!pq.empty()) {
            auto [height, pos] = pq.top();
            pq.pop();
            int x = pos.first;
            int y = pos.second;
            // Check if this cell has been visited to avoid processing it again
            if (visited[x][y]) continue;

            // Mark the cell as visited
            visited[x][y] = true;
            remaining_area++;

            // Process the adjacent cells
            if (x > 0 && !visited[x - 1][y]) {
                pq.push({A[x - 1][y], {x - 1, y}});
                visited[x - 1][y] = true;
            }
            if (x < H - 1 && !visited[x + 1][y]) {
                pq.push({A[x + 1][y], {x + 1, y}});
                visited[x + 1][y] = true;
            }
            if (y > 0 && !visited[x][y - 1]) {
                pq.push({A[x][y - 1], {x, y - 1}});
                visited[x][y - 1] = true;
            }
            if (y < W - 1 && !visited[x][y + 1]) {
                pq.push({A[x][y + 1], {x, y + 1}});
                visited[x][y + 1] = true;
            }
        }
        cout << remaining_area << "\n";
    }

    return 0;
}

This fixed version of the code uses a priority queue to process the sections of land that are adjacent to the sea level and are above the sea level, starting with the sections that are directly adjacent to the sea. It then updates the sea level by one unit each year and processes the sections of land that are now at the new sea level or above, updating the remaining area of land above sea level accordingly. The code also avoids processing a cell more than once by using a map to mark visited cells.
