```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H, W, Y;
    cin >> H >> W >> Y;

    vector<vector<int>> A(H, vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    // We will process cells in increasing order of elevation
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    vector<vector<bool>> sunk(H, vector<bool>(W, false));

    // Mark and add all border cells to the priority queue
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (i == 0 || i == H - 1 || j == 0 || j == W - 1) {
                pq.emplace(A[i][j], i, j);
                sunk[i][j] = true;
            }
        }
    }

    // We'll store the number of remaining land cells
    int totalLand = H * W;
    
    // To store the answer for each year
    vector<int> result(Y + 1, 0); // index 0 unused
    for (int y = 1; y <= Y; ++y) {
        result[y] = totalLand;
    }

    while (!pq.empty()) {
        auto [height, x, y] = pq.top();
        pq.pop();

        // If this cell is already processed or it's above current sea level, skip
        if (!sunk[x][y] || height > Y) continue;

        // This cell sinks at 'height' years, so reduce the area for that year
        if (height >= 1 && height <= Y) {
            result[height]--;
        }

        // Propagate to neighbors
        for (auto [dx, dy] : d) {
            int nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !sunk[nx][ny]) {
                sunk[nx][ny] = true;
                pq.emplace(A[nx][ny], nx, ny);
            }
        }
    }

    // Output results
    for (int y = 1; y <= Y; ++y) {
        cout << result[y] << "\n";
    }

    return 0;
}
```
