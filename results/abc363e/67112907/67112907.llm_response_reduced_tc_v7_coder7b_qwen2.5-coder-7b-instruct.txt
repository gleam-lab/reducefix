To solve this problem, we need to simulate the rising sea levels and determine how much of the island remains above water after each year. The key steps involve:

1. Parsing the input to get the initial heights of the island.
2. For each year, simulate the sinking of the land based on the current sea level.
3. Count the remaining land area after each simulation step.
4. Output the results for each year.

Here's the corrected and optimized C++ code:

```cpp
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1010;

int A[MAXN][MAXN];
bool visited[MAXN][MAXN];
int H, W, Y;

struct Cell {
    int x, y, height;
    bool operator<(const Cell& other) const {
        return height > other.height;
    }
};

void bfs(int sea_level) {
    queue<Cell> q;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!visited[i][j] && A[i][j] <= sea_level) {
                visited[i][j] = true;
                q.push({i, j, A[i][j]});
            }
        }
    }

    vector<pair<int, int>> directions = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    while (!q.empty()) {
        auto [x, y, h] = q.front();
        q.pop();
        for (auto [dx, dy] : directions) {
            int nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && A[nx][ny] <= sea_level) {
                visited[nx][ny] = true;
                q.push({nx, ny, A[nx][ny]});
            }
        }
    }
}

int countRemainingLand() {
    int count = 0;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            if (!visited[i][j]) {
                count++;
            }
        }
    }
    return count;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
        }
    }

    priority_queue<Cell> pq;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            pq.push({i, j, A[i][j]});
        }
    }

    for (int i = 0; i < Y; ++i) {
        while (!pq.empty() && pq.top().height <= i) {
            auto [x, y, h] = pq.top();
            pq.pop();
            bfs(i);
        }
        cout << countRemainingLand() << endl;
    }

    return 0;
}
```

### Explanation:
1. **Initialization**: We read the input dimensions and the initial heights of the island.
2. **Priority Queue**: We use a min-heap to store all cells along with their heights.
3. **BFS Simulation**: For each year, we perform a BFS starting from all cells that are currently below or equal to the sea level.
4. **Count Remaining Land**: After simulating the sinking, we count the number of unvisited cells, which represent the remaining land.
5. **Output**: We print the result for each year.

This approach ensures that we efficiently simulate the rising sea levels and correctly count the remaining land area. The time complexity is \(O(Y \log Y)\) due to the heap operations, which is efficient enough for the given constraints.
