#include <iostream>
#include <vector>
#include <queue>
#include <utility>

const int MAXN = 1005;

struct Cell {
    int h, x, y;
    bool operator<(const Cell& other) const {
        return h > other.h;
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(0);

    int H, W, Y;
    std::cin >> H >> W >> Y;

    std::vector<std::vector<int>> A(H, std::vector<int>(W));
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            std::cin >> A[i][j];
        }
    }

    std::priority_queue<Cell> pq;
    std::vector<std::vector<bool>> vis(H, std::vector<bool>(W, false));

    // Initialize priority queue with boundary cells
    for (int i = 0; i < H; ++i) {
        if (!vis[i][0]) pq.push({A[i][0], i, 0}), vis[i][0] = true;
        if (!vis[i][W-1]) pq.push({A[i][W-1], i, W-1}), vis[i][W-1] = true;
    }
    for (int j = 0; j < W; ++j) {
        if (!vis[0][j]) pq.push({A[0][j], 0, j}), vis[0][j] = true;
        if (!vis[H-1][j]) pq.push({A[H-1][j], H-1, j}), vis[H-1][j] = true;
    }

    int ans = H * W;
    while (!pq.empty()) {
        auto [h, x, y] = pq.top();
        pq.pop();

        if (h <= 0) continue;

        ans--;
        for (int k = 0; k < 4; ++k) {
            int nx = x + d[k][0], ny = y + d[k][1];
            if (nx < 0 || nx >= H || ny < 0 || ny >= W || vis[nx][ny]) continue;
            vis[nx][ny] = true;
            pq.push({A[nx][ny], nx, ny});
        }
    }

    for (int t = 1; t <= Y; ++t) {
        std::cout << ans << '\n';
    }

    return 0;
}
