#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>
#define vvl vector<vector<ll>>
#define vvb vector<vector<bool>>
#define pll pair<ll,ll>
#define pq priority_queue

// Direction vectors: up, right, down, left
ll dx[4] = {-1, 0, 1, 0};
ll dy[4] = {0, 1, 0, -1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll h, w, y;
    cin >> h >> w >> y;

    vvl grid(h, vl(w));
    ll total = h * w;

    // Read the grid
    rep(i, h) rep(j, w) cin >> grid[i][j];

    // Use a min-heap to process cells in increasing order of elevation
    priority_queue<pll, vector<pll>, greater<pll>> pq;

    // Mark visited cells
    vvb visited(h, vb(w, false));

    // Push all boundary cells into the heap (they are connected to sea)
    rep(i, h) {
        if (!visited[i][0]) {
            visited[i][0] = true;
            pq.emplace(grid[i][0], i * w + 0);
        }
        if (!visited[i][w - 1]) {
            visited[i][w - 1] = true;
            pq.emplace(grid[i][w - 1], i * w + (w - 1));
        }
    }

    rep(j, w) {
        if (!visited[0][j]) {
            visited[0][j] = true;
            pq.emplace(grid[0][j], 0 * w + j);
        }
        if (!visited[h - 1][j]) {
            visited[h - 1][j] = true;
            pq.emplace(grid[h - 1][j], (h - 1) * w + j);
        }
    }

    // Answer array for each year
    vl ans(y + 2, 0); // ans[i] is area remaining after i years
    for (ll i = 1; i <= y; ++i) ans[i] = total;

    ll year = 1;
    while (!pq.empty() && year <= y) {
        while (!pq.empty() && pq.top().first <= year) {
            ll val = pq.top().first;
            ll idx = pq.top().second;
            pq.pop();

            if (!visited[idx / w][idx % w]) continue;

            total--;
            ll x = idx / w;
            ll y_idx = idx % w;

            for (ll d = 0; d < 4; ++d) {
                ll nx = x + dx[d];
                ll ny = y_idx + dy[d];
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !visited[nx][ny]) {
                    visited[nx][ny] = true;
                    pq.emplace(grid[nx][ny], nx * w + ny);
                }
            }
        }
        ans[year++] = total;
    }

    // After processing, fill missing answers with previous values (for years not reached)
    for (ll i = 1; i <= y; ++i) {
        if (ans[i] == 0) ans[i] = ans[i - 1];
    }

    // Output result
    for (ll i = 1; i <= y; ++i) {
        cout << ans[i] << "\n";
    }

    return 0;
}
