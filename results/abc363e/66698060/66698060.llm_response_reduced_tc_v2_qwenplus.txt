#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

ll dx[4]={0,1,0,-1},dy[4]={1,0,-1,0};

int main(){
    ll h,w,y;cin >> h >> w >> y;
    vector<vl> room(h,vl(w));
    rep(i,h) rep(j,w) cin >> room[i][j];
    
    // Create events: when each cell gets submerged
    vector<tuple<ll,ll,ll>> cells;
    rep(i,h) rep(j,w) {
        cells.push_back({room[i][j], i, j});
    }
    sort(cells.begin(), cells.end());
    
    vb submerged(h*w, false);
    auto idx = [&](ll i, ll j) { return i * w + j; };
    
    ll remaining = h * w;
    ll ptr = 0;
    
    // Process each year
    for(ll year = 1; year <= y; year++) {
        // Add all cells with elevation <= current sea level
        while(ptr < cells.size() && get<0>(cells[ptr]) <= year) {
            ll i = get<1>(cells[ptr]);
            ll j = get<2>(cells[ptr]);
            ptr++;
            
            if(submerged[idx(i,j)]) continue;
            
            // Check if this cell is connected to the sea (border or already submerged neighbor)
            bool touches_sea = false;
            if(i == 0 || i == h-1 || j == 0 || j == w-1) {
                touches_sea = true;
            } else {
                rep(k,4) {
                    ll ni = i + dx[k], nj = j + dy[k];
                    if(0 <= ni && ni < h && 0 <= nj && nj < w) {
                        if(submerged[idx(ni,nj)]) {
                            touches_sea = true;
                            break;
                        }
                    }
                }
            }
            
            if(touches_sea) {
                queue<pair<ll,ll>> q;
                q.push({i,j});
                submerged[idx(i,j)] = true;
                remaining--;
                
                while(!q.empty()) {
                    auto [ci,cj] = q.front(); q.pop();
                    rep(k,4) {
                        ll ni = ci + dx[k], nj = cj + dy[k];
                        if(0 <= ni && ni < h && 0 <= nj && nj < w && !submerged[idx(ni,nj)] && room[ni][nj] <= year) {
                            submerged[idx(ni,nj)] = true;
                            remaining--;
                            q.push({ni,nj});
                        }
                    }
                }
            }
        }
        cout << remaining << endl;
    }
}
