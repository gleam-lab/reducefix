2. **Priority Queue**: We use a priority queue to process sections in order of their elevation. This helps in efficiently determining which sections sink as the sea level rises.
3. **Breadth-First Search (BFS)**: For each section that sinks, we check its adjacent sections (up, down, left, right) to see if they should also sink. This is done using BFS to ensure all connected sections at or below the current sea level are processed.
4. **Tracking Remaining Area**: We maintain a count of the remaining sections above sea level and update it each year as sections sink.

#define ll long long

ll dx[4] = {0, 1, 0, -1};
ll dy[4] = {1, 0, -1, 0};

int main() {

    ll ans = h * w;
    vector<vector<ll>> room(h, vector<ll>(w));
    vector<vector<bool>> used(h, vector<bool>(w, false));
    priority_queue<pair<ll, pair<ll, ll>>, vector<pair<ll, pair<ll, ll>>>, greater<pair<ll, pair<ll, ll>>>> pq;

    for (ll i = 0; i < h; i++) {

            cin >> room[i][j];
        }
    }

    // Push border cells into the priority queue
    for (ll i = 0; i < h; i++) {
        if (!used[i][0]) {
            pq.push({room[i][0], {i, 0}});
            used[i][0] = true;
        }
        if (!used[i][w - 1]) {
            pq.push({room[i][w - 1], {i, w - 1}});
            used[i][w - 1] = true;
        }
    }
    for (ll j = 1; j < w - 1; j++) {
        if (!used[0][j]) {
            pq.push({room[0][j], {0, j}});
            used[0][j] = true;
        }
        if (!used[h - 1][j]) {
            pq.push({room[h - 1][j], {h - 1, j}});
            used[h - 1][j] = true;
        }
    }

    ll current_sea_level = 0;
    for (ll year = 1; year <= y; year++) {
        current_sea_level = year;
        while (!pq.empty() && pq.top().first <= current_sea_level) {
            auto current = pq.top();
            pq.pop();

            ans--;
            for (ll k = 0; k < 4; k++) {
                ll ni = i + dx[k];
                ll nj = j + dy[k];

                    pq.push({room[ni][nj], {ni, nj}});

            }
        }

    }
    return 0;
}

1. **Initialization**: The code reads the island dimensions (H, W), the number of years (Y), and the elevation of each section. It initializes a priority queue to process sections in order of their elevation.
2. **Border Processing**: The border sections of the island are added to the priority queue first since they are adjacent to the sea and will sink first as the sea level rises.
3. **Yearly Simulation**: For each year, the sea level increases by 1. The priority queue processes sections with elevation less than or equal to the current sea level. Each processed section is marked as sunk, and its adjacent sections are checked and added to the queue if they meet the sinking criteria.
4. **Output**: After processing each year, the remaining area of the island is printed.
