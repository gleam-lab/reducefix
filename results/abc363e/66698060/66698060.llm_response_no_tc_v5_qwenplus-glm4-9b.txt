#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    vector<vector<ll>> a(h, vector<ll>(w));
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            cin >> a[i][j];
        }
    }

    // Initialize the queue and visited array
    vector<vector<bool>> visited(h, vector<bool>(w, false));
    vector<vector<pair<ll, ll>>> adj(h, vector<pair<ll, ll>>(w));
    ll remaining_area = 0;

    // Fill the adjacency list with unvisited neighbors
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            for (ll k = 0; k < 4; ++k) {
                ll ni = i + dx[k], nj = j + dy[k];
                if (ni >= 0 && ni < h && nj >= 0 && nj < w && !visited[ni][nj]) {
                    adj[i][j].push_back({ni, nj});
                }
            }
        }
    }

    // BFS to find initial remaining area
    queue<pair<ll, ll>> q;
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            if (!visited[i][j]) {
                visited[i][j] = true;
                remaining_area++;
                q.push({i, j});
            }
        }
    }

    // Process each year
    for (ll year = 0; year < y; ++year) {
        queue<pair<ll, ll>> new_q;
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();
            for (auto [nx, ny] : adj[x][y]) {
                if (a[nx][ny] > year + 1) {
                    new_q.push({nx, ny});
                    visited[nx][ny] = true;
                    remaining_area++;
                }
            }
        }
        q = new_q;
    }

    // Output the remaining area for each year
    for (ll year = 0; year < y; ++year) {
        cout << remaining_area << endl;
    }

    return 0;
}

const ll dx[4] = {-1, 0, 1, 0};
const ll dy[4] = {0, 1, 0, -1};
