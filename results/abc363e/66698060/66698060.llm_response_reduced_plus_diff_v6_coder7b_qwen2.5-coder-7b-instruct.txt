#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<int>
#define vvi vector<vector<int>>
#define pb push_back
#define all(x) x.begin(), x.end()

struct Cell {
    int h, x, y;
    bool operator>(const Cell& other) const { return h > other.h; }
};

class Solution {
public:
    vvi simulateIsland(vector<vvi>& grid, int n, int m, int y) {
        vvi result(n, vi(m, 0));
        priority_queue<Cell> pq;
        
        // Initialize priority queue with cells that have height less than or equal to the first sea level
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] <= 1) {
                    pq.push({grid[i][j], i, j});
                    result[i][j] = 1;
                }
            }
        }

        int directions[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        while (!pq.empty()) {
            Cell cell = pq.top();
            pq.pop();

            for (auto dir : directions) {
                int nx = cell.x + dir[0];
                int ny = cell.y + dir[1];

                if (nx >= 0 && nx < n && ny >= 0 && ny < m && !result[nx][ny] && grid[nx][ny] <= cell.h) {
                    pq.push({grid[nx][ny], nx, ny});
                    result[nx][ny] = 1;
                }
            }
        }

        // For each subsequent sea level
        for (int s = 2; s <= y; ++s) {
            priority_queue<Cell> new_pq;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < m; ++j) {
                    if (grid[i][j] <= s && !result[i][j]) {
                        new_pq.push({grid[i][j], i, j});
                        result[i][j] = 1;
                    }
                }
            }

            while (!new_pq.empty()) {
                Cell cell = new_pq.top();
                new_pq.pop();

                for (auto dir : directions) {
                    int nx = cell.x + dir[0];
                    int ny = cell.y + dir[1];

                    if (nx >= 0 && nx < n && ny >= 0 && ny < m && !result[nx][ny] && grid[nx][ny] <= cell.h) {
                        new_pq.push({grid[nx][ny], nx, ny});
                        result[nx][ny] = 1;
                    }
                }
            }
        }

        return result;
    }
};

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int h, w, y;
    cin >> h >> w >> y;

    vvi grid(h, vi(w));

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> grid[i][j];
        }
    }

    Solution sol;
    vvi result = sol.simulateIsland(grid, h, w, y);

    for (int i = 0; i < y; ++i) {
        int count = 0;
        for (int j = 0; j < h; ++j) {
            for (int k = 0; k < w; ++k) {
                if (result[j][k] == 1) {
                    ++count;
                }
            }
        }
        cout << count << '\n';
    }

    return 0;
}
