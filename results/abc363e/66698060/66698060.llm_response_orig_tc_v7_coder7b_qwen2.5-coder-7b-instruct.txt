#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1010;

struct Point {
    int x, y, h;
    bool operator<(const Point &p) const { return h > p.h; }
};

int n, m, k;
vector<vector<int>> grid(MAXN, vector<int>(MAXN));
priority_queue<Point> pq;
vector<vector<bool>> visited(MAXN, vector<bool>(MAXN));

int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

void bfs(int level) {
    queue<pair<int, int>> q;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (!visited[i][j] && grid[i][j] <= level) {
                visited[i][j] = true;
                q.push({i, j});
            }
        }
    }

    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx >= 0 && nx < n && ny >= 0 && ny < m && !visited[nx][ny] && grid[nx][ny] <= level) {
                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    cin >> n >> m >> k;

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> grid[i][j];
            pq.push({i, j, grid[i][j]});
        }
    }

    while (k--) {
        int level = pq.top().h;
        pq.pop();

        bfs(level);

        int remaining_area = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (!visited[i][j]) {
                    ++remaining_area;
                }
            }
        }

        cout << remaining_area << '\n';
    }

    return 0;
}
