```cpp
#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vl vector<ll>
#define vb vector<bool>
#define pll pair<ll,ll>
#define vvl vector<vl>
#define vvb vector<vb>
#define queue myqueue

// Directions: right, down, left, up
ll dx[4] = {0, 1, 0, -1};
ll dy[4] = {1, 0, -1, 0};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    
    vvl grid(h, vl(w));
    rep(i, h) rep(j, w) cin >> grid[i][j];

    // We will process cells in increasing order of elevation using a min-heap (priority queue)
    priority_queue<vl, vector<vl>, greater<vl>> pq;
    vvb visited(h, vb(w, false));

    // Add all boundary cells to the priority queue
    for (ll i = 0; i < h; ++i) {
        for (ll j : {0, w - 1}) {
            if (!visited[i][j]) {
                pq.push({grid[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }

    for (ll j = 1; j < w - 1; ++j) {
        for (ll i : {0, h - 1}) {
            if (!visited[i][j]) {
                pq.push({grid[i][j], i, j});
                visited[i][j] = true;
            }
        }
    }

    // total area above sea level
    ll remaining = h * w;

    // events[y] = number of cells submerged at sea level y
    vl events(100002, 0); // max Y is 1e5

    while (!pq.empty()) {
        vl curr = pq.top(); pq.pop();
        ll val = curr[0], x = curr[1], y_coord = curr[2];
        
        // For each cell we visit, it's being submerged
        events[val]++;
        remaining--;

        for (ll dir = 0; dir < 4; ++dir) {
            ll nx = x + dx[dir];
            ll ny = y_coord + dy[dir];
            if (nx >= 0 && nx < h && ny >= 0 && ny < w && !visited[nx][ny]) {
                pq.push({grid[nx][ny], nx, ny});
                visited[nx][ny] = true;
            }
        }
    }

    // Prefix sum over events to get cumulative submerged cells by year
    vl prefix_sum(100002, 0);
    for (ll i = 1; i <= 100000; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + events[i];
    }

    // Output the remaining area after each year from 1 to Y
    for (ll year = 1; year <= y; ++year) {
        cout << (h * w - prefix_sum[year]) << "\n";
    }
}
```
