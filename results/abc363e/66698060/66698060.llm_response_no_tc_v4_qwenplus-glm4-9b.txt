#include <bits/stdc++.h>
using namespace std;
#define ll long long

int h, w, y;

struct Cell {
    ll elevation, row, col;
};

vector<vector<ll>> grid;
vector<vector<bool>> used;
vector<Cell> pq;

bool compare(const Cell& a, const Cell& b) {
    return a.elevation > b.elevation;
}

int main() {
    cin >> h >> w >> y;
    grid.resize(h, vector<ll>(w));
    used.resize(h, vector<bool>(w, false));
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            cin >> grid[i][j];
        }
    }

    // Initialize the priority queue with the edges of the grid
    for (ll i = 0; i < h; ++i) {
        pq.push_back({grid[i][0], i, 0});
        used[i][0] = true;
        pq.push_back({grid[i][w - 1], i, w - 1});
        used[i][w - 1] = true;
    }
    for (ll j = 1; j < w - 1; ++j) {
        pq.push_back({grid[0][j], 0, j});
        used[0][j] = true;
        pq.push_back({grid[h - 1][j], h - 1, j});
        used[h - 1][j] = true;
    }

    // Sort the priority queue to ensure we process the lowest cells first
    sort(pq.begin(), pq.end(), compare);

    ll remaining_area = h * w;

    while (pq.size()) {
        Cell current = pq.top();
        pq.pop();
        if (current.elevation <= y) continue; // Only process cells that are above the current sea level

        // Mark the cell as used
        used[current.row][current.col] = true;
        remaining_area--;

        // Explore the 4 adjacent cells
        for (ll k = 0; k < 4; ++k) {
            ll ni = current.row + dx[k], nj = current.col + dy[k];
            if (ni >= 0 && ni < h && nj >= 0 && nj < w && !used[ni][nj] && grid[ni][nj] > y) {
                pq.push_back({grid[ni][nj], ni, nj});
                sort(pq.begin(), pq.end(), compare); // Keep the priority queue sorted
            }
        }
    }

    for (ll i = 0; i < y; ++i) {
        cout << remaining_area << endl;
    }

    return 0;
}
