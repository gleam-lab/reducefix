    ll idx = 0;

        // Add all cells with elevation <= current sea level
        while(idx < cells.size() && get<0>(cells[idx]) <= year) {
            auto [elev, i, j] = cells[idx];
            if (!used[i][j]) {

            }
            idx++;
        }
