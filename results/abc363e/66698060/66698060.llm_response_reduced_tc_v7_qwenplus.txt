#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

ll dx[4]={0,1,0,-1},dy[4]={1,0,-1,0};

int main(){
    ll h,w,y;cin >> h >> w >> y;
    vector<vl> room(h,vl(w));
    rep(i,h) rep(j,w) cin >> room[i][j];
    
    // Create a list of all cells with their elevations
    vector<tuple<ll,ll,ll>> cells;
    rep(i,h) rep(j,w) {
        cells.push_back({room[i][j], i, j});
    }
    
    // Sort by elevation in ascending order
    sort(cells.begin(), cells.end());
    
    // Union-Find or simply use BFS from the boundary? 
    // Instead: Process year by year, but simulate sinking from the outside
    // We'll use a priority queue to process cells in order of elevation
    priority_queue<tuple<ll,ll,ll>, vector<tuple<ll,ll,ll>>, greater<tuple<ll,ll,ll>>> pq;
    vector<vb> used(h,vb(w,false));
    ll ans = h * w;
    
    // Add all border cells to the priority queue
    rep(i,h) {
        pq.push({room[i][0], i, 0});
        used[i][0] = true;
        if (w > 1) {
            pq.push({room[i][w-1], i, w-1});
            used[i][w-1] = true;
        }
    }
    rep(j,1,w-1) {
        pq.push({room[0][j], 0, j});
        used[0][j] = true;
        if (h > 1) {
            pq.push({room[h-1][j], h-1, j});
            used[h-1][j] = true;
        }
    }
    
    // For each year, process all cells that sink at that sea level
    rep(year, y) {
        ll sea_level = year + 1;
        
        // Process all cells with elevation <= current sea level
        while (!pq.empty()) {
            auto [elev, i, j] = pq.top();
            if (elev > sea_level) break;
            
            pq.pop();
            ans--;
            
            // Check neighbors
            rep(k,4) {
                ll ni = i + dx[k], nj = j + dy[k];
                if (ni >= 0 && ni < h && nj >= 0 && nj < w && !used[ni][nj]) {
                    used[ni][nj] = true;
                    pq.push({room[ni][nj], ni, nj});
                }
            }
        }
        
        cout << ans << endl;
    }
}
