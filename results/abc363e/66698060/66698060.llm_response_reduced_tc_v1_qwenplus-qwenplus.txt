#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>
#define vc vector<char>
#define vvl vector<vector<ll>>
#define vvb vector<vector<bool>>
#define pll pair<ll,ll>
#define INF (1LL << 60)
#define PQ priority_queue

// Directions: up, right, down, left
ll dx[4] = {-1, 0, 1, 0};
ll dy[4] = {0, 1, 0, -1};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll H, W, Y;
    cin >> H >> W >> Y;

    vvl grid(H, vl(W));
    rep(i, H) rep(j, W) cin >> grid[i][j];

    // Visited matrix
    vvb visited(H, vb(W, false));

    // Priority queue to simulate BFS with elevation order
    using state = tuple<ll, ll, ll>; // elevation, x, y
    priority_queue<state, vector<state>, greater<state>> pq;

    // Push all boundary cells into the queue
    rep(i, H) {
        pq.push({grid[i][0], i, 0});
        pq.push({grid[i][W-1], i, W-1});
        visited[i][0] = true;
        visited[i][W-1] = true;
    }

    rep(j, W) {
        if (!visited[0][j]) {
            pq.push({grid[0][j], 0, j});
            visited[0][j] = true;
        }
        if (!visited[H-1][j]) {
            pq.push({grid[H-1][j], H-1, j});
            visited[H-1][j] = true;
        }
    }

    // Total area of the island
    ll totalArea = H * W;
    ll sunkCount = 0;

    // Array to store answers for each year
    vl ans(Y+2, 0); // ans[i] stores the area remaining at year i

    // Initialize all years with full area
    for(ll i = 1; i <= Y; ++i) ans[i] = totalArea;

    // Process events in increasing order of elevation (time)
    while(!pq.empty()) {
        ll h, r, c;
        tie(h, r, c) = pq.top();
        pq.pop();

        // If this cell would survive past year h, skip
        if (h > Y) continue;

        // This cell sinks at year h
        sunkCount++;
        ans[h]--;

        // Propagate to neighbors
        rep(d, 4) {
            ll nr = r + dx[d];
            ll nc = c + dy[d];

            if (nr >= 0 && nr < H && nc >= 0 && nc < W && !visited[nr][nc]) {
                visited[nr][nc] = true;
                pq.push({grid[nr][nc], nr, nc});
            }
        }
    }

    // Maintain the maximum remaining area so far
    for(ll i = Y; i >= 1; --i) {
        if(ans[i] < 0) ans[i] = 0;
        ans[i] = min(ans[i], ans[i+1]);
    }

    // Output results
    for(ll i = 1; i <= Y; ++i) {
        cout << ans[i] << '\n';
    }

    return 0;
}
