```cpp
#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll dx[4] = {0, 1, 0, -1};
ll dy[4] = {1, 0, -1, 0};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll h, w, y;
    cin >> h >> w >> y;

    vector<vector<ll>> A(h, vector<ll>(w));
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            cin >> A[i][j];
        }
    }

    // Total number of cells
    ll total = h * w;
    ll current_year = 0;
    ll result_index = 0;

    // Used array to mark visited cells
    vector<vector<bool>> used(h, vector<bool>(w, false));

    // Priority queue to process the lowest height first (min-heap)
    priority_queue<vector<ll>, vector<vector<ll>>, greater<vector<ll>>> pq;

    // Mark and add all boundary cells
    for (ll i = 0; i < h; ++i) {
        if (!used[i][0]) {
            pq.push({A[i][0], i, 0});
            used[i][0] = true;
        }
        if (!used[i][w - 1]) {
            pq.push({A[i][w - 1], i, w - 1});
            used[i][w - 1] = true;
        }
    }

    for (ll j = 1; j < w - 1; ++j) {
        if (!used[0][j]) {
            pq.push({A[0][j], 0, j});
            used[0][j] = true;
        }
        if (!used[h - 1][j]) {
            pq.push({A[h - 1][j], h - 1, j});
            used[h - 1][j] = true;
        }
    }

    // Process year by year
    vector<ll> results(y);

    while (current_year < y) {
        current_year++;
        while (!pq.empty() && pq.top()[0] <= current_year) {
            auto top = pq.top();
            pq.pop();
            ll x = top[1], y_coord = top[2];

            for (ll d = 0; d < 4; ++d) {
                ll nx = x + dx[d];
                ll ny = y_coord + dy[d];
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !used[nx][ny]) {
                    used[nx][ny] = true;
                    pq.push({A[nx][ny], nx, ny});
                }
            }

            total--;
        }
        results[result_index++] = total;
    }

    for (auto r : results) {
        cout << r << endl;
    }

    return 0;
}
```
