#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll dx[4] = {0, 1, 0, -1};
ll dy[4] = {1, 0, -1, 0};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    vector<vector<ll>> A(h, vector<ll>(w));
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            cin >> A[i][j];
        }
    }

    // Create a grid to track which cells are submerged
    vector<vector<bool>> submerged(h, vector<bool>(w, false));

    // Priority queue for BFS processing (min-heap by elevation)
    priority_queue<vector<ll>, vector<vector<ll>>, greater<vector<ll>>> pq;

    // Mark the border cells as processed and add them to the queue
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                submerged[i][j] = true;
                pq.push({A[i][j], i, j});
            }
        }
    }

    // Process the cells in order of increasing elevation
    while (!pq.empty()) {
        auto v = pq.top();
        pq.pop();
        ll val = v[0], x = v[1], y = v[2];

        for (ll k = 0; k < 4; ++k) {
            ll nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < h && ny >= 0 && ny < w && !submerged[nx][ny]) {
                submerged[nx][ny] = true;
                pq.push({A[nx][ny], nx, ny});
            }
        }
    }

    // For each year, compute how many cells are above sea level
    vector<ll> result(y + 2, 0);  // 1-based index up to Y
    vector<vector<ll>> heightCount(h, vector<ll>(w, 0));

    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            if (!submerged[i][j]) {
                heightCount[i][j] = A[i][j];
            }
        }
    }

    // Prepare a list of all non-submerged cell heights
    vector<ll> landHeights;
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            if (!submerged[i][j]) {
                landHeights.push_back(A[i][j]);
            }
        }
    }

    sort(landHeights.begin(), landHeights.end());

    // Compute remaining area for each year
    ll totalRemaining = landHeights.size();
    ll idx = 0;
    for (ll year = 1; year <= y; ++year) {
        while (idx < (ll)landHeights.size() && landHeights[idx] <= year) {
            totalRemaining--;
            idx++;
        }
        cout << totalRemaining << endl;
    }
}
