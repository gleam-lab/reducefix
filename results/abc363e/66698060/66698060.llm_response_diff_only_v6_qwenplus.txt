#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>

ll dx[4] = {0, 1, 0, -1};
ll dy[4] = {1, 0, -1, 0};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    
    vector<vl> room(h, vl(w));
    rep(i, h) rep(j, w) cin >> room[i][j];
    
    // Store cells by their elevation
    vector<vector<pair<ll, ll>>> cells_at_elevation(100001); // max elevation is 10^5
    
    rep(i, h) {
        rep(j, w) {
            cells_at_elevation[room[i][j]].push_back({i, j});
        }
    }
    
    // Track which cells are submerged
    vector<vb> submerged(h, vb(w, false));
    ll remaining = h * w;
    
    // Process each year from 1 to Y
    for (ll year = 1; year <= y; year++) {
        // Add all cells with elevation <= current sea level that haven't been processed yet
        if (year <= 100000) {
            for (auto &cell : cells_at_elevation[year]) {
                ll i = cell.first, j = cell.second;
                if (!submerged[i][j]) {
                    // Check if this cell is connected to the sea (on border or adjacent to submerged cell)
                    bool touches_sea = false;
                    
                    // Check if on border
                    if (i == 0 || i == h-1 || j == 0 || j == w-1) {
                        touches_sea = true;
                    } else {
                        // Check if adjacent to already submerged cell
                        rep(k, 4) {
                            ll ni = i + dx[k], nj = j + dy[k];
                            if (ni >= 0 && ni < h && nj >= 0 && nj < w && submerged[ni][nj]) {
                                touches_sea = true;
                                break;
                            }
                        }
                    }
                    
                    if (touches_sea) {
                        // Use BFS to flood all connected cells with elevation <= year
                        queue<pair<ll, ll>> q;
                        q.push({i, j});
                        if (!submerged[i][j]) {
                            submerged[i][j] = true;
                            remaining--;
                            
                            while (!q.empty()) {
                                auto [ci, cj] = q.front(); q.pop();
                                
                                rep(k, 4) {
                                    ll ni = ci + dx[k], nj = cj + dy[k];
                                    if (ni >= 0 && ni < h && nj >= 0 && nj < w && !submerged[ni][nj] && room[ni][nj] <= year) {
                                        submerged[ni][nj] = true;
                                        remaining--;
                                        q.push({ni, nj});
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        
        cout << remaining << endl;
    }
    
    return 0;
}
