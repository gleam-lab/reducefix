#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define rep(i,n) for(ll i=0;i<n;i++)
#define all(x) x.begin(), x.end()

struct Point {
    ll x, y, height;
    Point(ll _x, ll _y, ll _height) : x(_x), y(_y), height(_height) {}
    bool operator>(const Point& other) const {
        return height > other.height;
    }
};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    
    vector<vector<ll>> grid(h, vector<ll>(w));
    rep(i, h) rep(j, w) cin >> grid[i][j];

    priority_queue<Point, vector<Point>, greater<Point>> pq;
    vector<vector<bool>> visited(h, vector<bool>(w, false));

    rep(i, h) {
        pq.push({i, 0, grid[i][0]});
        pq.push({i, w-1, grid[i][w-1]});
        visited[i][0] = true;
        visited[i][w-1] = true;
    }

    rep(j, w) {
        if (j == 0 || j == w-1) continue;
        pq.push({0, j, grid[0][j]});
        pq.push({h-1, j, grid[h-1][j]});
        visited[0][j] = true;
        visited[h-1][j] = true;
    }

    ll ans = h * w;
    while (!pq.empty()) {
        auto [x, y, height] = pq.top();
        pq.pop();
        
        if (height <= y) continue;
        ans--;

        rep(dx, 4) {
            ll nx = x + dx, ny = y + dy[dx];
            if (nx >= 0 && nx < h && ny >= 0 && ny < w && !visited[nx][ny]) {
                visited[nx][ny] = true;
                pq.push({nx, ny, grid[nx][ny]});
            }
        }
    }

    cout << ans << endl;
    return 0;
}
