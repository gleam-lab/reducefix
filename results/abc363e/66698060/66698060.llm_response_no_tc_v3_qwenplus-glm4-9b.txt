#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for (ll i = 0; i < (n); i++)
#define vi vector<int>
#define vl vector<ll>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>

ll dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;
    vector<vl> A(h, vl(w));
    rep(i, h) rep(j, w) cin >> A[i][j];
    
    // Create a sea level map initialized to -1 (not visited)
    vector<vc> visited(h, vc(w, -1));
    
    // Directions for the DFS
    ll directions[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    // Queue for BFS (to determine the sea level)
    queue<pair<ll, ll>> q;
    
    // Initialize the BFS from the edges and set the visited status
    rep(i, h) {
        q.push({i, 0});
        q.push({i, w - 1});
        visited[i][0] = visited[i][w - 1] = 0;
    }
    rep(j, w) {
        if (j == 0 || j == w - 1) continue; // Already added in the loop above
        q.push({0, j});
        q.push({h - 1, j});
        visited[0][j] = visited[h - 1][j] = 0;
    }
    
    // Perform BFS to determine the sea levels
    while (!q.empty()) {
        ll x = q.front().first, y = q.front().second;
        q.pop();
        rep(k, 4) {
            ll nx = x + dx[k], ny = y + dy[k];
            if (0 <= nx && nx < h && 0 <= ny && ny < w) {
                if (visited[nx][ny] == -1) {
                    visited[nx][ny] = visited[x][y] + 1;
                    q.push({nx, ny});
                }
            }
        }
    }
    
    // Calculate the remaining area after Y years
    ll area = 0;
    rep(i, h) rep(j, w) {
        if (visited[i][j] > y || visited[i][j] == -1) {
            area++;
        }
    }
    
    // Output the remaining area for each year
    rep(i, y) {
        cout << area - (h * w - (h + w) * y / 2 + (h * w - (h + w) * (y - 1) / 2)) << endl;
    }
    return 0;
}
