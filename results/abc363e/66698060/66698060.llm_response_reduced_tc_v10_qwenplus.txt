#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,n) for(ll i=0;i<(ll)n;i++)
#define vi vector<int>
#define vl vector<ll>
#define vd vector<double>
#define vb vector<bool>
#define vs vector<string>
#define vc vector<char>
#define ull unsigned long long
#define chmax(a,b) a=max(a,b)
#define chmin(a,b) a=min(a,b)

ll dx[4]={0,1,0,-1},dy[4]={1,0,-1,0};

int main(){
    ll h,w,y;cin >> h >> w >> y;
    vector<vl> room(h,vl(w));
    rep(i,h) rep(j,w) cin >> room[i][j];
    
    // Create a list of all cells with their elevations and positions
    vector<tuple<ll,ll,ll>> cells;
    rep(i,h) rep(j,w) {
        cells.push_back({room[i][j], i, j});
    }
    
    // Sort by elevation in ascending order
    sort(cells.begin(), cells.end());
    
    // Union-Find or visited array to track which cells are submerged
    vb used(h*w, false);
    auto get_index = [&](ll i, ll j) { return i * w + j; };
    
    ll remaining = h * w;
    ll ptr = 0;
    
    // Process each year from 1 to y
    for (ll year = 1; year <= y; year++) {
        // Add all cells with elevation <= current sea level (year)
        while (ptr < cells.size() && get<0>(cells[ptr]) <= year) {
            ll i = get<1>(cells[ptr]);
            ll j = get<2>(cells[ptr]);
            ll idx = get_index(i, j);
            
            if (!used[idx]) {
                used[idx] = true;
                remaining--;
                
                // Check neighbors: if they are already marked but not yet processed through queue,
                // they will be handled naturally as we process all <= elevation cells.
                // But we don't need to propagate because sorting ensures we'll hit them eventually.
            }
            ptr++;
        }
        
        cout << remaining << endl;
    }
}
