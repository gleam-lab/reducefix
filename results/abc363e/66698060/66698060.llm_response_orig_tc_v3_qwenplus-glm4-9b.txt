
The problem is about simulating the process of the sea level rising and the island sinking accordingly. The goal is to compute the area of the island that remains above the sea level after a certain number of years.

The provided code seems to be a simulation that uses a priority queue to manage which parts of the island sink first as the sea level rises each year. However, the code has several issues:

1. **Incorrect Simulation of Sinking**: The code attempts to use a priority queue to simulate the sinking process, but it doesn't correctly check if the section is adjacent to the sea or the newly sunk sections.
2. **Inefficient Sinking Algorithm**: The sinking algorithm is very inefficient because it checks every year if any part of the island has sunk, which is not necessary.
3. **Inconsistent Indexing and Off-by-One Errors**: The code uses 0-based indexing, but the problem statement uses 1-based indexing, which introduces off-by-one errors.

To solve this problem, we can use a flood fill algorithm to find all connected sections of the island that are above the current sea level. As the sea level rises each year, we repeat the flood fill algorithm to find the new area above the sea level.

### Solution Steps:

1. **Mapping Water and Land**: Create a 2D array to map parts of the island that are currently underwater and parts that are above the current sea level.
2. **Initial Flood Fill**: Execute a flood fill from the edges of the island that are above the current sea level to map out the land.
3. **Adjusting for Rising Sea Level**: As the sea level rises, update the mapping to remove areas that are now underwater and perform another flood fill to find new land above the sea level.
4. **Repeat for Each Year**: Repeat the flood filling process for each year until the desired number of years is reached.

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

const int dx[] = {1, 0, 0, -1};
const int dy[] = {0, 1, -1, 0};
const int MAX = 1000;

void flood_fill(vector<vector<int>>& visited, const vector<vector<int>>& elevation, int x, int y, int currentSeaLevel) {
    queue<pair<int, int>> q;
    q.push({x, y});
    visited[x][y] = 1;
    
    while (!q.empty()) {
        auto [cx, cy] = q.front();
        q.pop();
        
        for (int i = 0; i < 4; ++i) {
            int nx = cx + dx[i];
            int ny = cy + dy[i];
            
            if (nx >= 0 && nx < MAX && ny >= 0 && ny < MAX && !visited[nx][ny] && elevation[nx][ny] > currentSeaLevel) {
                visited[nx][ny] = 1;
                q.push({nx, ny});
            }
        }
    }
}

int main() {
    int h, w, y;
    cin >> h >> w >> y;
    
    vector<vector<int>> elevation(h, vector<int>(w));
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> elevation[i][j];
        }
    }
    
    vector<vector<int>> visited(h, vector<int>(w, 0));
    int remaining_area = 0;
    
    // Find initial land above sea level
    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            if (elevation[i][j] > y && !visited[i][j]) {
                flood_fill(visited, elevation, i, j, y);
                remaining_area++;
            }
        }
    }
    
    // Output the result for each year
    for (int year = 1; year <= y; ++year) {
        cout << remaining_area << endl;
    }
    
    return 0;
}
