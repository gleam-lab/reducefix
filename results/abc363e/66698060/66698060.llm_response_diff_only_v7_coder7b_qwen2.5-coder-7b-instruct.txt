#include <bits/stdc++.h>
using namespace std;

const int MAX_H = 1000;
const int MAX_W = 1000;
const int MAX_Y = 100000;

vector<vector<int>> grid(MAX_H, vector<int>(MAX_W));
vector<vector<bool>> visited(MAX_H, vector<bool>(MAX_W));

void bfs(int h, int w, int sea_level) {
    queue<pair<int, int>> q;
    for (int i = 0; i < h; ++i) {
        if (grid[i][0] > sea_level) {
            q.push({i, 0});
            visited[i][0] = true;
        }
        if (grid[i][w - 1] > sea_level) {
            q.push({i, w - 1});
            visited[i][w - 1] = true;
        }
    }
    for (int j = 1; j < w - 1; ++j) {
        if (grid[0][j] > sea_level) {
            q.push({0, j});
            visited[0][j] = true;
        }
        if (grid[h - 1][j] > sea_level) {
            q.push({h - 1, j});
            visited[h - 1][j] = true;
        }
    }

    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();

        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (nx >= 0 && nx < h && ny >= 0 && ny < w && !visited[nx][ny] && grid[nx][ny] > sea_level) {
                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
}

int main() {
    int h, w, y;
    cin >> h >> w >> y;

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            cin >> grid[i][j];
        }
    }

    vector<int> results(y);
    for (int sea_level = 1; sea_level <= y; ++sea_level) {
        fill(visited.begin(), visited.end(), false);
        bfs(h, w, sea_level);
        int count = 0;
        for (int i = 0; i < h; ++i) {
            for (int j = 0; j < w; ++j) {
                if (!visited[i][j]) {
                    count++;
                }
            }
        }
        results[y - sea_level] = count;
    }

    for (int result : results) {
        cout << result << endl;
    }

    return 0;
}
