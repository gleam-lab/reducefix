#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1005;
int H, W, Y;
vector<vector<int>> A(MAXN, vector<int>(MAXN));

struct Cell {
    int x, y, level;
    bool operator<(const Cell& other) const {
        return level > other.level;
    }
};

priority_queue<Cell> pq;
vector<vector<bool>> visited(MAXN, vector<bool>(MAXN, false));

void bfs(int startLevel) {
    while (!pq.empty()) {
        auto [x, y, level] = pq.top();
        pq.pop();

        if (level != startLevel) break;

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx >= 0 && nx < H && ny >= 0 && ny < W && !visited[nx][ny] && A[nx][ny] <= level) {
                visited[nx][ny] = true;
                pq.emplace(nx, ny, level);
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr);

    cin >> H >> W >> Y;
    for (int i = 0; i < H; ++i) {
        for (int j = 0; j < W; ++j) {
            cin >> A[i][j];
            pq.emplace(i, j, A[i][j]);
        }
    }

    int ans = H * W;
    for (int t = 0; t < Y; ++t) {
        bfs(t + 1);
        ans -= count_if(begin(visited), end(visited), [](const auto& v) { return accumulate(begin(v), end(v), 0); });
        fill(begin(visited), end(visited), vector<bool>(W, false));
        cout << ans << '\n';
    }

    return 0;
}
