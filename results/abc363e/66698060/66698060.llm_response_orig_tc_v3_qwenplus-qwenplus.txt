#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

int main() {
    ll h, w, y;
    cin >> h >> w >> y;

    vector<vector<ll>> A(h, vector<ll>(w));
    for (ll i = 0; i < h; ++i) {
        for (ll j = 0; j < w; ++j) {
            cin >> A[i][j];
        }
    }

    // Create a visited matrix
    vector<vector<bool>> used(h, vector<bool>(w, false));

    // Priority queue to simulate BFS with minimum heap
    priority_queue<vector<ll>, vector<vector<ll>>, greater<vector<ll>>> pq;

    // Initialize the border cells as visited and add to PQ
    for (ll i = 0; i < h; ++i) {
        if (!used[i][0]) {
            pq.push({A[i][0], i, 0});
            used[i][0] = true;
        }
        if (!used[i][w-1]) {
            pq.push({A[i][w-1], i, w-1});
            used[i][w-1] = true;
        }
    }
    for (ll j = 1; j < w - 1; ++j) {
        if (!used[0][j]) {
            pq.push({A[0][j], 0, j});
            used[0][j] = true;
        }
        if (!used[h-1][j]) {
            pq.push({A[h-1][j], h-1, j});
            used[h-1][j] = true;
        }
    }

    ll total_area = h * w;
    vector<ll> result(y + 1);  // Store results for each year
    result[0] = total_area;

    ll removed = 0;
    ll current_year = 1;

    while (current_year <= y) {
        while (!pq.empty() && pq.top()[0] <= current_year) {
            ll ni = pq.top()[1];
            ll nj = pq.top()[2];
            pq.pop();

            if (!used[ni][nj]) {
                used[ni][nj] = true;
                removed++;
                for (ll k = 0; k < 4; ++k) {
                    ll nx = ni + dx[k];
                    ll ny = nj + dy[k];
                    if (nx >= 0 && nx < h && ny >= 0 && ny < w && !used[nx][ny]) {
                        pq.push({A[nx][ny], nx, ny});
                        used[nx][ny] = true;
                    }
                }
            }
        }
        result[current_year] = total_area - removed;
        current_year++;
    }

    // Output the results
    for (ll i = 1; i <= y; ++i) {
        cout << result[i] << endl;
    }

    return 0;
}
