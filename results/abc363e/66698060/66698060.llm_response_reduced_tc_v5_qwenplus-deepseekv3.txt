using namespace std;

ll dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

    vector<vector<ll>> room(h, vector<ll>(w));

            cin >> room[i][j];
        }
    }

    // Push all border cells into the priority queue
    for (ll i = 0; i < h; i++) {
        if (!used[i][0]) {
            pq.push({room[i][0], i, 0});
            used[i][0] = true;
        }
        if (w > 1 && !used[i][w-1]) {
            pq.push({room[i][w-1], i, w-1});
            used[i][w-1] = true;
        }
    }
    for (ll j = 1; j < w-1; j++) {
        if (!used[0][j]) {
            pq.push({room[0][j], 0, j});
            used[0][j] = true;
        }
        if (h > 1 && !used[h-1][j]) {
            pq.push({room[h-1][j], h-1, j});
            used[h-1][j] = true;
        }
    }

    for (ll current_year = 1; current_year <= y; current_year++) {
        while (!pq.empty() && pq.top()[0] <= current_year) {
            ll val = pq.top()[0];
            ll i = pq.top()[1];
            ll j = pq.top()[2];
            pq.pop();
            ans--;

            for (ll k = 0; k < 4; k++) {

                ll nj = j + dy[k];

The algorithm works by:

This ensures that we efficiently process cells in order of when they will be submerged, maintaining O(N log N) complexity where N is the number of cells.
