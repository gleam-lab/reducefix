using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> H(n);
    ll total_time = 0;

    // We will simulate the attack pattern using a queue to manage triple attacks
    queue<int> triple_attack_times;

    rep(i, n) {
        ll h;
        cin >> h;

        // While enemy still has health
        while (h > 0) {
            total_time++;

            // If T is multiple of 3, do triple damage
            if (total_time % 3 == 0) {
                h -= 3;
            } else {
                h -= 1;
            }
        }
    }

}
