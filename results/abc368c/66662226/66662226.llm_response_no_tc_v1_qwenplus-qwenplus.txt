```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;
    ll attack_at_t1 = 0;  // Number of enemies attacked at T ≡ 1 mod 3
    ll attack_at_t2 = 0;  // Number of enemies attacked at T ≡ 2 mod 3

    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<>> pq;

    rep(i, n) {
        pq.push({1, H[i]}); // {time when this enemy will be attacked again, remaining health}
    }

    while (!pq.empty()) {
        auto [t, h] = pq.top();
        pq.pop();

        T = t;
        ll damage = (t % 3 == 0) ? 3 : 1;
        h -= damage;

        if (h > 0) {
            pq.push({t + 1, h});
        }
    }

    cout << T << endl;
}
```
