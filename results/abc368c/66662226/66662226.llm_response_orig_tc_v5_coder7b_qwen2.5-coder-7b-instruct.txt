#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n)
    {
        cin >> a[i];
    }

    ll ans = 0;
    ll t = 0;
    while (true) {
        bool any = false;
        rep(i, n) {
            if (a[i] > 0) {
                any = true;
                if ((t + 1) % 3 == 0) {
                    a[i] -= 3;
                } else {
                    a[i]--;
                }
                ans++;
                if (a[i] <= 0) {
                    a[i] = 0;
                }
            }
        }
        if (!any) break;
        t++;
    }
    cout << ans << endl;
}
