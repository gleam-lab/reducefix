#include <iostream>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main() {
    int n;
    cin >> n;
    vector<int> health(n);
    long long ans = 0;

    // Read the health values of each enemy
    for (int i = 0; i < n; ++i) {
        cin >> health[i];
    }

    // Process each enemy
    for (int i = 0; i < n; ++i) {
        while (health[i] > 0) {
            if (ans % 3 == 0 && health[i] >= 3) {
                // If T is a multiple of 3, attack with 3 damage
                health[i] -= 3;
            } else {
                // Otherwise, attack with 1 damage
                health[i]--;
            }
            ans++;
        }
    }

    // Output the result
    cout << ans << endl;

    return 0;
}
