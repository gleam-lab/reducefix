using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<int> a(n);
    ll ans = 0;
    rep(i, n)
    {
        cin >> a[i];
        while (a[i] > 0)
        {
            ans++;
            if (ans % 3 == 0)
            {
                a[i] -= 3;
            }
            else
            {
                a[i]--;
            }
        }
    }
    cout << ans << endl;
}
