#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to determine number of attacks needed for one enemy
ll attacks_needed(ll h) {
    // We want to find minimum T such that:
    // Let x be number of triple attacks (at T divisible by 3)
    // Let y be number of single attacks (at T not divisible by 3)
    // We want: 3x + y >= h and T = x + y
    // And the sequence must respect timing constraints

    // Binary search on T
    ll low = 0, high = 3 * h;
    while (low < high) {
        ll mid = (low + high) / 2;
        ll triple_attacks = mid / 3;
        ll total_damage = 3 * triple_attacks + (mid - triple_attacks);
        if (total_damage >= h)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i)
        cin >> H[i];

    ll current_time = 0;
    for (int i = 0; i < n; ++i) {
        // For each enemy, we try to defeat it as soon as possible after current_time
        // First, simulate the time when this enemy can be defeated
        ll start_time = current_time;
        
        // Number of attacks available before next multiple of 3
        ll remaining_before_triple = 2 - (current_time % 3);
        if (remaining_before_triple <= 0)
            remaining_before_triple += 3;
        
        // Try first few attacks manually
        ll damage = 0;
        while (damage < H[i]) {
            current_time++;
            if (current_time % 3 == 0)
                damage += 3;
            else
                damage++;
        }
    }

    cout << current_time << endl;
    return 0;
}
