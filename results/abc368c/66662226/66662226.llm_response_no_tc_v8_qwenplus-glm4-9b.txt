#include <iostream>
using namespace std;
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (ll i = 0; i < n; ++i) {
        if (H[i] % 3 == 0) {
            T += H[i] / 3;
            H[i] = 0;
        } else {
            ll k = H[i] / 3;
            T += 3 * k + 1;
            H[i] -= 3 * k + 1;
        }
    }

    cout << T << endl;
    return 0;
}
