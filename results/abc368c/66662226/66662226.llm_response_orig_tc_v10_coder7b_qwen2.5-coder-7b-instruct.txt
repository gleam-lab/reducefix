#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    vector<long long> h(n);
    rep(i, n) cin >> h[i];

    long long t = 0;
    while(true){
        bool flag = true;
        for(int i=0;i<n;i++){
            if(h[i]>0){
                flag=false;
                if((t+1)%3==0){
                    h[i]-=3;
                }else{
                    h[i]--;
                }
                t++;
                if(h[i]<=0) h[i]=0;
            }
        }
        if(flag) break;
    }

    cout<<t<<endl;
    return 0;
}

- We initialize `t` as 0 which represents the number of turns we have taken so far.
- We use a while loop to continue attacking the enemies until all their healths become zero or less.
- Inside the while loop, we iterate over each enemy. If an enemy has health greater than zero, we decrement its health based on whether `t+1` is divisible by 3 or not. After decrementing the health, we increment `t`.
- The loop breaks when there are no enemies left with health greater than zero.
- Finally, we print the total number of turns taken.
