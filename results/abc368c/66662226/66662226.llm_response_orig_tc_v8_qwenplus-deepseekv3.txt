   - The sum is `k + 2 * (floor((T_start + k)/3) - floor(T_start/3))`.
   - Add `k` to `ans`.
3. After processing all enemies, output `ans`.

#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

ll calculate_k(ll T_start, ll H) {
    ll low = 1, high = 3 * H, k = high;
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll m = (T_start + mid) / 3 - T_start / 3;
        if (mid + 2 * m >= H) {
            k = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return k;
}

int main() {

    vector<ll> h(n);

    ll ans = 0;
    rep(i, n) {
        ll k = calculate_k(ans, h[i]);

}
