```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i = 0; i < (n); ++i)

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> h(n);
    rep(i, n) cin >> h[i];

    long long t = 0;
    bool alive = true;
    
    while (alive) {
        alive = false;
        for (int i = 0; i < n; ++i) {
            if (h[i] > 0) {
                alive = true;
                ++t;
                if (t % 3 == 0) h[i] -= 3;
                else h[i] -= 1;
            }
        }
    }

    cout << t << '\n';
}
```
