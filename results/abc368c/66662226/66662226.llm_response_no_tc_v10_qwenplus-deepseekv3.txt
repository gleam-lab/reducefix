4. **Sum Total Attacks**: Sum the total attacks (T) across all enemies, ensuring that each attack increments T correctly.

#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    int N;
    cin >> N;
    vector<ll> H(N);
    rep(i, N) cin >> H[i];

    ll T = 0;
    for (ll h : H) {
        if (h <= 0) continue;
        ll low = 0, high = h;
        ll best_k = h;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll m = (T + mid) / 3 - T / 3;
            ll sum = mid + 2 * m;
            if (sum >= h) {
                best_k = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        T += best_k;
    }

    cout << T << endl;
    return 0;
}
