```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n)
    {
        cin >> a[i];
    }
    ll ans = 0;
    while(true){
        bool flag = true;
        rep(i,n){
            if(a[i]>0){
                ans++;
                if(ans%3==0){
                    a[i]-=3;
                }else{
                    a[i]-=1;
                }
                if(a[i]>0){
                    flag = false;
                }
            }
        }
        if(flag) break;
    }
    cout << ans << endl;
}
```
