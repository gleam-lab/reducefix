#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to determine how many attacks are needed for one enemy
ll compute_attacks(int health) {
    // We want to find the minimal T such that total damage >= health.
    // Each attack alternates: 1,1,3,1,1,3,...
    // So every 3 attacks deal 5 damage.

    ll full_cycles = health / 5;
    ll remaining_health = health % 5;

    ll time = full_cycles * 3;

    while (remaining_health > 0) {
        time++;
        if (time % 3 == 0) {
            remaining_health -= 3;
        } else {
            remaining_health -= 1;
        }
    }

    return time;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<int> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll ans = 0;
    vector<ll> required_time(n);

    // Compute required times in parallel
    for (int i = 0; i < n; ++i) {
        required_time[i] = compute_attacks(H[i]);
    }

    // Use prefix max because we can only attack one enemy at a time
    ll current_max = 0;
    for (int i = 0; i < n; ++i) {
        current_max = max(current_max, required_time[i]);
        ans += current_max - current_max / 3;  // Count non-3x turns
        ans += (current_max / 3) * 2;         // Count 3x turns (each gives 2 damages of 3)
    }

    cout << ans << endl;
}
