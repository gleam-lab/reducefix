#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to determine the minimum number of attacks needed to reduce health to 0
// Each attack does 3 damage if T is a multiple of 3, else 1 damage
ll compute_attacks(int health) {
    // Let x be the number of full 5-damage cycles (3 turns): 3 turns give 5 damage total
    // Remaining damage handled per turn
    ll attacks = 0;
    while (health > 0) {
        attacks++;
        if (attacks % 3 == 0) {
            health -= 3;
        } else {
            health -= 1;
        }
    }
    return attacks;
}

int main() {
    int n;
    cin >> n;
    vector<int> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    // We process each enemy from left to right
    // For each enemy, simulate the attacks until its health drops to 0 or below
    ll T = 0;
    for (int i = 0; i < n; ++i) {
        if (H[i] <= 0) continue;

        // Simulate attacks on this enemy until its health is <= 0
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
        }
    }

    cout << T << endl;
}
