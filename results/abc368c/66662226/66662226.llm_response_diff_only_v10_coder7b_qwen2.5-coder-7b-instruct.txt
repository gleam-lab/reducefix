```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n)
    {
        cin >> h[i];
    }
    ll ans = 0;
    rep(i, n)
    {
        ans += (h[i] + 2) / 3;
    }
    cout << ans << endl;
}
```
