#include <bits/stdc++.h>

#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

ll compute_attacks(ll H, ll start_T) {
    ll low = 0;
    ll high = 3 * H;
    ll ans = high;
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll damage = 0;
        ll full_cycles = mid / 3;
        damage = full_cycles * 5;
        ll remaining = mid % 3;
        damage += remaining;
        if (damage >= H) {
            ans = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return ans;
}
