```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;
    // We'll process all enemies in a queue to simulate attacking only frontmost alive enemy
    queue<int> q;
    rep(i, n) if (H[i] > 0) q.push(i);

    while (!q.empty())
    {
        T++;
        int idx = q.front();
        if (T % 3 == 0)
        {
            H[idx] -= 3;
        }
        else
        {
            H[idx]--;
        }

        if (H[idx] > 0)
        {
            // Keep the enemy in queue if still alive
            q.push(idx);
        }
        q.pop();

        // Remove enemies from front of queue if they're dead or have been processed
        while (!q.empty() && H[q.front()] <= 0)
        {
            q.pop();
        }
    }

    cout << T << endl;
}
```
