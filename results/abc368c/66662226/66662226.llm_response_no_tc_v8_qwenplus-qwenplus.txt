```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;
    // We simulate the process using a queue to track enemies that still have health left
    queue<int> q;
    rep(i, n) q.push(i);

    while (!q.empty())
    {
        int idx = q.front();
        q.pop();

        T++;
        if (T % 3 == 0)
        {
            H[idx] -= 3;
        }
        else
        {
            H[idx]--;
        }

        if (H[idx] > 0)
        {
            q.push(idx); // enemy is still alive, push back into queue
        }
    }

    cout << T << endl;
}
```
