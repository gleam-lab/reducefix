```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n)
    {
        cin >> h[i];
    }

    ll t = 0;
    while (true)
    {
        bool any_positive = false;
        rep(i, n)
        {
            if (h[i] > 0)
            {
                any_positive = true;
                break;
            }
        }
        if (!any_positive)
        {
            break;
        }
        t++;
        if (t % 3 == 0)
        {
            rep(i, n)
            {
                if (h[i] > 0)
                {
                    h[i] -= 3;
                }
            }
        }
        else
        {
            rep(i, n)
            {
                if (h[i] > 0)
                {
                    h[i]--;
                }
            }
        }
    }
    cout << t << endl;
}
```
