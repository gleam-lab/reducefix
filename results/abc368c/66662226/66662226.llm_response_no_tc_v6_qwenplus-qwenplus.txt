#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to compute ceiling of division
ll ceil_div(ll a, ll b) {
    return (a + b - 1) / b;
}

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;
    // We will use a queue to simulate the attack cycle
    queue<pair<ll, ll>> q; // pair: (health, time when it will be hit by next normal attack)

    rep(i, n) {
        if (H[i] > 0) {
            q.push({H[i], T});
        }
    }

    while (!q.empty()) {
        auto [health, nextAttackTime] = q.front();
        q.pop();
        T++;

        // If this is a multiple of 3, special attack does 3 damage
        if (T % 3 == 0) {
            health -= 3;
        } else {
            health -= 1;
        }

        if (health > 0) {
            // Enemy still alive, schedule next attack
            q.push({health, T});
        }
    }

    cout << T << endl;
}
