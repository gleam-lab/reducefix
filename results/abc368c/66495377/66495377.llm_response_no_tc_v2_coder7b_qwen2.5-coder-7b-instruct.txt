#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll calculate_rounds_to_zero(ll h) {
    ll rounds = 0;
    while (h > 0) {
        rounds++;
        if (rounds % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
    }
    return rounds;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    ll total_rounds = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        total_rounds += calculate_rounds_to_zero(h);
    }

    cout << total_rounds << endl;
}
