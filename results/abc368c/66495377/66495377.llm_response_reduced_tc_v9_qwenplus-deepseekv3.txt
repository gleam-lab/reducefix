#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll compute_attacks(ll T_before, ll H) {
    ll low = 0, high = 2 * H;
    while (low < high) {
        ll mid = (low + high) / 2;
        ll k = (T_before + mid) / 3 - T_before / 3;
        ll damage = 2 * k + mid;
        if (damage >= H) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }
    ll T = 0;
    for (int i = 0; i < N; ++i) {
        ll m = compute_attacks(T, H[i]);
        T += m;
    }
    cout << T << endl;
    return 0;
}
