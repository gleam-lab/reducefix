#include <iostream>
#include <vector>

using namespace std;

long long solve(vector<long long>& hs) {
    long long total_attacks = 0;
    for (auto& h : hs) {
        long long attacks = 0;
        while (h > 0) {
            if ((total_attacks + 1) % 3 == 0) {
                h -= 3;
            } else {
                h -= 1;
            }
            attacks++;
        }
        total_attacks += attacks;
    }
    return total_attacks;
}

int main() {
    int n;
    cin >> n;
    vector<long long> hs(n);
    for (int i = 0; i < n; ++i) {
        cin >> hs[i];
    }
    cout << solve(hs) << endl;
    return 0;
}
