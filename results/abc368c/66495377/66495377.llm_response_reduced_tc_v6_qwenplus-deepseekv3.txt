#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll compute_k(ll T, ll H) {
    ll low = 0, high = 2 * H; // Upper bound: since each attack deals at least 1 damage, k won't exceed 2*H
    while (low < high) {
        ll mid = (low + high) / 2;
        ll count_multiples = (T + mid) / 3 - T / 3;
        ll sum = 2 * count_multiples + mid;
        if (sum >= H) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low;
}

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }
    
    ll T = 0;
    for (int i = 0; i < n; ++i) {
        ll k = compute_k(T, H[i]);
        T += k;
    }
    
    cout << T << endl;
    return 0;
