```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to calculate number of attacks required for a single enemy
ll num_attack(ll t, ll h) {
    ll cnt = 0;
    while (h > 0) {
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
        cnt++;
    }
    return cnt;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    ll total_attacks = 0;
    
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        total_attacks += num_attack(total_attacks, h);
    }

    cout << total_attacks << endl;
    return 0;
}
```
