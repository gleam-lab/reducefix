#include <iostream>
#include <vector>

using namespace std;

long long calculateAttacks(const vector<long long>& healths) {
    long long attacks = 0;
    long long current_time = 0;

    for (long long h : healths) {
        if (h > 0) {
            if (current_time % 3 == 0) {
                attacks += (h + 2) / 3;
            } else {
                attacks += h;
            }
        }
        current_time++;
    }

    return attacks;
}

int main() {
    int n;
    cin >> n;
    vector<long long> healths(n);

    for (int i = 0; i < n; ++i) {
        cin >> healths[i];
    }

    long long total_attacks = calculateAttacks(healths);

    cout << total_attacks << endl;

    return 0;
}
