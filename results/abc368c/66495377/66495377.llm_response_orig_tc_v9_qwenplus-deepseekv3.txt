#include <bits/stdc++.h>
using namespace std;
using ll = long long;

bool is_possible(const vector<ll>& H, ll T) {
    ll total_attacks = 0;
    for (ll h : H) {
        ll low = 0, high = T - total_attacks;
        ll best_Ti = high + 1;
        ll r = total_attacks % 3;
        while (low <= high) {
            ll mid = (low + high) / 2;
            ll strong_attacks = (r + mid) / 3;
            ll damage = mid + 2 * strong_attacks;
            if (damage >= h) {
                best_Ti = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        if (best_Ti > T - total_attacks) {
            return false;
        }
        total_attacks += best_Ti;
        if (total_attacks > T) {
            return false;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    ll left = 0;
    ll right = 1e18;
    ll answer = right;

    while (left <= right) {
        ll mid = (left + right) / 2;
        if (is_possible(H, mid)) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }

    cout << answer << '\n';
    return 0;
}
