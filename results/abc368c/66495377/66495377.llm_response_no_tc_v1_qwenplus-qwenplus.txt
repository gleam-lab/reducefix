#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to calculate the number of attacks needed for one enemy
ll compute_attacks(ll t, ll h) {
    ll low = 0;
    ll high = 2 * h; // Upper bound on number of attacks

    auto is_sufficient = [&](ll k) {
        // Total number of triple-damage attacks at T values that are multiples of 3
        ll triple_attacks = k / 3 + (t / 3 <= (t + k - 1) / 3 ? 0 : 1);
        ll total_damage = triple_attacks * 3 + k * 1;
        return total_damage >= h;
    };

    while (low < high) {
        ll mid = (low + high) / 2;
        if (is_sufficient(mid)) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }

    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll attack_needed = compute_attacks(t, H[i]);
        t += attack_needed;
    }

    cout << t << endl;
    return 0;
}
