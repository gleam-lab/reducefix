#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Calculates number of attacks needed to defeat an enemy with health h
ll num_attacks(ll t, ll h) {
    ll low = 0, high = 2e18;
    while (low < high) {
        ll mid = (low + high) / 2;
        ll damage = 0;
        
        // Count how many attacks happen in mid time units
        // First, count full cycles of 3
        ll full_cycles = mid / 3;
        damage += full_cycles * 3;  // 3-point attacks on multiples of 3
        
        // Remaining attacks are regular 1-point attacks
        damage += (mid - full_cycles);
        
        // Adjust for initial offset t
        ll attack_count = 0;
        ll cycle_start = t + 1;
        if (cycle_start % 3 == 0) cycle_start++;

        if (cycle_start % 3 == 1) {
            // We get one or two regular attacks before the 3-point pattern starts
            if (mid > 0) {
                damage--;
                attack_count++;
                mid--;
            }
            if (mid > 0) {
                damage--;
                attack_count++;
                mid--;
            }
        } else if (cycle_start % 3 == 2) {
            // We get one regular attack before the 3-point pattern starts
            if (mid > 0) {
                damage--;
                attack_count++;
                mid--;
            }
        }

        // Now apply the repeating 3-point attack pattern
        ll three_point_cycles = mid / 3;
        damage += three_point_cycles * 3;
        mid -= three_point_cycles * 3;

        // Apply remaining 1-point attacks
        while (mid > 0) {
            if (attack_count % 3 == 0) break; // 3-point attack slot
            damage++;
            mid--;
            attack_count++;
        }

        if (damage >= h)
            high = mid + attack_count;
        else
            low = mid + 1;
    }
    
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }
    
    ll total_time = 0;
    for (int i = 0; i < n; ++i) {
        ll attacks_needed = num_attacks(total_time, H[i]);
        total_time += attacks_needed;
    }
    
    cout << total_time << "\n";
    return 0;
}
