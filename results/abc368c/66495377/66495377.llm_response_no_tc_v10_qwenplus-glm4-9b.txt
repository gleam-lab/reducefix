#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    long long t = 0;
    for (int i = 0; i < n; ++i) {
        long long h;
        cin >> h;
        // Calculate how many full sets of 3 attacks we can perform on this enemy
        int full_sets = h / 3;
        // Calculate the remainder after performing full sets of 3 attacks
        int remainder = h % 3;
        // Each full set of 3 attacks takes 3 steps
        t += 3 * full_sets;
        // If there is a remainder, we need an additional step
        if (remainder > 0) {
            t += 1;
        }
    }
    cout << t << endl;
    return 0;
}
