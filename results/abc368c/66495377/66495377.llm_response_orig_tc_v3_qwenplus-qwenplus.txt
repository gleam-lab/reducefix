#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Check if all enemies are defeated with 't' total attacks
bool is_enough_attacks(const vector<ll>& h, ll t) {
    ll remaining = t;
    for (ll hi : h) {
        if (hi <= 0) continue;
        // Calculate how many attacks needed for this enemy
        ll full_cycles = remaining / 3;
        ll damage_from_full_cycles = full_cycles * 5; // 2 hits of 1 and 1 hit of 3 per 3 attacks
        
        ll remaining_attacks = remaining % 3;
        ll partial_damage = min(remaining_attacks, hi - damage_from_full_cycles);
        
        ll total_damage = damage_from_full_cycles + partial_damage;
        
        if (total_damage < hi) return false;
        remaining -= (hi + 2) / 3 * 3; // Number of attacks needed to defeat this enemy
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; ++i) {
        cin >> h[i];
    }

    // Binary search for the minimum T
    ll left = 0, right = 1e18; // Large upper bound
    while (left < right) {
        ll mid = (left + right) / 2;
        if (is_enough_attacks(h, mid)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }

    cout << left << endl;
    return 0;
}
