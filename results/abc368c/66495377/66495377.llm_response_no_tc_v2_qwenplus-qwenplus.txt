#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Calculates the number of attacks needed for an enemy with health h
ll required_attacks(ll h) {
    // Each 3 turns, we deal 3 + 1 + 1 = 5 damage (pattern: -3, -1, -1)
    ll full_cycles = h / 5;
    ll remaining_health = h % 5;
    
    ll attacks = full_cycles * 3;
    
    // Account for remaining health after full cycles
    if (remaining_health > 0) attacks += 1;
    if (remaining_health > 3) attacks += 1; // If more than 3 left, need one more attack
    
    return attacks;
}

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (int i = 0; i < n; ++i) {
        ll h = H[i];
        if (h <= 0) continue;

        // Determine number of attacks needed for this enemy
        ll attacks_needed = required_attacks(h);
        
        // Update total time T
        T += attacks_needed;
    }

    cout << T << endl;
    return 0;
}
