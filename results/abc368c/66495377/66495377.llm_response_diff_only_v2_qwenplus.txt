#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Simulate attacks starting from time T until enemy with health H is defeated
ll simulate_attack(ll t, ll h) {
    while (h > 0) {
        t++;
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
    }
    return t;
}

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll current_time = 0;
    for (int i = 0; i < n; ++i) {
        // Process each enemy in order
        if (H[i] <= 0) continue;
        
        current_time = simulate_attack(current_time, H[i]);
    }

    cout << current_time << endl;
    return 0;
}
