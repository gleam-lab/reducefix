#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll num_attack(ll t, ll h) {
    if (h == 0) return 0;
    
    // Each full cycle of 3 attacks: positions T ≡ 1,2,0 mod 3
    // In 3 consecutive attacks: damage = 1 + 1 + 3 = 5
    ll cycles = h / 5;
    ll rem = h % 5;
    
    ll cnt = 3 * cycles;
    t += cnt;
    
    while (rem > 0) {
        t++;
        cnt++;
        if (t % 3 == 0) {
            rem -= 3;
        } else {
            rem--;
        }
    }
    
    return cnt;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    ll t = 0;
    
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        ll attacks_needed = num_attack(t, h);
        t += attacks_needed;
    }

    cout << t << endl;
}
