#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Calculates the number of attacks needed to reduce health h to 0 or less
ll required_attacks(ll h) {
    // Each 3 attacks: two 1-damage and one 3-damage => total 5 damage
    ll full_cycles = h / 5;
    ll damage_from_full_cycles = full_cycles * 5;
    
    ll remaining_health = h - damage_from_full_cycles;
    
    if (remaining_health == 0) return full_cycles * 3;
    
    // After full cycles:
    // Attack 1: 1 damage
    // Attack 2: 1 damage
    // Attack 3: 3 damage
    if (remaining_health <= 2) return full_cycles * 3 + remaining_health;
    else if (remaining_health <= 4) return full_cycles * 3 + 2; // both 1-damage attacks
    else return full_cycles * 3 + 3; // all three attacks
}

int main() {
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (ll h : H) {
        ll attacks_needed = required_attacks(h);
        T += attacks_needed;
    }

    cout << T << endl;
}
