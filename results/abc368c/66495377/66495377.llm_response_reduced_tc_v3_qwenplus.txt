#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll num_attack(ll t, ll h) {
    // We want to compute the number of attacks needed starting from time T
    // In a cycle of 3: two normal attacks (-1) and one triple attack (-3)
    // So every 3 attacks deal 5 damage (1 + 1 + 3)

    ll total_attacks = 0;

    // Full cycles of 3 attacks that give 5 damage each
    ll full_cycles = h / 5;
    total_attacks += full_cycles * 3;  // 3 attacks per cycle

    // Remaining health after full cycles
    ll rem = h % 5;

    // Simulate the remaining attacks one by one
    ll current_t = t + total_attacks;
    while (rem > 0) {
        current_t++;
        if (current_t % 3 == 0) {
            rem -= 3;
        } else {
            rem--;
        }
        total_attacks++;
    }

    return total_attacks;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        ll attacks_needed = num_attack(t, h);
        t += attacks_needed;
    }

    cout << t << endl;
}
