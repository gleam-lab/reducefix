#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to calculate the number of attacks needed for a single enemy
ll calculate_attacks(ll t, ll h) {
    ll attacks = 0;
    
    // Binary search approach to find the minimum number of attacks needed
    ll low = 0, high = 2 * h; // upper bound estimation
    
    while (low < high) {
        ll mid = (low + high) / 2;
        ll damage = 0;
        
        // Count how many attacks are multiples of 3 in [1..mid]
        ll attack_t = t + mid;
        ll full_cycles = (attack_t - t) / 3;
        ll partial_cycle = max(0LL, (attack_t % 3) - (t % 3));
        
        ll three_damage_attacks = full_cycles + (t % 3 == 0 ? 1 : 0);
        if (attack_t >= t && attack_t % 3 == 0)
            three_damage_attacks += 1;
        
        // Subtract the extra ones beyond mid
        if (t + mid >= attack_t)
            three_damage_attacks -= 1;
        
        three_damage_attacks = (attack_t - t + 2) / 3;
        
        damage = three_damage_attacks * 3 + (mid - three_damage_attacks) * 1;
        
        if (damage >= h)
            high = mid;
        else
            low = mid + 1;
    }
    
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (int i = 0; i < n; ++i) {
        ll attacks_needed = calculate_attacks(T, H[i]);
        T += attacks_needed;
    }
    
    cout << T << endl;
    
    return 0;
}
