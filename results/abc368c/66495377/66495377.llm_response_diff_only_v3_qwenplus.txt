#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Simulate attacks starting from time T until enemy's health H becomes <= 0
ll simulate_enemy(ll& t, ll h) {
    ll attacks = 0;
    while (h > 0) {
        t++;
        attacks++;
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
    }
    return attacks;
}

int main() {
    int n;
    cin >> n;
    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        simulate_enemy(t, h);
    }
    cout << t << endl;
    return 0;
}
