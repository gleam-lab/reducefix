```cpp
#include <iostream>
#include <vector>

using namespace std;

long long solve(long long health) {
    long long attacks = 0;
    while (health > 0) {
        attacks++;
        if (attacks % 3 == 0) {
            health -= 3;
        } else {
            health -= 1;
        }
    }
    return attacks;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> health(n);
    for (auto& h : health) {
        cin >> h;
    }

    long long total_attacks = 0;
    for (const auto& h : health) {
        total_attacks += solve(h);
    }

    cout << total_attacks << '\n';

    return 0;
}
```
