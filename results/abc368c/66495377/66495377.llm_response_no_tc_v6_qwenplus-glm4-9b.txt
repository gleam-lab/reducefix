#include <iostream>
using namespace std;
using ll = long long;

ll calculate_attacks(ll health) {
    // When health is a multiple of 3, we can attack it completely in one go.
    while (health % 3 == 0) {
        health /= 3;
    }
    // After reducing to a multiple of 3, we can attack it in steps of 3.
    return health; // The number of attacks needed is the health in the end.
}

int main() {
    int n;
    cin >> n;
    ll total_attacks = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        total_attacks += calculate_attacks(h);
    }

    cout << total_attacks << endl;
    return 0;
}
