#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to determine the minimum number of attacks needed for one enemy
ll compute_attacks(ll t, ll h) {
    // We need to find the minimum T such that total damage >= h
    // Binary search on T
    ll low = 0, high = 2 * h; // upper bound generous enough
    while (low < high) {
        ll mid = (low + high) / 2;
        ll damage = 0;
        
        // Count how many times we hit with 3-damage and 1-damage in `mid` steps
        ll three_damage_count = mid / 3;
        ll one_damage_count = mid - three_damage_count;
        ll total_damage = one_damage_count * 1 + three_damage_count * 3;
        
        if (total_damage >= h)
            high = mid;
        else
            low = mid + 1;
    }

    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll attacks_needed = compute_attacks(t, H[i]);
        t += attacks_needed;
    }

    cout << t << endl;
}
