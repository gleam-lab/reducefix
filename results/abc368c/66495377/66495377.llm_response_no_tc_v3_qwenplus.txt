#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Simulate attacks starting from time t until enemy with health h is defeated
ll simulate_enemy(ll& t, ll h) {
    while (h > 0) {
        t++;
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
    }
    return t;
}

int main() {
    int n;
    cin >> n;
    ll t = 0;
    
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        simulate_enemy(t, h);
    }

    cout << t << endl;
    return 0;
}
