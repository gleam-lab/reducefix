#include <bits/stdc++.h>
using namespace std;
using ll = long long;

// Function to compute the number of attacks needed to reduce health h to 0 or less
// given that we start attacking at time t
ll compute_attacks(ll t, ll h) {
    // We will use binary search to find the minimum number of seconds needed
    ll low = 0, high = 2 * h; // upper bound is generous

    while (low < high) {
        ll mid = (low + high) / 2;
        
        // Calculate total damage after 'mid' attacks
        ll damage = 0;
        
        // Number of attacks in this period: mid
        // For every 3 seconds cycle: two 1-damage attacks and one 3-damage attack
        ll full_cycles = mid / 3;
        ll remaining = mid % 3;
        
        // Damage from full cycles: each cycle does 5 damage (1+1+3)
        damage += full_cycles * 5;
        
        // Damage from remaining attacks
        if (remaining >= 1) damage += 1;
        if (remaining >= 2) damage += 1;
        if (remaining >= 3) damage += 3;
        
        // If starting at time t, check which ticks fall on multiples of 3
        // Adjust based on starting time t's modulo 3
        ll offset = t % 3;
        ll effective_damage = 0;
        
        // Total attacks: mid
        // We simulate how many of those attacks fall on multiples of 3,
        // and how many don't, based on the offset
        
        ll three_damage_attacks = (mid + offset) / 3 - t / 3;
        effective_damage += three_damage_attacks * 3;
        effective_damage += (mid - three_damage_attacks) * 1;
        
        if (effective_damage >= h) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll attacks_needed = compute_attacks(t, H[i]);
        t += attacks_needed;
    }

    cout << t << endl;
}
