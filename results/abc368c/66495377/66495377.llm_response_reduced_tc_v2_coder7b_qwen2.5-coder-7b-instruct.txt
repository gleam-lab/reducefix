#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll calculate_attacks(ll h) {
    ll attacks = 0;
    while (h > 0) {
        if ((attacks + 1) % 3 == 0) {
            h -= 3;
        } else {
            h -= 1;
        }
        attacks++;
    }
    return attacks;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    ll total_attacks = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        total_attacks += calculate_attacks(h);
    }

    cout << total_attacks << endl;
}
