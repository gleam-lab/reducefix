#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll num_attack(ll t, ll h) {

    while (h > 0) {
        t++;
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h--;
        }
        cnt++;
    }
    return cnt;
}

}
