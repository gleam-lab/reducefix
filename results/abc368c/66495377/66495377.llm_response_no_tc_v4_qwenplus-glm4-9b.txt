#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int n;
    cin >> n;
    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        while (h > 0) {
            if (t % 3 == 0) {
                h -= 3;
            } else {
                h -= 1;
            }
            t++;
        }
    }

    cout << t << endl;
}
