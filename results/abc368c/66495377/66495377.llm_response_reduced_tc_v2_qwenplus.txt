#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll num_attack(ll t, ll h) {
    if (h == 0) return 0;
    
    // We want to find the minimum number of attacks starting from time T+1
    // such that total damage >= h.
    ll low = 0, high = h; // at most h attacks (if all deal 1 damage)
    while (low < high) {
        ll mid = (low + high) / 2;
        ll damage = 0;
        // Count how many of the next 'mid' attacks are multiples of 3
        ll start = t + 1;
        ll end = t + mid;
        ll triple_attacks = end / 3 - (start - 1) / 3;
        damage += triple_attacks * 3;
        damage += (mid - triple_attacks);
        
        if (damage >= h) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    return low;
}

int main() {
    int n;
    cin >> n;
    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        ll attacks_needed = num_attack(t, h);
        t += attacks_needed;
    }

    cout << t << endl;
}
