```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> h(n);
    for(int i = 0; i < n; i++) cin >> h[i];

    priority_queue<pair<int, int>> pq;
    for(int i = 0; i < n; i++) pq.push({h[i], i});

    int t = 0;
    while(!pq.empty()) {
        auto [health, idx] = pq.top(); pq.pop();
        if(health <= 0) continue;
        
        t++;
        if(t % 3 == 0) health -= 3;
        else health -= 1;
        
        pq.push({health, idx});
    }

    cout << t << endl;
    return 0;
}
```

This solution uses a priority queue (max heap) to always attack the frontmost enemy with the highest remaining health. It then updates the total number of attacks `t` and pushes the updated health back into the priority queue if it's still greater than zero. This approach ensures that we're always attacking the most vulnerable enemy first, which minimizes the total number of attacks needed. The time complexity is O(N log N), where N is the number of enemies, due to the operations on the priority queue.
