#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll simulate(int t, ll h) {
    ll cnt = 0;
    while(h > 0){
        t++;
        if(t % 3 == 0){
            h -= 3;
        } else {
            h -= 1;
        }
        cnt++;
    }
    return cnt;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin >> n;
    ll total_t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        total_t += simulate(total_t, h);
    }
    cout << total_t << '\n';
}
