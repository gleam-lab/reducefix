#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll num_attack(ll t, ll h) {
    ll cnt = 0;
    while (h > 0) {
        t++;
        if (t % 3 == 0) {
            h -= 3;
        } else {
            h--;
        }
        cnt++;
    }
    return cnt;
}

int main() {
    int n;
    cin >> n;
    ll t = 0;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        t += num_attack(t, h);
    }

    cout << t << endl;
}

This solution corrects the logic of the `num_attack` function. Instead of assuming a fixed damage per attack that scales with health, it correctly simulates each attack based on the current time `t`. The main loop continues until the health `h` of the enemy reaches zero or below, and it increments the count of attacks accordingly. The total number of attacks is accumulated in `t`, which is then printed at the end. This approach ensures that the problem constraints are respected and the solution is both correct and efficient.
