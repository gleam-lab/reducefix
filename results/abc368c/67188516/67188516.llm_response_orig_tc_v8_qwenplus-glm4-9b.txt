#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    
    long long T = 0;
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
        if (H[i] % 3 == 0) {
            T += H[i] / 3;
        } else {
            T += (H[i] + 2) / 3;
        }
    }
    
    cout << T << endl;
    
    return 0;
}
