#include <iostream>
#include <vector>

int main() {
    int N;
    std::cin >> N;
    std::vector<int> H(N);
    for(int i = 0; i < N; ++i) {
        std::cin >> H[i];
    }

    long long T = 0;
    while(true) {
        bool any_alive = false;
        for(int i = 0; i < N; ++i) {
            if(H[i] > 0) {
                any_alive = true;
                break;
            }
        }
        if(!any_alive) break;

        ++T;
        for(int i = 0; i < N; ++i) {
            if(H[i] > 0) {
                if(T % 3 == 0) H[i] -= 3;
                else H[i] -= 1;
                break;
            }
        }
    }

    std::cout << T << std::endl;
    return 0;
}

This solution correctly implements the logic described in the problem statement. It iterates through the list of enemies, reducing their health based on whether the current turn number is divisible by 3. The loop continues until no enemies have positive health left. This approach ensures that the problem constraints are respected and the solution is efficient enough to handle large inputs within the given limit.
