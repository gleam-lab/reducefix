```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    int pos = 0; // position in cycle: 0,1 -> -1 damage, 2 -> -3 damage

    for (int i = 0; i < N; ++i) {
        while (H[i] > 0) {
            T++;
            pos = (pos + 1) % 3;
            
            if (pos == 2) { // T is multiple of 3
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            
            // If current enemy is defeated, move to next enemy
            // But we need to handle the case where we might have damaged it enough
            // in one step to finish it
            if (H[i] <= 0) break;
        }
    }

    cout << T << endl;
    
    return 0;
}
```
