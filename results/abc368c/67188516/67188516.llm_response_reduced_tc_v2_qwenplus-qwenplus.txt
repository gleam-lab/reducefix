```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    long long carry = 0; // Tracks how many triple damage attacks are needed in the future

    for (int i = 0; i < N; ++i) {
        // First, apply any carried-over triple-damage attacks that would hit this enemy
        long long use_carry = min(carry, H[i]);
        H[i] -= use_carry;
        carry -= use_carry;

        if (H[i] > 0) {
            // Calculate how many attacks are needed on this enemy
            long long normal = H[i] / 3;
            long long rem = H[i] % 3;

            T += normal * 3;
            carry += normal;

            if (rem == 1) {
                T += 1;
                carry += 2;
            } else if (rem == 2) {
                T += 2;
                carry += 1;
            }
        }

        // Ensure we don't carry more than 2 triple-damage attacks
        carry = min(carry, 2LL);
    }

    cout << T << endl;
    return 0;
}
```
