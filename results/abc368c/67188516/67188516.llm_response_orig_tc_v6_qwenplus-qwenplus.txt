#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    // t3: number of triple attacks (T divisible by 3)
    // t1: number of single attacks (T not divisible by 3)
    long long t3 = 0, t1 = 0;
    
    // Binary search to find the minimal T such that all enemies are defeated
    long long low = 0, high = 3 * 1e18; // Generous upper bound

    while (low < high) {
        long long mid = (low + high) / 2;
        long long attack3 = mid / 3;   // number of attacks at times divisible by 3
        long long attack1 = mid - attack3; // remaining attacks

        bool done = true;
        for (int i = 0; i < N; ++i) {
            long long h = H[i];
            // First use as many triple attacks as possible
            long long need3 = min(t3, attack3);
            h -= need3 * 3;
            attack3 -= need3;

            // Then use single attacks
            long long need1 = min(t1, attack1);
            h -= need1 * 1;
            attack1 -= need1;

            // If still alive, try one more attack (either type)
            if (h > 0) {
                if (attack3 > 0 || attack1 > 0) {
                    if (attack3 > 0) {
                        attack3--;
                        h -= 3;
                    } else {
                        attack1--;
                        h -= 1;
                    }
                }
            }

            if (h > 0) {
                done = false;
                break;
            }

            // Restore attack counts for next iteration
            attack3 += need3;
            attack1 += need1;
        }

        if (done) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }

    cout << low << endl;

    return 0;
}
