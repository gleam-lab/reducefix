#include <iostream>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    long long bonus3 = 0, bonus2 = 0;

    for (int i = 0; i < N; ++i) {
        long long h = H[i];

        // Number of full cycles of 3 attacks: each cycle deals 5 damage (3+1+1)
        long long full_cycles = h / 5;
        T += full_cycles * 3;
        
        // Remaining damage needed after full cycles
        long long rem = h % 5;

        if (rem > 0) {
            // We need to place rem '1-damage' attacks in the next available slots
            // Each attack advances T by 1
            // But we can benefit from any bonus 3-damage attacks that may exist

            // First use up any existing bonus 3-damage opportunities
            long long use_bonus = min(bonus3, rem);
            T += use_bonus;
            bonus3 -= use_bonus;
            rem -= use_bonus;

            // Then use up any existing bonus 2-damage opportunities
            use_bonus = min(bonus2, rem);
            T += use_bonus;
            bonus2 -= use_bonus;
            rem -= use_bonus;

            // For remaining damage, advance time and create new bonuses
            while (rem > 0) {
                T++;
                if (T % 3 == 0) {
                    // This is a triple hit turn - deal 3 damage
                    rem -= 3;
                    bonus3 += 2;  // Next two turns can benefit from this
                    bonus2 -= 2;  // Clean up any excess
                    if (bonus2 < 0) bonus2 = 0;
                } else {
                    // Regular hit - deal 1 damage
                    rem--;
                    bonus3++;
                    bonus2++;
                }
            }
        }

        // Reduce bonuses based on position in sequence
        if (T % 3 == 0) {
            bonus3 = 0;
            bonus2 = 0;
        } else if (T % 3 == 1) {
            bonus3 = max(bonus3 - 2, 0LL);
            bonus2 = max(bonus2 - 1, 0LL);
        } else { // T % 3 == 2
            bonus3 = max(bonus3 - 1, 0LL);
            bonus2 = max(bonus2, 0LL);
        }
    }

    cout << T << endl;
    return 0;
}
