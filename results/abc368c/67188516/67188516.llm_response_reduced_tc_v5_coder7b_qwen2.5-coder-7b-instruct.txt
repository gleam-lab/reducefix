#include <iostream>
#include <vector>
#include <queue>

int main() {
  int N;
  std::cin >> N;
  
  std::priority_queue<int, std::vector<int>, std::greater<int>> pq;
  
  for(int i = 0; i < N; ++i) {
    int h;
    std::cin >> h;
    pq.push(h);
  }
  
  int T = 0;
  while(!pq.empty()) {
    int h = pq.top();
    pq.pop();
    
    ++T;
    if(T % 3 == 0) h -= 3;
    else h -= 1;
    
    if(h > 0) pq.push(h);
  }
  
  std::cout << T << std::endl;
  
  return 0;
}
