#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int i = 0; i < N; ++i) {
        if (H[i] % 3 == 0) {
            T += H[i] / 3; // Directly take the multiples of 3
        } else if (H[i] % 3 == 1) {
            T += (H[i] - 1) / 3; // For health % 3 == 1, we need at least one more to make it a multiple of 3
            T += 1; // Additional hit to decrement health to a multiple of 3
        } else { // H[i] % 3 == 2
            T += (H[i] - 2) / 3; // For health % 3 == 2, we need at least two more to make it a multiple of 3
            T += 2; // Additional hits to decrement health to a multiple of 3
        }
    }
    
    cout << T << endl;
    return 0;
}
