```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }
    
    long long T = 0;
    int enemy_idx = 0;
    
    while (enemy_idx < N) {
        T++;
        
        // Attack the frontmost enemy with health > 0
        if (H[enemy_idx] > 0) {
            if (T % 3 == 0) {
                H[enemy_idx] -= 3;
            } else {
                H[enemy_idx] -= 1;
            }
            
            // If current enemy's health becomes 0 or less, move to next enemy
            if (H[enemy_idx] <= 0) {
                enemy_idx++;
            }
        }
        // If current enemy is already defeated, skip (this shouldn't happen in normal flow)
        else {
            enemy_idx++;
        }
    }
    
    cout << T << endl;
    return 0;
}
```
