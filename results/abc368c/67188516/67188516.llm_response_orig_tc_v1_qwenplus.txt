#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    int pos = 0; // position in cycle: 0,1,2 -> attack power: 1,1,3

    for (int i = 0; i < N; ++i) {
        long long health = H[i];
        
        // Process complete cycles of 3 attacks (1+1+3 = 5 damage)
        long long full_cycles = health / 5;
        T += full_cycles * 3;
        health -= full_cycles * 5;
        
        if (health == 0) {
            continue;
        }
        
        // Handle remaining health (1 to 4)
        if (pos == 0) {
            // Next attack is at T+1 (damage 1), T+2 (damage 1), T+3 (damage 3)
            if (health <= 3) {
                T += health;
                pos = (pos + health) % 3;
            } else { // health == 4
                T += 3; // Use attacks: 1,1,3 (but only need 4 damage total)
                pos = 0; // After 3 attacks, back to position 0
            }
        } else if (pos == 1) {
            // Next attack is at T+1 (damage 3 because T+1 is multiple of 3), T+2 (damage 1), T+3 (damage 1)
            if (health == 1) {
                T += 1;
                pos = 2;
            } else if (health == 2) {
                T += 2;
                pos = 0;
            } else if (health == 3) {
                T += 1;
                pos = 2;
            } else { // health == 4
                T += 2;
                pos = 0;
            }
        } else { // pos == 2
            // Next attack is at T+1 (damage 1), T+2 (damage 3), T+3 (damage 1)
            if (health == 1) {
                T += 1;
                pos = 0;
            } else if (health == 2) {
                T += 2;
                pos = 1;
            } else if (health == 3) {
                T += 2;
                pos = 1;
            } else { // health == 4
                T += 3;
                pos = 0;
            }
        }
    }
    
    cout << T << endl;
    return 0;
}
