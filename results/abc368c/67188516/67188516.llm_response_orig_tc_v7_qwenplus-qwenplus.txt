```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    // Each attack does either 1 or 3 damage depending on T % 3 == 0
    // We simulate the attacks needed per enemy and track the current position in the 3-cycle

    long long T = 0, pos = 0;
    for (int i = 0; i < N; ++i) {
        long long h = H[i];

        if (h == 0) continue;

        // For each h, determine how many of each type of attack is needed:
        // Let x be number of triple-damage attacks (T % 3 == 0), and y be single-damage attacks
        // Total damage: 3x + y >= h
        // We want to minimize total actions x + y while satisfying the above constraint
        // Since we must apply attacks in sequence, we also need to ensure that triple-damage attacks
        // are only used at appropriate positions in the global timeline.

        // We can model this as a greedy simulation

        // We'll simulate until health <= 0
        while (h > 0) {
            T++;
            pos = T % 3;

            if (pos == 0 && h >= 3) {
                h -= 3;
            } else {
                h--;
            }
        }
    }

    cout << T << endl;
}
```
