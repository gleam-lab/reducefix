#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long solve(int N, vector<int>& H) {
    long long ans = 0;
    while(true) {
        bool alive = false;
        for(int i = 0; i < N; i++) {
            if(H[i] > 0) {
                alive = true;
                break;
            }
        }
        if(!alive) break;
        ans++;
        for(int i = 0; i < N; i++) {
            if(H[i] > 0) {
                if(ans % 3 == 0) H[i] -= 3;
                else H[i]--;
                if(H[i] <= 0) H[i] = 0;
            }
        }
    }
    return ans;
}

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i = 0; i < N; i++) cin >> H[i];
    cout << solve(N, H) << endl;
    return 0;
}
