#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool isPossible(long long T, const vector<long long>& H) {
    long long additional = 0;
    for (long long h : H) {
        long long t = T;
        long long triple = t / 3;
        long long single = t - triple;
        if (single + 3 * triple >= h) {
            continue;
        } else {
            return false;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }
    
    long long left = 0;
    long long right = *max_element(H.begin(), H.end()) * 3; // Upper bound
    
    while (left < right) {
        long long mid = left + (right - left) / 2;
        if (isPossible(mid, H)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    
    cout << left << endl;
    
    return 0;
}

### Explanation
1. **Binary Search Setup**: The binary search is initialized with `left` as 0 and `right` as a sufficiently large value (3 times the maximum health of any enemy).
2. **Checking Feasibility**: The `isPossible` function checks if a given T can reduce all enemies' health to 0 or below. For each enemy, it calculates the total damage inflicted by `T` attacks, where:
   - **Triple Attacks**: Every 3rd attack deals 3 damage (counted as `T / 3`).
   - **Single Attacks**: The remaining attacks deal 1 damage each (counted as `T - triple`).
3. **Binary Search Execution**: The search adjusts the range (`left` or `right`) based on whether the current mid value (T) can defeat all enemies. The loop continues until the minimal feasible T is found.
4. **Output Result**: The minimal T found by the binary search is printed, representing the total number of attacks needed to defeat all enemies.

This approach efficiently narrows down the possible values of T using binary search, ensuring optimal performance even for large input sizes.
