```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }

    long long T = 0;
    int current_enemy = 0;

    while (current_enemy < N) {
        T++;
        // Determine damage: -3 if T is multiple of 3, else -1
        long long damage = (T % 3 == 0) ? 3 : 1;

        // Attack the frontmost alive enemy
        H[current_enemy] -= damage;

        // If this enemy's health drops to 0 or below, move to next enemy
        if (H[current_enemy] <= 0) {
            current_enemy++;
        }
    }

    cout << T << endl;
    return 0;
}
```
