```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    while(true) {
        bool any_alive = false;
        for(int i = 0; i < N; ++i) {
            if(H[i] > 0) {
                any_alive = true;
                break;
            }
        }
        if(!any_alive) break;

        ++T;
        for(int i = 0; i < N; ++i) {
            if(H[i] > 0) {
                if(T % 3 == 0) {
                    H[i] -= 3;
                } else {
                    --H[i];
                }
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
