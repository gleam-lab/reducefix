#include <iostream>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    long long cycle3_damage = 0; // Total damage we can apply in full 3-attacks (every 3 turns)
    
    // We'll simulate the process by handling each enemy one by one
    for (int i = 0; i < N; ++i) {
        long long h = H[i];

        // Calculate how many full 3-attack cycles are possible
        long long full_cycles = h / 5;
        T += full_cycles * 3;
        h -= full_cycles * 5;

        // Now handle remaining health (0 to 4)
        if (h == 0) continue;

        // We need to determine when the next triple attack will happen
        long long next_triple = 3 - (T % 3);
        if (next_triple == 3) next_triple = 0;

        if (h <= 2) {
            if (next_triple >= h) {
                T += h;
            } else {
                T += 3; // wait till triple attack, use it for this h
            }
        } else { // h == 3 or 4
            if (next_triple == 0) {
                T += 1; // use triple attack immediately
            } else if (h == 3) {
                T += 3; // wait till next triple attack
            } else { // h == 4
                if (next_triple == 1) {
                    T += 2; // normal attack + triple attack
                } else { // next_triple == 2
                    T += 1; // wait 1 turn for triple attack
                    T += 1; // use triple attack and one more attack
                }
            }
        }
    }

    cout << T << endl;
    return 0;
}
