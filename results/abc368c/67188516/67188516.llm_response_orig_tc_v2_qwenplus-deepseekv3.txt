### Approaches
1. **Simulate the Attack Process**: We need to track the health of each enemy and the current time `T`. For each attack, we either deal 1 or 3 damage based on whether `T` is a multiple of 3.
2. **Efficient Calculation**: Instead of simulating each attack step-by-step (which would be too slow for large inputs), we can calculate the number of full cycles (each cycle consists of 3 attacks: 1, 1, 3 damage) needed for each enemy and then handle the remaining health after these cycles.
3. **Track Position**: The position (or phase) `pos` (T % 3) is crucial because it determines the damage pattern (1, 1, 3) for each enemy. We need to ensure that the timing of attacks aligns correctly with the damage pattern.

    }

    long long T = 0;
    long long pos = 0; // T % 3

    for (int i = 0; i < N; ++i) {
        long long h = H[i];
        if (h == 0) continue;

        // Calculate full cycles (each cycle is 3 attacks: 1,1,3)
        long long full_cycles = h / 5;
        T += full_cycles * 3;
        h %= 5;

        if (h == 0) {
            pos = (pos + full_cycles * 3) % 3;
            continue;
        }

        // Now handle the remaining h (0 < h < 5)
        while (h > 0) {
            T++;
            long long damage = (T % 3 == 0) ? 3 : 1;
            h -= damage;
        }
        pos = T % 3;
    }

    cout << T << endl;

    return 0;
}

#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    long long pos = 0; // T % 3

    for (long long h : H) {
        if (h == 0) continue;

        long long K = h / 5;
        long long rem = h % 5;

        T += K * 3;
        pos = (pos + K * 3) % 3;

        if (rem > 0) {
            long long additional_attacks = 0;
            if (pos == 0) {
                if (rem <= 2) {
                    additional_attacks = rem;
                } else {
                    additional_attacks = 3;
                }
            } else if (pos == 1) {
                if (rem == 1) {
                    additional_attacks = 1;
                } else {
                    additional_attacks = 2;
                }
            } else { // pos == 2
                if (rem <= 3) {
                    additional_attacks = 1;
                } else {
                    additional_attacks = 2;
                }
            }

            T += additional_attacks;
            pos = (pos + additional_attacks) % 3;
        }
    }

    cout << T << endl;

    return 0;
}
