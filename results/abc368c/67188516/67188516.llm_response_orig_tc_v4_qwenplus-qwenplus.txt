#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    long long attack_count = 0; // Number of attacks made so far
    vector<int> attack_mods; // Stores the mod3 of attack times where each enemy is hit

    for (int i = 0; i < N; ++i) {
        long long h = H[i];

        // Determine how many full cycles of 3 attacks we need
        long long full_cycles = h / 5;
        T += full_cycles * 3;
        attack_count += full_cycles * 2;

        long long remaining = h % 5;
        if (remaining == 0) continue;

        // For the remaining health, simulate optimal attacks
        vector<int> attack_times;
        long long curr_time = T + 1;
        while (attack_count >= 0 && attack_times.size() < remaining) {
            if (curr_time % 3 == 0) {
                if (attack_count >= 2) {
                    attack_times.push_back(curr_time);
                    attack_count -= 2;
                } else {
                    attack_count++;
                    curr_time++;
                }
            } else {
                attack_times.push_back(curr_time);
                attack_count++;
                curr_time++;
            }
        }

        T = attack_times.back();
        attack_count = 0; // Reset attack count after processing each enemy
    }

    cout << T << "\n";
    return 0;
}
