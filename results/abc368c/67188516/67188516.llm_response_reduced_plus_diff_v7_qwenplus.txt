```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    int enemy_index = 0;

    while (enemy_index < N) {
        T++;
        // Attack the frontmost enemy with health > 0
        if (H[enemy_index] > 0) {
            if (T % 3 == 0) {
                H[enemy_index] -= 3;
            } else {
                H[enemy_index] -= 1;
            }
            
            // If current enemy's health drops to 0 or below, move to next enemy
            if (H[enemy_index] <= 0) {
                enemy_index++;
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
