```cpp
#include <iostream>
#include <vector>

int main() {
  int N;
  std::cin >> N;
  std::vector<int> H(N);
  for(int i = 0; i < N; ++i) {
    std::cin >> H[i];
  }

  long long T = 0;
  while(true) {
    bool all_zero = true;
    for(int i = 0; i < N; ++i) {
      if(H[i] > 0) {
        all_zero = false;
        if((T + 1) % 3 == 0) {
          H[i] -= 3;
        } else {
          H[i] -= 1;
        }
        ++T;
      }
    }
    if(all_zero) break;
  }

  std::cout << T << std::endl;
  return 0;
}
```

This code reads the number of enemies and their healths, then iteratively attacks them as described in the problem statement, incrementing the time counter `T` each time an attack occurs. The loop continues until all enemies have been defeated (i.e., their healths are all zero). Finally, the total time taken `T` is printed.
