#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i = 0; i < N; i++) {
        cin >> H[i];
    }

    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    for(int i = 0; i < N; i++) {
        pq.push({H[i], i});
    }

    long long time = 0;
    while(!pq.empty()) {
        auto [h, idx] = pq.top();
        pq.pop();

        h -= (time % 3 == 0 ? 3 : 1);
        if(h > 0) {
            pq.push({h, idx});
        }
        time++;
    }

    cout << time << endl;
    return 0;
}

### Explanation:
1. **Priority Queue**: We use a min-heap (priority queue) to always process the smallest health first.
2. **Time Calculation**: For each enemy, we reduce its health based on whether `time` is a multiple of 3.
3. **Loop Until All Enemies Are Defeated**: The loop continues until the heap is empty, meaning all enemies have been defeated.

This approach ensures that we always prioritize defeating the weakest enemy first, which minimizes the total number of attacks needed. This strategy works efficiently even for large inputs due to the logarithmic time complexity of heap operations.
