```cpp
#include<iostream>
#include<vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i = 0; i < N; ++i) {
        cin >> H[i];
    }
    
    long long T = 0;
    for(int i = 0; i < N; ++i) {
        while(H[i] > 0) {
            T++;
            H[i]--;
            if(T % 3 == 0) {
                H[i] -= 2;
            }
        }
    }
    cout << T << endl;
    return 0;
}
```

This solution iterates over each enemy and reduces their health by 1 or 3 based on the value of T. It continues this process until the health of an enemy becomes 0 or less. The total number of operations performed (T) is then printed as the result. This approach ensures that the constraints are met efficiently, even for large inputs.
