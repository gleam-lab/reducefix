#include <iostream>
#include <vector>

using namespace std;
using ll = long long;

int main() {
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    ll T = 0;

    for (int i = 0; i < N; ++i) {
        // Determine how many attacks are needed for this enemy
        ll h = H[i];
        ll attacks = 0;

        // We simulate the attack pattern:
        // Every 3rd attack deals 3 damage, others deal 1 damage
        // So in every group of 3 attacks, total damage is 5 (1+1+3)

        ll full_cycles = h / 5;
        attacks += full_cycles * 3;
        ll remaining_damage = h % 5;

        if (remaining_damage > 0) {
            attacks += 1;
            if (remaining_damage > 1) {
                attacks += 1;
            }
        }

        // Now we need to consider when the previous attacks ended
        // to determine if our current attack timing gives a triple damage earlier
        ll attack_start_mod = T % 3;

        for (ll a = 0; a < attacks; ++a) {
            T++;
            if (T % 3 == 0 && a < attacks - 1) {
                // Triple damage attack
                T++;
            }
        }
    }

    cout << T << endl;
    return 0;
}
