```cpp
#include <iostream>
#include <vector>

using ll = long long;
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (int i = 0; i < N; ++i) {
        // Number of full cycles of 3 attacks: each cycle reduces health by 5 (1+1+3)
        ll full_cycles = H[i] / 5;
        T += full_cycles * 3;
        ll rem = H[i] % 5;

        if (rem == 0) continue;

        // Determine when the enemy will be attacked based on current T
        ll attack_time_mod = T % 3;

        // We need to simulate the remaining attacks needed for this enemy
        // Attack only happens when the enemy is frontmost, so we must advance T one by one
        // and attack until this enemy is defeated
        while (rem > 0) {
            T++;
            attack_time_mod = (attack_time_mod + 1) % 3;
            if (attack_time_mod == 0) {
                rem -= 3;
            } else {
                rem -= 1;
            }
        }
    }

    cout << T << endl;
}
```
