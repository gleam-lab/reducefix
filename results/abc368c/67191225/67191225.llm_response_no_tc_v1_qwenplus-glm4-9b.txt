#include <iostream>
using namespace std;

long long main() {
    long long N;
    cin >> N;
    long long H[N];
    for (long long i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    for (long long i = 0; i < N; ++i) {
        if (H[i] % 3 == 0) {
            T += H[i] / 3;
        } else if (H[i] % 3 == 1) {
            T += (H[i] + 2) / 3;
        } else if (H[i] % 3 == 2) {
            T += (H[i] + 1) / 3;
        }
    }

    cout << T << endl;
    return 0;
}
