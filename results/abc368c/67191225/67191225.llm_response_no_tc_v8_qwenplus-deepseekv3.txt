#include <iostream>
#include <vector>

using namespace std;
using ll = long long;

ll compute_steps(ll h) {
    ll low = 0, high = 2 * h;
    ll steps = 0;
    while (low <= high) {
        ll mid = (low + high) / 2;
        ll damage = (mid / 3) * 3 + (mid - mid / 3);
        if (damage >= h) {
            steps = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return steps;
}

int main() {
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    for (int i = 0; i < N; ++i) {
        ll steps = compute_steps(H[i]);
        T += steps;
    }
    cout << T << endl;
    
    return 0;
}
