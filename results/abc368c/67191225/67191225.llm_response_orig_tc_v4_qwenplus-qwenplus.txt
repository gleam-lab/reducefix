```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    // We'll use a priority queue to always attack the frontmost enemy with health > 0
    // Each element will be a pair of (current health, position)
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;

    // Initialize the priority queue with all enemies
    for (int i = 0; i < N; ++i) {
        if (H[i] > 0) {
            pq.emplace(H[i], i);
        }
    }

    ll T = 0;
    while (!pq.empty()) {
        T++;

        // Find which attacks are possible at this time step
        vector<pair<ll, int>> candidates;
        while (!pq.empty() && pq.top().second <= T) {
            // Only consider enemies that have been reachable by current time
            candidates.push_back(pq.top());
            pq.pop();
        }

        // Put back the ones we didn't choose
        bool attacked = false;
        for (auto [h, pos] : candidates) {
            if (T % 3 == 0) {
                h -= 3;
            } else {
                h -= 1;
            }
            if (h > 0) {
                pq.emplace(h, pos);
            }
            attacked = true;
            break;  // Attack only one enemy per turn
        }

        // If no attack happened, continue (this shouldn't happen)
        if (!attacked) {
            T--;
        }

        // Add remaining candidates back to the priority queue
        for (auto it = next(candidates.begin()); it != candidates.end(); ++it) {
            pq.emplace(it->first, it->second);
        }
    }

    cout << T << endl;
    return 0;
}
```
