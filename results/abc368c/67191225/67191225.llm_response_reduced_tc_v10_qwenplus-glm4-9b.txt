
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int health : H) {
        int full_attacks = health / 3; // Full attacks can be made when T is a multiple of 3
        health -= full_attacks * 3;
        T += full_attacks;
        
        if (health > 0) {
            T++; // Make one more attack to deal with the remaining health
        }
    }
    
    cout << T << endl;
}
```
