```cpp
#include<iostream>
#include<vector>

using namespace std;
using ll = long long;

int main()
{
    int N;
    cin >> N;
    vector<ll> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }

    ll T = 0;
    int current_enemy = 0;

    while (current_enemy < N) {
        T++;
        // Attack the frontmost enemy with health > 0
        if (H[current_enemy] > 0) {
            if (T % 3 == 0) {
                H[current_enemy] -= 3;
            } else {
                H[current_enemy] -= 1;
            }
            // If this enemy is defeated, move to next
            if (H[current_enemy] <= 0) {
                current_enemy++;
            }
        }
        // If current enemy is already defeated, we still attack it (but no effect),
        // but in our logic above, we only attack if health > 0. 
        // However, per problem: "attack the frontmost enemy with health 1 or more"
        // So we don't need to do anything if all front enemies are dead.
    }

    cout << T << endl;
    return 0;
}
```
