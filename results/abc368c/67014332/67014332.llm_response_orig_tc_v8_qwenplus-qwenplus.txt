```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N;
    cin >> N;
    vector<ll> H(N);
    for (ll i = 0; i < N; ++i) {
        cin >> H[i];
    }

    // We will simulate the process efficiently using a priority queue
    // Only attack the first enemy with health > 0 each time

    ll T = 0;
    ll idx = 0; // index of current enemy to attack

    while (idx < N) {
        T++;
        // Check if it's a multiple of 3
        if (T % 3 == 0) {
            H[idx] -= 3;
        } else {
            H[idx]--;
        }

        // Move idx forward while current enemy is dead
        while (idx < N && H[idx] <= 0) {
            idx++;
        }
    }

    cout << T << "\n";
    return 0;
}
```
