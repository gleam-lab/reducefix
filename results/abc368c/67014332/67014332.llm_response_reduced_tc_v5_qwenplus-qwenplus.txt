```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll t = 0, tri = 0; // tri counts how many attacks have been made since last triple attack

    for (ll h : H) {
        // While enemy still has health left
        while (h > 0) {
            t++;
            tri++;

            if (tri == 3) {
                h -= 3;
                tri = 0;
            } else {
                h -= 1;
            }
        }

        // Reset tri counter when moving to the next enemy
        tri = 0;
    }

    cout << t;
    return 0;
}
```
