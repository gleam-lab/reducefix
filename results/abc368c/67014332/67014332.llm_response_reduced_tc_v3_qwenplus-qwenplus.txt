#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n, ans = 0, tri = 1;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        ll h;
        cin >> h;
        
        // Calculate how many full attack cycles of 3 attacks are needed
        // Each cycle of 3 attacks deals 1*2 + 3*1 = 5 damage
        ll cycles = h / 5;
        ans += cycles * 3;
        h %= 5;

        // Handle remaining damage
        if (h > 0) {
            // Need to simulate the next few attacks
            // Determine when we can apply the triple damage
            bool triple_available = false;
            if (tri == 3) {
                triple_available = true;
                ++ans;
                tri = 1;
                h -= 3;
            } else {
                ++ans;
                ++tri;
                --h;
                if (h > 0) {
                    ++ans;
                    ++tri;
                    --h;
                    if (h > 0) {
                        ++ans;
                        tri = 1;
                        h -= 3;
                    }
                }
            }
        }
    }

    cout << ans << endl;
}
