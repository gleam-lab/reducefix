```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> h(n);
    for(int i = 0; i < n; i++) {
        cin >> h[i];
    }
    
    ll ans = 0;
    ll tri = 1;
    
    for(int i = 0; i < n; i++) {
        ll health = h[i];
        
        // Process in chunks of 5: each chunk takes 3 turns (1-1-3 pattern)
        // This gives us 5 damage over 3 turns
        ans += (health / 5) * 3;
        health %= 5;
        
        // Handle remaining health (0 to 4)
        while(health > 0) {
            if(tri == 3) {
                // Every 3rd turn deals 3 damage
                health -= 3;
                tri = 1;
                ans++;
            } else {
                // Other turns deal 1 damage
                health -= 1;
                tri++;
                ans++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
