#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

ll solve(ll x) {
    ll t = 0;
    while(x > 0) {
        if(x % 3 == 0) {
            t += x / 3;
            x /= 3;
        } else {
            x -= 1;
            t++;
        }
    }
    return t;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    ll n;
    cin >> n;
    vector<ll> h(n);
    for(ll &x : h) cin >> x;

    ll ans = 0;
    for(ll x : h) ans += solve(x);

    cout << ans << '\n';

    return 0;
}
