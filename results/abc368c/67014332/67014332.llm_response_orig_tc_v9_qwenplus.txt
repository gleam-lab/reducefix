```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; i++) {
        cin >> h[i];
    }
    
    ll ans = 0;
    ll tri = 1; // tracks current turn modulo 3
    
    for (int i = 0; i < n; i++) {
        ll health = h[i];
        
        // Process full cycles of 5 damage (3+1+1)
        // Each cycle takes 3 turns and deals 5 damage
        ll cycles = health / 5;
        ans += cycles * 3;
        health %= 5;
        
        // Handle remaining health (0 to 4)
        while (health > 0) {
            ans++;
            if (tri == 3) {
                health -= 3;
                tri = 1;
            } else {
                health -= 1;
                tri++;
            }
        }
    }
    
    cout << ans;
    return 0;
}
```
