#include<bits/stdc++.h>
using namespace std;
using ll = long long;
int main() {
    ll n, ans=0, tri=1;
    cin>>n;
    for(int i=0;i<n;i++) {
        ll h;
        cin>>h;
        // Calculate the number of full attacks of 3 and the remainder
        ans += (h + 2) / 3 * 3;
        h %= 3;
        // Reduce health by dealing with the remainder
        while (h > 0) {
            if (tri % 3 == 0) {
                tri = 1;
                h -= 3;
            } else {
                tri++;
                h--;
            }
            ans++;
        }
    }
    cout<<ans;
}
