```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> H(n);
    for (auto& h : H) cin >> h;

    ll T = 0, tri = 1; // Total time, and counter for every 3rd attack

    for (ll h : H) {
        // We simulate the attack sequence for each enemy until its health is reduced to 0 or less
        while (h > 0) {
            if (tri == 3) {
                // Every 3rd second, deal 3 damage
                h -= 3;
                tri = 1;
            } else {
                // Otherwise, deal 1 damage
                h -= 1;
                tri++;
            }
            T++;
        }
        tri = 1; // Reset tri after defeating an enemy
    }

    cout << T << endl;
}
```
