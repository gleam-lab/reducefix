```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
int main() {
    ll n, ans=0;
    cin>>n;
    vector<ll> health(n);
    for(int i=0;i<n;i++) {
        cin>>health[i];
    }
    sort(health.begin(), health.end());
    for(int i=0;i<n;i++) {
        ans+=(health[i]/3)*3;
        if(health[i]%3!=0)
            ans++;
    }
    cout<<ans;
}
```
