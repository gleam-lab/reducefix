```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; i++) {
        cin >> H[i];
    }

    ll T = 0;
    int pos = 0; // current enemy index

    while (pos < n) {
        T++;
        // Determine damage: 3 if T is multiple of 3, else 1
        ll damage = (T % 3 == 0) ? 3 : 1;
        
        // Attack the frontmost enemy with health > 0
        if (H[pos] > 0) {
            H[pos] -= damage;
            // If this enemy is defeated, move to next
            if (H[pos] <= 0) {
                pos++;
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
