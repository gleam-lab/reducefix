```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; i++) {
        cin >> h[i];
    }

    ll T = 0;
    int enemy_idx = 0;

    while (enemy_idx < n) {
        T++;
        // Attack the frontmost enemy with health > 0
        if (h[enemy_idx] > 0) {
            if (T % 3 == 0) {
                h[enemy_idx] -= 3;
            } else {
                h[enemy_idx] -= 1;
            }
        }

        // If current enemy's health drops to 0 or below, move to next enemy
        if (h[enemy_idx] <= 0) {
            enemy_idx++;
        }
    }

    cout << T << endl;
    return 0;
}
```
