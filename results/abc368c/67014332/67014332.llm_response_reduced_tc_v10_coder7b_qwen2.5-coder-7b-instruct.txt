```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n, ans=0, tri=1;
    cin >> n;
    vector<ll> h(n);
    for(ll &x: h) cin >> x;
    
    sort(h.begin(), h.end());
    
    for(ll &x: h) {
        if(x == 0) continue;
        
        ans += (x / 3);
        x %= 3;
        
        while(x > 0) {
            if(tri % 3 == 0) {
                x -= 3;
                ans++;
            } else {
                x--;
                ans++;
            }
            tri++;
        }
    }
    
    cout << ans;
}
```

This solution sorts the array first to always deal with the smallest health first which optimizes the process. It then iterates over each health, reducing it according to the rules specified, and keeps track of the total number of actions performed. The use of sorting and iterating through the array ensures that the solution is efficient even for large inputs.
