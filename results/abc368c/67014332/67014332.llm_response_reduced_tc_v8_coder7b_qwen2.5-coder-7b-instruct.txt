#include<bits/stdc++.h>
using namespace std;
using ll = long long;
int main() {
    ll n, ans=0, tri=1;
    cin>>n;
    vector<ll> h(n);
    for(int i=0;i<n;i++) {
        cin>>h[i];
    }
    sort(h.begin(), h.end());
    for(auto &x:h){
        if(x%3==0){
            ans+=x/3;
        }else{
            ans+=x/3+1;
        }
    }
    cout<<ans;
}
