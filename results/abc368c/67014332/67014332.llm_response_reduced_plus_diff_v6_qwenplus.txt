```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> h(n);
    for(int i = 0; i < n; i++) {
        cin >> h[i];
    }
    
    ll t = 0;
    int enemy_idx = 0;
    
    while (enemy_idx < n) {
        t++;
        // Attack the frontmost enemy with health > 0
        if (h[enemy_idx] > 0) {
            if (t % 3 == 0) {
                h[enemy_idx] -= 3;
            } else {
                h[enemy_idx] -= 1;
            }
            
            // If current enemy's health becomes 0 or less, move to next enemy
            if (h[enemy_idx] <= 0) {
                enemy_idx++;
            }
        }
    }
    
    cout << t << endl;
    return 0;
}
```
