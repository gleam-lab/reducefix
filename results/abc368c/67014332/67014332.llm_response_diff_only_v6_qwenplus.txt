```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; i++) {
        cin >> h[i];
    }

    ll T = 0;
    int enemy_idx = 0;

    while (enemy_idx < n) {
        T++;
        // Determine damage: -3 if T is multiple of 3, else -1
        ll damage = (T % 3 == 0) ? 3 : 1;

        // Attack the frontmost alive enemy
        if (h[enemy_idx] > 0) {
            h[enemy_idx] -= damage;
        }

        // If current enemy's health drops to 0 or below, move to next
        if (h[enemy_idx] <= 0) {
            enemy_idx++;
        }
    }

    cout << T << endl;
    return 0;
}
```
