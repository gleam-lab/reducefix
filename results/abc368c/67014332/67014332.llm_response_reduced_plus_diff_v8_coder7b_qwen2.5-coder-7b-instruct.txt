```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> arr(n);
    for(int i = 0; i < n; ++i){
        cin >> arr[i];
    }

    long long t = 0;
    for(long long &h : arr){
        while(h > 0){
            ++t;
            if(t % 3 == 0) h -= 3;
            else h -= 1;
        }
    }

    cout << t << endl;
    return 0;
}
```
