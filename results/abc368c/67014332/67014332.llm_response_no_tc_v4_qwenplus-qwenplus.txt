```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll n;
    cin >> n;
    vector<ll> H(n);
    for (int i = 0; i < n; ++i) {
        cin >> H[i];
    }

    ll T = 0;
    ll tri = 0; // Count how many attacks have been made since last triple attack

    for (int i = 0; i < n; ++i) {
        ll h = H[i];

        while (h > 0) {
            T++;
            tri++;

            if (tri == 3) {
                h -= 3;
                tri = 0;
            } else {
                h -= 1;
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
