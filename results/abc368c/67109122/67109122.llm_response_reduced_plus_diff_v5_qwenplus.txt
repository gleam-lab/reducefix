#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        // Process the current enemy
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            
            // If the current enemy is defeated, move to next
            if (H[i] <= 0) break;
            
            // Handle overflow from previous enemies: 
            // We need to simulate attacks on subsequent enemies when T increases
            // But note: we only attack the frontmost alive enemy
            // So if current enemy is still alive, we don't touch others
        }
    }
    
    cout << T << endl;
    return 0;
}
