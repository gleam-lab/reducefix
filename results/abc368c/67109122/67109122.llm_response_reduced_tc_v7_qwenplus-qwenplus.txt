#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    for (int i = 0; i < N; ++i) {
        // Determine how many full attack cycles we can apply on this enemy
        // Each cycle is 3 attacks: 1, 1, 3 -> total damage = 5
        // We want to know how many such full cycles we can apply from current T onwards

        // First, align the timing so that triple damage attacks hit this enemy optimally
        // We may need to wait a few turns to get T%3 == 0 to get a x3 attack

        // Number of full attack cycles needed
        long long full_cycles = H[i] / 5;
        long long remaining_health = H[i] % 5;

        T += full_cycles * 3;

        // Now simulate the remaining attacks needed
        while (remaining_health > 0) {
            T++;
            if (T % 3 == 0) {
                remaining_health -= 3;
            } else {
                remaining_health -= 1;
            }
        }
    }

    cout << T << endl;
    return 0;
}
