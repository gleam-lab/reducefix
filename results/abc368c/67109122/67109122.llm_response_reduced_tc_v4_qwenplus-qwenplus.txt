#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;

    for (int i = 0; i < N; ++i) {
        // Number of full attack cycles (3 turns: 3 + 1 + 1 damage)
        long long full_cycles = H[i] / 5;
        long long remaining = H[i] % 5;

        T += full_cycles * 3;

        // Simulate the remaining attacks needed
        while (remaining > 0) {
            if (T % 3 == 0) {
                remaining -= 3;
            } else {
                remaining -= 1;
            }
            T++;
        }
    }

    cout << T << endl;
    return 0;
}
