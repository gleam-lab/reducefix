#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    long long T = 0;
    int i = 0;
    while (i < N) {
        long long damage = 1;
        if (H[i] % 3 == 0) {
            damage = 3;
        }
        while (H[i] > 0) {
            H[i] -= damage;
            T++;
            damage = 3; // Change the damage to 3 for the next hit if T is a multiple of 3
        }
        i++;
    }
    cout << T << endl;
    return 0;
}
