    long long T = 0;
    long long attack = 1; // Start with the smallest possible attack
    
    for (int i = 1; i <= N; i++) {
        while (H[i] > 0) {
            if (T % 3 == 0 && attack >= 3) {
                H[i] -= 3;
                attack = 3;
            } else {
                H[i] -= attack;
                attack = 1;
            }
            T++;
        }
    }

    return 0;
}
