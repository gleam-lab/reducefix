#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 9);
    long long T = 0;
    for (int i = 0; i < N; i++) {
        cin >> H[i];
        while (H[i] > 0) {
            H[i] -= (T % 3 == 0 ? 3 : 1);
            T++;
        }
    }
    cout << T << endl;
    return 0;
}
