#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;

    // We simulate the process from left to right
    for (int i = 0; i < N; ++i) {
        // While current enemy is still alive
        while (H[i] > 0) {
            T++;

            // Check if we can use a triple attack
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i]--;
            }

            // Propagate damage to the next enemies if possible
            int j = i + 1;
            while (j < N && H[j] > 0) {
                if ((T + j - i) % 3 == 0) {
                    H[j] -= 3;
                } else {
                    H[j]--;
                }
                if (H[j] <= 0) {
                    j++;
                } else {
                    break;
                }
            }
        }
    }

    cout << T << endl;
    return 0;
}
