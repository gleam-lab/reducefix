    long long T = 0;
    while (true) {
        bool done = true;
        for (int i = 0; i < N; i++) {
            if (H[i] > 0) {
                done = false;
                if ((T + 1) % 3 == 0) H[i]--;
                else H[i] -= 2;
                T++;
            }
        }
        if (done) break;
    }

    cout << T << endl;
    return 0;
}
