
```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    long long T = 0;

    for (int i = 0; i < N; i++) {
        // Calculate the number of full 3 attacks and the remaining health
        int full3Attacks = H[i] / 3;
        int remainingHealth = H[i] % 3;

        // Add the full 3 attacks to T
        T += full3Attacks * 3;
        
        // If there is remaining health, add the next attack to T
        if (remainingHealth > 0) {
            T++;
        }
    }

    cout << T << endl;
    return 0;
}
```
