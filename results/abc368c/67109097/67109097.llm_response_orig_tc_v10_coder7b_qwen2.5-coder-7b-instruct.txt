#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> h(n);
    for(int i=0;i<n;++i){
        cin>>h[i];
    }
    
    long long t=0;
    while(true){
        bool done=true;
        for(int i=0;i<n;++i){
            if(h[i]>0){
                done=false;
                if((t+1)%3==0){
                    h[i]-=3;
                }else{
                    h[i]--;
                }
            }
        }
        if(done) break;
        ++t;
    }
    
    cout<<t<<endl;
    return 0;
}
