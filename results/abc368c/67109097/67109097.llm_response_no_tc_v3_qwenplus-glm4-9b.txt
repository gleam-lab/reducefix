#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    long long T = 0;

    for (int i = 0; i < N; i++) {
        if (H[i] > 0) {
            // Calculate the number of full rounds needed
            int full_rounds = H[i] / 3;
            T += full_rounds;
            H[i] -= full_rounds * 3;

            // If there is a remainder, handle it
            if (H[i] > 0) {
                // Find the nearest multiple of 3 that is greater than or equal to remaining health
                int remainder = H[i];
                while (remainder % 3 != 0) {
                    remainder--;
                }
                T += remainder;
                H[i] -= remainder;
            }
        }
    }

    cout << T << endl;
    return 0;
}
