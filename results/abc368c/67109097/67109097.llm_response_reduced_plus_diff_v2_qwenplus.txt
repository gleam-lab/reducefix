```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            
            // If current enemy is defeated, move to next one
            if (H[i] <= 0) break;
            
            // Check if there are remaining enemies with positive health before the current one
            bool found = false;
            for (int j = 0; j < i; j++) {
                if (H[j] > 0) {
                    found = true;
                    break;
                }
            }
            if (found) {
                // We need to go back to attack previous enemies
                // Reset to first enemy with positive health
                while (i > 0 && H[i-1] > 0) {
                    i--;
                }
                break;
            }
        }
    }
    
    cout << T << endl;
    return 0;
}
```
