#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 9);
    for (int i = 1; i <= N; i++) cin >> H[i];
    long long T = 0;
    long long sum = 0; // To keep track of cumulative health

    for (int i = 0; i < N; i++) {
        sum += H[i]; // Accumulate the total health of enemies seen so far
        if (sum % 3 == 0) {
            // If the cumulative health is a multiple of 3, we can attack multiple enemies at once
            T += (sum - H[i]) / 3;
            H[i] = 0;
        } else if (sum % 3 == 1) {
            // If the cumulative health modulo 3 is 1, we need to attack until it becomes 0 or 2
            while (sum % 3 == 1) {
                T++;
                sum -= 1; // Decrease sum by 1 to make it a multiple of 3
            }
            // Then we can attack all enemies with remaining health in the sum
            T += sum / 3;
            H[i] = 0;
        } else { // sum % 3 == 2
            // If the cumulative health modulo 3 is 2, we just attack the current enemy with health 2
            T += sum / 3 + 1;
            H[i] = 0;
        }
    }

    cout << T << endl;
    return 0;
}
