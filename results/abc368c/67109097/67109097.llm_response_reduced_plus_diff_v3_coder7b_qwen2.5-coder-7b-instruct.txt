#define trav(a,x) for(auto&a:x)
const ll mod=1e9+7;
ll n,h[200001],t=0;
void solve(){
    cin>>n;
    rep(i,0,n){
        cin>>h[i];
    }
    while(true){
        bool done=true;
        rep(i,0,n){
            if(h[i]>0){
                done=false;
                t++;
                h[i]-=(t%3==0)?3:1;
            }
        }
        if(done) break;
    }
    cout<<t<<endl;
}
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int tc=1;
    //cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
