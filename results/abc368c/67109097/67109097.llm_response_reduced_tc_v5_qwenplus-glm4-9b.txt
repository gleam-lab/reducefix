
```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int &h : H) cin >> h;

    long long T = 0;
    long long current = 0; // This will track the cumulative health we need to deal with

    for (int i = 0; i < N; ++i) {
        current += H[i];
        // While current health is exactly a multiple of 3, we can deal with it in a single hit of T
        while (current % 3 == 0) {
            current /= 3;
            T += 3;
        }
        // If current health is not a multiple of 3, attack it once
        current--;
        T++;
    }

    cout << T << endl;
    return 0;
}
```
