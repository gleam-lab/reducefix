```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i=0; i<N; i++) cin >> H[i]; // Adjusted index to start from 0
    
    long long T = 0;
    for(int i=0; i<N; i++) {
        if(H[i] % 3 == 0) { // Directly reduce by 3 if divisible by 3
            T += H[i] / 3;
        } else {
            T += H[i] / 3 + 1; // Otherwise, reduce by 3 as many times as possible and then once more
        }
    }
    
    cout << T << endl;
    return 0;
}
```
