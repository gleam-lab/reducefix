```cpp
#include<bits/stdc++.h>
#define mod 1000000007

using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> h(n);
    for(int i=0;i<n;i++){
        cin >> h[i];
    }
    
    long long t = 0;
    for(int i=0;i<n;i++){
        if(h[i]==0) continue;
        
        if((t+1)%3==0){
            t+=h[i]/3;
            if(h[i]%3!=0) t++;
        }else{
            t+=(h[i]+2)/3;
        }
    }
    cout<<t%mod<<endl;
    return 0;
}
```
