#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 9);
    for (int i = 1; i <= N; i++) cin >> H[i];
    long long T = 0;
    for (int i = 1; i <= N; i++) {
        long long H_i = H[i];
        long long a = H_i / 3;
        if (H_i % 3 == 0) {
            T += a;
        } else if (H_i % 3 == 1) {
            T += a + 2;  // Since 3*1 + 1*1 = 4, we add one more than a, and decrease it by 2 in the next step
        } else {  // H_i % 3 == 2
            T += a + 1;  // Since 3*1 + 2*1 = 5, we add one more than a, and decrease it by 1 in the next step
            T++;
        }
    }
    cout << T << endl;
    return 0;
}
