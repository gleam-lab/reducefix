```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;

    for (int i = 0; i < N; ++i) {
        // Number of full attack cycles needed: each cycle is 3 attacks (1 at T%3==0, and 2 at T%3!=0)
        long long full_cycles = H[i] / 5;
        long long remainder = H[i] % 5;

        T += full_cycles * 3;

        if (remainder > 0) {
            T += 1; // One more attack to finish remaining health <= 3
            if (remainder > 1) {
                T += 1; // Second attack needed
                if (remainder == 3) {
                    T += 1; // Third attack needed for remainder 3
                }
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
