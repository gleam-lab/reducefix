```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; ++i) {
        cin >> H[i];
    }

    long long T = 0;
    for (int i = 0; i < N; ++i) {
        // For each enemy, calculate how many full cycles of attacks are needed
        // Each cycle consists of:
        // - Ticks not multiple of 3: two attacks reducing health by 1
        // - Tick is multiple of 3: one attack reducing health by 3
        // Total reduction per cycle: 5 health

        long long h = H[i];
        if (h == 0) continue;

        // Compute number of full cycles
        long long full_cycles = h / 5;
        long long remaining_health = h % 5;

        T += full_cycles * 3; // 3 ticks per full cycle

        // Handle remaining health
        if (remaining_health > 0) {
            if (T % 3 == 0) {
                // If current T is multiple of 3, we can immediately use a triple damage tick
                T++;
            } else {
                // We may need to wait 1 or 2 ticks to get to a multiple of 3
                long long wait = (3 - (T % 3));
                T += wait;
                // But we might not need all the wait ticks
                if (wait >= remaining_health) {
                    T -= (wait - remaining_health);
                } else {
                    T += remaining_health - wait;
                }
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
