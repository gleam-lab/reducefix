#include<bits/stdc++.h>

using namespace std;

long long calculate_k(long long T, long long H_i) {
    long long low = 0, high = H_i;
    long long answer = H_i;  // at worst, H_i attacks (each dealing 1 damage)
    while (low <= high) {
        long long mid = (low + high) / 2;
        long long m = (T + mid) / 3 - T / 3;
        if (2 * m + mid >= H_i) {
            answer = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return answer;
}

    cin.tie(nullptr);
    
    int N;

    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        long long k = calculate_k(T, H[i]);
        T += k;
