```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for(int i = 0; i < N; i++) cin >> H[i];
    long long T = 0;
    priority_queue<int, vector<int>, greater<int>> pq(H.begin(), H.end());
    
    while(!pq.empty()) {
        int top = pq.top();
        pq.pop();
        
        if(top % 3 == 0) T += top / 3;
        else T += top / 3 + 1;
    }
    
    cout << T << endl;
    return 0;
}
```
