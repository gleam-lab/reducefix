#include<bits/stdc++.h>

    long long T = 0;
    
    while(true){
        bool any = false;
        for(int i=1;i<=N;i++){
            if(H[i]>0){
                any = true;
                T++;
                if(T%3==0) H[i]-=3;
                else H[i]--;
            }
        }
        if(!any) break;
    }

    return 0;
}
