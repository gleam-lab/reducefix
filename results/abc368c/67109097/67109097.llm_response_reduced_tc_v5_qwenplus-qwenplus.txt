```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }

    long long T = 0;
    for (int i = 0; i < N; i++) {
        // For each enemy, determine how many triple attacks and single attacks are needed
        long long h = H[i];
        // Each cycle of 3 attacks can reduce health by 5 (1+1+3)
        long long full_cycles = h / 5;
        long long remaining = h % 5;
        T += full_cycles * 3;
        
        // Handle remaining health
        if (remaining > 0) {
            if (remaining <= 2) {
                T += remaining; // Only single attacks needed
            } else {
                T += 3; // One triple attack (3) or combination like 1+3 or 3+1
            }
        }
    }

    cout << T << endl;
    return 0;
}
```
