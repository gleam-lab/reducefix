#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 9);
    for (int i = 1; i <= N; i++) cin >> H[i];
    long long T = 0;
    
    // Convert health values to minimum attacks needed up to next multiple of 3
    for (int i = 0; i < N; i++) {
        int health = H[i];
        int remainder = health % 3;
        if (remainder == 0) {
            H[i] = 0;
        } else if (remainder == 1) {
            H[i] = health - 1;
            T++;  // Increment T since we took an extra attack
        } else {  // remainder == 2
            H[i] = health - 2;
            T++;  // Increment T since we took an extra attack
        }
    }
    
    // Sum up all attacks needed to bring health to 0
    for (int i = 0; i < N; i++) {
        T += (H[i] + 2) / 3;  // Each remaining health point can be reduced in full groups of 3
    }
    
    cout << T << endl;
    return 0;
}
