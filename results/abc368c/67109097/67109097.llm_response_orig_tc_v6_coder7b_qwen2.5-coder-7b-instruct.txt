#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N + 1);
    for (int i = 1; i <= N; i++) cin >> H[i];
    long long T = 0;
    while (true) {
        bool alive = false;
        for (int i = 1; i <= N; i++) {
            if (H[i] > 0) {
                alive = true;
                if ((T + 1) % 3 == 0) H[i] -= 3;
                else H[i]--;
                if (H[i] < 0) H[i] = 0;
                T++;
            }
        }
        if (!alive) break;
    }
    cout << T << endl;
    return 0;
}
