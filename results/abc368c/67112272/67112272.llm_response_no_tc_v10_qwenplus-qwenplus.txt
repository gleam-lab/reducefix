```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];

    ll T = 0;

    // We simulate the process using a queue
    queue<int> q;
    rep(i, n) q.push(h[i]);

    while (!q.empty()) {
        T++;
        // If T is a multiple of 3, deal 3 damage, else deal 1 damage
        ll damage = (T % 3 == 0) ? 3 : 1;

        int current = q.front();
        current -= damage;

        if (current <= 0) {
            q.pop();
        } else {
            q.front() = current;
        }
    }

    cout << T << endl;
}
```
