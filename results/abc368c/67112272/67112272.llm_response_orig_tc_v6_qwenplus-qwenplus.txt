#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); i++)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];

    // We simulate the process with binary search to find minimal T such that all enemies are dead
    auto is_ok = [&](ll T) -> bool {
        vector<ll> remaining(n);
        rep(i, n) remaining[i] = h[i];

        // For each attack at time t from 1 to T
        // enemy index starts at 0
        ll attacks_at_3 = T / 3; // number of triple damage attacks
        ll attacks_not_3 = T - 2 * attacks_at_3; // number of single damage attacks

        // Each enemy can be hit by:
        // - attacks_not_3: every time not multiple of 3
        // - attacks_at_3: every time T divisible by 3

        // total damage per attack round: 1 + 1 + 3 = 5 over 3 steps

        // To distribute the attacks, we simulate how much damage can be applied

        // We use a priority queue or greedy approach to apply most damaging attacks first

        // Simulate applying the attacks:
        // We will use a greedy strategy:
        // For each enemy, compute how many times it's hit by triple and normal attacks

        ll remaining_triple = attacks_at_3;
        ll remaining_single = attacks_not_3;

        rep(i, n) {
            ll health = remaining[i];
            if (health <= 0) continue;

            // Try to kill this enemy using best possible attacks

            // First try triple attacks (damage 3)
            ll needed_triple = (health + 2) / 3; // ceil division
            if (needed_triple <= remaining_triple) {
                remaining_triple -= needed_triple;
                health -= needed_triple * 3;
            } else {
                health -= remaining_triple * 3;
                remaining_triple = 0;
            }

            if (health <= 0) continue;

            // Then try single attacks (damage 1)
            ll needed_single = health;
            if (needed_single <= remaining_single) {
                remaining_single -= needed_single;
                health = 0;
            } else {
                health -= remaining_single;
                remaining_single = 0;
            }

            if (health > 0) return false;
        }

        return true;
    };

    // Binary search on answer T
    ll low = 0;
    ll high = 3e18; // large enough upper bound

    while (high - low > 1) {
        ll mid = (low + high) / 2;
        if (is_ok(mid))
            high = mid;
        else
            low = mid;
    }

    cout << high << endl;

    return 0;
}
