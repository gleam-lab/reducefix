#include<bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); i++)
using ll = long long;

int main() {
    int n; cin >> n;
    vector<ll> h(n);
    ll total_health = 0, ans = 0;

    rep(i, n) {
        cin >> h[i];
        total_health += h[i];
    }
    
    // Calculate the minimum number of attacks needed
    ans = total_health;

    // Adjust for cases where the total health is not a multiple of 3
    if (total_health % 3 != 0) {
        total_health -= 1;
    }
    
    // Make the total health a multiple of 3
    while (total_health % 3 != 0) {
        total_health -= 1;
    }

    // Calculate the number of full cycles of 3 and the remainder
    ll full_cycles = total_health / 3;
    ll remainder = total_health % 3;
    
    // Add the number of full cycles
    ans += full_cycles * 3;
    
    // Add the number of attacks needed for the remainder
    if (remainder == 1) {
        ans += 1; // One attack to reduce health by 1
        remainder = 2; // Now we need 2 more to reach the next multiple of 3
    }
    if (remainder == 2) {
        ans += 2; // Two attacks to reduce health by 2
    }

    cout << ans << endl;
}
