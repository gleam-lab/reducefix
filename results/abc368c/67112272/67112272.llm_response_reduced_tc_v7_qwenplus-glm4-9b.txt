#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0; i < (n); i++)
using ll = long long;

int main() {
    int n; cin >> n;
    ll ans = 0;
    vector<ll> h(n);
    rep(i, n) {
        cin >> h[i];
        // For each enemy, calculate the number of full attacks needed based on their health
        // Each full attack deals 3 damage if T is a multiple of 3, otherwise 1 damage
        // Since the problem states that T increases each time, we can calculate the total damage
        // needed by multiplying the health by 3 and then adding the remainder when divided by 3
        ans += (h[i] + 2) / 3 * 3; // Adds the largest multiple of 3
        ans += (h[i] + 2) % 3;     // Adds the remainder, which will be 0, 1, or 2
    }
    cout << ans << endl;
}
