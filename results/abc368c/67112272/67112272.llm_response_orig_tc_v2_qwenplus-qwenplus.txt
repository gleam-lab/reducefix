#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); i++)
using ll = long long;

// Function to determine number of attacks needed for one enemy
ll compute_attacks(ll health) {
    // Binary search to find minimum T such that total damage >= health
    ll low = 1, high = 2 * health; // upper bound is safe

    while (low < high) {
        ll mid = (low + high) / 2;
        // Compute how many triple attacks we can have in `mid` steps
        ll triple_attacks = mid / 3;
        ll total_damage = triple_attacks * 3 + (mid - triple_attacks) * 1;
        if (total_damage >= health)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];

    ll ans = 0;
    for (ll health : h) {
        ans += compute_attacks(health);
    }

    cout << ans << endl;
}
