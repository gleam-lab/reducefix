```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); i++)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;

    // Priority queue to simulate the attack pattern correctly
    // We'll use a min-heap to process enemies needing attention earliest
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<>> pq;

    // Initialize: next time we can attack each enemy (first attack always at T=1)
    rep(i, n) {
        if (H[i] > 0) {
            pq.emplace(1, i);  // (time to attack, enemy index)
        }
    }

    while (!pq.empty()) {
        auto [t, idx] = pq.top();
        pq.pop();

        // Simulate the attack at time t
        H[idx] -= (t % 3 == 0) ? 3 : 1;
        T = max(T, t);

        // If enemy still has health left, schedule next attack
        if (H[idx] > 0) {
            pq.emplace(t + 1, idx);
        }
    }

    cout << T << endl;
    return 0;
}
```
