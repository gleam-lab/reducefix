#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0; i < (n); i++)
using ll = long long;

int main()
{
    ll n; cin >> n;
    ll t = 0;  // Initialize T to 0
    vector<ll> h(n);
    rep(i, n) {
        cin >> h[i];
        // Calculate the number of full attacks needed to bring the enemy's health to 0
        // and the remainder after these full attacks
        ll full_attacks = h[i] / 3;
        ll remainder = h[i] % 3;
        
        // Increase T by the number of full attacks
        t += full_attacks;
        
        // If there is a remainder, we need to attack once more
        if (remainder > 0) {
            // If the remaining health is greater than 2, it's more efficient to do two attacks
            // of 1 damage each, rather than one of 3 damage, unless the remaining health is 2
            // in which case we are forced to do two attacks of 1 damage each
            if (remainder == 2 || full_attacks == 0) {
                t += 2;
            } else {
                t += 1;
            }
        }
    }
    cout << t << endl;
}
