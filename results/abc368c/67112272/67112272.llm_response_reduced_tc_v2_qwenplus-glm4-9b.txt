#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main() {
    int n; cin >> n;
    ll ans = 0;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];
    
    for (ll i = 0; i < n; i++) {
        if (h[i] % 3 == 0) {
            ans += h[i] / 3;
        } else if (h[i] % 3 == 1) {
            ans += (h[i] - 1) / 3 + 1;
        } else { // h[i] % 3 == 2
            ans += h[i] / 3;
            ans += 1;
        }
    }
    
    cout << ans << endl;
}
