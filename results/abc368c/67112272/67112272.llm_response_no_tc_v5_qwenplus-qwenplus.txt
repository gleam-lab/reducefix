#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    ll n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];

    ll T = 0;
    for (ll i = 0; i < n; ++i) {
        // Determine how many attacks are needed for this enemy
        ll attacks_needed = 0;
        ll health = h[i];

        // We simulate until health is 0 or less
        while (health > 0) {
            T++;
            if (T % 3 == 0) {
                health -= 3;
            } else {
                health -= 1;
            }
        }
    }

    cout << T << endl;
    return 0;
}
