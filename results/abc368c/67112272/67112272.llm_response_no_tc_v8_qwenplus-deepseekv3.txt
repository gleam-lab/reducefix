### Approaches
The correct approach involves:
1. **Understanding the Attack Pattern**: The attacks alternate between dealing 1 damage (on turns not divisible by 3) and 3 damage (on turns divisible by 3). The key observation is that every 3 turns (T = 3, 6, 9, ...), the attack deals 3 damage instead of 1.
2. **Efficient Calculation**: For each enemy, the number of attacks needed to reduce its health to zero or less can be computed by considering that every 3 turns, the total damage is 1 + 1 + 3 = 5 over 3 turns. This means for every full cycle of 3 turns, 5 damage is dealt. The remaining health after these cycles can be handled by checking if the remaining health requires additional turns (1 or 2) beyond the full cycles.
3. **Synchronizing Attacks Across Enemies**: The total time T is the sum of individual times taken to defeat each enemy, but we must account for the fact that defeating an enemy might involve overlapping or sequential attacks. Specifically, the total T is the maximum of the individual times required for each enemy, but this is not straightforward because the attacks are sequential. Instead, we need to compute the minimal T such that the cumulative attacks (with their respective damages) reduce all enemies' health to zero or below.

        ll mid = (low + high) / 2;
        ll total = 0;
        for (int i = 0; i < n; ++i) {
            ll remaining = h[i];
            // The number of full 3-turn cycles (each dealing 5 damage in 3 turns)
            ll cycles = min(mid / 3, remaining / 5);
            remaining -= cycles * 5;

                } else {
                    remaining -= 1;
                }

                break;
            }
        }
        if (total <= mid) {
            ans = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

#include<bits/stdc++.h>
using namespace std;

using ll = long long;

ll compute_min_T(ll H) {
    ll min_T = LONG_LONG_MAX;
    for (ll r = 0; r <= 2; ++r) {
        ll remaining = H - r;
        ll k = (remaining <= 0) ? 0 : (remaining + 4) / 5; // ceil(remaining /5)
        ll T = 3 * k + r;
        if (T >= 0 && T + 2 * (T / 3) >= H) {
            min_T = min(min_T, T);
        }
    }
    return min_T;
}

int main() {
    int n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; ++i) {
        cin >> h[i];
    }
    
    ll total_T = 0;
    for (int i = 0; i < n; ++i) {
        ll H = h[i];
        ll min_T = LONG_LONG_MAX;
        for (ll r = 0; r <= 2; ++r) {
            ll remaining = H - r;
            ll k = (remaining <= 0) ? 0 : (remaining + 4) / 5;
            ll T = 3 * k + r;
            if (T + 2 * (T / 3) >= H) {
                min_T = min(min_T, T);
            }
        }
        total_T += min_T;
    }
    
    cout << total_T << endl;
    return 0;
}
