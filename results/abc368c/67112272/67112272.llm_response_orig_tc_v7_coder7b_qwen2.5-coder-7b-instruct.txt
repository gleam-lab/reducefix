```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n; cin >> n;
    vector<ll> h(n);
    rep(i,n){
        cin >> h[i];
    }

    ll ans = 0;
    while(true){
        bool alive = false;
        rep(i,n){
            if(h[i] > 0){
                alive = true;
                break;
            }
        }
        if(!alive) break;

        ans++;
        rep(i,n){
            if(h[i] > 0){
                if(ans % 3 == 0) h[i] -= 3;
                else h[i]--;
            }
        }
    }

    cout << ans << endl;
    return 0;
}
```
