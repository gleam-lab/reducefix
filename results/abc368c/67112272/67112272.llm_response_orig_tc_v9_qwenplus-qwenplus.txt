#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (ll i = 0; i < (n); ++i)
using ll = long long;

// Function to simulate the attack pattern and calculate total T needed
ll compute_time(ll h) {
    // Binary search bounds
    ll low = 0, high = 2 * h; // upper bound generous

    while (low < high) {
        ll mid = (low + high) / 2;
        // Compute how many enemies can be killed in `mid` time steps
        ll damage = mid / 3 * 5 + (mid % 3);
        if (damage >= h)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll ans = 0;
    for (ll h : H) {
        ans += compute_time(h);
    }

    cout << ans << endl;
}
