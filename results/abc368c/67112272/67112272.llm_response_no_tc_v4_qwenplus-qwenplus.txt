```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long64_t;

int main() {
    int n;
    cin >> n;
    vector<ll> h(n);
    for (int i = 0; i < n; ++i) {
        cin >> h[i];
    }

    ll time = 0;
    // We simulate the process using a priority queue (max heap)
    // to always attack the frontmost enemy that still has HP > 0.
    // We use a max-heap to prioritize enemies that need more attacks.

    // However, we must maintain the order of enemies.
    // So we simulate the process in rounds.

    // Instead of simulating every single attack step,
    // we can calculate how many full cycles of 3 attacks are needed,
    // and then handle remaining damage.

    // Each cycle of 3 seconds:
    // - 2 attacks do 1 damage each
    // - 1 attack does 3 damage
    // Total: 5 damage per 3 seconds

    // For each enemy:
    // We compute how many full 5-damage cycles are needed,
    // and how much extra damage is required after that.

    for (int i = 0; i < n; ++i) {
        // Compute number of full 3-second cycles needed
        ll full_cycles = h[i] / 5;
        if (h[i] % 5 != 0) full_cycles++;

        time += full_cycles * 3;
    }

    cout << time << endl;
    return 0;
}
```
