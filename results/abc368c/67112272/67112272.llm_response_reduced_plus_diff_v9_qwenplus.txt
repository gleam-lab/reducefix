```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> h(n);
    rep(i,n) cin >> h[i];
    
    ll T = 0;
    int idx = 0;
    
    while (idx < n) {
        T++;
        // Attack the frontmost enemy with health > 0
        if (h[idx] > 0) {
            if (T % 3 == 0) {
                h[idx] -= 3;
            } else {
                h[idx] -= 1;
            }
            // If current enemy's health becomes 0 or less, move to next enemy
            if (h[idx] <= 0) {
                idx++;
            }
        }
        // If we've processed all enemies, break
        if (idx >= n) break;
    }
    
    cout << T << endl;
}
```
