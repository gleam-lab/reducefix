#include<bits/stdc++.h>

using ll = long long;

    int n; cin >> n;
    ll ans = 0;
    vector<ll> h(n);
    
    rep(i, n) {
        cin >> h[i];
        // Calculate the number of full 3's and the remainder when divided by 3
        ll fullThrees = h[i] / 3;
        ll remainder = h[i] % 3;
        
        // Each full 3 requires 3 hits, so add that to the answer
        ans += fullThrees * 3;
        
        // The remainder will determine if we need to add more hits
        if (remainder == 0) {
            // If remainder is 0, no additional hits are needed
        } else if (remainder == 1) {
            // If remainder is 1, we can attack once without any effect, but we need to attack twice
            ans += 2;
        } else { // remainder == 2
            // If remainder is 2, we can attack twice to get to 3, which requires an additional hit
            ans += 1;
        }
    }
    
    cout << ans << endl;
}
