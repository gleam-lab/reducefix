```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> h(n);
    for(int i = 0; i < n; ++i) cin >> h[i];

    long long t = 0;
    while(true){
        bool all_zero = true;
        for(int i = 0; i < n; ++i){
            if(h[i] > 0){
                all_zero = false;
                if((t+1)%3 == 0) h[i] -= 3;
                else h[i] -= 1;
                t++;
            }
        }
        if(all_zero) break;
    }

    cout << t << '\n';
    return 0;
}
```
