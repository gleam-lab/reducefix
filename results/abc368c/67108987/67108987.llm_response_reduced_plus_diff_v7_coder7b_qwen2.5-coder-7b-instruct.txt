int main() {

    vector<long long> H(N);
    for(int i=0;i<N;i++)
        cin>>H[i];

    long long T = 0;
    while(1){
        bool flag = false;
        for(int i=0;i<N;i++){
            if(H[i]>0){
                if((T+1)%3==0)
                    H[i]-=3;
                else
                    H[i]--;
                T++;
                if(H[i]<=0)
                    flag = true;
            }
        }
        if(!flag)

    }
    cout<<T<<endl;
    return 0;
}
