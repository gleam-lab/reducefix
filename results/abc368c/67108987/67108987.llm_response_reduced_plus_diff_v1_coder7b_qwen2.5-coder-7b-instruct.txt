```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N + 1);
    for (int i = 1; i <= N; ++i) cin >> H[i];

    long long T = 0;
    while (true) {
        bool changed = false;
        for (int i = 1; i <= N; ++i) {
            if (H[i] > 0) {
                --T;
                if (T % 3 == 0) H[i] -= 3;
                else H[i] -= 1;
                if (H[i] <= 0) --N;
                changed = true;
                break;
            }
        }
        if (!changed) break;
    }

    cout << T << endl;
    return 0;
}
```

This solution iteratively decreases the health of each enemy based on the current value of `T` and checks if any enemy's health becomes zero or negative. If so, it decreases the count of remaining enemies and continues until no enemies have remaining health. The loop breaks once there are no changes made in a full iteration, indicating that all enemies' healths have been reduced to zero or less.
