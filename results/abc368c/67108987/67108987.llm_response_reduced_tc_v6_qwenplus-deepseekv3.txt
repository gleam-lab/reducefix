using namespace std;

long long compute_min_attacks(long long S_mod, long long H_i) {
    long long left = 0;
    long long right = H_i;
    long long answer = H_i;
    while (left <= right) {
        long long mid = left + (right - left) / 2;
        long long k = (S_mod + mid) / 3;
        long long damage = 2 * k + mid;
        if (damage >= H_i) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    return answer;
}

int main() {
    ios_base::sync_with_stdio(false);

    int N;

    }

    long long total_T = 0;
    for (int i = 0; i < N; i++) {
        long long S = total_T;
        long long S_mod = S % 3;
        long long T_i = compute_min_attacks(S_mod, H[i]);
        total_T += T_i;
