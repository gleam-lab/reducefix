```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 1);
    for (int i = 1; i <= N; i++) cin >> H[i];
    long long T = 0;
    while(true){
        bool flag=true;
        for(int i=1;i<=N;i++){
            if(H[i]>0){
                flag=false;
                T++;
                if(T%3==0)
                    H[i]-=3;
                else
                    H[i]--;
            }
        }
        if(flag) break;
    }
    cout<<T<<endl;
    return 0;
}
```

This program works as follows:
1. It takes input for number of enemies 'N' and their respective healths stored in array 'H'.
2. A loop continues until no enemy is left alive.
3. Inside the loop, it checks each enemy from the first one to the last.
4. If an enemy is still alive, it increases the time counter 'T' by 1 and reduces its health by 1 if 'T' is not a multiple of 3, otherwise, it reduces its health by 3.
5. The loop breaks when there are no enemies left alive.
6. Finally, it prints the total time taken 'T'.
