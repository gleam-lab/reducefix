#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        // Start with the maximum decrement, which is 3, and apply it as many times as possible
        T += H[i] / 3;
        H[i] %= 3;
        // Then apply the remaining decrement of 1
        T += H[i];
    }
    
    cout << T << endl;
    return 0;
}
