#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int INF=1e9+7;
struct node{
    int l,r,t;
};
node tr[400010];
void pushup(int u){
    tr[u].t=tr[u<<1].t+tr[u<<1|1].t;
}
void build(int u,int L,int R){
    tr[u]={L,R,0};
    if(L==R)return ;
    int mid=(L+R)>>1;
    build(u<<1,L,mid),build(u<<1|1,mid+1,R);
    pushup(u);
}
void update(int u,int x,int y){
    if(tr[u].l==x&&tr[u].r==y){
        tr[u].t++;
        return ;
    }
    int mid=(tr[u].l+tr[u].r)>>1;
    if(x<=mid)update(u<<1,x,y);
    else update(u<<1|1,x,y);
    pushup(u);
}
ll query(int u,int L,int R){
    if(tr[u].l==L&&tr[u].r==R)return tr[u].t;
    int mid=(tr[u].l+tr[u].r)>>1;
    if(R<=mid)return query(u<<1,L,R);
    if(L>mid)return query(u<<1|1,L,R);
    return query(u<<1,L,mid)+query(u<<1|1,mid+1,R);
}
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    int n,m;
    cin>>n;
    vector<ll> h(n+1);
    for(int i=1;i<=n;i++)cin>>h[i];
    build(1,1,n);
    ll ans=0;
    for(int i=n;i>=1;i--){
        int t=query(1,1,i-1);
        ans+=t;
        if(t&1)ans+=(h[i]-t/2)*2;
        else ans+=(h[i]-t/2)*2-1;
        update(1,i,h[i]);
    }
    cout<<ans<<endl;
    return 0;
}
