```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N + 1); // Adjusted index to match problem statement
    for(int i = 1; i <= N; ++i) cin >> H[i];
    
    long long T = 0;
    while(true){
        bool allZero = true;
        for(int i = 1; i <= N; ++i){
            if(H[i] > 0){
                allZero = false;
                ++T;
                if(T % 3 == 0) H[i] -= 3;
                else H[i] -= 1;
            }
        }
        if(allZero) break;
    }
    cout << T << endl;
    return 0;
}
```

This solution corrects the logic by ensuring that each enemy is only attacked once per turn, and it handles the case where an enemy's health might decrease by 3 units if T is a multiple of 3. The use of a `while` loop ensures that the process continues until all enemies' health drops to zero or below.
