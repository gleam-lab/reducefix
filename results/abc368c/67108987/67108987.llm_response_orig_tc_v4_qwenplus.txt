```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    
    long long T = 0;
    int i = 0;
    
    while (i < N) {
        // Skip enemies that are already dead
        if (H[i] <= 0) {
            i++;
            continue;
        }
        
        T++;
        
        // Check if current T is multiple of 3
        if (T % 3 == 0) {
            H[i] -= 3;
        } else {
            H[i] -= 1;
        }
        
        // If enemy is defeated, move to next enemy
        if (H[i] <= 0) {
            i++;
        }
    }
    
    cout << T << endl;
    return 0;
}
```
