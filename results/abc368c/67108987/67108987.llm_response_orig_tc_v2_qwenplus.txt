```cpp
#include<bits/stdc++.h>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        // Process each enemy
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            // If current enemy is defeated, move to next
            if (H[i] <= 0) break;
            
            // Skip enemies with health <= 0 at the front
            int j = i + 1;
            while (j < N && H[j] <= 0) j++;
            if (j >= N) break;
            
            // Move to next alive enemy
            i = j - 1; // i will be incremented in the loop
            break;
        }
    }
    
    cout << T << endl;
    return 0;
}
```
