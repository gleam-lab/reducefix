```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> h(n+1);
    for(int i=1;i<=n;i++) cin >> h[i];
    long long t = 0;
    for(int i=1;i<=n;i++){
        while(h[i]>0){
            t++;
            if(t%3==0) h[i]-=3;
            else h[i]--;
        }
    }
    cout << t << endl;
    return 0;
}
```
