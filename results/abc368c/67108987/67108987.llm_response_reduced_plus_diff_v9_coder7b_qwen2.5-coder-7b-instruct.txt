```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];

    long long T = 0;
    bool changed = true;

    while (changed) {
        changed = false;
        for (int i = 0; i < N; i++) {
            if (H[i] > 0) {
                changed = true;
                if (T % 3 == 0) {
                    H[i] -= 3;
                } else {
                    H[i]--;
                }
                T++;
            }
        }
    }

    cout << T << endl;
    return 0;
}
```

- **Initialization**: We read the number of enemies `N` and their health values into a vector `H`.
- **Main Loop**: We use a `while` loop to continue processing until no enemy has health left (`changed` becomes `false`).
- **Inner Loop**: For each enemy, if its health is greater than 0, we decrement its health according to the rule and increment the time `T`.
- **Output**: Finally, we print the total time `T` taken to defeat all enemies.
