#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        // Process enemy i until its health becomes 0 or less
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            
            // If current enemy is defeated, move to next enemy
            // But we need to handle the case where multiple enemies might be defeated in sequence
            if (H[i] <= 0) break;
            
            // If current enemy still has health, we continue attacking it
            // No need to do anything special - the loop continues
        }
    }
    
    cout << T << endl;
    return 0;
}
