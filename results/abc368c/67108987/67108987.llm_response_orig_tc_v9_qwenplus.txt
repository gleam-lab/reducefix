```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> H(N);
    for (int i = 0; i < N; i++) {
        cin >> H[i];
    }
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        while (H[i] > 0) {
            T++;
            if (T % 3 == 0) {
                H[i] -= 3;
            } else {
                H[i] -= 1;
            }
            
            // If the current enemy is defeated, move to next enemy
            // But we need to handle the case where multiple enemies might be defeated in sequence
            if (H[i] <= 0) {
                break;
            }
            
            // Check if there are remaining enemies with positive health
            bool found = false;
            for (int j = i + 1; j < N; j++) {
                if (H[j] > 0) {
                    i = j - 1; // Will be incremented to j in the next loop iteration
                    found = true;
                    break;
                }
            }
            if (!found) break;
        }
    }
    
    cout << T << endl;
    return 0;
}
```
