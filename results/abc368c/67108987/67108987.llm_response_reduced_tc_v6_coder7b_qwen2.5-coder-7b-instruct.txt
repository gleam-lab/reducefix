```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<long long> h(n+1), t(n+1);
    for(int i=1;i<=n;i++) cin >> h[i];

    for(int i=n;i>=1;i--) {
        if(h[i]>0) {
            t[i] = ceil((double)h[i]/3.0); // Calculate number of attacks needed to reduce health to 0
            if(i+1<=n) t[i] += t[i+1]; // Add remaining attacks from the next enemy
        }
    }

    cout << t[1] << endl;
    return 0;
}
```
