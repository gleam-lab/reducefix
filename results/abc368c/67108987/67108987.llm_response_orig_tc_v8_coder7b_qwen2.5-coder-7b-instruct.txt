```cpp
#include<bits/stdc++.h>
#define ll long long

using namespace std;

int main(){
    ios_base::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);

    int n;
    cin>>n;
    vector<ll> h(n+1), dp(n+1, LLONG_MAX);
    for(int i=1;i<=n;i++){
        cin>>h[i];
        if(h[i]==1){
            dp[i]=1;
        }
        else{
            dp[i]=min(dp[i], h[i]);
        }
    }

    for(int i=1;i<=n;i++){
        if(i>=2){
            dp[i]=min(dp[i], dp[i-1]+1);
        }
        if(i>=3){
            dp[i]=min(dp[i], dp[i-3]+3);
        }
    }

    cout<<dp[n]<<endl;
    return 0;
}
```

This code uses dynamic programming to find the minimum number of attacks required to defeat all enemies. The `dp` array keeps track of the minimum number of attacks needed to defeat the first `i` enemies. For each enemy, we calculate the minimum attacks required considering three possibilities: 
- Attacking the current enemy once,
- Attacking every third enemy starting from the current one, and
- Not attacking any enemy before the current one and directly attacking the current one. 

We take the minimum among these possibilities as the value for `dp[i]`. Finally, we output `dp[n]`, which gives us the minimum number of attacks required to defeat all enemies.
