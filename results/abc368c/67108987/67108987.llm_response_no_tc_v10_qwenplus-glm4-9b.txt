#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> H(N);
    for (int i = 0; i < N; i++) cin >> H[i];
    
    long long T = 0;
    for (int i = 0; i < N; i++) {
        // Calculate how many times we can subtract 3 from H[i]
        int multiplesOfThree = H[i] / 3;
        T += multiplesOfThree * 3;
        H[i] -= multiplesOfThree * 3;
        
        // Now we handle the remainder
        if (H[i] > 0) {
            T += 1; // Since we can't subtract 3, we subtract 1
            H[i] -= 1;
        }
        
        // If there's still health left, subtract 3
        if (H[i] > 0) {
            T += 1;
            H[i] -= 3;
        }
    }
    
    cout << T << endl;
    return 0;
}
