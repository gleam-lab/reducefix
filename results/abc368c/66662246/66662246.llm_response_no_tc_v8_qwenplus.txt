```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];
    
    ll t = 0;
    int current_enemy = 0;
    
    while (current_enemy < n) {
        t++;
        // Attack the frontmost enemy with health > 0
        if (h[current_enemy] > 0) {
            if (t % 3 == 0) {
                h[current_enemy] -= 3;
            } else {
                h[current_enemy] -= 1;
            }
            
            // If current enemy is defeated, move to next
            if (h[current_enemy] <= 0) {
                current_enemy++;
            }
        } else {
            // This enemy already has health <= 0, move to next
            current_enemy++;
        }
    }
    
    cout << t << endl;
}
```
