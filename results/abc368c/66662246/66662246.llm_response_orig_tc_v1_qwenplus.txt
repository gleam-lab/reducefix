```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];
    
    ll T = 0;
    int i = 0;
    
    while (i < n) {
        // Skip enemies already defeated
        if (a[i] <= 0) {
            i++;
            continue;
        }
        
        T++;
        
        // Check if current T is multiple of 3
        if (T % 3 == 0) {
            a[i] -= 3;
        } else {
            a[i]--;
        }
        
        // If enemy is defeated, move to next enemy
        if (a[i] <= 0) {
            i++;
        }
    }
    
    cout << T << endl;
}
```
