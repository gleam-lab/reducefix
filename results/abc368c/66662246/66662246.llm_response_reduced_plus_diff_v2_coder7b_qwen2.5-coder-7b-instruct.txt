```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define vi vector<ll>

void solve() {
    ll n;
    cin >> n;
    vi a(n);
    for(ll &x: a) cin >> x;
    
    ll time = 0;
    while(1){
        bool any = false;
        for(int i=0; i<n; i++){
            if(a[i]>0){
                any = true;
                if((time+1)%3==0){
                    a[i]-=3;
                }else{
                    a[i]-=1;
                }
                if(a[i]<0) a[i]=0;
            }
        }
        if(!any) break;
        time++;
    }

    cout<<time<<endl;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    ll t;
    cin>>t;
    while(t--){
        solve();
    }
    return 0;
}
```

This solution uses a simple greedy approach to solve the problem. It iterates over each element of the array and reduces its health by 1 or 3 based on whether the current time (incremented at each step) is divisible by 3. The process continues until no elements have remaining health. The total number of steps taken is then printed as the result. This approach ensures that the minimum number of steps is used to reduce all enemy health to zero.
