#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to compute the minimum T needed to reduce a single enemy's health to 0 or less
ll computeT(int h) {
    // We want to find the minimum number of attacks such that:
    // total damage >= h
    // Each attack does either 1 or 3 damage depending on whether T is a multiple of 3

    // Let x be the number of triple attacks (damage 3), and y be the number of single attacks (damage 1)
    // Then: 3x + y >= h
    // And total T = x + y
    // We want to minimize T subject to above constraints

    // We can binary search for minimal T such that:
    // For each T, we can do at most floor((T + 2)/3) triple attacks (say x)
    // And remaining are single attacks (y = T - x)

    ll low = 0, high = 3 * h; // Safe upper bound
    while (low < high) {
        ll mid = (low + high) / 2;
        ll tripleAttacks = mid / 3;
        ll singleAttacks = mid % 3;
        ll totalDamage = tripleAttacks * 3 + singleAttacks * 1;
        if (totalDamage >= h)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<int> H(n);
    rep(i, n) cin >> H[i];

    ll ans = 0;
    rep(i, n) {
        ans += computeT(H[i]);
    }

    cout << ans << endl;
}
