```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    ll ans = 0;
    rep(i, n) {
        ans += ceil((double)a[i] / 3.0);
    }

    cout << ans << endl;
}
```

This solution calculates the minimum number of attacks required to reduce the health of each enemy to zero or below by dividing each enemy's health by 3 and rounding up. This approach ensures that we account for both full reductions and partial reductions efficiently. The use of `ceil` function helps in handling the division correctly even when there is a remainder.
