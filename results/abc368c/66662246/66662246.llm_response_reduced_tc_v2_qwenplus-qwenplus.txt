#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to compute the minimum T needed to kill one enemy
ll calcT(int h) {
    // We want to find the minimum number of attacks to reduce health h to 0 or less
    // Let x be the number of triple attacks (at T divisible by 3)
    // Let y be the number of single attacks (at T not divisible by 3)
    // Then we have: 3x + y >= h
    // And we want to minimize T = x + y

    // Try all possible values of x in the range [0, h/3]
    // Since T = x + y and y = max(0, h - 3x), then T = x + max(0, h - 3x)

    // Optimal value is min over all possible x
    ll low = 0, high = h / 3;
    while (low < high) {
        ll mid = (low + high) / 2;
        if (mid + max(0LL, (ll)h - 3 * mid) > (mid + 1) + max(0LL, (ll)h - 3 * (mid + 1))) {
            low = mid + 1;
        } else {
            high = mid;
        }
    }

    return low + max(0LL, (ll)h - 3 * low);
}

int main() {
    int n;
    cin >> n;
    vector<int> H(n);
    rep(i, n) cin >> H[i];

    ll totalT = 0;
    rep(i, n) {
        totalT += calcT(H[i]);
    }

    cout << totalT << endl;
}
