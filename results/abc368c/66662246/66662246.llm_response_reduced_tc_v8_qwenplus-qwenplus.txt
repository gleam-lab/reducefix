#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to calculate the number of attacks needed for a single enemy
ll calc_attacks(ll h) {
    // Binary search to find minimum T such that:
    // floor(T/3) * 3 + (T % 3) * 1 >= h
    ll low = 0, high = 2 * h; // Safe upper bound
    while (low < high) {
        ll mid = (low + high) / 2;
        ll three_attacks = mid / 3;
        ll one_attacks = mid % 3;
        ll damage = three_attacks * 3 + one_attacks * 1;
        if (damage >= h)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll total_time = 0;
    priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<>> pq;

    // Initialize: Store enemies with their index and health
    rep(i, n) {
        if (H[i] > 0) {
            pq.emplace(calc_attacks(H[i]), i);
        }
    }

    while (!pq.empty()) {
        auto [t, idx] = pq.top();
        pq.pop();

        // Check if this enemy still needs attacks
        if (H[idx] <= 0) continue;

        // Attack once
        total_time++;
        
        // Apply special attack if time is multiple of 3
        if (total_time % 3 == 0) {
            H[idx] -= 3;
        } else {
            H[idx]--;
        }

        // If enemy still has health, schedule next attack
        if (H[idx] > 0) {
            pq.emplace(total_time + calc_attacks(H[idx]), idx);
        }
    }

    cout << total_time << endl;
}
