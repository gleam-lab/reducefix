#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to compute the minimum T needed to reduce a single enemy's health to 0 or less
ll computeTForHealth(ll h) {
    // Each 3 attacks:
    // - At T = 3k+1: damage 1
    // - At T = 3k+2: damage 1
    // - At T = 3k+3: damage 3
    // Total damage in one cycle of 3 attacks: 5
    ll full_cycles = h / 5;
    ll remaining_health = h % 5;
    ll t = full_cycles * 3;
    
    if (remaining_health == 0) return t;
    else if (remaining_health <= 2) return t + 1; // 1st or 2nd attack in next cycle
    else return t + 2; // Either 3rd attack in next cycle (3 damage), or two 1-damage attacks
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll ans = 0;
    vector<ll> required_times;
    
    for (ll h : H) {
        required_times.push_back(computeTForHealth(h));
    }

    sort(required_times.begin(), required_times.end());

    for (int i = 0; i < n; ++i) {
        ans = max(ans + required_times[i] - (i > 0 ? required_times[i - 1] : 0), 
                  required_times[i]);
    }
    
    cout << ans << endl;
}
