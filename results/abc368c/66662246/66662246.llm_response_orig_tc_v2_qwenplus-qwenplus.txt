#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

// Function to simulate attacks on a single enemy and return required T
ll simulate(int h) {
    ll t = 0;
    while (h > 0) {
        t++;
        if (t % 3 == 0)
            h -= 3;
        else
            h--;
    }
    return t;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<int> H(n);
    rep(i, n) cin >> H[i];

    // Binary search boundaries
    ll low = 0, high = 3 * 1e18; // large enough upper bound

    while (low < high) {
        ll mid = (low + high) / 2;
        ll total_kills = 0;

        // For each enemy, calculate how many hits it would receive in 'mid' time
        ll damage_3 = mid / 3;       // number of triple damage ticks
        ll damage_1 = mid - damage_3; // number of single damage ticks

        // Total damage applied by time 'mid'
        ll total_damage = damage_1 + damage_3 * 3;

        // Count how many enemies would be dead
        for (int h : H) {
            if (h <= total_damage)
                total_kills++;
        }

        // If all enemies are dead within 'mid' time
        if (total_kills == n)
            high = mid;
        else
            low = mid + 1;
    }

    cout << low << endl;
}
