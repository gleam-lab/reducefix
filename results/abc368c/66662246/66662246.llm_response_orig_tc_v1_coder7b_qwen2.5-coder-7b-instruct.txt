```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    ll ans = 0;
    priority_queue<pair<ll, int>> pq;
    rep(i, n) pq.push({-a[i], i});

    while (!pq.empty()) {
        auto [val, idx] = pq.top();
        pq.pop();
        val *= -1;

        if (val <= 0) break;

        if ((ans + 1) % 3 == 0) {
            val -= 3;
        } else {
            val--;
        }

        ans++;
        if (val > 0) pq.push({-val, idx});
    }

    cout << ans << endl;

    return 0;
}
```
