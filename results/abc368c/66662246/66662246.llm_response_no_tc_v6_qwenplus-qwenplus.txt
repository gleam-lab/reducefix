```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;
    rep(i, n)
    {
        // Determine how many attacks of type "3 damage" the current enemy needs
        ll threeDmgAttacks = H[i] / 3;
        ll remainingHealth = H[i] % 3;

        T += threeDmgAttacks * 3;

        if (remainingHealth > 0)
        {
            // Need to add one more attack for each remaining 1 damage
            T += remainingHealth;
        }
    }

    cout << T << endl;
}
```
