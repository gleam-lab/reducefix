```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];
    ll ans = 0;
    for(int i = 0; i < n; ++i){
        if(a[i] > 0){
            if((ans + 1) % 3 == 0){
                ans += a[i] / 3;
            }else{
                ans += a[i];
            }
        }
    }
    cout << ans << endl;
}
```
