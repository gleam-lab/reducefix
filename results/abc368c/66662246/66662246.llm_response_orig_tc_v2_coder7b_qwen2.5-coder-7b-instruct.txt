```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    ll ans = 0;
    priority_queue<ll, vector<ll>, greater<ll>> pq(a.begin(), a.end());
    while (!pq.empty()) {
        ll x = pq.top();
        pq.pop();

        if (x <= 0) break;

        ans++;
        if (ans % 3 == 0) {
            pq.push(x - 3);
        } else {
            pq.push(x - 1);
        }
    }

    cout << ans << endl;
}
```
