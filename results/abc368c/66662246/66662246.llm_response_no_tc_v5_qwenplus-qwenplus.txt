```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> H(n);
    rep(i, n) cin >> H[i];

    ll T = 0;

    // We simulate the process with a priority queue
    // At each step, we attack the frontmost enemy that still has health > 0

    // Instead of processing each attack one-by-one (which would be O(total_health)), 
    // we calculate how many attacks are needed to kill each enemy based on their health,
    // and track when triple damage ticks occur.

    // Let's keep a queue of enemies and simulate time T smartly

    queue<tuple<ll, ll>> q; // (time_mod_3, remaining_health, index)

    rep(i, n) {
        q.push({0, H[i], i});
    }

    while (!q.empty()) {
        auto [t_mod3, h, idx] = q.front();
        q.pop();

        T++;

        ll damage = (T % 3 == 0) ? 3 : 1;
        h -= damage;

        if (h <= 0) {
            // Enemy defeated, move to next
            if (idx + 1 < n) {
                q.push({T % 3, H[idx + 1], idx + 1});
            }
        } else {
            // Re-add this enemy to queue
            q.push({T % 3, h, idx});
        }
    }

    cout << T << endl;

    return 0;
}
```
