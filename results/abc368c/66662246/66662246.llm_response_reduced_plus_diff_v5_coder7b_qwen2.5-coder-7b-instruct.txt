```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];
    ll ans = 0;
    for(int i=0;i<n;i++){
        if(a[i]>0){
            int cnt=a[i]/3;
            if(a[i]%3!=0)cnt++;
            ans+=cnt;
            a[i]-=cnt*3;
        }
    }
    sort(a.begin(),a.end());
    reverse(a.begin(),a.end());
    int i=0;
    while(i<n && a[i]==0)i++;
    while(i<n){
        int cnt=a[i]/3;
        if(a[i]%3!=0)cnt++;
        ans+=cnt;
        a[i]-=cnt*3;
        for(int j=i+1;j<n;j++){
            if(a[j]>=cnt){
                a[j]-=cnt;
                break;
            }else{
                cnt-=a[j];
                a[j]=0;
            }
        }
        i++;
    }
    cout << ans << endl;
}
```
This solution works as follows:
1. It first counts the number of attacks required to reduce each enemy's health to zero, assuming no other enemies can help. 
2. It then sorts the array of remaining healths in descending order.
3. For each enemy with remaining health, it tries to distribute the attacks among the rest of the enemies. If an enemy can take all the attacks needed, it does so. Otherwise, it takes its share and reduces the health of subsequent enemies accordingly.
4. Finally, it prints the total number of attacks made.
