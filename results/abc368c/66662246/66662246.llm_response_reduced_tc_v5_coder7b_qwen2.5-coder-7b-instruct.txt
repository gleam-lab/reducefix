```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main()
{
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];
    ll ans = 0;
    rep(i, n)
    {
        int x = a[i] / 3;
        ans += x;
        a[i] -= x * 3;
        while (a[i] > 0)
        {
            ans++;
            if (a[i] % 3 == 0)
            {
                a[i] -= 3;
            }
            else
            {
                a[i]--;
            }
        }
    }
    cout << ans << endl;
}
```

In this code, we divide the health of each enemy by 3 instead of 5 before calculating the number of attacks needed. This is because every time an enemy's health is reduced by 3, it effectively reduces its health by 1 three times consecutively. Therefore, dividing the health by 3 gives us the correct number of full cycles of reducing health by 3 that can be performed before we need to reduce health by 1.
