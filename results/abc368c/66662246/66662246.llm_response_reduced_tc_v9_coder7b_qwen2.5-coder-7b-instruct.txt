```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> h(n);
    rep(i, n) cin >> h[i];

    ll ans = 0;
    while (true) {
        bool all_zero = true;
        rep(i, n) {
            if (h[i] > 0) {
                all_zero = false;
                if ((ans + 1) % 3 == 0) {
                    h[i] -= 3;
                } else {
                    h[i]--;
                }
                ans++;
            }
        }
        if (all_zero) break;
    }

    cout << ans << endl;
}
```

This solution iteratively reduces the health of each enemy based on the current value of `T` and checks if all enemies have been defeated. The loop continues until no enemy has any health left. This approach ensures that we correctly account for the changes in enemy health as per the problem statement.
