#include <iostream>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main() {
    int n;
    cin >> n;
    vector<int> health(n);
    for (int i = 0; i < n; ++i) {
        cin >> health[i];
    }

    long long T = 0;

    // For each enemy, calculate the total number of attacks needed.
    // If health is divisible by 3, we use the multiple of 3 attacks.
    // Otherwise, we use 1 attack and then use the rest of the health in multiples of 3.
    for (int i = 0; i < n; ++i) {
        if (health[i] % 3 == 0) {
            T += health[i] / 3;
        } else {
            T += health[i] / 3 + 1;
        }
    }

    cout << T << endl;

    return 0;
}
