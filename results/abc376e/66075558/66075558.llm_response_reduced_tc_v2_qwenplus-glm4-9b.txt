#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long
const int N = 2e5 + 9;

int solve() {
    int n, k;
    cin >> n >> k;
    vector<int> A(n), B(n);
    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n; ++i) cin >> B[i];
    
    vector<pair<int, int>> p(n);
    for (int i = 0; i < n; ++i) p[i] = {A[i], B[i]};
    sort(p.begin(), p.end(), [&](const pair<int, int>& a, const pair<int, int>& b) {
        if (a.first == b.first) return a.second < b.second;
        return a.first > b.first;
    });
    
    int sum = 0, best = 0;
    for (int i = 0; i < n; ++i) {
        sum += p[i].second;
        if (i >= k - 1) {
            best = max(best, sum);
        }
        if (i >= k - 1) {
            sum -= p[i - k].second;
        }
    }
    
    // Calculate the maximum of A in the first k elements
    int maxA = 0;
    for (int i = 0; i < k; ++i) {
        maxA = max(maxA, A[i]);
    }
    
    return maxA * best;
}

int32_t main() {
    int t;
    cin >> t;
    while (t--) {
        cout << solve() << '\n';
    }
    return 0;
}

This code first sorts the pairs of A and B values such that the A values are in descending order and B values are in ascending order when A values are equal. This allows us to find the best sum of B values for each A value in descending order.

We iterate through the sorted pairs and keep track of the sum of the best B values for the current A value. We also track the maximum sum of B values for the first k elements.

Finally, we return the product of the maximum A value in the first k elements and the best sum of B values found. This ensures that we are considering the minimum possible product of the maximum A value and the sum of B values for any subset of size k.
