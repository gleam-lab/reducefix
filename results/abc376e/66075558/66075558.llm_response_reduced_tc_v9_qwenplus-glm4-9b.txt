
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long
#define N 200005

int n, k;
int A[N], B[N];
vector<pair<int, int>> C;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        C.clear();
        for (int i = 0; i < n; ++i) {
            cin >> A[i];
            C.emplace_back(A[i], i);
        }
        for (int i = 0; i < n; ++i) {
            cin >> B[i];
        }

        sort(C.begin(), C.end());
        long long ans = 1e18;
        long long sumB = 0;
        for (int i = 0; i < n; ++i) {
            sumB += B[C[i].second];
            if (i + 1 >= k) {
                sumB -= B[C[i - k + 1].second];
                ans = min(ans, (long long)C[i].first * sumB);
            }
        }

        cout << ans << endl;
    }
    return 0;
}
```
