#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 200005;

int n, k, t;
long long a[N], b[N], ans;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> t;
    while (t--) {
        cin >> n >> k;
        vector<pair<int, int>>ab(N); // Pair of (A[i], B[i])
        for (int i = 0; i < n; ++i) {
            cin >> a[i] >> b[i];
            ab[i] = {a[i], b[i]};
        }

        // Sort by A[i] ascending
        sort(ab.begin(), ab.end());

        // Use prefix sum array for B[i]
        long long prefixSum = 0;
        vector<long long> prefixSums(n);
        for (int i = 0; i < n; ++i) {
            prefixSum += ab[i].second;
            prefixSums[i] = prefixSum;
        }

        // Calculate minimum result
        long long minResult = LLONG_MAX;
        for (int i = k - 1; i < n; ++i) {
            long long maxElement = ab[i].first;
            long long sumB = prefixSums[i] - (prefixSums[i - k + 1] - (i - k + 1) * ab[i - k + 1].second);
            minResult = min(minResult, maxElement * sumB);
        }

        cout << minResult << '\n';
    }

    return 0;
}
