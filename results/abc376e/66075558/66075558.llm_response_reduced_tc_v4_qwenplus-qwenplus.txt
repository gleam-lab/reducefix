#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

// Function to solve each test case
void solve() {
    int n, k;
    cin >> n >> k;
    
    vector<pair<int, int>> items(n);  // pair of (A_i, B_i)
    for (int i = 0; i < n; ++i) {
        cin >> items[i].first;  // A_i
    }
    for (int i = 0; i < n; ++i) {
        cin >> items[i].second;  // B_i
    }

    // Sort by A_i ascending because we want to try small max(A_i) first
    sort(items.begin(), items.end());

    // Min-heap to keep track of top k smallest B values
    priority_queue<int, vector<int>, greater<int>> pq;
    int sum_b = 0;
    int res = 1e18;  // Large initial value

    // Try each item as the max A_i in the subset
    for (int i = 0; i < n; ++i) {
        // We need to include B_i corresponding to current A_i
        sum_b += items[i].second;
        pq.push(items[i].second);

        // If we have more than k elements, remove the one with smallest B
        if (pq.size() > k) {
            sum_b -= pq.top();
            pq.pop();
        }

        // Once we've selected k elements, compute cost
        if (pq.size() == k) {
            res = min(res, items[i].first * sum_b);
        }
    }

    cout << res << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
