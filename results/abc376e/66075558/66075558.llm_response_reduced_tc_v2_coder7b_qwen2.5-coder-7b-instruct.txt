#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll;

struct Node {
    int val, id;
};

bool cmp(const Node &a, const Node &b) {
    if (a.val != b.val) return a.val < b.val;
    return a.id < b.id;
}

ll minCost(int n, int k, vector<int>& A, vector<int>& B) {
    vector<Node> nodes(n);
    for (int i = 0; i < n; ++i) {
        nodes[i] = {A[i], i};
    }
    sort(nodes.begin(), nodes.end(), cmp);

    vector<ll> prefixSum(n+1, 0);
    for (int i = 0; i < n; ++i) {
        prefixSum[i+1] = prefixSum[i] + B[nodes[i].id];
    }

    ll result = LLONG_MAX;
    for (int i = 0; i + k - 1 < n; ++i) {
        int maxVal = nodes[i+k-1].val;
        ll sum = prefixSum[min(n, i+k)] - prefixSum[i];
        result = min(result, (ll)maxVal * sum);
    }
    return result;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int T;
    cin >> T;
    while (T--) {
        int n, k;
        cin >> n >> k;
        vector<int> A(n), B(n);
        for (int i = 0; i < n; ++i) cin >> A[i];
        for (int i = 0; i < n; ++i) cin >> B[i];

        cout << minCost(n, k, A, B) << '\n';
    }

    return 0;
}
