```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

const int MAXN = 2e5 + 9, INF = 0x3f3f3f3f3f3f3f3f;

int n, k, ans;
vector<pair<int, int>> a, b;

void solve() {
    a.clear(), b.clear();
    cin >> n >> k;
    for (int i = 1; i <= n; ++i) {
        int ai, bi; cin >> ai >> bi;
        a.emplace_back(ai, i);
        b.emplace_back(bi, i);
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    long long sum_b = 0;
    for (int i = 0; i < k; ++i) {
        sum_b += b[i].first;
    }
    ans = sum_b * a[k - 1].first;

    for (int i = k; i < n; ++i) {
        sum_b -= b[i - k].first;
        sum_b += b[i].first;
        ans = min(ans, sum_b * a[i].first);
    }

    cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    int t; cin >> t;
    while (t--) solve();
    return 0;
}
```

This solution sorts both arrays `a` and `b` based on their first elements (the values from sequence `A`). It then calculates the initial sum of the smallest `k` elements from `b`, which corresponds to the product of the maximum element in `a` with the sum of the corresponding elements in `b`. This gives us an initial candidate for the answer. 

For each subsequent element in `a`, it updates the sum of the elements in `b` by subtracting the smallest element seen so far and adding the current element's corresponding value from `b`. It recalculates the product and updates the answer if this new product is smaller than the current best.

This approach ensures that we always consider the best possible subset of size `k` from both sequences, leading to an optimal solution. The time complexity is dominated by the sorting steps, making it efficient enough for the problem constraints.
