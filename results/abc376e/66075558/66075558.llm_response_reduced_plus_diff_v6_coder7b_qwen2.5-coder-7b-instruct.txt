#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define int long long
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;

int n, k;
pair<int, int> a[N];

bool cmp(pair<int, int>& pa, pair<int, int>& pb){
    if(pa.first != pb.first) return pa.first > pb.first;
    else return pa.second < pb.second;
}

signed main(){
    ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int t;
    cin >> t;
    while(t--){
        cin >> n >> k;
        vector<pair<int, int>> vec(n);
        for(int i = 0; i < n; ++i){
            cin >> vec[i].first;
            vec[i].second = i + 1;
        }
        sort(vec.begin(), vec.end(), cmp);
        long long res = inf;
        long long sum = 0;
        priority_queue<int> pq;
        for(int i = 0; i < n; ++i){
            sum += vec[i].first;
            pq.push(vec[i].first);
            if(pq.size() > k){
                sum -= pq.top();
                pq.pop();
            }
            if(i + 1 >= k){
                res = min(res, sum * vec[i].first);
            }
        }
        cout << res << '\n';
    }
    return 0;
}
