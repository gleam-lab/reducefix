```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

#define int long long

struct Item {
    int a, b;
};

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<Item> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].a;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].b;
        }

        // Sort items by A value ascending
        sort(items.begin(), items.end(), [](const Item& x, const Item& y) {
            return x.a < y.a;
        });

        // Min-heap to keep track of top K B values
        priority_queue<int, vector<int>, greater<>> min_heap;
        int total_b_sum = 0;
        int result = 1e18; // Large initial value

        // First, try with last K elements in sorted A list (largest A's)
        // and pick the minimum among those combinations

        for (int i = 0; i < N; ++i) {
            // Current item is included
            total_b_sum += items[i].b;
            min_heap.push(items[i].b);

            // If heap size exceeds K, remove smallest b (due to min-heap)
            if (min_heap.size() > K) {
                total_b_sum -= min_heap.top();
                min_heap.pop();
            }

            // Once we have K elements, calculate cost
            if (min_heap.size() == K) {
                result = min(result, items[i].a * total_b_sum);
            }
        }

        cout << result << "\n";
    }

    return 0;
}
```
