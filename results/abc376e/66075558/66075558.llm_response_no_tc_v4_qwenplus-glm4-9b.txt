#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> a(n), b(n);
        
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first >> b[i].first;
            a[i].second = b[i].first; // Store product of A_i and B_i
            b[i].second = i; // Store index of B_i
        }
        
        // Sort by A_i (first element), then by B_i (second element)
        sort(a.begin(), a.end(), [](const pair<int, int>& p1, const pair<int, int>& p2) {
            return p1.first < p2.first || (p1.first == p2.first && p1.second < p2.second);
        });
        
        // Prefix sum of B_i's
        vector<int> prefixSum(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            prefixSum[i + 1] = prefixSum[i] + b[i].first;
        }
        
        int ans = LLONG_MAX;
        // Try different starting points for the subset of size K
        for (int left = 0; left <= n - k; ++left) {
            int right = left + k - 1;
            int sumB = prefixSum[right + 1] - prefixSum[left];
            // Calculate the minimum possible value using the max A_i in the subset
            int maxA = a[right].first;
            ans = min(ans, maxA * sumB);
        }
        
        cout << ans << endl;
    }
    return 0;
}
