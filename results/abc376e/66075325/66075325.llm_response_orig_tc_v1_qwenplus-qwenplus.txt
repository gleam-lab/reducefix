```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

#define int long long

const int INF = 1e18;

int T, N, K;
pair<int, int> A[200005];
int B[200005];

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> T;
    while (T--) {
        cin >> N >> K;
        for (int i = 1; i <= N; ++i) {
            cin >> A[i].first;
            A[i].second = i;
        }
        for (int i = 1; i <= N; ++i) {
            cin >> B[i];
        }

        // Sort elements by A[i]
        sort(A + 1, A + N + 1);

        // Build prefix sums of B[i] in the sorted order
        int prefix_sum[200005] = {0};
        for (int i = 1; i <= N; ++i) {
            prefix_sum[i] = prefix_sum[i - 1] + B[A[i].second];
        }

        // Maintain a max heap to track the smallest K values of B among top candidates
        priority_queue<int> pq;
        int sum = 0;
        int result = INF;

        // We iterate through the A array sorted increasingly
        // For each element A[i], we assume it's the maximum A in the subset S
        // Then we pick K-1 smallest B from previous elements to minimize total sum * A[i]
        for (int i = 1; i <= N; ++i) {
            int idx = A[i].second;
            pq.push(B[idx]);
            sum += B[idx];

            // Keep only K largest B values in the heap
            while ((int)pq.size() > K) {
                sum -= pq.top();
                pq.pop();
            }

            // Once we have at least K elements, compute the candidate value
            if (i >= K) {
                result = min(result, A[i].first * sum);
            }
        }

        cout << result << endl;
    }

    return 0;
}
```
