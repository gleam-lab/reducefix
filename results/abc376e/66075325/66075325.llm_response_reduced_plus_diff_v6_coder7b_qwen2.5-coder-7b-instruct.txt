#include <vector>
#include <queue>
#include <algorithm>

#define int long long

struct Node {
    int val, idx;
};

struct Compare {
    bool operator()(Node &a, Node &b) {
        return a.val > b.val || (a.val == b.val && a.idx < b.idx);
    }
};

int main() {

            prefix_sum[i] = (i ? prefix_sum[i - 1] : 0) + B[i];
        }

        for (int i = 0; i < K; ++i) {
            pq.push({A[i], i});
        }
