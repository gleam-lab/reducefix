#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int32_t main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i].first;
            a[i].second = i + 1;
        }
        vector<pair<int, int>> b(n);
        for (int i = 0; i < n; i++) {
            cin >> b[i].first;
            b[i].second = i + 1;
        }
        sort(a.begin(), a.end(), greater<pair<int, int>>());
        sort(b.begin(), b.end(), greater<pair<int, int>>());

        int ans = INT_MAX;
        int sumB = 0;
        int count = 0;
        for (int i = 0; i < n; i++) {
            sumB += b[i].first;
            count++;
            if (count == k) {
                ans = min(ans, (sumB + b[i].first) * a[i].first);
                sumB -= b[i - k + 1].first;
                count--;
            }
        }
        cout << ans << endl;
    }
    return 0;
}
