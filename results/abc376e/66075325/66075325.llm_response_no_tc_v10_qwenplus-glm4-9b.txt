#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define MAXN 200005

int n, k, T;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while (T--) {
        cin >> n >> k;
        vector<pair<int, int>> a(n + 1), b(n + 1);
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first >> b[i].first;
            a[i].second = i + 1;
            b[i].second = i + 1;
        }
        sort(a.begin(), a.end());
        sort(b.begin(), b.end(), greater<pair<int, int>>());

        long long ans = LLONG_MAX;
        long long sum = 0;
        int count = 0;

        for (int i = k - 1; i < n; ++i) {
            sum += b[i].first;
            count++;
            while (count > k) {
                sum -= b[k - 1].first;
                count--;
            }
            ans = min(ans, a[i].first * sum);
        }

        cout << ans << '\n';
    }
    return 0;
}
