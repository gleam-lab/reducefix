#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <set>
using namespace std;

int compare(pair<int, int> a, pair<int, int> b) {
    return a.first > b.first;
}

int main() {
    int T, N, K;
    cin >> T;
    while (T--) {
        cin >> N >> K;
        vector<int> A(N + 1), B(N + 1);
        for (int i = 1; i <= N; i++) {
            cin >> A[i];
        }
        for (int i = 1; i <= N; i++) {
            cin >> B[i];
        }

        // Pair A and B elements and sort by A
        vector<pair<int, int>> AB(N + 1);
        for (int i = 1; i <= N; i++) {
            AB[i] = {A[i], B[i]};
        }
        sort(AB.begin() + 1, AB.end(), [](auto a, auto b) {
            return a.first < b.first;
        });

        // Use a multiset to keep track of the sum of B elements
        multiset<int> S;
        long long sumB = 0;
        for (int i = 1; i <= N; i++) {
            sumB += AB[i].second;
            S.insert(AB[i].second);
            if (i >= K) {
                // Remove the smallest element to keep the size of the set as K
                sumB -= *S.begin();
                S.erase(S.begin());
            }
        }

        // Calculate the minimum possible value
        long long ans = LLONG_MAX;
        for (int i = 1; i <= N - K + 1; i++) {
            // sumB is the sum of the last K B elements in the window
            // AB[i+K-1].first is the maximum of A in this window
            ans = min(ans, (sumB + AB[i + K - 1].second) * AB[i + K - 1].first);
        }

        cout << ans << endl;
    }
    return 0;
}
