#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;
        vector<pair<int, int>> A(N), B(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
            A[i].second = B[i].first = i;
        }
        for (int i = 0; i < N; ++i) {
            cin >> B[i].second;
        }
        sort(A.begin(), A.end());
        vector<pair<int, long long>> B_sorted(N);
        for (int i = 0; i < N; ++i) {
            B_sorted[i] = {B[A[i].second].second, A[i].first};
        }
        sort(B_sorted.begin(), B_sorted.end(), [](const pair<int, long long>& a, const pair<int, long long>& b) {
            if (a.first == b.first) return a.second < b.second;
            return a.first > b.first;
        });

        long long sum = 0, best_sum = 0, max_val = 0;
        for (int i = 0; i < N; ++i) {
            sum += B_sorted[i].second;
            if (i >= K - 1) {
                best_sum = sum;
            }
            if (i >= K - 1) {
                max_val = B_sorted[i - K + 1].first;
                sum -= B_sorted[i - K + 1].second;
                sum -= B_sorted[i].second;
                sum += B_sorted[i].first;
                best_sum -= B_sorted[i].second;
                best_sum += B_sorted[i].first;
                cout << max_val * best_sum << ' ';
                break;
            }
        }
        cout << '\n';
    }
    return 0;
}
