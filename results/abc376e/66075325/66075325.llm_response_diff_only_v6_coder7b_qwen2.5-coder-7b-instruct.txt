```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

const int INF = 1e18;

struct Data {
    int a, b, idx;
};

bool cmp(Data &a, Data &b) {
    return a.a > b.a || (a.a == b.a && a.b < b.b);
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<Data> data(n);
        for (int i = 0; i < n; ++i) {
            cin >> data[i].a;
            data[i].idx = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> data[i].b;
        }
        sort(data.begin(), data.end(), cmp);

        vector<int> prefix_sum(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            prefix_sum[i] = prefix_sum[i - 1] + data[i - 1].b;
        }

        long long ans = INF;
        for (int i = k - 1; i < n; ++i) {
            long long max_a = data[i].a;
            long long sum_b = prefix_sum[i + 1] - prefix_sum[i - k + 1] + data[i - k + 1].b;
            ans = min(ans, max_a * sum_b);
        }

        cout << ans << endl;
    }
    return 0;
}
```

### Explanation:
1. **Data Structure**: We use a struct `Data` to store the values of `A_i`, `B_i`, and their indices.
2. **Sorting**: We sort the array based on `A_i` in descending order. If two elements have the same `A_i`, we sort them based on `B_i` in ascending order. This ensures that we always consider the maximum `A_i` first.
3. **Prefix Sum**: We calculate the prefix sum of `B_i` to quickly compute the sum of any subarray of `B_i`.
4. **Finding Minimum Value**: We iterate through the sorted array and calculate the product of the maximum `A_i` in the current window of size `K` and the sum of `B_i` in that window. We keep track of the minimum value encountered.

This approach ensures that we efficiently find the minimum value of the required expression within the given constraints.
