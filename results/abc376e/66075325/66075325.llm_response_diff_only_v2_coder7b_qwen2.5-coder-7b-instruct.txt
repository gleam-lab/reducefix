```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define int long long
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, m, t, ans, k;
int s;

signed main() {
    cin >> t;
    while (t--) {
        cin >> n >> k;
        vector<pair<int, int>>a(n), b(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; i++) {
            cin >> b[i].first;
            b[i].second = i;
        }
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        ans = inf;
        for(int i=0;i<=n-k;i++){
            s = 0;
            for(int j=i;j<i+k;j++)
                s += b[j].first;
            ans = min(ans, (long long)a[k+i-1].first * s);
        }
        cout << ans << endl;
    }
    return 0;
}
```
