#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 200002;

int T, N, K;
vector<pair<int, int>> A, B;

long long solve() {
    sort(A.begin(), A.end());
    sort(B.begin(), B.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
        return a.first < b.first;
    });
    
    vector<long long> prefixSum(B.begin(), B.end());
    for (int i = 1; i < B.size(); ++i) {
        prefixSum[i] += prefixSum[i - 1];
    }
    
    long longbestSum = 0;
    long long minVal = LLONG_MAX / 2; // Arbitrary large value to ensure correctness
    
    for (int i = 0; i <= N - K; ++i) {
        int sum = prefixSum[i + K - 1] - (i > 0 ? prefixSum[i - 1] : 0);
        int maxA = A[i + K - 1].first;
        long long res = maxA * sum;
        bestSum = max(bestSum, res);
        minVal = min(minVal, maxA);
    }
    
    return min(bestSum, minVal * prefixSum[N - K]);
}

int main() {
    cin >> T;
    while (T--) {
        cin >> N >> K;
        A.resize(N + 1), B.resize(N + 1);
        for (int i = 1; i <= N; ++i) {
            cin >> A[i].first;
            A[i].second = i;
        }
        for (int i = 1; i <= N; ++i) {
            cin >> B[i].first;
            B[i].second = i;
        }
        cout << solve() << endl;
    }
    return 0;
}
