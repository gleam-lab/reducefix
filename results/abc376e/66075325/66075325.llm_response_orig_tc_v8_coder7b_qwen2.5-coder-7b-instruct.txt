#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define ll long long

ll minProduct(ll A[], ll B[], int N, int K) {
    vector<pair<ll, int>> vecA(A, A+N), vecB(B, B+N);
    sort(vecA.begin(), vecA.end());
    sort(vecB.begin(), vecB.end());

    ll ans = LLONG_MAX;
    priority_queue<ll> pq;
    ll sum = 0;

    for(int i=0; i<K; i++) {
        pq.push(vecB[i].first);
        sum += vecB[i].first;
    }

    for(int i=K; i<N; i++) {
        ans = min(ans, sum * vecA[i].first);
        sum -= pq.top();
        pq.pop();
        pq.push(vecB[i].first);
        sum += vecB[i].first;
    }

    ans = min(ans, sum * vecA[N-1].first);

    return ans;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    int T;
    cin >> T;
    while(T--) {
        int N, K;
        cin >> N >> K;
        
        ll A[N], B[N];
        for(int i=0; i<N; i++)
            cin >> A[i];
        for(int i=0; i<N; i++)
            cin >> B[i];
        
        cout << minProduct(A, B, N, K) << '\n';
    }
    
    return 0;
}
