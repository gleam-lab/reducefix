#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T;
    cin >> T;
    
    while (T--) {
        int N, K;
        cin >> N >> K;
        
        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
        }
        
        vector<int> B(N);
        for (int i = 0; i < N; ++i) {
            cin >> B[i];
        }
        
        // Pair A with B and sort by A in ascending order
        vector<pair<int, int>> AB(N);
        for (int i = 0; i < N; ++i) {
            AB[i] = {A[i].first, B[i]};
        }
        sort(AB.begin(), AB.end());
        
        // Min-heap to maintain top K smallest B values
        priority_queue<int, vector<int>, greater<>> maxHeap;
        int sumB = 0;
        int minResult = 1e18; // Large enough initial value
        
        // Start from the largest A values and maintain K elements
        for (int i = N - 1; i >= 0; --i) {
            if (maxHeap.size() < K) {
                sumB += AB[i].second;
                maxHeap.push(AB[i].second);
                
                if (maxHeap.size() == K) {
                    minResult = min(minResult, AB[i].first * sumB);
                }
            } else {
                // If current B is larger than smallest in heap, replace it
                if (AB[i].second > maxHeap.top()) {
                    sumB -= maxHeap.top();
                    maxHeap.pop();
                    
                    sumB += AB[i].second;
                    maxHeap.push(AB[i].second);
                    
                    minResult = min(minResult, AB[i].first * sumB);
                }
            }
        }
        
        cout << minResult << "\n";
    }
    
    return 0;
}
