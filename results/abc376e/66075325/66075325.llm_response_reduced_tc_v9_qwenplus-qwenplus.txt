#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

#define int long long

struct Item {
    int a, b;
};

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<Item> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].a;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].b;
        }

        // Sort items by A value
        sort(items.begin(), items.end(), [](const Item& x, const Item& y) {
            return x.a < y.a;
        });

        // Min-heap to maintain top K smallest B values
        priority_queue<int, vector<int>, greater<>> pq;
        int sum_b = 0;
        int result = 1e18;  // Large initial value

        // First, try the first K elements
        for (int i = 0; i < K; ++i) {
            pq.push(items[i].b);
            sum_b += items[i].b;
        }
        result = min(result, items[K - 1].a * sum_b);

        // Sliding window: add next item and possibly remove one
        for (int i = K; i < N; ++i) {
            pq.push(items[i].b);
            sum_b += items[i].b;

            if (pq.size() > K) {
                sum_b -= pq.top();
                pq.pop();
            }

            result = min(result, items[i].a * sum_b);
        }

        cout << result << "\n";
    }

    return 0;
}
