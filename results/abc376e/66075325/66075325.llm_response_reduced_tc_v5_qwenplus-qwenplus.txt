#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T;
    cin >> T;
    
    while (T--) {
        int N, K;
        cin >> N >> K;
        
        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
        }
        
        vector<int> B(N);
        for (int i = 0; i < N; ++i) {
            cin >> B[i].second();
        }
        
        // Pair A and B together for sorting
        for (int i = 0; i < N; ++i) {
            A[i].second = B[i];
        }
        
        // Sort items by A_i in ascending order
        sort(A.begin(), A.end());
        
        // Min-heap to maintain the largest K-1 B_i values
        priority_queue<int, vector<int>, greater<int>> min_heap;
        
        int sum_B = 0;
        int min_result = numeric_limits<int>::max();
        
        // We'll try each item as the max A_i in the subset
        for (int i = 0; i < N; ++i) {
            int current_A = A[i].first;
            int current_B = A[i].second;
            
            // If we have already selected K elements before this one
            if (min_heap.size() == K - 1) {
                // Try calculating with this element as the max A
                min_result = min(min_result, (sum_B + current_B) * current_A);
            }
            
            // Add current_B to the heap
            if (min_heap.size() < K - 1) {
                min_heap.push(current_B);
                sum_B += current_B;
                
                // When we have K-1 elements, prepare for next iteration
                if (min_heap.size() == K - 1) {
                    // No calculation here because we need one more element
                }
            } else {
                // Remove the smallest B from our selection
                if (current_B > min_heap.top()) {
                    sum_B -= min_heap.top();
                    min_heap.pop();
                    min_heap.push(current_B);
                    sum_B += current_B;
                }
            }
        }
        
        cout << min_result << "\n";
    }
    
    return 0;
}
