        for (int i = 0; i < n; i++) {
            b[i] = {bb[a[i].second], a[i].second};
        }
        
        int ans = 1e18;
        priority_queue<int> pq;
        int sum = 0;
        
        for (int i = 0; i < n; i++) {
            if (pq.size() < k - 1) {
                pq.push(b[i].first);
                sum += b[i].first;
            } else if (!pq.empty() && pq.top() > b[i].first) {
                sum -= pq.top();
                pq.pop();
                pq.push(b[i].first);
                sum += b[i].first;
