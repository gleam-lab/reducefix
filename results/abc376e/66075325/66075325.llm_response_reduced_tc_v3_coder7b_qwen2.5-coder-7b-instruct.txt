#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

const int INF = 1e18;

struct Node {
    int val, idx;
};

bool cmp(Node &a, Node &b) {
    return a.val > b.val;
}

long long solve(int n, int k, vector<Node> &A, vector<Node> &B) {
    sort(A.begin(), A.end(), cmp);
    sort(B.begin(), B.end());

    priority_queue<int> pq;
    long long sum = 0, ans = INF;

    for (int i = 0; i < k - 1; ++i) {
        pq.push(B[i].val);
        sum += B[i].val;
    }

    for (int i = k - 1; i < n; ++i) {
        pq.push(B[i].val);
        sum += B[i].val;
        ans = min(ans, (long long)A[i].val * (sum + B[i].val));
        sum -= pq.top();
        pq.pop();
    }

    return ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<Node> A(n), B(n);
        for (int i = 0; i < n; ++i) {
            cin >> A[i].val;
            A[i].idx = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> B[i].val;
            B[i].idx = i;
        }

        cout << solve(n, k, A, B) << '\n';
    }

    return 0;
}
