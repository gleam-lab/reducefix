        int n, k;

        }
        for (int i = 0; i < n; ++i) {
            cin >> ab[i].second;
        }
        sort(ab.begin(), ab.end());
        
        priority_queue<int> pq;
        int sum = 0;
        int ans = inf;
        
        for (int i = 0; i < n; ++i) {
            int current_a = ab[i].first;
            int current_b = ab[i].second;
            if (pq.size() < k - 1) {
                pq.push(current_b);
                sum += current_b;
            } else {
                if (k == 1) {
                    ans = min(ans, current_a * current_b);
                } else {
                    ans = min(ans, current_a * (sum + current_b));
                }
                if (!pq.empty() && current_b < pq.top()) {
                    sum -= pq.top();
                    pq.pop();
                    pq.push(current_b);

            }
        }

}
