#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define int long long

const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;
int n, k, ans;
pair<int, int> a[N], b[N];

signed main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int t;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        for (int i = 1; i <= n; i++) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 1; i <= n; i++) {
            cin >> b[i].first;
        }

        // Sort based on A_i values to maximize A_i for larger sums
        sort(a + 1, a + n + 1);
        // Sort based on B_i values to maximize the sum
        sort(b + 1, b + n + 1);

        int current_sum = 0, max_A = 0;
        for (int i = 0; i < k; i++) {
            current_sum += b[i].first;
            max_A = max(max_A, a[k - i].first);
        }

        ans = max_A * current_sum;

        for (int i = k; i < n; i++) {
            current_sum += b[i].first - b[i - k].first;
            max_A = max(max_A, a[n - i].first);
            ans = min(ans, max_A * current_sum);
        }

        cout << ans << endl;
    }
    return 0;
}
