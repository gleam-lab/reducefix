#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1e18;

void solve() {
    int n, k; cin >> n >> k;
    vector<pair<ll, ll>> ab(n);
    for (int i=0; i < n; ++i) cin >> ab[i].first;
    for (int i=0; i < n; ++i) cin >> ab[i].second;
    
    if (k == 1) {
        ll ans = INF;
        for (int i=0; i < n; ++i)
            ans = min(ans, ab[i].first * ab[i].second);
        cout << ans << '\n';
        return;
    }
    
    sort(ab.begin(), ab.end());
    
    priority_queue<ll> max_heap;
    ll sum_b = 0;
    for (int i=0; i < k-1; ++i) {
        sum_b += ab[i].second;
        max_heap.push(ab[i].second);
    }

    ll res = INF;
    for (int i=k-1; i < n; ++i) {
        ll a = ab[i].first;
        ll b = ab[i].second;
        res = min(res, (sum_b + b) * a);

        if (!max_heap.empty() && max_heap.top() > b) {
            sum_b -= max_heap.top();
            sum_b += b;
            max_heap.pop();
            max_heap.push(b);
        }
    }

    cout << res << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t; cin >> t;
    while(t--) solve();
}
