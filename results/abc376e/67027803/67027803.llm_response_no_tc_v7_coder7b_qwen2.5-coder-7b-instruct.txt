```cpp
#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL<<60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<int, int>> ab(n);
  for (int i=0; i<n; i++) cin >> ab[i].first;
  for (int i=0; i<n; i++) cin >> ab[i].second;
  sort(ab.rbegin(), ab.rend()); // Sort in descending order to maximize the product
  if (k == 1) {
    ll ans = inf;
    for (int i=0; i<n; i++) {
      ans = min(ans, static_cast<ll>(ab[i].first) * ab[i].second);
    }
    cout << ans << '\n';
    return;
  }
  priority_queue<int> q;
  int sum = 0;
  for (int i=0; i<k-1; i++) {
    q.push(ab[i].second);
    sum += ab[i].second;
  }
  ll ans = inf;
  for (int i=k-1; i<n; i++) {
    auto [a, b] = ab[i];
    ans = min(ans, (static_cast<ll>(sum) + b) * a);
    if (!q.empty() && q.top() < b) { // Check if the top element can be replaced
      sum -= q.top();
      sum += b;
      q.pop();
      q.push(b);
    }
  }
  cout << ans << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t = 1;  cin >> t;
  while (t--) solve();
}
```

### Explanation of Fixes:
1. **Sorting Order**: Changed the sorting order to descending (`rbegin`, `rend`) so that the largest elements are considered first, which helps in maximizing the product when multiplied by the smallest `K` elements of the second sequence.
2. **Avoiding Overflow**: Used `static_cast<ll>` to avoid integer overflow when calculating the product inside the loop.
3. **Efficiency**: Ensured that the priority queue operation to replace an element is only performed when necessary, improving overall efficiency.
4. **Edge Case Handling**: Correctly handled the edge case where `K = 1` by finding the minimum product directly without using a priority queue.
