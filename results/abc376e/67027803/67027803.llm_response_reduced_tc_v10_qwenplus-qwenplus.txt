#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n);
  for (int i=0; i<n; i++) cin >> ab[i].first;
  for (int i=0; i<n; i++) cin >> ab[i].second;

  // Sort by A_i ascending
  sort(ab.begin(), ab.end());

  if (k == 1) {
    ll ans = INF;
    for (int i = 0; i < n; i++) {
      ans = min(ans, ab[i].first * ab[i].second);
    }
    cout << ans << '\n';
    return;
  }

  // Max heap for B values of the k-1 smallest A's
  priority_queue<pair<ll, ll>> q;
  ll sum_b = 0;
  for (int i = 0; i < k - 1; i++) {
    q.push({ab[i].second, i});
    sum_b += ab[i].second;
  }

  ll ans = INF;

  // Try each element from k-1 to end as the max A
  for (int i = k - 1; i < n; i++) {
    auto [a, b] = ab[i];
    ans = min(ans, (sum_b + b) * a);

    // If we're not at the last element
    if (i < n - 1) {
      // Get the largest B value in our current subset
      auto [max_b_val, idx] = q.top();
      
      // If the new element has a smaller B value, replace it
      if (max_b_val > ab[i].second) {
        sum_b -= max_b_val;
        sum_b += ab[i].second;
        q.pop();
        q.push({ab[i].second, i});
      }
      
      // Always add the current element to the heap
      q.push({b, i});
    }
  }

  cout << ans << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t;
  cin >> t;
  while (t--) solve();
}
