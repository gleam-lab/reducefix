#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL<<60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n), ba(n);
  for (int i = 0; i < n; i++) {
    cin >> ab[i].first;
    ba[i].first = ab[i].first;
    ba[i].second = i;
  }
  for (int i = 0; i < n; i++) {
    cin >> ab[i].second;
    ba[i].second = ab[i].second;
  }
  sort(ab.begin(), ab.end(), greater<pair<ll, ll>>());

  if (k == 1) {
    ll ans = 0;
    for (int i = 0; i < n; i++) {
      ans = max(ans, ab[i].first * ab[i].second);
    }
    cout << ans << '\n';
    return;
  }

  ll ans = inf;
  multiset<ll> s;
  ll sum = 0;
  for (int i = 0; i < k - 1; i++) {
    s.insert(ab[i].second);
    sum += ab[i].second;
  }
  
  for (int i = k - 1; i < n; i++) {
    auto [a, b] = ab[i];
    ans = min(ans, (sum + b) * a);
    sum += b;
    s.insert(b);
    sum -= *s.begin();
    s.erase(s.begin());
  }

  cout << ans << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t; cin >> t;
  while (t--) solve();
}
