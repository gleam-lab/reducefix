#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
    int n, k;
    cin >> n >> k;
    vector<pair<ll, ll>> items(n);
    for (int i = 0; i < n; ++i) cin >> items[i].first;
    for (int i = 0; i < n; ++i) cin >> items[i].second;

    // Sort items by A_i (first value)
    sort(items.begin(), items.end());

    if (k == 1) {
        ll ans = INF;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, items[i].first * items[i].second);
        }
        cout << ans << '\n';
        return;
    }

    // Maintain a max-heap for the smallest k - 1 B_i values
    priority_queue<ll> max_heap;
    ll sum = 0;

    // Initialize with first k - 1 elements' B_i
    for (int i = 0; i < k - 1; ++i) {
        max_heap.push(items[i].second);
        sum += items[i].second;
    }

    ll ans = INF;

    // Try every possible item as the maximum A_i in subset
    for (int i = k - 1; i < n; ++i) {
        auto [a, b] = items[i];
        // Current candidate: max A_i is a, and we include this b in sum
        ans = min(ans, (sum + b) * a);

        // Maintain heap: try to keep only the smallest k - 1 B_i values
        if (!max_heap.empty() && max_heap.top() > b) {
            sum -= max_heap.top();
            sum += b;
            max_heap.pop();
            max_heap.push(b);
        }
    }

    cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin >> t;
    while (t--) solve();
}
