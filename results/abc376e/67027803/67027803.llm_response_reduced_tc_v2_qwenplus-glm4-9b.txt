#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL<<60;

ll solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n);
  for (int i=0; i<n; i++) cin >> ab[i].first;
  for (int i=0; i<n; i++) cin >> ab[i].second;
  sort(ab.begin(), ab.end());
  
  if (k == 1) {
    ll ans = inf;
    for (int i=0; i<n; i++) {
      ans = min(ans, ab[i].first * ab[i].second);
    }
    cout << ans << '\n';
    return 0;
  }
  
  ll max_a = 0, sum_b = 0, ans = inf;
  for (int i=0; i<n; i++) {
    if (i < k-1) {
      sum_b += ab[i].second;
      max_a = ab[i].first;
    } else {
      ans = min(ans, (sum_b + ab[i].second) * max_a);
      break;
    }
  }
  
  for (int i=k-2; i<n; i++) {
    max_a = max(max_a, ab[i].first);
    sum_b += ab[i].second - ab[i-k+1].second;
    ans = min(ans, (sum_b + ab[i].second) * max_a);
  }
  
  cout << ans << '\n';
  return 0;
}

int main() {
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t; cin >> t;
  while (t--) solve();
}
