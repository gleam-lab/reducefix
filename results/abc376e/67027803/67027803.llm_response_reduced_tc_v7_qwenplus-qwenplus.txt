using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n);
  for (int i = 0; i < n; i++) cin >> ab[i].first;
  for (int i = 0; i < n; i++) cin >> ab[i].second;

  sort(ab.begin(), ab.end());
  
  if (k == 1) {
    ll ans = INF;
    for (int i = 0; i < n; i++) {
      ans = min(ans, ab[i].first * ab[i].second);
    }
    cout << ans << '\n';
    return;
  }

  // We'll maintain the sum of the k-1 smallest B values seen so far
  // using a max heap to efficiently replace large B values when necessary
  priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<>> pq;
  multiset<pair<ll, ll>> remaining;
  
  ll sum = 0;
  for (int i = 0; i < n; ++i) {
    if (i < k - 1) {
      pq.emplace(ab[i].second, i);
      sum += ab[i].second;
    } else {
      remaining.emplace(ab[i].second, i);
    }
  }
  
  ll ans = INF;
  for (int i = k - 1; i < n; ++i) {
    auto [a, b] = ab[i];
    ans = min(ans, (sum + b) * a);
    
    if (i < n - 1) {
      // Consider adding current b (at index i) into our selection
      if (!pq.empty() && pq.top().first > b) {
        // Swap out larger b in pq with smaller b

      pair<ll, ll> small = *remaining.begin();
      remaining.erase(remaining.begin());
      
      pq.emplace(small);
      sum += small.first;
    }
  }

}
