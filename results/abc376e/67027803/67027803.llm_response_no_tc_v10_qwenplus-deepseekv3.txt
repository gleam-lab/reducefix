using ll = long long;
const ll inf = 1LL << 60;

void solve() {
    int n, k; cin >> n >> k;

    for (int i = 0; i < n; i++) cin >> ab[i].first;
    for (int i = 0; i < n; i++) cin >> ab[i].second;
    sort(ab.begin(), ab.end());
    
    if (k == 1) {
        ll ans = inf;
        for (int i = 0; i < n; i++) {
            ans = min(ans, ab[i].first * ab[i].second);
        }
        cout << ans << '\n';
        return;
    }
    
    priority_queue<ll, vector<ll>, greater<ll>> min_heap;
    ll sum = 0;
    for (int i = 0; i < k - 1; i++) {
        min_heap.push(ab[i].second);
        sum += ab[i].second;
    }
    
    ll ans = inf;
    for (int i = k - 1; i < n; i++) {
        ll current_sum = sum + ab[i].second;
        ans = min(ans, ab[i].first * current_sum);
        if (!min_heap.empty() && ab[i].second > min_heap.top()) {

            sum += ab[i].second;

        }
    }

}
