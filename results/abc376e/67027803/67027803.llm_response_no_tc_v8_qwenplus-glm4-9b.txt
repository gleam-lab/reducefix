#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL << 60;

void solve() {
  int n, k; cin >> n >> k;
  vector<ll> A(n);
  vector<ll> B(n);
  for (int i = 0; i < n; ++i) cin >> A[i];
  for (int i = 0; i < n; ++i) cin >> B[i];
  vector<pair<ll, ll>> ab(n);
  for (int i = 0; i < n; ++i) ab[i] = {A[i], B[i]};
  sort(ab.begin(), ab.end());

  ll ans = inf;
  ll sumB = 0;
  // Calculate the sum of B values for the first k-1 elements
  for (int i = 0; i < k - 1; ++i) {
    sumB += ab[i].second;
  }
  // Calculate the first possible answer using the first k-1 elements
  ans = min(ans, ab[k - 1].first * sumB);

  // Iterate over the remaining elements from k to n-1
  for (int i = k - 1; i < n; ++i) {
    // Calculate the new sum of B values excluding the current element
    sumB -= ab[i - k + 1].second;
    // Add the B value of the current element
    sumB += ab[i].second;
    // Update the answer
    ans = min(ans, ab[i].first * sumB);
  }

  // Output the result for the current test case
  cout << ans << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t = 1; cin >> t;
  while (t--) solve();
}
