```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
    int n, k;
    cin >> n >> k;
    vector<pair<ll, ll>> items(n);
    for (int i = 0; i < n; ++i) {
        cin >> items[i].first; // A_i
    }
    for (int i = 0; i < n; ++i) {
        cin >> items[i].second; // B_i
    }

    // Sort by A_i ascending
    sort(items.begin(), items.end());

    if (k == 1) {
        ll ans = INF;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, items[i].first * items[i].second);
        }
        cout << ans << '\n';
        return;
    }

    // Max-heap to maintain the k-1 smallest B's
    priority_queue<ll> max_heap;
    ll sum_b = 0;

    // Initialize with first k-1 elements' B values
    for (int i = 0; i < k - 1; ++i) {
        max_heap.push(items[i].second);
        sum_b += items[i].second;
    }

    ll result = INF;

    // Try every item from index k-1 to n-1 as the max A_i in subset
    for (int i = k - 1; i < n; ++i) {
        auto [a, b] = items[i];
        // Current subset: this element plus k-1 elements with smallest B so far
        result = min(result, a * (sum_b + b));

        // If we can replace the largest B in heap with a smaller one
        if (!max_heap.empty() && max_heap.top() > b) {
            sum_b -= max_heap.top();
            sum_b += b;
            max_heap.pop();
            max_heap.push(b);
        }
    }

    cout << result << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
