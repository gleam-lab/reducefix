#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL<<60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n);
  for (int i = 0; i < n; i++) cin >> ab[i].first;
  for (int i = 0; i < n; i++) cin >> ab[i].second;
  sort(ab.begin(), ab.end());
  if (k == 1) {
    ll ans = inf;
    for (int i = 0; i < n; i++) {
      ans = min(ans, ab[i].first * ab[i].second);
    }
    cout << ans << '\n';
    return;
  }
  priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
  ll sum = 0;
  for (int i = 0; i < k - 1; i++) {
    pq.push({ab[i].second, ab[i].first});
    sum += ab[i].second;
  }
  ll ans = inf;
  for (int i = k - 1; i < n; i++) {
    auto [b, a] = pq.top();
    pq.pop();
    ll current_expr = (sum + b) * a;
    ans = min(ans, current_expr);
    if (pq.empty() || pq.top().first < ab[i].second) {
      sum += ab[i].second - b;
      pq.push({ab[i].second, a});
    } else {
      pq.push({b, a});
    }
  }
  cout << ans << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t; cin >> t;
  while (t--) solve();
}
