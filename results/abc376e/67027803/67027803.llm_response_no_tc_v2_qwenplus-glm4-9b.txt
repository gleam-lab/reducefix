#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9 + 10;
const ll inf = 1LL << 60;

void solve() {
  int n, k; cin >> n >> k;
  vector<pair<ll, ll>> ab(n);
  for (int i = 0; i < n; i++) cin >> ab[i].first;
  for (int i = 0; i < n; i++) cin >> ab[i].second;
  
  // To optimize, sort pairs by the second element (B_i) in descending order
  sort(ab.begin(), ab.end(), [](const pair<ll, ll>& a, const pair<ll, ll>& b) {
    return a.second > b.second; // Sort in descending order of B_i
  });

  ll sum = 0;
  ll maxA = 0;
  
  // Find the sum of B_i for the top k elements and the maximum A_i from the same elements
  for (int i = 0; i < k; i++) {
    sum += ab[i].second;
    maxA = max(maxA, ab[i].first);
  }

  // The answer for this test case is the product of the maximum A_i for the top k and their sum of B_i
  cout << maxA * sum << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t; cin >> t;
  while (t--) solve();
}
