#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
    int n, k;
    cin >> n >> k;
    vector<pair<ll, ll>> items(n);
    for (int i = 0; i < n; ++i) {
        cin >> items[i].first; // A_i
    }
    for (int i = 0; i < n; ++i) {
        cin >> items[i].second; // B_i
    }

    // Sort by A_i ascending
    sort(items.begin(), items.end());

    if (k == 1) {
        ll ans = INF;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, items[i].first * items[i].second);
        }
        cout << ans << '\n';
        return;
    }

    // Max heap for largest k-1 B values among first i elements
    priority_queue<ll> max_heap;
    ll sum_b = 0;

    // Initialize with first k-1 elements
    for (int i = 0; i < k - 1; ++i) {
        max_heap.push(items[i].second);
        sum_b += items[i].second;
    }

    ll ans = INF;

    // Try each i from k-1 to n-1 as the max A element in subset
    for (int i = k - 1; i < n; ++i) {
        auto [a, b] = items[i];
        // We include this b in our subset and possibly remove the largest B so far
        ans = min(ans, (sum_b + b) * a);

        // Prepare for next iteration: maintain k-1 smallest B's
        if (i < n - 1) {
            if (max_heap.empty() || items[i].second < max_heap.top()) {
                if (!max_heap.empty()) {
                    sum_b += items[i].second - max_heap.top();
                    max_heap.pop();
                } else {
                    sum_b += items[i].second;
                }
                max_heap.push(items[i].second);
            }
        }
    }

    cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
