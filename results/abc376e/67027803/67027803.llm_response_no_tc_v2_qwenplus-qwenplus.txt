```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

void solve() {
    int n, k;
    cin >> n >> k;
    vector<pair<ll, ll>> ab(n);
    for (int i = 0; i < n; ++i) cin >> ab[i].first;
    for (int i = 0; i < n; ++i) cin >> ab[i].second;

    // Sort by A_i in ascending order
    sort(ab.begin(), ab.end());

    if (k == 1) {
        ll ans = INF;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, ab[i].first * ab[i].second);
        }
        cout << ans << '\n';
        return;
    }

    // Max heap to store B values of the smallest K-1 elements
    priority_queue<ll> max_heap;
    ll sum_b = 0;

    // Initialize with first K-1 elements
    for (int i = 0; i < k - 1; ++i) {
        max_heap.push(ab[i].second);
        sum_b += ab[i].second;
    }

    ll result = INF;

    // Try each possible maximum A_i from position k-1 onward
    for (int i = k - 1; i < n; ++i) {
        auto [a, b] = ab[i];
        // Current value: max(A) * sum(B) => a * (sum_b + b)
        result = min(result, a * (sum_b + b));

        // Add current b to heap if it's smaller than largest in current subset
        if (!max_heap.empty() && max_heap.top() > b) {
            sum_b -= max_heap.top();
            max_heap.pop();
            sum_b += b;
            max_heap.push(b);
        }
    }

    cout << result << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin >> t;
    while (t--) solve();
}
```
