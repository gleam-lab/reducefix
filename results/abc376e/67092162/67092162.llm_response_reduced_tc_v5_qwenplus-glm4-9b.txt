#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;
int t, n, k, a[maxN], b[maxN];
pair<int, int> p[maxN];
bool cmp(pair<int, int> p1, pair<int, int> p2) {
    if (p1.first != p2.first) return p1.first < p2.first;
    return p1.second < p2.second;
}
int main() 
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> t;
    while (t--) {
        cin >> n >> k;
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        long long sum = 0, ans;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
            cin >> b[i];
            p[i].first = a[i];
            p[i].second = b[i];
            pq.push({b[i], a[i]});
        }
        for (int i = 0; i < k; i++) {
            sum += pq.top().first;
            pq.pop();
        }
        ans = pq.top().second * sum;
        while (!pq.empty()) {
            sum -= pq.top().first;
            pq.pop();
            if (!pq.empty()) {
                sum += pq.top().first;
                pq.pop();
                ans = min(ans, pq.top().second * sum);
            }
        }
        cout << ans << "\n";
    }
    return 0;
}
