```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int maxN = 2e5 + 5;
int t, n, k;
int a[maxN], b[maxN];
pair<int, int> p[maxN];

bool cmp(pair<int, int> p1, pair<int, int> p2) {
    if (p1.first != p2.first) return p1.first < p2.first;
    return p1.second < p2.second;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> t;
    while (t--) {
        cin >> n >> k;
        priority_queue<int> pq;
        ll sum = 0;
        ll ans;
        
        for (int i = 0; i < n; i++) cin >> a[i];
        for (int i = 0; i < n; i++) cin >> b[i];
        
        for (int i = 0; i < n; i++) {
            p[i] = {a[i], b[i]};
        }
        
        sort(p, p + n, cmp);
        
        // Take the first K elements' B values
        for (int i = 0; i < k; i++) {
            sum += p[i].second;
            pq.push(p[i].second);
        }
        
        ans = (ll)p[k - 1].first * sum;
        
        // Try to improve by replacing larger B's with smaller ones
        for (int i = k; i < n; i++) {
            if (p[i].second < pq.top()) {
                sum -= pq.top();
                pq.pop();
                pq.push(p[i].second);
                sum += p[i].second;
            }
            ans = min(ans, (ll)p[i].first * sum);
        }
        
        cout << ans << "\n";
    }
    
    return 0;
}
```
