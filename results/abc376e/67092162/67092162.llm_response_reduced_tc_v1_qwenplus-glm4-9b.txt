#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;
int t, n, k;
int a[maxN], b[maxN];
pair<int, int> p[maxN];

bool cmp(const pair<int, int>& p1, const pair<int, int>& p2) {
    if (p1.first != p2.first) return p1.first < p2.first;
    return p1.second < p2.second;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> t;
    while (t--) {
        cin >> n >> k;
        long long sum = 0, ans = 0, max_a = 0;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }
        for (int i = 0; i < n; i++) {
            cin >> b[i];
            p[i].first = a[i];
            p[i].second = b[i];
        }
        
        sort(p, p + n, cmp);
        priority_queue<long long> pq;
        
        // Calculate sum of first k b[i] and max among first k a[i]
        for (int i = 0; i < k; i++) {
            pq.push(p[i].second);
            sum += p[i].second;
            max_a = max(max_a, p[i].first);
        }
        
        // Calculate the minimum possible value
        ans = max_a * sum;
        for (int i = k; i < n; i++) {
            sum += p[i].second - pq.top();
            pq.pop();
            pq.push(p[i].second);
            max_a = max(max_a, p[i].first);
            ans = min(ans, max_a * sum);
        }
        
        cout << ans << "\n";
    }
    return 0;
}
