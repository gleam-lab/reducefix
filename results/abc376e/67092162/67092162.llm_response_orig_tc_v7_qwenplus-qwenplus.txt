The main issue in your code is the incorrect handling of data types. You are using `int` to store intermediate results like `sum * a[i]`, which can easily overflow since:

- `A_i` and `B_i` can be up to `1e6`
- Sum of K `B_i`s can go up to `K * 1e6 = 2e5 * 1e6 = 2e11`
- Multiplying with max `A_i` (also 1e6) gives values up to `2e17`, which **exceeds the range of signed 32-bit integers** (`~2e9`). This causes integer overflow, resulting in negative or wrong values.

### Fix Summary:
- Replace all `int` variables involved in summation and multiplication with `long long` or `int64_t`.
- Make sure the `sum` variable used for B-values and the final result variable `ans` are of type `long long`.

#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;

int t, n, k;
long long a[maxN], b[maxN];
pair<long long, long long> p[maxN];

bool cmp(pair<long long, long long> p1, pair<long long, long long> p2) {
    return p1.first < p2.first;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> t;
    while (t--) {
        cin >> n >> k;
        priority_queue<long long> q;
        long long sum = 0, ans;
        for (int i = 1; i <= n; i++) cin >> a[i];
        for (int i = 1; i <= n; i++) cin >> b[i];
        for (int i = 1; i <= n; i++) {
            p[i].first = a[i];
            p[i].second = b[i];
        }

        sort(p + 1, p + n + 1, cmp);

        // Initialize first K elements
        for (int i = 1; i <= k; i++) {
            sum += p[i].second;
            q.push(p[i].second);
        }

        ans = p[k].first * sum;

        for (int i = k + 1; i <= n; i++) {
            if (p[i].second < q.top()) {
                sum -= q.top();
                q.pop();
                q.push(p[i].second);
                sum += p[i].second;
            }
            ans = min(ans, sum * p[i].first);
        }

        cout << ans << "\n";
    }

    return 0;
}
