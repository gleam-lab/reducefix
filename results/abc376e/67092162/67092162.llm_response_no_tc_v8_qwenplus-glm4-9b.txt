#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;
int t, n, k, a[maxN], b[maxN];

int main() 
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> t;
    while (t--) {
        cin >> n >> k;
        vector<pair<int, int>> p(n + 1);
        
        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            p[i] = {a[i], b[i]};
        }

        sort(p.begin() + 1, p.begin() + n + 1, [](const pair<int, int>& a, const pair<int, int>& b) {
            if (a.first == b.first) return a.second < b.second;
            return a.first > b.first;
        });

        long long max_a = p[k].first; // max A in the top k elements
        long long sum_b = 0; // sum of B in the top k elements
        for (int i = 1; i <= k; i++) {
            sum_b += p[i].second;
        }

        long long ans = max_a * sum_b;

        // We need to find the minimum possible value of the expression
        // by possibly replacing the smallest B's in the sum with larger A's
        long long min_sum_b = sum_b; // start with the current sum of B's
        long long best_result = ans;

        // Iterate over the remaining elements, they are sorted by A first, then by B
        for (int i = k + 1; i <= n; i++) {
            // If the current element's A is greater than the max A we have used so far, we need to check
            if (p[i].first > max_a) {
                max_a = p[i].first;
                // We need to replace one of the B's we have used with this new B to minimize the result
                min_sum_b = min(min_sum_b, sum_b - p[i].second + p[k].second);
                best_result = min(best_result, max_a * min_sum_b);
            }
        }

        cout << best_result << "\n";
    }
    return 0;
}
