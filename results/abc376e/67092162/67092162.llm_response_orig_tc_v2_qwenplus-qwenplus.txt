#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int T;
    cin >> T;
    
    while (T--) {
        int N, K;
        cin >> N >> K;
        
        vector<pair<int, int>> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort by A_i ascending
        sort(items.begin(), items.end());

        priority_queue<int> max_heap;
        long long sum_b = 0;

        // Take first K elements
        for (int i = 0; i < K; ++i) {
            sum_b += items[i].second;
            max_heap.push(items[i].second);
        }

        long long result = 1LL * items[K - 1].first * sum_b;

        // Try replacing the largest B_i so far with smaller ones to reduce sum
        for (int i = K; i < N; ++i) {
            if (!max_heap.empty() && items[i].second < max_heap.top()) {
                sum_b -= max_heap.top();
                max_heap.pop();
                sum_b += items[i].second;
                max_heap.push(items[i].second);
            }
            result = min(result, 1LL * items[i].first * sum_b);
        }

        cout << result << '\n';
    }

    return 0;
}
