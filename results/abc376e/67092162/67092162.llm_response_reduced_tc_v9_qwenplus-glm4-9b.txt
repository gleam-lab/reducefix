#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;
int t, n, k;
int a[maxN], b[maxN];
pair<int, int> p[maxN];
bool cmp(pair<int, int> p1, pair<int, int> p2) {
  if (p1.first != p2.first) return p1.first < p2.first;
  return p1.second < p2.second;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> t;
    while (t--) {
        cin >> n >> k;
        long long sum = 0, ans = 0;
        for (int i = 0; i < n; i++) cin >> a[i];
        for (int i = 0; i < n; i++) cin >> b[i];
        for (int i = 0; i < n; i++) {
            p[i].first = a[i];
            p[i].second = b[i];
        }
        sort(p, p + n, cmp);
        
        // Calculate the initial sum with the k largest elements
        for (int i = n - 1; i >= n - k; i--) {
            sum += p[i].second;
        }
        
        // Calculate the minimum value of the expression for the first k elements
        ans = max(p[n - 1].first, 0LL) * sum;
        
        // Slide the window from k to n and calculate the minimum value
        for (int i = n - k; i >= 0; i--) {
            sum -= p[i].second;
            sum += p[n - 1 - (i - (n - k))].second;
            int currentMax = max(p[i].first, p[n - 1 - (i - (n - k))].first);
            ans = min(ans, currentMax * sum);
        }
        
        cout << ans << '\n';
    }
    return 0;
}
