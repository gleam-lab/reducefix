#include <bits/stdc++.h>
using namespace std;
const int maxN = 2e5 + 5;

int t, n, k;
long long a[maxN], b[maxN];
pair<long long, long long> p[maxN];

bool cmp(pair<long long, long long>& p1, pair<long long, long long>& p2) {
    return p1.first < p2.first;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> t;
    while (t--) {
        cin >> n >> k;
        priority_queue<long long> q;
        long long sum = 0;
        long long ans;

        for (int i = 0; i < n; i++) cin >> a[i];
        for (int i = 0; i < n; i++) cin >> b[i];

        for (int i = 0; i < n; i++) {
            p[i].first = a[i];
            p[i].second = b[i];
        }

        sort(p, p + n, cmp);

        sum = 0;
        for (int i = 0; i < k; i++) {
            sum += p[i].second;
            q.push(p[i].second);
        }

        ans = p[k - 1].first * sum;

        for (int i = k; i < n; i++) {
            if (q.top() > p[i].second) {
                sum -= q.top();
                q.pop();
                q.push(p[i].second);
                sum += p[i].second;
            }
            ans = min(ans, sum * p[i].first);
        }

        cout << ans << "\n";
    }

    return 0;
}
