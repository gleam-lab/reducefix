```cpp
#include <bits/stdc++.h>
using namespace std;

const int maxN = 2e5 + 5;
int t, n, k, a[maxN], b[maxN];
pair<int, int> p[maxN];

bool cmp(pair<int, int> p1, pair<int, int> p2) {
    if (p1.first != p2.first) return p1.first > p2.first;
    return p1.second > p2.second;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> t;
    while (t--) {
        cin >> n >> k;
        priority_queue<int> q;
        long long sum = 0, ans = LLONG_MAX;
        
        for (int i = 1; i <= n; i++) cin >> a[i];
        for (int i = 1; i <= n; i++) cin >> b[i];
        
        for (int i = 1; i <= n; i++) {
            p[i] = {a[i], b[i]};
        }
        
        sort(p + 1, p + n + 1, cmp);
        
        for (int i = 1; i <= k; i++) {
            sum += p[i].second;
            q.push(-p[i].second); // Using negative to simulate min-heap with max-heap
        }
        
        ans = (long long)p[k].first * sum;
        
        for (int i = k + 1; i <= n; i++) {
            if (-q.top() > p[i].second) {
                sum -= -q.top();
                q.pop();
                q.push(-p[i].second);
                sum += p[i].second;
            }
            ans = min(ans, (long long)p[i].first * sum);
        }
        
        cout << ans << "\n";
    }
    return 0;
}
```

### Explanation of Changes:
1. **Priority Queue Usage**: Changed the priority queue to store negative values of `b[i]` to simulate a min-heap using a max-heap.
2. **Initialization of `ans`**: Initialized `ans` to `LLONG_MAX` to ensure any valid result will be smaller.
3. **Loop for Calculating Minimum Value**: Corrected the loop logic to properly update the sum and check for the minimum value.
4. **Type Casting**: Used `long long` for `ans` and intermediate calculations to avoid overflow issues.

This solution ensures that the problem constraints are met efficiently and correctly.
