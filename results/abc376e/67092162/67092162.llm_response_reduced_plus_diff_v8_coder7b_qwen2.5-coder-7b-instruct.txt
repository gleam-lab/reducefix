#include <bits/stdc++.h>
using namespace std;

struct Element {
    long long val, sum;
};

bool cmp(const Element &a, const Element &b) {
    if (a.val == b.val) return a.sum > b.sum;
    return a.val < b.val;
}

long long solve(long long n, long long k, vector<long long> &a, vector<long long> &b) {
    vector<Element> vec(n);
    for (long long i = 0; i < n; ++i) vec[i] = {a[i], b[i]};
    sort(vec.begin(), vec.end(), cmp);

    priority_queue<long long> pq;
    long long total_sum = 0, result = LLONG_MAX;
    for (long long i = 0; i < k; ++i) {
        total_sum += vec[i].sum;
        pq.push(-vec[i].sum);
    }

    result = min(result, total_sum * vec[k - 1].val);
    for (long long i = k; i < n; ++i) {
        total_sum -= pq.top(); pq.pop();
        total_sum += vec[i].sum;
        pq.push(-vec[i].sum);
        result = min(result, total_sum * vec[i].val);
    }
    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    long long t, n, k;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        vector<long long> a(n), b(n);
        for (long long i = 0; i < n; ++i) cin >> a[i];
        for (long long i = 0; i < n; ++i) cin >> b[i];

        cout << solve(n, k, a, b) << '\n';
    }
    return 0;
}
