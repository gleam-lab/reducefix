#include<bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<int>;
using P = pair<int, ll>;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<P> p(n);
        vector<ll> sumB(n);
        
        for (int i = 0; i < n; ++i) {
            cin >> p[i].first;
        }
        for (int i = 0; i < n; ++i) {
            cin >> p[i].second;
            sumB[i] = p[i].second;
        }

        // Sort the pairs first by A_i in ascending order
        sort(p.begin(), p.end(), [](const P& a, const P& b) { return a.first < b.first; });
        
        // Calculate prefix sums of B_i for efficient sum calculation
        for (int i = 1; i < n; ++i) {
            sumB[i] += sumB[i - 1];
        }
        
        // Initialize the answer to a large value
        ll ans = LLONG_MAX;
        
        // Iterate over all possible smallest values of A_i in the subset of size K
        for (int i = 0; i <= k; ++i) {
            // Calculate the sum of B_i for the first i elements
            ll current_sumB = (i == 0) ? 0 : sumB[i - 1];
            
            // The maximum value of A_i in the subset is p[i - 1].first
            ans = min(ans, p[i - 1].first * current_sumB);
        }
        
        // Output the minimum possible value for the current test case
        cout << ans << '\n';
    }
}
