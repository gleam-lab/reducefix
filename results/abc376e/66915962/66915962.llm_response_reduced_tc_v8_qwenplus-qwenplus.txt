#include<bits/stdc++.h>

        rep(i, n) {
            cin >> A[i].first;
            A[i].second = i;
        }
        
        // Read B values
        vector<int> B(n);
        rep(i, n) cin >> B[i];
        
        // Sort items by A value
        sort(A.begin(), A.end());
        
        // Initialize sum of B's and min-heap to track smallest K B's
        multiset<ll> heap;
        ll sumB = 0;
        
        // Initialize with first K elements based on sorted A
        rep(i, k) {
            sumB += B[A[i].second];
            heap.insert(B[A[i].second]);
        }
        
        ll ans = A[k-1].first * sumB;
        
        // Process remaining elements
        rep2(i, k, n) {
            // Remove the largest B in the current K set
            sumB -= *prev(heap.end());
            heap.erase(prev(heap.end()));
            
            // Add the new B value
            sumB += B[A[i].second];
            heap.insert(B[A[i].second]);
            
            // Update answer
            ans = min(ans, A[i].first * sumB);
        }
        
        cout << ans << '\n';
    }
}
