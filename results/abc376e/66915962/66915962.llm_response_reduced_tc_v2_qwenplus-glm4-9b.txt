#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
using ll = long long;
using P = pair<ll, ll>;
using VI = vector<ll>;

int main() {
    ll t;
    cin >> t;
    while (t--) {
        ll n, k;
        cin >> n >> k;
        vector<ll> a(n), b(n);
        for (ll i = 0; i < n; ++i) cin >> a[i];
        for (ll i = 0; i < n; ++i) cin >> b[i];

        // Pair elements of A and B, and sort by B values (descending)
        vector<P> arr(n);
        for (ll i = 0; i < n; ++i) arr[i] = {b[i], a[i]};
        sort(arr.rbegin(), arr.rend());

        ll maxA = arr[0].second;
        ll sumB = arr[0].first;
        ll ans = maxA * sumB;

        // Try to find a better solution by reducing sumB and keeping maxA
        for (ll j = 1; j < n; ++j) {
            ll currentMaxA = arr[j].second;
            ll currentSumB = arr[j].first + sumB;
            ans = min(ans, currentMaxA * currentSumB);
        }

        cout << ans << '\n';
    }
    return 0;
}
