#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<int>;
using P = pair<int, int>;
constexpr int INF = 1001001001;
constexpr ll LINF = 1001001001001001001ll;

#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, k, n) for (ll i = k; i < (ll)(n); i++)

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t;
    cin >> t;

    rep(qi, t) {
        int n, k;
        cin >> n >> k;
        vector<P> items(n);
        vector<int> b(n);
        rep(i, n) cin >> items[i].first;
        rep(i, n) {
            cin >> items[i].second;
            b[i] = items[i].second;
        }

        // Sort by A_i ascending
        sort(items.begin(), items.end());
        // Rebuild b in order of sorted A
        rep(i, n) b[i] = items[i].second;

        // Use a min-heap to maintain the K smallest B values among first i+1 elements
        priority_queue<int, vector<int>, greater<>> pq;
        ll sumB = 0;
        rep(i, k) {
            pq.push(b[i]);
            sumB += b[i];
        }

        ll ans = (ll)items[k - 1].first * sumB;

        rep2(i, k, n) {
            if (b[i] > pq.top()) {
                sumB -= pq.top();
                pq.pop();
                pq.push(b[i]);
                sumB += b[i];
            }
            ans = min(ans, (ll)items[i].first * sumB);
        }

        cout << ans << '\n';
    }

    return 0;
}
