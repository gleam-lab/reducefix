#include<bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<ll>;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    ll T;
    cin >> T;
    while (T--) {
        ll N, K;
        cin >> N >> K;
        VI A(N), B(N);
        for (ll i = 0; i < N; ++i) {
            cin >> A[i];
        }
        for (ll i = 0; i < N; ++i) {
            cin >> B[i];
        }
        
        // Sort A and B by A in descending order, and sort the indices accordingly
        vector<pair<ll, ll>> sorted_indices(N);
        for (ll i = 0; i < N; ++i) {
            sorted_indices[i] = {A[i], i};
        }
        sort(sorted_indices.rbegin(), sorted_indices.rend());
        
        ll ans = LINF;
        ll sumB = 0;
        multiset<ll> B_elements; // To keep track of the B[i] values that are currently included in the sum
        
        // Add the first K elements to the sum and multiset
        for (ll i = 0; i < K; ++i) {
            sumB += B[sorted_indices[i].second];
            B_elements.insert(B[sorted_indices[i].second]);
        }
        // Calculate the minimum possible value for this configuration
        ans = min(ans, sorted_indices[0].first * sumB);
        
        // Now try to minimize the expression by considering the rest of the elements
        for (ll i = K; i < N; ++i) {
            // Remove the element that is currently in the multiset and sumB
            ll removed_element = *B_elements.begin();
            sumB -= removed_element;
            B_elements.erase(B_elements.begin());
            
            // Add the next element in the sorted order
            sumB += B[sorted_indices[i].second];
            B_elements.insert(B[sorted_indices[i].second]);
            
            // Update the answer with the new maximum A_i and the current sum of B_i's
            ans = min(ans, sorted_indices[i].first * sumB);
        }
        
        // Output the result for the current test case
        cout << ans << '\n';
    }
    
    return 0;
}
