#include<bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<int>;
using P = pair<int, int>;
constexpr int INF = 1001001001;
constexpr ll LINF = 1001001001001001001ll;
#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, k, n) for (ll i = k; i < (ll)(n); i++)
#define OUT(x) cout << #x << ":" << x << endl

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t;
    cin >> t;
    while(t--){
        int n, k;
        cin >> n >> k;
        vector<ll> a(n), b(n);
        vector<pair<ll, int>> ab(n);
        rep(i, n){
            cin >> a[i];
            cin >> b[i];
            ab[i] = {a[i], b[i]};
        }
        sort(ab.begin(), ab.end());

        ll ans = LLONG_MAX;
        multiset<ll> bs;
        ll sumB = 0;
        rep(i, k){
            bs.insert(ab[i].second);
            sumB += ab[i].second;
        }

        ans = min(ans, ab[k-1].first * sumB);
        
        rep2(i, k, n){
            ll smax = *bs.rbegin();
            bs.erase(bs.find(smax));
            bs.insert(ab[i].second);
            sumB += ab[i].second - smax;

            ans = min(ans, ab[i].first * sumB);
        } 
        
        cout << ans << '\n';
    }
}
