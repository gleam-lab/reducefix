#include<bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<int>;
using P = pair<int, int>;
constexpr int INF = 1001001001;
constexpr ll LINF = 1001001001001001001ll;
#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, k, n) for (ll i = k; i < (ll)(n); i++)
#define OUT(x) cout << #x << ":" << x << endl

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int t;
    cin >> t;
    
    while(t--){
        int n, k;
        cin >> n >> k;
        vector<ll> a(n), b(n);
        
        rep(i, n) cin >> a[i];
        rep(i, n) cin >> b[i];
        
        vector<pair<ll, ll>> ab(n);
        rep(i, n) ab[i] = {a[i], b[i]};
        
        sort(ab.begin(), ab.end());
        
        ll ans = LINF;
        ll sumB = 0;
        multiset<ll> s;
        
        rep(i, k){
            s.insert(ab[i].second);
            sumB += ab[i].second;
        }
        
        ans = min(ans, ab[k-1].first * sumB);
        
        rep2(i, k, n){
            ll smax = *s.rbegin();
            s.erase(s.find(smax));
            s.insert(ab[i].second);
            sumB += ab[i].second - smax;
            
            ans = min(ans, ab[i].first * sumB);
        } 
        
        cout << ans << '\n';
    }

    return 0;
}
