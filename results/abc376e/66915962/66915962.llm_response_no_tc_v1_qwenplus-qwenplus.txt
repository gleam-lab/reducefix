#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<int, int>;

constexpr ll LINF = 1001001001001001001ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;
        vector<P> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort by A_i in ascending order
        sort(items.begin(), items.end());

        multiset<int> min_heap; // To keep track of smallest K B_i's
        ll sumB = 0;

        // Initialize with first K elements
        for (int i = 0; i < K; ++i) {
            sumB += items[i].second;
            min_heap.insert(items[i].second);
        }

        ll ans = 1LL * items[K - 1].first * sumB;

        // Try adding better combinations
        for (int i = K; i < N; ++i) {
            // If current B_i is smaller than max in heap, replace it
            if (items[i].second < *min_heap.rbegin()) {
                sumB -= *min_heap.rbegin();
                sumB += items[i].second;
                min_heap.erase(prev(min_heap.end()));
                min_heap.insert(items[i].second);
            }
            ans = min(ans, 1LL * items[i].first * sumB);
        }

        cout << ans << '\n';
    }

    return 0;
}
