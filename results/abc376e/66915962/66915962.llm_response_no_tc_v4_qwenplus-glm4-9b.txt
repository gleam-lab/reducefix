#define OUT(x) cout << #x << ":" << x << endl

int main(){
    int t;
    cin >> t;
    while(t--){
        int n, k;
        cin >> n >> k;
        vector<P> ab(n);
        for (int i = 0; i < n; ++i) {
            cin >> ab[i].first >> ab[i].second;
        }
        sort(ab.begin(), ab.end(), [](const P &a, const P &b) {
            if (a.first != b.first) return a.first > b.first;
            return a.second > b.second;
        });
        
        // Calculate the initial answer using the largest k elements
        ll ans = 0;
        ll sumB = 0;
        for (int i = n - 1; i >= n - k; --i) {
            sumB += ab[i].second;
            ans = min(ans, (ab[i].first * sumB));
        }
        
        cout << ans << '\n';
    }
}
