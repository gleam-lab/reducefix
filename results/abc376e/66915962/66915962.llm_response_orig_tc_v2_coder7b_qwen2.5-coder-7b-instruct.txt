#include<bits/stdc++.h>
using namespace std;
using ll = long long;
using VI = vector<int>;
using P = pair<int, int>;
constexpr int INF = 1001001001;
constexpr ll LINF = 1001001001001001001ll;
#define rep(i, n) for (ll i = 0; i < (ll)(n); i++)
#define rep2(i, k, n) for (ll i = k; i < (ll)(n); i++)
#define OUT(x) cout << #x << ":" << x << endl

int main(){
    int t;
    cin >> t;
    while(t--){
        int n, k;
        cin >> n >> k;
        vector<ll> a(n), b(n);
        rep(i, n) cin >> a[i];
        rep(i, n) cin >> b[i];

        priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> pq;
        rep(i, n) pq.push({b[i], a[i]});
        
        ll ans = LINF;
        ll sumB = 0;
        vector<ll> maxA(k);
        rep(i, k){
            sumB += pq.top().first;
            maxA[i] = pq.top().second;
            pq.pop();
        }

        ans = min(ans, sumB * maxA.back());

        rep2(i, k, n){
            sumB -= pq.top().first;
            maxA.erase(find(maxA.begin(), maxA.end(), pq.top().second));
            pq.pop();

            sumB += pq.top().first;
            maxA.push_back(pq.top().second);
            pq.pop();
            sort(maxA.begin(), maxA.end());
            
            ans = min(ans, sumB * maxA.back());
        }
        cout << ans << '\n';
    }
}
