#include<bits/stdc++.h>

        vector<ll> A(n), B(n);
        rep(i, n) cin >> A[i];
        rep(i, n) cin >> B[i];

        // Create a vector of pairs (A[i], B[i]) to sort by A[i]
        vector<P> AB(n);
        rep(i, n)_AB[i] = {A[i], B[i]};
        sort(AB.begin(), AB.end()); // Sort by A[i] values

        ll ans = LLONG_MAX;
        ll maxA = AB[k-1].first; // max A[i] in the first k elements
        ll sumB = 0;

        // Calculate sumB for the first k elements
        for (int i = 0; i < k; ++i) {
            sumB += AB[i].second;
        }

        // Calculate the answer using the first k elements
        ans = min(ans, maxA * sumB);

        // Use a min-heap to keep track of the smallest B[i] values
        priority_queue<int, vector<int>, greater<int>> min_heap;
        rep(i, k) {
            min_heap.push(AB[i].second);
        }

        // Consider the rest of the elements
        for (int i = k; i < n; ++i) {
            sumB -= min_heap.top(); // Remove the smallest B[i] from sumB
            min_heap.pop();
            sumB += AB[i].second; // Add the current B[i] to sumB
            min_heap.push(AB[i].second); // Push the current B[i] to the heap
            ll currentMaxA = AB[i].first; // max A[i] among the last k elements
            ans = min(ans, currentMaxA * sumB);
        }

        cout << ans << '\n';
    }
    return 0;
}
