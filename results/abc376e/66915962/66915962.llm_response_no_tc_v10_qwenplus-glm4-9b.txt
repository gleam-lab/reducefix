#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<int> a(n), b(n);
        ll sumB = 0;
        
        // Read and sort the A sequence with corresponding B values
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
            b[i] = a[i] * b[i];
            sumB += b[i];
        }
        sort(a.begin(), a.end());

        // Initialize the answer with the first possible combination
        ll ans = a[k-1] * sumB;

        // Try to find a better combination by replacing elements
        ll current_maxA = a[k-1];
        ll current_sumB = sumB;
        for (int i = 0; i < k; ++i) {
            // Remove the current element and find the next maximum A
            current_sumB -= b[i];
            
            // We need to find the maximum element in the remaining elements of A
            // that is greater than current_maxA
            ll next_maxA = current_maxA;
            while (!binary_search(a.begin() + k, a.end(), next_maxA)) {
                next_maxA--;
                if (next_maxA < current_maxA) break; // This should not happen
            }
            
            // Calculate the new sum B by considering the elements from k to n
            ll new_sumB = current_sumB - b[i];
            for (int j = i; j < n; ++j) {
                new_sumB += a[j] * b[j];
                if (a[j] == next_maxA) break;
            }
            
            // Update the answer if the new value is smaller
            ans = min(ans, next_maxA * new_sumB);
        }
        
        cout << ans << endl;
    }
    return 0;
}

This code reads the input for each test case, sorts the array A, and computes the initial answer by iterating over the elements of B and updating the sum and maximum A accordingly. It then tries to find a better combination by replacing an element in B with another (while keeping the size of the subset S the same) and calculates the new answer. The algorithm uses binary search to efficiently find the maximum element in the remaining array A that is greater than the current maximum A. The final answer for each test case is printed at the end.
