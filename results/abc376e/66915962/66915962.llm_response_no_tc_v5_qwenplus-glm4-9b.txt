#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<int> a(n), b(n);
        vector<pair<int, int>> ab(n); // Pair of (A[i], B[i])

        for (int i = 0; i < n; ++i) {
            cin >> a[i] >> b[i];
            ab[i] = {a[i], b[i]};
        }

        // Sort the pairs by A[i] in descending order
        sort(ab.begin(), ab.end(), [](const pair<int, int>& l, const pair<int, int>& r) {
            return l.first > r.first;
        });

        ll ans = LLONG_MAX;
        long long sumB = 0;
        set<int> sB; // To store the K largest B[i] values

        // Initial window of size K
        for (int i = 0; i < k; ++i) {
            sB.insert(ab[i].second);
            sumB += ab[i].second;
        }

        ans = min(ans, ab[k - 1].first * sumB);

        // Slide the window from position k to n-1
        for (int i = k; i < n; ++i) {
            sumB += ab[i].second - ab[i - k].second; // Update sumB by removing the smallest B[i-k] and adding the next B[i]

            // Find the new minimum in the current set of K B[i] values
            auto it = sB.begin();
            while (it != sB.end() && *it < ab[i - k].second) {
                it++;
            }

            // If it points to the smallest element, remove it
            if (it != sB.begin() && *it == ab[i - k].second) {
                sB.erase(it);
            }

            // Insert the new B[i] value
            sB.insert(ab[i].second);

            // Calculate the potential answer with the new maximum A[i]
            ans = min(ans, ab[i].first * sumB);
        }

        cout << ans << '\n';
    }

    return 0;
}
