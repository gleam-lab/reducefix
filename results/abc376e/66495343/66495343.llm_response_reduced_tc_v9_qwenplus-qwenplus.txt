#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int,int>
#define fi first
#define se second

const int INF = 1E18;

signed main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	
	int t;
	cin >> t;
	
	while (t--)
	{
		int n, k;
		cin >> n >> k;
		
		vector<int> a(n), b(n);
		for (int &x : a) cin >> x;
		for (int &x : b) cin >> x;
		
		// Pair A_i and B_i together
		vector<PII> items(n);
		for (int i = 0; i < n; ++i)
			items[i] = {a[i], b[i]};
		
		// Sort by A_i ascending
		sort(items.begin(), items.end());
		
		// Minimize max_A * sum_B
		int res = INF;
		int sum_b = 0;
		
		// We maintain the smallest K-1 B_i's among the first i-1 items
		// using a max-heap (so we can remove the largest if we exceed K-1)
		priority_queue<int> pq;
		
		for (int i = 0; i < n; ++i)
		{
			int cur_a = items[i].first;
			int cur_b = items[i].second;
			
			// We include this item in our set S
			// This will be the maximum A so far since list is sorted
			
			if ((int)pq.size() < k - 1)
			{
				// We need to collect at least K-1 elements for the heap
				sum_b += cur_b;
				pq.push(cur_b);
				
				// Not enough elements yet to form a valid subset of size K
				continue;
			}
			else
			{
				// We now have K elements including current one
				
				// If heap not full, add to it
				if (pq.size() < k)
				{
					sum_b += cur_b;
					pq.push(cur_b);
				}
				else
				{
					// Check if we should replace the biggest in heap
					if (cur_b > pq.top())
					{
						// Replace
						sum_b -= pq.top();
						sum_b += cur_b;
						pq.pop();
						pq.push(cur_b);
					}
				}
				
				// Now compute candidate result
				res = min(res, cur_a * sum_b);
			}
		}
		
		cout << res << "\n";
	}
	
	return 0;
}
