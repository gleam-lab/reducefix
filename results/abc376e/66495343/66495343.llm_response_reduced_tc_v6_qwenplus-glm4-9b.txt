#include<bits/stdc++.h>
using namespace std;
#define int long long

int t, n, k;
int a[200010], b[200010];

int solve() {
    sort(a + 1, a + n + 1);
    sort(b + 1, b + n + 1);
    
    long long ans = 0;
    long long sumB = 0;
    long long maxA = 0;
    int countK = 0;
    
    auto check = [&](long long c) {
        long long curSumB = 0, curMaxA = 0;
        int cnt = 0;
        for (int i = 1; i <= n; ++i) {
            if (a[i] > c) {
                curMaxA = max(curMaxA, a[i]);
                curSumB += b[i];
                cnt++;
            }
        }
        return curMaxA * curSumB > ans && cnt == k;
    };
    
    int l = 0, r = *max_element(a + 1, a + n + 1);
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid)) {
            ans = curMaxA * curSumB;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    
    return ans;
}

int main() {
    cin >> t;
    while (t--) {
        cin >> n >> k;
        for (int i = 0; i < n; ++i)
            cin >> a[i + 1];
        for (int i = 0; i < n; ++i)
            cin >> b[i + 1];
        cout << solve() << '\n';
    }
    return 0;
}
