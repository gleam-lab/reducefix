
```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

int a[200010], b[200010];
int sa[200010];

int32_t main() {
    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        for (int i = 0; i < n; i++) {
            cin >> b[i];
        }

        for (int i = 0; i < n; i++) {
            sa[i] = mini(a[i], b[i]);
        }

        sort(sa, sa + n, greater<pair<int, int>>());

        long long res = LLONG_MAX;
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            if (i == 0 || sa[i].first > sa[i - 1].first) {
                sum = 0;
            }
            sum += sa[i].second;
            res = min(res, (long long)sa[i].first * sum);
        }

        cout << res << '\n';
    }

    return 0;
}

pair<int, int> mini(int x, int y) {
    return {x, y};
}
```
