		for(int i=1;i<=n;i++)

		sort(sa+1,sa+1+n);
		
		int res=INF,sum=0;
		priority_queue<int> q;
		for(int i=n;i>=1;i--)
		{
			if(n-i+1<=k-1)
			{
				sum+=sa[i].se;
				q.push(sa[i].se);
				continue;
			}
			
			int na=sa[i].fi;
			sum+=sa[i].se;
			
			if(q.size()>k-1)
			{
				int big=q.top();
				q.pop();
				sum-=big;
			}
			
			q.push(sa[i].se);
			res=min(res,na*sum);
		}
