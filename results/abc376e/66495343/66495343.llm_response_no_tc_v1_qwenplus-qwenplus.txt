```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define PII pair<int, int>

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        vector<PII> pairs(n);
        for (int i = 0; i < n; ++i) {
            cin >> pairs[i].first;
        }
        for (int i = 0; i < n; ++i) {
            cin >> pairs[i].second;
        }

        // Sort by A_i in ascending order
        sort(pairs.begin(), pairs.end());

        int sum_b = 0;
        priority_queue<int> max_heap; // To keep the smallest (k-1) B_i's

        // Initialize sum of smallest k-1 B values
        for (int i = 0; i < k - 1; ++i) {
            sum_b += pairs[i].second;
            max_heap.push(pairs[i].second);
        }

        int res = INF;

        for (int i = k - 1; i < n; ++i) {
            // Current A is pairs[i].first, include it in the set
            sum_b += pairs[i].second;
            max_heap.push(pairs[i].second);

            // If heap size exceeds k-1, remove the largest B in heap
            if (max_heap.size() > k) {
                sum_b -= max_heap.top();
                max_heap.pop();
            }

            // Compute current cost and update minimum
            res = min(res, pairs[i].first * sum_b);
        }

        cout << res << '\n';
    }

    return 0;
}
```
