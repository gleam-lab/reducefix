#include <bits/stdc++.h>
using namespace std;

#define int long long
#define PII pair<int, int>

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<PII> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort items by A_i ascending
        sort(items.begin(), items.end());

        int sum_B = 0;
        priority_queue<int> max_heap; // To keep the smallest K B_i values
        int res = INF;

        for (int i = 0; i < N; ++i) {
            int A_i = items[i].first;
            int B_i = items[i].second;

            // Maintain a min-heap of size K to keep the smallest K B_i values
            if (max_heap.size() < K) {
                sum_B += B_i;
                max_heap.push(B_i); // Stored as positive since it's a max-heap
            } else {
                // If current B_i is smaller than the largest in heap, replace it
                if (!max_heap.empty() && B_i < max_heap.top()) {
                    sum_B -= max_heap.top();
                    max_heap.pop();
                    sum_B += B_i;
                    max_heap.push(B_i);
                }
            }

            if (max_heap.size() == K) {
                res = min(res, A_i * sum_B);
            }
        }

        cout << res << '\n';
    }

    return 0;
}
