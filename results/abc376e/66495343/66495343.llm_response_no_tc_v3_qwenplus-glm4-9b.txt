#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int, int>
#define fi first
#define se second

const int N = 2e5 + 10, INF = 0x3f3f3f3f, mod = 1e9 + 7;

int a[N], b[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int T;
    cin >> T;

    while (T--) {
        int n, k;
        cin >> n >> k;

        for (int i = 1; i <= n; i++)
            cin >> a[i];
        
        for (int i = 1; i <= n; i++)
            cin >> b[i];
        
        vector<PII> sa(n + 1); // PII of (value from A, sum of corresponding elements from B)
        for (int i = 1; i <= n; i++)
            sa[i] = {a[i], b[i]};

        // Sort by the value from A in ascending order
        sort(sa.begin() + 1, sa.begin() + 1 + n);

        // Initial sum of B elements for the first k elements
        long long sum = 0;
        for (int i = 1; i <= min(k, n); i++)
            sum += sa[i].se;

        long long res = INF;
        // Use a max heap to keep track of the top k elements from B
        priority_queue<int> maxheap;
        for (int i = 1; i <= k; i++) {
            maxheap.push(sa[i].se);
        }

        // Evaluate the expression for the first k elements
        res = min(res, sa[k].fi * sum);

        // Iterate over the rest of the elements
        for (int i = k + 1; i <= n; i++) {
            // Remove the smallest from the previous k elements sum
            sum -= maxheap.top();
            maxheap.pop();
            // Add the new element to the sum and max heap
            sum += sa[i].se;
            maxheap.push(sa[i].se);
            // Evaluate the expression with the new set of k elements
            res = min(res, sa[i].fi * sum);
        }

        cout << res << endl;
    }

    return 0;
}
