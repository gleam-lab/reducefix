```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int,int>
#define fi first
#define se second

const int INF = 1e18;

signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int t;
    cin >> t;

    while (t--)
    {
        int n, k;
        cin >> n >> k;

        vector<PII> items(n);
        for (int i = 0; i < n; ++i)
            cin >> items[i].first;
        
        for (int i = 0; i < n; ++i)
            cin >> items[i].second;

        // Sort by A value ascending
        sort(items.begin(), items.end());

        int res = INF;
        int sum_b = 0;
        priority_queue<int> max_heap; // To keep the smallest K-1 B values (as we want to remove largest when needed)

        for (int i = 0; i < n; ++i)
        {
            if (i < k - 1)
            {
                sum_b += items[i].se;
                max_heap.push(items[i].se);
            }
            else
            {
                // We consider selecting this item as the max A
                int curr_a = items[i].fi;
                sum_b += items[i].se;
                max_heap.push(items[i].se);

                // If heap size exceeds K, remove the largest B value
                if (max_heap.size() > k)
                {
                    sum_b -= max_heap.top();
                    max_heap.pop();
                }

                res = min(res, curr_a * sum_b);
            }
        }

        cout << res << "\n";
    }

    return 0;
}
```
