```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define PII pair<int, int>
#define fi first
#define se second

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        vector<PII> items(n);
        for (int i = 0; i < n; ++i) {
            cin >> items[i].fi; // A_i
        }
        for (int i = 0; i < n; ++i) {
            cin >> items[i].se; // B_i
        }

        // Sort items by A_i ascending
        sort(items.begin(), items.end());

        // Minimize (max_A_S) * (sum_B_S), where |S| = K
        int res = INF;
        int sum_b = 0;
        priority_queue<int> max_heap; // to keep track of the smallest (K-1) B's

        // We will process each item as the one that gives max A in subset S
        for (int i = 0; i < n; ++i) {
            int a_i = items[i].fi;
            int b_i = items[i].se;

            if (max_heap.size() < k - 1) {
                sum_b += b_i;
                max_heap.push(-b_i); // use min heap via negation
            } else {
                // Try adding current item and removing largest of the small ones
                sum_b += b_i;
                if (!max_heap.empty()) {
                    int removed = -max_heap.top();
                    sum_b -= removed;
                    max_heap.pop();
                    max_heap.push(-b_i);
                }

                res = min(res, a_i * sum_b);
            }
        }

        cout << res << "\n";
    }

    return 0;
}
```
