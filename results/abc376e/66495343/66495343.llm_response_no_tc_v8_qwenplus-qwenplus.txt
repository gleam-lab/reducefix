```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define PII pair<int, int>

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<PII> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort by A_i in ascending order
        sort(items.begin(), items.end());

        int sum_B = 0;
        priority_queue<int> max_heap; // To keep the largest K-1 B values

        // Initialize with first K-1 elements
        for (int i = 0; i < K - 1; ++i) {
            sum_B += items[i].second;
            max_heap.push(items[i].second);
        }

        int result = INF;

        // Now process from K-1 to N-1 (0-based)
        for (int i = K - 1; i < N; ++i) {
            // Current A_i is items[i].first
            // We have already selected K-1 items from before i

            // Add current item's B value
            sum_B += items[i].second;

            // Maintain only top K-1 B values among previously chosen ones
            if (max_heap.size() >= K) {
                int large = max_heap.top();
                max_heap.pop();
                sum_B -= large;
            }

            max_heap.push(items[i].second);

            // Compute current score: A_i * sum_B
            result = min(result, items[i].first * sum_B);
        }

        cout << result << '\n';
    }

    return 0;
}
```
