#include <bits/stdc++.h>
using namespace std;

#define int long long
#define PII pair<int, int>

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<PII> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort items by A_i in ascending order
        sort(items.begin(), items.end());

        int sum_B = 0;
        priority_queue<int> max_heap; // To keep the smallest K B_i values

        // Initialize with first K-1 elements
        for (int i = 0; i < K - 1; ++i) {
            sum_B += items[i].second;
            max_heap.push(items[i].second);
        }

        int result = INF;

        // Try including each item from position K-1 onward as the max A_i
        for (int i = K - 1; i < N; ++i) {
            sum_B += items[i].second;
            max_heap.push(items[i].second);

            // Maintain only K smallest B's
            if (max_heap.size() > K) {
                sum_B -= max_heap.top();
                max_heap.pop();
            }

            result = min(result, items[i].first * sum_B);
        }

        cout << result << '\n';
    }

    return 0;
}
