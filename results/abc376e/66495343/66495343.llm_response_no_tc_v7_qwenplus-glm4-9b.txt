#include <bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int, int>
#define fi first
#define se second

const int N = 2e5 + 10;

int a[N], b[N];
vector<PII> arr;

int solve(const vector<PII> &A, int K) {
    sort(A.begin(), A.end());
    int res = LLONG_MAX;
    int prefixSum = 0;
    vector<int> prefixProduct;
    
    for (int i = 0; i < (int)A.size(); ++i) {
        prefixSum += A[i].se;
        prefixProduct.push_back(A[i].fi * prefixSum);
    }
    
    for (int i = 0; i <= (int)A.size() - K; ++i) {
        res = min(res, prefixProduct[i + K - 1]);
    }
    
    return res;
}

int main() {
    int T;
    cin >> T;
    
    while (T--) {
        int n, k;
        cin >> n >> k;
        arr.resize(n);
        
        for (int i = 0; i < n; ++i) {
            cin >> a[i] >> b[i];
            arr[i] = {a[i], b[i]};
        }
        
        reverse(arr.begin(), arr.end());
        cout << solve(arr, k) << '\n';
    }
    
    return 0;
}
