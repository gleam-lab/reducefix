```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int,int>
#define fi first
#define se second
#define arr array<int,3>

const int N=2e5+10,INF=0x3f3f3f3f,mod=1e9+7;

int a[N],b[N];
PII sa[N];

signed main()
{
	ios::sync_with_stdio(0);cin.tie(0),cout.tie(0);
	int t;
	cin>>t;

	while(t--)
	{
		int n,k;
		cin>>n>>k;

		for(int i=1;i<=n;i++)
			cin>>a[i];
		
		for(int i=1;i<=n;i++)
			cin>>b[i];
			
		for(int i=1;i<=n;i++)
			sa[i]={a[i],b[i]};
		
		sort(sa+1,sa+1+n);
		
		int res=INF,sum=0;
		priority_queue<int,vector<int>,greater<>> pq;
		for(int i=n;i>=1;i--)
		{
			if(n-i+1<=k)
			{
				sum+=sa[i].se;
				pq.push(sa[i].se);
				continue;
			}
			
			int na=sa[i].fi;
			sum+=sa[i].se;
			
			if(pq.size()>k-1)
			{
				int small=pq.top();
				pq.pop();
				sum-=small;
			}
			
			pq.push(sa[i].se);
			res=min(res,na*sum);
		}
		
		cout<<res<<endl;
	}

}
```

Explanation of changes:
1. Replaced `priority_queue<int>` with `priority_queue<int,vector<int>,greater<>>` to ensure that we always extract the smallest element when needed.
2. Changed the loop to iterate from `n` to `1` instead of `1` to `n`. This ensures that we consider the largest elements first, which helps in minimizing the product as required.
3. Fixed the logic inside the if condition where the sum of `b_i` values is being updated based on whether `(n-i+1)` is less than or equal to `k`.
4. Kept other parts of the code unchanged as they were already correct and efficient.
