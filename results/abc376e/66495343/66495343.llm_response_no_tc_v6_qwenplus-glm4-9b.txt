
```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int, int>
#define fi first
#define se second

const int N = 2e5 + 10;

int a[N], b[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        for (int i = 1; i <= n; i++) cin >> a[i];
        for (int i = 1; i <= n; i++) cin >> b[i];

        vector<PII> sa(n + 1);
        for (int i = 1; i <= n; i++) {
            sa[i] = {a[i], b[i]};
        }

        // Sort by A_i descending first, B_i ascending in case of tie
        sort(sa.begin() + 1, sa.begin() + 1 + n, [](const PII &a, const PII &b) {
            if (a.fi != b.fi) return a.fi > b.fi;
            return a.se < b.se;
        });

        int res = LLONG_MAX;
        long long sum = 0;
        vector<long long> pq;
        for (int i = 1; i <= n; i++) {
            sum += sa[i].se;
            pq.push_back(sa[i].se);

            if (pq.size() > k) {
                sum -= pq.front();
                pq.erase(pq.begin());
            }

            if (pq.size() == k) {
                res = min(res, (long long)sa[i].fi * sum);
            }
        }

        cout << res << endl;
    }

    return 0;
}
```

### Explanation
1. **Sorting**: The elements are sorted primarily by `A_i` in descending order and by `B_i` in ascending order when `A_i` values are equal. This ensures that when we process the smallest `k` elements by `A_i`, the sum of corresponding `B_i` values will be minimized.
2. **Priority Queue**: We maintain a priority queue (min-heap) to always remove the smallest `B_i` value when the size of the priority queue exceeds `k`. This ensures that the sum of `B_i` values for the smallest `k` elements by `A_i` is minimized.
3. **Result Calculation**: For each element, we calculate the product of the maximum `A_i` in the current subset and the sum of `B_i` values in the subset. We keep track of the minimum value of this product across all processed subsets.
