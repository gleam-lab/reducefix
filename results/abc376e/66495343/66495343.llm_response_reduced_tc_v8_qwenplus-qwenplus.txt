```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair<int,int>
#define fi first
#define se second

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    
    int t;
    cin >> t;
    
    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<PII> items(n);
        for (int i = 0; i < n; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < n; ++i) {
            cin >> items[i].second;
        }
        
        // Sort by A_i ascending
        sort(items.begin(), items.end());
        
        int res = INF;
        int sum_b = 0;
        // Max heap to store the smallest B_i values
        priority_queue<int, vector<int>, greater<int>> pq;
        
        // First collect K-1 elements
        for (int i = 0; i < k - 1; ++i) {
            sum_b += items[i].se;
            pq.push(items[i].se);
        }
        
        // Now add one element at a time as max A_i and maintain top K-1 smallest B_i
        for (int i = k - 1; i < n; ++i) {
            sum_b += items[i].se;
            pq.push(items[i].se);
            
            // If we have more than K-1 elements in our heap
            if (pq.size() > k) {
                sum_b -= pq.top();
                pq.pop();
            }
            
            res = min(res, items[i].fi * sum_b);
        }
        
        cout << res << '\n';
    }
    
    return 0;
}
```
