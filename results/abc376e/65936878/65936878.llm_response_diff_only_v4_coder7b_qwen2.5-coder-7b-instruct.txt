```cpp
#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int t; cin >> t;
    while(t--) {
        int n, k; cin >> n >> k;
        vector<int> A(n), B(n);
        for(int i=0;i<n;i++) cin >> A[i];
        for(int i=0;i<n;i++) cin >> B[i];

        // Sort indices based on A[i] in descending order
        vector<int> idx(n);
        iota(idx.begin(), idx.end(), 0);
        sort(idx.begin(), idx.end(), [&](int a, int b){ return A[a] > A[b]; });

        priority_queue<int> pq;
        int total_sum = 0;
        for(int i=0;i<k;i++) {
            pq.push(B[idx[i]]);
            total_sum += B[idx[i]];
        }

        int result = LLONG_MAX;
        for(int i=k;i<n;i++) {
            result = min(result, (total_sum + B[idx[i]]) * A[idx[i]]);
            total_sum -= pq.top();
            pq.pop();
            pq.push(B[idx[i]]);
            total_sum += B[idx[i]];
        }
        cout << result << '\n';
    }
    return 0;
}
```

This solution addresses the problem by sorting the elements based on their corresponding `A` values in descending order and using a max-heap to efficiently manage the sum of selected `B` values. This approach ensures that we always consider the maximum possible `A` value multiplied by the smallest possible sum of `B` values among the chosen elements, thus minimizing the overall expression.
