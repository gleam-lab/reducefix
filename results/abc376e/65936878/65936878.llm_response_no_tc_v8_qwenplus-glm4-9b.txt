#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int32_t main() {
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<int> A(n), B(n);
        for (int i = 0; i < n; ++i) cin >> A[i];
        for (int i = 0; i < n; ++i) cin >> B[i];

        // Pair A[i] with B[i] and sort by A[i] descending
        vector<pair<int, int>> v(n);
        for (int i = 0; i < n; ++i) v[i] = {A[i], B[i]};
        sort(v.rbegin(), v.rend()); // Sort in descending order for optimal selections

        long long sumB = 0, maxA = v[0].first;
        long long ans = 1e18;

        // Calculate the best possible value for the first k elements
        for (int i = 0; i < k; ++i) {
            sumB += v[i].second;
            ans = min(ans, maxA * sumB);
        }

        // Calculate the best possible value for the remaining elements
        for (int i = k; i < n; ++i) {
            sumB += v[i].second - v[i - k].second; // Add current element's B value and remove the smallest
            maxA = v[i].first;
            ans = min(ans, maxA * sumB);
        }

        cout << ans << '\n';
    }
    return 0;
}
