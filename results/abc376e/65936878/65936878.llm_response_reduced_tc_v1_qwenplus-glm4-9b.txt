#include <bits/stdc++.h>
using namespace std;

int32_t main() {
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> v(n);
        for (int i = 0; i < n; ++i) cin >> v[i].f >> v[i].s;
        sort(v.begin(), v.end());

        // Calculate the sum of B_i for the first k elements
        long long sum_first_k_b = 0;
        for (int i = 0; i < k; ++i) sum_first_k_b += v[i].s;

        // Initialize the answer with the first possible minimum value
        long long ans = (v[0].f) * (sum_first_k_b);

        // Use a max-heap to keep track of the maximum sum of B_i in the top k elements
        priority_queue<pair<int, int>> max_heap;
        for (int i = 0; i < n; ++i) max_heap.push({v[i].s, i});

        // Iterate over the remaining elements
        for (int i = k; i < n; ++i) {
            // Remove the element with the smallest sum of B_i from the heap
            pair<int, int> top = max_heap.top();
            max_heap.pop();

            // Remove the contribution of this element from the sum of B_i
            sum_first_k_b -= top.first;

            // Add the current element's contribution to the sum of B_i
            sum_first_k_b += v[i].s;

            // Calculate the current answer and update the minimum
            ans = min(ans, (v[top.second].f) * (sum_first_k_b));

            // Add the current element back to the heap
            max_heap.push({v[i].s, i});
        }

        cout << ans << '\n';
    }
    return 0;
}
