        }

        sort(b.begin(), b.end());

        priority_queue<int> pq;
        long long sum = 0, res = LLONG_MAX;

        for(int i = 0; i < k; i++) {
            pq.push(-b[i].f);
            sum += b[i].f;
        }

        for(int i = k; i < n; i++) {
            res = min(res, (long long)a[i].f * sum);
            sum -= b[pq.top()];
            pq.pop();
            pq.push(-b[i].f);
            sum += b[i].f;
        }

        res = min(res, (long long)a[n-1].f * sum);
        cout << res << '\n';
    }

    return 0;
}
