
The problem asks to find the minimum possible value of the expression \((\max_{i \in S} A_i) \times (\sum_{i \in S} B_i)\) for a subset \(S\) of \(\{1, 2, \dots, N\}\) of size \(K\). Given multiple test cases, we need to solve each one efficiently.

### Key Observations:
1. The expression is maximized when \(\max_{i \in S} A_i\) is as large as possible, and \(\sum_{i \in S} B_i\) is also as large as possible.
2. To minimize the expression, we want to minimize \(A_i\) while maximizing \(\sum_{i \in S} B_i\).
3. Sorting the arrays \(A\) and \(B\) in descending order can help in achieving this.

### Strategy:
1. Sort both arrays \(A\) and \(B\) in descending order.
2. Use a max-heap to keep track of the \(B_i\) values of the smallest \(N-K\) elements in \(A\).
3. Iterate through the sorted arrays and calculate the minimum possible value of the expression.

### Optimized Solution:
```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int MAXN = 2e5 + 5;
int T, N, K;
vector<int> A, B;

struct MinHeap {
    priority_queue<int, vector<int>, greater<int>> pq;
    int val[MAXN];

    void push(int x) {
        if (pq.size() < N - K) {
            pq.push(x);
            val[pq.size()] = x;
        } else if (x > pq.top()) {
            pq.pop();
            pq.push(x);
            val[pq.size()] = x;
        }
    }

    int getMax() {
        // Find the maximum value in the top N-K elements
        int ind = upper_bound(val + 1, val + N - K + 1, pq.top()) - val;
        return val[ind];
    }
};

int solve() {
    sort(A.begin(), A.end(), greater<int>());
    sort(B.begin(), B.end(), greater<int>());

    MinHeap heap;
    int ans = LLONG_MAX;

    for (int i = 0; i < N; ++i) {
        heap.push(B[i]);
        if (i >= K) {
            ans = min(ans, heap.getMax() * A[i]);
        }
    }

    return ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> T;
    while (T--) {
        cin >> N >> K;
        A.resize(N), B.resize(N);
        for (int i = 0; i < N; ++i) cin >> A[i];
        for (int i = 0; i < N; ++i) cin >> B[i];
        cout << solve() << '\n';
    }

    return 0;
}
```

This code sorts the arrays \(A\) and \(B\) in descending order, uses a max-heap to track the smallest \(N-K\) elements in \(B\), and iterates through the arrays to find the minimum possible value of the expression. The complexity is dominated by the sorting step, making it \(O(N \log N)\) per test case.
