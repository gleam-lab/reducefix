#include <bits/stdc++.h>
#define int long long
using namespace std;

int solve(int n, int k, vector<int>& A, vector<int>& B) {
    // Create a vector of pairs containing A[j] and B[j]
    vector<pair<int, int>> v;
    for (int j = 0; j < n; ++j) {
        v.emplace_back(A[j], B[j]);
    }

    // Sort the vector based on A[j] descending and use a max-heap for B[j] values
    sort(v.rbegin(), v.rend());
    priority_queue<int> pq;
    long long sum_B = 0;
    long long ans = LLONG_MAX;

    // Iterate over the first min(k-1, n) elements
    for (int i = 0; i < min(k - 1, n); ++i) {
        pq.push(v[i].second);
        sum_B += v[i].second;
    }

    // Iterate over the rest of the elements
    for (int i = min(k - 1, n); i < n; ++i) {
        // Calculate the current expression value
        ans = min(ans, (sum_B + pq.top()) * v[i].first);

        // Update the sum_B and max-heap
        sum_B += v[i].second - pq.top();
        pq.pop();
        pq.push(v[i].second);
    }

    // If k-1 is greater than n, we cannot take all elements from B
    if (k - 1 > n) {
        ans = min(ans, sum_B * v[0].first);
    }

    return ans;
}

signed main() {
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<int> A(n), B(n);
        for (int i = 0; i < n; ++i) cin >> A[i];
        for (int i = 0; i < n; ++i) cin >> B[i];
        cout << solve(n, k, A, B) << '\n';
    }
    return 0;
}
