#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> a(n);
        
        for (int i = 0; i < n; i++) {
            cin >> a[i].first >> a[i].second;
        }

        // Sort the pairs by A component descending, then B component descending
        sort(a.rbegin(), a.rend());

        long long sumB = 0, maxA = a[0].first;  // Start with the largest A from the sorted list
        long long currentSumB = a[0].second;  // Start with the first B value
        long long minResult = maxA * currentSumB;  // Initialize result with the first possible combination

        // Iterate over the rest of the elements, removing the smallest B value and adding the next largest B value
        for (int i = 1; i < n; i++) {
            sumB += a[i].second;
            currentSumB += a[i].second;

            // Calculate the result for the current combination
            long long currentResult = maxA * currentSumB;
            minResult = min(minResult, currentResult);

            // If we have added too many elements, remove the smallest B value
            if (i >= k - 1) {
                sumB -= a[i - k + 1].second;
                currentSumB -= a[i - k + 1].second;
            }
        }

        cout << minResult << '\n';
    }

    return 0;
}
