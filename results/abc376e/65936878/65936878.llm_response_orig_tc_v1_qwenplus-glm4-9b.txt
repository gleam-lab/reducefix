\[
\left(\max_{i \in S} A_i\right) \times \left(\sum_{i \in S} B_i\right)
\]

We can think of this as trying to balance between two things: the maximum value in A, and the sum of the corresponding B values. The goal is to minimize their product.

### Corrected Algorithm

1. Pair up A and B values into a list of pairs.
2. Sort the pairs based on the values of A in descending order.
3. Use a priority queue to keep track of the K largest B values corresponding to the K largest A values.
4. Iterate over all possible subsets of size K:
   - Calculate the product of the maximum A value in the subset and the sum of the corresponding B values in the subset.
   - Update the minimum value if the current product is smaller.

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;
        vector<pair<int, int>> combined(N);
        for (int i = 0; i < N; ++i) {
            cin >> combined[i].first >> combined[i].second;
        }

        // Sort by A descending
        sort(combined.begin(), combined.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return a.first > b.first;
        });

        priority_queue<int, vector<int>, greater<int>> pq;
        long long sum_B = 0;
        long long min_product = LLONG_MAX;

        for (int i = 0; i < N; ++i) {
            pq.push(combined[i].second);
            sum_B += combined[i].second;

            if (pq.size() > K) {
                sum_B -= pq.top();
                pq.pop();
            }

            if (pq.size() == K) {
                min_product = min(min_product, static_cast<long long>(combined[i].first) * sum_B);
            }
        }

        cout << min_product << '\n';
    }

    return 0;
}
