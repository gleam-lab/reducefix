#include <bits/stdc++.h>
#define ll long long
using namespace std;

struct Node{
    ll first, second;
    bool operator<(const Node& other) const{
        return first > other.first || (first == other.first && second < other.second);
    }
};

int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int t; cin >> t;
    while(t--){
        int n, k; cin >> n >> k;
        vector<Node> vec(n);
        for(int i=0;i<n;++i){
            cin >> vec[i].first;
        }
        for(int i=0;i<n;++i){
            cin >> vec[i].second;
        }

        sort(vec.begin(), vec.end());
        
        priority_queue<ll> pq;
        ll sum = 0;
        for(int i=0;i<k-1;++i){
            pq.push(vec[i].second);
            sum += vec[i].second;
        }
        
        ll ans = LLONG_MAX;
        for(int i=k-1;i<n;++i){
            ll cur = sum + vec[i].second;
            ans = min(ans, cur * vec[i].first);
            
            if(!pq.empty()){
                ll top = pq.top();
                if(top > vec[i].second){
                    sum -= top;
                    sum += vec[i].second;
                    pq.pop();
                    pq.push(vec[i].second);
                }
            } else {
                sum += vec[i].second;
            }
        }
        cout << ans << '\n';
    }
    return 0;
}
