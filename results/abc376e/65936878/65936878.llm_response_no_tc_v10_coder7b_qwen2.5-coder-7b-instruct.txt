#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t; cin >> t;
    while(t--) {
        int n, k; cin >> n >> k;
        vector<int> a(n), b(n);
        for(int i=0;i<n;i++) cin >> a[i];
        for(int i=0;i<n;i++) cin >> b[i];

        // Create pairs of (a_i, b_i)
        vector<pair<int,int>> v(n);
        for(int i=0;i<n;i++) v[i] = {a[i], b[i]};

        // Sort the pairs based on a_i in descending order
        sort(v.begin(), v.end(), greater<pair<int,int>>());

        // Priority queue to keep track of the smallest b_i's
        priority_queue<int, vector<int>, greater<int>> pq;
        int sum_b = 0;
        for(int i=0;i<k;i++) {
            pq.push(v[i].s);
            sum_b += v[i].s;
        }

        // Initialize the answer with the current product
        int ans = v[0].f * sum_b;

        // Iterate through the remaining elements to find the minimum product
        for(int i=k;i<n;i++) {
            // Remove the smallest element from the sum and the priority queue
            sum_b -= pq.top();
            pq.pop();

            // Add the new element to the sum and the priority queue
            sum_b += v[i].s;
            pq.push(v[i].s);

            // Update the answer with the new product if it's smaller
            ans = min(ans, v[i].f * sum_b);
        }

        cout << ans << '\n';
    }
    return 0;
}
