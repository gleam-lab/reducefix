#include <bits/stdc++.h>

using namespace std;

struct Pair{
    int a, b;
};

bool comp(Pair &a, Pair &b){
    return (double)a.a / a.b < (double)b.a / b.b;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;

    while(t--){
        int n, k;
        cin >> n >> k;
        vector<Pair> v(n);
        
        for(int i = 0; i < n; i++){
            cin >> v[i].a >> v[i].b;
        }

        sort(v.begin(), v.end(), comp);

        priority_queue<int> pq;
        long long int sum = 0, res = INT_MAX;

        for(int i = 0; i < k; i++){
            pq.push(v[i].b);
            sum += v[i].b;
        }

        res = min(res, (long long)v[k-1].a * sum);

        for(int i = k; i < n; i++){
            if(pq.top() > v[i].b){
                sum -= pq.top();
                sum += v[i].b;
                pq.pop();
                pq.push(v[i].b);
            }
            res = min(res, (long long)v[i].a * sum);
        }

        cout << res << "\n";
    }
    
    return 0;
}
