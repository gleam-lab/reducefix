```cpp
#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> items(n);
        for (int i = 0; i < n; ++i) {
            cin >> items[i].f; // A_i
        }
        for (int i = 0; i < n; ++i) {
            cin >> items[i].s; // B_i
        }

        // Sort by A_i ascending because we want to control max(A_i) in a smart way
        sort(items.begin(), items.end());

        // Use min-heap to maintain the top K smallest B_i's from current prefix
        priority_queue<int, vector<int>, greater<>> pq;
        int sum_b = 0;

        // Initialize with first K elements
        for (int i = 0; i < k; ++i) {
            sum_b += items[i].s;
            pq.push(items[i].s);
        }

        // The minimum value is initially (max A in subset) * (sum of B)
        int ans = items[k - 1].f * sum_b;

        // Slide window over rest of the array
        for (int i = k; i < n; ++i) {
            // Always include current item since its A_i is larger (due to sorting),
            // and we're trying to minimize product. So if we include it, we must remove one from the set.
            int current_a = items[i].f;
            int current_b = items[i].s;

            // Try replacing the smallest B in our current set with this new B
            if (current_b > pq.top()) {
                sum_b += current_b - pq.top();
                pq.pop();
                pq.push(current_b);
            }

            // Update answer with current best set (which has current_a as max A_i)
            ans = min(ans, current_a * sum_b);
        }

        cout << ans << '\n';
    }

    return 0;
}
```
