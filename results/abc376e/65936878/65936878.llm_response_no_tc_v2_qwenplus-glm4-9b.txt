#include <bits/stdc++.h>
#define int long long
using namespace std;

int32_t main() {
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> v(n);
        for (int i = 0; i < n; i++) cin >> v[i].f;
        for (int i = 0; i < n; i++) cin >> v[i].s;
        
        // Sort by A[i] in descending order to prioritize larger A[i]
        sort(v.begin(), v.end(), greater<pair<int, int>>());
        
        // priority_queue to keep track of the largest K-1 B[i] values
        priority_queue<int, vector<int>, greater<int>> pq;
        long long sum_B = 0;
        
        // Calculate the sum of the smallest (N-K) B[i] values
        for (int i = n - k; i < n; i++) {
            sum_B += v[i].s;
            pq.push(v[i].s);
        }
        
        // Calculate the maximum value of (sum_B - B[i]) * A[i] for each i in the range [0, n-k)
        long long ans = 0;
        for (int i = 0; i < n - k; i++) {
            sum_B -= pq.top(); pq.pop();
            ans = max(ans, sum_B * v[i].f);
            pq.push(v[i].s);
        }
        
        cout << ans << '\n';
    }
    return 0;
}
