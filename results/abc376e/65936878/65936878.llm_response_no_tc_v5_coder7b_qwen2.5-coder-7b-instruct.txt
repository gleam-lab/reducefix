```cpp
#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while(t--) {
        int n, k; cin >> n >> k;
        vector<int> A(n), B(n);
        for(int i=0;i<n;i++) cin >> A[i];
        for(int i=0;i<n;i++) cin >> B[i];

        sort(A.begin(), A.end());
        sort(B.begin(), B.end());

        int ans = LLONG_MAX;
        for(int i=0;i<=n-k;i++) {
            int max_A = A[i+k-1];
            int sum_B = accumulate(B.begin()+i, B.begin()+i+k, 0LL);
            ans = min(ans, max_A * sum_B);
        }

        cout << ans << '\n';
    }
    return 0;
}
```
