```cpp
#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int t; cin >> t;
    while(t--) {
        int n, k; cin >> n >> k;
        vector<pair<int,int>> v(n);
        for(int i=0; i<n; ++i) cin >> v[i].f >> v[i].s;

        sort(v.begin(), v.end());

        priority_queue<int> pq;
        int sum = 0;
        for(int i=0; i<k-1; ++i) {
            pq.push(-v[i].s);
            sum += v[i].s;
        }

        int res = LLONG_MAX;
        for(int i=k-1; i<n; ++i) {
            res = min(res, (-pq.top() * v[i].f + sum + v[i].s));
            if(pq.top() > -v[i].s) {
                sum -= pq.top();
                pq.pop();
                pq.push(-v[i].s);
                sum += v[i].s;
            }
        }

        cout << res << '\n';
    }

    return 0;
}
```
