#include <bits/stdc++.h>

        int n, k; cin >> n >> k;

        for(int i=0;i<n;i++) cin >> v[i].f;

            }
            cout << min_val << '\n';
            continue;
        }
        
        priority_queue<int> q;
        int sum = 0;
        for(int i=0;i<k-1;i++) {
            q.push(v[i].s);
            sum += v[i].s;
        }
        
        int ans = LLONG_MAX;
        for(int i=k-1;i<n;i++) {
            ans = min(ans, (sum + v[i].s) * v[i].f);
            if(!q.empty() && q.top() > v[i].s) {
                sum -= q.top();
                q.pop();
                sum += v[i].s;
                q.push(v[i].s);
            }
        }
        cout << ans << '\n';
    }
}
