```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> items(n);
        for (int i = 0; i < n; ++i) cin >> items[i].first;
        for (int i = 0; i < n; ++i) cin >> items[i].second;
        
        // Sort by A_i ascending
        sort(items.begin(), items.end());
        
        // Max heap to maintain the K smallest B_i values
        priority_queue<int> max_heap;
        int sum_b = 0;
        
        // Initialize with first K-1 elements
        for (int i = 0; i < k - 1; ++i) {
            sum_b += items[i].second;
            max_heap.push(items[i].second);
        }
        
        int ans = LLONG_MAX;
        
        // Try each possible maximum A_i from position k-1 onward
        for (int i = k - 1; i < n; ++i) {
            // Current item's A_i and B_i
            int a = items[i].first;
            int b = items[i].second;
            
            // Create a temporary copy of the heap to avoid modifying the original
            priority_queue<int> temp = max_heap;
            int temp_sum = sum_b;
            
            // If we have less than K-1 elements, just add this element
             if (temp.size() < k - 1) {
                temp.push(b);
                temp_sum += b;
            } else {
                // Otherwise, consider adding current b and removing the largest in heap
                if (temp.empty() || b < temp.top()) {
                    temp_sum += b;
                    if (!temp.empty()) {
                        temp_sum -= temp.top();
                        temp.pop();
                    }
                    temp.push(b);
                }
            }
            
            // Now we have K elements in the heap
            ans = min(ans, a * (temp_sum + b));
        }
        
        cout << ans << '\n';
    }
    
    return 0;
}
```
