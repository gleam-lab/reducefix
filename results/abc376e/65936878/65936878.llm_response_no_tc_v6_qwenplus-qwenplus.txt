#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> items(n);
        for (int i = 0; i < n; ++i) cin >> items[i].f;
        for (int i = 0; i < n; ++i) cin >> items[i].s;

        // Sort items by A_i (first element of pair)
        sort(items.begin(), items.end());

        // Use a max heap to maintain the K smallest B_i values among first i items
        priority_queue<int> pq;
        int sum_b = 0;
        for (int i = 0; i < k - 1; ++i) {
            sum_b += items[i].s;
            pq.push(items[i].s);
        }

        int ans = LLONG_MAX;
        for (int i = k - 1; i < n; ++i) {
            // Try selecting current item as the max A in the subset
            ans = min(ans, items[i].f * (sum_b + items[i].s));
            
            // Add current B to heap if needed
            if (pq.size() < k) {
                pq.push(items[i].s);
                sum_b += items[i].s;
            } else if (!pq.empty() && pq.top() > items[i].s) {
                // Replace the largest B in heap with smaller one
                sum_b += items[i].s - pq.top();
                pq.pop();
                pq.push(items[i].s);
            }
        }
        cout << ans << '\n';
    }

    return 0;
}
