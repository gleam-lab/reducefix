#include <bits/stdc++.h>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<int> A(n), B(n);
        for (int i = 0; i < n; ++i) {
            cin >> A[i];
        }
        for (int i = 0; i < n; ++i) {
            cin >> B[i];
        }

        sort(A.begin(), A.end());
        sort(B.rbegin(), B.rend());

        int result = INT_MAX;
        for (int i = 0; i <= n - k; ++i) {
            int current_result = A[n-k+i] * accumulate(B.begin()+i, B.begin()+k+i, 0LL);
            result = min(result, current_result);
        }

        cout << result << endl;
    }

    return 0;
}
