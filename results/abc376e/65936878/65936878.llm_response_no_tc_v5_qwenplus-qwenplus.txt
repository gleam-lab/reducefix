```cpp
#include <bits/stdc++.h>
#define int long long
#define f first
#define s second
using namespace std;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t; cin >> t;
    while (t--) {
        int n, k; cin >> n >> k;
        vector<pair<int, int>> items(n);
        for (int i = 0; i < n; ++i) cin >> items[i].f;
        for (int i = 0; i < n; ++i) cin >> items[i].s;

        // Sort items by A_i (first element of pair)
        sort(items.begin(), items.end());

        // Use a max-heap to maintain the K smallest B_i values
        priority_queue<int> pq;
        int sum_b = 0;

        // Initialize with first K-1 elements
        for (int i = 0; i < k - 1; ++i) {
            sum_b += items[i].s;
            pq.push(items[i].s);
        }

        int ans = LLONG_MAX;

        // Try including each item from position k-1 onward as the max A_i
        for (int i = k - 1; i < n; ++i) {
            // Current item will be the maximum A_i in the subset
            int current_a = items[i].f;
            
            // Add current B_i to sum and to queue
            sum_b += items[i].s;
            pq.push(items[i].s);

            // Maintain only K smallest B_i's
            if (pq.size() > k) {
                sum_b -= pq.top();
                pq.pop();
            }

            // Calculate the current cost
            ans = min(ans, current_a * sum_b);
        }

        cout << ans << '\n';
    }

    return 0;
}
```
