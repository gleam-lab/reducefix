#include <bits/stdc++.h>
#define int long long
using namespace std;

int32_t main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i].f;
        }
        for (int i = 0; i < n; ++i) {
            cin >> a[i].s;
        }
        // Sort by the ratio of B_i / A_i because we want to maximize the sum of B_i while minimizing A_i
        sort(a.begin(), a.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return (double)a.s / a.f < (double)b.s / b.f;
        });

        // Minimize the maximum A_i in the subset while maximizing the sum of B_i
        int current_sum = 0;
        int max_A = a[0].f;
        for (int i = 1; i < n; ++i) {
            current_sum += a[i].s;
            if (i - 1 == k - 1) {
                current_sum -= a[i].s; // We do not include the last element in the sum for the next iteration
                max_A = a[i].f;
            }
        }

        cout << current_sum * max_A << '\n';
    }
    return 0;
}
