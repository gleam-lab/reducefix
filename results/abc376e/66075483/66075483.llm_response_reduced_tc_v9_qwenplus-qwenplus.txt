```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

#define int long long

struct Item {
    int a, b;
};

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<Item> items(N);
        for (int i = 0; i < N; ++i) {
            cin >> items[i].a;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].b;
        }

        // Sort items by A value
        sort(items.begin(), items.end(), [](const Item& x, const Item& y) {
            return x.a < y.a;
        });

        // Min-heap to maintain top K B values
        priority_queue<int, vector<int>, greater<int>> pq;
        int sum_b = 0;
        int result = 1e18;  // Large enough initial value

        // First K-1 elements in B
        for (int i = 0; i < K - 1; ++i) {
            sum_b += items[i].b;
            pq.push(items[i].b);
        }

        // Try every possibility with one of the remaining as max(A)
        for (int i = K - 1; i < N; ++i) {
            // Add current B to heap and sum
            pq.push(items[i].b);
            sum_b += items[i].b;

            // Maintain only K elements in heap
            if (pq.size() > K) {
                sum_b -= pq.top();
                pq.pop();
            }

            // Current maximum A is items[i].a
            result = min(result, items[i].a * sum_b);
        }

        cout << result << "\n";
    }

    return 0;
}
```
