```cpp
#include<iostream>
#include<algorithm>
using namespace std;

#define int long long
const int inf = 0x3f3f3f3f3f3f3f3f;

struct Node{
    int val,idx;
};

bool cmp(Node &a,Node &b){
    return a.val<b.val;
}

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0),cout.tie(0);

    int t,n,m,k;
    cin>>t;
    while(t--){
        cin>>n>>k;
        vector<Node> a(n+1),b(n+1);
        for(int i=1;i<=n;++i){
            cin>>a[i].val;a[i].idx=i;
        }
        for(int i=1;i<=n;++i){
            cin>>b[i].val;b[i].idx=i;
        }
        
        sort(a.begin()+1,a.end(),cmp);
        sort(b.begin()+1,b.end(),cmp);

        int ans=inf,sum=0;
        for(int i=1,j=n-k+1;i<=n;++i){
            sum+=b[j].val;
            if(i>=k){
                ans=min(ans,(long long)a[i].val*sum);
                sum-=b[j].val;
                ++j;
            }
        }
        cout<<ans<<'\n';
    }
    return 0;
}
```
