        prefix_sum_b[0] = b[0].first;
        for (int i = 1; i < n; ++i) {
            prefix_sum_b[i] = prefix_sum_b[i - 1] + b[i].first;
        }

        // Calculate the minimum value for the k elements in B
        for (int i = 0; i <= k - 1; ++i) {
            sum_b += b[i].first;
        }

        // Iterate over the remaining elements in A to find the minimum value
        for (int i = k; i < n; ++i) {

                long long next_sum_b = prefix_sum_b[n - 1] - (prefix_sum_b[i] - sum_b);
                min_value = min(min_value, max_a * next_sum_b);
