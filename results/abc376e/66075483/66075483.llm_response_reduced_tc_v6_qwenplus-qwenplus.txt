```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
        }

        vector<int> B(N);
        for (int i = 0; i < N; ++i) {
            cin >> B[i];
            A[i].second = B[i]; // Store B[i] directly in the pair
        }

        // Sort by A[i], ascending
        sort(A.begin(), A.end());

        // Use min-heap to keep track of top K smallest B values
        priority_queue<int, vector<int>, greater<>> pq;
        int sum = 0;
        int res = 1e18; // Initialize with large value

        // We'll process from i = K-1 to N-1
        // For each A[i], consider it as max(A_i), and find best K-1 elements from previous B[j]
        for (int i = 0; i < N; ++i) {
            if (i >= K - 1) {
                // We can form a subset of size K with current A[i] as max
                res = min(res, (sum + B[A[i].second]) * static_cast<int64_t>(A[i].first));
            }

            // Add current B to heap and sum
            if (i < N - 1) {
                pq.push(B[A[i].second]);
                sum += B[A[i].second];

                // If we have more than K-1 elements, remove the smallest one
                if (pq.size() > K - 1) {
                    sum -= pq.top();
                    pq.pop();
                }
            }
        }

        cout << res << "\n";
    }

    return 0;
}
```
