#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

#define int long long

const int INF = 1e18;

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    
    int T;
    cin >> T;
    
    while (T--) {
        int N, K;
        cin >> N >> K;
        
        vector<pair<int, int>> A(N), B(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
            A[i].second = i;
        }
        for (int i = 0; i < N; ++i) {
            cin >> B[i].first;
        }

        // Sort elements by A value
        sort(A.begin(), A.end());
        vector<int> sorted_B(N);
        for (int i = 0; i < N; ++i) {
            sorted_B[i] = B[A[i].second].first;
        }

        // Build prefix sums of smallest K-1 elements from the right
        priority_queue<int> max_heap;
        vector<int> suffix_sum(N + 1, 0);
        int sum = 0;
        
        for (int i = N - 1; i >= 0; --i) {
            max_heap.push(sorted_B[i]);
            sum += sorted_B[i];
            
            if (max_heap.size() > K - 1) {
                sum -= max_heap.top();
                max_heap.pop();
            }
            
            suffix_sum[i] = sum;
        }

        // Try all positions where we pick A[i] as max A and K-1 smallest B's to its left
        int result = INF;
        priority_queue<int, vector<int>, greater<>> min_heap;
        sum = 0;
        
        for (int i = 0; i < N; ++i) {
            // We consider A[i] as the maximum A in the subset
            // Take K-1 smallest B values from left (in the heap) + current B
            if (min_heap.size() == K - 1) {
                result = min(result, A[i].first * (sum + sorted_B[i]));
            }
            
            // Add current B to heap for future use
            min_heap.push(sorted_B[i]);
            sum += sorted_B[i];
            
            // Maintain heap size <= K-1
            if (min_heap.size() >= K) {
                sum -= min_heap.top();
                min_heap.pop();
            }
        }

        cout << result << "\n";
    }

    return 0;
}
