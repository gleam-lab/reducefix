```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define int long long
const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;

signed main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>>a(n), b(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; i++) {
            cin >> b[i].first;
            b[i].second = i;
        }
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());

        int ans = inf, s = 0;
        priority_queue<int> q;
        for (int i = 0; i < n - k + 1; i++) {
            q.push(b[i].first);
            s += b[i].first;
            if (i >= k - 1) {
                ans = min(ans, (long long)s * a[i + k - 1].first);
                s -= q.top();
                q.pop();
            }
        }

        cout << ans << endl;
    }
    return 0;
}
```
