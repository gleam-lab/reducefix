#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long
const int N = 2e5 + 9;

int n, k, ans, L;
vector<pair<int, int>> a, b;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> k;
    a.resize(n);
    b.resize(n);
    
    for (int i = 0; i < n; i++) {
        cin >> a[i].first;
        a[i].second = i;
    }
    
    for (int i = 0; i < n; i++) {
        cin >> b[i].first;
        b[i].second = i;
    }

    sort(a.begin(), a.end());
    sort(b.begin(), b.end(), [](const pair<int, int>& l, const pair<int, int>& r) {
        return l.first < r.first;
    });
    
    // We use a prefix sum to accumulate the B values
    vector<int> prefixSum(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        prefixSum[i] = prefixSum[i - 1] + b[i - 1].first;
    }

    // We iterate over the A values, considering only the top k for each
    int j = 0;
    for (int i = 0; i < n; i++) {
        while (j < k && b[j].second < a[i].second) {
            j++;
        }
        if (j == 0) {
            L = prefixSum[k - 1] * a[i].first;
        } else {
            L = prefixSum[j] * a[i].first;
        }
        ans = min(ans, L);
    }

    cout << ans << endl;

    return 0;
}
