        sort(a, a + n);
        for (int i = 0; i < n; ++i) {
            b[i] = b[a[i].second];
        }
        priority_queue<int, vector<int>, greater<int>> min_heap;
        int sum = 0;
        int ans = inf;
        for (int i = 0; i < n; ++i) {
            if (min_heap.size() < k - 1) {
                min_heap.push(b[i]);
                sum += b[i];
            } else {
                if (k == 1) {
                    ans = min(ans, a[i].first * b[i]);
                } else {
