#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define int long long

const int N = 2e5 + 9, inf = 0x3f3f3f3f3f3f3f3f;

struct Node {
    int val, idx;
    bool operator<(const Node & other) const { return val > other.val; }
};

signed main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<Node> a(n), b(n);
        for(int i = 0; i < n; ++i) {
            cin >> a[i].val;
            a[i].idx = i;
        }
        for(int i = 0; i < n; ++i) {
            cin >> b[i].val;
            b[i].idx = i;
        }
        
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        
        vector<pair<int, int>> sorted_b(n);
        for(int i = 0; i < n; ++i) {
            sorted_b[b[i].idx] = make_pair(i, b[i].val);
        }
        
        int ans = inf, sum = 0;
        priority_queue<int> pq;
        
        for(int i = 0; i < n; ++i) {
            pq.push(sorted_b[i].second);
            sum += sorted_b[i].second;
            
            if(pq.size() > k) {
                sum -= pq.top();
                pq.pop();
            }
            
            if(pq.size() == k) {
                ans = min(ans, (long long)a[sorted_b[i].first].val * sum);
            }
        }
        
        cout << ans << endl;
    }
    return 0;
}
