```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

const int INF = 1e18;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<pair<int, int>> items(N); // A_i -> B_i
        for (int i = 0; i < N; ++i) {
            cin >> items[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> items[i].second;
        }

        // Sort items by A_i in ascending order
        sort(items.begin(), items.end());

        // Min-heap to keep track of top K smallest B values
        priority_queue<int, vector<int>, greater<int>> heap;
        int sum_b = 0;
        int res = INF;

        // Initialize with the last K-1 elements (largest A's)
        for (int i = N - 1; i >= max(0LL, N - K); --i) {
            heap.push(items[i].second);
            sum_b += items[i].second;
        }

        // Try each i as the max A_i (items[i].first)
        for (int i = N - K - 1; i >= 0; --i) {
            // Maintain a window of K elements including current i
            heap.push(items[i].second);
            sum_b += items[i].second;

            // If heap size > K, remove the smallest B
            if (heap.size() > K) {
                sum_b -= heap.top();
                heap.pop();
            }

            // Evaluate cost with current max A_i
            res = min(res, items[i].first * sum_b);
        }

        // Also try all combinations where max A is among the first K elements
        // This handles edge cases when all K best are at the beginning
        priority_queue<int> temp_heap;
        sum_b = 0;
        for (int i = 0; i < K; ++i) {
            temp_heap.push(items[i].second);
            sum_b += items[i].second;
        }
        res = min(res, items[K - 1].first * sum_b);

        for (int i = K; i < N; ++i) {
            // Remove the largest B currently in the set
            if (!temp_heap.empty()) {
                sum_b -= temp_heap.top();
                temp_heap.pop();
            }

            // Add new smaller or equal B
            temp_heap.push(items[i].second);
            sum_b += items[i].second;

            // Now the maximum A in the current set is items[i].first
            res = min(res, items[i].first * sum_b);
        }

        cout << res << "\n";
    }

    return 0;
}
```
