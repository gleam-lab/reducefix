#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (a); i < (b); i++)
#define rrep(i, a, b) for(ll i = (a); i >= (b); i--)
#define all(x) (x).begin(), (x).end()

inline void scan() {}
template<class Head, class... Tail>
inline void scan(Head& head, Tail&... tail) {
    cin >> head;
    scan(tail...);
}

void solve() {
    ll n, k;
    scan(n, k);
    vector<pll> items(n);
    for (auto &x : items) scan(x.first);
    for (auto &x : items) scan(x.second);

    // Sort by A_i ascending
    sort(all(items));

    // Min-heap to maintain top K B_i values
    multiset<ll, greater<ll>> top_k;
    ll sum_b = 0;

    // Initialize with first K elements
    for (ll i = 0; i < k; ++i) {
        top_k.insert(items[i].second);
        sum_b += items[i].second;
    }

    ll ans = sum_b * items[k - 1].first;

    // Sliding window
    for (ll i = k; i < n; ++i) {
        // Remove the largest B in current set
        sum_b -= *top_k.begin();
        top_k.erase(top_k.begin());

        // Add new B value
        top_k.insert(items[i].second);
        sum_b += items[i].second;

        // Update answer with current max A_i (items[i].first)
        ans = min(ans, sum_b * items[i].first);
    }

    cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    ll t;
    scan(t);
    while (t--) solve();
    
    return 0;
}
