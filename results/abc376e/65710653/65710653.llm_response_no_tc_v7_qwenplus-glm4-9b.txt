#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, a, b) for(int i = (a); i < (b); ++i)
#define rrep(i, a, b) for(int i = (a); i >= (b); --i)

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<ll, ll>> a(n);
        rep(i, 0, n) {
            cin >> a[i].first;
            a[i].second = i;
        }
        vector<ll> b(n);
        rep(i, 0, n) {
            cin >> b[i];
        }

        sort(a.begin(), a.end(), [](const pair<ll, ll>& p1, const pair<ll, ll>& p2) {
            return p1.first > p2.first;
        });

        multiset<ll, greater<ll>> smallest_b;
        ll sum_b = 0;
        for (int i = 0; i < k; ++i) {
            smallest_b.insert(b[a[i].second]);
            sum_b += b[a[i].second];
        }

        ll ans = sum_b * a[k - 1].first;
        for (int i = k; i < n; ++i) {
            sum_b -= *smallest_b.rbegin();
            smallest_b.erase(smallest_b.find(*smallest_b.rbegin()));
            sum_b += b[a[i].second];
            ans = min(ans, sum_b * a[i].first);
        }

        cout << ans << '\n';
    }

    return 0;
}
