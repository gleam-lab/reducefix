3. **Track Minimum Value**: Keep track of the minimum value of the product obtained from the above step.

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

    cin.tie(nullptr);

    int t;
    cin >> t;
    
    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<int> a(n), b(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
        }
        for (int i = 0; i < n; ++i) {
            cin >> b[i];
        }
        
        // Sort arrays
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        
        // Calculate initial sum of B for the first window
        ll current_sum = accumulate(b.begin(), b.begin() + k, 0LL);
        ll result = a[k - 1] * current_sum;
        
        // Sliding window to find the minimum value
        for (int i = k; i < n; ++i) {
            current_sum += b[i] - b[i - k];
            result = min(result, static_cast<ll>(a[i]) * current_sum);
        }
        
        cout << result << '\n';
    }

    return 0;
}

2. **Initial Sum Calculation**: The initial sum of the first `k` elements of `b` is calculated.
3. **Sliding Window**: The window slides through the array `b`, updating the sum by adding the next element and subtracting the element that falls out of the window. The product of the current maximum element in `a` and the updated sum is compared with the current result and updated if necessary.
4. **Output**: Finally, the minimum product found is printed.

This approach ensures that you efficiently find the minimum value using a sliding window technique, achieving a time complexity of \(O(N \log N)\) due to sorting and \(O(N)\) for the sliding window, making it suitable for large inputs within the given constraints.
