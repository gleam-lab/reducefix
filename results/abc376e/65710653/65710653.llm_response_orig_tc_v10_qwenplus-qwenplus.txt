#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, a, b) for (ll i = (a); i < (b); ++i)
#define per(i, a, b) for (ll i = (b)-1; i >= (a); --i)

struct Item {
    ll a, b;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    ll T;
    cin >> T;

    while (T--) {
        ll N, K;
        cin >> N >> K;

        vector<Item> items(N);
        for (ll i = 0; i < N; ++i) cin >> items[i].a;
        for (ll i = 0; i < N; ++i) cin >> items[i].b;

        // Sort by A in ascending order
        sort(items.begin(), items.end(), [](const Item& x, const Item& y) {
            return x.a < y.a;
        });

        multiset<ll> activeBs;
        ll sumB = 0;
        // Take first K elements' B values
        for (ll i = 0; i < K; ++i) {
            activeBs.insert(items[i].b);
            sumB += items[i].b;
        }

        // Initial cost
        ll res = sumB * items[K - 1].a;

        // Try replacing the max B in current set with smaller Bs from remaining items
        for (ll i = K; i < N; ++i) {
            if (items[i].b > *activeBs.begin()) {
                sumB -= *activeBs.begin();
                activeBs.erase(activeBs.begin());
                activeBs.insert(items[i].b);
                sumB += items[i].b;
            }
            res = min(res, sumB * items[i].a);
        }

        cout << res << '\n';
    }

    return 0;
}
