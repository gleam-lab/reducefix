#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#define rep(i, a, b) for(ll i = (a); i < (b); ++i)
#define rrep(i, a, b) for(ll i = (a); i >= (b); --i)
#define all(v) (v).begin(),(v).end()
#define each(x,v) for(auto& x : v)

template<typename T> bool chmax(T &a, const T &b) { if (a < b) return a = b, true; return false; }
template<typename T> bool chmin(T &a, const T &b) { if (a > b) return a = b, true; return false; }

void solve() {
    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;
        vector<int> A(N), B(N);
        for(int i = 0; i < N; ++i) cin >> A[i];
        for(int i = 0; i < N; ++i) cin >> B[i];

        vector<pair<int, int>> AB(N);
        for(int i = 0; i < N; ++i) AB[i] = {A[i], B[i]};
        sort(all(AB));

        multiset<int> ms;
        ll sum_b = 0;
        for(int i = 0; i < K; ++i) {
            ms.insert(AB[i].second);
            sum_b += AB[i].second;
        }

        ll res = 1LL * AB[K - 1].first * sum_b;

        for(int i = K; i < N; ++i) {
            // remove max from set
            sum_b -= *ms.rbegin();
            ms.erase(prev(ms.end()));

            // add new element
            ms.insert(AB[i].second);
            sum_b += AB[i].second;

            // update answer
            chmin(res, 1LL * AB[i].first * sum_b);
        }

        cout << res << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
