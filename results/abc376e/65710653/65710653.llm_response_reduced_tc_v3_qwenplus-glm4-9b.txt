#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

int main() {
    int T;
    cin >> T;
    
    while (T--) {
        long long N, K;
        cin >> N >> K;
        
        vector<long long> A(N), B(N);
        for (long long &value : A) cin >> value;
        for (long long &value : B) cin >> value;
        
        auto comp = [&](const pair<long long, long long>& a, const pair<long long, long long>& b) {
            return a.first > b.first || (a.first == b.first && a.second < b.second);
        };
        
        vector<pair<long long, long long>> C(N);
        for (long long i = 0; i < N; ++i) {
            C[i] = {A[i], B[i]};
        }
        
        sort(C.begin(), C.end(), comp);
        
        long long maxA = C[K-1].first;
        long long sumB = 0;
        for (long long i = 0; i < K; ++i) {
            sumB += C[i].second;
        }
        
        long long result = maxA * sumB;
        
        for (long long i = K; i < N; ++i) {
            sumB += C[i].second - C[K-1].second;
            result = min(result, maxA * sumB);
            maxA = max(maxA, C[i].first);
        }
        
        cout << result << endl;
    }
    
    return 0;
}
