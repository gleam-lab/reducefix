#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (a); i < (b); i++)
#define rrep(i, a, b) for(ll i = (a); i >= (b); i--)
#define all(x) (x).begin(), (x).end()

void solve() {
    ll N, K;
    cin >> N >> K;
    vector<pll> items(N);
    for (ll i = 0; i < N; ++i) cin >> items[i].first;
    for (ll i = 0; i < N; ++i) cin >> items[i].second;

    // Sort by A_i ascending
    sort(all(items));

    multiset<ll> top_k_b;
    ll sum_b = 0;

    // Initialize with first K elements
    rep(i, 0, K) {
        sum_b += items[i].second;
        top_k_b.insert(items[i].second);
    }

    ll min_result = sum_b * items[K-1].first;

    // Try every possible window of K elements
    rep(i, K, N) {
        // Remove the largest B in current set to make space
        sum_b -= *top_k_b.rbegin();
        top_k_b.erase(top_k_b.find(*top_k_b.rbegin()));
        
        // Add new element
        sum_b += items[i].second;
        top_k_b.insert(items[i].second);

        // Calculate current value and update minimum
        min_result = min(min_result, sum_b * items[i].first);
    }

    cout << min_result << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    ll T;
    cin >> T;
    while (T--) {
        solve();
    }
    
    return 0;
}
