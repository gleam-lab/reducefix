#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<long long, long long>;

#define rep(i, a, b) for(ll i = (a); i < (b); i++)
#define rrep(i, a, b) for(ll i = (a); i >= (b); i--)
#define all(x) begin(x), end(x)

void solve() {
    ll n, k;
    cin >> n >> k;
    vector<pll> pairs(n);
    for (ll i = 0; i < n; ++i) cin >> pairs[i].first;
    for (ll i = 0; i < n; ++i) cin >> pairs[i].second;

    // Sort by A_i in ascending order
    sort(all(pairs));

    // Min-heap to keep track of smallest K B_i values
    multiset<ll, greater<ll>> top_k;  // Stores top K B values in descending order (so we can remove max)
    ll sum_b = 0;

    // Initialize with first K elements
    rep(i, 0, k) {
        top_k.insert(pairs[i].second);
        sum_b += pairs[i].second;
    }

    ll ans = sum_b * pairs[k - 1].first;

    // Sliding window
    rep(i, k, n) {
        // If current B is smaller than max in set, replace it
        if (!top_k.empty() && pairs[i].second < *top_k.begin()) {
            sum_b -= *top_k.begin();
            top_k.erase(top_k.begin());
            top_k.insert(pairs[i].second);
            sum_b += pairs[i].second;
        }
        ans = min(ans, sum_b * pairs[i].first);
    }

    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    ll t;
    cin >> t;
    while (t--) {
        solve();
    }
}
