#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (n); ++i)
#define all(x) begin(x), end(x)

typedef long long ll;
typedef pair<int, int> pii;

struct Segtree {
    vector<ll> dat;
    int n;
    
    Segtree(int _n) : n(_n), dat(2*n, 0) {}
    
    void add(int k, ll x) {
        k += n;
        dat[k] += x;
        while(k > 1) {
            k >>= 1;
            dat[k] = max(dat[2*k], dat[2*k+1]);
        }
    }
    
    ll query(int a, int b) {
        a += n; b += n;
        ll res = 0;
        while(a < b) {
            if(a & 1) res = max(res, dat[a++]);
            if(b & 1) res = max(res, dat[--b]);
            a >>= 1; b >>= 1;
        }
        return res;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t; cin >> t;
    while(t--) {
        int n, k; cin >> n >> k;
        vector<pii> ab(n);
        rep(i, n) cin >> ab[i].first >> ab[i].second;
        
        sort(all(ab));
        reverse(all(ab));
        
        Segtree st(n);
        ll sum = 0;
        rep(i, k) {
            st.add(i, ab[i].second);
            sum += ab[i].second;
        }
        
        ll ans = sum * ab[k-1].first;
        for(int i = k; i < n; ++i) {
            sum -= st.query(i-k, i);
            st.add(i-k, ab[i-k].second);
            st.add(i, ab[i].second);
            sum += ab[i].second;
            ans = min(ans, sum * ab[i].first);
        }
        
        cout << ans << "\n";
    }
    
    return 0;
}
