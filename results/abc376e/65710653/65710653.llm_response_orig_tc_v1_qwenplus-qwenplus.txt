#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (a); i < (b); ++i)
#define per(i, a, b) for(ll i = (b)-1; i >= (a); --i)

void solve() {
    ll N, K;
    cin >> N >> K;

    vector<pll> items(N);
    for (ll i = 0; i < N; ++i) {
        cin >> items[i].first;
    }
    for (ll i = 0; i < N; ++i) {
        cin >> items[i].second;
    }

    sort(items.begin(), items.end());

    multiset<ll> ms;
    ll sum = 0;
    for (ll i = 0; i < K; ++i) {
        ms.insert(items[i].second);
        sum += items[i].second;
    }

    ll res = sum * items[K - 1].first;

    for (ll i = K; i < N; ++i) {
        // Remove the largest B value in the current set
        sum -= *ms.rbegin();
        ms.erase(prev(ms.end()));

        // Add new element
        ms.insert(items[i].second);
        sum += items[i].second;

        res = min(res, sum * items[i].first);
    }

    cout << res << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll T;
    cin >> T;

    while (T--) {
        solve();
    }

    return 0;
}
