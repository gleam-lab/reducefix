#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define fi first
#define se second
#define pb push_back
#define rep(i, a, b) for (ll i = (a); i < (b); i++)
#define rrep(i, a, b) for (ll i = (a)-1; i >= (b); i--)

void solve() {
    ll N, K;
    cin >> N >> K;
    vector<pll> items(N);
    for (ll i = 0; i < N; ++i) cin >> items[i].fi;
    for (ll i = 0; i < N; ++i) cin >> items[i].se;

    // Sort by A_i ascending
    sort(items.begin(), items.end());

    // Use min-heap to maintain top K B_i values
    priority_queue<ll, vector<ll>, greater<ll>> pq;
    ll sum = 0;

    // Initialize with first K elements
    for (ll i = 0; i < K; ++i) {
        pq.push(items[i].se);
        sum += items[i].se;
    }

    ll ans = sum * items[K - 1].fi;

    // Sliding window
    for (ll i = K; i < N; ++i) {
        // Remove the smallest B in the current set
        if (!pq.empty() && items[i].se > pq.top()) {
            sum -= pq.top();
            pq.pop();
            pq.push(items[i].se);
            sum += items[i].se;
        }
        ans = min(ans, sum * items[i].fi);
    }

    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    ll T;
    cin >> T;
    while (T--) solve();
}
