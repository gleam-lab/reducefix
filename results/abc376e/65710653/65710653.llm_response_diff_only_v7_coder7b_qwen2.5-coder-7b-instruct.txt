```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
const int INF = 2e9;

struct Node {
    int val;
    int idx;
};

bool cmp(Node a, Node b) {
    return a.val > b.val || (a.val == b.val && a.idx < b.idx);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<Node> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i].val;
            a[i].idx = i;
        }
        vector<int> b(n);
        for (int i = 0; i < n; ++i) {
            cin >> b[i];
        }

        sort(a.begin(), a.end(), cmp);

        priority_queue<int> pq;
        long long sum = 0;
        for (int i = 0; i < k; ++i) {
            pq.push(b[a[i].idx]);
            sum += b[a[i].idx];
        }

        long long res = sum * a[k - 1].val;
        for (int i = k; i < n; ++i) {
            sum += b[a[i].idx] - pq.top();
            pq.pop();
            pq.push(b[a[i].idx]);
            res = min(res, sum * a[i].val);
        }

        cout << res << "\n";
    }

    return 0;
}
```

### Explanation of Fixes:
1. **Sorting**: The original code sorted the pairs `(A_i, B_i)` by `A_i` in descending order. However, it was not handling the case where two elements have the same `A_i`. This led to incorrect results because the original code did not consider the indices when sorting. The updated code uses a custom comparator that sorts primarily by `A_i` in descending order and secondarily by index in ascending order to handle ties correctly.

2. **Priority Queue**: Instead of using a multiset to keep track of the sums, we use a priority queue to efficiently manage the smallest element for removal and addition. This helps in maintaining the correct sum efficiently.

3. **Efficiency**: By using a priority queue instead of a multiset, the time complexity is reduced from O(N log N) to O(N log K), which is more efficient given the constraints.

4. **Input/Output Optimization**: Added `ios_base::sync_with_stdio(false)` and `cin.tie(nullptr)` to optimize I/O operations, which can significantly improve performance on large inputs.
