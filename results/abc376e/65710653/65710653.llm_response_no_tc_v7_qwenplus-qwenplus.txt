#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (a); i < (b); ++i)
#define rrep(i, a, b) for(ll i = (a); i >= (b); --i)
#define all(v) v.begin(), v.end()

void solve() {
    ll N, K;
    cin >> N >> K;
    vector<pair<ll, ll>> items(N); // A_i, B_i
    for (ll i = 0; i < N; ++i) cin >> items[i].first;
    for (ll i = 0; i < N; ++i) cin >> items[i].second;

    // Sort by A_i in increasing order
    sort(all(items));

    // Min-heap to keep track of smallest K B_i values
    priority_queue<ll, vector<ll>, greater<ll>> pq;
    ll sum_B = 0;
    for (ll i = 0; i < K; ++i) {
        pq.push(items[i].second);
        sum_B += items[i].second;
    }

    ll ans = sum_B * items[K - 1].first;

    // Sliding window
    for (ll i = K; i < N; ++i) {
        // Remove the largest B from current K elements if it's bigger than current item's B
        if (items[i].second > pq.top()) {
            sum_B -= pq.top();
            pq.pop();
            pq.push(items[i].second);
            sum_B += items[i].second;
        }
        ans = min(ans, sum_B * items[i].first);
    }

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    ll T;
    cin >> T;
    while (T--) {
        solve();
    }
}
