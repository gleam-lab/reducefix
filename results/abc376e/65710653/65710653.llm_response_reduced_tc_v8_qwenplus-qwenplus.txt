#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (a); i < (b); i++)
#define rrep(i, a, b) for(ll i = (a)-1; i >= (b); i--)
#define all(x) (x).begin(), (x).end()

inline void scan() {}
template<class Head, class... Tail>
inline void scan(Head& head, Tail&... tail) {
    std::cin >> head;
    scan(tail...);
}
#define LL(...) ll __VA_ARGS__; scan(__VA_ARGS__)
#define print(...) ([](auto&&... args) { ((std::cout << args << " "), ...); cout << "\n"; }(__VA_ARGS__))

void solve() {
    LL(n, k);
    vector<pll> items(n);
    for (auto &x : items) cin >> x.first;
    for (auto &x : items) cin >> x.second;

    // Sort by A_i ascending
    sort(all(items), [](const pll &a, const pll b) {
        return a.first < b.first;
    });

    // Max-heap to maintain the K smallest B_i values
    multiset<ll, greater<>> top_k_b;
    ll sum_b = 0;

    // Initialize with first K elements
    rep(k) {
        top_k_b.insert(items[i].second);
        sum_b += items[i].second;
    }

    ll ans = sum_b * items[k - 1].first;

    // Slide window
    rep(i, 0, n - k) {
        ll curr_a = items[i].first;
        ll remove_b = items[i].second;
        ll add_a = items[i + k].first;
        ll add_b = items[i + k].second;

        // Remove the larger value to keep smallest K
        if (top_k_b.count(remove_b)) {
            sum_b -= remove_b;
            top_k_b.erase(top_k_b.find(remove_b));
        } else {
            // We need to remove one of the larger elements in top_k_b
            ll max_in_set = *top_k_b.begin();
            sum_b -= max_in_set;
            top_k_b.erase(top_k_b.begin());
        }

        // Add new element
        top_k_b.insert(add_b);
        sum_b += add_b;

        // Current maximum A is add_a
        ans = min(ans, sum_b * add_a);
    }

    print(ans);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    LL(t);
    rep(t) solve();
}
