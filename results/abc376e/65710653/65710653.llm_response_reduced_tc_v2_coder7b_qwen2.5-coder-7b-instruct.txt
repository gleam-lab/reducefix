#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for(int i = 0; i < (n); ++i)
#define rrep(i, n) for(int i = (n)-1; i >= 0; --i)
#define de(x) cout << #x << "=" << x << endl
#define rep1(i, n) for(int i = 1; i <= (n); ++i)

typedef long long int ll;
const ll MOD = 1e9+7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int t;
    cin >> t;
    while(t--) {
        int n, k;
        cin >> n >> k;
        vector<int> a(n), b(n);
        rep(i, n) cin >> a[i];
        rep(i, n) cin >> b[i];

        vector<pair<int, int>> ab(n);
        rep(i, n) ab[i] = {a[i], b[i]};
        sort(ab.begin(), ab.end());

        priority_queue<int> pq;
        int sum = 0;
        rep(i, k) {
            pq.push(ab[i].second);
            sum += ab[i].second;
        }

        int res = sum * ab[k-1].first;
        for(int i = k; i < n; ++i) {
            sum -= pq.top();
            pq.pop();
            pq.push(ab[i].second);
            sum += ab[i].second;
            res = min(res, sum * ab[i].first);
        }

        cout << res << endl;
    }

    return 0;
}
