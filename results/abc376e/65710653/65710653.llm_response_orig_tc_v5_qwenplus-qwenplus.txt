#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;

#define rep(i, a, b) for(ll i = (ll)(a); i < (ll)(b); i++)
#define per(i, a, b) for(ll i = (ll)(b)-1; i >= (ll)(a); i--)
#define all(v) begin(v), end(v)

void solve() {
    ll T;
    cin >> T;

    while (T--) {
        ll N, K;
        cin >> N >> K;

        vector<pll> A(N);
        for (auto &[x, y] : A) cin >> x;
        for (auto &[x, y] : A) cin >> y;

        sort(all(A));

        multiset<ll> ms;
        ll sum = 0;
        for (int i = 0; i < K; ++i) {
            ms.insert(A[i].second);
            sum += A[i].second;
        }

        ll res = sum * A[K - 1].first;

        for (int i = K; i < N; ++i) {
            ms.insert(A[i].second);
            sum += A[i].second;

            if (ms.size() > K) {
                sum -= *ms.rbegin();
                ms.erase(prev(ms.end()));
            }

            res = min(res, sum * A[i].first);
        }

        cout << res << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
