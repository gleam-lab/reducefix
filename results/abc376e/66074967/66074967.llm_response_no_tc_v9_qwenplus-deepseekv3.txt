2. **Priority Queue Usage**: The code uses a priority queue to keep track of the top \(K-1\) elements, but this doesn't correctly account for the sum of the smallest \(K-1\) \(B_i\) values when combined with the next \(A_i\).

### Approach
1. **Sort by \(A_i\)**: First, sort both \(A\) and \(B\) based on \(A_i\) in ascending order. This allows us to consider subsets where the maximum \(A_i\) is fixed.
2. **Prefix Sum and Priority Queue**: For each \(i\) from \(K\) to \(N\), fix \(A_i\) as the maximum \(A\) in the subset. The sum of \(B_i\) should be the sum of the smallest \(K-1\) \(B_j\) values for \(j < i\) plus \(B_i\). This can be efficiently computed using a max-heap (priority queue) to keep track of the smallest \(K-1\) elements.

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

const int inf = 1e18;

signed main() {
    ios_base::sync_with_stdio(false);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> a(n);
        vector<int> b(n);
        
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> b[i];
        }
        
        sort(a.begin(), a.end());
        vector<int> sorted_b(n);
        for (int i = 0; i < n; ++i) {
            sorted_b[i] = b[a[i].second];
        }
        
        priority_queue<int> pq;
        int sum = 0;
        int ans = inf;
        
        for (int i = 0; i < n; ++i) {
            int current_sum = sum + sorted_b[i];
            int current_max = a[i].first;
            if (pq.size() == k - 1) {
                ans = min(ans, current_sum * current_max);
            }
            pq.push(sorted_b[i]);
            sum += sorted_b[i];
            if (pq.size() > k - 1) {
                sum -= pq.top();
                pq.pop();
            }
        }
        
        cout << ans << '\n';
    }
    return 0;
}
