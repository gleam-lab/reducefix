        sort(a + 1, a + n + 1);
        sort(b + 1, b + n + 1);

        int max_a = 0, sum_b = 0;

            max_a = max(max_a, a[i].first);
            sum_b += b[a[i].second].first;
        }

            sum_b -= b[a[i - k].second].first;
            sum_b += b[a[i].second].first;
            ans = min(ans, max_a * sum_b);
        }

        cout << ans << endl;

}
