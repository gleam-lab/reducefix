```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define int long long
#define ll long long

const int N = 2e5 + 9, inf = 1000000009;

void solve(int n, int k, vector<pair<ll, int>>& a, vector<pair<ll, int>>& b) {
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    vector<ll> prefix_sum(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + b[i - 1].first;
    }

    ll ans = inf;
    for (int i = k - 1; i < n; ++i) {
        ll max_a = a[i].first;
        ll sum_b = prefix_sum[i + 1] - prefix_sum[i - k + 1];
        ans = min(ans, max_a * sum_b);
    }

    cout << ans << endl;
}

signed main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        
        vector<pair<ll, int>> a(n), b(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> b[i].first;
            b[i].second = i;
        }

        solve(n, k, a, b);
    }

    return 0;
}
```

This solution involves sorting both arrays `a` and `b`, then calculating prefix sums of array `b`. It iterates through the sorted array `a`, maintaining a window of size `k` for array `b` to calculate the minimum value of the required expression. This approach ensures correctness and efficiency within the given constraints.
