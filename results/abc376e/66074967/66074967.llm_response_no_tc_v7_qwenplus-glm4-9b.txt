```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define int long long

int solve(int n, int k, vector<int>& A, vector<int>& B) {
    vector<pair<int, int>> AB;
    for (int i = 0; i < n; i++) {
        AB.emplace_back(A[i], B[i]);
    }
    sort(AB.begin(), AB.end());

    int min_value = INT_MAX;
    int prefix_sum = 0;
    for (int i = 0; i <= n - k; i++) {
        int sumB = 0;
        for (int j = i; j < i + k && j < n; j++) {
            sumB += AB[j].second;
        }
        int maxA = AB[i + k - 1].first;
        min_value = min(min_value, maxA * sumB);
    }
    return min_value;
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;
        vector<int> A(N), B(N);
        for (int i = 0; i < N; i++) {
            cin >> A[i];
        }
        for (int i = 0; i < N; i++) {
            cin >> B[i];
        }
        cout << solve(N, K, A, B) << endl;
    }
    return 0;
}
```

### Explanation:

1. **Sorting pairs**: Instead of separately sorting the arrays `A` and `B`, we combine them into a single vector of pairs `(value from A, corresponding index from B)`. This allows us to sort them by the values from `A` and still know the indices from `B`.

2. **Prefix sums**: We iterate through every possible subset of size `k` of the sorted pairs. For each subset, we calculate the sum of values from `B` for the indices in that subset. This is done using a simple loop that adds the `B` values for each member of a subset.

3. **Max value from A**: We take the maximum value from `A` for the current subset by simply accessing the corresponding `A` value for the largest element in the subset (since the pairs are sorted by `A` values).

4. **Compute the answer**: For each subset, we compute the expression and keep track of the minimum value.

This approach ensures that we are only iterating through each possible subset once, making the solution efficient even for large values of `N` and `K`.
