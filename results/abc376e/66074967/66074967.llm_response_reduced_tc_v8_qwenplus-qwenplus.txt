#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
            A[i].second = i;
        }

        vector<int> B(N);
        for (int i = 0; i < N; ++i) {
            cin >> B[i];
        }

        // Sort items by A[i]
        sort(A.begin(), A.end());

        // Maintain a min-heap to keep track of the largest K-1 B values
        priority_queue<int, vector<int>, greater<>> pq;
        int sumB = 0;
        int res = 1LL * 1e18; // Large enough initial value

        // We'll process from i = 0 to i = N-1 as the max A_i
        for (int i = 0; i < N; ++i) {
            int curA = A[i].first;
            int curB = B[A[i].second];

            // Always add current B to the pool
            pq.push(curB);
            sumB += curB;

            // If we have more than K elements, remove the smallest B
            if (pq.size() > K) {
                sumB -= pq.top();
                pq.pop();
            }

            // Once we have exactly K elements, compute the score
            if (pq.size() == K) {
                res = min(res, curA * sumB);
            }
        }

        cout << res << '\n';
    }

    return 0;
}
