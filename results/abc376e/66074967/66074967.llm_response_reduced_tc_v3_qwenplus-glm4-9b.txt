#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long
const int N = 2e5 + 9, inf = 1000000009;

int n, k, ans, s;
vector<pair<int, int>> a, b;

int main() {
    int t;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        a.clear();
        b.clear();
        s = 0;
        ans = inf;

        for (int i = 0; i < n; i++) {
            int x, y;
            cin >> x;
            s += y = x;
            a.emplace_back(x, i);
        }

        sort(a.begin(), a.end(), greater<pair<int, int>>());

        for (int j = 0; j < k; j++) {
            b.emplace_back(a[j].first, a[j].second);
        }

        for (int i = 0; i < k; i++) {
            int x = b[i].second;
            ans = min(ans, (s - b[i].first + a[b[i].second + 1].first) * b[i].first);
            s -= b[i].first;
            s += a[b[i].second + 1].first;
            b.push_back(a[b[i].second + 1]);
        }

        cout << ans << endl;
    }
    return 0;
}
