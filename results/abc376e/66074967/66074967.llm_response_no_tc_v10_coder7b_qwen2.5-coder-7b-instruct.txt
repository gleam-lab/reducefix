```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define int long long
const int N = 2e5 + 9, inf = 1000000009;

signed main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>>a(n), b(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; i++) {
            cin >> b[i].first;
            b[i].second = i;
        }
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        long long ans = inf;
        for (int i = 0; i <= n - k; i++) {
            ans = min(ans, static_cast<long long>(a[i + k - 1].first) * (b[i].first + b[i + 1].first + ... + b[i + k - 1].first));
        }
        cout << ans << endl;
    }
    return 0;
}
```
