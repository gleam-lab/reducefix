#include<iostream>
#include<algorithm>
using namespace std;

#define ll long long
#define INF 1e18

ll minProduct(int N, int K, vector<ll>& A, vector<ll>& B) {
    // Sort both arrays based on their values
    vector<pair<ll, int>> sortedA(N), sortedB(N);
    for(int i=0; i<N; ++i){
        sortedA[i] = {A[i], i};
        sortedB[i] = {B[i], i};
    }
    sort(sortedA.begin(), sortedA.end());
    sort(sortedB.begin(), sortedB.end());

    // Priority queue to keep track of top K elements in B
    priority_queue<ll, vector<ll>, greater<ll>> pq;
    ll sum = 0;
    for(int i=0; i<K; ++i){
        pq.push(sortedB[i].first);
        sum += sortedB[i].first;
    }

    // Initialize result with a large number
    ll result = INF;
    for(int i=K-1; i<N; ++i){
        // Calculate product for current subset
        result = min(result, (pq.top() * sortedA[i].first));
        
        // Remove the smallest element from previous window
        sum -= pq.top();
        pq.pop();
        
        // Add the next largest element to the window
        pq.push(sortedB[i].first);
        sum += sortedB[i].first;
    }
    return result;
}

int main(){
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    int T; cin >> T;
    while(T--){
        int N, K; cin >> N >> K;
        vector<ll> A(N), B(N);
        for(auto &a: A) cin >> a;
        for(auto &b: B) cin >> b;

        cout << minProduct(N, K, A, B) << '\n';
    }

    return 0;
}
