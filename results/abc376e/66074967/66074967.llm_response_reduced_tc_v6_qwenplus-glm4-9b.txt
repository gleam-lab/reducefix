#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
using namespace std;

#define int long long
const int N = 2e5 + 9;

int32_t main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<int> a(n + 1), b(n + 1);
        for (int i = 1; i <= n; ++i) {
            cin >> a[i];
        }
        for (int i = 1; i <= n; ++i) {
            cin >> b[i];
        }

        // Pairing A and B and sorting by A descending
        vector<pair<int, int>> ab(n + 1);
        for (int i = 1; i <= n; ++i) {
            ab[i] = {a[i], b[i]};
        }
        sort(ab.rbegin(), ab.rend()); // Sort descending by A

        // Prefix sum of B to calculate sum of first k elements quickly
        vector<int> prefixSum(n + 1);
        partial_sum(ab.begin() + 1, ab.begin() + k + 1, prefixSum.begin() + 1);
        prefixSum[0] = 0; // Make it zero-indexed

        int ans = INT_MAX;
        for (int i = 1; i <= k; ++i) {
            // Prefix sum of B up to i-th element
            int sumB = prefixSum[i];
            // Minimum value of A in the next elements (elements with index > i)
            int minA = ab[i + 1].first;
            ans = min(ans, minA * sumB);
        }
        cout << ans << '\n';
    }

    return 0;
}
