#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
        }
        for (int i = 0; i < N; ++i) {
            cin >> A[i].second;
        }

        // Sort items by A_i in ascending order
        sort(A.begin(), A.end());

        // Min-heap to keep the K largest B_i values
        priority_queue<int, vector<int>, greater<>> pq;
        int sum_B = 0;
        int result = 1e18; // Large initial value

        // Process the elements in increasing order of A_i
        for (int i = 0; i < N; ++i) {
            int a_val = A[i].first;
            int b_val = A[i].second;

            pq.push(b_val);
            sum_B += b_val;

            // If we have more than K elements, remove the smallest B_i
            if (pq.size() > K) {
                sum_B -= pq.top();
                pq.pop();
            }

            // Once we have at least K elements, calculate the cost
            if (pq.size() == K) {
                result = min(result, a_val * sum_B);
            }
        }

        cout << result << '\n';
    }

    return 0;
}
