#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll;
const ll INF = 1e9;

void solve() {
    ll n, k;
    cin >> n >> k;
    vector<pair<ll, ll>> a(n), b(n);
    for(ll i = 0; i < n; ++i) {
        cin >> a[i].first;
        a[i].second = i;
    }
    for(ll i = 0; i < n; ++i) {
        cin >> b[i].first;
        b[i].second = i;
    }
    
    // Sort based on A values
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    
    ll sum_b = 0, ans = INF;
    priority_queue<ll, vector<ll>, greater<ll>> pq;
    
    for(ll i = 0; i < n; ++i) {
        pq.push(b[i].first);
        sum_b += b[i].first;
        
        // Maintain a window of size k
        if(pq.size() > k) {
            sum_b -= pq.top();
            pq.pop();
        }
        
        // If we have enough elements to form a valid subset
        if(i >= k - 1) {
            ans = min(ans, (ll)a[i].first * sum_b);
        }
    }
    
    cout << ans << '\n';
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    int t;
    cin >> t;
    while(t--) solve();
    return 0;
}
