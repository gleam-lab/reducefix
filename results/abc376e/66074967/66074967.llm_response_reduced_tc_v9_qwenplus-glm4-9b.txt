#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long
const int N = 2e5 + 9;

int t, n, k;
vector<pair<int, int>> a, b;

int main() {
    cin >> t;
    while (t--) {
        cin >> n >> k;
        a.assign(n, {0, 0});
        b.assign(n, {0, 0});
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> b[i].first;
            b[i].second = i;
        }
        sort(a.begin(), a.end());
        sort(b.begin(), b.end(), [](const pair<int, int>& x, const pair<int, int>& y) {
            return x.first < y.first;
        });

        long long ans = LLONG_MAX;
        long long sumB = 0;
        for (int i = 0; i < n; ++i) {
            sumB += b[i].first;
            if (i >= k - 1) {
                sumB -= b[i - k + 1].first;
                ans = min(ans, a[i].first * sumB);
            }
        }
        cout << ans << '\n';
    }
    return 0;
}
