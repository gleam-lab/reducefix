```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef pair<int, int> pii;

int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);

    int t; cin >> t;
    while(t--){
        int n, k; cin >> n >> k;
        vector<pii> a(n), b(n);
        for(int i = 0; i < n; i++){
            cin >> a[i].first;
            a[i].second = i;
        }
        for(int i = 0; i < n; i++)
            cin >> b[i].first;
        
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());

        vector<int> prefix_b(n + 1);
        for(int i = 0; i < n; i++)
            prefix_b[i + 1] = prefix_b[i] + b[i].first;

        int ans = 1e9;
        for(int i = 0; i + k - 1 < n; i++){
            int max_a = a[i + k - 1].first;
            int sum_b = prefix_b[i + k] - prefix_b[i];
            ans = min(ans, max_a * sum_b);
        }
        cout << ans << '\n';
    }
    return 0;
}
```
