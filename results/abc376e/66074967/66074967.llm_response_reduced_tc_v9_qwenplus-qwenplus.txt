#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        int N, K;
        cin >> N >> K;

        vector<pair<int, int>> A(N);
        for (int i = 0; i < N; ++i) {
            cin >> A[i].first;
            A[i].second = i;
        }

        vector<int> B(N);
        for (int i = 0; i < N; ++i) {
            cin >> B[i];
        }

        // Sort elements based on A_i in ascending order
        sort(A.begin(), A.end());

        // We will use a max heap to keep track of the smallest K B_i values
        // under consideration. The heap will store B_i values.
        priority_queue<int> max_heap;
        int sum_B = 0;
        int result = 1e18; // Large enough initial value

        for (int i = 0; i < N; ++i) {
            int current_A = A[i].first;
            int current_B = B[A[i].second];

            max_heap.push(current_B);
            sum_B += current_B;

            // If we have more than K elements in the heap, remove the largest one
            if (max_heap.size() > K) {
                sum_B -= max_heap.top();
                max_heap.pop();
            }

            // Once we have exactly K elements, start considering results
            if (max_heap.size() == K) {
                result = min(result, sum_B * current_A);
            }
        }

        cout << result << "\n";
    }

    return 0;
}
