#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
#define int long long

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<pair<int, int>> a(n), b(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i].first;
            a[i].second = i;
        }
        for (int i = 0; i < n; ++i) {
            cin >> b[i].first;
        }

        // Sort items by A value
        sort(a.begin(), a.end());

        // Create B array in the order of sorted A
        vector<int> sorted_b(n);
        for (int i = 0; i < n; ++i) {
            sorted_b[i] = b[a[i].second].first;
        }

        // Use a max heap to maintain the K smallest B values among first i+1 elements
        priority_queue<int> pq;
        int sum = 0;
        int result = 1e18; // Initialize with a large number

        for (int i = 0; i < n; ++i) {
            pq.push(sorted_b[i]);
            sum += sorted_b[i];

            // If we have more than k elements, remove the largest one
            if (pq.size() > k) {
                sum -= pq.top();
                pq.pop();
            }

            // Once we have considered at least k elements, we can form a valid subset
            if (i >= k - 1) {
                result = min(result, a[i].first * sum);
            }
        }

        cout << result << '\n';
    }

    return 0;
}
