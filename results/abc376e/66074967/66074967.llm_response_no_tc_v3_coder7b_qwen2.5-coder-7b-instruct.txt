```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define ll long long
const int MAXN = 2e5 + 9;

ll n, k, t, ans;
vector<pair<ll, ll>> a, b;

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> t;
    while (t--) {
        a.clear(), b.clear();
        cin >> n >> k;
        for (int i = 0; i < n; ++i) {
            ll val;
            cin >> val;
            a.emplace_back(val, i + 1);
        }
        for (int i = 0; i < n; ++i) {
            ll val;
            cin >> val;
            b.emplace_back(val, i + 1);
        }

        sort(a.begin(), a.end());
        sort(b.begin(), b.end());

        vector<ll> prefixSum(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            prefixSum[i] = prefixSum[i - 1] + b[i - 1].first;
        }

        ans = LLONG_MAX;
        for (int i = 0; i <= n - k; ++i) {
            ll maxVal = a[i + k - 1].first;
            ll sum = prefixSum[min(i + k, n)] - prefixSum[i];
            ans = min(ans, maxVal * sum);
        }

        cout << ans << '\n';
    }

    return 0;
}
```
