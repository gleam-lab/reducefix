#include<iostream>
#include<algorithm>
using namespace std;

#define int long long
const int INF = 1e9;

int n, k, t, ans;
vector<pair<int, int>> a, b;

signed main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> t;
    while (t--) {
        a.clear(), b.clear();
        cin >> n >> k;
        
        // Read and store pairs for A and B
        for(int i = 1; i <= n; ++i){
            int x; cin >> x;
            a.push_back({x, i});
        }
        for(int i = 1; i <= n; ++i){
            int x; cin >> x;
            b.push_back({x, i});
        }
        
        // Sort arrays based on the first element of pairs
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());

        // Initialize variables
        long long max_a = 0, sum_b = 0, current_sum = 0;
        
        // Iterate through sorted arrays to find the minimum result
        for(int i = 0; i < n; ++i){
            max_a = max(max_a, (long long)a[i].first);
            current_sum += b[i].first;
            if(i >= k) current_sum -= b[i-k].first;
            if(i >= k-1) ans = min(ans, max_a * current_sum);
        }

        cout << ans << '\n';
    }
    return 0;
}
