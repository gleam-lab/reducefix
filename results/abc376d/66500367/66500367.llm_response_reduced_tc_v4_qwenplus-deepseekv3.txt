using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
const int INF = 1e9;
vector<int> g[N];
int n, m, dep[N];

int bfs(int start) {
    fill(dep, dep + n + 1, -1);
    queue<pii> q;
    q.push({start, -1});
    dep[start] = 0;
    int res = INF;
    while (!q.empty()) {
        int u = q.front().first, parent = q.front().second;
        q.pop();
        for (int v : g[u]) {
            if (v == parent) continue;
            if (dep[v] == -1) {
                dep[v] = dep[u] + 1;
                q.push({v, u});
            } else {
                res = min(res, dep[u] + dep[v] + 1);
            }
        }
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    
    int ans = INF;
    for (int i = 1; i <= n; i++) {
        ans = min(ans, bfs(i));
    }
    
    cout << (ans == INF ? -1 : ans) << '\n';
