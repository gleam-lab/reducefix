```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair <int, int>;

const int N = 2e5 + 5;
vector <int> g[N];
int n, m, dep[N];

int bfs () {
	queue <pii> q;
	q.push ({1, 0});
	dep[1] = 1;
	while (!q.empty ()) {
		int u = q.front ().first, p = q.front ().second;
		q.pop ();
		for (auto v : g[u])
			if (v != p) {
				if (dep[v])
					return dep[u] + dep[v] - 1;
				dep[v] = dep[u] + 1;
				q.push ({v, u});
			}
	}
	return -1;
}

int main () {
	ios :: sync_with_stdio (false);
	cin.tie (nullptr);
	cout.tie (nullptr);
	
	cin >> n >> m;
	for (int i = 1; i <= m; i ++) {
		int u, v;
		cin >> u >> v;
		g[u].push_back (v);
		g[v].push_back(u); // Ensure undirected graph
	}

	cout << bfs () << '\n';
	return 0;
}
```
This solution fixes the issue of not returning the correct result for the tree diameter problem. It ensures that the graph is treated as undirected by adding both edges (u, v) and (v, u). The BFS function calculates the depth of each node and returns the sum of the depths of two nodes at the maximum distance from the root minus one to account for the path length between them.
