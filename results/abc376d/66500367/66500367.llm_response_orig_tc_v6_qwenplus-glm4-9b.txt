const int N = 2e5 + 5;

void bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;

    while (!q.empty()) {

            if (dep[v] == 0) { // If v has not been visited
                dep[v] = dep[u] + 1;
                q.push(v);
            }
        }
    }
}

        g[u].push_back(v);
        g[v].push_back(u); // Assuming undirected graph
    }

    bfs();

    int max_depth = 0;
    for (int i = 1; i <= n; i++) {
