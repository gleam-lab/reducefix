const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], parent[N];

int bfs() {
    int shortest_cycle = INT_MAX;
    for (int i = 1; i <= n; i++) {
        if (!dep[i]) {
            queue<int> q;
            q.push(i);
            dep[i] = 1;
            parent[i] = -1;
            
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                
                for (int v : g[u]) {
                    if (!dep[v]) {
                        dep[v] = dep[u] + 1;
                        parent[v] = u;
                        q.push(v);
                    } else if (v != parent[u]) {
                        shortest_cycle = min(shortest_cycle, dep[u] + dep[v] - 1);
                    }
                }
            }
        }
    }
    return shortest_cycle == INT_MAX ? -1 : shortest_cycle;
}
