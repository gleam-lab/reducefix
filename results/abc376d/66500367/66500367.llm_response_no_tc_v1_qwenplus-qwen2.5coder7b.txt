```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

int bfs() {
    queue<pii> q;
    q.push({1, 0});
    dep[1] = 1;
    while (!q.empty()) {
        int u = q.front().first, p = q.front().second;
        q.pop();
        for (auto v : g[u]) {
            if (v != p && !dep[v]) { // Ensure not revisiting parent and check if already visited
                dep[v] = dep[u] + 1;
                q.push({v, u});
            }
        }
    }
    return *max_element(dep + 1, dep + n + 1); // Return the maximum depth found
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Add this line to make it an undirected graph
    }

    cout << bfs() << '\n';
    return 0;
}
```
This solution includes the necessary modification to handle an undirected graph and correctly implements BFS to find the maximum depth of any node in the tree. The function `bfs` returns the longest path from the root node (node 1).
