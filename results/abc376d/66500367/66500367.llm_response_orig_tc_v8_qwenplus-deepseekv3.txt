After analyzing the incorrect code and the failing test case, I identified that the issue lies in the BFS implementation for finding the shortest cycle in the graph. The original code incorrectly returns the depth of one of the nodes when it finds a visited node, which doesn't necessarily represent the shortest cycle length.

Here's the corrected C++ code that properly finds the shortest cycle in an undirected graph:

```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, par[N], dep[N];

int bfs() {
    int shortest_cycle = INT_MAX;
    for (int i = 1; i <= n; i++) {
        if (dep[i] == 0) {
            queue<int> q;
            q.push(i);
            dep[i] = 1;

            while (!q.empty()) {
                int u = q.front();
                q.pop();

                for (int v : g[u]) {
                    if (dep[v] == 0) {
                        dep[v] = dep[u] + 1;
                        par[v] = u;
                        q.push(v);
                    } else if (par[u] != v && par[v] != u) {
                        shortest_cycle = min(shortest_cycle, dep[u] + dep[v] - 1);
                    }
                }
            }
        }
    }
    return shortest_cycle == INT_MAX ? -1 : shortest_cycle;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Add this line for undirected graph
    }

    cout << bfs() << '\n';
    return 0;
}
```

Key fixes made:
1. Added parent tracking (par array) to properly identify cycles
2. Modified the BFS to check for cycles by looking at visited nodes that aren't the immediate parent
3. Calculated cycle length as dep[u] + dep[v] - 1 when finding a back edge
4. Kept track of the shortest cycle found across all BFS runs
5. Added the reverse edge for undirected graphs (g[v].push_back(u))
6. Initialized all nodes with dep 0 and processed each connected component
