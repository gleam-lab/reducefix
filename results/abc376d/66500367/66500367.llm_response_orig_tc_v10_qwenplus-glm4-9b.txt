The issue in the code is that it returns the depth of the first node it encounters that has already been visited, which is incorrect for the BFS traversal which should calculate the depth for all nodes.

#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair <int, int>;

const int N = 2e5 + 5;
vector <int> g[N];
int n, m, dep[N];

void bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (auto v : g[u]) {
            if (!dep[v]) {  // If v has not been visited
                dep[v] = dep[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);  // Since the graph is undirected
    }

    bfs();

    int maxDep = *max_element(dep + 1, dep + n + 1);  // Find the maximum depth
    cout << maxDep << '\n';
    return 0;
}

3. After BFS traversal, it finds the maximum value in the `dep` array to get the maximum distance from the root node to any other node.
4. Since the graph is undirected, an edge from `u` to `v` is added if an edge from `v` to `u` is also added. This ensures that both directions are considered for each edge.
