bool dfs(int u, int c) {
    color[u] = c;
    vis[u] = true;
    for (int v : g[u]) {
        if (!vis[v]) {
            if (!dfs(v, !c)) return false;
        } else if (color[v] == color[u]) return false;
    }
    return true;
}
