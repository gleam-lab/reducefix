#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 200005;
vector<int> g[N];
int n, m, dep[N], parent[N];

int bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;
    parent[1] = -1;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (auto v : g[u]) {
            if (!dep[v]) {
                dep[v] = dep[u] + 1;
                parent[v] = u;
                q.push(v);
            } else if (v != parent[u]) {
                // Found a cycle, return the sum of depths from both nodes to LCA
                int cycle_length = dep[u] + dep[v] - 2 * depLCA(u, v) + 1;
                return cycle_length;
            }
        }
    }
    return -1;
}

// Helper function to find Lowest Common Ancestor (LCA)
int lca(int u, int v) {
    if (dep[u] < dep[v]) swap(u, v);
    while (dep[u] > dep[v]) {
        u = parent[u];
    }
    while (u != v) {
        u = parent[u];
        v = parent[v];
    }
    return u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    cout << bfs() << '\n';
    return 0;
}
