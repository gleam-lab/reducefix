
```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair <int, int>;

const int N = 2e5 + 5;
vector <int> g[N];
int n, m, dep[N];

int bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : g[u]) {
            if (dep[v] != 0) { // Check for visited nodes
                return dep[u];
            }
            dep[v] = dep[u] + 1;
            q.push(v);
        }
    }
    return -1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs() << '\n';
    return 0;
}
```

### Analysis and Fixes
1. **Check if `v` has been visited**: Before updating the depth of `v`, I added a check to ensure that `dep[v]` is not zero. This prevents reprocessing the same node and ensures that we only process each node once.
2. **Changed queue type**: The queue is now `queue<int>` instead of `queue<pii>`. The previous use of `pii` was unnecessary as we only need the node integer for each step.
3. **Edge case handling**: The check for zero depth (i.e., `dep[v] != 0`) also handles the edge case where the graph is disconnected. If the BFS completes without finding a cycle, it will return `-1`.
