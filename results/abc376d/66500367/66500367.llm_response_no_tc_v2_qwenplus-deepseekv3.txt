#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

int bfs() {
    int shortest_cycle = -1;
    for (int start = 1; start <= n; ++start) {
        queue<pii> q;
        q.push({start, -1});
        fill(dep, dep + n + 1, -1);
        dep[start] = 1;
        while (!q.empty()) {
            int u = q.front().first, p = q.front().second;
            q.pop();
            for (auto v : g[u]) {
                if (v == p) continue;
                if (dep[v] == -1) {
                    dep[v] = dep[u] + 1;
                    q.push({v, u});
                } else {
                    int cycle_length = dep[u] + dep[v] - 1;
                    if (shortest_cycle == -1 || cycle_length < shortest_cycle) {
                        shortest_cycle = cycle_length;
                    }
                }
            }
        }
    }
    return shortest_cycle;
}

        g[u].push_back(v);
        g[v].push_back(u); // Undirected graph
