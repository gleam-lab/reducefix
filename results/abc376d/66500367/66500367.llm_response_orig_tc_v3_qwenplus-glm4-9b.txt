using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

int bfs() {
    queue<pii> q;
    q.push({1, 0});
    dep[1] = 1;
    int maxDepth = 0; // To track the maximum depth
    while (!q.empty()) {
        int u = q.front().first, p = q.front().second;
        q.pop();
        int currentDepth = dep[u];
        maxDepth = max(maxDepth, currentDepth); // Update the maximum depth
        for (int v : g[u])

                if (dep[v]) // This condition is not needed for the purpose of finding max depth
                    continue;
                dep[v] = dep[u] + 1;
                q.push({v, u});
            }
    }
    return maxDepth;
}

    cin.tie(nullptr);
    cout.tie(nullptr);

    for (int i = 1; i <= m; i++) {
