
```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], ans;

int bfs() {
    queue<pii> q;
    q.push({1, 0});
    dep[1] = 1;
    while (!q.empty()) {
        int u = q.front().first, p = q.front().second;
        q.pop();
        for (auto v : g[u])
            if (v != p) {
                if (dep[v])
                    ans = dep[u];
                dep[v] = dep[u] + 1;
                q.push({v, u});
            }
    }
    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs() << '\n';
    return 0;
}
```

### Explanation:
1. Added a global variable `ans` to store the answer of the BFS traversal.
2. Modified the BFS function to return the value of `ans` instead of `-1`, which indicates that the BFS traversal found a cycle.
3. Updated the main function to print the result of the BFS function.
