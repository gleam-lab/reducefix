The initial approach uses BFS to detect cycles by checking if a node has been visited before, but it incorrectly returns the depth of the node when a cycle is detected without verifying if it forms a cycle with the parent node. 

The correct approach involves performing BFS and for each node, checking if any of its adjacent nodes (except the parent) has already been visited. If such a node is found, it means there's a cycle, and the length of the cycle can be calculated as the sum of the depths of the current node and the adjacent node plus one. However, if no cycle is found after BFS completes, we return -1.

#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], parent[N];

int bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;
    parent[1] = -1;
    int result = -1;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (auto v : g[u]) {
            if (v == parent[u]) continue;
            if (dep[v] == 0) {
                dep[v] = dep[u] + 1;
                parent[v] = u;
                q.push(v);
            } else {
                int cycle_length = dep[u] + dep[v] - 1;
                if (result == -1 || cycle_length < result) {
                    result = cycle_length;
                }
            }
        }
    }
    return result;
}

int main() {

    cout.tie(nullptr);

    for (int i = 1; i <= m; i++) {

}
