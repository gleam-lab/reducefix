#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

// Function to perform BFS and find the maximum depth of any cycle
int bfs() {
    queue<pii> q;
    q.push({1, 0}); // Start from node 1 with parent 0 (no parent)
    dep[1] = 1; // Initialize depth of node 1 as 1
    
    while (!q.empty()) {
        int u = q.front().first, p = q.front().second;
        q.pop();
        
        for (auto v : g[u]) {
            if (v == p) continue; // Skip the parent node
            
            if (dep[v]) { // If node v is already visited, it means we found a cycle
                return dep[u]; // Return the depth of node u
            }
            
            dep[v] = dep[u] + 1; // Update the depth of node v
            q.push({v, u}); // Push node v into the queue with current node u as its parent
        }
    }
    
    return -1; // No cycle found, return -1
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v); // Add edge from u to v
        g[v].push_back(u); // Add edge from v to u (for undirected graph)
    }

    cout << bfs() << '\n'; // Output the result of BFS
    return 0;
}
