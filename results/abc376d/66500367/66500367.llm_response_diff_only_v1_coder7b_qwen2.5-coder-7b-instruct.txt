#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

// Initialize depth array to zero
void init() {
    for(int i = 1; i <= n; ++i)
        dep[i] = 0;
}

// Function to find the diameter of the tree using BFS
int bfs() {
    queue<pii> q;
    q.push({1, 0}); // Start from node 1 with distance 0
    dep[1] = 1;

    while(!q.empty()) {
        int u = q.front().first, p = q.front().second;
        q.pop();
        
        for(auto v : g[u]) {
            if(v == p) continue; // Skip parent node
            if(dep[v]) return dep[u]; // If already visited, return current depth
            dep[v] = dep[u] + 1; // Update depth
            q.push({v, u}); // Push child node with parent node as reference
        }
    }

    return -1; // Return -1 if no cycle is found
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for(int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Add both directions since it's an undirected graph
    }

    init(); // Initialize depth array
    cout << bfs() << '\n';

    return 0;
}
