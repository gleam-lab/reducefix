#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], parent[N];

int bfs() {
    queue<int> q;
    q.push(1);
    dep[1] = 1;
    parent[1] = -1; // No parent for the root node
    int cycle_length = -1;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (auto v : g[u]) {
            if (v == parent[u]) continue; // Skip the parent node
            if (dep[v] == 0) { // Not visited
                dep[v] = dep[u] + 1;
                parent[v] = u;
                q.push(v);
            } else { // Visited, cycle detected
                // The cycle length is dep[u] + dep[v] - 1
                int current_cycle_length = dep[u] + dep[v] - 1;
                if (cycle_length == -1 || current_cycle_length < cycle_length) {
                    cycle_length = current_cycle_length;
                }
            }
        }
    }
    return cycle_length;
}

int main() {
    ios::sync_with_stdio(false);

    for (int i = 1; i <= m; i++) {

        g[u].push_back(v);
        g[v].push_back(u); // Since the graph is undirected
    }

}

2. **Parent Tracking**: The `parent` array keeps track of the parent of each node to avoid revisiting the parent during BFS.
3. **Cycle Detection**: When a visited node is encountered (that isn't the parent), a cycle is detected. The cycle length is calculated as the sum of the depths of the current node and the encountered node minus one.
4. **Cycle Length Calculation**: The shortest cycle length is tracked and returned, or -1 if no cycle is found.
