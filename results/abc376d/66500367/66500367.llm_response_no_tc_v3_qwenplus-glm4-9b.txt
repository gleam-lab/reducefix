#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, par[N];

int bfs () {
    queue<pii> q;
    q.push({1, -1});
    par[1] = 1; // Start from node 1 with parent -1
    while (!q.empty()) {
        int u = q.front().first;
        int p = q.front().second;
        q.pop();

        for (int v : g[u]) {
            if (v == p)
                continue; // Skip the parent node
            if (par[v])
                return par[u]; // Found a cycle
            par[v] = u;
            q.push({v, u});
        }
    }
    return -1; // No cycle found
}

int main () {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Ensure bidirectional graph for cycle detection
    }

    cout << bfs() << '\n';
    return 0;
}
