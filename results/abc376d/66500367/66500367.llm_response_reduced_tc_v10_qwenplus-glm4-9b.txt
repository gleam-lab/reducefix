#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair <int, int>;

const int N = 2e5 + 5;
vector <int> g[N];
int n, m, dist[N];

int bfs() {
    queue<int> q;
    q.push(1);
    dist[1] = 1;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : g[u]) {
            if (dist[v] == 0) {  // This ensures we only visit nodes that haven't been visited yet
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    return dist[n];  // We return the distance to the nth node
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs() << '\n';
    return 0;
}
