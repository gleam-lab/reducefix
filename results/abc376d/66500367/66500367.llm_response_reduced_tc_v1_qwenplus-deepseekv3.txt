To fix this code, we need to correctly identify the shortest cycle in the graph. The original code attempts to find the shortest cycle starting from node 1 using BFS, but it incorrectly returns the depth of the node when it encounters a visited node, which does not necessarily indicate a cycle. To correctly find the shortest cycle, we should use BFS to explore the graph and keep track of the parent node to avoid revisiting it. When we encounter a node that has already been visited and it's not the parent, we have found a cycle. The length of the cycle is then the sum of the depths of the current node and the encountered node plus one.

### Approaches
1. **BFS for Shortest Cycle Detection**: We use BFS starting from each node to explore the graph. For each node, we maintain a depth array to keep track of the depth of nodes from the starting node. If we encounter a node that has already been visited and it's not the parent node, we calculate the cycle length as the sum of the depths of the current node and the encountered node plus one.
2. **Early Termination**: Once a cycle is detected during BFS, we return the cycle length immediately. If no cycle is found after BFS completes, we return -1.

#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m;

int bfs(int start) {
    vector<int> dep(n + 1, -1);
    vector<int> parent(n + 1, -1);
    queue<int> q;
    q.push(start);
    dep[start] = 0;
    int min_cycle = INT_MAX;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : g[u]) {
            if (dep[v] == -1) {
                dep[v] = dep[u] + 1;
                parent[v] = u;
                q.push(v);
            } else if (v != parent[u]) {
                min_cycle = min(min_cycle, dep[u] + dep[v] + 1);
            }
        }
    }

    return min_cycle == INT_MAX ? -1 : min_cycle;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {

        g[u].push_back(v);
        g[v].push_back(u);
    }

    int min_cycle = INT_MAX;
    for (int i = 1; i <= n; ++i) {
        int cycle = bfs(i);
        if (cycle != -1) {
            min_cycle = min(min_cycle, cycle);
        }
    }

    cout << (min_cycle == INT_MAX ? -1 : min_cycle) << '\n';
    return 0;
}
