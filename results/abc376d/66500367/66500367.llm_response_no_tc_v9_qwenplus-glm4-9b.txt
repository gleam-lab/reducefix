
```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], par[N];

int bfs () {
    queue<pii> q;
    q.push({1, 0});
    dep[1] = 1;
    par[1] = 1; // Set parent to the node itself to detect cycles

    while (!q.empty()) {
        int u = q.front().first;
        int p = q.front().second;
        q.pop();

        for (auto v : g[u]) {
            if (v != p) {
                if (dep[v]) { // If v is already visited, there's a cycle
                    return dep[u];
                }
                dep[v] = dep[u] + 1;
                par[v] = u; // Update parent to u
                q.push({v, u});
            }
        }
    }
    return -1; // No cycle found
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs() << '\n';
    return 0;
}
```

### Explanation:
1. **Initialization of `par[N]`:** I added an array `par[N]` to keep track of the parent of each node. This helps in identifying the cycle once it's detected.
2. **Cycle Detection:** When visiting a node `v` for the first time, I check if it's already visited by checking `dep[v]`. If it is, then there's a cycle, and we return `dep[u]`.
3. **Parent Update:** I update `par[v]` to `u` to keep track of the parent of each node, which helps in identifying the path of the cycle if needed.
4. **Return Value:** The function returns `-1` if no cycle is found, indicating that the graph is a tree.
