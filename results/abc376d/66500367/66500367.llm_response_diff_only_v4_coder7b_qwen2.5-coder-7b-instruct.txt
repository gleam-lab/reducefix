```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair <int, int>;

const int N = 2e5 + 5;
vector <int> g[N];
int n, m, dep[N];

int bfs () {
	queue <int> q;
	q.push (1);
	dep[1] = 1;
	while (!q.empty ()) {
		int u = q.front ();
		q.pop ();
		for (auto v : g[u]) {
			if (!dep[v]) {
				dep[v] = dep[u] + 1;
				q.push (v);
			}
		}
	}
	return dep[n] ? dep[n] : -1;
}

int main () {
	ios :: sync_with_stdio (false);
	cin.tie (nullptr);
	cout.tie (nullptr);
	
	cin >> n >> m;
	for (int i = 1; i <= m; i ++) {
		int u, v;
		cin >> u >> v;
		g[u].push_back (v);
		g[v].push_back (u); // Ensure the graph is undirected
	}

	cout << bfs () << '\n';
	return 0;
}
```

### Explanation of Fixes
1. **Graph Representation**: The original code assumed an undirected graph but did not add reverse edges. Added `g[v].push_back(u)` to ensure the graph is undirected.
2. **BFS Initialization**: Changed the BFS initialization to start from node 1 directly without setting a parent parameter.
3. **Edge Case Handling**: The function now returns `-1` if there is no path from node 1 to node `n`, ensuring correct handling of disconnected graphs.
