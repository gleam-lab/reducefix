The BFS approach was incorrect because it didn't properly handle bidirectional edges and didn't correctly track the parent-child relationships needed to detect cycles.

Here's the corrected code that properly finds the shortest cycle in an undirected graph:

```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N], parent[N];

int bfs() {
    int shortest_cycle = INT_MAX;
    for (int i = 1; i <= n; i++) {
        if (dep[i]) continue;
        
        queue<int> q;
        q.push(i);
        dep[i] = 1;
        parent[i] = -1;
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            
            for (int v : g[u]) {
                if (v == parent[u]) continue;
                
                if (!dep[v]) {
                    dep[v] = dep[u] + 1;
                    parent[v] = u;
                    q.push(v);
                } else {
                    shortest_cycle = min(shortest_cycle, dep[u] + dep[v] - 1);
                }
            }
        }
    }
    return shortest_cycle == INT_MAX ? -1 : shortest_cycle;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Add both directions for undirected graph
    }
    
    cout << bfs() << '\n';
    return 0;
}
```

1. Added proper handling of bidirectional edges by pushing both (u,v) and (v,u) to the adjacency list
2. Implemented a parent array to track parent-child relationships during BFS
3. Modified the cycle detection logic to find the shortest cycle by comparing all found cycles
4. Added initialization and proper BFS from each unvisited node (though the problem likely ensures connectivity)
5. Return -1 if no cycle is found (though the problem likely guarantees a cycle)

1. Performing BFS from each node
2. Tracking parent nodes to avoid immediate backtracking
3. When encountering an already visited node that's not the parent, calculating the cycle length
4. Keeping track of the shortest cycle found
