#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using pii = pair<int, int>;

const int N = 2e5 + 5;
vector<int> g[N];
int n, m, dep[N];

// Function to perform BFS and find the depth of the tree
int bfs() {
    queue<pii> q;
    q.push({1, 0}); // Start from node 1 with parent -1 and depth 0
    dep[1] = 1;     // Depth of root is 1

    while (!q.empty()) {
        int u = q.front().first, p = q.front().second; // Current node and its parent
        q.pop();

        for (auto v : g[u]) {
            if (v != p) { // Avoid revisiting the parent node
                if (dep[v]) {
                    return dep[u]; // If we encounter a visited node, it means there's a cycle, return the current depth
                }
                dep[v] = dep[u] + 1; // Mark the neighbor as visited with increased depth
                q.push({v, u});      // Push the neighbor into the queue
            }
        }
    }

    return -1; // Return -1 if no cycle is found
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v); // Add edge u -> v
        g[v].push_back(u); // Since the graph is undirected, add edge v -> u
    }

    cout << bfs() << '\n'; // Output the result of BFS
    return 0;
}
