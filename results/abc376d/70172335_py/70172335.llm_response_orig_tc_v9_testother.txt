from collections import defaultdict, deque

def main():
    n, m = map(int, input().split())
    edge = defaultdict(list)
    
    for _ in range(m):
        a, b = map(lambda x: int(x) - 1, input().split())
        edge[a].append(b)
    
    # BFS to find shortest cycle containing vertex 0 (original vertex 1)
    INF = float("inf")
    dist = [INF] * n
    dist[0] = 0
    q = deque()
    q.append(0)
    
    min_cycle = INF
    
    while q:
        u = q.popleft()
        for v in edge[u]:
            if v == 0:
                # Found a cycle back to vertex 0
                min_cycle = min(min_cycle, dist[u] + 1)
            elif dist[v] == INF:
                # First time visiting this vertex
                dist[v] = dist[u] + 1
                q.append(v)
    
    print(min_cycle if min_cycle != INF else -1)

if __name__ == "__main__":
    main()
