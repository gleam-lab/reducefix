from collections import defaultdict, deque

def main():
    n, m = map(int, input().split())
    graph = defaultdict(list)
    
    for _ in range(m):
        a, b = map(int, input().split())
        a -= 1
        b -= 1
        graph[a].append(b)
    
    # Use BFS to find shortest cycle containing vertex 0 (original vertex 1)
    # We'll track the distance from vertex 0 and use parent tracking to avoid immediate backtracking
    dist = [-1] * n
    parent = [-1] * n
    
    q = deque()
    q.append(0)
    dist[0] = 0
    
    found_cycle = False
    min_cycle_length = float('inf')
    
    while q:
        u = q.popleft()
        
        for v in graph[u]:
            if v == 0 and dist[u] >= 1:  # Found a cycle back to vertex 0
                found_cycle = True
                min_cycle_length = min(min_cycle_length, dist[u] + 1)
            elif dist[v] == -1:  # Not visited yet
                dist[v] = dist[u] + 1
                parent[v] = u
                q.append(v)
    
    if found_cycle:
        print(min_cycle_length)
    else:
        print(-1)

if __name__ == "__main__":
    main()
