from collections import defaultdict, deque

def main():
    n, m = map(int, input().split())
    edge = defaultdict(list)
    
    for _ in range(m):
        a, b = map(lambda x: int(x) - 1, input().split())
        edge[a].append(b)
    
    # BFS from vertex 0 (vertex 1 in 1-indexed)
    INF = float("inf")
    dist = [INF] * n
    dist[0] = 0
    q = deque()
    q.append(0)
    
    while q:
        u = q.popleft()
        for v in edge[u]:
            if dist[v] == INF:
                dist[v] = dist[u] + 1
                q.append(v)
    
    # Find minimum cycle containing vertex 0
    ans = INF
    
    # Check all edges that lead back to vertex 0
    for v in edge[0]:
        if dist[v] != INF:
            ans = min(ans, dist[v] + 1)
    
    # For other vertices, check if there's a path from 0 to v and from v to 0
    for u in range(n):
        if u == 0 or dist[u] == INF:
            continue
            
        # BFS from u to find shortest path back to 0
        dist_from_u = [INF] * n
        dist_from_u[u] = 0
        q = deque()
        q.append(u)
        
        while q:
            cur = q.popleft()
            for nxt in edge[cur]:
                if dist_from_nxt] == INF:
                    dist_from_nxt] = dist_from_cur] + 1
                    q.append(nxt)
        
        if dist_from_nxt[0] != INF:
            ans = min(ans, dist[u] + dist_from_nxt[0])
    
    print(ans if ans != INF else -1)

if __name__ == "__main__":
    main()
