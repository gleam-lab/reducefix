```cpp
#include <bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;

typedef long long ll;
const int MAXN = 2e5 + 10;
vector<int> adj[MAXN];
queue<int> q;
int dist[MAXN];
bool visited[MAXN];

void bfs(int start) {
    fill(dist, dist + MAXN, INF);
    dist[start] = 0;
    q.push(start);
    visited[start] = true;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : adj[u]) {
            if (!visited[v]) {
                dist[v] = dist[u] + 1;
                visited[v] = true;
                q.push(v);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, u, v;
    cin >> n >> m;

    for (int i = 0; i < m; ++i) {
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    bfs(1); // Start BFS from node 1

    int min_dist = INT_MAX;
    bool found = false;

    for (int i = 1; i <= n; ++i) {
        if (dist[i] != INF) {
            min_dist = min(min_dist, dist[i]);
            found = true;
        }
    }

    if (!found || min_dist == 0) {
        cout << "-1" << endl;
    } else {
        cout << min_dist << endl;
    }

    return 0;
}
```

This solution uses a breadth-first search (BFS) to find the shortest path from node 1 to all other nodes in the graph. It initializes the distance array with infinity and then updates the distances as it explores the graph. Finally, it outputs the minimum distance found, or `-1` if no path exists from node 1 to any other node. This approach ensures that the solution is efficient and handles all edge cases appropriately.
