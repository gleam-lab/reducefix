#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN];
bool vis[MAXN];
int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    q.push(1);
    step[1] = 0; // Initialize step count of the starting node
    while(!q.empty()){
        ll top=q.front();
        vis[top]=true;
        q.pop();
        for(ll i=0;i<vec[top].size();i++){
            ll to=vec[top][i];
            if(!vis[to]){ // Check if the node is not visited
                step[to] = step[top] + 1; // Set step count
                q.push(to);
            }
        }
    }
    if(step[1] == 0){ // If the first node is not visited, print -1
        printf("-1");
    } else {
        printf("%lld",step[1] + 1); // Output the step count to node 1 plus one
    }
    return 0;
}
