Here's the fixed solution that properly handles finding the shortest cycle containing node 1:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll dist[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    queue<ll> q;
    memset(dist, -1, sizeof(dist));
    dist[1] = 0;
    q.push(1);
    
    ll res = INF;
    while(!q.empty()){
        ll u = q.front();
        q.pop();
        
        for(ll v : vec[u]){
            if(v == 1){
                res = min(res, dist[u] + 1);
                continue;
            }
            if(dist[v] == -1){
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    if(res == INF) printf("-1\n");
    else printf("%lld\n", res);
    return 0;
}
```

Key fixes:
1. Instead of immediately returning when finding any path back to node 1, we now track the shortest path length (minimum cycle)
2. Properly initialized distance array with -1 (unvisited)
3. Only push nodes to queue when they haven't been visited before
4. Continue searching even after finding one path back to node 1 to ensure we find the shortest cycle
5. After BFS completes, check if we found any cycle (res != INF) or output -1 if none found

This approach ensures we find the shortest cycle containing node 1 by performing a complete BFS and tracking the minimum cycle length found. The complexity remains O(N + M) which is efficient for the problem constraints.
