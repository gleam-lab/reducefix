ll dist[MAXN];

    ll res = INF;
    
    while(!q.empty()){
        ll u = q.front();
        q.pop();
        
        for(ll v : vec[u]){
            if(v == 1){
                res = min(res, dist[u] + 1);
            }
            if(dist[v] == -1){
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    if(res != INF){
        printf("%lld", res);
    } else {

    }
    return 0;
}
