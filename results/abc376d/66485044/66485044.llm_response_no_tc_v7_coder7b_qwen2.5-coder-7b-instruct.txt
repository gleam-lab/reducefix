```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN = 2e5 + 10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN];
bool vis[MAXN];

int main() {
    ll n, m, u, v;
    scanf("%lld %lld", &n, &m);
    for (ll i = 1; i <= m; i++) {
        scanf("%lld %lld", &u, &v);
        vec[u].push_back(v);
    }
    q.push(1);
    step[1] = 0;
    vis[1] = true;
    while (!q.empty()) {
        ll top = q.front();
        q.pop();
        for (ll i = 0; i < vec[top].size(); i++) {
            ll to = vec[top][i];
            if (!vis[to]) {
                step[to] = step[top] + 1;
                vis[to] = true;
                q.push(to);
            }
        }
    }
    if (vis[n]) {
        printf("%lld\n", step[n]);
    } else {
        printf("-1\n");
    }
    return 0;
}
```

### Explanation of Fixes
1. **Initialization**: Initialize `step[1]` to 0 because the distance from node 1 to itself is 0.
2. **Visited Check**: Before pushing a node into the queue, check if it has already been visited to avoid processing the same node multiple times.
3. **Output Handling**: After the BFS loop, check if node `n` was visited. If it was, print the distance (`step[n]`). Otherwise, print `-1` indicating that node `n` is not reachable from node 1.
4. **Edge Case Handling**: The original code had a logical error where it would terminate prematurely if it found a path back to node 1 from any other node except node 1 itself. This was incorrect as we need to find the shortest path to node `n`, not just any path to node 1.
