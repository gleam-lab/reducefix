```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll dist[MAXN];
bool vis[MAXN];

int main() {
    ll n, m, u, v;
    scanf("%lld %lld", &n, &m);
    for (ll i = 1; i <= m; i++) {
        scanf("%lld %lld", &u, &v);
        vec[u].push_back(v);
    }
    
    memset(dist, -1, sizeof(dist));
    queue<ll> q;
    q.push(1);
    dist[1] = 0;
    ll min_cycle = INF;
    
    while (!q.empty()) {
        ll u = q.front();
        q.pop();
        
        for (ll v : vec[u]) {
            if (v == 1) {
                min_cycle = min(min_cycle, dist[u] + 1);
            }
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    if (min_cycle != INF) {
        printf("%lld\n", min_cycle);
    } else {
        printf("-1\n");
    }
    
    return 0;
}
```

1. **Graph Representation**: The graph is represented using an adjacency list `vec` where each node points to its neighbors.
2. **Initialization**: The `dist` array is initialized to -1 to indicate unvisited nodes, and the distance to node 1 is set to 0.
3. **BFS Execution**: A BFS is initiated from node 1. For each node `u` dequeued, its neighbors `v` are examined.
4. **Cycle Detection**: If a neighbor `v` is node 1, it means a cycle is found. The cycle length is `dist[u] + 1`, and we update `min_cycle` if this length is smaller than the current minimum.
5. **Distance Update**: For unvisited neighbors `v`, the distance is updated and `v` is enqueued.
6. **Result Output**: After BFS completes, if a cycle involving node 1 was found (`min_cycle` is not INF), its length is printed; otherwise, -1 is printed indicating no such cycle exists.

This approach ensures that the shortest cycle involving node 1 is found efficiently using BFS, leveraging the property that BFS explores nodes level by level, thus finding the shortest paths in unweighted graphs.
