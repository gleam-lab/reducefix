#include <bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN = 2e5 + 10;

vector<ll> vec[MAXN];
queue<ll> q;
ll dist[MAXN][2]; // dist[node][0] is distance from node to 1, dist[node][1] is distance from node to cycle back

bool visited[MAXN];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    ll n, m;
    cin >> n >> m;

    for (ll i = 0; i < m; ++i) {
        ll u, v;
        cin >> u >> v;
        vec[u].push_back(v);
    }

    // BFS to compute shortest path to node 1
    memset(dist, 0x3f, sizeof(dist));
    q.push(1);
    dist[1][0] = 0;

    while (!q.empty()) {
        ll u = q.front();
        q.pop();

        for (ll v : vec[u]) {
            if (dist[v][0] > dist[u][0] + 1) {
                dist[v][0] = dist[u][0] + 1;
                q.push(v);
            }
        }
    }

    // BFS from nodes that point to 1 to find shortest cycle
    for (ll u = 1; u <= n; ++u) {
        for (ll v : vec[u]) {
            if (v == 1) {
                if (dist[u][0] != INF) {
                    dist[1][1] = min(dist[1][1], dist[u][0] + 1);
                }
            }
        }
    }

    // For each node, check if it can form a cycle: u -> ... -> 1 -> ... -> u
    for (ll u = 1; u <= n; ++u) {
        if (dist[u][0] != INF && dist[u][1] == INF) {
            for (ll v : vec[u]) {
                if (dist[v][0] != INF) {
                    dist[u][1] = min(dist[u][1], dist[u][0] + dist[v][0] + 1);
                }
            }
        }
    }

    if (dist[1][1] == INF) {
        cout << -1 << endl;
    } else {
        cout << dist[1][1] << endl;
    }

    return 0;
}
