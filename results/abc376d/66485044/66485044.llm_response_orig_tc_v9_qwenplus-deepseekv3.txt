Here's the corrected solution:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll dist[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    fill(dist, dist+MAXN, INF);
    dist[1] = 0;
    
    queue<ll> q;
    q.push(1);
    
    ll res = INF;
    
    while(!q.empty()){
        ll u = q.front();
        q.pop();
        
        for(ll v : vec[u]){
            if(v == 1){
                res = min(res, dist[u] + 1);
            }
            if(dist[v] > dist[u] + 1){
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    if(res == INF) printf("-1\n");
    else printf("%lld\n", res);
    
    return 0;
}
```

Key improvements made:
1. Replaced the original BFS implementation with a cleaner version that properly tracks distances
2. Used a distance array initialized to INF to properly track shortest paths
3. When we encounter node 1 during BFS from node 1, we update the minimum cycle length
4. Added proper handling for the case where no cycle exists (output -1)
5. Removed unnecessary vis array since distance tracking is sufficient
6. The algorithm now correctly finds the shortest cycle starting and ending at node 1 by:
   - Performing BFS from node 1
   - Tracking shortest distance to each node
   - Whenever we find a path back to node 1, we record the cycle length
   - Finally outputting the minimum cycle length found

This solution has O(N + M) time complexity, which is optimal for this problem using BFS.
