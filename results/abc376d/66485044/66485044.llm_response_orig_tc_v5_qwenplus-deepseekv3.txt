Based on the error analysis, the original code was failing to find the shortest cycle from node 1 back to itself. The output was 16 when it should have been 3. This suggests the BFS implementation wasn't correctly detecting immediate cycles or wasn't properly tracking steps.

Here's the corrected solution that properly detects cycles from node 1 back to itself:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN];
bool vis[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    memset(step, -1, sizeof(step));
    step[1] = 0;
    q.push(1);
    
    ll min_cycle = INF;
    
    while(!q.empty()){
        ll top=q.front();
        q.pop();
        
        for(ll to : vec[top]){
            if(to == 1){
                min_cycle = min(min_cycle, step[top] + 1);
                continue;
            }
            if(step[to] == -1){
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }
    
    if(min_cycle != INF)
        printf("%lld\n", min_cycle);
    else
        printf("-1\n");
    
    return 0;
}
```

Key changes made:
1. Initialized step array to -1 (unvisited) instead of 0
2. Set step[1] to 0 explicitly
3. Removed the vis array since we're using step array for tracking visited nodes
4. Simplified the cycle detection logic - whenever we find node 1 in neighbors, we calculate the cycle length
5. Track the minimum cycle length found
6. Only output -1 if no cycle was found

This solution properly performs a BFS from node 1 and checks for any edges that return to node 1, keeping track of the shortest such cycle. The complexity remains O(N + M) which is optimal for this problem.
