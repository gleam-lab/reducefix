After analyzing the incorrect solution and the failing test case, I identified that the issue lies in the BFS implementation for finding the shortest cycle containing node 1. The incorrect approach was prematurely terminating when finding any cycle back to node 1, rather than finding the shortest such cycle.

Here's the corrected solution that properly implements BFS to find the shortest cycle containing node 1:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll dist[MAXN];
ll parent[MAXN];
ll min_cycle = INF;

void bfs(ll start) {
    queue<ll> q;
    q.push(start);
    dist[start] = 0;
    
    while(!q.empty()) {
        ll u = q.front();
        q.pop();
        
        for(ll v : vec[u]) {
            if(v == start && u != start) {
                min_cycle = min(min_cycle, dist[u] + 1);
            }
            if(dist[v] == -1) {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                q.push(v);
            }
        }
    }
}

int main() {
    ll n, m, u, v;
    scanf("%lld %lld", &n, &m);
    for(ll i = 1; i <= m; i++) {
        scanf("%lld %lld", &u, &v);
        vec[u].push_back(v);
    }
    
    memset(dist, -1, sizeof(dist));
    memset(parent, -1, sizeof(parent));
    bfs(1);
    
    if(min_cycle == INF) {
        printf("-1\n");
    } else {
        printf("%lld\n", min_cycle);
    }
    return 0;
}
```

Key improvements made:
1. Properly initialized distance array with -1 to indicate unvisited nodes
2. Used BFS to explore all paths from node 1, tracking distances
3. Checked for cycles back to node 1 when visiting neighbors
4. Kept track of the minimum cycle length found
5. Handled the case where no cycle exists (output -1)

The algorithm now correctly finds the shortest cycle containing node 1 by:
- Performing BFS from node 1
- For each node's neighbors, checking if it can return to node 1 (forming a cycle)
- Keeping track of the minimum such cycle length
- Outputting the result or -1 if no cycle exists

This solution has a time complexity of O(N + M) which is optimal for this problem.
