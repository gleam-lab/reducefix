The original code attempts to find the shortest cycle starting and ending at node 1 in a directed graph using DFS, but it has several issues:
1. It doesn't reset the `vis` array when backtracking, which prevents discovering all possible cycles
2. The condition `if (v != 1 && vis[v]) continue` incorrectly skips visited nodes even when they might be part of valid cycles
3. The approach is inefficient for large graphs as it doesn't use BFS which is more suitable for shortest path problems

Here's the fixed code using BFS for finding the shortest cycle:

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;
const int INF = 1e18;

vector<int> g[N];
int n, m;

int findShortestCycle() {
    int min_cycle = INF;
    queue<pair<int, int>> q;
    vector<int> dist(n + 1, INF);
    vector<int> parent(n + 1, -1);
    
    dist[1] = 0;
    q.push({1, -1});
    
    while (!q.empty()) {
        auto [u, p] = q.front();
        q.pop();
        
        for (int v : g[u]) {
            if (v == p) continue;
            if (dist[v] == INF) {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                q.push({v, u});
            } else if (v == 1) {
                min_cycle = min(min_cycle, dist[u] + 1);
            }
        }
    }
    
    return min_cycle == INF ? -1 : min_cycle;
}

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    
    int ans = findShortestCycle();
    cout << ans << endl;
    
    return 0;
}
```

The fixed code:
1. Uses BFS which is more appropriate for shortest path problems
2. Properly tracks distances and parent nodes
3. Checks for cycles by looking when we encounter node 1 again
4. Handles the case when no cycle exists by returning -1
5. Has better time complexity (O(N + M)) compared to the original DFS approach
