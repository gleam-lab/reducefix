#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m, ans = 1e18;

// BFS is more suitable for shortest cycle detection from node 1
int bfs()
{
    vector<int> dist(N, -1);
    queue<int> q;
    q.push(1);
    dist[1] = 0;

    while (!q.empty())
    {
        int u = q.front();
        q.pop();

        for (int v : g[u])
        {
            if (dist[v] == -1)
            {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
            else if (v != 1 && dist[v] != dist[u] + 1 + 1 - 1) // avoid immediate back edge
            {
                // Found a cycle through already visited node
                return dist[u] + dist[v] + 1;
            }
        }
    }
    return -1;
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Undirected graph
    }

    cout << bfs();

    return 0;
}
