#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m;

// This function was incorrectly trying to find cycles.
// A proper cycle detection or shortest cycle finding needs BFS/DFS with parent tracking
// or advanced techniques. The original DFS logic had multiple issues:
// - It didn't properly detect cycles
// - It reused the `vis` array incorrectly across different paths
// - It tried to optimize with `cnt >= ans` but that could prune valid paths

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    // We'll use BFS from each node to find shortest cycle
    int ans = 1e18;
    for (int start = 1; start <= n; ++start) {
        vector<int> dist(n + 1, -1);
        vector<int> parent(n + 1, -1);
        queue<int> q;
        
        q.push(start);
        dist[start] = 0;
        
        bool found = false;
        while (!q.empty() && !found) {
            int u = q.front();
            q.pop();
            
            for (int v : g[u]) {
                if (dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    parent[v] = u;
                    q.push(v);
                } else if (parent[u] != v) {
                    // Found a cycle
                    ans = min(ans, dist[u] + dist[v] + 1);
                    found = true;
                }
            }
        }
    }

    cout << (ans == 1e18 ? -1 : ans) << endl;
    return 0;
}
