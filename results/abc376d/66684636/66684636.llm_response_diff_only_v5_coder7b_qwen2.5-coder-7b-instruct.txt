```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector<int> g[N];
int n, m, ans = INT_MAX;

void dfs(int u, int cnt)
{
    vis[u] = true;
    if (cnt >= ans) return;
    if (u == 1 && cnt != 0)
    {
        ans = min(ans, cnt);
        return;
    }
    for (int v : g[u])
    {
        if (v != 1 && vis[v]) continue;
        dfs(v, cnt + 1);
    }
    vis[u] = false; // Backtrack to allow other paths through this node
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    dfs(1, 0);
    if (ans != INT_MAX) cout << ans;
    else cout << -1;
    return 0;
}
```

### Explanation of Changes
1. **Initialization of `ans`**: Changed from `1e18` to `INT_MAX` for better readability and to handle overflow issues more gracefully.
2. **Backtracking**: Added `vis[u] = false;` at the end of the DFS function to backtrack and allow other paths through the current node.
3. **Edge Case Handling**: The condition `if (u == 1 && cnt != 0)` ensures that we only consider valid paths that reach node 1 with a non-zero count, as starting from node 1 without any steps is not considered a valid path.
