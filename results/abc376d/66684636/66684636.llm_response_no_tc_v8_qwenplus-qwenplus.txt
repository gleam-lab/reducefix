#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m;

int bfs()
{
    queue<pair<int, int>> q;
    vector<int> dist(n + 1, -1);
    dist[1] = 0;

    for (int v : g[1])
    {
        q.push({v, 1});
        dist[v] = 1;
    }

    while (!q.empty())
    {
        auto [u, d] = q.front();
        q.pop();

        if (u == 1)
            return d;

        for (int v : g[u])
        {
            if (dist[v] == -1 || (v == 1 && d + 1 < dist[v]))
            {
                dist[v] = d + 1;
                q.push({v, d + 1});
            }
        }
    }

    return -1;
}

signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs() << endl;

    return 0;
}
