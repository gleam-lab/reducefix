const int N = 2e5 + 5;
const int INF = 1e18;

vector<int> g[N];
int dist[N];
int n, m, ans = INF;

void bfs() {
    queue<pair<int, int>> q;
    q.push({1, 0});
    dist[1] = 0;
    
    while (!q.empty()) {
        auto [u, d] = q.front();
        q.pop();
        
        for (int v : g[u]) {
            if (v == 1) {
                ans = min(ans, d + 1);
                continue;
            }
            if (dist[v] > d + 1) {
                dist[v] = d + 1;
                q.push({v, d + 1});
            }
        }
    }
}

signed main() {

    for (int i = 1; i <= m; i++) {

        cin >> u >> v;
        g[u].push_back(v);
    }
    
    fill(dist, dist + N, INF);
    bfs();
    
    if (ans != INF) cout << ans;
    else cout << -1;
    
    return 0;
}
