```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector<int> g[N];
int n, m, ans = 1e18;

void dfs(int u, int cnt) {
    vis[u] = true;
    if (cnt > ans) return;
    if (u == 1 && cnt != 0) {
        ans = min(ans, cnt);
        return;
    }
    for (int v : g[u]) {
        if (!vis[v]) dfs(v, cnt + 1);
    }
    vis[u] = false; // Backtrack to allow other paths to explore this node
}

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    dfs(1, 0);
    if (ans != (int)1e18) cout << ans;
    else cout << -1;
    return 0;
}
```
This solution corrects the logic of the depth-first search (DFS) function. It ensures that nodes can be revisited if necessary by resetting their visit status after exploring all their children. This is crucial for finding the shortest path back to node 1 from any other node in the graph. The condition `if (cnt > ans)` is used to early exit the DFS when the current path length exceeds the best-known answer, optimizing performance.
