```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m, ans = -1;

void solve()
{
    // We're looking for the shortest cycle that starts and ends at node 1
    vector<int> dist(n + 1, -1);
    queue<int> q;
    
    q.push(1);
    dist[1] = 0;

    while (!q.empty())
    {
        int u = q.front();
        q.pop();

        for (int v : g[u])
        {
            if (dist[v] == -1)
            {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
            else if (v != 1 && u != 1)
            {
                // Found a cycle back to node 1 through another path
                ans = dist[u] + dist[v] + 1;
                return;
            }
        }
    }
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    solve();

    cout << ans << endl;
    return 0;
}
```
