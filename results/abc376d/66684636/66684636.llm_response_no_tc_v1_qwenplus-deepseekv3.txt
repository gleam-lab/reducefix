int findShortestCycle() {

    for (int start = 1; start <= n; ++start) {
        vector<int> dist(n + 1, INF);
        vector<int> parent(n + 1, -1);
        queue<int> q;
        q.push(start);
        dist[start] = 0;

            q.pop();

                if (dist[v] == INF) {
                    dist[v] = dist[u] + 1;

        }
    }

}
