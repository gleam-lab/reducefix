#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m;

// BFS is more appropriate for finding shortest cycle
int findShortestCycle()
{
    int ans = 1e18;
    
    for (int start = 1; start <= n; ++start)
    {
        vector<int> dist(n + 1, -1);
        vector<int> parent(n + 1, -1);
        queue<int> q;
        
        dist[start] = 0;
        q.push(start);
        
        while (!q.empty())
        {
            int u = q.front();
            q.pop();
            
            for (int v : g[u])
            {
                if (dist[v] == -1)
                {
                    dist[v] = dist[u] + 1;
                    parent[v] = u;
                    q.push(v);
                }
                else if (parent[u] != v && parent[v] != u)
                {
                    // Found a cycle that's not just back-and-forth
                    int cycleLength = dist[u] + dist[v] + 1;
                    if (cycleLength >= 3) // Valid cycle must have at least 3 nodes
                        ans = min(ans, cycleLength);
                }
            }
        }
    }
    
    return (ans == 1e18) ? -1 : ans;
}

signed main()
{
    cin >> n >> m;
    
    for (int i = 1; i <= m; ++i)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Since it's an undirected graph
    }
    
    cout << findShortestCycle();
    
    return 0;
}
