#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector<int> g[N];
int n, m, ans = 1e18;

void bfs(int start, int &ans)
{
    queue<int> q;
    q.push(start);
    vis[start] = true;
    int depth = 0;
    while (!q.empty())
    {
        int u = q.front();
        q.pop();
        depth++;
        for (int v : g[u])
        {
            if (!vis[v])
            {
                vis[v] = true;
                q.push(v);
            }
        }
    }
    ans = min(ans, depth);
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Ensure undirected graph
    }
    fill(vis, vis + N, false);
    ans = 1e18;
    for (int i = 1; i <= n; i++)
    {
        if (!vis[i])
        {
            bfs(i, ans);
        }
    }
    if (ans != 1e18) cout << ans;
    else cout << -1;
    return 0;
}
