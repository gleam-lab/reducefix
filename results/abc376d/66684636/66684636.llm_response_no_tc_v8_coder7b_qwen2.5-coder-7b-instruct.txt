```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector <int> g[N];
int n, m, ans = 1e18;

void dfs(int u, int cnt)
{
    vis[u] = true;
    if (cnt >= ans) return;
    if (u == 1 && cnt != 0)
    {
        ans = min(ans, cnt);
        return;
    }
    for (int v : g[u])
    {
        if (!vis[v])
        {
            dfs(v, cnt + 1);
        }
    }
    vis[u] = false; // Backtrack
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    dfs(1, 0);
    if (ans != (int)1e18) cout << ans;
    else cout << -1;
    return 0;
}
```

### Explanation of Fixes:
1. **Backtracking**: Added `vis[u] = false;` after the recursive call to ensure that other paths can explore this node again.
2. **Edge Case Handling**: Ensured that the function correctly handles cases where there is no valid path from the source to the target node (i.e., when `ans` remains unchanged).
