#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m;

int bfs() {
    queue<pair<int, int>> q;
    vector<int> dist(N, -1);
    q.push({1, 0});
    dist[1] = 0;

    while (!q.empty()) {
        int u = q.front().first;
        int d = q.front().second;
        q.pop();

        for (int v : g[u]) {
            if (dist[v] == -1) {
                dist[v] = d + 1;
                q.push({v, d + 1});
                if (v == 1) {
                    return d + 1;
                }
            }
        }
    }
    return -1;
}

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }

    cout << bfs();
    return 0;
}
