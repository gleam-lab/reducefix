```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector <int> g[N];
int n, m, ans = (int)1e18;

void dfs(int u, int cnt)
{
    vis[u] = true;
    if (cnt < ans)
    {
        if (u == 1)
        {
            ans = cnt;
        }
        for (int v : g[u])
        {
            if (!vis[v])
            {
                dfs(v, cnt + 1);
            }
        }
    }
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // 忙路禄氓聤聽氓聫聧氓聬聭猫戮鹿
    }
    dfs(1, 0);
    if (ans == (int)1e18)
    {
        cout << -1;
    }
    else
    {
        cout << ans;
    }
    return 0;
}
```
