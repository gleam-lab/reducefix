```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector<int> g[N];
int n, m, ans = INT_MAX;

void dfs(int u, int cnt)
{
    vis[u] = true;
    if (cnt >= ans) return;
    if (u == 1 && cnt != 0)
    {
        ans = min(ans, cnt);
        return;
    }
    for (int v : g[u])
    {
        if (v != 1 && vis[v]) continue;
        dfs(v, cnt + 1);
    }
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    memset(vis, false, sizeof(vis));
    dfs(1, 0);
    if (ans != INT_MAX) cout << ans;
    else cout << -1;
    return 0;
}
```

### Explanation of Fixes
1. **Initialization of `ans`**: Changed `ans = 1e18` to `ans = INT_MAX`. This is more appropriate because we are looking for the minimum number of steps, and `INT_MAX` ensures that any valid path will be smaller.
2. **Resetting Visited Array**: Added `memset(vis, false, sizeof(vis));` before calling `dfs` to ensure that the visited array is reset for each test case, preventing incorrect results from previous runs.
3. **Edge Case Handling**: The function correctly handles the case where there is no path from node 1 back to itself except through other nodes, returning `-1` in such cases.
