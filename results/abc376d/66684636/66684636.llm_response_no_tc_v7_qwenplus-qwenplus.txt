```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m;

// BFS is more suitable for finding shortest cycle
int bfs()
{
    vector<int> dist(n + 1, -1);
    queue<int> q;
    
    // Start from node 1
    q.push(1);
    dist[1] = 0;
    
    while (!q.empty())
    {
        int u = q.front();
        q.pop();
        
        for (int v : g[u])
        {
            // If not visited
            if (dist[v] == -1)
            {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
            // If it's a back edge to 1, check if it forms a cycle
            else if (v == 1)
            {
                return dist[u] + 1;
            }
        }
    }
    
    return -1; // No cycle found
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    
    cout << bfs();
    
    return 0;
}
```
