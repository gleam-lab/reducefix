
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 2e5 + 5;
vector<int> g[N];
bool vis[N];
int n, m, ans = 1e18;

void dfs(int u, int cnt) {
    vis[u] = true;
    if (cnt >= ans) return;
    for (int v : g[u]) {
        if (!vis[v]) {
            dfs(v, cnt + 1);
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    dfs(1, 0);
    if (ans == 1e18) cout << -1;
    else cout << ans;
    return 0;
}
```
