Here's the fixed code using BFS which is more suitable for finding shortest paths/cycles in unweighted graphs:

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;
const int INF = 1e18;

vector<int> g[N];
int dist[N];
int n, m, ans = INF;

void bfs() {
    queue<pair<int,int>> q;
    for(int v : g[1]) {
        q.push({v, 1});
        dist[v] = 1;
    }
    
    while(!q.empty()) {
        auto [u, d] = q.front();
        q.pop();
        
        for(int v : g[u]) {
            if(v == 1) {
                ans = min(ans, d + 1);
                continue;
            }
            if(dist[v] > d + 1) {
                dist[v] = d + 1;
                q.push({v, d + 1});
            }
        }
    }
}

signed main() {
    cin >> n >> m;
    for(int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    
    fill(dist, dist + n + 1, INF);
    bfs();
    
    if(ans != INF) cout << ans;
    else cout << -1;
    return 0;
}
```

Key changes made:
1. Replaced DFS with BFS since BFS naturally finds shortest paths in unweighted graphs
2. Maintained a distance array to track shortest path lengths from node 1
3. When we encounter node 1 during BFS, we update the answer with the cycle length
4. Only push nodes to the queue if we find a shorter path to them
5. Initialized distances to INF to represent unreachable nodes

This approach will correctly find the shortest cycle containing node 1 in O(N + M) time complexity, which is optimal for this problem. The BFS ensures we find the shortest cycle first before longer ones, unlike DFS which might find longer cycles first.
