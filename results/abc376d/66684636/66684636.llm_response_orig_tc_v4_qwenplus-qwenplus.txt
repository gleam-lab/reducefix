#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

vector<int> g[N];
int n, m, ans = -1;

// BFS-based approach to find the shortest cycle starting from node 1
void bfs(int start) {
    queue<pair<int, int>> q;  // {node, distance}
    vector<int> dist(n + 1, -1);
    vector<int> parent(n + 1, -1);

    q.push({start, 0});
    dist[start] = 0;

    while (!q.empty()) {
        auto [u, d] = q.front();
        q.pop();

        for (int v : g[u]) {
            if (dist[v] == -1) {  // not visited
                dist[v] = d + 1;
                parent[v] = u;
                q.push({v, d + 1});
            } else if (parent[u] != v && parent[v] != u) {  // avoid immediate parent and child check
                // Found a cycle
                int cycle_len = d + dist[v] + 1;
                if (ans == -1 || cycle_len < ans)
                    ans = cycle_len;
            }
        }
    }
}

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);  // since graph is undirected
    }

    bfs(1);

    cout << ans << endl;
    return 0;
}
