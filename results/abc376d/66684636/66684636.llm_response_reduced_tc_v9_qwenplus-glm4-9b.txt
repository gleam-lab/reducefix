#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 2e5 + 5;

bool vis[N];
vector <int> g[N];
int n, m, ans = 1e18;

void dfs(int u, int cnt)
{
    vis[u] = true;
    if (cnt > ans)	return;
    for (int v : g[u])
    {
        if (v != 1 && vis[v])	continue;
        dfs(v, cnt + 1);
    }
}

void bfs(int start) {
    queue<int> q;
    q.push(start);
    vis[start] = true;
    int dist = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : g[u]) {
            if (!vis[v]) {
                vis[v] = true;
                q.push(v);
                if (v == 1) {
                    ans = min(ans, dist); // Update answer with current level distance to node 1
                }
            }
        }
        dist++;
    }
}

signed main()
{
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vis[1] = true; // Start BFS from node 1 if it exists
    bfs(1);

    if (ans == 1e18) cout << -1;
    else cout << ans;
    return 0;
}
