const int N = 2e5 + 5;

bool vis[N], inStack[N];

void dfs(int u, int cnt) {
    vis[u] = true;
    inStack[u] = true;
    for (int v : g[u]) {
        if (!vis[v]) {

        } else if (inStack[v]) {

        }
    }
    inStack[u] = false;
}
