const int N = 2e5 + 5;
const int INF = 1e18;

vector<int> g[N];
int dist[N];
int parent[N];
int n, m, ans = INF;

void bfs(int start) {
    queue<int> q;
    q.push(start);
    dist[start] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : g[u]) {
            if (v == start && dist[u] + 1 >= 3) {
                ans = min(ans, dist[u] + 1);
                continue;
            }
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                q.push(v);
            } else if (parent[u] != v && parent[v] != u) {
                ans = min(ans, dist[u] + dist[v] + 1);
            }
        }
    }
}

signed main() {

    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u); // Assuming undirected graph for cycle detection
    }
    
    memset(dist, -1, sizeof(dist));

    bfs(1);
    
    if (ans != INF) {
