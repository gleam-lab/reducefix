```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 200005;
ll n, m, mn = 200005, vis[MAXN];
vector<int> b[MAXN];

void dfs(int x, int cnt) {
    vis[x] = 1;
    ll ans = cnt + 1;
    bool found = false;
    for (int i = 0; i < b[x].size(); i++) {
        if (!vis[b[x][i]]) {
            dfs(b[x][i], ans);
            found = true;
        }
    }
    if (!found && cnt > 0) {
        mn = min(mn, cnt);
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
        b[v].push_back(u); // Add this line to make it an undirected graph
    }
    dfs(1, 0);
    if (mn == 200005) {
        cout << -1;
    } else {
        cout << mn;
    }
    return 0;
}
```
