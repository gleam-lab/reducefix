
```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll n,m,mn=200005,vis[200005];
vector<int> b[200005];

void dfs(int x, int cnt) {
    vis[x] = 1;
    ll ans = cnt + 1;
    for (int i = 0; i < b[x].size(); i++) {
        if (!vis[b[x][i]]) {
            if (b[x][i] == 1) { // Assuming '1' represents a special node
                mn = min(mn, ans);
                return;
            } else {
                dfs(b[x][i], ans);
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
        if (v != 1 && u != 1 && !b[1].empty()) { // Ensure we don't add self-loops or nodes to themselves
            b[v].push_back(u); // Ensure bidirectional edge
        }
    }
    dfs(1, 0);
    if (mn == 200005) {
        cout << "-1";
    } else {
        cout << mn;
    }
    return 0;
}
```
