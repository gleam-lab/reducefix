
```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll n, m, mn=200005, vis[200005];
vector<int> b[200005];
int ans = 0;

void dfs(int x, int cnt) {
    vis[x] = 1;
    ll current = cnt + 1;
    for (int i = 0; i < b[x].size(); i++) {
        if (!vis[b[x][i]]) {
            dfs(b[x][i], current);
            current = mn; // Reset current to mn after exploring a child
        }
    }
    ans = min(ans, current);
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
    }
    dfs(1, 0);
    if (mn == 200005) {
        cout << -1;
    } else {
        cout << ans;
    }
    return 0;
}
```
