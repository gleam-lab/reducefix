```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 200005;

vector<int> adj[MAXN];
bool visited[MAXN];

// Depth-first search function to find the minimum cycle length starting from node 'start'
void dfs(int start, int parent, int depth) {
    visited[start] = true;
    for (int neighbor : adj[start]) {
        if (!visited[neighbor]) {
            dfs(neighbor, start, depth + 1);
        } else if (neighbor != parent) { // Check if it's not the parent to avoid counting back edges
            mn = min(mn, depth - visited[neighbor] + 1); // Calculate cycle length
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // Assuming undirected graph
    }

    memset(visited, false, sizeof(visited));
    int mn = INT_MAX;

    // Iterate through all nodes to handle disconnected components
    for (int i = 1; i <= n; ++i) {
        if (!visited[i]) {
            dfs(i, -1, 0); // -1 indicates no parent initially
        }
    }

    if (mn == INT_MAX) {
        cout << -1 << endl; // No cycle found
    } else {
        cout << mn << endl;
    }

    return 0;
}
```
This solution addresses several issues:
1. It correctly handles undirected graphs by adding both directions of each edge.
2. It uses a more general approach to DFS that works for any connected component of the graph.
3. It initializes `mn` with `INT_MAX` to ensure any valid cycle length will be smaller.
4. It checks for cycles by ensuring the neighbor is not the parent when revisiting nodes.
