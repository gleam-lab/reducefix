```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 2e5 + 5;
ll n, m, mn = 200005, vis[MAXN];
vector<int> b[MAXN];

void dfs(int x, int cnt) {
    vis[x] = 1;
    ll ans = cnt + 1;
    bool has_one = false;
    for (int i = 0; i < b[x].size(); i++) {
        if (!vis[b[x][i]]) {
            dfs(b[x][i], ans);
        } else if (b[x][i] == 1) {
            has_one = true;
        }
    }
    if (has_one) {
        mn = min(mn, ans);
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
        if (v == 1) {
            b[v].push_back(u); // Make the graph undirected
        }
    }
    dfs(1, 0);
    if (mn == 200005) {
        cout << -1;
    } else {
        cout << mn;
    }
    return 0;
}
```

This solution ensures that the graph is treated as undirected by adding an edge from `v` to `u` whenever an edge from `u` to `v` is read. This adjustment helps in correctly identifying cycles involving the node `1`. The DFS function now checks for nodes already visited to prevent redundant processing and uses a boolean flag to track if any path leads back to node `1`, updating the minimum cycle length accordingly.
