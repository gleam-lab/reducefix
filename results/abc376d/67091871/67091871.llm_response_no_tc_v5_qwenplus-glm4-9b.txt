
```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int n, m;
const int MAXN = 200005;
int vis[MAXN];
vector<int> adj[MAXN];
int mn = INT_MAX;

void dfs(int node, int depth) {
    vis[node] = 1;
    for (int neighbor : adj[node]) {
        if (!vis[neighbor]) {
            dfs(neighbor, depth + 1);
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    // Start DFS from node 1, with initial depth of 0
    dfs(1, 0);
    
    // If no valid path found, output -1
    if (mn == INT_MAX) {
        cout << -1;
    } else {
        cout << mn - 1; // Subtract 1 because depth starts from 0
    }
    return 0;
}
```
