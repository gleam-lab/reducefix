I analyzed the problem and your code. The issue is that your DFS approach doesn't guarantee finding the shortest cycle containing node 1 - it just finds the first cycle it encounters. For the given input, there's actually a shorter cycle (1→2→4→5→1) of length 4, but your code finds a longer one first.

Here's the corrected solution using BFS which is more suitable for shortest path problems:

```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main(){
    ll n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n+1);
    for(int i=0; i<m; i++){
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    ll min_cycle = LONG_MAX;
    vector<ll> dist(n+1, -1);
    queue<pair<int,int>> q; // {node, parent}

    // Initialize BFS from node 1
    q.push({1, -1});
    dist[1] = 0;

    while(!q.empty()){
        auto [u, parent] = q.front();
        q.pop();

        for(int v : adj[u]){
            if(v == 1 && u != parent){ // Found a cycle back to 1
                min_cycle = min(min_cycle, dist[u] + 1);
            }
            if(dist[v] == -1 || (v == 1 && dist[u] + 1 < dist[v])){
                dist[v] = dist[u] + 1;
                q.push({v, u});
            }
        }
    }

    if(min_cycle == LONG_MAX){
        cout << -1;
    } else {
        cout << min_cycle;
    }
    return 0;
}
```

Key improvements:
1. Uses BFS instead of DFS to find shortest paths
2. Properly tracks distances and updates them
3. Specifically checks for cycles back to node 1
4. Handles the case where no cycle exists
5. Uses more standard C++ practices (vector instead of raw arrays)

The BFS approach is more appropriate here because:
- It naturally finds shortest paths in unweighted graphs
- We can stop early if we find a cycle back to node 1
- It avoids the issues of DFS getting stuck in long paths before finding shorter ones
