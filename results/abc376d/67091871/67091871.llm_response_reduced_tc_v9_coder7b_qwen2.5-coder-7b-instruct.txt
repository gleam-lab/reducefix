```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int MAXN = 200005;
int n, m, mn = INT_MAX, vis[MAXN];
vector<int> b[MAXN];

void dfs(int x, int cnt) {
    vis[x] = 1;
    ll ans = cnt + 1;
    bool hasOne = false;
    for (int i = 0; i < b[x].size(); i++) {
        if (!vis[b[x][i]]) {
            dfs(b[x][i], ans);
        } else if (b[x][i] == 1) {
            hasOne = true;
        }
    }
    if (hasOne) {
        mn = min(mn, ans);
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
        if (v == 1) {
            b[v].push_back(u); // Make sure to add the reverse edge
        }
    }
    dfs(1, 0);
    if (mn == INT_MAX) {
        cout << -1;
    } else {
        cout << mn;
    }
    return 0;
}
```
