#define ll long long

ll n, m, mn = 200005;

void bfs(int start) {
    queue<int> q;
    q.push(start);
    dist[start] = 0;
    
    while(!q.empty()) {
        int u = q.front();
        q.pop();
        
        for(int v : b[u]) {
            if(v == start) {
                mn = min(mn, dist[u] + 1);
            }
            if(dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

}
