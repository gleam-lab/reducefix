#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll n, m;
vector<ll> adj[200005];
ll dist[200005];

void bfs() {
    queue<ll> q;
    for (ll i = 1; i <= n; ++i) {
        dist[i] = -1;
    }
    dist[1] = 0;
    q.push(1);
    
    while (!q.empty()) {
        ll u = q.front();
        q.pop();
        for (ll v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
                if (v == 1) {
                    cout << dist[v];
                    exit(0);
                }
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (ll i = 0; i < m; ++i) {
        ll u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    bfs();
    
    cout << -1 << endl;
    return 0;
}
