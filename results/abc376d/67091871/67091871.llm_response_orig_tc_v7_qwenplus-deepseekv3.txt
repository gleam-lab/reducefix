Here's the corrected C++ code that uses BFS (Breadth-First Search) instead of DFS, which is more appropriate for finding shortest paths in unweighted graphs:

```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll n, m, mn = LLONG_MAX;
vector<vector<int>> adj;
vector<int> dist;
vector<int> parent;

void bfs(int start) {
    queue<int> q;
    q.push(start);
    dist[start] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                q.push(v);
            } else if (v == start && parent[u] != start) {
                mn = min(mn, dist[u] + 1);
            }
        }
    }
}

int main() {
    cin >> n >> m;
    adj.resize(n + 1);
    dist.assign(n + 1, -1);
    parent.assign(n + 1, -1);
    
    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    bfs(1);
    
    if (mn == LLONG_MAX) {
        cout << -1;
    } else {
        cout << mn;
    }
    
    return 0;
}
```

Key improvements:
1. Switched from DFS to BFS for shortest path finding
2. Properly initialized distance and parent arrays
3. Added cycle detection when revisiting the start node
4. Used proper data structures and initialization
5. Handled the case when no cycle is found (output -1)

The BFS approach is more suitable here because:
1. It naturally finds shortest paths in unweighted graphs
2. It's more efficient for this problem (O(N + M) complexity)
3. It correctly handles the cycle detection by checking when we can return to the start node via a different path
