```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 2e5 + 5;
vector<int> adj[MAXN];
bool vis[MAXN];

// Function to perform DFS and find the minimum distance to node 1
void dfs(int node, int parent, int &minDist) {
    vis[node] = true;
    if(node != 1 && adj[node].size() == 1) { // If it's not the root and has only one neighbor, it's a leaf
        minDist = min(minDist, abs(parent - node));
        return;
    }
    for(auto child : adj[node]) {
        if(!vis[child]) {
            dfs(child, node, minDist);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int n, m;
    cin >> n >> m;
    
    for(int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    int minDist = INT_MAX;
    memset(vis, false, sizeof(vis));
    dfs(1, -1, minDist);

    if(minDist == INT_MAX) {
        cout << "-1";
    } else {
        cout << minDist;
    }

    return 0;
}
```
