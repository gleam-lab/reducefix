```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int MAXN = 200005;

ll n, m, mn = INT_MAX;
bool vis[MAXN];
vector<int> adj[MAXN];

void dfs(int node, int depth) {
    vis[node] = true;
    if (depth < mn) mn = depth;
    for (int neighbor : adj[node]) {
        if (!vis[neighbor]) {
            dfs(neighbor, depth + 1);
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // Assuming undirected graph
    }

    dfs(1, 0);

    if (mn == INT_MAX) {
        cout << -1;
    } else {
        cout << mn;
    }

    return 0;
}
```
