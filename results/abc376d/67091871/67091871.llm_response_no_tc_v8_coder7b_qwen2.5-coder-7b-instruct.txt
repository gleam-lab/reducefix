#include<bits/stdc++.h>
#define ll long long
using namespace std;

const int N = 200005;
vector<int> adj[N];
bool vis[N];
int mn = INT_MAX;

void dfs(int node, int dist) {
    if (dist > mn) return; // Early exit if current distance exceeds minimum found so far
    vis[node] = true;
    for (int neighbor : adj[node]) {
        if (!vis[neighbor]) {
            dfs(neighbor, dist + 1);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // Assuming undirected graph
    }

    dfs(1, 0); // Start DFS from node 1 with initial distance 0

    if (mn == INT_MAX) {
        cout << -1 << '\n';
    } else {
        cout << mn << '\n';
    }

    return 0;
}

1. **Edge Case Handling**: Added an early exit condition in `dfs` to stop further exploration if the current path length exceeds the minimum distance already found (`mn`). This prevents unnecessary computations.
2. **Graph Representation**: Ensured that the adjacency list is correctly set up assuming the graph is undirected.
3. **Initialization**: Used `INT_MAX` instead of a large number like `200005` for initializing `mn`, as it better represents the unvisited state.
4. **Efficiency**: Improved the efficiency by breaking out of the loop early when a longer path is detected.
5. **Code Readability**: Enhanced readability by using `ios_base::sync_with_stdio(false); cin.tie(NULL);` at the beginning of the `main` function to optimize input operations.
