```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll n, m, mn = 200005, vis[200005];
vector<int> b[200005];

void dfs(int x, int cnt) {
    if (vis[x] == 1) return; // Prevent revisiting processed nodes
    vis[x] = 1;
    ll currentMn = cnt + 1;
    for (int i = 0; i < b[x].size(); ++i) {
        if (vis[b[x][i]] == 0 || b[x][i] == 1) {
            if (b[x][i] == 1) {
                mn = min(mn, currentMn);
                return;
            } else {
                dfs(b[x][i], currentMn);
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        b[u].push_back(v);
        b[v].push_back(u); // Ensure bidirectional adjacency
    }
    
    // Initialize all nodes as unvisited
    for (int i = 1; i <= n; ++i) {
        vis[i] = 0;
    }
    
    dfs(1, 0);
    
    if (mn == 200005) {
        cout << -1;
    } else {
        cout << mn;
    }
    return 0;
}
```

This code fixes the problem by:
1. Ensuring bidirectional adjacency to avoid separate operations for each direction.
2. Adding a check to prevent revisiting nodes that have already been processed.
3. Ensure that the minimum is not set to a default value that won't be reached.
4. Properly initializing all nodes as unvisited before the DFS starts.
