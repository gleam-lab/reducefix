```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll n, m;
vector<int> adj[200005];

// BFS is more appropriate for shortest path in unweighted graph
ll bfs() {
    queue<pair<ll, ll>> q;  // {node, distance}
    q.push({1, 0});
    
    vector<ll> vis(200005, 0);
    vis[1] = 1;
    
    while (!q.empty()) {
        ll node = q.front().first;
        ll dist = q.front().second;
        q.pop();
        
        if (node == n) return dist;
        
        for (auto it : adj[node]) {
            if (!vis[it]) {
                vis[it] = 1;
                q.push({it, dist + 1});
            }
        }
    }
    return -1;  // no path found
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    cin >> n >> m;
    
    for (ll i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    cout << bfs() << "\n";
    
    return 0;
}
```
