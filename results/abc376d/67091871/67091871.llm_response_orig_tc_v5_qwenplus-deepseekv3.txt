Here's the fixed code using BFS for shortest path finding:

```cpp
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    ll n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n+1);
    for(int i=0; i<m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    queue<pair<int,int>> q;
    vector<int> dist(n+1, -1);
    
    // Start BFS from node 1
    q.push({1, 0});
    dist[1] = 0;
    int min_cycle = INT_MAX;
    
    while(!q.empty()) {
        auto [u, d] = q.front();
        q.pop();
        
        for(int v : adj[u]) {
            if(v == 1) {
                // Found a cycle back to node 1
                min_cycle = min(min_cycle, d+1);
                continue;
            }
            if(dist[v] == -1) {
                dist[v] = d+1;
                q.push({v, d+1});
            }
        }
    }
    
    if(min_cycle == INT_MAX) {
        cout << -1 << endl;
    } else {
        cout << min_cycle << endl;
    }
    
    return 0;
}
```

Key improvements:
1. Switched from DFS to BFS which is more suitable for finding shortest paths
2. Maintained a distance array to track the shortest path lengths
3. When we find a direct edge back to node 1, we immediately check if it forms a shorter cycle
4. The BFS ensures we find the shortest cycle first since it explores nodes level by level
5. Removed unnecessary global variables and simplified the code structure

This approach has a time complexity of O(N + M) which is optimal for this problem and will correctly find the shortest cycle starting and ending at node 1.
