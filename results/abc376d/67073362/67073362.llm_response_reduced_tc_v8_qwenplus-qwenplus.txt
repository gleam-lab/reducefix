const int N = 2e5 + 10, INF = 1e9 + 7;
vector<int> a[N];

void solve(){
    int n, m;
    cin >> n >> m;
    for(int i = 0; i < m; i++){
        int u, v;
        cin >> u >> v;
        a[u].push_back(v);
    }

    // BFS from node 1
    vector<int> dist(n + 1, -1);
    queue<int> q;
    q.push(1);
    dist[1] = 0;

    while(!q.empty()){
        int u = q.front();
        q.pop();
        for(auto v : a[u]){
            if(dist[v] == -1){
                dist[v] = dist[u] + 1;
                q.push(v);
                if(v == 1){
                    cout << dist[v];
                    return;
                }
            }
        }
    }

    cout << -1;
}

signed main(){
    IOS;
    int T = 1;
    while(T--){
        solve();
    }
    return 0;
