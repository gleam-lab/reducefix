```cpp
#include <bits/stdc++.h>
using namespace std;

#define IOS ios::sync_with_stdio(false); cin.tie(0); cout.tie(0)
#define int long long

const int MAXN = 2e5 + 5;

int n, m;
vector<int> adj[MAXN];
int visited[MAXN];
int min_cycle = -1;

void dfs(int u, int depth, int parent) {
    visited[u] = 1; // visiting
    for (int v : adj[u]) {
        if (visited[v] == 0) {
            dfs(v, depth + 1, u);
            if (min_cycle != -1) return; // already found
        } else if (visited[v] == 1 && v != parent) {
            // cycle detected
            min_cycle = depth;
            return;
        }
    }
    visited[u] = 2; // visited
}

void solve() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    for (int i = 1; i <= n; ++i) {
        if (visited[i] == 0) {
            dfs(i, 1, -1);
            if (min_cycle != -1) break;
        }
    }

    cout << min_cycle << endl;
}
#undef int
int main() {
    IOS;
    solve();
    return 0;
}
```
