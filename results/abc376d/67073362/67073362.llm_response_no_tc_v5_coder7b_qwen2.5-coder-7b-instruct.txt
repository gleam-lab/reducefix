using namespace std;

const int N = 2e5 + 10, M = 2e6 + 10, INF = 1e9 + 7, P = 998244353;
vector<int> a[N];
bool visited[N];

void dfs(int u, int depth, int& min_edges) {
    visited[u] = true;
    for(auto v : a[u]) {
        if(!visited[v]) {
            dfs(v, depth + 1, min_edges);
        } else if(v == 1 && depth > 1) { // Cycle found and it does not include itself
            min_edges = min(min_edges, depth);
        }
    }
}

void solve() {
    int n, m;
    cin >> n >> m;
    for(int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        a[u].pb(v);
    }

    memset(visited, false, sizeof(visited));
    int min_edges = INF;

    for(int i = 1; i <= n; ++i) {
        if(!visited[i]) {
            dfs(i, 1, min_edges);
        }
    }

    if(min_edges == INF)
        cout << -1;
    else
        cout << min_edges;
}

signed main() {
    IOS;
    int T = 1;
    while(T--) {
        solve();
    }
    return 0;
}

This solution introduces a DFS approach to detect cycles starting from vertex 1. It uses an array `visited` to keep track of visited nodes during the traversal. When a back-edge is encountered (i.e., a node is revisited), it checks if the current path length is greater than 1 (to ensure we're not counting a self-loop as part of the cycle). If a cycle is detected, it updates `min_edges` with the minimum path length. This ensures that only cycles containing more than one edge are considered, adhering to the problem's requirements.
