After analyzing the code, I found several issues that need to be fixed:

1. The main problem is finding the shortest cycle starting and ending at node 1 in a directed graph.
2. The current approach using DFS doesn't correctly track the shortest path because it doesn't properly handle visited nodes and path tracking.
3. The global variable `mi` is being modified without proper reset between test cases.
4. The `g` array is being used both as a visited marker and for counting, which causes incorrect behavior.

Here's the corrected version using BFS (more suitable for shortest path problems) with proper tracking of visited nodes and path lengths:

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<climits>
using namespace std;

const int N = 2e5 + 10;
const int INF = INT_MAX;

vector<int> adj[N];

void solve() {
    int n, m;
    cin >> n >> m;
    
    // Clear the adjacency list
    for(int i = 0; i <= n; i++) {
        adj[i].clear();
    }
    
    // Read edges
    for(int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INF;
    queue<pair<int, int>> q;
    vector<int> dist(n + 1, INF);
    
    // Start BFS from node 1
    q.push({1, 1});
    dist[1] = 1;
    
    while(!q.empty()) {
        auto [u, d] = q.front();
        q.pop();
        
        for(int v : adj[u]) {
            if(v == 1) {
                // Found a cycle back to node 1
                min_cycle = min(min_cycle, d);
                continue;
            }
            if(dist[v] > d + 1) {
                dist[v] = d + 1;
                q.push({v, d + 1});
            }
        }
    }
    
    if(min_cycle == INF) {
        cout << -1 << endl;
    } else {
        cout << min_cycle << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T = 1;
    // cin >> T;
    while(T--) {
        solve();
    }
    
    return 0;
}
```

Key improvements:
1. Uses BFS instead of DFS for finding shortest paths
2. Properly tracks distances using a distance array
3. Clears the adjacency list between test cases (though currently only one test case is run)
4. Uses a more standard approach for graph representation
5. Simplifies the code by removing unnecessary macros and includes
6. Properly handles the case when no cycle is found (outputs -1)

The algorithm now correctly finds the shortest cycle starting and ending at node 1 by performing a BFS and checking whenever we can return to node 1 from any neighbor.
