#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 10, INF = 1e9 + 7;

vector<vector<int>> graph(N);
bool visited[N];
int min_cycle_length = INF;

// BFS to detect shortest cycle starting from node 1
void bfs(int start) {
    vector<int> dist(N, INF);
    queue<int> q;
    q.push(start);
    dist[start] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : graph[u]) {
            if (dist[v] == INF) {
                dist[v] = dist[u] + 1;
                q.push(v);
            } else if (v == 1) { // Cycle back to node 1
                min_cycle_length = min(min_cycle_length, dist[u] + 1);
            }
        }
    }
}

void solve() {
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
    }
    
    bfs(1);
    
    cout << (min_cycle_length == INF ? -1 : min_cycle_length);
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    int T = 1;
    while (T--) {
        solve();
    }
    
    return 0;
}
