```cpp
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

const int N = 2e5 + 10;
vector<int> a[N];
int mi = INT_MAX, g[N];

void dfs(int u, int depth) {
    g[u] = depth;
    for (auto v : a[u]) {
        if (!g[v]) {
            dfs(v, depth + 1);
        } else if (v == 1) {
            mi = min(mi, depth + 1);
        }
    }
}

void solve() {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        a[u].push_back(v);
    }
    dfs(1, 0);
    if (mi == INT_MAX) {
        cout << -1 << endl;
    } else {
        cout << mi << endl;
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int T = 1;
    while (T--) {
        solve();
    }
    return 0;
}
```
