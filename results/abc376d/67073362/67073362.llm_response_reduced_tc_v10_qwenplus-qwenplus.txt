#include <bits/stdc++.h>
#define IOS ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define endl "\n"
#define PII pair<int, int>
using namespace std;
typedef long long LL;
const int N = 2e5 + 10, INF = 1e9 + 7;

vector<vector<int>> graph(N);
int dist[N];

void solve() {
    int n, m;
    cin >> n >> m;
    
    for(int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
    }
    
    // Initialize distances to infinity
    for(int i = 1; i <= n; ++i) {
        dist[i] = INF;
    }
    
    // BFS from node 1
    queue<int> q;
    q.push(1);
    dist[1] = 0;
    
    while(!q.empty()) {
        int u = q.front();
        q.pop();
        
        for(int v : graph[u]) {
            if(dist[v] > dist[u] + 1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    // The minimum cycle length is the distance back to node 1 from any neighbor of node 1
    int min_cycle = INF;
    for(int v : graph[1]) {
        if(dist[v] != INF) {
            min_cycle = min(min_cycle, dist[v] + 1);
        }
    }
    
    cout << (min_cycle == INF ? -1 : min_cycle) << endl;
}

signed main() {
    IOS;
    int T = 1;
    while(T--) {
        solve();
    }
    return 0;
}
