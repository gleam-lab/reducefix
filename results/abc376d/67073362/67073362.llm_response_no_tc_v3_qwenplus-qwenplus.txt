#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 10, INF = 1e9 + 7;

vector<vector<int>> graph(N);
bool visited[N];
int min_cycle_length = INF;

// BFS to find shortest cycle involving node 1
void bfs(int start) {
    queue<pair<int, int>> q; // {node, distance}
    vector<int> dist(N, INF);
    
    q.push({start, 0});
    dist[start] = 0;
    
    while (!q.empty()) {
        auto [u, d] = q.front(); q.pop();
        
        for (int v : graph[u]) {
            if (dist[v] == INF) {
                dist[v] = d + 1;
                q.push({v, d + 1});
            } 
            else if (v == start && d + 1 >= 3) { // Cycle found back to start
                min_cycle_length = min(min_cycle_length, d + 1);
            }
        }
    }
}

void solve() {
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
    }
    
    bfs(1);
    
    cout << (min_cycle_length == INF ? -1 : min_cycle_length) << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T = 1;
    while (T--) {
        solve();
    }
    
    return 0;
}
