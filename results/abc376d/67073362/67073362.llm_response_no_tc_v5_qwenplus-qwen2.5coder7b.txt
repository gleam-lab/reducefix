#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

const int MAXN = 2e5 + 10;

vector<int> adj[MAXN];
int depth[MAXN];

void dfs(int node, int parent, int d) {
    depth[node] = d;
    for (auto child : adj[node]) {
        if (child != parent) {
            dfs(child, node, d + 1);
        }
    }
}

int find_min_depth(int n) {
    dfs(1, -1, 0);
    int max_depth = 0;
    for (int i = 1; i <= n; ++i) {
        max_depth = max(max_depth, depth[i]);
    }
    return max_depth;
}

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    int result = find_min_depth(n);
    cout << result << endl;

    return 0;
}
