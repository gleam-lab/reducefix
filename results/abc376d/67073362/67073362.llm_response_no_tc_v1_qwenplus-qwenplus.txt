#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 10, INF = 1e9 + 7;

vector<vector<int>> adj;
int dist[N], in_queue[N];

void solve() {
    int n, m;
    cin >> n >> m;
    adj.assign(n + 1, vector<int>());
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    // BFS initialization
    queue<int> q;
    memset(dist, -1, sizeof(dist));
    dist[1] = 0;
    q.push(1);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    cout << (dist[1] == 0 ? -1 : dist[1]);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
