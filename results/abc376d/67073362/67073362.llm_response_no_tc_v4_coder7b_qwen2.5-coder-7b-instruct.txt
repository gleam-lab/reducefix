const int N = 2e5 + 10, M = 2e6 + 10, INF = 1e9 + 7, P = 998244353;
vector<int> a[N];
int vis[N], mi = INF, g[N];

bool dfs(int u, int deep){
    g[u] = deep;
    if(vis[u]) {
        mi = min(mi, deep - vis[u] + 1);
        return true;
    }
    vis[u] = deep;
    for(auto i: a[u]){
        if(dfs(i, deep + 1))
            return true;
    }
    vis[u] = 0;
    return false;
}

    }
    bool found = false;
    for(int i = 1; i <= n; i++){
        if(!vis[i] && dfs(i, 1)){
            found = true;
            break;
        }
    }

}

In the original code, there were a few issues:
1. The `dfs` function was not properly resetting visited nodes after exploring all paths from a node, which led to incorrect cycle detection.
2. The global variable `mi` should be reset between test cases if multiple cases are being handled in one run.
3. There was no check to ensure that a cycle containing vertex 1 was actually found before printing the result.
