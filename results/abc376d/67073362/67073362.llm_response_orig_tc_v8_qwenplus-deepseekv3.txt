The issue in the original code is that it uses a depth-first search (DFS) approach to find the shortest cycle back to node 1, which is not efficient and can miss the shortest path in certain cases. Instead, we should use a breadth-first search (BFS) approach, which is more suitable for finding shortest paths in unweighted graphs.

Here's the fixed code that uses BFS to find the shortest cycle starting and ending at node 1:

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<climits>
using namespace std;

const int N = 2e5 + 10;
vector<int> adj[N];
int dist[N];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int n, m;
    cin >> n >> m;
    
    for(int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    
    // Perform BFS from node 1 to find shortest cycle back to 1
    queue<pair<int, int>> q;
    q.push({1, 0});
    
    for(int i = 1; i <= n; i++) dist[i] = -1;
    dist[1] = 0;
    
    while(!q.empty()) {
        auto [u, d] = q.front();
        q.pop();
        
        for(int v : adj[u]) {
            if(v == 1) {
                min_cycle = min(min_cycle, d + 1);
                continue;
            }
            if(dist[v] == -1) {
                dist[v] = d + 1;
                q.push({v, d + 1});
            }
        }
    }
    
    if(min_cycle == INT_MAX) {
        cout << -1 << endl;
    } else {
        cout << min_cycle << endl;
    }
    
    return 0;
}
```

Key improvements:
1. Uses BFS instead of DFS to find the shortest cycle
2. Tracks distances from node 1 to all other nodes
3. Whenever we find a direct edge back to node 1, we update the minimum cycle length
4. Properly initializes and updates distance arrays
5. Handles the case where no cycle exists (outputs -1)

This approach ensures we find the shortest cycle containing node 1 (if one exists) in O(N + M) time complexity, which is optimal for this problem. The BFS guarantees that the first time we find a path back to node 1 will be the shortest possible path.
