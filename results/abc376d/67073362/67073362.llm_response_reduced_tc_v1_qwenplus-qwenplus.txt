#include<bits/stdc++.h>
#define IOS ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
using namespace std;

const int N = 2e5 + 10;
vector<int> adj[N];

// BFS to find shortest cycle starting from node 1
int findShortestCycle(int n) {
    vector<int> dist(n + 1, -1);
    queue<int> q;
    
    dist[1] = 0;
    q.push(1);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            } else if (v == 1) {
                // Found a cycle back to node 1
                return dist[u] + 1;
            }
        }
    }
    return -1; // No cycle found
}

void solve() {
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    cout << findShortestCycle(n) << endl;
}

signed main() {
    IOS
    int T = 1;
    while (T--) {
        solve();
    }
    return 0;
}
