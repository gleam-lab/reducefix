I analyzed the problem and your code. The issue is that your current DFS approach doesn't correctly track the shortest cycle from node 1 back to itself. The code marks nodes as visited (g[u]++) but doesn't properly handle the shortest path calculation. Here's the fixed version using BFS which is more suitable for shortest path problems:

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<climits>
using namespace std;

const int N = 2e5 + 10;
const int INF = INT_MAX;

vector<int> adj[N];
int dist[N];

void solve() {
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    fill(dist, dist + n + 1, INF);
    dist[1] = 0;
    
    queue<int> q;
    q.push(1);
    
    int min_cycle = INF;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : adj[u]) {
            if (v == 1) {
                min_cycle = min(min_cycle, dist[u] + 1);
            }
            if (dist[v] > dist[u] + 1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    
    cout << (min_cycle == INF ? -1 : min_cycle) << endl;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int T = 1;
    while (T--) {
        solve();
    }
    return 0;
}
```

Key improvements:
1. Replaced DFS with BFS which naturally finds shortest paths in unweighted graphs
2. Properly initialized distance array and used it to track shortest distances
3. Specifically checked for cycles back to node 1 (when v == 1)
4. Simplified the code structure and removed unnecessary includes/macros
5. Handled the case where no cycle exists (output -1)

This solution will correctly find the shortest cycle starting and ending at node 1, as required by the problem. The BFS approach ensures we find the shortest path first, which is crucial for this problem.
