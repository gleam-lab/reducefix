#include <bits/stdc++.h>
#define IOS ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define endl "\n"
#define PII pair<int, int>
using namespace std;
typedef long long LL;
const int N = 2e5 + 10, INF = 1e9 + 7;

vector<int> adj[N];
int dist[N];

void solve() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) adj[i].clear();
    
    while (m--) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    // BFS from node 1
    fill(dist, dist + n + 1, INF);
    queue<int> q;
    q.push(1);
    dist[1] = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : adj[u]) {
            if (dist[v] > dist[u] + 1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    if (dist[1] == INF)
        cout << -1;
    else
        cout << dist[1];
    cout << endl;
}

signed main() {
    IOS;
    int T = 1;
    // cin >> T;
    while (T--) solve();
    return 0;
}
