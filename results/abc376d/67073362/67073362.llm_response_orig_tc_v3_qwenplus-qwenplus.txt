#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9 + 7;

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    queue<pair<int, int>> q;
    vector<bool> visited(n + 1, false);
    q.push({1, 0});
    visited[1] = true;

    int ans = -1;
    while (!q.empty()) {
        auto [u, dist] = q.front();
        q.pop();

        if (u == 1 && dist > 0) {
            ans = dist;
            break;
        }

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                q.push({v, dist + 1});
            }
        }
    }

    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T = 1;
    //cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
