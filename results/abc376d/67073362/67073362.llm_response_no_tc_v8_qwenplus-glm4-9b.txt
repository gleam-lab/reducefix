const int N = 2e5 + 10, M = 2e6 + 10, INF = numeric_limits<int>::max(), P = 998244353;
vector<int> a[N];
int mi = INF, g[N];

void bfs(int start) {
    queue<int> q;
    q.push(start);
    g[start] = 1;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : a[u]) {
            if (!g[v]) {
                g[v] = 1;
                q.push(v);
            }
        }
    }
}

void solve() {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        a[u].push_back(v);
    }

    // Run BFS from node 1
    bfs(1);

    // Find the minimum distance to node 1 for all nodes
    set<int> distances;
    for (int i = 1; i <= n; i++) {
        if (g[i]) {
            distances.insert(g[i]);
        }
    }

    // If there are no nodes connected to node 1
    if (distances.empty()) {
        cout << -1 << endl;
    } else {
        cout << *distances.begin() << endl;
    }
}

int main() {
    IOS;
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
