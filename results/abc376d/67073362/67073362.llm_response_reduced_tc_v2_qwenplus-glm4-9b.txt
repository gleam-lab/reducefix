#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <tuple>
#include <cstring>
#include <memory>
using namespace std;

const int N = 200010;
vector<int> graph[N];
int minDepth = N + 1, visited[N];

void bfs(int start) {
    queue<pair<int, int>> q;
    q.push({start, 0});
    visited[start] = 1;
    while (!q.empty()) {
        auto [node, depth] = q.front();
        q.pop();
        for (int neighbor : graph[node]) {
            if (!visited[neighbor]) {
                visited[neighbor] = 1;
                q.push({neighbor, depth + 1});
                if (neighbor == 1) minDepth = min(minDepth, depth + 1);
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u);
    }
    bfs(1);
    if (minDepth == N + 1) {
        cout << -1 << endl;
    } else {
        cout << minDepth - 1 << endl;
    }
    return 0;
}
