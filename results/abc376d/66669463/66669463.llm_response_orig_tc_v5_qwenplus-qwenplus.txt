```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9 + 7;
int n, m;
vector<int> adj[200005];
int dist[200005];
bool inQueue[200005];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    // BFS from node 1
    fill(dist, dist + n + 1, INF);
    queue<int> q;
    dist[1] = 0;
    q.push(1);
    inQueue[1] = true;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        inQueue[u] = false;
        for (int v : adj[u]) {
            if (dist[v] > dist[u] + 1) {
                dist[v] = dist[u] + 1;
                if (!inQueue[v]) {
                    q.push(v);
                    inQueue[v] = true;
                }
            }
        }
    }

    if (dist[1] == INF)
        cout << -1 << endl;
    else
        cout << dist[1] << endl;

    return 0;
}
```
