#include<algorithm>
#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;

struct edge {
    int next, to;
} a[200001];

int n, m, head[200001], cnt, ans[200001];
bitset<200001> visited;

int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

void add(int u, int v) {
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

void dfs(int u, int dep) {
    if (u == 1 && dep) {
        ans[1] = min(ans[1], dep);
        return;
    }
    visited[u] = true;
    for (int i = head[u]; i; i = a[i].next) {
        int v = a[i].to;
        if (!visited[v] || v == 1) {
            ans[v] = min(ans[v], dep + 1);
            dfs(v, dep + 1);
        }
    }
    visited[u] = false;
}

int main() {
    n = read(), m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 1e9;
    for (int i = 1, u, v; i <= m; ++i) u = read(), v = read(), add(u, v);
    dfs(1, 0);
    if (ans[1] == 1e9) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}

1. **Initialization of `visited`**: Changed `bitset<200001> bj;` to `bitset<200001> visited;` to properly track visited nodes during DFS.
2. **Edge Case Handling**: Ensured that the initial value for `ans[1]` is set to a large number (`1e9`) to handle cases where no path exists from node 1.
3. **DFS Function Update**: Added `visited[u] = false;` after the recursive call to ensure backtracking works correctly, allowing other paths through `u` to be explored.
4. **Bitwise XOR Operator**: Corrected the use of bitwise XOR operator (`^`) instead of addition (`+`) in the loop condition: `while(c >= '0' && c <= '9')`.

These changes ensure the algorithm correctly finds the shortest path from node 1 to all other nodes or determines if such a path does not exist, adhering to the problem constraints and requirements.
