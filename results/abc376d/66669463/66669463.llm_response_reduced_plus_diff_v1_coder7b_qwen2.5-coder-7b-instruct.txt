```cpp
#include <algorithm>
#include <iostream>
#include <cstdio>
#include <bitset>
using namespace std;

const int MAXN = 2e5 + 5;
struct edge {
    int next, to;
} a[MAXN << 1];
int n, m, head[MAXN], cnt, ans[MAXN];
bitset<MAXN> bj;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ '0'), c = getchar();
    return res;
}

inline void add(const int u, const int v) {
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

void dfs(const int u, const int dep) {
    if (u == 1 && dep) {
        ans[1] = min(ans[1], dep);
        return;
    }
    bj[u] = true;
    for (int i = head[u], v; i; i = a[i].next) {
        v = a[i].to;
        if ((!bj[v] || v == 1) && dep + 1 < ans[v]) {
            ans[v] = dep + 1;
            dfs(v, dep + 1);
        }
    }
    return;
}

int main() {
    n = read(), m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 100000000;
    for (int i = 1, u, v; i <= m; ++i) u = read(), v = read(), add(u, v), add(v, u); // Add both directions since it's an undirected graph
    dfs(1, 0);
    if (ans[1] == 100000000) printf("-1");
    else printf("%d\n", ans[1]);
    return 0;
}
```
