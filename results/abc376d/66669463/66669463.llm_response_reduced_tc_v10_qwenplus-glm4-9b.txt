#include <algorithm>
#include <iostream>
#include <cstdio>
#include <bitset>
using namespace std;

struct edge {
    int next, to;
} edge[200001];

int n, m, head[200001], cnt, ans[200001];
bitset<200001> visited;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res * 10) + (c - '0'), c = getchar();
    return res;
}

inline void add(int u, int v) {
    edge[cnt].next = head[u];
    edge[cnt].to = v;
    head[u] = cnt++;
}

void dfs(int u, int dep) {
    if (visited[u]) return;
    visited[u] = 1;
    for (int i = head[u], v; i != -1; i = edge[i].next) {
        v = edge[i].to;
        if (!visited[v] && dep + 1 < ans[v]) ans[v] = dep + 1, dfs(v, dep + 1);
    }
    return;
}

int main() {
    n = read(), m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 100000000;
    for (int i = 1, u, v; i <= m; ++i) {
        u = read(), v = read();
        add(u, v);
        add(v, u); // Ensure undirected graph
    }
    dfs(1, 0);
    if (ans[1] == 100000000) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}
