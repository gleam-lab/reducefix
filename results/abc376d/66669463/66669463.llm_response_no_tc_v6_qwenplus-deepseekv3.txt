        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    queue<pair<int, int>> q;
    
    // Initialize BFS from node 1's neighbors

        q.push({v, 1});

        q.pop();
        
        for (int v : adj[u]) {
            if (v == 1) {

                continue;
            }
            if (dist[v] == -1 || dist[u] + 1 < dist[v]) {
                dist[v] = dist[u] + 1;
                q.push({v, u});
            }
        }
    }

        cout << "-1" << endl;

        cout << min_cycle << endl;
    }

    return 0;
