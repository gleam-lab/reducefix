```cpp
#include <algorithm>
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
using namespace std;

// Use adjacency list for graph representation
vector<vector<int>> adj;
const int INF = 1e9;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    adj.resize(n + 1);  // Nodes are 1-based

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    // BFS from node 1
    vector<int> dist(n + 1, INF);
    queue<int> q;
    q.push(1);
    dist[1] = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : adj[u]) {
            if (dist[v] == INF) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    if (dist[1] == INF) {
        cout << -1 << '\n';
    } else {
        cout << dist[1] << '\n';
    }

    return 0;
}
```
