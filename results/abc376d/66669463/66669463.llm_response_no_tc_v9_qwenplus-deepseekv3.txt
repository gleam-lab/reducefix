After analyzing the code, I found several issues that need to be fixed:

1. The depth-first search (DFS) approach is incorrect for finding shortest cycles because DFS doesn't guarantee finding the shortest path in an unweighted graph (BFS is more suitable).
2. The visited tracking (`bj`) is incorrectly preventing exploration of shorter paths.
3. The cycle detection logic is flawed - it only checks if we return to node 1.
4. The initial ans[] values are too large and could be more efficiently set to INT_MAX.

Here's the corrected version using BFS for shortest cycle detection containing node 1:

```cpp
#include<algorithm>
#include<iostream>
#include<cstdio>
#include<queue>
#include<climits>
using namespace std;

struct edge {
    int next, to;
} a[200001];
int n, m, head[200001], cnt, ans = INT_MAX;
int dist[200001];

inline int read() {
    int res = 0;
    char c = getchar();
    while(c < '0' || c > '9') c = getchar();
    while(c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

inline void add(const int u, const int v) {
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

void bfs() {
    queue<int> q;
    for(int i = 1; i <= n; ++i) dist[i] = -1;
    
    dist[1] = 0;
    q.push(1);
    
    while(!q.empty()) {
        int u = q.front();
        q.pop();
        
        for(int i = head[u], v; i; i = a[i].next) {
            v = a[i].to;
            if(v == 1) {  // found a cycle back to node 1
                ans = min(ans, dist[u] + 1);
            }
            if(dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    n = read(); m = read();
    for(int i = 1, u, v; i <= m; ++i) {
        u = read(), v = read();
        add(u, v);
    }
    
    bfs();
    
    if(ans == INT_MAX) printf("-1");
    else printf("%d", ans);
    
    return 0;
}
```

Key improvements:
1. Replaced DFS with BFS to find shortest paths
2. Properly tracks distances from node 1
3. Efficiently detects cycles back to node 1 during BFS
4. Uses INT_MAX for initialization which is more appropriate
5. Simplified the overall structure and removed unnecessary variables
6. Better handling of the case where no cycle exists

This solution will correctly find the shortest cycle containing node 1 or return -1 if no such cycle exists.
