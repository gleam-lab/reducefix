#include<algorithm>
#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;

struct Edge {
    int next, to;
};

Edge edges[200001];
int n, m, head[200001], cnt = 0, ans[200001];
bitset<200001> visited;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

inline void add(const int u, const int v) {
    edges[++cnt].next = head[u];
    edges[cnt].to = v;
    head[u] = cnt;
}

void dfs(const int u, const int depth) {
    if (u == 1 && depth) {
        ans[1] = min(ans[1], depth);
        return;
    }
    visited[u] = true;
    for (int i = head[u], v; i; i = edges[i].next) {
        v = edges[i].to;
        if (!visited[v] || v == 1) {
            if (depth + 1 < ans[v]) {
                ans[v] = depth + 1;
                dfs(v, depth + 1);
            }
        }
    }
    return;
}

int main() {
    n = read(), m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 100000000;
    for (int i = 1, u, v; i <= m; ++i) u = read(), v = read(), add(u, v);
    dfs(1, 0);
    if (ans[1] == 100000000) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}
