#include <iostream>
#include <vector>
#include <climits>
using namespace std;

vector<int> adj[200001];
int visited[200001];
int depth[200001];

void dfs(int node, int dep) {
    visited[node] = 1;
    depth[node] = dep;

    for (int neighbor : adj[node]) {
        if (!visited[neighbor]) {
            dfs(neighbor, dep + 1);
            depth[node] = min(depth[node], depth[neighbor]);
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    dfs(1, 0);

    cout << depth[1] << endl;
}
