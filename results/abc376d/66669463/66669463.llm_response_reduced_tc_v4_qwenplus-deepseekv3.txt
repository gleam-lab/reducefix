```cpp
#include <algorithm>
#include <iostream>
#include <cstdio>
#include <bitset>
#include <queue>
using namespace std;

struct edge {
    int next, to;
} a[200001];
int n, m, head[200001], cnt, ans[200001];
bitset<200001> bj;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

inline void add(const int u, const int v) {
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

void bfs() {
    queue<pair<int, int>> q;
    q.push({1, 0});
    ans[1] = 0;
    bj[1] = 1;

    while (!q.empty()) {
        auto [u, dep] = q.front();
        q.pop();

        for (int i = head[u], v; i; i = a[i].next) {
            v = a[i].to;
            if (v == 1) {
                if (dep + 1 < ans[1]) {
                    ans[1] = dep + 1;
                }
                continue;
            }
            if (!bj[v] && dep + 1 < ans[v]) {
                ans[v] = dep + 1;
                bj[v] = 1;
                q.push({v, dep + 1});
            }
        }
    }
}

int main() {
    n = read(); m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 100000000;
    for (int i = 1, u, v; i <= m; ++i) u = read(), v = read(), add(u, v);
    bfs();
    if (ans[1] == 100000000) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}
```
