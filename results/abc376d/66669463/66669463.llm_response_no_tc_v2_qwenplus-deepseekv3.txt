I analyzed the code and found the following issues:

1. The algorithm tries to find the shortest cycle containing node 1 using DFS, but DFS is not suitable for shortest path problems in general graphs (especially when there are cycles).
2. The visited marking (bj) is not properly managed - it prevents exploring alternative paths that might lead to shorter cycles.
3. The initial distance for node 1 should be 0, not a large number, since we're starting from it.

Here's the fixed code using BFS (Breadth-First Search) which is more appropriate for shortest path problems:

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<climits>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n+1);
    for(int i=0; i<m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    vector<int> dist(n+1);
    queue<int> q;
    
    // BFS from node 1's neighbors
    for(int start : adj[1]) {
        fill(dist.begin(), dist.end(), -1);
        dist[start] = 1;
        q.push(start);
        
        while(!q.empty()) {
            int u = q.front();
            q.pop();
            
            for(int v : adj[u]) {
                if(v == 1) {
                    min_cycle = min(min_cycle, dist[u] + 1);
                    continue;
                }
                if(dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    
    if(min_cycle == INT_MAX) {
        cout << "-1\n";
    } else {
        cout << min_cycle << "\n";
    }
    
    return 0;
}
```

Key improvements:
1. Uses BFS instead of DFS for finding shortest paths
2. Properly initializes distances and manages the queue
3. Starts BFS from each neighbor of node 1 to find the shortest cycle containing it
4. Uses more modern C++ features like vectors instead of manual memory management
5. Handles the case where no cycle exists (outputs -1)
6. More readable and maintainable structure

The algorithm now correctly finds the shortest cycle containing node 1 by performing BFS from each of its neighbors and checking when we can return to node 1.
