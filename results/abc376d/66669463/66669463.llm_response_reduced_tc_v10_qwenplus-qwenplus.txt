#include <algorithm>
#include <iostream>
#include <cstdio>
#include <queue>
#include <bitset>
using namespace std;

struct edge
{
    int next, to;
} a[200001];

int n, m, head[200001], cnt, ans[200001];
bitset<200001> visited;
inline int read()
{
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9')
        c = getchar();
    while (c >= '0' && c <= '9')
        res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

inline void add(const int u, const int v)
{
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

int main()
{
    n = read(), m = read();
    for (int i = 1; i <= m; ++i)
    {
        int u = read(), v = read();
        add(u, v);
    }

    queue<int> q;
    q.push(1);
    visited[1] = true;
    ans[1] = 0;

    while (!q.empty())
    {
        int u = q.front();
        q.pop();
        for (int i = head[u]; i; i = a[i].next)
        {
            int v = a[i].to;
            if (!visited[v])
            {
                visited[v] = true;
                ans[v] = ans[u] + 1;
                q.push(v);
            }
        }
    }

    cout << (ans[n] ? ans[n] : -1);
}
