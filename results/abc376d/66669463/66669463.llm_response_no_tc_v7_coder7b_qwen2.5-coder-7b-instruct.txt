#include<algorithm>
#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;

const int MAXN = 200001;

struct Edge {
    int next, to;
} edges[MAXN];

int n, m, head[MAXN], cnt;
int ans[MAXN];
bitset<MAXN> visited;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c - '0'), c = getchar();
    return res;
}

inline void addEdge(int u, int v) {
    edges[++cnt].next = head[u];
    edges[cnt].to = v;
    head[u] = cnt;
}

void dfs(int u, int depth) {
    if (u == 1 && depth) {
        ans[1] = min(ans[1], depth);
        return;
    }
    visited[u] = true;
    for (int i = head[u]; i; i = edges[i].next) {
        int v = edges[i].to;
        if (!visited[v] || v == 1) {
            if (depth + 1 < ans[v]) {
                ans[v] = depth + 1;
                dfs(v, depth + 1);
            }
        }
    }
    return;
}

int main() {
    n = read(), m = read();
    for (int i = 1; i <= n; ++i) ans[i] = 100000000;
    for (int i = 1, u, v; i <= m; ++i) {
        u = read(), v = read();
        addEdge(u, v);
        addEdge(v, u); // Assuming undirected graph
    }
    dfs(1, 0);
    if (ans[1] == 100000000) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}

### Explanation of Changes:
1. **Graph Representation**: The original code assumed a directed graph but did not account for the reverse edges when adding connections. This was corrected by adding both `u -> v` and `v -> u`.
2. **Bitset Initialization**: The bitset `bj` is renamed to `visited` for better clarity and initialized correctly.
3. **Answer Array Initialization**: The answer array `ans` is initialized with a large value (`100000000`) to ensure that any valid distance found will be smaller.
4. **DFS Function**: The DFS function now checks if the current node has been visited before or if it is the starting node (1). It updates the answer array only if the new path length is shorter than the previously recorded one.
5. **Output Handling**: The program checks if the answer remains unchanged after processing all nodes. If so, it prints `-1`, indicating no valid path exists from the root node (1).
