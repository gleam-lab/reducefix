After analyzing the code, I found several issues that need to be fixed:

1. The graph is being treated as undirected when it's actually directed (based on how edges are added)
2. The DFS approach isn't suitable for finding shortest cycles because it explores all paths without priority
3. The visited marking (bj) interferes with finding alternative paths
4. The cycle detection logic is flawed as it only checks if it returns to node 1

Here's the fixed code using BFS (more suitable for shortest path problems) with proper cycle detection:

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<climits>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n+1);
    for(int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    
    for(int start = 1; start <= n; ++start) {
        vector<int> dist(n+1, -1);
        queue<int> q;
        q.push(start);
        dist[start] = 0;
        
        while(!q.empty()) {
            int u = q.front();
            q.pop();
            
            for(int v : adj[u]) {
                if(v == start) {
                    min_cycle = min(min_cycle, dist[u] + 1);
                    continue;
                }
                if(dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    
    if(min_cycle != INT_MAX) {
        cout << min_cycle << endl;
    } else {
        cout << -1 << endl;
    }
    
    return 0;
}
```

Key improvements:
1. Uses BFS instead of DFS for shortest path finding
2. Checks for cycles from every node (not just node 1)
3. Properly tracks distances and detects cycles when returning to the starting node
4. Handles the case where no cycle exists
5. More efficient and cleaner implementation using standard C++ features

This solution will correctly find the shortest directed cycle in the graph or return -1 if none exists.
