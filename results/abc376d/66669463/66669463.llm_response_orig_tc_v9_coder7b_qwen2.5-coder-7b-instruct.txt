#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

const int MAXN = 200001;

vector<int> adj[MAXN];
bool visited[MAXN];

void dfs(int u, vector<int>& dist) {
    visited[u] = true;
    for (int v : adj[u]) {
        if (!visited[v]) {
            dist[v] = dist[u] + 1;
            dfs(v, dist);
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> dist(n + 1, INT_MAX);
    dist[1] = 0;
    dfs(1, dist);

    int min_dist = INT_MAX;
    for (int i = 1; i <= n; ++i) {
        min_dist = min(min_dist, dist[i]);
    }

    cout << min_dist << endl;

    return 0;
}
