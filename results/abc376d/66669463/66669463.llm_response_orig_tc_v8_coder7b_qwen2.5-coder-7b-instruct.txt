#include <iostream>
#include <vector>
#include <queue>
using namespace std;

const int MAXN = 2e5 + 10;
vector<int> adj[MAXN];

int bfs(int start, int n) {
    vector<bool> visited(n + 1, false);
    queue<pair<int, int>> q;
    q.push({start, 0});
    visited[start] = true;
    
    while (!q.empty()) {
        auto [node, dist] = q.front();
        q.pop();
        
        if (dist > 0 && node == 1) return dist - 1;
        
        for (int neighbor : adj[node]) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                q.push({neighbor, dist + 1});
            }
        }
    }
    
    return -1;
}

int main() {
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    int result = bfs(1, n);
    cout << (result == -1 ? "IMPOSSIBLE" : to_string(result)) << endl;
    
    return 0;
}
