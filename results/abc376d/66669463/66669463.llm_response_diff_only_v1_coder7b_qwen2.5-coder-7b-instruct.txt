#include<algorithm>
#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;

const int MAXN = 200001;

struct Edge {
    int next, to;
} edges[MAXN * 2];

int n, m, head[MAXN], cnt, ans[MAXN];
bitset<MAXN> visited;

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c - '0'), c = getchar();
    return res;
}

inline void add(int u, int v) {
    edges[++cnt].next = head[u];
    edges[cnt].to = v;
    head[u] = cnt;
}

void dfs(int u, int depth) {
    if (u == 1 && depth != 0) {
        ans[1] = min(ans[1], depth);
        return;
    }
    visited[u] = true;
    for (int i = head[u], v; i; i = edges[i].next) {
        v = edges[i].to;
        if (!visited[v] || v == 1) {
            if (depth + 1 < ans[v]) {
                ans[v] = depth + 1;
                dfs(v, depth + 1);
            }
        }
    }
    visited[u] = false;
}

int main() {
    n = read();
    m = read();
    fill(ans + 1, ans + n + 1, 100000000);
    for (int i = 1, u, v; i <= m; ++i) {
        u = read();
        v = read();
        add(u, v);
    }
    dfs(1, 0);
    if (ans[1] == 100000000) printf("-1");
    else printf("%d", ans[1]);
    return 0;
}

### Explanation of Fixes:
1. **Edge Case Handling**: The original code did not handle the case where `u` is `1` and `depth` is `0` correctly. This has been addressed by checking if `u` is `1` and `depth` is not zero before updating `ans[1]`.
2. **Visited Bitset Reset**: After processing all neighbors of a node, the `visited` bitset for that node was not reset, which could lead to incorrect results if the same node is processed again. This has been corrected by resetting the `visited` status after the DFS call for each neighbor.
3. **Initialization of `ans` Array**: The `ans` array was initialized with a large number (`100000000`) but without considering the possibility of an unconnected graph. This has been adjusted to ensure it is properly initialized.
4. **Variable Naming and Formatting**: Improved variable names and formatting for better readability and consistency with common coding standards.
