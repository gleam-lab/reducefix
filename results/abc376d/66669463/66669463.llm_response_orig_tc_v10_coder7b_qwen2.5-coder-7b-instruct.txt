#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAXN = 2e5 + 10;

vector<int> adj[MAXN];
int n, m, dist[MAXN];

void bfs() {
    fill(dist, dist + n + 1, INT_MAX);
    dist[1] = 0;
    queue<int> q;
    q.push(1);

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : adj[u]) {
            if (dist[v] == INT_MAX) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    bfs();

    int ans = *max_element(dist + 1, dist + n + 1);
    cout << ans << endl;

    return 0;
}
