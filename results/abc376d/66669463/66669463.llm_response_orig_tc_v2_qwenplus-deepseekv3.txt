The issue in the original code is that it's using a depth-first search (DFS) approach to find the shortest cycle containing node 1, which is not the correct approach for this problem. DFS doesn't naturally find the shortest cycle in an unweighted graph. Instead, we should use a breadth-first search (BFS) approach from node 1 to find the shortest cycle that includes node 1.

Here's the corrected code using BFS:

```cpp
#include <algorithm>
#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    vector<int> dist(n + 1, -1);
    queue<int> q;
    
    for (int start : adj[1]) {
        dist.assign(n + 1, -1);
        dist[start] = 1;
        q.push(start);
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            
            for (int v : adj[u]) {
                if (v == 1) {
                    min_cycle = min(min_cycle, dist[u] + 1);
                    break;
                }
                if (dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    q.push(v);
                }
            }
            if (min_cycle == 3) break; // Early exit if we found the smallest possible cycle
        }
        if (min_cycle == 3) break; // Early exit if we found the smallest possible cycle
    }
    
    cout << (min_cycle == INT_MAX ? -1 : min_cycle) << '\n';
    return 0;
}
```

This solution works by:
1. Building an adjacency list for the graph
2. For each neighbor of node 1, perform a BFS to find the shortest path back to node 1
3. Keep track of the minimum cycle length found
4. Early termination if we find a cycle of length 3 (the smallest possible cycle)
5. Output the result or -1 if no cycle was found

The BFS approach ensures we find the shortest cycle containing node 1 because BFS explores nodes in order of their distance from the starting point. Each time we find a path back to node 1, we know it's the shortest path from that particular neighbor.
