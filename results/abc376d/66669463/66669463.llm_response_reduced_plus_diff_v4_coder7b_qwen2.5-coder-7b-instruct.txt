#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

const int MAXN = 100001;

vector<int> adj[MAXN];
bool visited[MAXN];

void bfs(int start, vector<int>& distances) {
    queue<int> q;
    q.push(start);
    visited[start] = true;
    distances[start] = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                distances[v] = distances[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> distances(n + 1, INT_MAX);

    bfs(1, distances);

    int result = *max_element(distances.begin() + 2, distances.end());
    cout << (result == INT_MAX ? -1 : result) << endl;

    return 0;
}
