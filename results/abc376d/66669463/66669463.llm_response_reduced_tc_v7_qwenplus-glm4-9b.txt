#include <algorithm>
#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

const int MAXN = 200001;
vector<int> graph[MAXN];
int dist[MAXN], minDist[MAXN];

void dijkstra(int start, int n) {
    fill(dist, dist + n + 1, INT_MAX);
    fill(minDist, minDist + n + 1, INT_MAX);
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

    dist[start] = 0;
    minDist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();

        if (d > dist[u]) continue;

        for (int v : graph[u]) {
            int nextDist = d + 1;
            if (nextDist < dist[v]) {
                dist[v] = nextDist;
                pq.push({nextDist, v});
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        graph[i].clear();
    }

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u); // Ensure bidirectional graph
    }

    dijkstra(1, n);

    if (minDist[1] == INT_MAX) {
        cout << -1 << endl;
    } else {
        cout << minDist[1] << endl;
    }

    return 0;
}
