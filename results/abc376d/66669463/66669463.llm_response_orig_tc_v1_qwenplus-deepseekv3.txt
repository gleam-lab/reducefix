Based on the error analysis, the original code was incorrectly finding a cycle length of 56 when the expected shortest cycle was 3. The issue likely stems from how the depth-first search (DFS) was tracking cycles and updating the minimum cycle length.

Here's the corrected approach:
1. We need to find the shortest cycle starting and ending at node 1
2. Use BFS for each neighbor of node 1 to find the shortest path back to node 1
3. Track the minimum cycle length found (distance + 1 for the edge back to node 1)

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n+1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    
    // For each neighbor of node 1, perform BFS to find shortest path back to 1
    for (int neighbor : adj[1]) {
        vector<int> dist(n+1, -1);
        queue<int> q;
        q.push(neighbor);
        dist[neighbor] = 1;
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            
            if (u == 1) {
                min_cycle = min(min_cycle, dist[u]);
                break;
            }
            
            for (int v : adj[u]) {
                if (dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    
    if (min_cycle == INT_MAX) {
        cout << "-1\n";
    } else {
        cout << min_cycle << '\n';
    }
    
    return 0;
}
```

Key improvements:
1. Replaced DFS with BFS (Breadth-First Search) which naturally finds shortest paths
2. For each neighbor of node 1, we perform BFS to find the shortest path back to node 1
3. The cycle length is the path length found + 1 (for the edge from 1 to neighbor)
4. We keep track of the minimum cycle length found
5. If no cycle is found (min_cycle remains INT_MAX), we output -1

This approach ensures we find the shortest cycle containing node 1 efficiently with a time complexity of O(N + M) in the worst case, which is optimal for this problem.
