#include <algorithm>
#include <iostream>
#include <cstdio>
#include <queue>
using namespace std;

struct Edge {
    int to, next;
} a[200001];

int n, m, head[200001], cnt, ans[200001];
bool visited[200001];

inline int read() {
    int res = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') res = (res << 1) + (res << 3) + (c ^ 48), c = getchar();
    return res;
}

inline void add(int u, int v) {
    a[++cnt].next = head[u];
    a[cnt].to = v;
    head[u] = cnt;
}

void bfs() {
    queue<int> q;
    q.push(1);
    visited[1] = true;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int i = head[u]; i; i = a[i].next) {
            int v = a[i].to;
            if (!visited[v]) {
                visited[v] = true;
                ans[v] = ans[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    n = read();
    m = read();
    
    for (int i = 1; i <= n; ++i) ans[i] = -1;
    ans[1] = 0;
    
    for (int i = 1; i <= m; ++i) {
        int u = read(), v = read();
        add(u, v);
    }
    
    bfs();
    
    printf("%d", ans[1] == 0 ? -1 : ans[1]);
}
