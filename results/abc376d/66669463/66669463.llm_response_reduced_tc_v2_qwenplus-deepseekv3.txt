```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }
    
    int min_cycle = INT_MAX;
    vector<int> distance(n + 1);
    queue<int> q;
    
    for (int neighbor : adj[1]) {
        if (neighbor == 1) {
            min_cycle = 1;
            break;
        }
        distance.assign(n + 1, -1);
        distance[neighbor] = 1;
        q.push(neighbor);
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            
            for (int v : adj[u]) {
                if (v == 1) {
                    min_cycle = min(min_cycle, distance[u] + 1);
                    continue;
                }
                if (distance[v] == -1) {
                    distance[v] = distance[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    
    cout << (min_cycle == INT_MAX ? -1 : min_cycle) << '\n';
    return 0;
}
```

This solution:
1. Uses BFS which is more appropriate for shortest path problems in unweighted graphs
2. For each neighbor of node 1, performs a BFS to find the shortest path back to node 1
3. Keeps track of the minimum cycle length found
4. Handles the case where node 1 has a direct self-loop (cycle length = 1)
5. Returns -1 if no cycle exists that includes node 1

The time complexity is O(N + M) in the worst case, which is optimal for this problem. The space complexity is O(N + M) to store the graph and distances.
