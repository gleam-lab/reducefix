Here's the fixed code that properly tracks distances and checks for cycles involving node 0:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    vector<vector<int>> G(N);
    for(int i = 0; i < M; i++) {
        int s, v;
        cin >> s >> v;
        s--; v--;
        G[s].push_back(v);
    }

    int min_cycle = INT_MAX;
    vector<int> dist(N, -1);
    queue<int> q;
    q.push(0);
    dist[0] = 0;

    while(!q.empty()) {
        int u = q.front();
        q.pop();

        for(int v : G[u]) {
            if(v == 0) {
                min_cycle = min(min_cycle, dist[u] + 1);
            }
            if(dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    if(min_cycle != INT_MAX) {
        cout << min_cycle << endl;
    } else {
        cout << -1 << endl;
    }
    return 0;
}
```

3. Set distance of node 0 to 0 at start
4. When visiting neighbors, first check if it's node 0 to detect cycles
5. Only update distances and push to queue if node hasn't been visited
6. Simplified the output logic
