#include<bits/stdc++.h>

template<typename T> bool chmax(T &m, const T q) {
    if (m < q) { m = q; return true; } else return false;
}

    if (m > q) { m = q; return true; } else return false;
}

int main() {
    int N, M;
    cin >> N >> M;
    vvi G(N);
    vi dist(N, INF);
    dist[0] = 0; // Start from node 0, which is indexed as 0 in the graph
    rep(i, 0, M) {
        int s, v;
        cin >> s >> v;
        s--; v--;
        G[s].pb(v);
    }
    vb vis(N, false);
    queue<int> que;
    que.push(0);
    while (!que.empty()) {
        int pos = que.front();
        que.pop();
        if (vis[pos]) continue;
        vis[pos] = true;
        for (auto to : G[pos]) {
            if (!vis[to]) {
                que.push(to);
                dist[to] = dist[pos] + 1;
            } else {
                chmin(dist[pos], dist[to] + 1);
            }
        }
    }
    int min_length = INF;
    rep(i, 0, N) {
        if (dist[i] != INF) {
            chmin(min_length, dist[i]);
        }
    }
    if (min_length == INF) cout << -1 << el;
    else cout << min_length << el;
}
