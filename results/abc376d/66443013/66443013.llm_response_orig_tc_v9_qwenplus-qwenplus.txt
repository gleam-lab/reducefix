#include <bits/stdc++.h>
using namespace std;
using ll = long long;

map<ll, vector<ll>> graph;
map<ll, bool> visited;

ll min_cycle_length = -1;
const ll MAX_DEPTH = 20; // Sufficiently large to detect small cycles back to node 1

// DFS with path tracking to detect cycle back to node 1
void dfs(ll node, ll depth, const vector<ll>& path) {
    if (depth > MAX_DEPTH) return; // Prune deep paths to avoid excessive recursion

    visited[node] = true;

    for (auto neighbor : graph[node]) {
        if (neighbor == 1) {
            // Found a cycle back to node 1
            if (depth >= 1) { // At least one intermediate node in the cycle
                if (min_cycle_length == -1 || depth + 1 < min_cycle_length)
                    min_cycle_length = depth + 1;
            }
        } else if (!visited[neighbor]) {
            dfs(neighbor, depth + 1, path);
        } else {
            // Optional: check for shorter cycles via already visited nodes
            // Not needed here since we're interested in shortest only from 1 -> ... -> 1
        }
    }

    visited[node] = false; // Backtrack
}

int main() {
    ll n, m;
    cin >> n >> m;

    for (ll i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        visited[a] = false;
        visited[b] = false;
    }

    vector<ll> path;
    visited[1] = true;
    for (auto neighbor : graph[1]) {
        dfs(neighbor, 1, path);
    }

    cout << min_cycle_length << endl;
}
