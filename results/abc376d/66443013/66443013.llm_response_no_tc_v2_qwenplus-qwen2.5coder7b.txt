```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

void dfs(ll n, ll hen) {
    seen[n] = true;
    for (auto x : graph[n]) {
        if (x != 1 && seen[x]) continue;
        if (x == 1) {
            if (hen != 1) {
                ans = min(ans, hen + 1);
            }
            continue;
        }
        dfs(x, hen + 1);
    }
}

int main() {
    ll n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        graph[b].push_back(a); // Add this line to make it undirected graph
        if (!seen.count(a)) seen[a] = false;
        if (!seen.count(b)) seen[b] = false;
    }
    dfs(1, 0);
    if (ans != inf) cout << ans << endl;
    else cout << "-1" << endl;
    return 0;
}
```

In the original code, there was no line to add an edge from `b` to `a`, making the graph directed. This could lead to incorrect results when performing DFS. The corrected version includes this line to ensure the graph is undirected. Additionally, I've added a return statement at the end of the `main` function for good practice.
