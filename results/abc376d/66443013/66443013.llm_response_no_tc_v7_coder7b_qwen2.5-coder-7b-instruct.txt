```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

void dfs(ll n, ll hen) {
    seen[n] = true;
    for (auto x : graph[n]) {
        if (x != 1 && seen[x]) continue;
        if (x == 1) {
            if (hen != 1) {
                ans = min(ans, hen + 1);
            }
            continue;
        }
        dfs(x, hen + 1);
    }
}

int main() {
    ll n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        graph[b].push_back(a); // Ensure bidirectional edges
        if (!seen.count(a)) seen[a] = false;
        if (!seen.count(b)) seen[b] = false;
    }
    dfs(1, 0);
    if (ans != inf) cout << ans << endl;
    else cout << "-1" << endl;
}
```

### Explanation of Fixes
1. **Bidirectional Edges**: The original code did not handle bidirectional edges correctly. This was corrected by adding `graph[b].push_back(a);` to ensure that both directions of the edge are considered.
2. **Edge Case Handling**: The code now handles the case where there is no path to node 1 by checking if `ans` remains unchanged at `inf`.
3. **Readability**: The code structure has been maintained for readability, ensuring that each function and loop serves its intended purpose clearly.
