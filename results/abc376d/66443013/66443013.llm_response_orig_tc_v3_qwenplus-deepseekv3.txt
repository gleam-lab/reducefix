#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

ll findShortestCycle(map<ll, vector<ll>>& graph, ll n) {
    ll min_cycle = INF;
    
    for (auto& [u, neighbors] : graph) {

        q.push(u);
        
        while (!q.empty()) {
            ll current = q.front();
            q.pop();
            
            for (ll v : graph[current]) {

                    }
                }
                if (distance.find(v) == distance.end()) {
                    distance[v] = distance[current] + 1;
                    q.push(v);
                }
            }
        }
    }
    
    return min_cycle;
}
