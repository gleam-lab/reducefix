#include<bits/stdc++.h>
using namespace std;
using ll=long long;
map<ll,vector<ll>>graph;
map<ll,bool>seen;
const ll inf=1LL<<60;
ll ans=inf;

// Function to perform DFS and find the minimum path length
void dfs(ll node, ll distance){
  seen[node] = true; // Mark the current node as visited
  for(auto neighbor : graph[node]){
    if(neighbor != 1 && seen[neighbor]) continue; // Skip revisiting nodes except for node 1
    if(neighbor == 1){
      if(distance != 0){ // Ensure we don't count the initial step to node 1 as part of the path
        ans = min(ans, distance + 1); // Update the answer with the minimum path length found
      }
      continue;
    }
    dfs(neighbor, distance + 1); // Recur for all unvisited neighbors
  }
}

int main(){
  ll n, m;
  cin >> n >> m;
  for(int i = 0; i < m; i++){
    ll a, b;
    cin >> a >> b;
    graph[a].push_back(b);
    graph[b].push_back(a); // Since it's an undirected graph, add both directions
    if(!seen.count(a)) seen[a] = false;
    if(!seen.count(b)) seen[b] = false;
  }
  dfs(1, 0); // Start DFS from node 1 with an initial distance of 0
  if(ans != inf) cout << ans << endl;
  else cout << "-1" << endl;
}
