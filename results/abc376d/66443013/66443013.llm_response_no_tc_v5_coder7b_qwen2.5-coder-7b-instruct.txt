```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
vector<vector<int>> graph;
vector<bool> visited;

void dfs(int node, int parent, int &cycle_length) {
    visited[node] = true;
    for (auto neighbor : graph[node]) {
        if (!visited[neighbor]) {
            dfs(neighbor, node, cycle_length);
        } else if (neighbor != parent) {
            cycle_length = min(cycle_length, abs(node - neighbor));
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    graph.resize(n + 1);
    visited.resize(n + 1, false);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
    }

    int result = INT_MAX;
    for (int i = 1; i <= n; ++i) {
        if (!visited[i]) {
            int cycle_length = INT_MAX;
            dfs(i, -1, cycle_length);
            result = min(result, cycle_length);
        }
    }

    if (result == INT_MAX) {
        cout << "-1" << endl;
    } else {
        cout << result << endl;
    }

    return 0;
}
```

### Explanation:
1. **Graph Representation**: We use an adjacency list to represent the graph.
2. **DFS Function**: The `dfs` function is used to detect cycles starting from each node. It keeps track of the current path length (`cycle_length`) and updates it when a backedge is found.
3. **Cycle Detection**: If a backedge is detected (i.e., a node is revisited but not as the parent), we calculate the cycle length.
4. **Main Logic**: For each unvisited node, we initiate a DFS to find cycles and keep track of the minimum cycle length.
5. **Output**: Finally, we output the minimum cycle length or `-1` if no cycle containing vertex 1 is found.
