#include <bits/stdc++.h>
using namespace std;
using ll = long long;

map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

// Use BFS instead of DFS to find shortest path
void bfs(ll start) {
    queue<pair<ll, ll>> q; // {node, distance}
    q.push({start, 0});
    seen[start] = true;

    while (!q.empty()) {
        auto [node, dist] = q.front();
        q.pop();

        for (auto neighbor : graph[node]) {
            if (neighbor == 1) {
                ans = min(ans, dist + 1);
                return; // Found shortest path to node 1
            }
            if (!seen[neighbor]) {
                seen[neighbor] = true;
                q.push({neighbor, dist + 1});
            }
        }
    }
}

int main() {
    ll n, m;
    cin >> n >> m;

    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        if (!seen.count(a)) seen[a] = false;
        if (!seen.count(b)) seen[b] = false;
    }

    bfs(1);

    if (ans != inf) cout << ans << endl;
    else cout << "-1" << endl;
}
