### Approaches
1. **Problem Analysis**: The task is to find the shortest cycle that includes node 1 in a directed graph. If no such cycle exists, we should return -1.
2. **Intuition**: BFS is suitable for finding the shortest path in unweighted graphs. We can use BFS to explore all possible paths from node 1, keeping track of the distance from node 1. If we encounter node 1 again during the BFS, we have found a cycle, and we can update the shortest cycle length accordingly.
3. **Algorithm Selection**: BFS is chosen because it explores all nodes at the present depth before moving on to nodes at the next depth level, ensuring the shortest path is found first.
4. **Complexity Analysis**: The BFS approach runs in O(N + M) time, where N is the number of nodes and M is the number of edges, which is efficient for the problem constraints.

```cpp
#include<bits/stdc++.h>
using namespace std;
using ll=long long;

int main() {
    ll n, m;
    cin >> n >> m;
    map<ll, vector<ll>> graph;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
    }

    ll ans = -1;
    queue<pair<ll, ll>> q; // {current node, distance from 1}
    q.push({1, 0});
    vector<ll> dist(n + 1, -1);
    dist[1] = 0;

    while (!q.empty()) {
        auto [current, d] = q.front();
        q.pop();
        for (auto neighbor : graph[current]) {
            if (neighbor == 1) {
                if (ans == -1 || d + 1 < ans) {
                    ans = d + 1;
                }
            }
            if (dist[neighbor] == -1) {
                dist[neighbor] = d + 1;
                q.push({neighbor, d + 1});
            }
        }
    }

    cout << ans << endl;
    return 0;
}
```
