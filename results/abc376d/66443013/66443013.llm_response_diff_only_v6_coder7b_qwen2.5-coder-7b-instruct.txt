```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

void dfs(ll n, ll hen) {
    seen[n] = true;
    for (auto x : graph[n]) {
        if (x == 1 && hen != 1) {
            ans = min(ans, hen + 1);
            continue;
        }
        if (!seen[x]) dfs(x, hen + 1);
    }
}

int main() {
    ll n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        graph[b].push_back(a); // Ensure undirected graph
        seen[a] = false;
        seen[b] = false;
    }
    dfs(1, 0);
    if (ans != inf) cout << ans << endl;
    else cout << "-1" << endl;
    return 0;
}
```
This solution includes the necessary corrections to ensure the graph is treated as undirected and removes unnecessary checks that were leading to incorrect results. The `dfs` function now correctly updates the answer when it reaches the target node (node 1). Additionally, the `seen` map is initialized for all nodes before the DFS starts.
