#include <bits/stdc++.h>
using namespace std;
using ll = long long;

map<ll, vector<ll>> graph;
map<ll, bool> visited;
const ll inf = 1LL << 60;
ll ans = inf;

// BFS function to find shortest cycle involving node 1
void bfs(ll start) {
    queue<pair<ll, ll>> q;  // {node, distance}
    map<ll, ll> dist;

    // Initialize distances
    for (auto &[node, _] : graph) {
        dist[node] = inf;
    }

    q.push({start, 0});
    dist[start] = 0;

    while (!q.empty()) {
        auto [u, d] = q.front();
        q.pop();

        for (auto v : graph[u]) {
            if (dist[v] == inf) {
                dist[v] = d + 1;
                q.push({v, dist[v]});
            } else if (v != start) {
                // If already visited and not parent, it's part of a cycle
                ans = min(ans, dist[u] + dist[v] + 1);  // u -> v is one edge
            }
        }
    }
}

int main() {
    ll n, m;
    cin >> n >> m;

    for (int i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
    }

    bfs(1);

    if (ans != inf)
        cout << ans << endl;
    else
        cout << -1 << endl;

    return 0;
}
