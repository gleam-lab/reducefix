#include<bits/stdc++.h>
using namespace std;
using ll=long long;
map<ll,vector<ll>>graph;
map<ll,ll>dist;
const ll inf=1LL<<60;

void bfs(ll start) {
    queue<ll> q;
    q.push(start);
    dist[start] = 0;
    while (!q.empty()) {
        ll node = q.front();
        q.pop();
        for (ll adj : graph[node]) {
            if (dist[adj] == inf) {
                dist[adj] = dist[node] + 1;
                q.push(adj);
            }
        }
    }
}

int main() {
    ll n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        if (dist.count(a) == 0) dist[a] = inf;
        if (dist.count(b) == 0) dist[b] = inf;
    }
    bfs(1);
    
    ll max_dist = 0;
    for (auto& p : dist) {
        if (p.first != 1) {
            max_dist = max(max_dist, p.second);
        }
    }
    
    if (max_dist == inf) {
        cout << "-1" << endl;
    } else {
        cout << max_dist << endl;
    }
}
