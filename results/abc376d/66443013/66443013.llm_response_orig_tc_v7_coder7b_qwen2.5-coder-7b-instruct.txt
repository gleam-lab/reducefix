```cpp
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

void dfs(ll n, ll hen) {
    seen[n] = true;
    for (auto x : graph[n]) {
        if (x != 1 && seen[x]) continue;
        if (x == 1) {
            if (hen != 1) {
                ans = min(ans, hen + 1);
            }
            continue;
        }
        dfs(x, hen + 1);
    }
    // seen[n] = false; // Uncommenting this line can lead to incorrect results because it allows revisiting nodes which is not allowed in this problem
}

int main() {
    ll n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        graph[b].push_back(a); // Assuming undirected graph
        if (!seen.count(a)) seen[a] = false;
        if (!seen.count(b)) seen[b] = false;
    }
    dfs(1, 0);
    if (ans != inf) cout << ans << endl;
    else cout << "-1" << endl;
    return 0;
}
```

### Explanation of Changes:
1. **Graph Representation**: Added `graph[b].push_back(a);` to handle an undirected graph correctly.
2. **Edge Cases**: The existing code handles edge cases like empty input and single elements implicitly through the way it initializes and processes the graph.
3. **Complexity**: The time complexity remains O(V + E), where V is the number of vertices and E is the number of edges, as each vertex and edge is processed at most once.
