using namespace std;

using ll = long long;
map<ll, vector<ll>> graph;
map<ll, bool> seen;
const ll inf = 1LL << 60;
ll ans = inf;

void dfs(ll n, ll hen) {
    seen[n] = true;
    for (auto x : graph[n]) {
        if (x != 1 && seen[x]) continue;
        if (x == 1) {
            if (hen != 1) {
                ans = min(ans, hen + 1);
            }
            continue;
        }
        dfs(x, hen + 1);
    }
}

        graph[a].push_back(b);

        if (!seen.count(a)) seen[a] = false;

    dfs(1, 0);
    if (ans != inf) cout << ans << endl;

    return 0;
