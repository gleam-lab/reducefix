vector<vector<ll>>graph;

  seen[n]=true;
  for(auto x:graph[n]){
    if(seen[x]&&x!=1)continue; // Only consider back edges

        ans=min(ans,hen+1);
      }
      continue;
    }
    dfs(x,hen+1);
  }
  seen[n]=false; // Backtrack
}
