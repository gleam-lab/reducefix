I analyzed the code and found several issues that need to be fixed:

1. The BFS implementation is incorrect - it marks nodes as visited when they are popped from the queue rather than when they are discovered, which can lead to duplicate entries in the queue.

2. The condition for checking if we've reached node 1 is inside the neighbor loop, but we should also check the starting node (1) immediately.

3. The step counting logic is correct but could be made more robust by initializing steps to -1 or INF.

Here's the fixed code:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN];
bool vis[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    memset(step, -1, sizeof(step));
    q.push(1);
    step[1] = 0;
    vis[1] = true;
    
    while(!q.empty()){
        ll top = q.front();
        q.pop();
        
        if(top == 1 && step[top] > 0) {
            printf("%lld", step[top]);
            return 0;
        }
        
        for(ll i=0; i<vec[top].size(); i++){
            ll to = vec[top][i];
            if(to == 1) {
                printf("%lld", step[top] + 1);
                return 0;
            }
            if(!vis[to]){
                vis[to] = true;
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }
    
    printf("-1");
    return 0;
}
```

Key improvements:
1. Proper BFS implementation by marking nodes as visited when they are added to the queue
2. Initialize step counts to -1 (unvisited)
3. Check for the cycle at node 1 both when processing the node itself and when processing its neighbors
4. More robust step counting with proper initialization
5. The code now correctly handles cases where node 1 is immediately reachable (distance 0) vs when it's part of a cycle (distance > 0)
