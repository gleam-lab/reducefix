```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN];
bool vis[MAXN];
int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    q.push(1);
    step[1] = 0; // Initialize starting point's step as 0
    while(!q.empty()){
        ll top=q.front();
        vis[top]=true;
        q.pop();
        for(ll i=0;i<vec[top].size();i++){
            ll to=vec[top][i];
            if(step[to] == INF){ // Check if node is not visited yet
                step[to]=step[top]+1;
                q.push(to);
            }
        }
    }
    if(step[n] != INF) { // Check if destination is reachable
        printf("%lld\n", step[n]);
    } else {
        printf("-1\n"); // If not reachable
    }
    return 0;
}
```
In this solution:
- The `step` array is initialized with `INF` to indicate that all nodes are initially unreachable except the starting node which is set to `0`.
- A check is added to ensure that only unvisited nodes (where `step[to] == INF`) are processed to avoid unnecessary operations.
- After processing all possible paths from the source, the code checks if the destination node (`n`) is reachable by verifying `step[n]`. If it is not reachable, `-1` is printed; otherwise, the shortest path length is printed.
