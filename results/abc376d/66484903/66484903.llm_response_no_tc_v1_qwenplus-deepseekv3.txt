The original code attempts to find the shortest cycle involving node 1 in a directed graph using BFS. However, there are a few issues:

1. The BFS doesn't correctly handle cycles because it marks nodes as visited too early (before checking if they lead back to node 1).
2. The step counting is incorrect for cycles since it doesn't account for the initial step from node 1 to its neighbors.

Here's the corrected version that properly tracks the shortest cycle involving node 1:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll step[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    ll min_cycle = INF;
    queue<ll> q;
    
    // Initialize BFS from node 1's neighbors
    for(auto neighbor : vec[1]){
        q.push(neighbor);
        step[neighbor] = 1;
    }
    
    while(!q.empty()){
        ll top = q.front();
        q.pop();
        
        for(auto to : vec[top]){
            if(to == 1){
                min_cycle = min(min_cycle, step[top] + 1);
                continue;
            }
            if(step[to] == 0){
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }
    
    printf("%lld", min_cycle == INF ? -1 : min_cycle);
    return 0;
}
```

Key improvements:
1. Starts BFS from node 1's neighbors instead of node 1 itself
2. Properly tracks steps taken to return to node 1
3. Uses step array for both distance tracking and visited status
4. Finds the minimum cycle length among all possible cycles involving node 1
5. Handles the case where no cycle exists (outputs -1)

The algorithm now correctly finds the shortest directed cycle that includes node 1, or returns -1 if no such cycle exists.
