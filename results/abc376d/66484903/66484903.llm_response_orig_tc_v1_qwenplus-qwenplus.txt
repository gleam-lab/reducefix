```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int MAXN = 200010;

vector<ll> adj[MAXN];
ll dist[MAXN];

void bfs(ll start) {
    queue<ll> q;
    q.push(start);
    dist[start] = 0;

    while (!q.empty()) {
        ll u = q.front();
        q.pop();

        for (ll v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    ll n, m;
    cin >> n >> m;

    for (ll i = 0; i < m; ++i) {
        ll u, v;
        cin >> u >> v;
        adj[u].push_back(v);
    }

    // Initialize distance array to -1 (unvisited)
    memset(dist, -1, sizeof(dist));

    bfs(1);

    // Output the distance from node 1 to node n
    cout << dist[n] << endl;

    return 0;
}
```
