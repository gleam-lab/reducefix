```cpp
#include <bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;

typedef long long ll;
const int MAXN = 2e5 + 10;
vector<int> vec[MAXN];
queue<int> q;
bool vis[MAXN];
int step[MAXN];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, u, v;
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        cin >> u >> v;
        vec[u].push_back(v);
        vec[v].push_back(u); // Assuming the graph is undirected
    }

    q.push(1);
    vis[1] = true;
    step[1] = 0;

    while (!q.empty()) {
        int top = q.front();
        q.pop();

        for (int to : vec[top]) {
            if (!vis[to]) {
                vis[to] = true;
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }

    bool found = false;
    for (int i = 2; i <= n; i++) {
        if (step[i] != INF) {
            cout << step[i] << " ";
            found = true;
        } else {
            cout << "-1 ";
        }
    }

    if (!found) {
        cout << "-1";
    }

    return 0;
}
```

### Explanation:
1. **Graph Representation**: The graph is represented using an adjacency list `vec`.
2. **Initialization**:
   - We initialize the queue with the starting node (node 1).
   - We mark node 1 as visited (`vis[1] = true`) and set its distance to 0 (`step[1] = 0`).
3. **Breadth-First Search (BFS)**:
   - While the queue is not empty, we process each node.
   - For each neighbor of the current node, if it has not been visited, we mark it as visited, update its distance, and push it into the queue.
4. **Output**:
   - After the BFS completes, we iterate through all nodes and print their distances from node 1. If a node is unreachable, we print `-1`.

This solution ensures that we find the shortest paths efficiently using BFS and handles all edge cases appropriately.
