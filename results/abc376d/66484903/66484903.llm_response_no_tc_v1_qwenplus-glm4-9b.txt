#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
bool visited[MAXN];
ll distance[MAXN];

int main(){
    ll n, m;
    scanf("%lld %lld", &n, &m);
    
    // Initialize all distances to -1 (indicating unvisited)
    for (ll i = 1; i <= n; ++i) {
        distance[i] = -1;
    }
    
    for (ll i = 1; i <= m; ++i) {
        ll u, v;
        scanf("%lld %lld", &u, &v);
        vec[u].push_back(v);
    }
    
    // BFS initialization
    queue<ll> q;
    q.push(1);
    visited[1] = true;
    distance[1] = 0; // Distance to itself is 0
    
    while (!q.empty()) {
        ll current = q.front();
        q.pop();
        
        for (ll neighbor : vec[current]) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                distance[neighbor] = distance[current] + 1;
                q.push(neighbor);
                
                // If we reach node 1 again, print the distance
                if (neighbor == 1) {
                    printf("%lld\n", distance[neighbor]);
                    return 0;
                }
            }
        }
    }
    
    // If we exit the loop without printing, no path exists
    printf("-1\n");
    return 0;
}
