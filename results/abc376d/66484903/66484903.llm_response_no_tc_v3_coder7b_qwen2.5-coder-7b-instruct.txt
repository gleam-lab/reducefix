int step[MAXN];
bool vis[MAXN];
int main(){
    int n,m,u,v;
    scanf("%d %d",&n,&m);
    for(int i=1;i<=m;i++){
        scanf("%d %d",&u,&v);
        vec[u].push_back(v);
    }
    q.push(1);
    memset(step,-1,sizeof(step));
    step[1] = 0;
    while(!q.empty()){
        int top=q.front();
        vis[top]=true;
        q.pop();
        for(int i=0;i<vec[top].size();i++){
            int to=vec[top][i];
            if(step[to]==-1){
                step[to]=step[top]+1;
                q.push(to);
            }else if(vis[to]){
                printf("%d\n",step[top]-step[to]+1);
                return 0;
            }
        }
    }
    printf("-1\n");
    return 0;
}
