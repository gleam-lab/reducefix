typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
bool vis[MAXN];
ll step[MAXN];

int main(){
    ll n, m, u, v;
    scanf("%lld %lld", &n, &m);
    for(ll i = 0; i < m; ++i){
        scanf("%lld %lld", &u, &v);
        vec[u].push_back(v);
    }
    queue<ll> q;
    q.push(1);
    step[1] = 0; // Initialize the step count for the starting node

    while(!q.empty()){
        ll top = q.front();
        vis[top] = true;
        q.pop();

        for(ll i = 0; i < vec[top].size(); i++){
            ll to = vec[top][i];
            if(to == 1){
                printf("%lld\n", step[top] + 1);
                return 0;
            }
            if(!vis[to]){
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }

    printf("-1\n");
    return 0;
}
