The original code attempts to find the shortest cycle containing node 1 in a directed graph using BFS. However, it has several issues:
1. It doesn't handle the case where node 1 has no cycle (properly returns -1)
2. The BFS updates steps and marks nodes as visited too early, which can miss shorter paths
3. It checks for `to==1` after pushing to the queue, which is too late for some cases

Here's the fixed code that properly finds the shortest cycle containing node 1:

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
ll step[MAXN];

int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
    }
    
    queue<ll> q;
    fill(step, step+MAXN, INF);
    step[1] = 0;
    q.push(1);
    
    ll min_cycle = INF;
    
    while(!q.empty()){
        ll top = q.front();
        q.pop();
        
        for(ll to : vec[top]){
            if(to == 1){
                min_cycle = min(min_cycle, step[top] + 1);
            }
            if(step[to] > step[top] + 1){
                step[to] = step[top] + 1;
                q.push(to);
            }
        }
    }
    
    printf("%lld", (min_cycle == INF) ? -1 : min_cycle);
    return 0;
}
```

Key improvements:
1. Properly initializes step distances to INF
2. Checks for cycles (return to node 1) during neighbor processing
3. Updates step distances only when a shorter path is found
4. Handles the case where no cycle exists (returns -1)
5. Uses a cleaner BFS implementation with better variable names

The code now correctly finds the shortest cycle containing node 1 or returns -1 if no such cycle exists.
