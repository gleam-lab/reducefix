The graph is represented using an adjacency list (`vec`), and we need to find the shortest path from the vertex `1` to all other vertices. The adjacency list is built using `n` edges given as pairs of vertices `u` and `v`.

Your code has several issues:

2. The queue `q` should be initialized with the start node (vertex `1`), but it is not initialized properly.
3. You are using an adjacency list but iterating through it from index `0` to `vec[top].size()`, which is unnecessary because `vec[top]` is of type `vector<ll>` and might not have a fixed size.

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
vector<ll> vec[MAXN];
queue<ll> q;
ll step[MAXN] = {INF}; // Initialize with INF
bool vis[MAXN];
int main(){
    ll n,m,u,v;
    scanf("%lld %lld",&n,&m);
    for(ll i=1;i<=m;i++){
        scanf("%lld %lld",&u,&v);
        vec[u].push_back(v);
        vec[v].push_back(u); // Since it is undirected
    }
    q.push(1);
    step[1] = 0; // Starting point
    vis[1] = true;
    while(!q.empty()){
        ll top=q.front();
        vis[top] = true;
        q.pop();
        for(ll i=0;i<vec[top].size();i++){
            ll to=vec[top][i];
            if(to==1){
                printf("%lld",step[top]+1);
                return 0;
            }
            if(!vis[to]){
                step[to]=step[top]+1;
                q.push(to);
            }
        }
    }
    printf("-1");
    return 0;
}
```

Additionally, it adds an undirected edge in both directions when building the graph. The shortest path to the start node is always `0`, and you can return immediately upon reaching it.
