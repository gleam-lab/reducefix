// Function to detect cycle using DFS and keep track of path length
bool dfs(int node, int parent, int &cycleLength) {
    visited[node] = true;
    for(auto neighbor : vec[node]) {
        if (!visited[neighbor]) {
            if(dfs(neighbor, node, cycleLength)) {
                return true; // Cycle found
            }

            // Backedge found, update cycle length

            return true;
        }
    }
    return false;
}
