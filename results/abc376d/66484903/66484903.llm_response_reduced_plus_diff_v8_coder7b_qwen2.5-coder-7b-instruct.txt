#include <bits/stdc++.h>

typedef long long ll;
const int MAXN = 2e5 + 10;
vector<int> adj[MAXN];
bool visited[MAXN];
int dist[MAXN];

void bfs(int start) {
    queue<int> q;
    q.push(start);
    visited[start] = true;
    dist[start] = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (auto v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    int n, m, u, v;

    for (int i = 0; i < m; i++) {
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // Assuming undirected graph
    }

    memset(visited, false, sizeof(visited));
    memset(dist, -1, sizeof(dist));

    bfs(1);

    int target = 1;
    for (int i = 2; i <= n; i++) {
        if (dist[i] < dist[target]) {
            target = i;
        }
    }

    cout << dist[target] << endl;

    return 0;
}
