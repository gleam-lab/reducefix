            if(dist[nex] == -1){
                dist[nex] = dist[pos] + 1;

            }

        for(int v : G[u]){
            if(v == 0){
                ans = min(ans, dist[u] + 1);
            }
        }
    }
    
    if(ans == INT_MAX) cout << -1 << endl;

}
