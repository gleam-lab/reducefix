#include <bits/stdc++.h>
using namespace std;
const long long INF = 1LL << 60;
const long long MOD = 998244353LL;

int main(){
    int N, M;
    cin >> N >> M;
    vector<vector<int>> G(N);
    for(int i=0; i<M; i++){
        int u, v;
        cin >> u >> v;
        u--, v--;
        G[u].push_back(v);
    }

    // BFS to find shortest path from node 0 to all other nodes
    vector<int> dist(N, -1);
    queue<int> que;
    que.push(0);
    dist[0] = 0;
    
    while(!que.empty()){
        int pos = que.front();
        que.pop();
        for(int nex : G[pos]){
            if(dist[nex] == -1){
                dist[nex] = dist[pos] + 1;
                que.push(nex);
            }
        }
    }

    // Find minimum distance from any node that has a direct edge back to 0
    int ans = INT_MAX;
    for(int i = 0; i < N; i++){
        if(dist[i] != -1){
            for(int neighbor : G[i]){
                if(neighbor == 0){
                    ans = min(ans, dist[i]);
                    break;
                }
            }
        }
    }

    if(ans == INT_MAX){
        cout << -1 << endl;
    } else {
        cout << ans + 1 << endl; // +1 for the final edge back to 0
    }

    return 0;
}
