#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n;
    
    vector<long long> x(n + 2); // Use long long to prevent overflow issues
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }

    vector<long long> p(n + 2);
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
    }

    // Build prefix sum of p
    vector<long long> sum(n + 2);
    for (int i = 1; i <= n; ++i) {
        sum[i] = sum[i - 1] + p[i];
    }

    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;

        // Find first x >= L
        int l = lower_bound(x.begin() + 1, x.begin() + n + 1, L) - x.begin();

        // Find first x > R
        int r = upper_bound(x.begin() + 1, x.begin() + n + 1, R) - x.begin();

        if (l <= r - 1) {
            cout << sum[r - 1] - sum[l - 1] << '\n';
        } else {
            cout << 0 << '\n'; // No elements in range
        }
    }

    return 0;
}
