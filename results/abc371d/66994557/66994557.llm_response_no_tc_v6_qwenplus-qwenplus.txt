```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
    }
    
    // Prefix sum of p
    vector<long long> prefix_sum(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }

    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        // Find first index where x[i] >= L
        int l = lower_bound(x.begin() + 1, x.begin() + n + 1, L) - x.begin();
        // Find first index where x[i] > R
        int r = upper_bound(x.begin() + 1, x.begin() + n + 1, R) - x.begin();
        
        if (l <= r - 1) {
            cout << prefix_sum[r - 1] - prefix_sum[l - 1] << '\n';
        } else {
            cout << 0 << '\n'; // No elements in range
        }
    }
    
    return 0;
}
```
