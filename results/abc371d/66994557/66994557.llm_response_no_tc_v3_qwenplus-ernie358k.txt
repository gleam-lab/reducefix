```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1), sum(n + 1, 0);
    
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        // Since x[i] are the original values and we need to find their indices,
        // we use `lower_bound` and `upper_bound` on the sorted version of x.
        // However, since we directly use indices in `sum`, we should ensure the bounds are correct.
        // First, we need to sort a copy of x and then use these bounds to find the corresponding indices.
        vector<int> sorted_x(x.begin(), x.end());
        sort(sorted_x.begin(), sorted_x.end());
        
        auto it_l = lower_bound(sorted_x.begin(), sorted_x.end(), L);
        auto it_r = upper_bound(sorted_x.begin(), sorted_x.end(), R);
        
        int l = it_l - sorted_x.begin() + 1;  // Convert to 1-based index
        int r = it_r - sorted_x.begin();      // Upper bound gives the position just after the last occurrence, so no +1 here
        
        // If `r` is out of bounds (i.e., no element is greater than R), adjust it to n.
        if (r == n + 1) {
            r = n;
        }
        
        // Output the sum of p values for the range defined by L and R.
        cout << sum[r] - sum[l - 1] << endl;
    }
    
    return 0;
}
```
