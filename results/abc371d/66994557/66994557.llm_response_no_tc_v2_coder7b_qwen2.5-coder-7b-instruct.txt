#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
long long sum[200005];

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i]; // Corrected variable name to avoid confusion with array 'sum'
    }
    
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        int l = lower_bound(x + 1, x + n + 1, L) - x;
        int r = upper_bound(x + 1, x + n + 1, R) - x;
        
        if (l <= r && r <= n) { // Ensure the indices are within valid range
            cout << sum[r] - sum[l - 1] << endl;
        } else {
            cout << "0" << endl; // Handle out-of-range queries gracefully
        }
    }
    
    return 0;
}
