#include<bits/stdc++.h>
using namespace std;
int n, q, x[200005], p[200005], fs[200005], ls[200005];
int sum[2000005];
long long tsum[200005]; // Total sum array including prefix sums

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
    }
    for (int i = 1; i <= n; i++) {
        sum[i] = sum[i - 1] + p[i];
    }

    // Precompute prefix sums for lower bound optimization
    for (int i = 1; i <= n; i++) {
        tsum[i] = sum[i] + x[i]; // Total sum including p[i]
    }

    // Binary search to find the closest indices to the left of L and R
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        int l = 1, r = n, mid;
        while (l < r) {
            mid = l + (r - l + 1) / 2;
            if (x[mid] >= L) {
                r = mid;
            } else {
                l = mid;
            }
        }
        fs[l] = l == 1 ? 0 : tsum[l - 1];
        while (l > 1) {
            if (x[l] - x[l - 1] < R - L) {
                fs[l - 1] = tsum[l - 2];
                l--;
            } else {
                break;
            }
        }

        l = 1, r = n;
        while (l < r) {
            mid = l + (r - l + 1) / 2;
            if (x[mid] <= R) {
                l = mid;
            } else {
                r = mid;
            }
        }
        ls[l] = l == n ? tsum[n] : tsum[l];
        while (l < n) {
            if (x[l + 1] - x[l] < R - L) {
                ls[l + 1] = tsum[l];
                l++;
            } else {
                break;
            }
        }

        cout << (ls[l + 1] - fs[l]) << '\n';
    }
}
