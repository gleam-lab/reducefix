```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
int sum[200005];

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        int l = lower_bound(x + 1, x + n + 1, L) - x;
        int r = upper_bound(x + 1, x + n + 1, R) - x;
        if (l > r) {
            cout << "0\n"; // In case lower bound is greater than upper bound
        } else {
            cout << sum[r - 1] - sum[l - 1] << "\n";
        }
    }
    return 0;
}
```

### Explanation:
1. **Boundary and Edge Cases**: Added a check if `l > r` to handle cases where the range is invalid or `L` is greater than `R`. This avoids accessing `sum` out of bounds.
2. **Efficiency**: The use of `lower_bound` and `upper_bound` ensures that the solution runs in O(n log n) time complexity due to sorting and the binary search operations for each query.
3. **Code Readability**: The code is structured to be clear and maintainable, with appropriate variable names and separation of operations.
