#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, q;
    cin >> n;
    vector<int> x(n);
    for (int i = 0; i < n; i++) {
        cin >> x[i];
    }
    
    vector<int> p(n);
    for (int i = 0; i < n; i++) {
        cin >> p[i];
    }
    
    vector<long long> sum(n);
    sum[0] = p[0];
    for (int i = 1; i < n; i++) {
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        auto l = lower_bound(x.begin(), x.end(), L);
        auto r = upper_bound(x.begin(), x.end(), R);
        
        if (l == x.end() || *l > R) {
            cout << 0 << endl;
        } else {
            long long result = sum[*r - 1] - (l == x.begin() ? 0 : sum[l - 1]);
            cout << result << endl;
        }
    }
    
    return 0;
}
