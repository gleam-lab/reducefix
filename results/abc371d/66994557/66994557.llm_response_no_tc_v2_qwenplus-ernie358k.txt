```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1), sum(n + 1, 0);
    
    // Input the x array (with 1-based index)
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    // Input the p array (with 1-based index)
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        // Convert L and R to 0-based index for lower_bound and upper_bound
        auto it_l = lower_bound(x.begin() + 1, x.end(), L) - x.begin();
        auto it_r = upper_bound(x.begin() + 1, x.end(), R) - x.begin();
        
        // Print the sum of p values for the range [L, R)
        cout << sum[it_r] - sum[it_l] << endl;
    }
    
    return 0;
}
```
