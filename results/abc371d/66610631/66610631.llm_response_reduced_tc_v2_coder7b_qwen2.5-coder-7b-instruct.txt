```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 2e5 + 7;
vector<long long> prefix_sum(N, 0);

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    
    int n, q;
    cin >> n;
    vector<int> x(n + 1), p(n + 1);
    
    // Read input vectors
    for (int i = 1; i <= n; ++i) cin >> x[i];
    for (int i = 1; i <= n; ++i) cin >> p[i];
    
    // Calculate prefix sums
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }
    
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Adjust indices for 0-based indexing used internally
        --l, --r;
        
        // Query range sum using prefix sums
        long long result = prefix_sum[r] - prefix_sum[l];
        cout << result << '\n';
    }
    
    return 0;
}
```
