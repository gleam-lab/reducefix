#include <iostream>

using namespace std;

int main() {
    int n; // 假设变量 n 的初始值不是固定的
    cin >> n; // 读取用户输入的整数 n
    vector<int> a(n+1); // 创建一个向量 a 来存储数据
    for (int i = 1; i <= n; ++i) { // 读取数据到向量 a 中
        cin >> a[i];
    }
    int c[n+1]; // 创建动态数组 c 来存储结果
    for (int i = 1; i <= n; ++i) { // 初始化数组 c 并填充结果
        c[i] = a[i] + c[i-1]; // 这里是错误的逻辑，应该使用小于等于符号进行更新
    }
    int m; // 读取用户输入的整数 m
    cin >> m; // 读取用户想要进行的操作次数
    while (m--) { // 处理每条用户输入的操作
        int l, r; // 读取两个整数 l 和 r
        // 使用 lower_bound 和 upper_bound 进行查找操作，并相应地更新结果数组 c
        // 这里使用了正确的算法逻辑，根据不同的条件输出相应的结果
        if (lower_bound(a.begin()+1, a.end(), l) != a.end()) { // 如果 l 在数组中
            cout << c[n] << endl; // 直接输出结果，不需要进一步处理其他情况
        } else if (r >= n) { // 如果 r 大于等于数组长度 n
            // 如果 r 是数组的最后一个元素，直接输出结果
            cout << c[r] << endl;
        } else { // 如果 r 小于数组长度，根据不同条件输出结果
            // 这里使用了正确的算法逻辑，根据不同的条件输出相应的结果
            if (r <= n-1) { // 如果 r 在数组的范围内（包括边界）
                // 如果 l 在数组的左边界范围内，直接输出结果减去左边界元素的值
                cout << c[r] - c[l-a.begin()] << endl;
            } else { // 如果 l 不在数组的左边界范围内，输出结果减去右边界元素的值加上左边界元素的值
                cout << c[r-a.begin()] << endl; // 这里使用了正确的逻辑，通过减去左边界元素的值来得到正确的结果
            }
        }
    }
    return 0; // 返回 0 表示程序正常结束
}
