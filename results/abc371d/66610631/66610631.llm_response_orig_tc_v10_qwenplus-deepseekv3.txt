    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for(int i = 1; i <= n; i++) {
        cin >> b[i];
        c[i] = c[i-1] + b[i];  // Cumulative sum
    }
    
    cin >> m;
    while(m--) {
        int l, r;
        cin >> l >> r;
        
        auto l_it = lower_bound(a.begin()+1, a.end(), l);
        auto r_it = upper_bound(a.begin()+1, a.end(), r);
        
        int l_idx = l_it - a.begin();
        int r_idx = r_it - a.begin();
        
        ll sum = 0;
        if(l_idx <= r_idx) {
            if(l_idx > 0 && r_idx > 0) {
                sum = c[r_idx-1] - (l_idx > 1 ? c[l_idx-2] : 0);

                sum = c[r_idx-1];
            }
        }

    }
    
    return 0;
}
