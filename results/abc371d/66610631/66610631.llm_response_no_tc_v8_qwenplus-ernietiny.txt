#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, m;
    cin >> n; // Number of elements in the array a
    vector<int> a(n + 1); // Array a with initial values initialized
    for (int i = 1; i <= n; ++i) {
        cin >> a[i]; // Input elements for array a
    }
    
    // Calculate the modified value of c[n] based on the given range
    int modified_value = c(n); // Assuming c is a function that returns the value of array element in the given range
    
    while (m--) { // Process each range of elements
        int l, r; // Range of elements to be processed
        cin >> l >> r; // Input the lower and upper bounds of the range
        
        // Find the correct index to store the modified value
        auto it = lower_bound(a.begin() + 1, a.end(), l); // Use lower_bound to find the index of element within the given range
        if (it == a.end()) { // If the element is not found in the range, do nothing
            cout << modified_value << endl; // Output the original value if element is not present in the range
        } else { // If the element is found, calculate the modified value accordingly
            if (r == n) { // If the range is just one element, output the modified value directly
                cout << modified_value << endl;
            } else { // Otherwise, calculate the modified value by subtracting the difference between the original value and the modified value of the previous element in the range
                cout << (modified_value - c[it - a.begin() - 1]) << endl; // Subtracting the modified value of previous element to get the modified value for current range
            }
        }
    }
    return 0;
}
