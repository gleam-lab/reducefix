#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;

const int N = 1e6 + 7;
int n, m;
vector<int> a(N + 1, 0), b(N + 1, 0), c(N + 1, 0);

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> b[i];
        c[i] = b[i] + c[i - 1];
    }
    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;
        auto l_index = lower_bound(a.begin() + 1, a.end(), l) - (a.begin() + 1);
        auto r_index = upper_bound(a.begin() + 1, a.end(), r) - (a.begin() + 1) - 1;
        if (r_index == n) cout << c[n] << endl;
        else if (l_index == 0) cout << c[r_index] << endl;
        else if (r_index == n - 1) cout << c[n] - c[l_index] << endl;
        else cout << c[r_index] - c[l_index] << endl;
    }
    return 0;
}
