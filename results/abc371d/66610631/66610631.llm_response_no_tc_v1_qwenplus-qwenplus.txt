#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 1e6+7;

int main() {
    int n, m;
    cin >> n;
    vector<int> a(n+1, 0), b(n+1, 0), c(n+1, 0);
    
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    for(int i = 1; i <= n; ++i) {
        cin >> b[i];
        c[i] = b[i] + c[i-1];
    }
    
    cin >> m;
    while(m--) {
        int l, r;
        cin >> l >> r;
        
        // Find first index where a[i] >= l (inclusive)
        auto l_i = lower_bound(a.begin()+1, a.end(), l);
        // Find first index where a[i] > r (exclusive)
        auto r_i = upper_bound(a.begin()+1, a.end(), r);
        
        int left_idx = l_i - a.begin();
        int right_idx = r_i - a.begin();
        
        if (left_idx > n || right_idx < 2) {
            cout << 0 << endl;
            continue;
        }
        
        // We subtract 1 from both because prefix sums are 1-based
        int result = c[right_idx-1] - c[left_idx-1];
        cout << result << endl;
    }
    
    return 0;
}
