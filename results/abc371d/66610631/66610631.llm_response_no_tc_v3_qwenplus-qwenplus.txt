#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 1e6+7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n;
    
    vector<int> a(n+2, 0); // Extra space to handle boundaries easily
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    
    vector<ll> prefix_b(n+2, 0); // Use long long to avoid overflow
    for(int i = 1; i <= n; ++i) {
        cin >> prefix_b[i];
        prefix_b[i] += prefix_b[i-1];
    }
    
    cin >> m;
    while(m--) {
        int l, r;
        cin >> l >> r;
        
        // Find leftmost index where a[i] >= l
        int left_idx = lower_bound(a.begin()+1, a.end(), l) - a.begin();
        
        // Find rightmost index where a[i] <= r
        int right_idx = upper_bound(a.begin()+1, a.end(), r) - a.begin() - 1;
        
        if(left_idx > right_idx) {
            cout << 0 << "\n";
        } else {
            cout << prefix_b[right_idx] - prefix_b[left_idx - 1] << "\n";
        }
    }
    
    return 0;
}
