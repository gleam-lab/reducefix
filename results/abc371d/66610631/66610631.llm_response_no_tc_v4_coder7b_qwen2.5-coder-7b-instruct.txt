#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 2e5+7;
int main() {
	int n,q;
	cin >> n;
	vector<int>pos(n+1),pop(n+1),prefix_pop(n+1);
	for(int i=1;i<=n;i++){
		cin >> pos[i];
	}
	for(int i=1;i<=n;i++){
		cin >> pop[i];
		prefix_pop[i] = prefix_pop[i-1] + pop[i];
	}
	cin >> q;
	while(q--){
		int l,r;
		cin >> l >> r;
		int l_idx = lower_bound(pos.begin(),pos.end(),l)-pos.begin();
		int r_idx = upper_bound(pos.begin(),pos.end(),r)-pos.begin()-1;
		cout << (prefix_pop[r_idx]-(l_idx==0?0:prefix_pop[l_idx-1])) << endl;
	}
	return 0;
}
