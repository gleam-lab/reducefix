#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;

const int N = 1e6+7;
int a[N], b[N], c[N];

int main() {
    int n, m;
    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for(int i = 1; i <= n; i++) {
        cin >> b[i];
        c[i] = b[i] + (i > 1 ? c[i-1] : 0);
    }
    cin >> m;
    while(m--) {
        int l, r;
        cin >> l >> r;
        auto lb = lower_bound(a + 1, a + n + 1, l);
        auto rb = upper_bound(a + 1, a + n + 1, r);
        if(lb == a + n || rb == a + 1) {
            cout << c[n] << endl;
        } else if(lb == a + 1 && rb == a + n + 1) {
            cout << c[n] - c[0] << endl;
        } else {
            cout << c[rb - a - 1] - c[lb - a - 1] << endl;
        }
    }
    return 0;
}
