#include <bits/stdc++.h>
using namespace std;
using LL = long long;
const int N = 2e6 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<LL> a(n), b(n), prefix_sum(n + 1, 0);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n; ++i) {
        cin >> b[i];
        prefix_sum[i + 1] = prefix_sum[i] + b[i];
    }

    // Precompute prefix sums of b
    vector<LL> c(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        c[i] = c[i - 1] + b[i - 1];
    }

    int m;
    cin >> m;
    while (m--) {
        LL l, r;
        cin >> l >> r;

        // Find the range in a where values are between l and r inclusive
        auto left = lower_bound(a.begin(), a.end(), l);
        auto right = upper_bound(a.begin(), a.end(), r);

        // Compute the sum of corresponding b values using prefix sum
        int start = left - a.begin();
        int end = right - a.begin();

        cout << c[end] - c[start] << "\n";
    }

    return 0;
}
