#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 1e6 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n;

    vector<int> a(n + 1), b(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> b[i];
    }

    // Sort the values and build prefix sum based on sorted indices
    vector<pair<int, int>> pairs(n + 1); // index from 1 to n
    for (int i = 1; i <= n; ++i) {
        pairs[i] = {a[i], b[i]};
    }

    // sort the pairs by a's value
    sort(pairs.begin() + 1, pairs.end());

    // build prefix sum of b's value according to the sorted order
    vector<ll> prefix(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        prefix[i] = prefix[i - 1] + pairs[i].second;
    }

    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;

        // Binary search to find the range [left, right] in `pairs` with a[i] in [l, r]
        int left = 1, right = n;

        // Find first a[i] >= l
        int lo = 1, hi = n;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (pairs[mid].first < l)
                lo = mid + 1;
            else
                hi = mid - 1;
        }
        int L = lo;

        // Find last a[i] <= r
        lo = 1, hi = n;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (pairs[mid].first > r)
                hi = mid - 1;
            else
                lo = mid + 1;
        }
        int R = hi;

        if (R < L) {
            cout << "0\n";
        } else {
            cout << prefix[R] - prefix[L - 1] << "\n";
        }
    }

    return 0;
}
