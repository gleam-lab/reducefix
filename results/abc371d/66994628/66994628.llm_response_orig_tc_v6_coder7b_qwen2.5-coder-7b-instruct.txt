#include<bits/stdc++.h>
using namespace std;

int n, q;
vector<int> x, p, prefix_sum;

void preprocess() {
    sort(x.begin(), x.end());
    vector<int> sorted_x = x;
    map<int, int> rank;
    for (int i = 0; i < n; ++i) {
        if (!rank.count(sorted_x[i])) {
            rank[sorted_x[i]] = i + 1;
        }
    }
    for (int i = 0; i < n; ++i) {
        x[i] = rank[x[i]];
    }
}

int main() {
    cin >> n >> q;
    x.resize(n), p.resize(n);
    for (int i = 0; i < n; ++i) cin >> x[i];
    for (int i = 0; i < n; ++i) cin >> p[i];
    
    preprocess();
    
    prefix_sum.resize(n + 1);
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + p[i - 1];
    }

    while (q--) {
        int l, r;
        cin >> l >> r;
        l = lower_bound(x.begin(), x.end(), l) - x.begin() + 1;
        r = upper_bound(x.begin(), x.end(), r) - x.begin();
        cout << prefix_sum[r - 1] - prefix_sum[l - 1] << endl;
    }

    return 0;
}
