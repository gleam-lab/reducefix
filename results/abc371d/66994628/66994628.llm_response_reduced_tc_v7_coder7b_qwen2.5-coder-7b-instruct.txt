#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
long long sum[200005];

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(x + 1, x + n + 1, l);
        auto it_r = upper_bound(x + 1, x + n + 1, r);
        if (it_l == x + n + 1 || *it_l > r) it_l--;
        if (it_r == x + 1 || *it_r < l) it_r++;
        l = it_l - x;
        r = it_r - x;
        cout << sum[r] - sum[l] << endl;
    }
    return 0;
}
