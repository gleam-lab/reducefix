#include<bits/stdc++.h>
using namespace std;
int n, q;
int x[200005], p[200005];
long long sum[200005];

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        // Adjusting for 0-based indexing of array
        l = lower_bound(x + 1, x + n + 1, l) - x;
        r = upper_bound(x + 1, x + n + 1, r) - x;
        // Ensure l is at least 1 to avoid accessing sum[0]
        if (l == 0) l = 1;
        // Calculate the sum for the range [l, r)
        cout << sum[r - 1] - sum[l - 1] << endl;
    }
    return 0;
}
