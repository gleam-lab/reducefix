```cpp
#include<bits/stdc++.h>
using namespace std;
int n,q;
long long sum[100005]; // 假设输入的数字长度不超过这个范围

int main(){
	cin >> n; // 输入数据点数量
	for(int i = 1; i <= n; i++){ // 循环读取每个数据点
		long long x, p; // 分别存储数据点的数值和对应概率
		cin >> x >> p; // 读取数值和概率
		sum[i] = sum[i-1] + p * x; // 根据概率计算每个数据点的期望值
	}

	while(q--){ // 按照题目要求进行操作
		long long l, r; // 存储查询区间的左端点和右端点
		cin >> l >> r; // 读取查询区间左端点和右端点
		// 根据查询区间内的概率计算期望值差值
		long long diff = sum[r] - sum[l-1]; // 计算期望值差值
		cout << diff << endl; // 输出结果
	}
}
```
