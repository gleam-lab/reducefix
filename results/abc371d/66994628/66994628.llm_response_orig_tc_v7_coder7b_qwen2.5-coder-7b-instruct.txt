```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    
    int n, q;
    cin >> n;
    vector<long long> x(n + 1), p(n + 1), prefix_sum(n + 1);
    
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }
    
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Adjusting indices for zero-based indexing used internally
        --l, --r;
        
        long long result = prefix_sum[r] - prefix_sum[l];
        cout << result << '\n';
    }
    
    return 0;
}
```

This solution uses a `vector` instead of an array for dynamic memory management and handles input/output efficiently with `ios_base::sync_with_stdio(false)` and `cin.tie(0)`. The prefix sum is calculated once and reused for each query, ensuring efficient computation even for large inputs. The indices are adjusted to be zero-based within the function to maintain consistency with standard C++ practices.
