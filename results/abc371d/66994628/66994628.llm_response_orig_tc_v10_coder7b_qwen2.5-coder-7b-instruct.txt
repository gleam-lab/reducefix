```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
long long sum[200005];

int main() {
    cin >> n;
    for(int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for(int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i-1] + p[i];
    }
    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(x+1, x+n+1, l);
        auto it_r = upper_bound(x+1, x+n+1, r);
        l = it_l - x;
        r = it_r - x;
        if(l == 1 && r == n+1) {
            cout << sum[n] << endl;
        } else if(l == 1) {
            cout << sum[r-1] << endl;
        } else if(r == n+1) {
            cout << sum[n] - sum[l-2] << endl;
        } else {
            cout << sum[r-1] - sum[l-2] << endl;
        }
    }
    return 0;
}
```
