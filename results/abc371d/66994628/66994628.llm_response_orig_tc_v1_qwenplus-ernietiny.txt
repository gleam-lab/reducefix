```cpp
#include<bits/stdc++.h>
using namespace std;
int n,q;
long long sum[100005]; // 使用长整型数组来存储前缀和

int main(){
	cin>>n; // 输入数量
	long long a,b; // 定义长整型变量来存储查询区间左右端点
	for(int i=1;i<=n;i++){ // 遍历每一个查询区间
		cin>>a>>b; // 输入查询区间的左右端点
		sum[a] = sum[b] = sum[b-a] + n; // 初始化前缀和数组
	}
	while(q--){ // 对每一个查询进行操作
		cin>>x; // 输入要查询的区间左端点
		long long l = lower_bound(sum+1,sum+1+n,x)-sum; // 找到小于等于查询区间的左端点对应的下标
		long long r = upper_bound(sum+1,sum+1+n,x)-sum-1; // 找到大于查询区间的右端点对应的下标
		cout<<sum[r-1]-sum[l-1]<<endl; // 输出结果
	}
}
```
