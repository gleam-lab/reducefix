    }

        sum[i] = sum[i-1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int l_val, r_val;
        cin >> l_val >> r_val;
        int l = lower_bound(x.begin()+1, x.end(), l_val) - x.begin();

    return 0;
