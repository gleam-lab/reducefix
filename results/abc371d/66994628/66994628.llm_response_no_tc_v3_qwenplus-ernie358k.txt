#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1), sum(n + 1, 0);
    
    // Read x values, note that we use x[0] as a dummy to avoid off-by-one errors
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    // Read p values
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
    }
    
    // Compute prefix sum for p values
    for (int i = 1; i <= n; ++i) {
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Use lower_bound and upper_bound on x values
        auto it_l = lower_bound(x.begin() + 1, x.end(), l);
        auto it_r = upper_bound(x.begin() + 1, x.end(), r);
        
        int idx_l = it_l - x.begin();
        int idx_r = it_r - x.begin();
        
        // Print the result using prefix sums, adjust for 1-based indexing
        cout << sum[idx_r - 1] - sum[idx_l - 1] << endl;
    }
    
    return 0;
}
