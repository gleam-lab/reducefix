#include<bits/stdc++.h>
using namespace std;

int main(){
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1);
    for(int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    for(int i = 1; i <= n; ++i) {
        cin >> p[i];
    }
    
    // Prefix sum of p
    vector<long long> sum(n + 1, 0);
    for(int i = 1; i <= n; ++i) {
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        
        // Find first index where x[i] >= l
        int left = lower_bound(x.begin() + 1, x.end(), l) - x.begin();
        
        // Find first index where x[i] > r
        int right = upper_bound(x.begin() + 1, x.end(), r) - x.begin();
        
        if (left == n + 1 || x[left] > r) {
            cout << 0 << endl;
        } else {
            cout << sum[right - 1] - sum[left - 1] << endl;
        }
    }
    
    return 0;
}
