#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n;
    
    vector<long long> x(n + 2); // Extra space to avoid boundary checks
    vector<long long> p(n + 2);
    vector<long long> prefix_sum(n + 2, 0);

    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }

    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }

    cin >> q;

    while (q--) {
        long long l, r;
        cin >> l >> r;

        // Find first index where x[i] >= l
        int left_idx = lower_bound(x.begin() + 1, x.begin() + n + 1, l) - x.begin();

        // Find first index where x[i] > r
        int right_idx = upper_bound(x.begin() + 1, x.begin() + n + 1, r) - x.begin();

        // Calculate the sum using prefix sums
        long long result = prefix_sum[right_idx - 1] - prefix_sum[left_idx - 1];

        cout << result << '\n';
    }

    return 0;
}
