```cpp
#include<bits/stdc++.h>
using namespace std;
int n,q,x[200005],p[200005];
long long sum[200005];
int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>x[i];
    }
    for(int i=1;i<=n;i++){
        cin>>p[i];
        sum[i] = sum[i-1] + p[i];
    }
    cin>>q;
    while(q--){
        int l,r;
        cin>>l>>r;
        l=lower_bound(x+1,x+n+1,l)-x;
        r=upper_bound(x+1,x+n+1,r)-x;
        cout<<(long long)sum[r-1]-(l==1?0:sum[l-2])<<endl;
    }
    return 0;
}
```
