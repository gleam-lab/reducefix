```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, q;
    cin >> n;
    vector<int> x(n + 1), p(n + 1), prefix_sum(n + 1);
    
    for(int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    for(int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        
        // Adjusting indices because array is 1-based
        --l, --r;
        
        // Finding the correct range using binary search
        auto it_l = lower_bound(x.begin(), x.end(), l);
        auto it_r = upper_bound(x.begin(), x.end(), r);
        
        if(it_l == x.end() || *it_l > r) {
            it_l--;
        }
        
        if(it_r == x.begin()) {
            it_r++;
        } else {
            it_r--;
        }
        
        int start_index = distance(x.begin(), it_l) + 1;
        int end_index = distance(x.begin(), it_r);
        
        cout << prefix_sum[end_index] - prefix_sum[start_index - 1] << endl;
    }
    
    return 0;
}
```
