#include<bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
int n, q, x[MAXN], p[MAXN];
long long sum[MAXN];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    cin >> n;
    for(int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for(int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i-1] + p[i];
    }

    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(x+1, x+n+1, l);
        auto it_r = upper_bound(x+1, x+n+1, r);
        int idx_l = it_l - x;
        int idx_r = it_r - x;
        cout << sum[idx_r-1] - sum[idx_l-1] << '\n';
    }

    return 0;
}
