#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin >> n; // Number of elements
    
    // Create a sorted array for sum calculation
    vector<int> sum(n + 1);
    for (int i = 0; i < n; ++i) {
        cin >> sum[i]; // Each element will be added to the sum array
    }
    
    int fixed_sum = 0; // Fixed sum to compare with the expected output
    
    // Main logic starts here
    for (int i = 1; i <= n; ++i) {
        int lower_bound_val, upper_bound_val; // Boundary values for finding range in sorted array
        cin >> lower_bound_val >> upper_bound_val; // Read the range you want to compare with expected output
        
        // Calculate the difference between the fixed sum and the actual range
        fixed_sum -= sum[upper_bound_val - 1] - sum[lower_bound_val - 1]; // Avoid subtracting 0 and compare against the expected output
        
        // Output the difference if it's within the acceptable range
        if (fixed_sum >= 0) {
            cout << fixed_sum << endl; // The actual range is within the expected output range, so print the difference here
        }
    }
    return 0; // Return statement should not be necessary here as it's just a part of the solution.
}
