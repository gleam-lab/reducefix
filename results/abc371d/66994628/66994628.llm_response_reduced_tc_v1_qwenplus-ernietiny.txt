```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin >> n; // Number of elements
    
    // Create a sorted array for sum calculation
    vector<int> sum(n + 1);
    for (int i = 0; i < n; ++i) {
        cin >> sum[i]; // Elements for each position
    }
    
    // Sort the array to handle range queries efficiently
    sort(sum.begin(), sum.end());
    
    int q; // Number of queries
    cin >> q; // Number of queries to process
    
    while (q--) { // Process each query
        int l, r; // Left and right indices of the element to query
        cin >> l >> r; // Elements to query and their indices
        
        // Calculate the difference between the two sums at the given indices
        int diff = sum[r] - sum[l - 1]; // Assuming the array is sorted in ascending order
        cout << diff << endl; // Output the result
    }
    
    return 0; // Return 0 to indicate successful completion of the program
}
```
