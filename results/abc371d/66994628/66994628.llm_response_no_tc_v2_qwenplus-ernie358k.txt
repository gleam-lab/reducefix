#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    
    vector<int> x(n + 1), p(n + 1), sum(n + 1, 0);
    
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Find the indices of the closest elements in x array
        auto it_l = lower_bound(x.begin() + 1, x.end(), l) - x.begin();
        auto it_r = upper_bound(x.begin() + 1, x.end(), r) - x.begin();
        
        // Since we are using 1-based indexing for p array, adjust the indices accordingly
        // Ensure we do not go out of bounds and handle cases where no elements are found within range
        int adjusted_l = max(it_l, 1);
        int adjusted_r = min(it_r, n);
        
        // Print the sum of p values within the adjusted range
        cout << sum[adjusted_r] - sum[adjusted_l - 1] << endl;
    }
    
    return 0;
}
