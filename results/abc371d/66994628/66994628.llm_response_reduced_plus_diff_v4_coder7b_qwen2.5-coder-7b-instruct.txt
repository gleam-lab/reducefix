#include<bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int n, q;
    cin >> n;
    vector<int> x(n), p(n + 1);
    for(int i = 0; i < n; ++i) {
        cin >> x[i];
    }
    partial_sum(p.begin(), p.end(), p.begin());
    
    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        --l; --r;
        long long ans = p[r + 1] - p[l];
        cout << ans << '\n';
    }
    return 0;
}
