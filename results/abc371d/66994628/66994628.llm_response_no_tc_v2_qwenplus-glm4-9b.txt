#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200005;

int n, q;
int x[MAXN], p[MAXN], sum[MAXN];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        if (l > r) {
            swap(l, r);
        }
        auto lower_idx = lower_bound(x + 1, x + n + 1, l) - x;
        auto upper_idx = upper_bound(x + 1, x + n + 1, r) - x;
        if (lower_idx > upper_idx) {
            cout << "0" << endl;
        } else {
            cout << sum[upper_idx - 1] - sum[lower_idx - 1] << endl;
        }
    }
    return 0;
}
