#include <bits/stdc++.h>

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int N;
    cin >> N;
    vector<int> X(N);
    vector<ll> P(N+1, 0);
    rep(i, N) cin >> X[i];
    rep(i, N){
        int p;
        cin >> p;
        P[i+1] = P[i] + p;
    }

    // Coordinate compression (discretization)
    vector<int> sorted_X = X;
    sort(sorted_X.begin(), sorted_X.end());
    map<int, int> x_to_idx;
    int idx = 0;
    for (auto x : sorted_X) {
        if (!x_to_idx.count(x)) {
            x_to_idx[x] = idx++;
        }
    }

    // Build prefix sum over compressed coordinates
    vector<ll> compressed_sum(idx + 1, 0);
    rep(i, N) {
        compressed_sum[x_to_idx[X[i]] += P[i+1] - P[i];
    }

    // Prefix sum of compressed values
    vector<ll> prefix_sum(idx + 1, 0);
    for (int i = 1; i <= idx; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + compressed_sum[i - 1];
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;
        // Binary search to find the first X >= L and last X <= R
        auto l_iter = lower_bound(sorted_X.begin(), sorted_X.end(), L);
        auto r_iter = upper_bound(sorted_X.begin(), sorted_X.end(), R);

        int l_idx = (l_iter == sorted_X.end()) ? idx : x_to_idx[*l_iter];
        int r_idx = (r_iter == sorted_X.begin()) ? 0 : x_to_idx[*(prev(r_iter))];

        if (l_iter == sorted_X.end() || r_iter == sorted_X.begin() || L > R) {
            cout << 0 << endl;
        } else {
            cout << prefix_sum[r_idx + 1] - prefix_sum[l_idx] << endl;
        }
    }
}
