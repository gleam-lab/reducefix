    vector<ll> P(N+1, 0);
    rep(i, N) cin >> X[i];
    rep(i, N){
        ll p;
        cin >> p;
        P[i+1] = P[i] + p;
    }

    // Build prefix sum array and coordinate compression map
    vector<int> sorted_X = X;
    sort(all(sorted_X));
    map<int, int> compressed;
    int idx = 0;
    for (auto x : sorted_X) {
        if (!compressed.count(x)) {
            compressed[x] = idx++;
        }
    }

    // Build prefix sum per compressed coordinate
    vector<ll> prefix(idx + 1, 0);
    rep(i, N) {
        prefix[compressed[X[i]] += 1] = P[i + 1];
    }

    // For fast prefix sum queries
    vector<ll> acc(idx + 1, 0);
    for (int i = 1; i <= idx; ++i) {
        acc[i] = acc[i - 1] + prefix[i];
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;

        // Binary search to find lower and upper bounds in compressed coordinates
        auto l_iter = lower_bound(sorted_X.begin(), sorted_X.end(), L);
        auto r_iter = upper_bound(sorted_X.begin(), sorted_X.end(), R);

        int l_idx = (l_iter == sorted_X.end()) ? idx : compressed[*l_iter];
        int r_idx = (r_iter == sorted_X.end()) ? idx : compressed[*prev(r_iter)];

        if (l_iter == sorted_X.end()) {
            cout << 0 << endl;
            continue;
        }

        ll ans = acc[r_idx + 1] - acc[l_idx];
        cout << ans << endl;
    }

    return 0;
}
