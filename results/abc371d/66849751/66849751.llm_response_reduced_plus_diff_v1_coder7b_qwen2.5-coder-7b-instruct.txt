```cpp
#include <iostream>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N;
    cin >> N;
    vector<long long> A(N), P(N + 1);
    map<long long, int> pos_map;

    for(int i = 0; i < N; ++i) {
        cin >> A[i];
        P[i + 1] = P[i] + A[i];
        pos_map[A[i]] = i;
    }

    int Q;
    cin >> Q;
    while(Q--) {
        int L, R;
        cin >> L >> R;
        auto itL = pos_map.lower_bound(L);
        auto itR = pos_map.upper_bound(R);
        
        long long sum = 0;
        if(itL != pos_map.end()) {
            sum += P[itL->second + 1];
        }
        if(itR != pos_map.begin()) {
            --itR;
            sum -= P[itR->second + 1];
        }
        cout << sum << '\n';
    }

    return 0;
}
```
