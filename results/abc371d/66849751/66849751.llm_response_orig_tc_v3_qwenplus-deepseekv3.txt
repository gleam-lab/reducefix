### Approaches

#include <bits/stdc++.h>

using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<int> X(N);
    vector<ll> P(N + 1, 0);
    
    for (int i = 0; i < N; ++i) {
        cin >> X[i];
    }
    
    for (int i = 0; i < N; ++i) {
        int p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }

    vector<pair<int, int>> x_indexed(N);
    for (int i = 0; i < N; ++i) {
        x_indexed[i] = {X[i], i};
    }
    sort(x_indexed.begin(), x_indexed.end());

    vector<int> sorted_X(N);
    for (int i = 0; i < N; ++i) {
        sorted_X[i] = x_indexed[i].first;
    }

    int Q;
    cin >> Q;
    while (Q--) {
        int L, R;
        cin >> L >> R;
        
        auto it_l = lower_bound(sorted_X.begin(), sorted_X.end(), L);
        auto it_r = upper_bound(sorted_X.begin(), sorted_X.end(), R);
        
        int l = it_l - sorted_X.begin();
        int r = it_r - sorted_X.begin();
        
        ll ans = P[x_indexed[r - 1].second + 1] - P[x_indexed[l].second];
        cout << ans << '\n';
    }

    return 0;
}

1. **Input Handling**: The code reads the number of elements `N`, followed by the arrays `X` (values) and `P` (weights). The prefix sum array `P` is computed to store cumulative sums of weights.
2. **Sorting and Indexing**: The values in `X` are paired with their indices and sorted to facilitate binary search operations.
3. **Binary Search for Queries**: For each query `(L, R)`, the lower and upper bounds of the range are found using `lower_bound` and `upper_bound`. These bounds determine the indices of elements within the query range.
4. **Prefix Sum Calculation**: The sum of weights for elements within the query range is computed using the prefix sum array `P`, ensuring efficient O(1) range sum queries.
