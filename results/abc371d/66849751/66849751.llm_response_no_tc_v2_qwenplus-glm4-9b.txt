#include <bits/stdc++.h>

#define all(x) (x).begin(), (x).end()
#define pb(a) push_back(a)

int main() {
    int N;
    cin >> N;
    vector<int> P(N + 1, 0); // Initialize P with zeros
    vector<int> X(N);
    
    rep(i, N) cin >> X[i];
    rep(i, N) cin >> P[i + 1]; // Start from i+1 because P[0] is already set to 0

    // Precompute P
    rep(i, N) {
        P[i + 1] += P[i];
    }

    map<int, int> x_to_ptr;
    rep(i, N) {
        x_to_ptr.emplace(X[i], i);
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;
        auto X_l = x_to_ptr.lower_bound(L);
        auto X_r = x_to_ptr.upper_bound(R);

        if (X_l == x_to_ptr.begin() && X_l->first > L) {
            // No valid L in range
            cout << 0 << endl;
        } else {
            // There is at least one valid L in the range
            int l = (X_l == x_to_ptr.begin()) ? 0 : (*X_l).second + 1;
            int r = (X_r == x_to_ptr.end()) ? N : (*X_r).second;
            cout << P[r] - P[l] << endl;
        }
    }
}
