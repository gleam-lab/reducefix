#include <iostream>
#include <vector>
#include <map>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> P(N+1);
    P[0] = 0;
    int sum = 0;
    for (int i = 1; i <= N; ++i) {
        cin >> sum;
        P[i] = P[i - 1] + sum;
    }

    int Q;
    cin >> Q;
    while (Q--) {
        int L, R;
        cin >> L >> R;
        int low = 1, high = N;
        while (high - low > 1) {
            int mid = (low + high) / 2;
            if (L <= P[mid]) {
                high = mid;
            } else {
                low = mid;
            }
        }
        int left_idx = high;
        low = 1, high = N;
        while (high - low > 1) {
            int mid = (low + high) / 2;
            if (R <= P[mid]) {
                high = mid;
            } else {
                low = mid;
            }
        }
        int right_idx = high;
        cout << P[right_idx] - P[left_idx - 1] << endl;
    }
    return 0;
}
