#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;

#define rep(i,n) for (int i=0; i<(n); ++i)
#define reps(i,n) for (int i=0; i<=(n); ++i)
#define all(x) (x).begin(), (x).end()
#define rall(x) (x).rbegin(), (x).rend()
#define pb(a) push_back(a)

int main() {
    int N;
    cin >> N;
    vector<int> P(N + 1), X(N);
    
    rep(i, N) cin >> X[i];
    P[0] = 0;
    rep(i, N) {
        int p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }

    // Create a sorted vector of pairs (X[i], prefix_sum up to i)
    vector<pair<int, int>> points;
    rep(i, N) {
        points.emplace_back(X[i], P[i + 1]);
    }
    sort(all(points));

    // Prepare X and prefix sum arrays for binary search usage
    vector<int> sorted_X(N), prefix_sums(N + 1);
    rep(i, N) {
        sorted_X[i] = points[i].first;
        prefix_sums[i + 1] = points[i].second;
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;

        // Use binary search to find the range
        int left = lower_bound(sorted_X.begin(), sorted_X.end(), L) - sorted_X.begin();
        int right = upper_bound(sorted_X.begin(), sorted_X.end(), R) - sorted_X.begin();

        int ans = prefix_sums[right] - prefix_sums[left];
        cout << ans << endl;
    }

    return 0;
}
