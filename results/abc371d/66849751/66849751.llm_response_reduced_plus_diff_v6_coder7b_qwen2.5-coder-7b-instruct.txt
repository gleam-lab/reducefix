#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;
#define rep(i, n) for(int i = 0; i < (n); ++i)
#define all(x) (x).begin(), (x).end()

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<ll> A(N), P(N + 1);
    P[0] = 0;
    rep(i, N) {
        cin >> A[i];
        P[i + 1] = P[i] + A[i];
    }

    map<ll, int> pos;
    rep(i, N) {
        pos[P[i]] = i;
    }

    int Q;
    cin >> Q;
    while(Q--) {
        ll L, R;
        cin >> L >> R;
        auto itL = pos.lower_bound(L);
        auto itR = pos.upper_bound(R);
        if(itL == pos.end()) itL--;
        if(itR == pos.begin()) itR++;
        else itR--;

        ll ans = P[itR->first] - P[itL->first];
        cout << ans << '\n';
    }

    return 0;
}
