    rep(i, N) cin >> X[i];
    rep(i, N) {
        int p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }

    // Create a sorted vector of pairs (X[i], i) to allow binary search
    vector<pair<int, int>> x_index(N);
    rep(i, N) x_index[i] = {X[i], i};
    sort(all(x_index));

    // Extract sorted X values for binary search
    vector<int> sorted_X(N);
    rep(i, N) sorted_X[i] = x_index[i].first;

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;

        // Find left bound using lower_bound
        int left_idx = lower_bound(sorted_X.begin(), sorted_X.end(), L) - sorted_X.begin();
        
        // Find right bound using upper_bound
        int right_idx = upper_bound(sorted_X.begin(), sorted_X.end(), R) - sorted_X.begin();

        // Prefix sum query
        int ans = P[right_idx] - P[left_idx];
        cout << ans << endl;
    }
}
