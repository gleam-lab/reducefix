#include <bits/stdc++.h>

using namespace std;

int main(){
    int N;
    cin >> N; // Number of test cases
    vector<int> X(N); // Test case input
    for (int i = 0; i < N; ++i) cin >> X[i]; // Populate X array

    // Define mapping for efficient range queries
    map<int, int> x_to_idx; // Map X index to corresponding element in P array
    for (int i = 0; i < N; ++i) x_to_idx[X[i]] = i + 1; // Index X with original values

    // Calculate patched values in P array
    for (int i = 1; i <= N; ++i) P[i] = P[i - 1] + X[i - 1]; // Adding patched values for each query range

    int Q; // Number of queries
    cin >> Q; // Number of queries to process

    // Patched queries with range checks and calculation
    while (Q--) {
        int L, R; // Query ranges to calculate answer
        cin >> L >> R; // Query ranges in descending order of X index
        auto it = x_to_idx.lower_bound(L); // Find index of the element to query
        if (it == end(x_to_idx)) { // If element not found, return default value (0 in this case)
            cout << 0 << endl; // Return default value for this case
            continue; // Skip to the next query range
        }
        int ans = P[R] - P[it->second]; // Calculate the answer for the patched value range
        cout << ans << endl; // Output the answer for the query range
    }

    return 0; // Return the correct solution code block
}
