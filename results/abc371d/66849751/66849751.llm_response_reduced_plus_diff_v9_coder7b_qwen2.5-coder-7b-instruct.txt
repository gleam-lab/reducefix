```cpp
#include <iostream>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    vector<long long> X(N);
    vector<long long> P(N + 1);
    map<long long, int> x_to_ptr;

    for(int i = 0; i < N; ++i) {
        cin >> X[i];
        P[i + 1] = P[i] + X[i];
    }

    for(int i = 0; i < N; ++i) {
        x_to_ptr[X[i]] = i;
    }

    int Q;
    cin >> Q;
    while(Q--) {
        int L, R;
        cin >> L >> R;
        auto it_L = x_to_ptr.lower_bound(L);
        auto it_R = x_to_ptr.upper_bound(R);

        long long left_sum = (it_L == x_to_ptr.begin()) ? 0 : P[it_L->second];
        long long right_sum = P[it_R == x_to_ptr.end() ? N : it_R->second];

        cout << right_sum - left_sum << '\n';
    }

    return 0;
}
```
