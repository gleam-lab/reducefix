#include <bits/stdc++.h>

using namespace std;
using ll = long long;

int main() {
    int N;
    cin >> N;
    vector<ll> P(N + 1, 0);
    vector<int> X(N);
    for (int i = 0; i < N; ++i) {
        cin >> X[i];
    }
    for (int i = 0; i < N; ++i) {
        int p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }

    int Q;
    cin >> Q;
    rep(i, Q) {
        int L, R;
        cin >> L >> R;
        auto X_l = lower_bound(X.begin(), X.end(), L);
        auto X_r = upper_bound(X.begin(), X.end(), R);

        int l = distance(X.begin(), X_l);
        int r = distance(X.begin(), X_r);

        cout << P[r] - P[l] << endl;
    }

    return 0;
}
