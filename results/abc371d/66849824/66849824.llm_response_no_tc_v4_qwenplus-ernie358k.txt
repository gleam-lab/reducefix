#include <iostream>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;

#define rep(i, n) for (int i = 0; i < (n); ++i)
#define all(x) (x).begin(), (x).end()

int dx[4] = {1, 0, -1, 0};
int dy[4] = {0, 1, 0, -1};

int main() {
    int N;
    cin >> N;
    vector<ll> P(N + 1), X(N);
    P[0] = 0;
    rep(i, N) cin >> X[i];
    rep(i, N) {
        ll p; // Changed int to ll to match the type of P array
        cin >> p;
        P[i + 1] = P[i] + p;
    }
    map<int, int> x_to_index; // Renamed from x_to_ptr for clarity
    rep(i, N) {
        x_to_index.emplace(X[i], i);
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;
        auto it_l = x_to_index.lower_bound(L);
        int l = (it_l == x_to_index.end()) ? N : it_l->second + 1; // +1 because we need the next index in the prefix sum array
        auto it_r = x_to_index.upper_bound(R);
        int r = (it_r == x_to_index.end()) ? N : it_r->second; // No +1 here because we need the index within the range

        // If l is still N after adjustment, it means all points are to the right of L
        // If r is still N, it means all points are to the left of R (but since we're using upper_bound, it naturally includes R=max case)
        // We use P[r] - P[l-1] to get the sum of weights between indices l-1 and r-1 (inclusive of corresponding X values L to R)
        ll ans = (l <= r) ? P[r] - P[l - 1] : 0; // Ensure l <= r to avoid invalid range access
        cout << ans << endl;
    }
}
