#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;

#define rep(i,n) for (int i=0; i<(n); ++i)
#define reps(i,n) for (int i=0; i<=(n); ++i)
#define all(x) (x).begin(), (x).end()
#define rall(x) (x).rbegin(), (x).rend()
#define pb(a) push_back(a)
#define Yes(b) cout << ((b)?"Yes":"No") << endl
#define YES(b) cout << ((b)?"YES":"NO") << endl

int main(){
    int N;
    cin >> N;
    vector<ll> P(N+1), X(N);
    P[0] = 0;
    rep(i, N) cin >> X[i];
    rep(i, N){
        ll p;
        cin >> p;
        P[i+1] = P[i] + p;
    }
    
    // Create a sorted vector of pairs instead of map to handle binary search correctly
    vector<pair<ll, ll>> x_p(N);
    rep(i, N) {
        x_p[i] = {X[i], i};
    }
    sort(all(x_p));

    // Build prefix sums of P and prepare indexed_X
    vector<ll> indexed_P(N+1);
    rep(i, N) {
        indexed_P[i+1] = P[i+1];
    }

    int Q;
    cin >> Q;
    rep(q, Q){
        ll L, R;
        cin >> L >> R;

        // Binary search on the sorted vector
        int left_idx = lower_bound(all(x_p), make_pair(L, -1LL)) - x_p.begin();
        int right_idx = upper_bound(all(x_p), make_pair(R, numeric_limits<ll>::max())) - x_p.begin();

        ll ans = 0;
        if (left_idx < right_idx) {
            ans = indexed_P[right_idx] - indexed_P[left_idx];
        }
        cout << ans << endl;
    }
}
