#include <iostream>
#include <vector>
#include <map>

using namespace std;
using ll = long long;

const int MAXN = 1e5 + 5;

int dx[4] = {1, 0, -1, 0};
int dy[4] = {0, 1, 0, -1};

int main() {
    int N;
    cin >> N;
    vector<ll> P(MAXN);
    vector<int> X(N);
    P[0] = 0;
    for (int i = 0; i < N; ++i) {
        cin >> X[i];
    }
    for (int i = 0; i < N; ++i) {
        int p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }

    map<int, int> x_to_ptr;
    for (int i = 0; i < N; ++i) {
        x_to_ptr[X[i]] = i;
    }

    int Q;
    cin >> Q;
    for (int q = 0; q < Q; ++q) {
        int L, R;
        cin >> L >> R;
        auto X_l = x_to_ptr.lower_bound(L);
        auto X_r = x_to_ptr.upper_bound(R);
        int l = (X_l != x_to_ptr.begin()) ? (*prev(X_l)).second : 0;
        int r = (X_r != x_to_ptr.end()) ? (*X_r).second : N - 1;
        ll ans = P[r] - P[l];
        cout << ans << endl;
    }
    return 0;
}
