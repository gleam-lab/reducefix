    int N;
    cin >> N;
    vector<ll> A(N+1);
    vector<ll> P(N+1);
    A[0] = 0;
    P[0] = 0;
    rep(i,N){
        cin >> A[i+1];
        P[i+1] = P[i] + A[i+1];
    }
    
    map<ll, int> pos;
    for(int i=1;i<=N;++i){
        pos[A[i]] = i;
    }

    int Q;
    cin >> Q;
    while(Q--){
        int L, R;
        cin >> L >> R;
        int left = pos[L];
        int right = pos[R];
        
        if(left > right){
            swap(left,right);
        }
        
        ll sum = P[right] - P[left];
        cout << sum << '\n';
    }
    return 0;
}
