```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;

#define rep(i,n) for(int i=0;i<n;++i)
#define all(x) (x).begin(), (x).end()

int main() {
    int N, Q;
    cin >> N >> Q;
    vector<ll> X(N), P(N);
    rep(i, N) cin >> X[i];
    rep(i, N) cin >> P[i];

    // Calculate prefix sum array P
    rep(i, N-1) P[i+1] += P[i];

    while(Q--) {
        int L, R;
        cin >> L >> R;
        // Find the first position that is not less than L
        auto it_L = lower_bound(all(X), L);
        // Find the first position that is greater than R
        auto it_R = upper_bound(all(X), R);

        // If L is larger than any element in X, no villages are within range
        if(it_L == X.end()) {
            cout << 0 << '\n';
        } else {
            int idx_L = it_L - X.begin();
            int idx_R = it_R - X.begin() - 1;
            // Calculate the number of villagers in the range [L, R]
            ll result = P[idx_R];
            if(idx_L > 0) result -= P[idx_L - 1];
            cout << result << '\n';
        }
    }

    return 0;
}
```
