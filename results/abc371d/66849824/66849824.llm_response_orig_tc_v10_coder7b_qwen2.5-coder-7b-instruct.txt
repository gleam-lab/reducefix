        P[i+1] = P[i] + p;
    }

        x_to_ptr[X[i]] = i;
    }

        ll L, R;
        cin >> L >> R;
        auto it_l = x_to_ptr.lower_bound(L);
        auto it_r = x_to_ptr.upper_bound(R);
        if(it_l == x_to_ptr.end() || (*it_l).first > R){
            it_l = prev(x_to_ptr.end());
        }
        if(it_r != x_to_ptr.begin() && (*prev(it_r)).first >= L){
            it_r = prev(it_r);
        }
        int l = (*it_l).second;
        int r = (*it_r).second;
        ll ans = P[r+1] - P[l];
        cout << ans << endl;
    }
}
