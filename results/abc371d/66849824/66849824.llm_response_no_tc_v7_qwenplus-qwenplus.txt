#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;

#define rep(i,n) for (int i=0; i<(n); ++i)
#define reps(i,n) for (int i=0; i<=(n); ++i)
#define all(x) (x).begin(), (x).end()
#define rall(x) (x).rbegin(), (x).rend()
#define pb(a) push_back(a)
#define Yes(b) cout << ((b)?"Yes":"No") << endl
#define YES(b) cout << ((b)?"YES":"NO") << endl

int dx[4] = {1, 0, -1, 0};
int dy[4] = {0, 1, 0, -1};

int main() {
    int N;
    cin >> N;
    vector<ll> P(N + 1), X(N);
    
    P[0] = 0;
    rep(i, N) cin >> X[i];
    rep(i, N) {
        int p;
        cin >> p;
        P[i+1] = P[i] + p;
    }

    // Map from position value to index in sorted order
    map<int, int> x_to_idx;
    vector<int> sorted_indices(N);
    iota(sorted_indices.begin(), sorted_indices.end(), 0); // Fill with 0, 1, ..., N-1
    sort(sorted_indices.begin(), sorted_indices.end(), [&](int a, int b) {
        return X[a] < X[b];
    });

    rep(i, N) {
        x_to_idx[X[sorted_indices[i]]] = i;
    }

    // Build prefix sum array based on sorted indices
    vector<ll> sorted_P(N + 1);
    sorted_P[0] = 0;
    rep(i, N) {
        sorted_P[i+1] = sorted_P[i] + P[sorted_indices[i]+1] - P[sorted_indices[i]];
    }

    int Q;
    cin >> Q;
    rep(q, Q) {
        int L, R;
        cin >> L >> R;

        // Find left and right bounds in the sorted array
        auto l_it = x_to_idx.lower_bound(L);
        auto r_it = x_to_idx.upper_bound(R);

        int left = l_it == x_to_idx.end() ? N : l_it->second;
        int right = r_it == x_to_idx.end() ? N : r_it->second;

        cout << sorted_P[right] - sorted_P[left] << endl;
    }
}
