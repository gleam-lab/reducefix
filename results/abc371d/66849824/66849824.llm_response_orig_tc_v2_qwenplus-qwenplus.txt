        P[i + 1] = P[i] + p;

    vector<pair<ll, int>> sorted_X(N);

    // Build prefix sum array based on sorted X
    vector<ll> sorted_P(N + 1);
    sorted_P[0] = 0;
    rep(i, N) sorted_P[i + 1] = sorted_P[i] + P[i + 1] - P[i];

        auto left = lower_bound(sorted_X.begin(), sorted_X.end(), make_pair(L, 0));

        int l_idx = left - sorted_X.begin();
        int r_idx = right - sorted_X.begin();

        // Query prefix sum
        ll ans = sorted_P[r_idx] - sorted_P[l_idx];
        cout << ans << "\n";
    }

    return 0;
}
