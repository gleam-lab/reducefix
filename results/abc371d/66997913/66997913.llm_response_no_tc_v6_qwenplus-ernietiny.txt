#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n; // 获取数组长度
  vector<int> x(n); // 创建输入数组
  for (int i = 0; i < n; i++) cin >> x[i]; // 读取每个元素并存储到数组中
  
  // 处理p向量，保证p中元素的值在范围内
  for (int i = 0; i < n; i++) {
    if (x[i] < p[i]) { // 如果p中的元素小于当前x的元素，更新p的值
      p[i] = max(p[i], x[i] + 1); // p中元素的当前值应为当前x元素加上1后的最大值
    }
  }
  
  int q; // 获取询问次数
  cin >> q; // 读取询问次数
  vector<ll> s(q+1); // 创建输出向量，用于存储答案
  for (int i = 0; i < q; i++) { // 对每个询问进行处理
    int l, r; // 读取左端点和右端点值
    cin >> l >> r; // 读取左端点和右端点的值
    // 由于需要找到右端点值在数组中的下标，需要先对p向量进行排序
    sort(p.begin(), p.begin() + n); // 对p向量进行排序，确保下标正确
    int rt = upper_bound(x.begin(), x.end(), r) - x.begin(); // 获取右端点在数组中的下标
    ll ans = s[rt] - s[lower_bound(x.begin(), x.end(), l) - x.begin()]; // 根据排序后的p向量计算答案
    cout << ans << endl; // 输出答案
  }
  return 0;
}
