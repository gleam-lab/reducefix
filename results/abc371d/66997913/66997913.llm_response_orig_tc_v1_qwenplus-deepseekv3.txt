using namespace std;
using ll = long long;

    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    vector<int> x(n);
    for (int i = 0; i < n; i++) cin >> x[i];
    
    vector<ll> p(n), s(n+1);
    for (int i = 0; i < n; i++) {
        cin >> p[i];
        s[i+1] = s[i] + p[i];
    }
    
    int q;
    cin >> q;
    for (int i = 0; i < q; i++) {
        int l, r;
        cin >> l >> r;
        int rt = upper_bound(x.begin(), x.end(), r) - x.begin();
        int lt = lower_bound(x.begin(), x.end(), l) - x.begin();
        ll ans = s[rt] - s[lt];
        cout << ans << '\n';
    }
    
    return 0;
}
