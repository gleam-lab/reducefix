#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  sort(x.begin(), x.end()); // Sort the array to use binary search effectively
  
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i]+p[i];
  }
  
  int q;
  cin >> q;
  for (int i = 0; i < q; i++) {
    int l, r;
    cin >> l >> r;
    
    // Find the rightmost index of any element <= r
    auto it_r = upper_bound(x.begin(), x.end(), r);
    if (it_r != x.begin()) --it_r; // Adjust if no such element exists
    
    // Find the leftmost index of any element >= l
    auto it_l = lower_bound(x.begin(), x.end(), l);
    
    int ans = s[it_r - x.begin()] - s[it_l - x.begin()];
    cout << ans << endl;
  }
  return 0;
}
