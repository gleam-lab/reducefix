#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i] + p[i];
  }

  int q;
  cin >> q;
  while(q--) {
    int l, r;
    cin >> l >> r;
    int lt = lower_bound(begin(x), end(x), l) - begin(x);
    int rt = upper_bound(begin(x), end(x), r) - begin(x);
    ll ans = s[rt] - s[lt];
    cout << ans << '\n';
  }
  return 0;
}
