#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);

  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  sort(x.begin(), x.end());
  vector<ll> s(n + 1);
  for (int i = 0; i < n; i++) {
    cin >> s[i + 1];
    s[i + 1] += s[i];
  }

  int q;
  cin >> q;
  while (q--) {
    int l, r;
    cin >> l >> r;
    auto itl = lower_bound(x.begin(), x.end(), l);
    auto itr = upper_bound(x.begin(), x.end(), r);
    int lt = distance(x.begin(), itl);
    int rt = distance(x.begin(), itr);
    if (lt == rt) {
      cout << 0 << '\n';
    } else {
      cout << s[rt] - s[lt] << '\n';
    }
  }
  return 0;
}
