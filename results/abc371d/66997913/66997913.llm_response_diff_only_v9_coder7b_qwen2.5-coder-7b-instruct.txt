    cin >> s[i];
    s[i+1] = s[i] + s[i]; // Assuming s[i] should be added to the previous sum, correct if logic differs
  }

  cin >> q;
  while (q--) {
    int l, r;
    cin >> l >> r;
    auto it1 = lower_bound(x.begin(), x.end(), l); // Find first element not less than l
    auto it2 = upper_bound(x.begin(), x.end(), r); // Find first element greater than r
    
    int idx1 = it1 - x.begin(); // Convert iterator to index
    int idx2 = it2 - x.begin(); // Convert iterator to index
    
    if (idx1 >= n || idx2 <= 0) continue; // Check for out-of-bound indices

}
