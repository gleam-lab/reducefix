#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i] + p[i];
  }
  
  int q;
  cin >> q;
  for (int i = 0; i < q; i++) {
    int l, r;
    cin >> l >> r;
    // Correct the bounds by adding 1 to the lower bound
    int rt = upper_bound(x.begin(), x.end(), r) - x.begin() + 1;
    int lt = lower_bound(x.begin(), x.end(), l) - x.begin() + 1;
    // Ensure that rt is not out of bounds and that lt is not zero
    if (rt > n || lt <= 0) {
      cout << "0" << endl;
    } else {
      ll ans = s[rt] - s[lt - 1];
      cout << ans << endl;
    }
  }
  return 0;
}
