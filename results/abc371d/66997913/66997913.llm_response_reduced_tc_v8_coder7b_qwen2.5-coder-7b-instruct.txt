```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);
  
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i] + p[i];
  }
  
  int q;
  cin >> q;
  while(q--) {
    int l, r;
    cin >> l >> r;
    auto it_l = lower_bound(x.begin(), x.end(), l);
    auto it_r = upper_bound(x.begin(), x.end(), r);
    int lt = distance(x.begin(), it_l);
    int rt = distance(x.begin(), it_r);
    int ans = s[rt] - s[lt];
    cout << ans << '\n';
  }
  return 0;
}
```
