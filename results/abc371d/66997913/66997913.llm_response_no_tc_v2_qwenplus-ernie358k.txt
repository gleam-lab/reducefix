```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> x(n);
    for (int i = 0; i < n; i++) cin >> x[i];
    vector<ll> p(n), s(n + 1, 0);
    for (int i = 0; i < n; i++) {
        cin >> p[i];
        s[i + 1] = s[i] + p[i];
    }
    
    sort(x.begin(), x.end()); // Sorting x to use binary search correctly
    
    int q;
    cin >> q;
    for (int i = 0; i < q; i++) {
        int l, r;
        cin >> l >> r;
        auto it_r = upper_bound(x.begin(), x.end(), r); // Find the first element greater than r
        auto it_l = lower_bound(x.begin(), x.end(), l); // Find the first element greater than or equal to l
        
        // Since we sorted x, we need to adjust the indices accordingly
        // it_l points to the first element >= l, so no change needed for left bound
        // But it_r points to the first element > r, so we decrement it to get the largest <= r (if exists)
        if (it_r != x.begin()) it_r--;
        
        // Calculate the indices in the sorted array which correspond to the original positions
        // (However, since we are only interested in the prefix sums and bounds, we don't actually need these indices back)
        // Instead, we directly use the prefix sums array s
        
        // `it_l - x.begin()` gives the index in the sorted array which is the smallest index >= l in the original array
        // `it_r - x.begin() + 1` gives the index in the sorted array which is just past the largest index <= r in the original array
        // Since s[0] is 0, s[it_l - x.begin()] will give the cumulative sum up to the smallest index >= l
        // And s[it_r - x.begin() + 1] will give the cumulative sum up to just past the largest index <= r
        ll ans = s[min(static_cast<int>(it_r - x.begin()) + 1, n)] - s[max(0, static_cast<int>(it_l - x.begin()))];
        cout << ans << endl;
    }
    return 0;
}
```
