```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n; // 假设输入的数字数量是固定的
  vector<int> x(n); // 存储每个数字的向量
  for (int i = 0; i < n; i++) cin >> x[i]; // 读取每个数字

  vector<ll> p(n), s(n+1); // p存储每个数字对应的分数，s存储对应的总和
  for (int i = 0; i < n; i++) { // 遍历每个数字对应的分数
    cin >> p[i]; // 读取分数
    s[i+1] = s[i] + p[i]; // 更新总和
  }

  int q; // 询问次数
  cin >> q; // 询问次数
  while (q--) { // 对于每个询问进行操作
    int l, r; // 询问的两个数字
    cin >> l >> r; // 读取两个数字
    int ans = s[r] - s[l-1]; // 使用索引来获取对应的答案
    cout << ans << endl; // 输出结果
  }
  return 0;
}
```
