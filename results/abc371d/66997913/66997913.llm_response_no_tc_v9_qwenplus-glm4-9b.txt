#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> x(n);
    for (int i = 0; i < n; i++) cin >> x[i];
    vector<ll> p(n), s(n+1);
    for (int i = 0; i < n; i++) {
        cin >> p[i];
        s[i+1] = s[i] + p[i];
    }
    
    int q;
    cin >> q;
    for (int i = 0; i < q; i++) {
        int l, r;
        cin >> l >> r;
        // Use upper_bound to find the position for r and lower_bound for l
        auto rt = upper_bound(x.begin(), x.end(), r) - x.begin();
        auto lt = lower_bound(x.begin(), x.end(), l) - x.begin();
        
        // Calculate the sum of subarray from l to r
        ll ans = s[rt] - s[lt];
        cout << ans << endl;
    }
    return 0;
}
