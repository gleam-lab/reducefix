#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false); cin.tie(nullptr);
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i]+p[i];
  }
  
  int q;
  cin >> q;
  while(q--) {
    int l, r;
    cin >> l >> r;
    auto it_l = upper_bound(x.begin(), x.end(), l-1);
    auto it_r = lower_bound(x.begin(), x.end(), r);
    int idx_l = it_l - x.begin();
    int idx_r = it_r - x.begin();
    int ans = s[idx_r] - s[idx_l];
    cout << ans << '\n';
  }
  return 0;
}
