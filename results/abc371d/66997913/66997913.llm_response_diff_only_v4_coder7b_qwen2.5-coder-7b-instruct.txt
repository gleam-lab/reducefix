#include <bits/stdc++.h>

using ll = long long;

int main() {
  int n;

  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  sort(x.begin(), x.end()); // Ensure x is sorted for binary search to work correctly
  
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i] + p[i]; // Accumulate prefix sums
  }
  
  int q;
  cin >> q;
  while(q--) { // Use a loop instead of a for loop with q iterations
    int l, r;
    cin >> l >> r;
    auto itl = lower_bound(x.begin(), x.end(), l); // Find first element not less than l
    auto itr = upper_bound(x.begin(), x.end(), r); // Find first element greater than r
    
    int idxl = itl - x.begin(); // Calculate index for lower bound
    int idxr = itr - x.begin(); // Calculate index for upper bound
    
    if(idxl == n || x[idxl] > r) idxl--; // Adjust if out of bounds or greater than r
    if(idxr == 0 || x[idxr-1] < l) idxr++; // Adjust if out of bounds or less than l
    
    ll ans = s[min(idxr, n)] - s[max(0, idxl)]; // Calculate answer using adjusted indices
    cout << ans << endl;
  }
  return 0;
}
