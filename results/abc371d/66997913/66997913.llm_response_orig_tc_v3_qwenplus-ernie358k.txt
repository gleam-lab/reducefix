using namespace std;
using ll = long long;

const int MAXN = 200005; // Maximum possible value of n + some extra for safety

struct SegmentTree {
    vector<ll> tree;
    int n;

    SegmentTree(int _n) : n(_n) {
        tree.resize(4 * n, 0);
    }

    void build(const vector<ll>& arr, int node = 1, int start = 0, int end = 0) {
        if (start == end) {
            tree[node] = arr[start];
        } else {
            int mid = (start + end) / 2;
            build(arr, node * 2, start, mid);
            build(arr, node * 2 + 1, mid + 1, end);
            tree[node] = tree[node * 2] + tree[node * 2 + 1];
        }
    }

    ll queryRange(int L, int R) {
        return queryRangeUtil(L, R, 1, 0, n - 1);
    }

private:
    ll queryRangeUtil(int L, int R, int node, int start, int end) {
        if (R < start || end < L) return 0;
        if (L <= start && end <= R) return tree[node];

        int mid = (start + end) / 2;
        ll leftSum = queryRangeUtil(L, R, node * 2, start, mid);
        ll rightSum = queryRangeUtil(L, R, node * 2 + 1, mid + 1, end);
        return leftSum + rightSum;
    }
};
