#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  int n;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  sort(x.begin(), x.end()); // Sort the array to use binary search
  vector<ll> p(n), s(n+1);
  for (int i = 0; i < n; i++) {
    cin >> p[i];
    s[i+1] = s[i] + p[i]; // Prefix sum array
  }
  
  int q;
  cin >> q;
  for (int i = 0; i < q; i++) {
    int l, r;
    cin >> l >> r;
    int rt = upper_bound(x.begin(), x.end(), r) - x.begin(); // Find first element greater than r
    int lt = lower_bound(x.begin(), x.end(), l) - x.begin(); // Find first element not less than l
    if (lt >= n || x[lt] > r) lt--; // Adjust if the found index is out of bounds or does not satisfy the condition
    int ans = s[rt] - s[lt + 1]; // Correctly calculate the range sum using prefix sums
    cout << ans << endl;
  }
  return 0;
}
