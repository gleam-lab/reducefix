#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
  int n;
  cin >> n; // Number of elements in the vector x
  vector<int> x(n);
  for (int i = 0; i < n; i++) cin >> x[i]; // Read elements from x
  
  // Create the two vectors for storing modified sums
  vector<ll> p(n), s(n+1);
  // Fill p with the modified sums from the original input
  for (int i = 0; i < n; i++) {
    p[i] = s[i+1] - x[i]; // Calculate the modified sum
  }
  
  int q;
  cin >> q; // Number of queries
  vector<pair<int, int>> query; // To store the query results
  for (int i = 0; i < q; i++) {
    int l, r; // Query values l and r to be compared
    cin >> l >> r; // Read the query values
    // Calculate the answer for the current query using binary search on modified sum array
    int ans = binary_search(s.begin(), s.end(), r - x[upper_bound(x.begin(), x.end(), l) - x.begin()]); // Upper_bound is used to find the index of the minimum element in the modified sum array
    cout << ans << endl; // Output the answer for the current query
  }
  return 0;
}
