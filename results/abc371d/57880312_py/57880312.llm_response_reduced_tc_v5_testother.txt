import sys
import bisect

def main():
    input = sys.stdin.read
    data = input().split()
    
    N = int(data[0])
    X = list(map(int, data[1:1 + N]))
    P = list(map(int, data[1 + N:1 + 2 * N]))
    
    # Create the prefix sum array
    prefix = [0] * (N + 1)  # Make it 1-indexed with extra space to avoid index checks
    for i in range(N):
        prefix[i + 1] = prefix[i] + P[i]
    
    # Handle the queries
    results = []
    query_start_index = 1 + 2 * N
    for i in range(Q := int(data[query_start_index])):
        Li = int(data[query_start_index + 1 + 2 * i])
        Ri = int(data[query_start_index + 1 + 2 * i + 1])
        
        # Find leftmost index >= Li
        left_index = bisect.bisect_left(X, Li)
        # Find rightmost index <= Ri
        right_index = bisect.bisect_right(X, Ri) - 1
        
        if left_index <= right_index and right_index >= 0 and left_index < N:
            total_villagers = prefix[right_index + 1] - prefix[left_index]
            results.append(total_villagers)
        else:
            results.append(0)
    
    # Print all results for the queries
    sys.stdout.write('\n'.join(map(str, results)) + '\n')

if __name__ == "__main__":
    main()
