#include <iostream>
#include <vector>
#include <map>

using namespace std;
using ll = long long;

int main() {
    int N;
    cin >> N;
    vector<ll> P(N + 1, 0); // Initialize P with 0
    vector<int> X(N);
    
    for (int i = 0; i < N; ++i) {
        cin >> X[i];
    }
    
    for (int i = 0; i < N; ++i) {
        ll p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }
    
    // Mapping X values to their respective indices
    map<int, int> x_to_ptr;
    for (int i = 0; i < N; ++i) {
        x_to_ptr[X[i]] = i;
    }
    
    int Q;
    cin >> Q;
    
    while (Q--) {
        int L, R;
        cin >> L >> R;
        
        // Using upper_bound and lower_bound to find the closest indices
        auto X_l = x_to_ptr.lower_bound(L);
        auto X_r = x_to_ptr.upper_bound(R);
        
        // If the range goes out of the bounds, adjust accordingly
        if (X_l == x_to_ptr.begin()) {
            cout << 0 << endl; // If L is less than the smallest key
        } else {
            --X_l; // Go back to the last key less than L
            cout << P[x_to_ptr[X_l]] - P[x_to_ptr[L]] << endl;
        }
        
        if (X_r == x_to_ptr.end()) {
            cout << 0 << endl; // If R is greater than the largest key
        } else {
            cout << P[x_to_ptr[X_r]] - P[x_to_ptr[R]] << endl;
        }
    }
    
    return 0;
}
