        cin >> X[i];
        cin >> P[i + 1];
        P[i + 1] += P[i];
    }
    map<ll, int> x_to_p;
    rep(i, N) {
        x_to_p[X[i]] = P[i + 1];
    }

    int Q;

        ll L, R;
        cin >> L >> R;
        auto it_l = x_to_p.lower_bound(L);
        auto it_r = x_to_p.upper_bound(R);
        ll ans = (it_r != end(x_to_p)) ? it_r->second : P[N];
        if(it_l != begin(x_to_p)) {
            --it_l;
            ans -= it_l->second;
        }
        cout << ans << endl;
    }
    return 0;
}
