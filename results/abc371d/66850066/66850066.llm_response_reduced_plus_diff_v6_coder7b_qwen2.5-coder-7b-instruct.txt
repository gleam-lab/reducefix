#include <bits/stdc++.h>

    int N;

    vector<ll> P(N + 1), X(N);
    P[0] = 0;
    rep(i, N) cin >> X[i];
    rep(i, N) {
        ll p;
        cin >> p;
        P[i + 1] = P[i] + p;
    }
    map<int, int> x_to_ptr;
    rep(i, N) {
        x_to_ptr[X[i]] = i;
    }

    int Q;
    cin >> Q;
    while (Q--) {
        int L, R;
        cin >> L >> R;
        auto it_l = x_to_ptr.lower_bound(L);
        auto it_r = x_to_ptr.upper_bound(R);
        if (it_l == x_to_ptr.end()) {
            L = x_to_ptr.rbegin()->second + 1;
        } else {
            L = it_l->second;
        }
        if (it_r == x_to_ptr.begin()) {
            R = 0;
        } else {
            --it_r;
            R = it_r->second + 1;
        }
        cout << P[R] - P[L] << endl;
    }

    return 0;
}
