#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using Graph = vector<vector<int>>;

#define rep(i,n) for (int i=0; i<(n); ++i)
#define reps(i,n) for (int i=0; i<=(n); ++i)
#define all(x) (x).begin(), (x).end()
#define rall(x) (x).rbegin(), (x).rend()
#define pb(a) push_back(a)
#define Yes(b) cout << ((b)?"Yes":"No") << endl

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int N;
    cin >> N;
    vector<ll> P(N+1), X(N);
    P[0] = 0;
    rep(i, N) cin >> X[i];
    rep(i, N){
        ll p;
        cin >> p;
        P[i+1] = P[i] + p;
    }

    // Build prefix sum of values
    vector<pair<ll, ll>> sortedX_P;
    rep(i, N) {
        sortedX_P.emplace_back(X[i], P[i+1]);
    }
    sort(all(sortedX_P));

    // Extract just the X for binary search
    vector<ll> sortedX;
    rep(i, N) {
        sortedX.push_back(sortedX_P[i].first);
    }

    int Q;
    cin >> Q;
    rep(q, Q){
        ll L, R;
        cin >> L >> R;
        
        auto it_l = lower_bound(sortedX.begin(), sortedX.end(), L);
        auto it_r = upper_bound(sortedX.begin(), sortedX.end(), R);

        int l_idx = it_l - sortedX.begin();
        int r_idx = it_r - sortedX.begin();

        ll ans = sortedX_P[r_idx-1].second - sortedX_P[l_idx].second + 
                 (l_idx == 0 ? P[0] : 0); // handle edge case when l_idx == 0

        cout << ans << endl;
    }
}
