#include<bits/stdc++.h>
using namespace std;
int n,q,x[200005],p[200005];
map<int,int>sum;
int main(){
	cin>>n;
    for(int i=1;i<=n;i++){
		cin>>x[i];
	}
	for(int i=1;i<=n;i++){
		cin>>p[i];
	     sum[x[i]]=sum[x[i]]+p[i];
	}
	cin>>q;
	while(q--){
		int L,R;
		cin>>L>>R;
	   auto it_l=sum.lower_bound(L);
	   auto it_r=sum.upper_bound(R);
		cout<<(it_r->second-it_l->second)<<endl;	
	}
}
