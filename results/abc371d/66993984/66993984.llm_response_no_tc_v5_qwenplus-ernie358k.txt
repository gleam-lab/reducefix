#include <iostream>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    vector<int> x(n + 1), p(n + 1); // Using vector for better memory management and bounds checking
    map<int, int> prefixSum;

    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }

    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefixSum[x[i]] += p[i]; // Using map to accumulate prefix sums based on x values
    }

    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;

        // We need to find the prefix sum for all x values in range [L, R]
        // Since we use a map, prefixSum[R] will give the sum up to and including R
        // and prefixSum.lower_bound(L) - 1 will give the sum just before L (if L exists)
        auto itL = prefixSum.lower_bound(L);
        int sumL = (itL == prefixSum.begin()) ? 0 : (--itL)->second;
        int sumR = prefixSum.count(R) ? prefixSum.at(R) : (prefixSum.lower_bound(R) != prefixSum.end() ? prefixSum.at(prev(prefixSum.lower_bound(R))->first) : 0);
        // If R is not in the map, we take the sum just before R if it exists, otherwise 0

        // Correcting for edge case where L is less than the smallest x value
        if (itL == prefixSum.begin() && L < itL->first) {
            sumL = 0;
        }

        // Correcting for edge case where R is greater than the largest x value
        if (prefixSum.count(R) == 0 && R > (--prefixSum.end())->first) {
            sumR = prefixSum.rbegin()->second;
        }

        cout << sumR - sumL << endl;
    }

    return 0;
}

### Explanation of Fixes:
1. **Data Structures**: Changed arrays to vectors for better memory management and bounds checking. Used a `map` to store prefix sums based on the values of `x`.
2. **Prefix Sum Calculation**: Accumulated prefix sums directly into the map based on the `x` values. This allows for efficient range sum queries.
3. **Range Query**: Used `lower_bound` and `upper_bound` logic adapted for the map to find the correct prefix sums for the range `[L, R]`. Handled edge cases where `L` is less than the smallest `x` value and `R` is greater than the largest `x` value.
4. **Edge Cases**: Ensured that if `L` is smaller than the smallest key in the map, the sum before `L` is zero. Similarly, handled the case where `R` is greater than the largest key in the map by taking the sum up to the largest key present.
5. **Code Readability**: Improved code readability by using vector and map instead of raw arrays, and by adding comments to explain key sections.
