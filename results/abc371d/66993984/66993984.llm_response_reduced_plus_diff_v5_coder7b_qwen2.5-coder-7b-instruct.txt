```cpp
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
int x[MAXN], p[MAXN];
long long sum[MAXN];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    cin >> n;
    for(int i = 1; i <= n; ++i) cin >> x[i];
    for(int i = 1; i <= n; ++i) cin >> p[i];
    
    // Calculate prefix sums
    for(int i = 1; i <= n; ++i) sum[i] = sum[i-1] + p[i];
    
    cin >> q;
    while(q--) {
        int L, R;
        cin >> L >> R;
        
        // Adjust indices since array is 1-indexed
        --L, --R;
        
        // Calculate the sum of elements between L and R
        long long result = sum[R] - sum[L];
        cout << result << '\n';
    }
    
    return 0;
}
```
