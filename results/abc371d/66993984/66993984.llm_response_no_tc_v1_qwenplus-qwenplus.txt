```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
long long sum[200005];  // Use prefix sums instead of map for efficiency

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    
    // Prefix sum initialization
    sum[0] = 0;
    for(int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int L, R;
        cin >> L >> R;
        
        // Find first index where x[i] >= L
        int c = lower_bound(x + 1, x + n + 1, L) - x;
        // Find first index where x[i] > R
        int d = upper_bound(x + 1, x + n + 1, R) - x;
        
        // All elements from c to d-1 are in the range [L, R]
        cout << sum[d - 1] - sum[c - 1] << '\n';
    }
    
    return 0;
}
```
