#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
int x[MAXN], p[MAXN];
long long sum[MAXN];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int n, q;
    cin >> n;
    for(int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for(int i = 1; i <= n; ++i) {
        cin >> p[i];
        sum[i] = sum[i-1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int L, R;
        cin >> L >> R;
        
        auto it1 = lower_bound(x + 1, x + n + 1, L);
        auto it2 = upper_bound(x + 1, x + n + 1, R);
        
        int idx1 = it1 - x;
        int idx2 = it2 - x;
        
        long long result = sum[idx2 - 1] - sum[idx1 - 1];
        cout << result << '\n';
    }
    
    return 0;
}
