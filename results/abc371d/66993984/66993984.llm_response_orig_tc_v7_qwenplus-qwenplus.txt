#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n;
    
    vector<long long> x(n + 2); // Positions (1-based)
    vector<long long> p(n + 2); // Weights (1-based)
    vector<long long> prefix_sum(n + 2); // Prefix sums of p

    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }

    cin >> q;
    while (q--) {
        long long L, R;
        cin >> L >> R;

        // Find first index where x[i] >= L
        auto left = lower_bound(x.begin() + 1, x.begin() + n + 1, L);
        // Find last index where x[i] <= R
        auto right = upper_bound(x.begin() + 1, x.begin() + n + 1, R);

        int l_idx = left - x.begin();
        int r_idx = (right - x.begin()) - 1;

        if (l_idx > r_idx) {
            cout << 0 << '\n';
        } else {
            cout << prefix_sum[r_idx] - prefix_sum[l_idx - 1] << '\n';
        }
    }

    return 0;
}
