```cpp
#include<bits/stdc++.h>
using namespace std;
int n,q,x[200005],p[200005];
vector<pair<int,int>> v;
map<int,int> sum;

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> x[i];
        v.push_back({x[i], i});
    }
    sort(v.begin(), v.end());
    
    for(int i = 1; i <= n; i++) {
        p[v[i-1].second] = i;
    }

    for(int i = 1; i <= n; i++) {
        if(i > 1) sum[i] = sum[i-1];
        sum[i] += p[i];
    }

    cin >> q;
    while(q--) {
        int l, r;
        cin >> l >> r;
        int cl = lower_bound(v.begin(), v.end(), {l, INT_MAX}) - v.begin();
        int cr = upper_bound(v.begin(), v.end(), {r, INT_MAX}) - v.begin() - 1;
        cout << sum[cr] - sum[cl-1] << '\n';
    }
    return 0;
}
```
