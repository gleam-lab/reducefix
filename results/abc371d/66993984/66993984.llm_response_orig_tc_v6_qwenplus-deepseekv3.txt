The issue with the original code is that it's using integer variables to store cumulative sums which can overflow when dealing with large inputs (200000 elements with potentially large values). The incorrect output shows negative numbers which is a clear sign of integer overflow. We need to use 64-bit integers (long long) to handle the large sums properly.

Here's the fixed code:

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, q, x[200005], p[200005];
ll sum[200005];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for(int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i-1] + p[i];
    }
    
    cin >> q;
    while(q--) {
        int L, R;
        cin >> L >> R;
        int c = lower_bound(x+1, x+n+1, L) - x;
        int d = upper_bound(x+1, x+n+1, R) - x;
        cout << sum[d-1] - sum[c-1] << '\n';
    }
    return 0;
}
```

3. Added `ios_base::sync_with_stdio(false)` and `cin.tie(NULL)` for faster I/O
4. Changed `endl` to `'\n'` for faster output
