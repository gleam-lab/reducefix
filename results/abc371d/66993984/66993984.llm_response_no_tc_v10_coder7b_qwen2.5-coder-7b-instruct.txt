		int L,R;
		cin>>L>>R;
		int result = 0;
		for(auto it=sum.begin();it!=sum.end();++it){ // Iterate over all unique elements in x
			if(it->first >= L && it->first <= R){
				result += it->second; // Add their contribution to the result if within the range [L, R]
			}
		}
