The issue with the original code appears to be integer overflow when calculating and storing the prefix sums. The input values can be very large (up to 200000 elements with potentially large values), so we need to use a larger data type to store the prefix sums.

#include<bits/stdc++.h>
using namespace std;
int n,q,x[200005],p[200005];
long long sum[200005]; // Changed to long long to prevent overflow

    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>x[i];
    }
    for(int i=1;i<=n;i++){
        cin>>p[i];
        sum[i]=sum[i-1]+p[i]; // Calculate prefix sum
    }
    cin>>q;
    while(q--){
        int L,R;
        cin>>L>>R;
        int c=lower_bound(x+1,x+n+1,L)-x;
        int d=upper_bound(x+1,x+n+1,R)-x;
        cout<<sum[d-1]-sum[c-1]<<endl;
    }
    return 0;
}
