        sum[i + 1] = sum[i] + p[i];
    }
    cin >> q;
    while (q--) {

        cin >> L >> R;
        if (L > R) {
            swap(L, R); // Ensure L is less than or equal to R
        }
        int c = lower_bound(x, x + n, L) - x;

            cout << 0 << endl; // If no elements are included, sum is 0
