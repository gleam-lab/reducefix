#include<bits/stdc++.h>
using namespace std;
int n, q, x[200005], p[200005];
map<int, int> sum;
map<int, int> pos;

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
        pos[x[i]] = i;  // Store the position of each element
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        sum[i] = sum[i - 1] + p[i];  // Calculate prefix sums
    }
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        int l = pos[L];  // Find the position of L in x
        int r = pos[R];  // Find the position of R in x
        cout << sum[r] - sum[l - 1] << endl;  // Calculate the range sum
    }
}
