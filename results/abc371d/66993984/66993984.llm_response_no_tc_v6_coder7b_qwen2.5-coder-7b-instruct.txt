```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
map<int, int> prefixSum;

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefixSum[i] = prefixSum[i - 1] + p[i];
    }
    
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        // Adjusting indices to match the 0-based index of the map
        auto itL = prefixSum.lower_bound(L);
        auto itR = prefixSum.upper_bound(R);
        
        // If there's no element greater than or equal to L, itL will be end()
        // Similarly, if all elements are less than R, itR will be end()
        if (itL == prefixSum.end()) {
            cout << "0\n";
            continue;
        }
        
        // If there's an element greater than R, we need to subtract the value at that position from the upper bound
        if (itR != prefixSum.begin()) --itR;
        
        cout << (*itR).second - (*itL).second << '\n';
    }
    return 0;
}
```
