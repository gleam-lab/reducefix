```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
map<int, int> prefixSum;

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefixSum[x[i]] += p[i]; // Accumulate prefix sums based on x values
    }

    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        int total = 0;
        for (int i = L; i <= R; ++i) {
            total += prefixSum[i]; // Sum up the prefix sums for the range [L, R]
        }
        cout << total << endl;
    }
    return 0;
}
```
