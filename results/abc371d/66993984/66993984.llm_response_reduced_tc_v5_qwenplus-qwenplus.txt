#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, q;
ll x[200005], p[200005];
ll prefix_sum[200005];

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n;
    for(int i = 1; i <= n; ++i){
        cin >> x[i];
    }
    for(int i = 1; i <= n; ++i){
        cin >> p[i];
        prefix_sum[i] = prefix_sum[i-1] + p[i];
    }
    
    cin >> q;
    while(q--){
        int L, R;
        cin >> L >> R;
        
        // Find first index where x[i] >= L
        int left = lower_bound(x + 1, x + n + 1, L) - x;
        // Find first index where x[i] > R
        int right = upper_bound(x + 1, x + n + 1, R) - x;
        
        cout << prefix_sum[right - 1] - prefix_sum[left - 1] << endl;
    }
    
    return 0;
}
