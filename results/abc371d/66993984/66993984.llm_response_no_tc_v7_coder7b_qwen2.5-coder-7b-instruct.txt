        auto it_c = lower_bound(x + 1, x + n + 1, L);
        int c = it_c - x;

        // Find the first position strictly greater than R
        auto it_d = upper_bound(x + 1, x + n + 1, R);
        int d = it_d - x;

        // Adjust indices to match the problem's requirement
        if (it_c != x + 1 && (*it_c == L)) --c;
