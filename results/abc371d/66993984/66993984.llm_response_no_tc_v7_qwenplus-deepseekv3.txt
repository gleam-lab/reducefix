        prefix_sum[i] = prefix_sum[i-1] + p[i];

    while(q--) {
        int L, R;
        cin >> L >> R;
        
        // Find first x[i] >= L
        int left = lower_bound(x.begin()+1, x.end(), L) - x.begin();
        // Find first x[i] > R
        int right = upper_bound(x.begin()+1, x.end(), R) - x.begin();
        
        // The valid range is [left, right-1]

}
