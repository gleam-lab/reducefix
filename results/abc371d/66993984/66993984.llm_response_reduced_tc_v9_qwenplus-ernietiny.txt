#include <bits/stdc++.h>

int main() {

    map<int, int> sum;
    for (int i = 1; i <= n; ++i) {
        sum[i] = sum[i - 1]; // Initialize sum with the previous cumulative sum
    }
    
    // Read patches
    for (int i = 0; i < n; ++i) {
        int patch;
        cin >> patch; // Patch values will be positive integers
        sum[patch]++; // Add patch to its corresponding sum entry
    }
    
    // Calculate the expected difference between the sorted subarray and the index of the minimum element
    int expectedDiff = sum[n] - n - 1; // Index of the sorted subarray minus one due to sorted order and index 1 (due to sum[n])
    
    // Perform the requested queries
    while (true) {
        int L, R; // Range to query
        cin >> L >> R; // Range query values will be positive integers
        
        // Check if the range is empty or has only one element, and return the expected output accordingly
        if (sum.empty() || sum.size() == 1) {
            cout << expectedDiff << endl; // Return expected output if the range is valid
            break; // Exit loop if range is valid and output is correct
        }

        int currentIndex = lower_bound(sum.begin(), sum.end(), L) - sum.begin(); // Index of sorted subarray ending at L-th element in sorted order

        cout << patchDifference << endl; // Output the expected difference if it's within acceptable range or an error otherwise
    }
    return 0; // Return a valid exit code from the main function to indicate successful completion of program execution
}
