```cpp
#include<bits/stdc++.h>
using namespace std;

int n, q, x[200005], p[200005];
map<int, int> prefix_sum;

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        prefix_sum[x[i]] += p[i];
    }
    
    cin >> q;
    while (q--) {
        int L, R;
        cin >> L >> R;
        
        // Find the first element greater than or equal to L
        auto it1 = prefix_sum.lower_bound(L);
        // Find the first element strictly greater than R
        auto it2 = prefix_sum.upper_bound(R);
        
        if (it1 != prefix_sum.end()) {
            cout << (*it1).second;
        } else {
            cout << 0;
        }
        
        if (it2 != prefix_sum.begin()) {
            --it2;
            cout << " " << (*it2).second;
        } else {
            cout << " 0";
        }
        
        cout << endl;
    }
    
    return 0;
}
```

This solution uses `std::map` to store the prefix sums of the array `x`. The `lower_bound` function is used to find the smallest key that is not less than `L`, and the `upper_bound` function is used to find the smallest key that is greater than `R`. The difference between these two values gives the required sum.

The output format ensures that both the start and end values are printed correctly. If there are no valid keys in the range, it prints 0 accordingly. This approach handles all edge cases and provides the correct output for the given problem.
