#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<int> a(n + 1), b(n + 1);
    vector<ll> c(n + 1, 0);

    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> b[i];
        c[i] = c[i - 1] + b[i];
    }

    int m;
    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;

        auto it_l = lower_bound(a.begin() + 1, a.end(), l);
        auto it_r = upper_bound(a.begin() + 1, a.end(), r);

        int i = it_l - a.begin();
        int j = it_r - a.begin();

        if (i > n) {
            cout << "0\n";
        } else if (j == 0) {
            cout << "0\n";
        } else {
            if (j > n) {
                j = n;
            }
            cout << c[j] - c[i - 1] << '\n';
        }
    }

    return 0;
}

1. **Input Reading**: The code reads the size `n` of the arrays, followed by the arrays `a` and `b`. The cumulative sum array `c` is computed from `b`.
2. **Query Processing**: For each query, it reads the range `[l, r]` and uses `lower_bound` to find the first element in `a` that is not less than `l` and `upper_bound` to find the first element greater than `r`.
3. **Index Handling**: The indices `i` and `j` are derived from the iterators returned by the binary search functions. If `i` exceeds the array size, the result is zero. 4. **Output**: The result for each query is the difference between the cumulative sums at positions `j` and `i-1`, which gives the sum of elements in `b` corresponding to the elements in `a` that fall within the range `[l, r]`.

This approach ensures that the solution is efficient and handles large input sizes and ranges correctly without overflow.
