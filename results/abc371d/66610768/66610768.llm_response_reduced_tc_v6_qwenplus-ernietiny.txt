#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 1e6 + 7;

int main() {
    int n, m;
    cin >> n; // Number of elements in array a
    vector<int> a(n + 1); // Preallocate space for array a
    for (int i = 0; i < n; ++i) {
        cin >> a[i]; // Read elements from array a
    }
    vector<int> c(n + 2); // To store intermediate results
    for (int i = 0; i < n; ++i) { // Process elements one by one
        c[i + 1] = c[i] + a[i]; // Compute new value for element i+1
    }
    while (m--) { // Process each test case
        int l, r; // Read low and high bounds of the range to compare
        cin >> l >> r; // Read the range [l, r] to compare with c[]
        // Search for the minimum value in the first subarray (up to l)
        auto i = lower_bound(a.begin(), a.end(), l); // Using std::lower_bound for O(N log N) performance
        if (r >= a[n]) { // If range is within the current subarray (range starts at l)
            cout << c[n] - c[i - 1] << endl; // Output the difference of current subarray value and c[] value
        } else { // If range extends beyond current subarray, use upper_bound for linear search time complexity
            cout << c[r - 1] - c[i - 2] << endl; // Output the difference of next subarray value and c[] value
        }
    }
    return 0;
}
