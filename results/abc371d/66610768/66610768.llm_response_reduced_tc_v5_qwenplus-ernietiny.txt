#include<bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n; // Number of elements in the array a
    vector<int> a(n+1); // Create an array of size n+1 to store the elements
    for (int i = 1; i <= n; ++i) {
        cin >> a[i]; // Read the elements from the array
    }
    
    // Sort the array in ascending order to handle range queries efficiently
    sort(a.begin(), a.end());
    
    int m; // Number of queries
    cin >> m; // Number of queries to process
    
    while (m--) { // Process each query
        int l, r; // Range queries: left and right indices
        cin >> l >> r; // Read the left and right indices of the range to be checked
        // Calculate the difference between the target value and the current value in the range
        int target = a[r] - a[l]; // Calculate target value for the range query
        // Calculate the difference between the previous value and the current value in the subarray starting from index n (to avoid overflow)
        int prev_diff = target - (n >= r ? a[n] : 0); // If subarray starts from index n, calculate difference between previous subarray values and target value
        // Output the result
        if (r >= a[n]) { // If range is within the bounds of the array
            cout << target << endl; // Output the target value as expected
        } else { // If range is outside the bounds or query is negative, output the previous subarray value as negative value minus the current subarray value
            cout << prev_diff << endl; // This should be equal to the difference between previous subarray values and target value
        }
    }
    return 0;
}
