#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 1e6+7;

int main() {
    int n, m;
    cin >> n;
    vector<int> a(n + 1, 0), b(n + 1, 0), c(n + 1, 0);
    
    // Read array 'a'
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    // Calculate prefix sum of array 'b' stored in array 'c'
    for (int i = 1; i <= n; i++) {
        cin >> b[i];
        c[i] = b[i] + c[i - 1];
    }
    
    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;
        
        // Find the position where 'l' should be inserted to maintain sorted order
        auto l_i = lower_bound(a.begin() + 1, a.end(), l) - a.begin();
        
        // Find the position just after the last occurrence of 'r' in 'a'
        auto r_j = upper_bound(a.begin() + 1, a.end(), r) - a.begin();
        
        if (r >= a[n]) {
            cout << c[n] - c[l_i - 1] << endl;
        } else {
            cout << c[r_j - 1] - c[l_i - 1] << endl;
        }
    }
    return 0;
}
