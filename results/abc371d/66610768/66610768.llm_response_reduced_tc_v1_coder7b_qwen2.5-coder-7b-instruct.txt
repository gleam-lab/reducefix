```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 2e5 + 7;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int n, q;
    cin >> n;
    vector<int> x(n + 1, 0), p(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
    }

    vector<LL> prefix_sum(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        prefix_sum[i] = prefix_sum[i - 1] + p[i];
    }

    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(x.begin() + 1, x.end(), l);
        auto it_r = upper_bound(x.begin() + 1, x.end(), r);

        int idx_l = distance(x.begin(), it_l);
        int idx_r = distance(x.begin(), it_r);

        if (idx_r == 0 || idx_l > n || idx_r > n) {
            cout << "0\n";
        } else {
            cout << prefix_sum[min(idx_r - 1, n)] - prefix_sum[max(0, idx_l - 1)] << "\n";
        }
    }

    return 0;
}
```

### Explanation
1. **Reading Input**: The code reads the number of villages `n` and their positions and populations into vectors `x` and `p`.
2. **Prefix Sum Calculation**: A prefix sum array `prefix_sum` is calculated where `prefix_sum[i]` holds the total population up to the `i-th` village.
3. **Handling Queries**: For each query, the code finds the indices of the leftmost and rightmost villages within the range `[L_i, R_i]`. It then calculates the total population in this range using the prefix sum array.
4. **Edge Cases Handling**: The code checks if the queried range is valid and handles cases where no villages fall within the range.

This solution ensures that all edge cases are handled correctly and efficiently computes the required results.
