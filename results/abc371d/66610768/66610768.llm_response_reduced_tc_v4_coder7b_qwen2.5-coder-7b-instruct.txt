#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 2e5+7;
int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    int n,q;
    cin >> n;
    vector<ll>pos(n),pop(n);
    for(int i=0;i<n;i++)cin >> pos[i];
    for(int i=0;i<n;i++)cin >> pop[i];
    vector<ll>prefix_pop(n+1);
    prefix_pop[0] = 0;
    for(int i=0;i<n;i++)prefix_pop[i+1] = prefix_pop[i] + pop[i];
    
    cin >> q;
    while(q--){
        int l,r;
        cin >> l >> r;
        auto it_l = lower_bound(pos.begin(),pos.end(),l);
        auto it_r = upper_bound(pos.begin(),pos.end(),r);
        int idx_l = it_l - pos.begin();
        int idx_r = it_r - pos.begin() - 1;
        if(idx_r >= n)idx_r = n-1;
        cout << prefix_pop[idx_r+1] - prefix_pop[idx_l] << '\n';
    }
    return 0;
}
