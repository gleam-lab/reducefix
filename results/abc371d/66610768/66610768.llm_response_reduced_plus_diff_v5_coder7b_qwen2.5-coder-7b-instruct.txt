#include <bits/stdc++.h>
using namespace std;

using LL = __int128_t;
constexpr int N = 1e6 + 7;

vector<LL> prefix_sums;

void preprocess(const vector<int>& a) {
    prefix_sums.assign(N, 0);
    for (int i = 1; i <= a.size(); ++i) {
        prefix_sums[i] = prefix_sums[i - 1] + a[i - 1];
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n, m;
    cin >> n;
    vector<int> a(n), b(n), c(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n; ++i) {
        cin >> b[i];
        c[i] = b[i] + c[i - 1];
    }

    preprocess(c);

    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(begin(a), end(a), l);
        auto it_r = upper_bound(begin(a), end(a), r);

        int idx_l = distance(begin(a), it_l);
        int idx_r = distance(begin(a), it_r);

        if (idx_r == 0 || idx_l >= n) {
            cout << 0 << '\n';
            continue;
        }

        LL sum = prefix_sums[idx_r] - prefix_sums[idx_l - 1];
        cout << sum << '\n';
    }

    return 0;
}
