#include <bits/stdc++.h>
using namespace std;
using LL = long long;
const int N = 2e5 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<LL> a(n), b(n);
    for (auto &x : a) cin >> x;
    for (auto &x : b) cin >> x;

    // prefix sum of b
    vector<LL> prefix_b(n + 1, 0);
    for (int i = 0; i < n; ++i)
        prefix_b[i + 1] = prefix_b[i] + b[i];

    // coordinate compression on a values
    vector<LL> sorted_a = a;
    sort(sorted_a.begin(), sorted_a.end());

    // build mapping from compressed index to original index
    map<LL, int> idx_map;
    for (int i = 0; i < n; ++i)
        idx_map[a[i]] = i + 1;  // use 1-based index

    int m;
    cin >> m;
    while (m--) {
        LL l, r;
        cin >> l >> r;

        // find range in a that covers [l, r]
        auto l_it = lower_bound(a.begin(), a.end(), l);
        auto r_it = upper_bound(a.begin(), a.end(), r);

        int start_idx = l_it - a.begin();
        int end_idx = r_it - a.begin();

        // sum b[i] where a[i] is in [l, r]
        if (start_idx < end_idx) {
            cout << prefix_b[end_idx] - prefix_b[start_idx] << "\n";
        } else {
            cout << "0\n";
        }
    }

    return 0;
}
