```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 2 * 1e5 + 10;
ll a[MAXN], b[MAXN], c[MAXN];

int main() {
    ios::sync_with_stdio(false); cin.tie(0);

    int n, q;
    cin >> n >> q;

    for(int i = 1; i <= n; ++i) cin >> a[i];
    sort(a + 1, a + n + 1);

    for(int i = 1; i <= n; ++i) {
        cin >> b[i];
        c[i] = c[i - 1] + b[i];
    }

    while(q--) {
        int l, r;
        cin >> l >> r;

        auto l_it = lower_bound(a + 1, a + n + 1, l);
        auto r_it = upper_bound(a + 1, a + n + 1, r);

        int l_idx = l_it - a, r_idx = r_it - a;

        if(l >= a[n]) cout << c[n] - c[l_idx - 1] << '\n';
        else cout << c[r_idx - 1] - c[l_idx - 1] << '\n';
    }

    return 0;
}
```
