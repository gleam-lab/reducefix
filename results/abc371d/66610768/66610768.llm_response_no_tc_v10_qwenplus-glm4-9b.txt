#include<bits/stdc++.h>

    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    for(int i = 1; i <= n; i++) {
        cin >> b[i];
        c[i] = c[i - 1] + b[i];
    }
    cin >> m;
    while(m--) {
        int l, r;
        cin >> l >> r;
        auto l_i = lower_bound(a.begin() + 1, a.end(), l);
        auto r_j = upper_bound(a.begin() + 1, a.end(), r);
        int i = l_i - a.begin();
        int j = r_j - a.begin();
        if (r >= a[n]) {
            if (i == 0) {
                cout << c[n] << endl;
            } else {
                cout << c[n] - c[i - 1] << endl;
            }
        } else {
            if (i == 0) {
                cout << c[j - 1] << endl;
            } else {
                cout << c[j - 1] - c[i - 1] << endl;
            }
        }
    }
    return 0;
}
