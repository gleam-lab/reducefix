#include<bits/stdc++.h>
using namespace std;
using LL = __int128;
typedef long long ll;
const int N = 1e6+7;
int main() {
	int n,m;
	cin >> n;
	vector<int>a(n+1,0),b(n+1,0),c(n+1,0);
	for(int i=1;i<=n;i++){
		cin >> a[i];
	}
	for(int i=1;i<=n;i++){
		cin >> b[i];
		c[i]=b[i]+c[i-1];
	}
	cin >> m;
	while(m--){
		int l,r;
		cin >> l >> r;
		auto l_i = lower_bound(a.begin()+1,a.end(),l);
		auto r_j = upper_bound(a.begin()+1,a.end(),r);
		int i = max(1, (int)(l_i-a.begin()));
		int j = min(n, (int)(r_j-a.begin())-1);
		if(i <= j) cout << c[j] - c[i-1] << endl;
		else cout << 0 << endl; // Handle case where no elements are within the range [l, r]
	}
	return 0;
}
