#include<bits/stdc++.h>
using namespace std;

using LL = __int128_t;

const int N = 1e6 + 7;

vector<LL> pre_sum;

void build_pre_sum(const vector<int>& a, int n) {
    pre_sum.assign(n + 1, 0);
    for(int i = 1; i <= n; ++i) {
        pre_sum[i] = pre_sum[i - 1] + a[i - 1];
    }
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0);

    int n, q;
    cin >> n >> q;
    
    vector<int> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];

    build_pre_sum(a, n);

    while(q--) {
        int l, r;
        cin >> l >> r;
        if(l == 1 && r == n) {
            cout << pre_sum[r] << '\n';
        } else {
            cout << pre_sum[r] - pre_sum[l - 1] << '\n';
        }
    }

    return 0;
}
