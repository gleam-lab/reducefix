#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 1e6 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n;
    vector<int> a(n + 1, 0), b(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> b[i];
    }

    // Prefix sum of b values
    vector<ll> prefix_sum(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        prefix_sum[i] = prefix_sum[i - 1] + b[i];
    }

    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;

        // Find leftmost index in a where a[i] >= l
        auto left = lower_bound(a.begin() + 1, a.end(), l);
        int li = distance(a.begin(), left);

        // Find rightmost index in a where a[i] <= r
        auto right = upper_bound(a.begin() + 1, a.end(), r);
        int rj = distance(a.begin(), right) - 1;

        if (li > rj) {
            cout << 0 << '\n';
        } else {
            cout << prefix_sum[rj] - prefix_sum[li - 1] << '\n';
        }
    }

    return 0;
}
