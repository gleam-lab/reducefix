#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n; // 输入数据点个数
    vector<int> a(n + 1); // 初始化数组a
    for (int i = 1; i <= n; ++i) { // 输入每个数据点的值
        cin >> a[i];
    }
    
    // 处理数据，避免越界访问数组元素
    for (int i = 1; i <= n; ++i) { // 初始化c数组，并填充数据
        c[i] = a[i] + c[i - 1]; // 这里应该是累加而不是直接赋值给c数组元素
    }
    
    // 处理输入的m个查询
    while (m--) { // 处理每个查询
        int l, r; // 输入查询区间左端点和右端点
        cin >> l >> r; // 输入查询区间左端点和右端点
        // 使用二分查找找到区间内所有满足条件的元素值
        auto l_i = lower_bound(a.begin() + 1, a.end(), l); // 二分查找左边界
        auto r_j = upper_bound(a.begin() + 1, a.end(), r); // 二分查找右边界
        // 检查区间右端点是否超出数组范围，避免越界访问数组元素
        if (r >= a[n]) { // 如果区间右端点在数组范围内，输出对应的值
            cout << c[n] - c[l_i - a.begin()] << endl; // 这里应该是减法操作，并输出结果
        } else { // 如果超出范围，输出区间内满足条件的元素值差
            cout << c[r_j - a.begin()] - c[l_i - a.begin()] << endl; // 这里应该是减法操作并输出结果
        }
    }
    return 0; // 返回0表示程序正常结束
}
