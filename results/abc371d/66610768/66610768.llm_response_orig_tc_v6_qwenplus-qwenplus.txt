#include <bits/stdc++.h>
using namespace std;

using LL = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<LL> a(n), b(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n; ++i) {
        cin >> b[i];
    }

    // Pair a[i] and b[i], then sort by a[i]
    vector<pair<LL, LL>> pairs(n);
    for (int i = 0; i < n; ++i) {
        pairs[i] = {a[i], b[i]};
    }
    sort(pairs.begin(), pairs.end());

    // Prefix sum of b[i] in sorted order
    vector<LL> prefix(n + 1, 0);
    for (int i = 0; i < n; ++i) {
        prefix[i + 1] = prefix[i] + pairs[i].second;
    }

    int m;
    cin >> m;
    while (m--) {
        LL l, r;
        cin >> l >> r;

        // Binary search to find the range [start, end) in sorted a[i] that satisfies l <= a[i] <= r
        auto start = lower_bound(pairs.begin(), pairs.end(), make_pair(l, numeric_limits<LL>::min())) - pairs.begin();
        auto end = upper_bound(pairs.begin(), pairs.end(), make_pair(r, numeric_limits<LL>::max())) - pairs.begin();

        cout << prefix[end] - prefix[start] << "\n";
    }

    return 0;
}
