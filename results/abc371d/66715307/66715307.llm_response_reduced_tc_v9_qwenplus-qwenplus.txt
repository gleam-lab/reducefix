```cpp
#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) cin >> x[i];
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Find first index where x[i] >= l using binary search
        int left = 1, right = n;
        while (left < right) {
            int mid = (left + right) / 2;
            if (x[mid] < l) left = mid + 1;
            else right = mid;
        }
        int lower_idx = left;

        // Find last index where x[i] <= r using binary search
        left = 1, right = n;
        while (left < right) {
            int mid = (left + right + 1) / 2; // +1 to avoid infinite loop in upper bound
            if (x[mid] <= r) left = mid;
            else right = mid - 1;
        }
        int upper_idx = left;

        if (x[lower_idx] >= l && x[upper_idx] <= r)
            cout << a[upper_idx] - a[lower_idx - 1] << '\n';
        else
            cout << 0 << '\n';
    }

    return 0;
}
```
