#include <bits/stdc++.h>
using namespace std;
int n, x[200050], p[200050], a[200050];

int main() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> x[i];
    for (int i = 0; i < n; ++i) {
        cin >> p[i];
        a[i + 1] = a[i] + p[i];  // Offset the prefix sum array to start from index 1
    }
    cin >> n;  // Input should be corrected to n since we already read n
    for (int i = 0; i < n; ++i) {
        int l, r; cin >> l >> r;
        int l1 = 0, r1 = n - 1, l2 = 0, r2 = n - 1;
        while (l1 < r1) {
            int mid = l1 + (r1 - l1) / 2;
            if (x[mid] >= l) r1 = mid;
            else l1 = mid + 1;
        }
        while (l2 < r2) {
            int mid = l2 + (r2 - l2) / 2;
            if (x[mid] <= r) l2 = mid + 1;
            else r2 = mid;
        }
        cout << a[r2] - a[l1] << endl;  // Correct the index range for prefix sums
    }
    return 0;
}
