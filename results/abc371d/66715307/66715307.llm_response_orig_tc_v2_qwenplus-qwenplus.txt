#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200050;

int n;
long long x[MAXN], p[MAXN], a[MAXN];
vector<pair<long long, int>> sorted_x;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }

    // Prepare sorted list of (x[i], index) for binary search
    vector<pair<long long, int>> points(n);
    for (int i = 1; i <= n; ++i) {
        points[i - 1] = {x[i], i};
    }
    sort(points.begin(), points.end());

    int q;
    cin >> q;

    while (q--) {
        long long l, r;
        cin >> l >> r;

        // Binary search to find the first index where x >= l
        int left = 0, right = n - 1;
        int first_idx = n; // not found value
        while (left <= right) {
            int mid = (left + right) / 2;
            if (points[mid].first >= l) {
                first_idx = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        // Binary search to find the last index where x <= r
        left = 0;
        right = n - 1;
        int last_idx = -1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (points[mid].first <= r) {
                last_idx = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        if (first_idx > last_idx) {
            cout << 0 << endl;
        } else {
            int start = points[first_idx].second;
            int end = points[last_idx].second;
            // Ensure that the original array is sorted, otherwise we need to sort indices accordingly
            cout << a[end] - a[start - 1] << endl;
        }
    }

    return 0;
}
