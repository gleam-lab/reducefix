#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, x[200050], p[200050], q, a[200050];
    cin >> n; // Number of elements in array x
    for (int i = 1; i <= n; ++i) cin >> x[i]; // Read the elements of array x
    
    // Simple modified array to store the differences between adjacent elements
    int diff[200050]; // Initialize with all elements as 0
    for (int i = 1; i < n; ++i) {
        diff[i] = a[i] - a[i - 1]; // Calculate the difference between adjacent elements
    }
    
    // Read the query value
    cin >> q; // Number of queries to process
    
    // Process each query separately
    for (int i = 1; i <= q; ++i) {
        int l, r; // Left and right indices for querying the difference
        cin >> l >> r; // Read the indices for the query
        // Calculate the difference between the two adjacent elements at the query index
        int diff_value = diff[r] - diff[l - 1]; // Using negative index to handle cases where r is greater than n-1
        cout << diff_value << endl; // Output the difference value for the query
    }
    return 0;
}
