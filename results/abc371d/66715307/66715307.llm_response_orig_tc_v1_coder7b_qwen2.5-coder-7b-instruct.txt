#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, q;
    cin >> n;
    vector<int> x(n), p(n);
    for (int i = 0; i < n; ++i) cin >> x[i];
    for (int i = 0; i < n; ++i) cin >> p[i];
    
    vector<long long> prefix_sums(n + 1, 0);
    for (int i = 1; i <= n; ++i) prefix_sums[i] = prefix_sums[i - 1] + p[i - 1];

    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        auto it_l = lower_bound(x.begin(), x.end(), l);
        auto it_r = upper_bound(x.begin(), x.end(), r);
        int idx_l = distance(x.begin(), it_l);
        int idx_r = distance(x.begin(), it_r);

        cout << prefix_sums[idx_r] - prefix_sums[idx_l - 1] << endl;
    }

    return 0;
}
