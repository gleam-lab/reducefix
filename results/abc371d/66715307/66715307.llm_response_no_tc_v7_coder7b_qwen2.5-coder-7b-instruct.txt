        a[i] = a[i - 1] + p[i];
    }
    cin >> q;

        int l, r; cin >> l >> r;
        int posL = lower_bound(x + 1, x + n + 1, l) - x;
        int posR = upper_bound(x + 1, x + n + 1, r) - x;

    }

}
