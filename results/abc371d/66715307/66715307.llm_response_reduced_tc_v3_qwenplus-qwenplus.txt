#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> x[i];
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        a[i] = a[i-1] + p[i];
    }
    cin >> q;
    for (int i = 1; i <= q; i++) {
        int l, r;
        cin >> l >> r;
        
        // Find first index where x[i] >= l using binary search
        int left = 1, right = n;
        while (left < right) {
            int mid = (left + right) / 2;
            if (x[mid] >= l) right = mid;
            else left = mid + 1;
        }
        int idx_l = left;

        // Find last index where x[i] <= r using binary search
        left = 1, right = n;
        while (left < right) {
            int mid = (left + right + 1) / 2; // ceil division
            if (x[mid] <= r) left = mid;
            else right = mid - 1;
        }
        int idx_r = left;

        // Check if the range is valid
        if (idx_l > idx_r || x[idx_l] > r || x[idx_r] < l) {
            cout << 0 << endl;
        } else {
            cout << a[idx_r] - a[idx_l - 1] << endl;
        }
    }

    return 0;
}
