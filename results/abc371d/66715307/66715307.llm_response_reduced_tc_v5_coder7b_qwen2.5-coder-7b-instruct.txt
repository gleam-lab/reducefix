#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> x(n + 1), p(n + 1);
    for (int i = 1; i <= n; ++i) cin >> x[i];
    for (int i = 1; i <= n; ++i) cin >> p[i];

    // Calculate prefix sum array for cumulative population
    vector<long long> a(n + 1);
    for (int i = 1; i <= n; ++i) a[i] = a[i - 1] + p[i];

    int q;
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        int pos_l = lower_bound(x.begin(), x.end(), l) - x.begin();
        int pos_r = upper_bound(x.begin(), x.end(), r) - x.begin() - 1;
        cout << a[pos_r] - a[pos_l - 1] << endl;
    }
    return 0;
}
