#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;
    std::vector<int> x(n + 1), p(n + 1);
    
    // Read the array x
    for(int i = 1; i <= n; ++i) std::cin >> x[i];

    // Read the array p and calculate prefix sums
    for(int i = 1; i <= n; ++i) {
        std::cin >> p[i];
        p[i] += p[i - 1]; 
    }

    int q;
    std::cin >> q;

    while(q--) {
        int l, r;
        std::cin >> l >> r;

        // Find the first index where x[index] >= l using binary search
        auto it_l = std::lower_bound(x.begin() + 1, x.end(), l) - x.begin();
        
        // Find the first index where x[index] > r using binary search
        auto it_r = std::upper_bound(x.begin() + 1, x.end(), r) - x.begin();

        // Calculate the sum between indices [it_l, it_r)
        long long result = p[it_r - 1] - (it_l == 1 ? 0 : p[it_l - 2]);
        std::cout << result << '\n';
    }
    return 0;
}
