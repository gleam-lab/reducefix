#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

// Function to find the first index greater than or equal to 'val'
int binarySearchFirst(int val, int start, int end) {
    while (start <= end) {
        int mid = start + (end - start) / 2;
        if (x[mid] >= val) {
            end = mid - 1;
        } else {
            start = mid + 1;
        }
    }
    return start;
}

// Function to find the last index less than or equal to 'val'
int binarySearchLast(int val, int start, int end) {
    while (start <= end) {
        int mid = start + (end - start) / 2;
        if (x[mid] <= val) {
            start = mid + 1;
        } else {
            end = mid - 1;
        }
    }
    return end;
}

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }
    sort(x + 1, x + n + 1); // Sort the array to use binary search
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }
    cin >> q;
    for (int i = 1; i <= q; i++) {
        int l, r; cin >> l >> r;
        int l1 = binarySearchFirst(l, 1, n);
        int r1 = binarySearchLast(r, 1, n);
        cout << a[r1] - a[l1 - 1] << endl;
    }
    return 0;
}

### Explanation of Fixes:
1. **Binary Search Functions**: Added helper functions `binarySearchFirst` and `binarySearchLast` to perform binary search for finding the first and last indices respectively where the value is greater than or equal to `l` and less than or equal to `r`.
2. **Sorting**: The array `x` is sorted before performing binary searches to ensure that the binary search operations work correctly.
3. **Efficiency**: Binary search reduces the time complexity of finding the required indices from O(n) to O(log n), making the solution more efficient.
4. **Edge Cases**: The code now handles edge cases such as when `l` and `r` are outside the range of the array values by using appropriate boundary checks within the binary search functions.
