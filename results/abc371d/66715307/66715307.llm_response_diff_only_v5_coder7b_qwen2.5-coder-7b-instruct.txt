#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

int findLeftBound(int arr[], int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] >= target) {
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    return left;
}

int findRightBound(int arr[], int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] <= target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return right - 1;
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> x[i];
    sort(x, x + n); // Sort the array to use binary search
    for (int i = 1; i < n; i++) p[i] = p[i - 1] + x[i];
    cin >> q;
    for (int i = 0; i < q; i++) {
        int l, r;
        cin >> l >> r;
        int leftIndex = findLeftBound(x, l);
        int rightIndex = findRightBound(x, r);
        cout << p[rightIndex] - p[leftIndex] << endl;
    }
    return 0;
}
