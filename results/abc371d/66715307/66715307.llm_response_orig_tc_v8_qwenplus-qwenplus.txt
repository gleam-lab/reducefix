#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200050;

int n;
long long x[MAXN], p[MAXN], a[MAXN];
vector<pair<long long, int>> sorted_x;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
    }
    for (int i = 1; i <= n; ++i) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }

    // Prepare sorted x with indices
    vector<pair<long long, int>> indexed_x(n);
    for (int i = 1; i <= n; ++i) {
        indexed_x[i - 1] = {x[i], i};
    }
    sort(indexed_x.begin(), indexed_x.end());

    // Build prefix sum array based on sorted x
    long long prefix_sum[MAXN];
    prefix_sum[0] = 0;
    for (int i = 0; i < n; ++i) {
        prefix_sum[i + 1] = prefix_sum[i] + p[indexed_x[i].second];
    }

    int q;
    cin >> q;
    while (q--) {
        long long l, r;
        cin >> l >> r;

        // Binary search to find left and right bounds in sorted x
        auto left = lower_bound(indexed_x.begin(), indexed_x.end(), make_pair(l, 0));
        auto right = upper_bound(indexed_x.begin(), indexed_x.end(), make_pair(r, numeric_limits<int>::max()));

        int left_idx = left - indexed_x.begin();
        int right_idx = right - indexed_x.begin() - 1;

        if (left_idx > right_idx) {
            cout << 0 << "\n";
        } else {
            cout << prefix_sum[right_idx + 1] - prefix_sum[left_idx] << "\n";
        }
    }

    return 0;
}
