```cpp
#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; ++i) cin >> x[i];
    for (int i = 1; i <= n; ++i) {
        p[i] = x[i]; // Store original positions in p
    }

    // Build prefix sum of weights
    a[0] = 0;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        a[i] += a[i - 1];
    }

    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;

        // Find leftmost index where position >= l
        int left = 1, right = n;
        while (left < right) {
            int mid = (left + right) / 2;
            if (p[mid] >= l) right = mid;
            else left = mid + 1;
        }
        int L = left;

        // If all positions are less than l, output 0
        if (p[L] < l) {
            cout << 0 << '\n';
            continue;
        }

        // Find rightmost index where position <= r
        left = 1, right = n;
        while (left < right) {
            int mid = (left + right + 1) / 2;
            if (p[mid] <= r) left = mid;
            else right = mid - 1;
        }
        int R = left;

        // If valid range found
        if (L <= R && p[R] <= r) {
            cout << a[R] - a[L - 1] << '\n';
        } else {
            cout << 0 << '\n';
        }
    }

    return 0;
}
```
