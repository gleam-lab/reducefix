#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, x[200050], p[200050], q, a[200050] = {0};
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> x[i];
        if (i > 1) a[i] = a[i - 1];
        cin >> p[i];
        a[i] += p[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        auto it1 = lower_bound(x + 1, x + n + 1, l);
        auto it2 = upper_bound(x + 1, x + n + 1, r);
        cout << a[it2 - x - 1] - (it1 == x + 1 ? 0 : a[it1 - x]) << endl;
    }
    return 0;
}
