#include <iostream>
#include <vector>
using namespace std;

int n;
vector<int> x, p, a;

int main() {
    cin >> n;
    x.resize(n + 1);
    p.resize(n + 1);
    a.resize(n + 1);

    for (int i = 1; i <= n; i++) {
        cin >> x[i];
    }

    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }

    int q;
    cin >> q;
    for (int i = 1; i <= q; i++) {
        int l, r;
        cin >> l >> r;

        int l1 = 1, r1 = n, l2 = 1, r2 = n;
        while (l1 < r1) {
            int mid = l1 + (r1 - l1) / 2;
            if (x[mid] >= l) r1 = mid;
            else l1 = mid + 1;
        }
        while (l2 < r2) {
            int mid = l2 + (r2 - l2 + 1) / 2;
            if (x[mid] <= r) l2 = mid;
            else r2 = mid - 1;
        }

        // Ensure l2 and l1 are within the bounds and l2 is greater than or equal to l1
        if (l1 > l2) l2 = l1;
        cout << a[l2] - a[l1 - 1] << endl;
    }

    return 0;
}
