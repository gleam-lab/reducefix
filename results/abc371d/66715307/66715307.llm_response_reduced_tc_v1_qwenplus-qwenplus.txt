#include <bits/stdc++.h>
using namespace std;

int n, x[200050], p[200050], q, a[200050];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) cin >> x[i];
    for (int i = 1; i <= n; ++i) {
        p[i] = p[i-1] + x[i];
    }
    cin >> q;
    while (q--) {
        int l, r;
        cin >> l >> r;
        
        // Binary search to find the first index where x[i] >= l
        int left = 1, right = n, L = n + 1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (x[mid] >= l) {
                L = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        // Binary search to find the last index where x[i] <= r
        left = 1, right = n, R = 0;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (x[mid] <= r) {
                R = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        if (L <= R) {
            cout << p[R] - p[L - 1] << endl;
        } else {
            cout << 0 << endl;
        }
    }

    return 0;
}
