#include <bits/stdc++.h>
using namespace std;
int n, x[200050], p[200050], a[200050];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> x[i];
    a[0] = 0;
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        a[i] = a[i - 1] + p[i];
    }
    cin >> n; // Read q from input, previously initialized to n which might be incorrect
    for (int q = 0; q < n; q++) {
        int l, r; cin >> l >> r;
        int l1 = 1, r1 = n, l2 = 1, r2 = n;
        while (l1 < r1) {
            int mid = l1 + (r1 - l1) / 2;
            if (x[mid] >= l) r1 = mid;
            else l1 = mid + 1;
        }
        while (l2 < r2) {
            int mid = l2 + (r2 - l2 + 1) / 2;
            if (x[mid] <= r) l2 = mid;
            else r2 = mid - 1;
        }
        cout << a[l2] - a[l1 - 1] << endl; // Fix the index for a[l1 - 1]
    }
    return 0;
}
