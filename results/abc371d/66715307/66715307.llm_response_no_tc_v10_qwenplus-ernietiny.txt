```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n; // 输入整数n表示需要处理的记录数
    cin >> n; // 读入记录数
    
    // 创建数组用于存储各位置上的元素值
    vector<int> x(n); // x数组初始化，存储x[1]到x[n]的值
    vector<int> p(n), a(n); // p数组存储单个数值，a数组存储计算结果
    
    // 读入每个位置上的数值和对应的权值
    for (int i = 1; i <= n; ++i) {
        cin >> x[i] >> p[i]; // 读入每个位置的数值和权值
        a[i] = p[i]; // 将权值赋值给结果数组中的相应位置
    }
    
    // 读入需要输出的记录数q
    cin >> q; // 读入需要输出的记录数
    
    // 根据权值进行排序，确保结果正确性
    sort(a.begin(), a.end()); // 对结果数组进行排序，保证相同权值出现顺序正确
    
    // 处理每个查询记录的逻辑部分
    for (int i = 0; i < q; ++i) { // 处理q个查询记录
        int l, r; // 记录查询左边界和右边界的位置索引
        cin >> l >> r; // 读入查询左边界和右边界的位置索引
        // 使用二分查找找到满足条件的中间位置索引l2和r2
        int l1 = lower_bound(a.begin(), a.end(), l) - a.begin() + 1; // 使用二分查找找到满足条件的中间位置索引l1
        int r2 = upper_bound(a.begin(), a.end(), r) - a.begin(); // 使用二分查找找到满足条件的中间位置索引r2
        // 输出查询结果
        cout << a[l2] - a[l1 - 1] << endl; // 输出查询结果，这里假设查询结果为整数类型，直接输出即可
    }
    
    return 0; // 返回0表示程序正常结束
}
```
