
```cpp
#include <bits/stdc++.h>
using namespace std;
int n, x[200050], p[200050], q, a[200050];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> x[i];
    for (int i = 0; i < n; i++) cin >> p[i];
    a[0] = p[0];
    for (int i = 1; i < n; i++) a[i] = a[i - 1] + p[i];
    cin >> q;
    while (q--) {
        int l, r; cin >> l >> r;
        int l1 = 0, r1 = n - 1, l2 = 0, r2 = n - 1;
        while (l1 < r1) {
            int mid = (l1 + r1 + 1) >> 1;
            if (x[mid] >= l) r1 = mid - 1;
            else l1 = mid;
        }
        while (l2 < r2) {
            int mid = (l2 + r2 + 1) >> 1;
            if (x[mid] <= r) l2 = mid;
            else r2 = mid - 1;
        }
        cout << a[r2] - (l1 > 0 ? a[l1 - 1] : 0) << endl;
    }
    return 0;
}
```
