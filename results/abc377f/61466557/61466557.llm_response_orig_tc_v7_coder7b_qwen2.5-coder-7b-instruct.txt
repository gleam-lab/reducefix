```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;

// Function to check if a position is within bounds
bool inBounds(pll pos, ll n) {
    return 1 <= pos.first && pos.first <= n && 1 <= pos.second && pos.second <= n;
}

// Main function to solve the problem
void solve() {
    ll n, m; cin >> n >> m;
    unordered_set<pll> pieces;
    for(ll i = 0; i < m; ++i) {
        ll x, y; cin >> x >> y;
        pieces.insert({x, y});
    }
    
    ll total_squares = n * n;
    ll occupied_by_rows = 0, occupied_by_cols = 0, occupied_by_diagonals = 0;
    
    // Count squares occupied by rows, columns, and diagonals
    for(ll i = 1; i <= n; ++i) {
        for(ll j = 1; j <= n; ++j) {
            if(pieces.count({i, j})) {
                occupied_by_rows++;
                occupied_by_cols++;
                if(pieces.count({i, j + i - 1})) occupied_by_diagonals++;
                if(pieces.count({i, j - i + 1})) occupied_by_diagonals++;
            }
        }
    }
    
    // Calculate the number of valid positions
    ll valid_positions = total_squares - (occupied_by_rows + occupied_by_cols - occupied_by_diagonals);
    
    // Output the result
    cout << valid_positions << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    
    int t = 1;
    cin >> t;
    while(t--) solve();
    return 0;
}
```

This solution uses `unordered_set` to store the positions of the pieces for efficient lookup. It counts how many squares are occupied by rows, columns, and diagonals, ensuring each diagonal is counted only once. Finally, it calculates the number of valid positions by subtracting the occupied squares from the total number of squares. This approach ensures correct handling of all edge cases and large inputs up to the specified constraints.
