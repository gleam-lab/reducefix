#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;

const int MAXN = 1000005;

bool row[MAXN], col[MAXN], diag1[2 * MAXN], diag2[2 * MAXN];

void solve() {
    ll n, m; cin >> n >> m;
    for(ll i = 0; i < m; ++i) {
        ll a, b; cin >> a >> b;
        row[a-1] = true;
        col[b-1] = true;
        diag1[a + b - 2] = true;
        diag2[a - b + MAXN - 1] = true;
    }
    ll ans = n * n;
    for(ll i = 0; i < n; ++i) {
        if(row[i]) ans -= n;
        if(col[i]) ans -= n;
    }
    for(ll i = 0; i < 2 * MAXN; ++i) {
        if(diag1[i]) ans -= n;
        if(diag2[i]) ans -= n;
    }
    for(ll i = 0; i < n; ++i) {
        if(row[i] && col[i]) ans += 1;
    }
    for(ll i = 0; i < 2 * MAXN; ++i) {
        if(diag1[i] && diag2[i]) ans += 1;
    }
    for(ll i = 0; i < n; ++i) {
        if(row[i] && diag1[i] && diag2[i]) ans -= 1;
    }
    for(ll i = 0; i < n; ++i) {
        if(col[i] && diag1[i] && diag2[i]) ans -= 1;
    }
    for(ll i = 0; i < n; ++i) {
        if(row[i] && col[i] && diag1[i] && diag2[i]) ans += 1;
    }
    cout << ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    solve();
    return 0;
}
