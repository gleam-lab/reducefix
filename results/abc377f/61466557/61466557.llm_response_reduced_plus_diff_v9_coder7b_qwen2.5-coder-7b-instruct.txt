// Function to calculate the number of valid positions
ll countValidPositions(ll n, const vector<pii>& queens) {
    set<ll> rows, cols, diag1, diag2;
    for(auto [i, j] : queens) {
        rows.insert(i);
        cols.insert(j);
        diag1.insert(i - j);
        diag2.insert(i + j);
    }

    ll totalPositions = n * n;
    ll invalidPositions = 0;

    invalidPositions += rows.size() * n; // Rows
    invalidPositions += cols.size() * n; // Columns
    invalidPositions -= rows.size() * cols.size(); // Overcounted intersections

    for(auto d : diag1) {

    }

}
