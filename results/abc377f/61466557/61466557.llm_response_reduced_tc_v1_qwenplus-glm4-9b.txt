#include <iostream>
#include <vector>
using namespace std;

long long solve(long long n, long long m, vector<pair<long long, long long>>& pieces) {
    long long total_empty_squares = n * n;

    // Mark squares captured by pieces in rows and columns
    vector<bool> row_captured(n + 1, false);
    vector<bool> col_captured(n + 1, false);
    for (auto& piece : pieces) {
        if (row_captured[piece.first] == false) {
            row_captured[piece.first] = true;
            total_empty_squares -= (n - 1);
        }
        if (col_captured[piece.second] == false) {
            col_captured[piece.second] = true;
            total_empty_squares -= (n - 1);
        }
    }

    // Count the number of empty squares in diagonals (i+j) and (i-j)
    vector<bool> diag1_captured(2 * n + 1, false);
    vector<bool> diag2_captured(2 * n + 1, false);
    for (auto& piece : pieces) {
        diag1_captured[piece.first + piece.second] = true;
        diag2_captured[piece.first - piece.second] = true;
    }

    for (long long i = 1; i <= 2 * n; ++i) {
        long long diag_count = (diag1_captured[i] ? 1 : 0) + (diag2_captured[i] ? 1 : 0);
        long long diag_size = (i <= n ? i : n) + (i >= n ? i - n + 1 : 0);
        total_empty_squares -= (diag_size - diag_count);
    }

    return total_empty_squares;
}

int main() {
    long long n, m;
    cin >> n >> m;

    vector<pair<long long, long long>> pieces(m);
    for (long long i = 0; i < m; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    cout << solve(n, m, pieces) << endl;

    return 0;
}
