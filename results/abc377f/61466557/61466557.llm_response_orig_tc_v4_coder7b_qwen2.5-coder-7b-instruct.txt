#include <bits/stdc++.h>
using namespace std;

const long long MAX_N = 1e9;

long long n, m;
vector<pair<long long, long long>> pieces;

set<long long> rows, cols, diag1, diag2;

void preprocess() {
    for (auto [r, c] : pieces) {
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r - c);
        diag2.insert(r + c);
    }
}

long long count_safe_squares() {
    long long total = n * n;
    long long occupied_by_rows = rows.size() * n;
    long long occupied_by_cols = cols.size() * n;
    long long occupied_by_diags = diag1.size() * n + diag2.size() * n;
    long long counted_twice = (rows.size() * cols.size() + diag1.size() * diag2.size()) * n;
    long long counted_three_times = (rows.size() * diag1.size() + rows.size() * diag2.size() + cols.size() * diag1.size() + cols.size() * diag2.size()) * n;
    long long counted_four_times = (diag1.size() * diag2.size()) * n;
    return total - occupied_by_rows - occupied_by_cols - occupied_by_diags + counted_twice - counted_three_times + counted_four_times;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> m;
    pieces.resize(m);
    for (auto &p : pieces) {
        cin >> p.first >> p.second;
    }

    preprocess();

    long long safe_squares = count_safe_squares();
    cout << safe_squares << endl;

    return 0;
}
