#include <iostream>
#include <vector>
#include <set>
using namespace std;

long long n, m;
vector<pair<int, int>> pieces;

bool inBounds(int r, int c, int n) {
    return (1 <= r && r <= n) && (1 <= c && c <= n);
}

int main() {
    cin >> n >> m;
    pieces.resize(m);
    for (int i = 0; i < m; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    set<pair<int, int>> captured;
    set<int> rows, cols, diag1, diag2, anti_diag1, anti_diag2;

    for (auto& p : pieces) {
        rows.insert(p.first);
        cols.insert(p.second);
        diag1.insert(p.first + p.second);
        diag2.insert(p.first - p.second);
        anti_diag1.insert(p.first - p.second);
        anti_diag2.insert(p.first + p.second);
    }

    long long totalEmpty = n * n;
    long long nonCapturable = 0;

    // Subtract squares in rows and columns where pieces are placed
    for (auto& r : rows) {
        nonCapturable += n - 1;
    }
    for (auto& c : cols) {
        nonCapturable += n - 1;
    }

    // Subtract squares on diagonals and anti-diagonals where pieces are placed
    for (auto& d : diag1) {
        int minCol = max(1, d - n + 1);
        int maxCol = min(n, d);
        if (minCol <= maxCol) {
            nonCapturable += maxCol - minCol + 1;
        }
    }

    for (auto& d : diag2) {
        int minCol = max(1, 2 * n - d);
        int maxCol = min(n, d);
        if (minCol <= maxCol) {
            nonCapturable += maxCol - minCol + 1;
        }
    }

    for (auto& d : anti_diag1) {
        int minRow = max(1, d - n + 1);
        int maxRow = min(n, d);
        if (minRow <= maxRow) {
            nonCapturable += maxRow - minRow + 1;
        }
    }

    for (auto& d : anti_diag2) {
        int minRow = max(1, 2 * n - d - 1);
        int maxRow = min(n, d - 1);
        if (minRow <= maxRow) {
            nonCapturable += maxRow - minRow + 1;
        }
    }

    cout << totalEmpty - nonCapturable << endl;

    return 0;
}
