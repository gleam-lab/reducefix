#include <bits/stdc++.h>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>
using namespace std;
using namespace __gnu_pbds;

typedef long long ll;
typedef pair<ll, ll> pll;
typedef tree<int,null_type,less<>,rb_tree_tag,tree_order_statistics_node_update> ordered_set;

#define all(x) (x).begin(), (x).end()
#define FOR(i, a, b) for (int i = (a); i < (b); ++i)
#define pb push_back

bool inBounds(const pll &p, ll n) {
    return 1 <= p.first && p.first <= n && 1 <= p.second && p.second <= n;
}

void solve() {
    ll N, M;
    cin >> N >> M;
    
    set<ll> rows, cols;
    set<ll> diag1, diag2;  // i-j and i+j
    vector<pll> queens(M);
    
    for (ll i = 0; i < M; ++i) {
        ll r, c;
        cin >> r >> c;
        queens[i] = {r, c};
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r - c);
        diag2.insert(r + c);
    }

    ll total = N * N;
    ll attacked = 0;

    // Rows and columns
    attacked += (ll)rows.size() * N;
    attacked += (ll)cols.size() * N;
    // Subtract overcounted cells (intersections of rows and columns with pieces)
    set<pair<ll, ll>> rc_intersections;
    for (ll r : rows) {
        for (ll c : cols) {
            rc_intersections.insert({r, c});
        }
    }

    // Diagonals
    set<pair<ll, ll>> d1_intersections;
    for (ll d : diag1) {
        attacked += (N - abs(d));
        // Intersections with rows and cols
        for (ll r : rows) {
            ll c = r - d;
            if (inBounds({r, c}, N)) d1_intersections.insert({r, c});
        }
        for (ll c : cols) {
            ll r = c + d;
            if (inBounds({r, c}, N)) d1_intersections.insert({r, c});
        }
    }

    set<pair<ll, ll>> d2_intersections;
    for (ll d : diag2) {
        ll cnt;
        if (d <= N) cnt = d;
        else cnt = 2 * N - d + 1;
        attacked += (cnt - 1); // Because the diagonal includes the piece itself

        // Intersections with rows and cols
        for (ll r : rows) {
            ll c = d - r;
            if (inBounds({r, c}, N)) d2_intersections.insert({r, c});
        }
        for (ll c : cols) {
            ll r = d - c;
            if (inBounds({r, c}, N)) d2_intersections.insert({r, c});
        }

        // Intersections with diag1
        for (ll od : diag1) {
            if ((d + od) % 2 == 0) {
                ll r = (d + od) / 2;
                ll c = (d - od) / 2;
                if (inBounds({r, c}, N)) d2_intersections.insert({r, c});
            }
        }
    }

    // Adjust for overlaps
    attacked -= rc_intersections.size();
    attacked -= d1_intersections.size();
    attacked -= d2_intersections.size();

    // Remove occupied cells
    attacked += M;

    ll result = N * N - attacked;
    cout << result << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
