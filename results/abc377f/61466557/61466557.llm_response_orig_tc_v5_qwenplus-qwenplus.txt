bool inBounds(const pll &a, const ll n) {
    return 1 <= a.fst && a.fst <= n && 1 <= a.snd && a.snd <= n;
}

void solve() {
    ll n, m;
    re(n, m);

    set<pll> pieces;
    set<ll> rows, cols, diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll a, b;
        re(a, b);
        pieces.insert({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    ll total = n * n;

    // Remove squares attacked by queens
    total -= (ll)rows.size() * n;
    total -= (ll)cols.size() * n;
    total += (ll)rows.size() * (ll)cols.size(); // Overlap between row and column

    // For diagonal 1: a - b
    set<pll> intersections;
    for (ll d : diag1) {
        ll count = n - abs(d);
        total -= count;

        // Add back overlaps
        intersections.clear();
        for (ll r : rows) {
            ll c = r - d;
            if (1 <= c && c <= n) intersections.insert({r, c});
        }
        for (ll c : cols) {
            ll r = c + d;
            if (1 <= r && r <= n) intersections.insert({r, c});
        }
        total += intersections.size();
    }

    // For diagonal 2: a + b
    for (ll d : diag2) {
        ll count;
        if (d <= n)
            count = d - 1;
        else
            count = 2 * n - d + 1;
        total -= count;

        // Add back overlaps
        intersections.clear();
        for (ll r : rows) {
            ll c = d - r;
            if (1 <= c && c <= n) intersections.insert({r, c});
        }
        for (ll c : cols) {
            ll r = d - c;
            if (1 <= r && r <= n) intersections.insert({r, c});
        }
        for (ll od : diag1) {
            ll r = (d + od) / 2;
            ll c = (d - od) / 2;
            if ((d + od) % 2 == 0 && 1 <= r && r <= n && 1 <= c && c <= n)
                intersections.insert({r, c});
        }
        total += intersections.size();
    }

    // Remove duplicate counts from overlapping diagonals
    for (ll dr : diag1) {
        for (ll dc : diag2) {
            ll r = (dr + dc) / 2;
            ll c = (dc - dr) / 2;
            if ((dr + dc) % 2 == 0 && 1 <= r && r <= n && 1 <= c && c <= n)
                if (pieces.count({r, c}) == 0)
                    total += 1;
        }
    }

    // Remove overlap of rows with diag1
    for (ll r : rows) {
        for (ll d : diag1) {
            ll c = r - d;
            if (1 <= c && c <= n)
                if (pieces.count({r, c}) == 0)
                    total += 1;
        }
    }

    // Remove overlap of cols with diag1
    for (ll c : cols) {
        for (ll d : diag1) {
            ll r = c + d;
            if (1 <= r && r <= n)
                if (pieces.count({r, c}) == 0)
                    total += 1;
        }
    }

    // Remove overlap of rows with diag2
    for (ll r : rows) {
        for (ll d : diag2) {
            ll c = d - r;
            if (1 <= c && c <= n)
                if (pieces.count({r, c}) == 0)
                    total += 1;
        }
    }

    // Remove overlap of cols with diag2
    for (ll c : cols) {
        for (ll d : diag2) {
            ll r = d - c;
            if (1 <= r && r <= n)
                if (pieces.count({r, c}) == 0)
                    total += 1;
        }
    }

    // Now remove occupied cells which were not subtracted earlier
    for (auto [x, y] : pieces) {
        // Check if this square was counted as an intersection or already removed
        bool already_removed = false;
        if (rows.count(x) || cols.count(y)) already_removed = true;
        if (!already_removed && diag1.count(x - y)) already_removed = true;
        if (!already_removed && diag2.count(x + y)) already_removed = true;
        if (!already_removed) total -= 1;
    }

    cout << max(total, 0LL) << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t = 1;
    //cin >> t;
    while (t--) solve();

    return 0;
}
