```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;

    set<i64> row_set, col_set, diag1_set, diag2_set;
    set<pair<i64, i64>> piece_set;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        piece_set.emplace(a, b);
        row_set.insert(a);
        col_set.insert(b);
        diag1_set.insert(a + b);
        diag2_set.insert(a - b);
    }

    i64 total_attacked = 0;

    // Count squares attacked by rows and columns
    total_attacked += row_set.size() * N;
    total_attacked += col_set.size() * N;

    // Count squares attacked by diagonals (a+b)
    for (i64 sum_val : diag1_set) {
        // x + y = sum_val
        i64 min_x = max(1LL, sum_val - N);
        i64 max_x = min(N, sum_val - 1);
        if (min_x <= max_x) {
            total_attacked += (max_x - min_x + 1);
        }
    }

    // Count squares attacked by diagonals (x - y)
    for (i64 diff_val : diag2_set) {
        // x - y = diff_val => y = x - diff_val
        i64 min_x = max(1LL, 1 + diff_val);
        i64 max_x = min(N, N + diff_val);
        if (min_x <= max_x) {
            total_attacked += (max_x - min_x + 1);
        }
    }

    // Subtract overlaps

    // Rows and columns intersections
    for (i64 r : row_set) {
        for (i64 c : col_set) {
            total_attacked--; // (r,c) is counted twice
        }
    }

    // Rows and diag1 intersections
    for (i64 r : row_set) {
        for (i64 d : diag1_set) {
            i64 c = d - r;
            if (1 <= c && c <= N) {
                total_attacked--;
            }
        }
    }

    // Rows and diag2 intersections
    for (i64 r : row_set) {
        for (i64 d : diag2_set) {
            i64 c = r - d;
            if (1 <= c && c <= N) {
                total_attacked--;
            }
        }
    }

    // Cols and diag1 intersections
    for (i64 c : col_set) {
        for (i64 d : diag1_set) {
            i64 r = d - c;
            if (1 <= r && r <= N) {
                total_attacked--;
            }
        }
    }

    // Cols and diag2 intersections
    for (i64 c : col_set) {
        for (i64 d : diag2_set) {
            i64 r = d + c;
            if (1 <= r && r <= N) {
                total_attacked--;
            }
        }
    }

    // Diag1 and diag2 intersections
    for (i64 d1 : diag1_set) {
        for (i64 d2 : diag2_set) {
            // Solve:
            // x + y = d1
            // x - y = d2
            // => x = (d1 + d2)/2, y = (d1 - d2)/2
            if ((d1 + d2) % 2 == 0) {
                i64 x = (d1 + d2) / 2;
                i64 y = (d1 - d2) / 2;
                if (1 <= x && x <= N && 1 <= y && y <= N) {
                    total_attacked--;
                }
            }
        }
    }

    // Remove the pieces themselves as they are not empty
    i64 occupied = M;

    // Total valid positions = all grid cells - attacked cells + non-empty cells that are in attacked area
    i64 ans = N * N - total_attacked + occupied;

    cout << ans << "\n";

    return 0;
}
```
