#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<i64> row_set, col_set, diag1_set, diag2_set;
    vector<array<i64, 2>> pieces(M);

    for (int i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        row_set.insert(a);
        col_set.insert(b);
        diag1_set.insert(a + b);
        diag2_set.insert(a - b);
    }

    // Total squares minus occupied ones
    i64 total_squares = N * N;
    i64 occupied = M;

    // Count how many unique positions are attacked
    set<pair<i64, i64>> attacked;

    for (auto [a, b] : pieces) {
        // Rows and columns
        for (i64 x = 1; x <= N; ++x) {
            if (x != a) attacked.insert({x, b});
            if (x != b) attacked.insert({a, x});
        }

        // Diagonals
        i64 sum = a + b;
        i64 diff = a - b;

        // Points on diagonal 1: x + y = sum
        for (i64 x = max(1LL, sum - N); x <= min(N, sum - 1); ++x) {
            i64 y = sum - x;
            if (x >= 1 && y >= 1 && x <= N && y <= N && (x != a || y != b))
                attacked.insert({x, y});
        }

        // Points on diagonal 2: x - y = diff
        if (diff >= -(N - 1) && diff <= N - 1) {
            i64 start_x = max(1LL, diff + 1);
            i64 end_x = min(N, N + diff);
            for (i64 x = start_x; x <= end_x; ++x) {
                i64 y = x - diff;
                if (x != a || y != b)
                    attacked.insert({x, y});
            }
        }
    }

    // Remove positions already occupied by existing pieces
    for (auto [a, b] : pieces)
        attacked.erase({a, b});

    // Result is total squares - occupied - attacked
    i64 result = total_squares - M - attacked.size();

    cout << result << "\n";

    return 0;
}
