```cpp
#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    vector<pair<i64, i64>> positions(M);
    for(i64 i = 0; i < M; ++i){
        cin >> positions[i].first >> positions[i].second;
    }

    set<pair<i64, i64>> occupied_positions(positions.begin(), positions.end());

    i64 count = 0;
    for(i64 i = 1; i <= N; ++i){
        for(i64 j = 1; j <= N; ++j){
            if(!occupied_positions.count({i, j}) && 
               !occupied_positions.count({i, N - j + 1}) &&
               !occupied_positions.count({N - i + 1, j}) &&
               !occupied_positions.count({N - i + 1, N - j + 1})){
                count++;
            }
        }
    }

    cout << count << '\n';

    return 0;
}
```
This solution iterates over all possible positions on the board and checks if they are not already occupied and also not diagonally adjacent to any occupied position. It counts all such valid positions and outputs the result. This solution works within the time constraints provided.
