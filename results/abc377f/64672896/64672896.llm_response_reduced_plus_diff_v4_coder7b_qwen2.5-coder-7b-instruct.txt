#include<bits/stdc++.h>

int main() {

    i64 N, M;
    cin >> N >> M;
    
    vector<array<int,2>> pieces(M);
    set<pair<int,int>> occupied;
    
    for(int i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        occupied.insert({a, b});
    }
    
    unordered_set<int> rows, cols, diagonals1, diagonals2;
    
    for(auto [a, b] : pieces) {
        rows.insert(a);
        cols.insert(b);
        diagonals1.insert(a + b);
        diagonals2.insert(a - b);
    }
    
    i64 count = 0;
    
    for(i64 i = 1; i <= N; ++i) {
        for(i64 j = 1; j <= N; ++j) {
            if(!rows.count(i) && !cols.count(j) && !diagonals1.count(i + j) && !diagonals2.count(i - j)) {
                count++;
            }
        }
    }
    
    cout << count << '\n';
    
    return 0;
}

1. We read the input values `N` and `M`.
2. We store the positions of the existing pieces in a vector `pieces` and a set `occupied`.
3. We use sets `rows`, `cols`, `diagonals1`, and `diagonals2` to keep track of which rows, columns, and diagonals already have pieces.
4. We iterate through each cell `(i, j)` in the grid. If the cell is not occupied by any piece, we increment the counter `count`.
5. Finally, we print the value of `count`.
