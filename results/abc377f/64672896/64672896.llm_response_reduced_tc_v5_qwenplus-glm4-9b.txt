#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;
    vector<pair<long long, long long>> positions(M);
    vector<pair<long long, long long>> unique_positions;

    // Read positions and remove duplicates
    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        positions[i] = {a, b};
        unique_positions.push_back({a, b});
    }
    sort(unique_positions.begin(), unique_positions.end());

    // Collect unique rows, columns, diagonals, and anti-diagonals
    vector<long long> rows, cols, diag1, diag2, anti_diag1, anti_diag2;
    for (const auto& p : unique_positions) {
        rows.push_back(p.first);
        cols.push_back(p.second);
        diag1.push_back(p.first + p.second);
        diag2.push_back(p.first - p.second);
        anti_diag1.push_back(p.first - (N + 1 - p.second));
        anti_diag2.push_back(p.first + (N + 1 - p.second));
    }

    // Remove duplicates from rows, cols, diag1, diag2, anti_diag1, anti_diag2
    sort(rows.begin(), rows.end());
    sort(cols.begin(), cols.end());
    sort(diag1.begin(), diag1.end());
    sort(diag2.begin(), diag2.end());
    sort(anti_diag1.begin(), anti_diag1.end());
    sort(anti_diag2.begin(), anti_diag2.end());
    rows.erase(unique(rows.begin(), rows.end()), rows.end());
    cols.erase(unique(cols.begin(), cols.end()), cols.end());
    diag1.erase(unique(diag1.begin(), diag1.end()), diag1.end());
    diag2.erase(unique(diag2.begin(), diag2.end()), diag2.end());
    anti_diag1.erase(unique(anti_diag1.begin(), anti_diag1.end()), anti_diag1.end());
    anti_diag2.erase(unique(anti_diag2.begin(), anti_diag2.end()), anti_diag2.end());

    // Set the number of taken positions in each row, column, diagonal, and anti-diagonal
    vector<long long> row_count(N + 1, 0), col_count(N + 1, 0);
    vector<long long> diag1_count(2 * N, 0), diag2_count(2 * N, 0);
    vector<long long> anti_diag1_count(2 * N, 0), anti_diag2_count(2 * N, 0);
    for (const auto& p : unique_positions) {
        row_count[p.first]++;
        col_count[p.second]++;
        diag1_count[p.first + p.second]++;
        diag2_count[p.first - p.second]++;
        anti_diag1_count[p.first - (N + 1 - p.second)]++;
        anti_diag2_count[p.first + (N + 1 - p.second)]++;
    }

    // Calculate the number of free squares
    long long free_squares = N * N;
    for (const auto& r : rows) {
        free_squares -= N - row_count[r];
    }
    for (const auto& c : cols) {
        free_squares -= N - col_count[c];
    }
    for (const auto& d : diag1) {
        free_squares -= 2 * (diag1_count[d] - 1);
    }
    for (const auto& d : diag2) {
        free_squares -= 2 * (diag2_count[d] - 1);
    }
    for (const auto& d : anti_diag1) {
        free_squares -= 2 * (anti_diag1_count[d] - 1);
    }
    for (const auto& d : anti_diag2) {
        free_squares -= 2 * (anti_diag2_count[d] - 1);
    }

    cout << free_squares << endl;
    return 0;
}
