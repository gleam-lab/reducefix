```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <map>

using namespace std;
using i64 = long long;

bool isSafe(int x, int y, const vector<pair<int, int>>& pieces) {
    for (auto& p : pieces) {
        int a, b;
        tie(a, b) = p;
        if (x == a || y == b || abs(x - y) == abs(a - b)) {
            return false;
        }
    }
    return true;
}

int main() {
    i64 N, M;
    cin >> N >> M;
    vector<pair<int, int>> pieces(M);
    for (i64 i = 0; i < M; ++i) {
        i64 x, y;
        cin >> x >> y;
        --x, --y; // Convert to 0-based index
        pieces[i] = {x, y};
    }

    i64 safeCount = 0;
    for (i64 i = 0; i < N; ++i) {
        for (i64 j = 0; j < N; ++j) {
            if (isSafe(i, j, pieces)) {
                ++safeCount;
            }
        }
    }

    cout << safeCount << endl;

    return 0;
}
```

This code defines a `isSafe` function that checks if a piece can be placed at a given position (x, y) without being captured by any of the given pieces. It then iterates over all squares of the grid and counts the number of safe squares. The result is printed to the standard output.

Note: This code assumes that the grid is a square grid of size \( N \times N \) and that the positions of the pieces are given in 1-based index format (i.e., (1, 1) is the top-left corner of the grid). The pieces' positions are converted to 0-based index for easier array manipulation.
