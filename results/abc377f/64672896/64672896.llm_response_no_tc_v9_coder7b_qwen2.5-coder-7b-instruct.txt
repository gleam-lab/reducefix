#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    ll N, M;
    cin >> N >> M;
    
    vector<pair<ll, ll>> pieces(M);
    set<ll> sums, diffs;
    unordered_set<ll> rows, cols;

    for(int i = 0; i < M; ++i){
        ll x, y;
        cin >> x >> y;
        pieces[i] = {x, y};
        sums.insert(x + y);
        diffs.insert(x - y);
        rows.insert(x);
        cols.insert(y);
    }

    ll total_empty_squares = N * N - M;
    ll captured_by_row = N - rows.size();
    ll captured_by_col = N - cols.size();

    for(auto& [_, y] : pieces){
        if(rows.count(y)){
            total_empty_squares -= 1;
        }
    }

    for(auto& [x, _] : pieces){
        if(cols.count(x)){
            total_empty_squares -= 1;
        }
    }

    for(auto sum : sums){
        if(sum > 0 && sum <= N){
            total_empty_squares -= 1;
        }
    }

    for(auto diff : diffs){
        if(diff >= -(N-1) && diff <= N-1){
            total_empty_squares -= 1;
        }
    }

    cout << total_empty_squares << '\n';

    return 0;
}
