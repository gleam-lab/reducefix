#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<array<int, 2>> g(M);
    vector<i64> A, B, C, D;
    for (i64 i = 0; i < M; i++) {
        i64 a, b;
        cin >> a >> b;
        g[i] = {a, b};
        C.push_back(a);
        D.push_back(b);
        A.push_back(a + b);
        B.push_back(a - b);
    }
    sort(C.begin(), C.end());
    sort(D.begin(), D.end());
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());

    // Remove duplicates from A and B
    A.erase(unique(A.begin(), A.end()), A.end());
    B.erase(unique(B.begin(), B.end()), B.end());

    // Calculate the number of attackable squares for each line
    i64 count_lines = 0;
    for (i64 a : A) count_lines += min(2 * N - a, (i64)1e9) - max(2 * N + a, (i64)1) + 1;
    for (i64 b : B) count_lines += min(2 * N - b, (i64)1e9) - max(2 * N + b, (i64)1) + 1;

    // Remove duplicates from C and D
    set<i64> temp;
    for (i64 c : C) temp.insert(c);
    C.assign(temp.begin(), temp.end());
    temp.clear();
    for (i64 d : D) temp.insert(d);
    D.assign(temp.begin(), temp.end());

    // Calculate the number of attackable squares for each row and column
    i64 count_rows = 0, count_cols = 0;
    for (i64 c : C) count_rows += min(N, c) - max(1LL, N - c) + 1;
    for (i64 d : D) count_cols += min(N, d) - max(1LL, N - d) + 1;

    // Calculate the number of attackable squares for each diagonal
    i64 count_dia1 = 0, count_dia2 = 0;
    for (i64 c : C) count_dia1 += min(2 * N - c, c) - max(c - 1LL, 1) + 1;
    for (i64 d : D) count_dia2 += min(2 * N - d, d) - max(d - 1LL, 1) + 1;

    // Calculate the total number of attackable squares
    i64 total_attackable = count_lines + count_rows + count_cols + count_dia1 + count_dia2;

    // Calculate the number of safe squares
    i64 safe_squares = (N * N) - total_attackable;

    cout << safe_squares << endl;

    return 0;
}
