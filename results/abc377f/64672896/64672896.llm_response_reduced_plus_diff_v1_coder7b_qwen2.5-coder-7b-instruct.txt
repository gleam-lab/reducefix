#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {

    cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    vector<pair<i64, i64>> pieces(M);
    set<pair<i64, i64>> positions;

    for(i64 i = 0; i < M; ++i){
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        positions.insert({a, b});
    }

    unordered_set<i64> sum_lines, diff_lines, row_lines, col_lines;

    for(auto &piece : pieces){
        sum_lines.insert(piece.first + piece.second);
        diff_lines.insert(piece.first - piece.second);
        row_lines.insert(piece.first);
        col_lines.insert(piece.second);
    }

    i64 count = 0;

    for(i64 i = 1; i <= N; ++i){
        for(i64 j = 1; j <= N; ++j){
            if(positions.find({i, j}) != positions.end())
                continue;

            bool can_place = true;

            if(row_lines.find(i) != row_lines.end())
                can_place = false;
            
            if(col_lines.find(j) != col_lines.end())
                can_place = false;

            if(sum_lines.find(i + j) != sum_lines.end())
                can_place = false;

            if(diff_lines.find(i - j) != diff_lines.end())
                can_place = false;

            if(can_place)
                ++count;
        }
    }

    cout << count << '\n';

    return 0;
}

### Explanation:
1. **Efficient Data Structures**: We use sets (`sum_lines`, `diff_lines`, `row_lines`, `col_lines`) to store all unique sums, differences, rows, and columns occupied by pieces. This allows us to quickly check if placing a piece at a specific position would result in a capture.
   
2. **Iterate Over Grid**: We iterate over each cell `(i, j)` in the grid. If the cell is already occupied, we skip it.

3. **Check Capture Conditions**: For each empty cell, we check if it can be attacked by any of the existing pieces using the precomputed sets. If none of the attack conditions are met, the cell is valid for placing a new piece.

4. **Count Valid Positions**: We count all valid cells where no piece can capture a new piece and output this count.

This approach ensures that the solution is efficient even for large values of \(N\) and \(M\), as the time complexity is \(O(N \times M + N^2)\), which is feasible within the given constraints.
