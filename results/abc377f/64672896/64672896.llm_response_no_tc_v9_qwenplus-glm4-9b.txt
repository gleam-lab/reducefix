#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<array<int,2>> pieces(M);
    vector<i64> rows, cols, diag1, diag2;

    for (int i = 0; i < M; ++i) {
        i64 r, c;
        cin >> r >> c;
        --r, --c; // Convert to zero-indexed
        pieces[i] = {r, c};
        rows.push_back(r);
        cols.push_back(c);
        diag1.push_back(r + c);
        diag2.push_back(r - c);
    }

    // Unique the rows, cols, diag1, diag2 and map them back to indices
    sort(rows.begin(), rows.end());
    sort(cols.begin(), cols.end());
    sort(diag1.begin(), diag1.end());
    sort(diag2.begin(), diag2.end());

    vector<i64> unique_rows(rows.begin(), unique(rows.begin(), rows.end()) - 1);
    vector<i64> unique_cols(cols.begin(), unique(cols.begin(), cols.end()) - 1);
    vector<i64> unique_diag1(diag1.begin(), unique(diag1.begin(), diag1.end()) - 1);
    vector<i64> unique_diag2(diag2.begin(), unique(diag2.end()) - 1);

    // Create maps for quick index lookup
    map<i64, i64> row_map, col_map, diag1_map, diag2_map;
    for (i64 i = 0; i < unique_rows.size(); ++i) {
        row_map[unique_rows[i]] = i;
    }
    for (i64 i = 0; i < unique_cols.size(); ++i) {
        col_map[unique_cols[i]] = i;
    }
    for (i64 i = 0; i < unique_diag1.size(); ++i) {
        diag1_map[unique_diag1[i]] = i;
    }
    for (i64 i = 0; i < unique_diag2.size(); ++i) {
        diag2_map[unique_diag2[i]] = i;
    }

    // Calculate the number of valid positions
    i64 ans = N * N; // Total number of squares
    for (const auto& p : pieces) {
        ans -= row_map[p[0]] + 1; // This row is not available
        ans -= col_map[p[1]] + 1; // This column is not available
        ans -= diag1_map[p[0] + p[1]] + 1; // This diagonal is not available
        ans -= diag2_map[p[0] - p[1]] + 1; // This anti-diagonal is not available
    }

    cout << ans << "\n";
    return 0;
}
