#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using i64 = long long;

int main() {
    i64 N, M;
    cin >> N >> M;
    vector<pair<i64, i64>> pieces(M);
    
    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        a--; b--; // Convert to 0-indexed
        pieces[i] = {a, b};
    }

    // Total number of squares
    i64 total_squares = N * N;
    
    // Mark the squares captured by each piece
    vector<vector<bool>> captured(N, vector<bool>(N, false));
    
    for (const auto& p : pieces) {
        i64 x = p.first, y = p.second;
        // Rows and columns
        for (i64 i = 0; i < N; ++i) {
            captured[i][y] = true;
            captured[x][i] = true;
        }
        // Diagonals x+y and x-y
        for (i64 i = 0; i < N; ++i) {
            if (x + y <= N - 1) captured[x + y][i] = true;
            if (x - y >= 0) captured[x - y][i] = true;
        }
    }

    // Calculate the number of safe squares
    i64 safe_squares = 0;
    for (i64 i = 0; i < N; ++i) {
        for (i64 j = 0; j < N; ++j) {
            if (!captured[i][j]) {
                safe_squares++;
            }
        }
    }
    
    cout << safe_squares << endl;

    return 0;
}
