To solve this problem efficiently, we must avoid iterating over the entire N×N grid (since N can be as large as 10⁹). Instead, we compute the number of **safe** squares using **set operations** and **inclusion-exclusion principle**, tracking only the affected rows, columns, and diagonals.

Here's a breakdown of the approach:

### Key Observations:
- Each piece attacks in four directions: same row, same column, main diagonal (`i - j`), and anti-diagonal (`i + j`).
- We track all unique attacked rows, columns, and diagonals.
- The total number of attacked positions is calculated by combining:
  - All attacked rows × N
  - All attacked columns × N
  - All attacked main diagonals (each contributes up to N points)
  - All attacked anti-diagonals (also up to N points each)
- Then subtract overlaps:
  - Intersections between rows and columns
  - Intersections between rows and diagonals
  - Intersections between columns and diagonals
  - Intersections between both diagonals

We use sets for efficient lookups and avoid double-counting.

---

Here is the corrected and optimized C++ code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<i64> rows, cols, diag1, diag2;
    vector<pair<i64, i64>> pieces(M);

    for (int i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total number of attacked cells
    i64 attacked = 0;

    // Rows and columns contribute N cells each
    attacked += (i64)rows.size() * N;
    attacked += (i64)cols.size() * N;

    // Diagonals also contribute up to N cells each
    attacked += (i64)diag1.size() * N;
    attacked += (i64)diag2.size() * N;

    // Subtract overlaps

    // Row & Column intersections
    for (auto r : rows) {
        if (cols.count(r)) {
            attacked -= 1; // cell (r, r) counted twice
        }
    }

    // Row & Diag1 intersections
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = d - r;
            if (1 <= c && c <= N) {
                attacked -= 1;
            }
        }
    }

    // Row & Diag2 intersections
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = r - d;
            if (1 <= c && c <= N) {
                attacked -= 1;
            }
        }
    }

    // Col & Diag1 intersections
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = d - c;
            if (1 <= r && r <= N) {
                attacked -= 1;
            }
        }
    }

    // Col & Diag2 intersections
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d + c;
            if (1 <= r && r <= N) {
                attacked -= 1;
            }
        }
    }

    // Diag1 & Diag2 intersections
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            i64 x = (d1 + d2) / 2;
            i64 y = (d1 - d2) / 2;
            if ((d1 + d2) % 2 == 0 && (d1 - d2) % 2 == 0 &&
                1 <= x && x <= N && 1 <= y && y <= N) {
                attacked -= 1;
            }
        }
    }

    // Now subtract the exact number of attacked cells from total cells
    cout << N * N - attacked << "\n";

    return 0;
}
```

### Summary:
- Efficiently computes the attack zones without iterating through the full grid.
- Uses sets to track unique attack lines.
- Applies inclusion-exclusion to count overlapping attack zones.
- Works within constraints due to its complexity being O(M²), which is acceptable since M ≤ 1000.
