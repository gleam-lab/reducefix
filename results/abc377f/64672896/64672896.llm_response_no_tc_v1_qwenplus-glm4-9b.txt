#include<bits/stdc++.h>
using namespace std;

using i64 = long long;

int main() {
    i64 N, M;
    cin >> N >> M;
    vector<pair<i64, i64>> pieces(M);
    vector<i64> rows(N + 1, 0), cols(N + 1, 0), diag1(N + 1, 0), diag2(N + 1, 0);
    
    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        rows[a]++;
        cols[b]++;
        diag1[a + b]++;
        diag2[a - b]++;
    }
    
    i64 occupied = 0;
    for (i64 i = 1; i <= N; ++i) {
        occupied += rows[i];
        occupied += cols[i];
        occupied += diag1[i];
        occupied += diag2[i];
    }
    
    i64 totalEmpty = N * N - occupied;
    
    // Subtract the cases where the empty square is in the same row or column as an occupied square
    for (i64 i = 1; i <= N; ++i) {
        occupied += rows[i] * (N - rows[i]);
        occupied += cols[i] * (N - cols[i]);
        occupied += diag1[i] * (N - diag1[i]);
        occupied += diag2[i] * (N - diag2[i]);
    }
    
    // Subtract the cases where the empty square is in the same diagonal as an occupied square
    for (i64 i = 0; i <= 2 * N; ++i) {
        occupied += diag1[i] * (N - diag1[i]);
        occupied += diag2[i] * (N - diag2[i]);
    }
    
    totalEmpty -= occupied;
    
    // Subtract the cases where the empty square is in the same anti-diagonal as an occupied square
    occupied = 0;
    for (i64 i = 1; i <= N; ++i) {
        occupied += (cols[i] + rows[N + 1 - i]) * (N - cols[i] - rows[N + 1 - i]);
    }
    
    totalEmpty -= occupied;
    
    cout << totalEmpty << endl;
    
    return 0;
}
