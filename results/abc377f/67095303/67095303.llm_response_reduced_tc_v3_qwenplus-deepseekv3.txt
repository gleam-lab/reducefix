2. **Diagonals (i + j = d)**: For each diagonal defined by `i + j = d`, calculate the number of squares on this diagonal that are not already under attack by any row or column.
3. **Diagonals (i - j = d)**: Similarly, for each diagonal defined by `i - j = d`, calculate the number of squares on this diagonal not under attack by any row or column.
4. **Intersections**: Ensure that intersections between diagonals and rows or columns are correctly accounted for to avoid double-counting.

void solve() {

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    int safe_rows = N - rows.size();
    int safe_cols = N - cols.size();
    int ans = safe_rows * safe_cols;

    // For diag1 (i + j = d)
    for (int d : diag1) {
        int i_min = max(1LL, d - N);
        int i_max = min(N, d - 1);
        int cnt = i_max - i_min + 1;

        set<int> affected_rows;
        for (int i : rows) {
            if (d - i >= 1 && d - i <= N) {
                affected_rows.insert(i);
            }
        }
        set<int> affected_cols;
        for (int j : cols) {
            if (d - j >= 1 && d - j <= N) {
                affected_cols.insert(d - j);
            }
        }

        int total_affected = affected_rows.size() + affected_cols.size();
        for (int i : affected_rows) {
            if (affected_cols.count(i)) {
                total_affected--;
            }
        }

        cnt -= affected_rows.size();
        cnt -= affected_cols.size();
        cnt += total_affected;
        ans -= cnt;
    }

    // For diag2 (i - j = d)
    for (int d : diag2) {
        int i_min = max(1LL, 1 + d);
        int i_max = min(N, N + d);
        int cnt = i_max - i_min + 1;

        for (int i : rows) {
            if (i - d >= 1 && i - d <= N) {
                affected_rows.insert(i);
            }

        for (int j : cols) {
            if (j + d >= 1 && j + d <= N) {
                affected_cols.insert(j + d);
            }
        }

        int total_affected = affected_rows.size() + affected_cols.size();
        for (int i : affected_rows) {
            if (affected_cols.count(i)) {
                total_affected--;
            }
        }

        cnt -= affected_rows.size();
        cnt -= affected_cols.size();
        cnt += total_affected;
        ans -= cnt;
    }

    // Now, handle intersections between diag1 and diag2
    for (int d1 : diag1) {
        for (int d2 : diag2) {
            if ((d1 + d2) % 2 != 0) continue;
            int i = (d1 + d2) / 2;

                }
            }
        }
    }

2. **Diagonals (i + j = d)**: For each diagonal, we calculate the number of squares on the diagonal that are not already attacked by any row or column. This is done by adjusting the count of squares on the diagonal by subtracting those that are in attacking rows or columns.
3. **Diagonals (i - j = d)**: Similarly, for each diagonal `i - j = d`, we adjust the count by subtracting squares attacked by rows or columns.
4. **Intersections**: Finally, we handle intersections between diagonals of type `i + j = d` and `i - j = d` to ensure squares at their intersection are not double-counted.
