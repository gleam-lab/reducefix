#include <bits/stdc++.h>
#define int long long
#define PII pair<int, int>

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> row_set, col_set, diag1_set, diag2_set;
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        row_set.insert(a);
        col_set.insert(b);
        diag1_set.insert(a + b);
        diag2_set.insert(a - b);
    }

    // Total empty squares not attacked by existing pieces
    int total_empty = n * n - m;

    // First subtract all squares in same row or column as any piece
    // We need to be careful not to double-count

    // Number of squares in any of the occupied rows or columns:
    // = #rows * n + #cols * n - #rows * #cols
    int rows = row_set.size();
    int cols = col_set.size();
    int attacked_by_row_or_col = rows * n + cols * n - rows * cols;

    // Subtract these from total empty squares
    total_empty -= attacked_by_row_or_col;

    // Now handle diagonal attacks carefully
    // We will only count those that are NOT already in any of the rows or columns

    // For each diag1 = a + b
    for (int d : diag1_set) {
        // Determine how many valid (x,y) on this diagonal satisfy 1<=x<=n, 1<=y<=n
        // x + y = d
        int low = max(1LL, d - n);
        int high = min(n, d - 1);
        int count = high - low + 1;

        // Now subtract those already counted via rows or cols
        for (int r : row_set) {
            int c = d - r;
            if (1 <= c && c <= n)
                count--;
        }
        for (int c : col_set) {
            int r = d - c;
            if (1 <= r && r <= n && row_set.find(r) == row_set.end())
                count--;
        }

        total_empty -= count;
    }

    // For each diag2 = a - b
    for (int d : diag2_set) {
        // x - y = d => x = y + d
        int low_y = max(1LL, 1 - d);  // y >= 1 - d
        int high_y = min(n, n - d);  // y <= n - d
        int count = max(0LL, high_y - low_y + 1);

        // Subtract those already counted via rows or cols
        for (int c : col_set) {
            int r = d + c;
            if (1 <= r && r <= n)
                count--;
        }
        for (int r : row_set) {
            int c = r - d;
            if (1 <= c && c <= n && col_set.find(c) == col_set.end())
                count--;
        }

        total_empty -= count;
    }

    cout << total_empty << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }

    return 0;
}
