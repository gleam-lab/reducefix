#include<bits/stdc++.h>

int qpowm(int a, int b, int m) {
	int r = 1;
	while (b) {
		if (b & 1)(r *= a) %= m;
		(a *= a) %= m, b >>= 1;
	}
	return r;
}

int qpow(int a, int b) {
	int r = 1;
	while (b) {
		if (b & 1)r *= a;
		a *= a;
		b >>= 1;
	}
	return r;
}

void solve() {
	int n, m;
	cin >> n >> m;
	vector<set<int>> h(n + 1), v(n + 1), d1(2 * n + 1), d2(2 * n + 1);

	for (int i = 1, x, y; i <= m; i++) {
		cin >> x >> y;
		h[x].insert(y);
		v[y].insert(x);
		d1[x + y].insert(x);
		d2[x - y].insert(x);
	}

	int ans = n * n - m;

	for (int i = 1; i <= n; i++) {
		if (!h[i].empty()) {
			int maxv = *h[i].rbegin(), minv = *h[i].begin();
			ans -= (maxv - minv + 1) - h[i].size();
		}
	}

	for (int i = 1; i <= n; i++) {
		if (!v[i].empty()) {
			int maxv = *v[i].rbegin(), minv = *v[i].begin();
			ans -= (maxv - minv + 1) - v[i].size();
		}
	}

	for (int i = -n; i <= n; i++) {
		if (!d1[i + n].empty()) {
			int maxv = *d1[i + n].rbegin(), minv = *d1[i + n].begin();
			ans -= (maxv - minv + 1) - d1[i + n].size();
		}
	}

	for (int i = -n; i <= n; i++) {
		if (!d2[i + n].empty()) {
			int maxv = *d2[i + n].rbegin(), minv = *d2[i + n].begin();
			ans -= (maxv - minv + 1) - d2[i + n].size();
		}
	}

	cout << ans << endl;
}

signed main() {
	int _ = 1;
	while (_) {
		solve();
	}
	return 0;
}
