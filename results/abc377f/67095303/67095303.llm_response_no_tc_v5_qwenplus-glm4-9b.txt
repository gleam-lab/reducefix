#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    vector<pair<int, int>> points(m);
    for (int i = 0; i < m; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    set<int> rows, cols, diag1, diag2;

    // Collect all relevant rows, columns, and diagonals
    for (const auto& p : points) {
        rows.insert(p.first);
        cols.insert(p.second);
        diag1.insert(p.first + p.second);
        diag2.insert(p.first - p.second);
    }

    // Calculate the number of empty squares
    long long emptySquares = n * n;

    // Subtract squares occupied by existing pieces
    emptySquares -= rows.size() * n;
    emptySquares -= cols.size() * n;

    // Subtract squares on both diagonals
    emptySquares -= diag1.size() * n;
    emptySquares -= diag2.size() * n;

    // Subtract squares at the intersection of two diagonals
    set<int> commonDiagonals;
    for (int d : diag1) {
        if (diag2.count(d)) {
            commonDiagonals.insert(d);
        }
    }
    emptySquares -= commonDiagonals.size() * n;

    // Output the result
    cout << emptySquares << endl;

    return 0;
}
