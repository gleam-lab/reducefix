#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int N, M;
    cin >> N >> M;

    set<int> rows, cols;
    set<int> diag1, diag2; // diag1: i+j, diag2: i-j

    for (int i = 0; i < M; ++i) {
        int r, c;
        cin >> r >> c;
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r + c);
        diag2.insert(r - c);
    }

    // Total safe cells = Total cells not in any attacked row or column
    int ans = (N - rows.size()) * (N - cols.size());

    // Now subtract the number of cells on diagonals that are not already excluded by row/col elimination
    for (int d : diag1) {
        int count = 0;
        // Check intersection with rows
        for (int r : rows) {
            int c = d - r;
            if (c >= 1 && c <= N) {
                count++;
            }
        }
        // Check intersection with columns
        for (int c : cols) {
            int r = d - c;
            if (r >= 1 && r <= N) {
                count++;
            }
        }
        // Count total cells on this diagonal
        int len;
        if (d <= N + 1) {
            len = d - 1;
        } else {
            len = 2 * N + 1 - d;
        }
        ans -= (len - count); // Subtract those cells that are on this diagonal and not already counted out
    }

    for (int d : diag2) {
        int count = 0;
        // Check intersection with rows
        for (int r : rows) {
            int c = r - d;
            if (c >= 1 && c <= N) {
                count++;
            }
        }
        // Check intersection with columns
        for (int c : cols) {
            int r = d + c;
            if (r >= 1 && r <= N) {
                count++;
            }
        }
        // Count total cells on this diagonal
        int len;
        if (d >= 0) {
            len = N - d;
        } else {
            len = N + d;
        }
        ans -= (len - count); // Subtract those cells that are on this diagonal and not already counted out
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T = 1;
    // For multiple test cases
    // cin >> T;
    while (T--) {
        solve();
    }

    return 0;
}
