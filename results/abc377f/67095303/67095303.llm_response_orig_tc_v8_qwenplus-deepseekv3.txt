#include<bits/stdc++.h>

void solve() {
    int N, M;
    cin >> N >> M;
    set<int> rows, cols, diag1, diag2;
    
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }
    
    int total_covered = 0;
    
    // Rows and columns
    total_covered += rows.size() * N;
    total_covered += cols.size() * N;
    total_covered -= rows.size() * cols.size();
    
    // Diagonals (i + j = d)
    for (int d : diag1) {
        int min_i = max(1LL, d - N);
        int max_i = min(N, d - 1);

        int cnt = max_i - min_i + 1;
        total_covered += cnt;
        
        // Subtract overlaps with rows
        for (int r : rows) {
            if (d - r >= 1 && d - r <= N) {
                total_covered--;

        for (int c : cols) {

                total_covered--;

                total_covered--;
            }
        }

    // Diagonals (i - j = e)
    for (int e : diag2) {
        int min_j = max(1LL, 1 - e);
        int max_j = min(N, N - e);
        if (min_j > max_j) continue;
        int cnt = max_j - min_j + 1;
        total_covered += cnt;
        
        // Subtract overlaps with rows
        for (int r : rows) {
            if (r - e >= 1 && r - e <= N) {
                total_covered--;
            }
        }
        
        // Subtract overlaps with columns
        for (int c : cols) {
            if (e + c >= 1 && e + c <= N) {
                total_covered--;
            }
        }
        
        // Subtract overlaps with other diagonals (i + j = d)
        for (int d : diag1) {

                total_covered--;
            }
        }

    // Total safe squares
    int total_squares = N * N;
    int safe_squares = total_squares - total_covered + M; // Adding M because existing pieces are already counted in rows/cols/diags

}

2. **Coverage Calculation**:

   - **Diagonals (i + j = d)**: For each diagonal, we calculate the number of squares it covers, subtract overlaps with rows, columns, and other diagonals (i - j = e).
   - **Diagonals (i - j = e)**: Similarly, we calculate coverage for these diagonals and subtract overlaps.
3. **Safe Squares**: The total safe squares are obtained by subtracting the total covered squares from the total grid squares and adding the count of existing pieces (since they were subtracted twice in the overlap calculations).

This approach ensures that all overlaps are correctly accounted for, providing the accurate count of safe squares.
