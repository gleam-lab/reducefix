#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    set<pair<int, int>> pieces;
    set<int> row, col, diag1, diag2, anti_diag1, anti_diag2;

    for (long long i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        pieces.insert({a, b});
        row.insert(a);
        col.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
        anti_diag1.insert(-a + b);
        anti_diag2.insert(-a - b);
    }

    long long total_empty_squares = n * n;
    long long used_squares = row.size() + col.size() + diag1.size() + diag2.size() + anti_diag1.size() + anti_diag2.size();

    // Subtract the total number of squares occupied by pieces and their lines
    total_empty_squares -= used_squares;

    // Subtract the intersections of these lines
    for (auto p : pieces) {
        int r = p.first;
        int c = p.second;
        for (auto dr : diag1) {
            if (r + c == dr) {
                used_squares++;
            }
        }
        for (auto dr : diag2) {
            if (r - c == dr) {
                used_squares++;
            }
        }
        for (auto dr : anti_diag1) {
            if (-r + c == dr) {
                used_squares++;
            }
        }
        for (auto dr : anti_diag2) {
            if (-r - c == dr) {
                used_squares++;
            }
        }
    }

    cout << total_empty_squares - used_squares << endl;

    return 0;
}
