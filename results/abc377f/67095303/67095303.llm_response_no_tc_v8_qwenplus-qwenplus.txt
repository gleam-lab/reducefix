#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int N, M;
    cin >> N >> M;

    set<int> rows, cols;
    set<int> diag1, diag2; // diag1: i+j, diag2: i-j

    for (int i = 0; i < M; ++i) {
        int r, c;
        cin >> r >> c;
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r + c);
        diag2.insert(r - c);
    }

    // Start with total number of cells not in any attacked row or column
    int ans = (N - (int)rows.size()) * (N - (int)cols.size());

    // Subtract squares attacked by diagonal 1 (i+j = d)
    for (int d : diag1) {
        // Calculate how many squares are on this diagonal within the grid
        int lo = max(1LL, d - N);
        int hi = min(N, d - 1);
        if (lo > hi) continue;
        int diag_len = hi - lo + 1;

        // Count how many of these are already excluded by attacked rows or columns
        int blocked = 0;
        for (int r : rows) {
            int c = d - r;
            if (1 <= c && c <= N) blocked++;
        }
        for (int c : cols) {
            int r = d - c;
            if (1 <= r && r <= N && rows.find(r) == rows.end()) blocked++;
        }

        ans -= (diag_len - blocked);
    }

    // Subtract squares attacked by diagonal 2 (i-j = d)
    for (int d : diag2) {
        // Calculate how many squares are on this diagonal within the grid
        int lo = max(1LL, 1 + d);
        int hi = min(N, N + d);
        if (lo > hi) continue;
        int diag_len = hi - lo + 1;

        // Count how many of these are already excluded by attacked rows or columns
        int blocked = 0;
        for (int r : rows) {
            int c = r - d;
            if (1 <= c && c <= N) blocked++;
        }
        for (int c : cols) {
            int r = c + d;
            if (1 <= r && r <= N && rows.find(r) == rows.end()) blocked++;
        }

        ans -= (diag_len - blocked);
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    return 0;
}
