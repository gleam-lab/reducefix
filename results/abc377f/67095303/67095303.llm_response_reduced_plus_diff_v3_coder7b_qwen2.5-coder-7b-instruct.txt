#define ll long long
using namespace std;

const ll MOD = 1e9 + 7;

ll qpow(ll a, ll b) {
    ll res = 1;
    while(b > 0) {
        if(b & 1) 
            res = res * a % MOD;
        a = a * a % MOD;
        b >>= 1;
    }
    return res;
}

ll modinv(ll a) {
    return qpow(a, MOD - 2);
}

void solve() {
    ll n, m;
    cin >> n >> m;
    
    vector<ll> row(n + 1), col(n + 1), diag1(2 * n + 1), diag2(2 * n + 1);

    for(int i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        --x, --y;
        row[x]++, col[y]++;
        diag1[x + y]++;
        diag2[n - x + y]++;
    }

    ll ans = (n * n - accumulate(row.begin(), row.end(), 0LL)) % MOD;
    ans = (ans * modinv(n - accumulate(col.begin(), col.end(), 0LL))) % MOD;
    ans = (ans * modinv(n - accumulate(diag1.begin(), diag1.end(), 0LL))) % MOD;
    ans = (ans * modinv(n - accumulate(diag2.begin(), diag2.end(), 0LL))) % MOD;
    
    cout << (ans + MOD) % MOD << '\n';
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int t;
    cin >> t;
    while(t--)
        solve();
    return 0;
}
