#include <bits/stdc++.h>
#define int long long
#define PII pair<int, int>

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols;
    set<int> diag1, diag2;

    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total empty cells not attacked by row or column
    int ans = (n - rows.size()) * (n - cols.size());

    // Subtract the intersections on diagonal 1: i+j=d
    for (int d : diag1) {
        set<int> blocked;

        // Intersections with occupied rows
        for (int r : rows) {
            int c = d - r;
            if (c >= 1 && c <= n)
                blocked.insert(r);
        }

        // Intersections with occupied columns
        for (int c : cols) {
            int r = d - c;
            if (r >= 1 && r <= n)
                blocked.insert(r);
        }

        // Count number of points on this diagonal
        int cnt;
        if (d <= n + 1) {
            cnt = d - 1;
        } else {
            cnt = 2 * n + 1 - d;
        }

        ans -= (cnt - blocked.size());
    }

    // Subtract the intersections on diagonal 2: i-j=d
    for (int d : diag2) {
        set<int> blocked;

        // Intersections with occupied rows
        for (int r : rows) {
            int c = r - d;
            if (c >= 1 && c <= n)
                blocked.insert(r);
        }

        // Intersections with occupied columns
        for (int c : cols) {
            int r = d + c;
            if (r >= 1 && r <= n)
                blocked.insert(r);
        }

        // Intersections with diag1
        for (int pd : diag1) {
            // Solve:
            // i + j = pd
            // i - j = d
            // => i = (pd + d)/2
            // => j = (pd - d)/2

            if ((pd + d) % 2 == 0 && (pd - d) % 2 == 0) {
                int i = (pd + d) / 2;
                int j = (pd - d) / 2;
                if (i >= 1 && i <= n && j >= 1 && j <= n)
                    blocked.insert(i);
            }
        }

        // Count number of points on this diagonal
        int cnt;
        if (d >= 0) {
            cnt = n - d;
        } else {
            cnt = n + d;
        }

        ans -= (cnt - blocked.size());
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    solve();

    return 0;
}
