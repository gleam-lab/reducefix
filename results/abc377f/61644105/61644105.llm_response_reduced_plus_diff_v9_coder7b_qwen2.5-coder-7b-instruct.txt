```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);

    ll n, m; cin >> n >> m;
    
    set<ll> rowSet, colSet, diagSumSet, antiDiagSumSet;
    vector<pair<ll, ll>> positions(m);
    
    for(ll i = 0; i < m; ++i){
        ll x, y; cin >> x >> y;
        positions[i] = {x, y};
        rowSet.insert(x);
        colSet.insert(y);
        diagSumSet.insert(x+y);
        antiDiagSumSet.insert(x-y);
    }
    
    ll totalPositions = n*n;
    ll occupiedPositions = m;
    
    ll availablePositions = totalPositions - occupiedPositions;
    
    for(auto [x, y]: positions){
        availablePositions -= 
            (rowSet.count(x) - 1 + colSet.count(y) - 1 +
             diagSumSet.count(x+y) - 1 + antiDiagSumSet.count(x-y) - 1);
    }
    
    cout << availablePositions << endl;
    
    return 0;
}
```
