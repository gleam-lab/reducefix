```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;

    set<long long> rows, cols;
    set<pair<long long, long long>> diag1, diag2;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        rows.insert(a);
        cols.insert(b);
        diag1.insert({a + b, a}); // (x + y)
        diag2.insert({a - b, a}); // (x - y)
    }

    long long total_empty = N * N - M;
    long long attacked = 0;

    // Rows and columns
    attacked += rows.size() * N;
    attacked += cols.size() * N;

    // Diagonals (a+b)
    for (auto [sum_ab, x] : diag1) {
        attacked += N;
    }

    // Anti-diagonals (a-b)
    for (auto [diff_ab, x] : diag2) {
        attacked += N;
    }

    // Subtract overlaps

    // Row-Column intersections
    for (auto r : rows) {
        if (cols.find(r) != cols.end()) {
            attacked -= 1; // intersection point
        }
    }

    // Row-Diagonal overlaps
    for (auto r : rows) {
        for (auto [sum_ab, x] : diag1) {
            long long c = sum_ab - r;
            if (c >= 1 && c <= N) {
                attacked -= 1;
            }
        }
        for (auto [diff_ab, x] : diag2) {
            long long c = diff_ab + r;
            if (c >= 1 && c <= N) {
                attacked -= 1;
            }
        }
    }

    // Column-Diagonal overlaps
    for (auto c : cols) {
        for (auto [sum_ab, x] : diag1) {
            long long r = sum_ab - c;
            if (r >= 1 && r <= N) {
                attacked -= 1;
            }
        }
        for (auto [diff_ab, x] : diag2) {
            long long r = diff_ab + c;
            if (r >= 1 && r <= N) {
                attacked -= 1;
            }
        }
    }

    // Diagonal-Anti-diagonal overlaps
    for (auto [sum_ab, x] : diag1) {
        for (auto [diff_ab, x2] : diag2) {
            long long r = (sum_ab + diff_ab) / 2;
            long long c = (sum_ab - diff_ab) / 2;
            if (r >= 1 && r <= N && c >= 1 && c <= N) {
                attacked -= 1;
            }
        }
    }

    // Subtract the pieces themselves as they are not empty
    attacked -= M;

    long long result = max(0LL, total_empty - attacked);
    cout << result << endl;

    return 0;
}
```
