```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> positions(m);
    for(int i = 0; i < m; ++i) {
        cin >> positions[i].first >> positions[i].second;
    }

    sort(positions.begin(), positions.end());

    int total_squares = n * n;
    int empty_squares = total_squares;

    for(int i = 0; i < m; ++i) {
        int x = positions[i].first, y = positions[i].second;
        bool found = false;

        for(int j = 0; j < i; ++j) {
            int x2 = positions[j].first, y2 = positions[j].second;
            if(x == x2 || y == y2 || x + y == x2 + y2 || x - y == x2 - y2) {
                found = true;
                break;
            }
        }

        if(found) {
            --empty_squares;
        } else {
            break;
        }
    }

    cout << empty_squares << endl;

    return 0;
}
```
