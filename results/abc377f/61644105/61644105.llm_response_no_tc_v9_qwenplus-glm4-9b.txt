
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

long long n, m;

struct Point {
    int x, y;
};

long long calculateEmptyPositions(int row, int col, const vector<Point>& occupied) {
    long long total = n * n;
    int count = 0;

    // Count covered positions by existing pieces
    for (const auto& p : occupied) {
        if (p.x == row || p.y == col) {
            count++;
        } else if (abs(p.x - p.y) == abs(row - col)) {
            count++;
        } else if (abs(p.x + p.y) == abs(row + col)) {
            count++;
        }
    }

    return total - count;
}

int main() {
    cin >> n >> m;
    vector<Point> occupied(m);
    for (int i = 0; i < m; i++) {
        cin >> occupied[i].x >> occupied[i].y;
    }

    long long ans = 0;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            ans += calculateEmptyPositions(i, j, occupied);
        }
    }

    cout << ans << endl;
    return 0;
}
```

This code calculates the number of empty squares where a piece can be placed without being captured. It uses a brute-force method to check every empty square and checks if it is safe by checking against all given pieces. The `calculateEmptyPositions` function returns the number of empty positions for a given square `(row, col)`. The main function iterates over all squares and sums up the count of safe positions.
