
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
using namespace std;

using LL = long long;

int n, m;
vector<pair<int, int>> points;

bool isCapture(int x, int y, int px, int py) {
    // Check if (x, y) can be captured by (px, py)
    return (x == px || y == py) || 
           (x + y == px + py) || 
           (x - y == px - py);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> n >> m;
    points.resize(m);
    for (int i = 0; i < m; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    set<pair<int, int>> reachable;
    for (auto& p : points) {
        reachable.insert(p);
        reachable.insert({p.first, n - p.second + 1});
        reachable.insert({n - p.first + 1, p.second});
        reachable.insert({p.first - p.second + n + 1, p.second});
        reachable.insert({p.first + p.second + 1, p.second});
    }

    int freeSquares = n * n - m;
    for (int x = 1; x <= n; ++x) {
        for (int y = 1; y <= n; ++y) {
            bool canPlace = true;
            for (auto& p : points) {
                if (isCapture(x, y, p.first, p.second)) {
                    canPlace = false;
                    break;
                }
            }
            if (canPlace) {
                if (reachable.find({x, y}) == reachable.end()) {
                    freeSquares++;
                }
            }
        }
    }

    cout << freeSquares << endl;
    return 0;
}
```

This code initializes a set of points that are reachable by any of the existing pieces. It then calculates the total number of free squares and subtracts those that are already considered occupied (either by pieces or by their reachable zones). The code checks each square to see if it can be placed on without being captured by any existing piece. If a square is not in the reachable set, it is considered free and the count of free squares is incremented. The final count of free squares is then printed.
