```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    
    vector<long long> rows(n+1), cols(n+1), diags(2*n+1), adiags(2*n+1);
    
    for(long long i = 0; i < m; ++i){
        long long a, b;
        cin >> a >> b;
        rows[a]++;
        cols[b]++;
        diags[a+b]++;
        adiags[a-b+n-1]++;
    }
    
    long long ans = n*n;
    for(long long i = 1; i <= n; ++i){
        ans -= rows[i] + cols[i] + diags[i] + adiags[i];
    }
    cout << ans/2 << endl;
    return 0;
}
```

This solution calculates the number of empty squares that cannot be captured by any existing pieces by counting how many squares each line (horizontal, vertical, diagonal, anti-diagonal) contains and subtracting this total from the total number of squares in the grid. Since each piece can capture multiple squares, we divide the final answer by two to correct for over-counting. This approach ensures the solution runs efficiently even for large inputs.
