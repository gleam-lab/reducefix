```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int MAX_M = 2000;

ll N;
int M;

set<ll> rows, cols;
set<ll> diag1, diag2; // diag1: i-j, diag2: i+j

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> N >> M;

    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total squares under attack = |rows|*N + |cols|*N + |diag1|*(N) + |diag2|*(N) 
    // But we need to subtract overlaps using inclusion-exclusion

    ll ans = N * N;

    // Subtract attacked squares
    ans -= (ll)rows.size() * N;
    ans -= (ll)cols.size() * N;
    ans -= (ll)diag1.size() * (N);
    ans -= (ll)diag2.size() * (N);

    // Add back over-subtracted intersections
    for (ll r : rows) {
        for (ll c : cols) {
            // (r,c) counted in both row and column
            ans += 1;
        }
    }

    // Handle row & diag1
    for (ll r : rows) {
        for (ll d : diag1) {
            ll c = r - d;
            if (1 <= c && c <= N) {
                ans += 1;
            }
        }
    }

    // Handle row & diag2
    for (ll r : rows) {
        for (ll d : diag2) {
            ll c = d - r;
            if (1 <= c && c <= N) {
                ans += 1;
            }
        }
    }

    // Handle col & diag1
    for (ll c : cols) {
        for (ll d : diag1) {
            ll r = d + c;
            if (1 <= r && r <= N) {
                ans += 1;
            }
        }
    }

    // Handle col & diag2
    for (ll c : cols) {
        for (ll d : diag2) {
            ll r = d - c;
            if (1 <= r && r <= N) {
                ans += 1;
            }
        }
    }

    // Handle diag1 & diag2
    for (ll d1 : diag1) {
        for (ll d2 : diag2) {
            ll r = (d2 + d1) / 2;
            ll c = (d2 - d1) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N) {
                ans += 1;
            }
        }
    }

    // Subtract triple overlaps
    for (ll r : rows) {
        for (ll c : cols) {
            ll d1 = r - c;
            ll d2 = r + c;
            if (diag1.count(d1)) ans -= 1;
            if (diag2.count(d2)) ans -= 1;
        }
    }

    // Add back quadruple overlaps
    for (ll r : rows) {
        for (ll c : cols) {
            ll d1 = r - c;
            ll d2 = r + c;
            if (diag1.count(d1) && diag2.count(d2)) {
                ans += 1;
            }
        }
    }

    // Subtract already occupied cells
    ans -= M;

    cout << ans << endl;

    return 0;
}
```
