#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e9 + 10;
const int INF = 1e9 + 10;

typedef long long ll;
typedef pair<int, int> pii;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    set<pii> st;
    for(int i = 0; i < m; ++i){
        int x, y;
        cin >> x >> y;
        st.insert({x, y});
    }

    ll res = 0;
    for(int i = 1; i <= n; ++i){
        for(int j = 1; j <= n; ++j){
            if(st.count({i, j})) continue;
            bool flag = true;
            for(int k = 1; k <= n; ++k){
                if(st.count({i, k}) || st.count({k, j}) || st.count({i + j - k, k}) || st.count({i - j + k, k})){
                    flag = false;
                    break;
                }
            }
            if(flag) res++;
        }
    }

    cout << res << '\n';

    return 0;
}

This solution uses a set to store all the positions of the pieces and then checks each empty cell to see if it can be attacked by any piece. If not, it increments the result counter. The time complexity is O(n^3) which should be sufficient given the problem constraints.
