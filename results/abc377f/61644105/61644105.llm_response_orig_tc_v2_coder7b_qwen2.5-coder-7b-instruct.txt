```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int N = 1e9+7;

ll n, m, cnt[2][2*N];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    for(int i=1;i<=m;i++){
        ll x,y,d; cin >> x >> y;
        d = (x+y)%2;
        cnt[d][x-y+N]++;
        cnt[d][x+y]++;
    }

    ll ans = n*n;
    for(int i=1;i<=2*N;i++){
        ans -= (cnt[0][i]*cnt[1][i]);
    }
    
    cout << ans << "\n";

    return 0;
}
```

This solution counts the number of squares that cannot be captured by checking how many pairs of diagonals they intersect. It uses prefix sums to count these intersections efficiently. The answer is then calculated as the total number of squares minus the intersections. This approach ensures a time complexity of O(M log N), which should handle the upper constraint of M=10^3 and N=10^9 effectively.
