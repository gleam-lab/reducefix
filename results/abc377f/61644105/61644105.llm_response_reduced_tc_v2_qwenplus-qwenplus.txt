```cpp
#include <bits/stdc++.h>

using namespace std;

using i64 = int64_t;

int main() {
    i64 N, M;
    cin >> N >> M;

    set<pair<i64, i64>> pieces;
    set<i64> rows, cols;
    set<i64> diag1, diag2; // diag1: x+y, diag2: x-y

    for (i64 i = 0; i < M; ++i) {
        i64 x, y;
        cin >> x >> y;
        pieces.emplace(x, y);
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x + y);
        diag2.insert(x - y);
    }

    // Total squares not attacked by any piece
    i64 total = N * N;

    // Subtract all squares attacked in the same row
    total -= (i64)rows.size() * N;

    // Subtract all squares attacked in the same column
    total -= (i64)cols.size() * N;

    // Subtract all squares on each diagonal of type x+y=a
    for (auto d : diag1) {
        i64 cnt = min(d - 1, N - 1) - max(i64(0), d - N) + 1;
        total -= cnt;
    }

    // Subtract all squares on each diagonal of type x-y=b
    for (auto d : diag2) {
        i64 cnt;
        if (d < 0) {
            cnt = min(N + d, N) - max(i64(0), d) + 1;
        } else {
            cnt = min(N - d, N) - max(i64(0), -d) + 1;
        }
        total -= cnt;
    }

    // Add back overlaps (Inclusion-Exclusion Principle)

    // Rows and columns overlap
    total += (i64)rows.size() * cols.size();

    // Rows and diag1 (x+y=a)
    for (auto x : rows) {
        for (auto a : diag1) {
            i64 y = a - x;
            if (y >= 1 && y <= N) {
                total += 1;
            }
        }
    }

    // Rows and diag2 (x-y=b)
    for (auto x : rows) {
        for (auto b : diag2) {
            i64 y = x - b;
            if (y >= 1 && y <= N) {
                total += 1;
            }
        }
    }

    // Columns and diag1 (x+y=a)
    for (auto y : cols) {
        for (auto a : diag1) {
            i64 x = a - y;
            if (x >= 1 && x <= N) {
                total += 1;
            }
        }
    }

    // Columns and diag2 (x-y=b)
    for (auto y : cols) {
        for (auto b : diag2) {
            i64 x = b + y;
            if (x >= 1 && x <= N) {
                total += 1;
            }
        }
    }

    // Diag1 and diag2 overlaps
    for (auto a : diag1) {
        for (auto b : diag2) {
            i64 x = (a + b) / 2;
            i64 y = (a - b) / 2;
            if ((a + b) % 2 == 0 && (a - b) % 2 == 0 && x >= 1 && x <= N && y >= 1 && y <= N) {
                total += 1;
            }
        }
    }

    // Subtract overlaps of three types
    for (auto x : rows) {
        for (auto y : cols) {
            if (diag1.count(x + y) || diag2.count(x - y)) {
                total -= 1;
            }
        }
    }

    // Add back overlaps of all four types
    for (const auto& [x, y] : pieces) {
        if (rows.count(x) && cols.count(y) && diag1.count(x + y) && diag2.count(x - y)) {
            total += 1;
        }
    }

    // Subtract already occupied squares
    total -= M;

    cout << total << endl;

    return 0;
}
```
