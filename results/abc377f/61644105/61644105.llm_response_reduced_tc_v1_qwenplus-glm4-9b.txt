#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

const int MOD = 1000000007;

int N, M;
vector<pair<int, int>> pieces;
vector<pair<int, int>> rows, cols, ups, downs;

ll count_empty_spaces() {
    // Start with all squares
    ll total = N * N;
    
    // Subtract squares that are in the same rows or columns as any piece
    for (auto& p : pieces) {
        total -= 2; // one for row, one for column
    }
    
    // Subtract squares that are in the same diagonals as any piece
    for (auto& p : pieces) {
        int x = p.first, y = p.second;
        total -= (ups[1 - (x + y) & 1].end() - lower_bound(ups[1 - (x + y) & 1].begin(), ups[1 - (x + y) & 1].end(), x - y)) - 1;
        total -= (downs[1 - (x - y) & 1].end() - lower_bound(downs[1 - (x - y) & 1].begin(), downs[1 - (x - y) & 1].end(), x + y)) - 1;
    }
    
    return total;
}

int main() {
    cin >> N >> M;
    pieces.resize(M);
    rows.resize(N);
    cols.resize(N);
    ups.resize(2, {0, N + 1});
    downs.resize(2, {0, N + 1});
    
    for (int i = 0; i < M; ++i) {
        int x, y;
        cin >> x >> y;
        --x, --y; // Convert to 0-based index
        pieces[i] = {x, y};
        
        rows[x].push_back(y);
        cols[y].push_back(x);
        
        ups[1 - (x + y) & 1].push_back(x - y);
        downs[1 - (x - y) & 1].push_back(x + y);
    }
    
    // Sort the columns and rows to facilitate range queries
    for (auto& row : rows) sort(row.begin(), row.end());
    for (auto& col : cols) sort(col.begin(), col.end());
    sort(ups[0].begin(), ups[0].end());
    sort(ups[1].begin(), ups[1].end());
    sort(downs[0].begin(), downs[0].end());
    sort(downs[1].begin(), downs[1].end());
    
    // Calculate the number of empty spaces
    cout << count_empty_spaces() << endl;
    
    return 0;
}
