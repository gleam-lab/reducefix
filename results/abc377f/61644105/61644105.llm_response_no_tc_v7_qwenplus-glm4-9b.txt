
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long n, m;
vector<pair<int, int>> pieces;
vector<int> row_piece, col_piece, diag1_piece, diag2_piece, anti_diag1_piece, anti_diag2_piece;

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        pieces.emplace_back(a, b);
        row_piece.push_back(a);
        col_piece.push_back(b);
        diag1_piece.push_back(a + b);
        diag2_piece.push_back(a - b + n);
        anti_diag1_piece.push_back(a + b - n);
        anti_diag2_piece.push_back(a - b);
    }

    // Sort and remove duplicates
    sort(row_piece.begin(), row_piece.end());
    sort(col_piece.begin(), col_piece.end());
    sort(diag1_piece.begin(), diag1_piece.end());
    sort(diag2_piece.begin(), diag2_piece.end());
    sort(anti_diag1_piece.begin(), anti_diag1_piece.end());
    sort(anti_diag2_piece.begin(), anti_diag2_piece.end());

    row_piece.erase(unique(row_piece.begin(), row_piece.end()), row_piece.end());
    col_piece.erase(unique(col_piece.begin(), col_piece.end()), col_piece.end());
    diag1_piece.erase(unique(diag1_piece.begin(), diag1_piece.end()), diag1_piece.end());
    diag2_piece.erase(unique(diag2_piece.begin(), diag2_piece.end()), diag2_piece.end());
    anti_diag1_piece.erase(unique(anti_diag1_piece.begin(), anti_diag1_piece.end()), anti_diag1_piece.end());
    anti_diag2_piece.erase(unique(anti_diag2_piece.begin(), anti_diag2_piece.end()), anti_diag2_piece.end());

    // Calculate the number of invalid positions
    long long invalid_positions = 0;

    // Invalid by row or column
    for (int r : row_piece) invalid_positions++;
    for (int c : col_piece) invalid_positions++;
    // Invalid by diagonals
    for (int d : diag1_piece) invalid_positions++;
    for (int d : diag2_piece) invalid_positions++;
    for (int d : anti_diag1_piece) invalid_positions++;
    for (int d : anti_diag2_piece) invalid_positions++;

    // Subtract overcounted positions
    // Overcounted by 2 (e.g., both row and diagonal)
    for (int r : row_piece) {
        if (find(diag1_piece.begin(), diag1_piece.end(), r) != diag1_piece.end()) invalid_positions--;
        if (find(diag2_piece.begin(), diag2_piece.end(), r) != diag2_piece.end()) invalid_positions--;
        if (find(anti_diag1_piece.begin(), anti_diag1_piece.end(), r) != anti_diag1_piece.end()) invalid_positions--;
        if (find(anti_diag2_piece.begin(), anti_diag2_piece.end(), r) != anti_diag2_piece.end()) invalid_positions--;
    }
    for (int c : col_piece) {
        if (find(diag1_piece.begin(), diag1_piece.end(), c + n) != diag1_piece.end()) invalid_positions--;
        if (find(diag2_piece.begin(), diag2_piece.end(), c + n) != diag2_piece.end()) invalid_positions--;
        if (find(anti_diag1_piece.begin(), anti_diag1_piece.end(), c + n) != anti_diag1_piece.end()) invalid_positions--;
        if (find(anti_diag2_piece.begin(), anti_diag2_piece.end(), c + n) != anti_diag2_piece.end()) invalid_positions--;
    }

    // Total number of valid positions
    long long total_positions = n * n;
    long long valid_positions = total_positions - invalid_positions;

    cout << valid_positions << endl;

    return 0;
}
```
