```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

ll n, m;

vector<ll> row, col, up[2], down[2];

bool exist(vector<ll>& vec, ll x) {
    return binary_search(vec.begin(), vec.end(), x);
}

ll count(vector<ll>& vec, ll l, ll r) {
    return upper_bound(vec.begin(), vec.end(), r) - lower_bound(vec.begin(), vec.end(), l);
}

void solve() {
    cin >> n >> m;
    for (ll i = 0; i < m; ++i) {
        ll x, y; cin >> x >> y;
        row.push_back(y);
        col.push_back(x);
        up[(x+y)&1].push_back(y-x);
        down[(x+y)&1].push_back(x+y);
    }
    sort(row.begin(), row.end());
    row.erase(unique(row.begin(), row.end()), row.end());
    sort(col.begin(), col.end());
    col.erase(unique(col.begin(), col.end()), col.end());
    sort(up[0].begin(), up[0].end());
    up[0].erase(unique(up[0].begin(), up[0].end()), up[0].end());
    sort(up[1].begin(), up[1].end());
    up[1].erase(unique(up[1].begin(), up[1].end()), up[1].end());
    sort(down[0].begin(), down[0].end());
    down[0].erase(unique(down[0].begin(), down[0].end()), down[0].end());
    sort(down[1].begin(), down[1].end());
    down[1].erase(unique(down[1].begin(), down[1].end()), down[1].end());

    ll ans = n*n - m;
    for (auto& r : row) ans -= m - count(up[0], 1-r, m-1-r) - count(up[1], 1+r, m+1+r);
    for (auto& c : col) ans -= m - count(down[0], 1-c, m-1+c) - count(down[1], 1+c, m+1-c);

    cout << ans << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
```
