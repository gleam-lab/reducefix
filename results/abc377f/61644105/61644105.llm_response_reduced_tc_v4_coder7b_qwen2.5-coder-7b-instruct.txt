```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    vector<pair<int, int>> positions(m);
    for(int i = 0; i < m; ++i) {
        cin >> positions[i].first >> positions[i].second;
    }
    
    set<int> rows, cols, diag1, diag2;
    for(auto& pos : positions) {
        rows.insert(pos.first);
        cols.insert(pos.second);
        diag1.insert(pos.first + pos.second);
        diag2.insert(pos.first - pos.second);
    }
    
    int total_empty_squares = n * n;
    total_empty_squares -= rows.size();
    total_empty_squares -= cols.size();
    total_empty_squares -= diag1.size();
    total_empty_squares -= diag2.size();
    
    for(auto& pos : positions) {
        if(rows.count(pos.first)) --total_empty_squares;
        if(cols.count(pos.second)) --total_empty_squares;
        if(diag1.count(pos.first + pos.second)) --total_empty_squares;
        if(diag2.count(pos.first - pos.second)) --total_empty_squares;
    }
    
    cout << total_empty_squares << endl;
    
    return 0;
}
```
