#include <bits/stdc++.h>

using namespace std;

using i64 = int64_t;
using Coord = pair<i64, i64>;

i64 N, M;

set<i64> rows, cols;
set<i64> diag1, diag2;

int main() {
    scanf("%lld %lld", &N, &M);
    
    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        scanf("%lld %lld", &a, &b);
        rows.insert(a);
        cols.insert(b);
        diag1.insert(b - a);   // diagonal: y - x
        diag2.insert(a + b);   // anti-diagonal: x + y
    }
    
    i64 total_safe = N * N;
    
    // Remove all squares in the same row or column as any piece
    total_safe -= (i64)(N * (rows.size() + cols.size()));
    
    // Remove all squares on the same diagonal or anti-diagonal
    total_safe -= (i64)(diag1.size() * (2 * N - 1));
    total_safe -= (i64)(diag2.size() * (2 * N - 1));
    
    // Add back over-subtracted intersections:
    // Squares that are in both a row and a column of a piece
    total_safe += (i64)(rows.size() * cols.size());
    
    // Squares that are in a row and on a diagonal
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = r + d;
            if (1 <= c && c <= N)
                ++total_safe;
        }
    }
    
    // Squares that are in a row and on an anti-diagonal
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = d - r;
            if (1 <= c && c <= N)
                ++total_safe;
        }
    }
    
    // Squares that are in a column and on a diagonal
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = c - d;
            if (1 <= r && r <= N)
                ++total_safe;
        }
    }
    
    // Squares that are in a column and on an anti-diagonal
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d - c;
            if (1 <= r && r <= N)
                ++total_safe;
        }
    }
    
    // Squares that are on a diagonal and anti-diagonal
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            i64 r = (d2 - d1) / 2;
            i64 c = (d2 + d1) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N && (d2 - d1) % 2 == 0 && (d2 + d1) % 2 == 0)
                ++total_safe;
        }
    }

    // Subtract back those counted three times:
    // In row, col, and diagonal
    for (auto r : rows) {
        auto c = cols.begin();
        for (auto d : diag1) {
            i64 cc = r + d;
            if (cc == *c) ++c;
        }
    }
    
    // In row, col, and anti-diagonal
    for (auto r : rows) {
        for (auto c : cols) {
            if (diag2.count(r + c))
                --total_safe;
        }
    }

    // Finally, subtract occupied cells
    total_safe -= M;

    printf("%lld\n", total_safe);

    return 0;
}
