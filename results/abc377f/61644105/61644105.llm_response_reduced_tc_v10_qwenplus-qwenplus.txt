#include <bits/stdc++.h>

using namespace std;

using i64 = int64_t;
using pii = pair<int, int>;

#define MAX_M 1000

i64 N;
int M;
set<int> rows, cols;
set<i64> diag1, diag2;

i64 compute_total_cells() {
    return N * N;
}

i64 compute_conflict_count() {
    // Each row contributes (N - count_in_row) cells -> total M*N
    // Similarly for columns
    i64 ans = 1LL * M * N;
    ans += 1LL * M * N;

    // Diag1: y - x in diag1
    ans += 1LL * diag1.size() * N;
    ans += 1LL * diag2.size() * N;

    return ans;
}

i64 compute_2_intersection_count() {
    i64 ans = 0;

    // Row-Column intersections
    ans += 1LL * rows.size() * cols.size();

    // Row-diag1 intersections
    for (int r : rows) {
        for (int c = 1; c <= N; ++c) {
            if (diag1.count((c - r))) {
                ans++;
            }
        }
    }

    // Row-diag2 intersections
    for (int r : rows) {
        for (int c = 1; c <= N; ++c) {
            if (diag2.count(r + c)) {
                ans++;
            }
        }
    }

    // Column-diag1 intersections
    for (int c : cols) {
        for (int r = 1; r <= N; ++r) {
            if (diag1.count(c - r)) {
                ans++;
            }
        }
    }

    // Column-diag2 intersections
    for (int c : cols) {
        for (int r = 1; r <= N; ++r) {
            if (diag2.count(r + c)) {
                ans++;
            }
        }
    }

    // diag1-diag2 intersections
    for (i64 d1 : diag1) {
        for (i64 d2 : diag2) {
            int x = (d2 - d1) / 2;
            int y = (d2 + d1) / 2;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                ans++;
            }
        }
    }

    return ans;
}

i64 compute_3_intersection_count() {
    i64 ans = 0;

    // Row-col-diag1
    for (int r : rows) {
        for (int c : cols) {
            if (diag1.count(c - r)) {
                ans++;
            }
        }
    }

    // Row-col-diag2
    for (int r : rows) {
        for (int c : cols) {
            if (diag2.count(r + c)) {
                ans++;
            }
        }
    }

    // Row-diag1-diag2
    for (int r : rows) {
        for (i64 d1 : diag1) {
            i64 d2 = r + ((d1 + r) % N + N) % N;
            if (d2 < 1 || d2 > 2 * N) continue;
            if (diag2.count(d2)) {
                ans++;
            }
        }
    }

    // Col-diag1-diag2
    for (int c : cols) {
        for (i64 d1 : diag1) {
            int y = d1 + c;
            int x = c;
            i64 d2 = x + y;
            if (y < 1 || y > N) continue;
            if (diag2.count(d2)) {
                ans++;
            }
        }
    }

    return ans;
}

i64 compute_4_intersection_count() {
    i64 ans = 0;

    // Row-col-diag1-diag2
    for (int r : rows) {
        for (int c : cols) {
            if (diag1.count(c - r) && diag2.count(r + c)) {
                ans++;
            }
        }
    }

    return ans;
}

int main() {
    scanf("%lld %d", &N, &M);
    for (int i = 0; i < M; ++i) {
        int a, b;
        scanf("%d %d", &a, &b);
        rows.insert(a);
        cols.insert(b);
        diag1.insert(1LL * b - a);
        diag2.insert(1LL * a + b);
    }

    i64 total = compute_total_cells();
    i64 conflict = compute_conflict_count();
    i64 two_inter = compute_2_intersection_count();
    i64 three_inter = compute_3_intersection_count();
    i64 four_inter = compute_4_intersection_count();

    i64 ans = total - conflict + two_inter - three_inter + four_inter;

    printf("%lld\n", ans);

    return 0;
}
