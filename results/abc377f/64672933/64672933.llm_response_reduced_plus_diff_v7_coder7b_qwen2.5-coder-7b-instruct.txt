        cin >> a >> b;
        g[i] = {a, b};
        occupied.insert({a, b});
    }
    
    vector<i64> sums(N + N, 0);
    vector<i64> diffs(N + N, 0);
    vector<i64> cols(N + 1, 0);
    vector<i64> rows(N + 1, 0);

        sums[a + b]++;
        diffs[a - b]++;
        rows[a]++;
        cols[b]++;
    }
    
    i64 ans = 0;
    
    for (i64 r = 1; r <= N; ++r) {
        for (i64 c = 1; c <= N; ++c) {
            if (!occupied.count({r, c}) &&

                ans++;
            }
        }
    }
    
    cout << ans << "\n";
    return 0;
}
