```cpp
#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);
    i64 N, M;
    cin >> N >> M;
    vector<pair<i64, i64>> G(M);
    set<i64> S, T;
    for (int i = 0; i < M; i++) {
        i64 a, b;
        cin >> a >> b;
        G[i] = {a, b};
        S.insert(a + b);
        T.insert(a - b);
    }
    i64 ans = N * N - M;
    for (auto [a, b] : G) {
        if (S.find(a + b) != S.end()) ans--;
        if (T.find(a - b) != T.end()) ans--;
        for (auto c : G) {
            if (a + b == c.first + c.second || a - b == c.first - c.second) {
                if (a == c.first && b == c.second) continue;
                if (abs(a - c.first) + abs(b - c.second) == 1) ans++;
            }
        }
    }
    cout << ans << '\n';
    return 0;
}
```
