#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<i64> rows(N + 1), cols(N + 1), diag1(2 * N + 1), diag2(2 * N + 1), antidiag1(2 * N + 1), antidiag2(2 * N + 1);
    
    for (i64 i = 1; i <= N; ++i) {
        rows[i] = cols[i] = diag1[i + N] = diag2[i] = antidiag1[i + N] = antidiag2[i] = 1;
    }

    for (i64 i = 1; i <= M; ++i) {
        i64 a, b;
        cin >> a >> b;
        rows[a] = cols[b] = diag1[a + b] = diag2[a + b - N] = antidiag1[a - b + N] = antidiag2[a - b] = 0;
    }

    i64 ans = N * N;
    for (i64 i = 1; i <= N; ++i) {
        ans -= rows[i];
        ans -= cols[i];
        ans -= diag1[i + N];
        ans -= diag2[i];
        ans -= antidiag1[i + N];
        ans -= antidiag2[i];
    }

    cout << ans << "\n";
    return 0;
}
