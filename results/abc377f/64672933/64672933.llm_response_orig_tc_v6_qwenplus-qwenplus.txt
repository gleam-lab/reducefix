#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;

    set<i64> row_set, col_set, diag1_set, diag2_set;
    set<pair<i64, i64>> piece_set;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        row_set.insert(a);
        col_set.insert(b);
        diag1_set.insert(a + b);
        diag2_set.insert(a - b);
        piece_set.emplace(a, b);
    }

    i64 total_captured = 0;

    // Count cells captured by rows and columns
    total_captured += row_set.size() * N;
    total_captured += col_set.size() * N;

    // Count cells captured by diagonals
    total_captured += diag1_set.size() * N;
    total_captured += diag2_set.size() * N;

    // Subtract overcounts: intersections of rows and columns
    for (auto r : row_set) {
        for (auto c : col_set) {
            if (!piece_set.count({r, c})) {
                total_captured -= 1;
            }
        }
    }

    // Subtract overcounts: intersections of rows and diag1
    for (auto r : row_set) {
        for (auto d1 : diag1_set) {
            i64 c = d1 - r;
            if (1 <= c && c <= N) {
                if (!piece_set.count({r, c})) {
                    total_captured -= 1;
                }
            }
        }
    }

    // Subtract overcounts: intersections of rows and diag2
    for (auto r : row_set) {
        for (auto d2 : diag2_set) {
            i64 c = r - d2;
            if (1 <= c && c <= N) {
                if (!piece_set.count({r, c})) {
                    total_captured -= 1;
                }
            }
        }
    }

    // Subtract overcounts: intersections of columns and diag1
    for (auto c : col_set) {
        for (auto d1 : diag1_set) {
            i64 r = d1 - c;
            if (1 <= r && r <= N) {
                if (!piece_set.count({r, c})) {
                    total_captured -= 1;
                }
            }
        }
    }

    // Subtract overcounts: intersections of columns and diag2
    for (auto c : col_set) {
        for (auto d2 : diag2_set) {
            i64 r = d2 + c;
            if (1 <= r && r <= N) {
                if (!piece_set.count({r, c})) {
                    total_captured -= 1;
                }
            }
        }
    }

    // Subtract overcounts: intersections of diag1 and diag2
    for (auto d1 : diag1_set) {
        for (auto d2 : diag2_set) {
            i64 r = (d1 + d2) / 2;
            i64 c = d1 - r;
            if ((d1 + d2) % 2 == 0 && 1 <= r && r <= N && 1 <= c && c <= N) {
                if (!piece_set.count({r, c})) {
                    total_captured -= 1;
                }
            }
        }
    }

    // Add back over-subtracted counts from triple overlaps
    for (auto r : row_set) {
        for (auto c : col_set) {
            for (auto d1 : diag1_set) {
                i64 expected_c = d1 - r;
                if (expected_c == c && 1 <= c && c <= N) {
                    if (!piece_set.count({r, c})) {
                        total_captured += 1;
                    }
                }
            }
            for (auto d2 : diag2_set) {
                i64 expected_c = r - d2;
                if (expected_c == c && 1 <= c && c <= N) {
                    if (!piece_set.count({r, c})) {
                        total_captured += 1;
                    }
                }
            }
        }
    }

    // Total squares minus captured squares minus occupied squares
    i64 ans = N * N - total_captured - M;
    cout << ans << "\n";

    return 0;
}
