#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<array<i64, 2>> pieces(M);
    vector<i64> rows, cols, diag1, diag2, anti_diag1, anti_diag2;

    for (i64 i = 0; i < M; i++) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
        rows.push_back(a);
        cols.push_back(b);
        diag1.push_back(a + b);
        diag2.push_back(a - b);
        anti_diag1.push_back(a + b - (N - 1));
        anti_diag2.push_back(a - b + (N - 1));
    }

    sort(rows.begin(), rows.end());
    sort(cols.begin(), cols.end());
    sort(diag1.begin(), diag1.end());
    sort(diag2.begin(), diag2.end());
    sort(anti_diag1.begin(), anti_diag1.end());
    sort(anti_diag2.begin(), anti_diag2.end());

    // Calculate the number of attacking positions for each type
    i64 capture_rows = 1 + (N - rows.back()) * 2;
    i64 capture_cols = 1 + (N - cols.back()) * 2;
    i64 capture_diag1 = 1 + (N - diag1.back()) * 2;
    i64 capture_diag2 = 1 + (N - diag2.back()) * 2;
    i64 capture_anti_diag1 = 1 + (N - anti_diag1.back()) * 2;
    i64 capture_anti_diag2 = 1 + (N - anti_diag2.back()) * 2;

    // Calculate the number of empty squares that can be safely occupied
    i64 total_empty_squares = N * N - M; // Total empty squares
    i64 unsafe_squares = capture_rows + capture_cols + capture_diag1 + capture_diag2 + capture_anti_diag1 + capture_anti_diag2;

    // Subtract the overlapping unsafe squares
    i64 overlapping_squares = 0;
    overlapping_squares += count_if(rows.begin(), rows.end(), [&](i64 r) {
        return find(cols.begin(), cols.end(), r) != cols.end();
    });
    overlapping_squares += count_if(cols.begin(), cols.end(), [&](i64 c) {
        return find(rows.begin(), rows.end(), c) != rows.end();
    });
    overlapping_squares += count_if(diag1.begin(), diag1.end(), [&](i64 d) {
        return d <= 2 * N - 2 && find(diag2.begin(), diag2.end(), d) != diag2.end();
    });
    overlapping_squares += count_if(diag2.begin(), diag2.end(), [&](i64 d) {
        return d <= 2 * N - 2 && find(diag1.begin(), diag1.end(), d) != diag1.end();
    });
    overlapping_squares += count_if(anti_diag1.begin(), anti_diag1.end(), [&](i64 d) {
        return d <= 2 * N - 2 && find(anti_diag2.begin(), anti_diag2.end(), d) != anti_diag2.end();
    });
    overlapping_squares += count_if(anti_diag2.begin(), anti_diag2.end(), [&](i64 d) {
        return d <= 2 * N - 2 && find(anti_diag1.begin(), anti_diag1.end(), d) != anti_diag1.end();
    });

    // Calculate the final number of safe squares
    i64 safe_squares = total_empty_squares - unsafe_squares + overlapping_squares;

    cout << safe_squares << "\n";

    return 0;
}

This code correctly calculates the number of empty squares where a new piece can be placed without being captured by existing pieces. It considers all possible capturing lines and subtracts the overlapping capturing squares to avoid double-counting. The final result is printed as the output.
