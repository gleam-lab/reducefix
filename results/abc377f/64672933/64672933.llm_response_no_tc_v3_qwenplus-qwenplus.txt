#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<pair<i64, i64>> pieces;
    set<i64> rows, cols, diag1, diag2;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces.insert({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total dangerous cells: those in same row, column, or diagonals
    // We'll calculate total dangerous cells, then subtract from N^2 - M

    i64 total_dangerous = 0;

    // Count all cells in the same rows and columns
    total_dangerous += (i64)rows.size() * N;
    total_dangerous += (i64)cols.size() * N;

    // Count all cells in the two diagonals
    total_dangerous += (i64)diag1.size() * N;
    total_dangerous += (i64)diag2.size() * N;

    // Subtract over-counted cells that are in both row and column
    for (auto r : rows) {
        for (auto c : cols) {
            if (pieces.count({r, c})) continue; // skip cells with pieces
            total_dangerous -= 1;
        }
    }

    // Subtract over-counted cells that are on diagonal1 and row/column
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = d - r;
            if (c < 1 || c > N) continue;
            if (pieces.count({r, c})) continue;
            total_dangerous -= 1;
        }
    }

    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = d - c;
            if (r < 1 || r > N) continue;
            if (pieces.count({r, c})) continue;
            total_dangerous -= 1;
        }
    }

    // Subtract over-counted cells that are on diagonal2 and row/column
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = r - d;
            if (c < 1 || c > N) continue;
            if (pieces.count({r, c})) continue;
            total_dangerous -= 1;
        }
    }

    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d + c;
            if (r < 1 || r > N) continue;
            if (pieces.count({r, c})) continue;
            total_dangerous -= 1;
        }
    }

    // Subtract over-counted cells that are on both diagonals
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            i64 r = (d1 + d2) / 2;
            i64 c = (d1 - d2) / 2;
            if ((d1 + d2) % 2 != 0 || (d1 - d2) % 2 != 0) continue;
            if (r < 1 || r > N || c < 1 || c > N) continue;
            if (pieces.count({r, c})) continue;
            total_dangerous -= 1;
        }
    }

    // Now subtract cells with existing pieces
    total_dangerous -= M;

    i64 total_empty = N * N - M;
    i64 safe_cells = total_empty - total_dangerous;

    cout << safe_cells << "\n";

    return 0;
}
