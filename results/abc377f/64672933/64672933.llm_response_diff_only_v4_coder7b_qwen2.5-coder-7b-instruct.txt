#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    ll N, M; cin >> N >> M;
    vector<pair<ll,ll>> pieces(M);
    for(ll i=0;i<M;++i){
        ll a, b; cin >> a >> b;
        pieces[i] = {a-1, b-1}; // Adjusting to zero-based index
    }

    set<pair<ll,ll>> occupied;
    for(auto [x, y]: pieces){
        occupied.insert({x, y});
    }

    ll total_squares = N*N;
    ll empty_squares = total_squares - M;

    // Check each row
    for(ll i=0;i<N;++i){
        bool found = false;
        for(ll j=0;j<N;++j){
            if(!occupied.count({i, j})){
                if(!found){
                    ++empty_squares;
                    found = true;
                }
            }else{
                found = false;
            }
        }
    }

    // Check each column
    for(ll j=0;j<N;++j){
        bool found = false;
        for(ll i=0;i<N;++i){
            if(!occupied.count({i, j})){
                if(!found){
                    ++empty_squares;
                    found = true;
                }
            }else{
                found = false;
            }
        }
    }

    // Check diagonals
    for(ll d=-N+1;d<=N-1;++d){
        bool found = false;
        for(ll i=0;i<N;++i){
            ll j = i - d;
            if(j>=0 && j<N){
                if(!occupied.count({i, j})){
                    if(!found){
                        ++empty_squares;
                        found = true;
                    }
                }else{
                    found = false;
                }
            }
        }
    }

    for(ll d=1;d<=2*N-2;++d){
        bool found = false;
        for(ll i=N-1;i>=0;--i){
            ll j = i + d - N + 1;
            if(j>=0 && j<N){
                if(!occupied.count({i, j})){
                    if(!found){
                        ++empty_squares;
                        found = true;
                    }
                }else{
                    found = false;
                }
            }
        }
    }

    cout << empty_squares << endl;

    return 0;
}
