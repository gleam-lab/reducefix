```cpp
#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);
    i64 n, m; cin >> n >> m;
    vector<array<i64,2>> g(m);
    vector<i64> a(n), b(n);
    for(int i=0;i<m;++i){
        cin >> g[i][0] >> g[i][1];
        --g[i][0]; --g[i][1];
        a[g[i][0]]++, b[g[i][1]]++;
    }
    i64 cnt = 0;
    for(i64 i=0;i<n;++i) cnt += a[i]*b[i];
    cout << n*n-cnt << '\n';
    return 0;
}
```
