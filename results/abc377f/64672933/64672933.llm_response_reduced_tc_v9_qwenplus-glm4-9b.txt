#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <map>

using namespace std;
using i64 = long long;

const int MOD = 1000000007;

int main() {
    i64 N, M;
    cin >> N >> M;
    vector<pair<i64, i64>> points(M);
    for (i64 i = 0; i < M; ++i) {
        cin >> points[i].first >> points[i].second;
    }

    // Sort points by rows and columns
    sort(points.begin(), points.end());

    // Calculate the number of points in the same row and column
    vector<i64> row_counts(N + 1, 0), col_counts(N + 1, 0);
    for (const auto& p : points) {
        row_counts[p.first] += 1;
        col_counts[p.second] += 1;
    }

    // Calculate the number of points on the same diagonals
    vector<i64> diag1_counts(2 * N + 1, 0), diag2_counts(2 * N + 1, 0);
    i64 k1 = 0, k2 = 0;
    for (i64 i = 1; i <= N; ++i) {
        diag1_counts[k1 + points[k1].first] += 1;
        diag1_counts[k1 + points[k1].first + N - points[k1].second] += 1;
        diag2_counts[k2 + points[k2].first - points[k2].second] += 1;
        diag2_counts[k2 + points[k2].first + points[k2].second - N] += 1;
        k1 += row_counts[i];
        k2 += col_counts[i];
    }

    // Calculate the number of valid positions
    i64 valid_positions = 0;
    for (i64 i = 1; i <= N; ++i) {
        // Horizontal position i
        valid_positions += N - row_counts[i];
        // Vertical position i
        valid_positions += N - col_counts[i];
        // Diagonal i + N (main diagonal)
        valid_positions += N - diag1_counts[i + N];
        // Diagonal i - 1 (secondary diagonal)
        valid_positions += N - diag2_counts[i - 1];
    }

    // Output the result
    cout << valid_positions << '\n';

    return 0;
}
