#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<array<i64, 2>> pieces(M);
    for (i64 i = 0; i < M; i++) {
        i64 a, b;
        cin >> a >> b;
        pieces[i] = {a, b};
    }

    // Maps to count occurrences of each line (sum and difference) and column
    unordered_map<i64, i64> line_count, col_count, diag1_count, diag2_count;

    // Populate the maps with piece positions
    for (i64 i = 0; i < M; i++) {
        i64 sum = pieces[i][0] + pieces[i][1];
        i64 diff = pieces[i][0] - pieces[i][1];
        line_count[sum]++;
        col_count[diff]++;
        diag1_count[pieces[i][0] + pieces[i][1] - 1]++;
        diag2_count[pieces[i][0] - pieces[i][1] + 1]++;
    }

    // Calculate the number of valid positions for placing a new piece
    i64 valid_positions = N * N; // Start with total number of squares
    for (auto& p : pieces) {
        i64 a = p[0], b = p[1];

        // Subtract positions in the same row and column
        valid_positions -= (line_count[a] - 1) + (col_count[b] - 1);

        // Subtract positions in the same diagonal (sum)
        valid_positions -= (diag1_count[a + b - 1] - 1);

        // Subtract positions in the same anti-diagonal (difference)
        // Ensure that the difference is non-negative since i-j can be negative
        valid_positions -= (diag2_count[a - b + 1] - 1);
    }

    cout << valid_positions << "\n";

    return 0;
}
