```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    i64 N, M;
    cin >> N >> M;
    
    set<pair<i64, i64>> pieces;
    set<i64> rows, cols, diag1, diag2;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces.emplace(a, b);

        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total threatened squares
    i64 threatened = 0;

    // Rows and columns contribute N squares each
    threatened += rows.size() * N;
    threatened += cols.size() * N;

    // Diagonal sums (a+b)
    for (auto d : diag1) {
        i64 lo = max(1LL, d - N);
        i64 hi = min(N, d - 1);
        threatened += (hi - lo + 1);
    }

    // Diagonal diffs (a-b)
    for (auto d : diag2) {
        i64 lo = max(1LL, d + 1);
        i64 hi = min(N, N + d);
        threatened += (hi - lo + 1);
    }

    // Overlaps between row/col/diag1/diag2

    // Count overlaps between rows and diag1
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = d - r;
            if (c >= 1 && c <= N) {
                threatened--;
            }
        }
    }

    // Count overlaps between rows and diag2
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = r - d;
            if (c >= 1 && c <= N) {
                threatened--;
            }
        }
    }

    // Count overlaps between cols and diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = d - c;
            if (r >= 1 && r <= N) {
                threatened--;
            }
        }
    }

    // Count overlaps between cols and diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d + c;
            if (r >= 1 && r <= N) {
                threatened--;
            }
        }
    }

    // Count overlaps between diag1 and diag2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 != 0) continue;
            i64 r = (d1 + d2) / 2;
            i64 c = d1 - r;
            if (r >= 1 && c >= 1 && r <= N && c <= N) {
                threatened--;
            }
        }
    }

    // Subtract overlaps with other rows/cols/diags within same category
    // Row-row overlaps
    for (auto it = rows.begin(); it != rows.end(); ++it) {
        auto next = it;
        while (++next != rows.end()) {
            threatened--;
        }
    }

    // Col-col overlaps
    for (auto it = cols.begin(); it != cols.end(); ++it) {
        auto next = it;
        while (++next != cols.end()) {
            threatened--;
        }
    }

    // diag1-diag1 overlaps
    for (auto it = diag1.begin(); it != diag1.end(); ++it) {
        auto next = it;
        while (++next != diag1.end()) {
            threatened--;
        }
    }

    // diag2-diag2 overlaps
    for (auto it = diag2.begin(); it != diag2.end(); ++it) {
        auto next = it;
        while (++next != diag2.end()) {
            threatened--;
        }
    }

    // Now remove the actual occupied cells from count
    for (auto [r, c] : pieces) {
        threatened--;
    }

    i64 total_empty = N * N - M;
    i64 safe_positions = total_empty - threatened;

    cout << safe_positions << "\n";

    return 0;
}
```
