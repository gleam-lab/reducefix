```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    
    set<pair<i64, i64>> occupied;
    set<i64> rows, cols, diag1, diag2;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        occupied.emplace(a, b);
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total squares under attack
    i64 attacked = 0;

    // Rows and columns contribute N squares each
    attacked += rows.size() * N;
    attacked += cols.size() * N;

    // Diagonals of the form x + y = c
    for (auto d : diag1) {
        i64 len;
        if (d >= 2 && d <= N + 1) {
            len = d - 1;
        } else if (d > N + 1 && d <= 2 * N) {
            len = 2 * N + 1 - d;
        } else {
            len = 0;
        }
        attacked += len;
    }

    // Diagonals of the form x - y = c
    for (auto d : diag2) {
        i64 len;
        if (d >= -(N - 1) && d <= -1) {
            len = N + d;
        } else if (d >= 0 && d <= N - 1) {
            len = N - d;
        } else {
            len = 0;
        }
        attacked += len;
    }

    // Subtract overlaps

    // Row-column intersections: (r, c) where r in rows and c in cols
    for (auto r : rows) {
        for (auto c : cols) {
            attacked--;
        }
    }

    // Row-diag1 intersections: solve x = r, x + y = d => y = d - r
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 y = d - r;
            if (y >= 1 && y <= N) {
                attacked--;
            }
        }
    }

    // Row-diag2 intersections: x = r, x - y = d => y = r - d
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 y = r - d;
            if (y >= 1 && y <= N) {
                attacked--;
            }
        }
    }

    // Col-diag1 intersections: y = c, x + y = d => x = d - c
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 x = d - c;
            if (x >= 1 && x <= N) {
                attacked--;
            }
        }
    }

    // Col-diag2 intersections: y = c, x - y = d => x = d + c
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 x = d + c;
            if (x >= 1 && x <= N) {
                attacked--;
            }
        }
    }

    // Diag1-diag2 intersections: solve x + y = d1, x - y = d2 => x = (d1 + d2)/2, y = (d1 - d2)/2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 == 0) {
                i64 x = (d1 + d2) / 2;
                i64 y = (d1 - d2) / 2;
                if (x >= 1 && x <= N && y >= 1 && y <= N) {
                    attacked--;
                }
            }
        }
    }

    // Subtract already occupied squares from attacked count
    i64 total_occupied = 0;
    for (auto p : occupied) {
        bool in_attack = false;
        i64 x = p.first, y = p.second;
        if (rows.count(x) || cols.count(y) || diag1.count(x + y) || diag2.count(x - y)) {
            in_attack = true;
        }
        if (in_attack) total_occupied++;
    }

    i64 total_safe = N * N - attacked - (M - total_occupied);
    cout << total_safe << "\n";

    return 0;
}
```
