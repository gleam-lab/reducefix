#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<pair<i64, i64>> pieces;
    set<i64> rows, cols, diag1, diag2;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces.insert({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total dangerous cells
    i64 total_dangerous = 0;

    // Count danger from rows
    total_dangerous += rows.size() * N;

    // Count danger from columns
    total_dangerous += cols.size() * N;

    // Count danger from diagonal1: x + y = c
    for (auto d : diag1) {
        // Number of cells on diagonal x + y = d
        i64 count = 0;
        if (d >= 2 && d <= N + 1)
            count = min(d - 1, N) - max<i64>(d - N, 1) + 1;
        else if (d > N + 1 && d <= 2 * N)
            count = min(d - 1, N) - max<i64>(d - N, 1) + 1;
        total_dangerous += count;
    }

    // Count danger from diagonal2: x - y = c
    for (auto d : diag2) {
        // Number of cells on diagonal x - y = d
        i64 count = 0;
        i64 x_min = max<i64>(1, 1 + d);
        i64 x_max = min(N, N + d);
        count = max<i64>(0, x_max - x_min + 1);
        total_dangerous += count;
    }

    // Subtract overlaps

    // Overlap between row and column
    for (auto r : rows) {
        for (auto c : cols) {
            if (pieces.count({r, c}) == 0)
                total_dangerous -= 1;
        }
    }

    // Overlap between row and diag1
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = d - r;
            if (c >= 1 && c <= N)
                total_dangerous -= 1;
        }
    }

    // Overlap between row and diag2
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = r - d;
            if (c >= 1 && c <= N)
                total_dangerous -= 1;
        }
    }

    // Overlap between col and diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = d - c;
            if (r >= 1 && r <= N)
                total_dangerous -= 1;
        }
    }

    // Overlap between col and diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d + c;
            if (r >= 1 && r <= N)
                total_dangerous -= 1;
        }
    }

    // Overlap between diag1 and diag2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            i64 r = (d1 + d2);
            i64 c = (d1 - d2);
            if (r % 2 == 0 && c % 2 == 0) {
                r /= 2;
                c /= 2;
                if (r >= 1 && r <= N && c >= 1 && c <= N)
                    total_dangerous -= 1;
            }
        }
    }

    // Add back triple overlaps that were subtracted too many times
    for (auto r : rows) {
        for (auto c : cols) {
            for (auto d1 : diag1) {
                i64 check_c = d1 - r;
                if (check_c == c && c >= 1 && c <= N)
                    total_dangerous += 1;
            }
            for (auto d2 : diag2) {
                i64 check_r = d2 + c;
                if (check_r == r && r >= 1 && r <= N)
                    total_dangerous += 1;
            }
        }
    }

    // Subtract existing pieces since they are not empty
    total_dangerous += M;

    i64 total_empty_safe = N * N - total_dangerous;

    cout << total_empty_safe << "\n";

    return 0;
}
