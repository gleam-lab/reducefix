    vector<pair<i64, i64>> pieces(M);

        i64 a, b;

    i64 total_empty_squares = N * N;

        total_empty_squares -= (diag1.size() - 1); // Remove current diagonal 1
        total_empty_squares -= (diag2.size() - 1); // Remove current diagonal 2

    }
