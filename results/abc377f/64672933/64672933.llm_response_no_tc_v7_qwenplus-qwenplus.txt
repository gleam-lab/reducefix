```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<pair<i64, i64>> pieces;
    set<i64> rows, cols, diag1, diag2;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        pieces.emplace(a, b);
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total squares under attack
    set<pair<i64, i64>> attacked;

    // Rows
    for (auto r : rows) {
        for (i64 c = 1; c <= N; ++c) {
            attacked.emplace(r, c);
        }
    }

    // Columns
    for (auto c : cols) {
        for (i64 r = 1; r <= N; ++r) {
            attacked.emplace(r, c);
        }
    }

    // Diagonals: x + y = k
    for (auto d : diag1) {
        // y = d - x
        for (i64 x = max(1LL, d - N); x <= min(N, d - 1); ++x) {
            i64 y = d - x;
            if (y >= 1 && y <= N)
                attacked.emplace(x, y);
        }
    }

    // Diagonals: x - y = k
    for (auto d : diag2) {
        // y = x - d
        for (i64 x = max(1LL, d + 1); x <= min(N, N + d); ++x) {
            i64 y = x - d;
            if (y >= 1 && y <= N)
                attacked.emplace(x, y);
        }
    }

    // Remove existing pieces from attacked set if they were added multiple times
    for (const auto& p : pieces) {
        attacked.erase(p);
    }

    // Total safe positions = total grid size - attacked squares - already occupied squares
    i64 totalSafe = N * N - attacked.size() - pieces.size();
    cout << totalSafe << "\n";

    return 0;
}
```
