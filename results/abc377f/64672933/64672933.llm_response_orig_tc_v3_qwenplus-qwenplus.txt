```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    i64 N, M;
    cin >> N >> M;

    set<i64> rows, cols;
    set<i64> diag1, diag2; // x+y and x-y

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    i64 total_captured = 0;

    // Rows
    total_captured += rows.size() * N;

    // Columns
    total_captured += cols.size() * N;

    // Diagonals: a + b
    total_captured += diag1.size() * N;

    // Diagonals: a - b
    total_captured += diag2.size() * N;

    // Subtract overlaps: intersections between rows and columns
    for (auto r : rows) {
        for (auto c : cols) {
            total_captured--;
        }
    }

    // Subtract overlaps: intersections between rows and diagonal1
    for (auto r : rows) {
        for (auto d : diag1) {
            i64 c = d - r;
            if (c >= 1 && c <= N)
                total_captured--;
        }
    }

    // Subtract overlaps: intersections between rows and diagonal2
    for (auto r : rows) {
        for (auto d : diag2) {
            i64 c = r - d;
            if (c >= 1 && c <= N)
                total_captured--;
        }
    }

    // Subtract overlaps: intersections between columns and diagonal1
    for (auto c : cols) {
        for (auto d : diag1) {
            i64 r = d - c;
            if (r >= 1 && r <= N)
                total_captured--;
        }
    }

    // Subtract overlaps: intersections between columns and diagonal2
    for (auto c : cols) {
        for (auto d : diag2) {
            i64 r = d + c;
            if (r >= 1 && r <= N)
                total_captured--;
        }
    }

    // Subtract overlaps: intersections between diagonal1 and diagonal2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            i64 x = (d1 + d2) / 2;
            i64 y = (d1 - d2) / 2;
            if ((d1 + d2) % 2 == 0 && (d1 - d2) % 2 == 0 && x >= 1 && y >= 1 && x <= N && y <= N)
                total_captured--;
        }
    }

    // Add back over-subtracted points where all three lines intersect
    for (auto r : rows) {
        for (auto c : cols) {
            i64 d1 = r + c;
            i64 d2 = r - c;
            if (diag1.count(d1) && diag2.count(d2))
                total_captured++;
        }
    }

    cout << N * N - total_captured << "\n";

    return 0;
}
```
