```cpp
#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;

    set<i64> rowSet, colSet, diag1Set, diag2Set;
    set<pair<i64, i64>> pieceSet;

    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        cin >> a >> b;
        rowSet.insert(a);
        colSet.insert(b);
        diag1Set.insert(a + b);
        diag2Set.insert(a - b);
        pieceSet.emplace(a, b);
    }

    // Total number of squares under attack by existing pieces
    i64 attackedSquares = 0;

    // Rows and columns contribute N each
    attackedSquares += (i64)rowSet.size() * N;
    attackedSquares += (i64)colSet.size() * N;

    // Diagonals
    for (i64 d : diag1Set) {
        i64 len = min(d - 1, N) - max(i64(1), d - N) + 1;
        attackedSquares += len;
    }

    for (i64 d : diag2Set) {
        i64 x_min = max(i64(1), i64(1) + d);
        i64 x_max = min(N, N + d);
        if (x_min <= x_max) {
            attackedSquares += x_max - x_min + 1;
        }
    }

    // Now subtract overlaps

    // Intersections between rows and columns
    for (i64 r : rowSet) {
        for (i64 c : colSet) {
            if (pieceSet.count({r, c}) == 0) {
                attackedSquares--;
            }
        }
    }

    // Intersections between rows and diag1
    for (i64 r : rowSet) {
        for (i64 d : diag1Set) {
            i64 c = d - r;
            if (c >= 1 && c <= N) {
                if (pieceSet.count({r, c}) == 0) {
                    attackedSquares--;
                }
            }
        }
    }

    // Intersections between rows and diag2
    for (i64 r : rowSet) {
        for (i64 d : diag2Set) {
            i64 c = r - d;
            if (c >= 1 && c <= N) {
                if (pieceSet.count({r, c}) == 0) {
                    attackedSquares--;
                }
            }
        }
    }

    // Intersections between cols and diag1
    for (i64 c : colSet) {
        for (i64 d : diag1Set) {
            i64 r = d - c;
            if (r >= 1 && r <= N) {
                if (pieceSet.count({r, c}) == 0) {
                    attackedSquares--;
                }
            }
        }
    }

    // Intersections between cols and diag2
    for (i64 c : colSet) {
        for (i64 d : diag2Set) {
            i64 r = d + c;
            if (r >= 1 && r <= N) {
                if (pieceSet.count({r, c}) == 0) {
                    attackedSquares--;
                }
            }
        }
    }

    // Intersections between diag1 and diag2
    for (i64 d1 : diag1Set) {
        for (i64 d2 : diag2Set) {
            i64 r = (d1 + d2);
            i64 c = (d1 - d2);
            if (r % 2 == 0 && c % 2 == 0) {
                r /= 2;
                c /= 2;
                if (r >= 1 && c >= 1 && r <= N && c <= N) {
                    if (pieceSet.count({r, c}) == 0) {
                        attackedSquares--;
                    }
                }
            }
        }
    }

    // Subtract overcounted squares that are already occupied by pieces
    attackedSquares -= M;

    i64 totalEmpty = N * N - M;
    i64 safeSquares = totalEmpty - attackedSquares;

    cout << safeSquares << "\n";

    return 0;
}
```
