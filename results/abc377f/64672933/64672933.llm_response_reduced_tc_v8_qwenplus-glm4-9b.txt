#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    i64 N, M;
    cin >> N >> M;
    vector<pair<i64, i64>> pieces(M);
    for (i64 i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }
    
    // Sort pieces based on the sum and difference of their coordinates
    sort(pieces.begin(), pieces.end(), [](const pair<i64, i64>& a, const pair<i64, i64>& b) {
        if (a.first + a.second != b.first + b.second) return a.first + a.second < b.first + b.second;
        return a.first - a.second < b.first - b.second;
    });
    
    // For each unique sum and difference, calculate the number of captured squares
    i64 ans = N * N;  // Start with all squares
    for (i64 i = 0; i < M; ++i) {
        i64 sum = pieces[i].first + pieces[i].second;
        i64 diff = pieces[i].first - pieces[i].second;
        
        // Count the number of pieces that can capture the current piece
        i64 captures = 0;
        for (i64 j = 0; j < i; ++j) {
            if (pieces[i].first + pieces[i].second == pieces[j].first + pieces[j].second ||
                pieces[i].first - pieces[i].second == pieces[j].first - pieces[j].second) {
                captures++;
            }
        }
        
        // Remove the captured squares from the total
        ans -= captures;
    }
    
    cout << ans << "\n";
    return 0;
}

It sorts the pieces based on the sum of their coordinates and then checks which pieces can capture the current piece based on the sum and difference of their coordinates. The result is the total number of squares minus the number of squares that are captured.
