```cpp
#include<bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    i64 N, M; cin >> N >> M;
    vector<pair<i64, i64>> pieces(M);
    set<pair<i64, i64>> occupied;
    for(i64 i = 0; i < M; ++i){
        i64 a, b; cin >> a >> b;
        pieces[i] = {a, b};
        occupied.insert({a, b});
    }
    
    i64 count = 0;
    for(i64 i = 1; i <= N; ++i){
        for(i64 j = 1; j <= N; ++j){
            bool valid = true;
            if(occupied.find({i, j}) != occupied.end()) continue;
            
            for(auto& [a, b]: pieces){
                if(a == i || b == j || a + b == i + j || a - b == i - j){
                    valid = false;
                    break;
                }
            }
            
            if(valid) ++count;
        }
    }
    
    cout << count << '\n';
    return 0;
}
```

This solution checks each cell individually to see if it's occupied and also if it would be captured by any other piece. If a cell isn't occupied and no other piece threatens it, it increments the counter. This approach ensures all possible placements are considered while respecting the constraints efficiently.
