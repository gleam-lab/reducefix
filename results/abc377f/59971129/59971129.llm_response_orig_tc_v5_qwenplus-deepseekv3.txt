#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main() {
    ll N, M;
    cin >> N >> M;
    
    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: i-j, diag2: i+j
    
    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }
    
    ll R = rows.size();
    ll C = cols.size();
    ll D1 = diag1.size();
    ll D2 = diag2.size();
    
    // Total squares covered by rows: R * N
    // Total squares covered by columns: C * N
    // Overlap between rows and columns: R * C (each row and column intersects at one square)
    ll row_col = R * N + C * N - R * C;
    
    // Now, add squares covered by diag1 and diag2, not already in row_col
    // For each diagonal in diag1, count the number of squares not in any existing row or column
    ll diag1_total = 0;
    for (ll d : diag1) {
        // Diagonal i - j = d → j = i - d
        ll i_min = max(1LL, 1 + d);
        ll i_max = min(N, N + d);
        if (i_min > i_max) continue;
        ll count = i_max - i_min + 1;
        // Subtract squares that are in existing rows or columns
        for (ll i = i_min; i <= i_max; ++i) {
            ll j = i - d;
            if (rows.count(i) || cols.count(j)) {
                count--;
            }
        }
        diag1_total += max(0LL, count);
    }
    
    ll diag2_total = 0;
    for (ll d : diag2) {
        // Diagonal i + j = d → j = d - i

        for (ll i = i_min; i <= i_max; ++i) {
            ll j = d - i;
            if (rows.count(i) || cols.count(j)) {
                count--;
            }
        }
        diag2_total += max(0LL, count);
    }
    
    // Now, add squares where diag1 and diag2 intersect, not already in rows or columns
    ll diag_diag = 0;
    for (ll d1 : diag1) {
        for (ll d2 : diag2) {
            // i - j = d1, i + j = d2 → 2i = d1 + d2, 2j = d2 - d1

                }

    return 0;
}
