#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define mp make_pair
#define pb push_back

const ll INF = 1e18;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll n, m;
    cin >> n >> m;

    set<ll> r, c, d1, d2;
    vector<array<ll, 2>> pieces(m);
    
    for(ll i = 0; i < m; ++i) {
        cin >> pieces[i][0] >> pieces[i][1];
        r.insert(pieces[i][0]);
        c.insert(pieces[i][1]);
        d1.insert(pieces[i][0] + pieces[i][1]);
        d2.insert(pieces[i][0] - pieces[i][1]);
    }

    ll ans = n * n - r.size() * c.size();
    
    for(ll i = 0; i < m; ++i) {
        ll x = pieces[i][0], y = pieces[i][1];
        if(r.count(x) == 1 && c.count(y) == 1) {
            continue;
        }
        
        auto itr = r.lower_bound(x);
        if(itr != r.end()) {
            ans -= (*itr - x) * (c.count(y) ? 1 : 0);
        }
        if(itr != begin(r)) {
            --itr;
            ans -= (x - *itr) * (c.count(y) ? 1 : 0);
        }
        
        itr = c.lower_bound(y);
        if(itr != end(c)) {
            ans -= (c.count(x) ? 1 : 0) * (*itr - y);
        }
        if(itr != begin(c)) {
            --itr;
            ans -= (y - *itr) * (c.count(x) ? 1 : 0);
        }
        
        itr = d1.lower_bound(x + y);
        if(itr != end(d1)) {
            ans -= (*itr - x - y) * (c.count(x) ? 1 : 0);
        }
        if(itr != begin(d1)) {
            --itr;
            ans -= (x + y - *itr) * (c.count(x) ? 1 : 0);
        }
        
        itr = d2.lower_bound(x - y);
        if(itr != end(d2)) {
            ans -= (*itr - x + y) * (c.count(x) ? 1 : 0);
        }
        if(itr != begin(d2)) {
            --itr;
            ans -= (x - y - *itr) * (c.count(x) ? 1 : 0);
        }
    }

    cout << ans << endl;

    return 0;
}
