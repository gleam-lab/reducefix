```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main() {
    ll N, M;
    cin >> N >> M;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: a-b, diag2: a+b

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total safe cells = total cells not in any attacked row or column
    ll ans = (N - (ll)rows.size()) * (N - (ll)cols.size());

    // Now subtract the number of cells that are on some diagonal but not in attacked rows or cols
    // Process diag1: a - b
    for (ll d : diag1) {
        // The line is y = x - d
        // Number of integer points on this diagonal within [1, N] x [1, N]
        ll len = N - abs(d);
        if (len <= 0) continue;

        // Subtract cells already covered by attacked rows or columns
        for (ll r : rows) {
            ll c = r - d;
            if (1 <= c && c <= N)
                --len;
        }

        for (ll c : cols) {
            ll r = c + d;
            if (1 <= r && r <= N)
                --len;
        }

        // Avoid double-counting cells that are in both attacked rows and cols
        map<pair<ll, ll>, bool> seen;
        for (ll r : rows) {
            for (ll c : cols) {
                ll x = r, y = c;
                if (x - y == d)
                    if (!seen[{x, y}])
                        ++len, seen[{x, y}] = true;
            }
        }

        ans -= len;
    }

    // Process diag2: a + b
    for (ll d : diag2) {
        // The line is y = -x + d
        // Number of integer points on this diagonal within [1, N] x [1, N]
        ll len = N - abs(d - (1 + N));
        if (len <= 0) continue;

        // Subtract cells already covered by attacked rows or columns
        for (ll r : rows) {
            ll c = d - r;
            if (1 <= c && c <= N)
                --len;
        }

        for (ll c : cols) {
            ll r = d - c;
            if (1 <= r && r <= N)
                --len;
        }

        // Avoid double-counting cells that are in both attacked rows and cols
        map<pair<ll, ll>, bool> seen;
        for (ll r : rows) {
            for (ll c : cols) {
                ll x = r, y = c;
                if (x + y == d)
                    if (!seen[{x, y}])
                        ++len, seen[{x, y}] = true;
            }
        }

        ans -= len;
    }

    // Finally, ensure we don't double count intersection of diagonals
    // Intersections of diag1 and diag2 may overlap with attacked rows/cols
    // But our above handling avoids over-subtraction using 'seen'

    cout << ans << endl;
    return 0;
}
```
