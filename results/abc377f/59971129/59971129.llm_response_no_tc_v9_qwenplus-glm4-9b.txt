#include <iostream>
#include <set>
using namespace std;

typedef long long ll;

int main() {
    ll n, m;
    cin >> n >> m;
    set<ll> rows, cols, diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll r, c;
        cin >> r >> c;
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r - c);
        diag2.insert(r + c);
    }

    ll free_rows = n - rows.size();
    ll free_cols = n - cols.size();
    ll free_diag1 = n - diag1.size();
    ll free_diag2 = n - diag2.size();

    // Calculate the number of free squares
    ll total_free_squares = n * n - m; // Total squares minus pieces
    total_free_squares -= free_rows;   // Subtract rows occupied by pieces
    total_free_squares -= free_cols;   // Subtract columns occupied by pieces
    total_free_squares -= free_diag1;  // Subtract diagonal occupied by pieces
    total_free_squares -= free_diag2;  // Subtract diagonal occupied by pieces

    cout << total_free_squares << endl;
    return 0;
}
