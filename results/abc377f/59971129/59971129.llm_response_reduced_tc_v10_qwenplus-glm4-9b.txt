#include <iostream>
#include <vector>
#include <set>
#include <map>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;
    set<long long> rows, cols, diagonals1, diagonals2;

    long long a, b;
    for (int i = 0; i < m; ++i) {
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diagonals1.insert(a + b);
        diagonals2.insert(a - b);
    }

    // Calculate the number of empty rows, columns, and diagonals
    long long free_rows = n - rows.size();
    long long free_cols = n - cols.size();
    long long free_diag1 = n - diagonals1.size();
    long long free_diag2 = n - diagonals2.size();

    // Calculate the number of captured squares by existing pieces
    long long captured = 0;

    // Rows and columns captured by rows and columns
    for (const auto& row : rows) {
        captured += (n - row + 1) * (n - row + 1);
    }
    for (const auto& col : cols) {
        captured += (n - col + 1) * (n - col + 1);
    }

    // Diagonals captured by rows and columns
    for (const auto& diag : diagonals1) {
        captured += (diag < n ? n - diag : 0) * (diag < n ? n - diag : 0);
    }
    for (const auto& diag : diagonals2) {
        captured += (diag < n ? n - diag : 0) * (diag < n ? n - diag : 0);
    }

    // Calculate the total number of squares
    long long total_squares = n * n;

    // Subtract the number of captured squares from the total to get the number of free squares
    long long free_squares = total_squares - captured;

    // Output the result
    cout << free_squares << endl;

    return 0;
}
