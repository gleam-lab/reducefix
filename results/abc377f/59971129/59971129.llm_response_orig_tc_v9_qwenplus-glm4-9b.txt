#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 1005;
vector<int> rowCap[N], colCap[N], diag1Cap[N * 2 + 3], diag2Cap[N * 2 + 3];

int main() {
    ll n, m;
    cin >> n >> m;
    for (ll i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        rowCap[a].push_back(i);
        colCap[b].push_back(i);
        diag1Cap[a + b].push_back(i);
        diag2Cap[a - b + N].push_back(i);
    }

    ll totalEmpty = n * n;
    for (ll i = 1; i <= n; ++i) {
        // Calculate the number of valid squares in row i
        ll rowTaken = rowCap[i].size() * 2;
        for (ll j = 0; j < rowCap[i].size(); ++j) {
            ll k = j + 1;
            while (k < rowCap[i].size() && rowCap[i][j] + k == rowCap[i][k]) {
                rowTaken += 2;
                ++k;
            }
            break;
        }
        totalEmpty -= rowTaken;

        // Calculate the number of valid squares in column i
        ll colTaken = colCap[i].size() * 2;
        for (ll j = 0; j < colCap[i].size(); ++j) {
            ll k = j + 1;
            while (k < colCap[i].size() && colCap[i][j] + k == colCap[i][k]) {
                colTaken += 2;
                ++k;
            }
            break;
        }
        totalEmpty -= colTaken;

        // Calculate the number of valid squares in diagonal 1 (i+j)
        ll diag1Taken = diag1Cap[i + 1].size() * 2;
        for (ll j = 0; j < diag1Cap[i + 1].size(); ++j) {
            ll k = j + 1;
            while (k < diag1Cap[i + 1].size() && diag1Cap[i + 1][j] + k == diag1Cap[i + 1][k]) {
                diag1Taken += 2;
                ++k;
            }
            break;
        }
        totalEmpty -= diag1Taken;

        // Calculate the number of valid squares in diagonal 2 (i-j+1)
        ll diag2Taken = diag2Cap[i + N].size() * 2;
        for (ll j = 0; j < diag2Cap[i + N].size(); ++j) {
            ll k = j + 1;
            while (k < diag2Cap[i + N].size() && diag2Cap[i + N][j] + k == diag2Cap[i + N][k]) {
                diag2Taken += 2;
                ++k;
            }
            break;
        }
        totalEmpty -= diag2Taken;
    }

    cout << totalEmpty << endl;
    return 0;
}
