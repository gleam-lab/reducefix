typedef long long ll;
#define N 1005
#define mk make_pair
ll n,m,a[N],b[N],ans;
set<ll>s,t,u,v;//s被选行，t被选列 
map<pair<ll,ll>,bool>vis;
int main(){
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=m;++i){
        scanf("%lld%lld",&a[i],&b[i]);

        t.insert(b[i]);
        u.insert(a[i]-b[i]);
        v.insert(a[i]+b[i]);
    }
    ans=(n-s.size())*(n-t.size());
    for(auto x:u){
        ll tmp=n-abs(x);
        for(auto y:s){
            if(y-x>=1&&y-x<=n&&!vis[mk(y,y-x)]){
                --tmp;
                vis[mk(y,y-x)]=1;
            }
        }
        for(auto y:t){
            if(x+y>=1&&x+y<=n&&!vis[mk(x+y,y)]){
                --tmp;
                vis[mk(x+y,y)]=1;
            }
        }
        ans-=tmp;
        vis.clear();
    }
    for(auto x:v){
        ll tmp=n-abs(x-n);
        for(auto y:s){

                --tmp;
                vis[mk(p/2,q/2)]=1;
            }
        }
        ans-=tmp;
        vis.clear();
    }
    printf("%lld",ans+1);
    return 0;
}
