#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;

#define all(x) (x).begin(), (x).end()

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    set<pll> occupied;
    set<ll> rows, cols, diag1, diag2;

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        occupied.insert({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total empty squares: total = N*N - M
    // Safe squares = total - number of squares threatened or occupied

    // First count the number of squares attacked by existing pieces
    set<pll> attacked;

    for (const auto& p : occupied) {
        ll a = p.first, b = p.second;
        // Row
        for (ll j = 1; j <= N; ++j) {
            if (j != b) attacked.insert({a, j});
        }
        // Column
        for (ll i = 1; i <= N; ++i) {
            if (i != a) attacked.insert({i, b});
        }
        // Diagonal 1 (a-b constant)
        ll d = a - b;
        for (ll i = max(1LL, d+1); i <= N && i-d >= 1; ++i) {
            ll j = i - d;
            if (j >= 1 && j <= N && (i != a || j != b))
                attacked.insert({i, j});
        }
        // Diagonal 2 (a+b constant)
        d = a + b;
        for (ll i = max(1LL, d-N); i <= min(N, d-1); ++i) {
            ll j = d - i;
            if (j >= 1 && j <= N && (i != a || j != b))
                attacked.insert({i, j});
        }
    }

    // Now calculate safe squares
    ll total_safe = N * N - M - attacked.size();

    cout << total_safe << endl;

    return 0;
}
