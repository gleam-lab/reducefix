#include <bits/stdc++.h>
using namespace std;

#define MAXN 1000000000

typedef pair<int, int> PII;

int n, m;
set<int> s, t, u, v;

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 0; i < m; ++i) {
        int a, b;
        scanf("%d %d", &a, &b);
        s.insert(a);
        t.insert(b);
        u.insert(a - b);
        v.insert(a + b);
    }

    long long emptySquares = n * n;
    emptySquares -= s.size(); // Subtract squares in selected rows
    emptySquares -= t.size(); // Subtract squares in selected columns

    // Subtract the intersections of selected rows and columns
    set<PII> intersections;
    for (auto &p : s) {
        for (auto &q : t) {
            intersections.insert({p, q});
        }
    }

    emptySquares -= intersections.size();

    // Subtract the diagonals intersecting selected rows and columns
    set<PII> diag1, diag2;
    for (auto &p : s) {
        for (auto &q : t) {
            diag1.insert({p, p + q});
            diag2.insert({p, p - q});
        }
    }

    emptySquares -= diag1.size();
    emptySquares -= diag2.size();

    // Subtract the squares that are points on both diagonals
    set<int> diag1Points, diag2Points;
    for (auto &p : diag1) diag1Points.insert(p.second);
    for (auto &p : diag2) diag2Points.insert(p.second);

    for (auto &p : diag1Points) {
        if (diag2Points.count(p)) {
            emptySquares--;
        }
    }

    cout << emptySquares << endl;
    return 0;
}
