#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<n;++i)
#define pb push_back
#define all(x) (x).begin(),(x).end()
#define rall(x) (x).rbegin(),(x).rend()
#define sz(x) ((int)(x).size())
#define clr(x) memset(x,0,sizeof(x))
#define mp make_pair
#define fi first
#define se second
#define db double
#define sqr(x) ((x)*(x))

typedef long long ll;
const int maxn = 1e9 + 7;

vector<int> r, c, d1, d2;

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr);
    
    int n, m; cin >> n >> m;
    r.resize(m), c.resize(m), d1.resize(m), d2.resize(m);
    rep(i, m) cin >> r[i] >> c[i], --r[i], --c[i];
    set<int> s, t, u, v;
    rep(i, m) s.insert(r[i]), t.insert(c[i]), u.insert(r[i] - c[i]), v.insert(r[i] + c[i]);
    int ans = (n - s.size()) * (n - t.size());
    for (auto& x : u) {
        int tmp = n - abs(x);
        for (auto& y : s) if (abs(y - x) >= 1 && abs(y - x) <= n) --tmp;
        for (auto& y : t) if (x + y >= 1 && x + y <= n) --tmp;
        ans -= tmp;
    }
    for (auto& x : v) {
        int tmp = n - abs(x);
        for (auto& y : s) if (abs(x - y) >= 1 && abs(x - y) <= n) --tmp;
        for (auto& y : t) if (x + y >= 1 && x + y <= n) --tmp;
        for (auto& y : u) if (x == y || x == 2 * y) --tmp;
        ans -= tmp;
    }
    cout << ans + 1 << '\n';
    return 0;
}
