```cpp
#include <bits/stdc++.h>
using namespace std;

#define ll long long

ll n, m, ans = 0;
vector<bool> row(1000001), col(1000001), diag1(2000001), diag2(2000001);

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        row[a] = true;
        col[b] = true;
        diag1[a + b] = true;
        diag2[a - b + 1000000] = true;
    }
    for (int i = 1; i <= n; ++i) {
        if (!row[i]) ans++;
        if (!col[i]) ans++;
        if (!diag1[i]) ans++;
        if (!diag2[i]) ans++;
    }
    cout << ans / 2 << endl; // Each valid position is counted twice
    return 0;
}
```
