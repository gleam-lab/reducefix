```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
#define mk make_pair

int main() {
    ll N, M;
    cin >> N >> M;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: a-b, diag2: a+b

    map<pair<ll, ll>, bool> occupied;

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
        occupied[mk(a, b)] = true;
    }

    // Total empty squares
    ll total_empty = (N - rows.size()) * (N - cols.size());

    // Now we need to subtract the number of points on diagonals that are not actually empty
    set<pair<ll, ll>> attacked;

    // For each diagonal of type a - b = d
    for (ll d : diag1) {
        // Points on this diagonal satisfy a - b = d => a = b + d
        ll start_b = max(1LL, 1LL - d);
        ll end_b = min(N, N - d);

        if (start_b > end_b) continue;

        ll count = end_b - start_b + 1;
        total_empty -= count;

        // Add these points to attacked set
        for (ll b = start_b; b <= end_b; ++b) {
            ll a = b + d;
            attacked.insert({a, b});
        }
    }

    // For each diagonal of type a + b = s
    for (ll s : diag2) {
        // Points on this diagonal satisfy a + b = s => b = s - a
        ll start_a = max(1LL, s - N);
        ll end_a = min(N, s - 1);

        if (start_a > end_a) continue;

        ll count = end_a - start_a + 1;
        total_empty -= count;

        // Add these points to attacked set
        for (ll a = start_a; a <= end_a; ++a) {
            ll b = s - a;
            attacked.insert({a, b});
        }
    }

    // Now add back intersections (points counted twice)
    ll intersection_count = 0;
    for (auto p : attacked) {
        if (occupied.count(p)) continue;
        if (rows.count(p.first) || cols.count(p.second)) continue;
        ll d1 = p.first - p.second;
        ll d2 = p.first + p.second;
        if (diag1.count(d1) && diag2.count(d2)) {
            intersection_count++;
        }
    }

    total_empty += intersection_count;

    cout << total_empty << endl;

    return 0;
}
```
