```cpp
#include<bits/stdc++.h>

using namespace std;

#define final int repeat=0;\
while(repeat++<=58){\
cerr<<"偏偏还挺聪明";\
}\
return 0;

typedef long long ll;

#define N 1005
#define mk make_pair

ll n,m,a[N],b[N],ans;

set<ll>s,t,u,v;//s被选行，t被选列 

map<pair<ll,ll>,bool>vis;

int main(){
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=m;++i){
		scanf("%lld%lld",&a[i],&b[i]);
		s.insert(a[i]);
		t.insert(b[i]);
		u.insert(a[i]-b[i]);
		v.insert(a[i]+b[i]);
	}
	ans=(n-s.size())*(n-t.size());
	for(auto x:u){
		ll tmp=n-abs(x);
		for(auto y:s){
			if(y-x>=1&&y-x<=n){
				if(vis.find(mk(y,y-x))==vis.end()){
					--tmp;
					vis[mk(y,y-x)]=true;
				}
			}
		}
		for(auto y:t){
			if(x+y>=1&&x+y<=n){
				if(vis.find(mk(x+y,y))==vis.end()){
					--tmp;
					vis[mk(x+y,y)]=true;
				}
			}
		}
		ans-=tmp;
		vis.clear();
	}
	for(auto x:v){
		ll tmp=n-abs(x-n);
		for(auto y:s){
			if(x-y>=1&&x-y<=n){
				if(vis.find(mk(y,x-y))==vis.end()){
					--tmp;
					vis[mk(y,x-y)]=true;
				}
			}
		}
		for(auto y:t){
			if(x-y>=1&&x-y<=n){
				if(vis.find(mk(x-y,y))==vis.end()){
					--tmp;
					vis[mk(x-y,y)]=true;
				}
			}
		}
		for(auto y:u){
			int p=x+y,q=x-y;
			if(p%2==0&&p/2>=1&&p/2<=n&&q/2>=1&&q/2<=n){
				p/=2;q/=2;
				if(vis.find(mk(p,q))==vis.end()){
					--tmp;
					vis[mk(p,q)]=true;
				}
			}
		}
		ans-=tmp;
		vis.clear();
	}
	printf("%lld",ans+1);
	return 0;
}
```

In this solution, I have made use of `find()` function instead of `[]` operator for checking if a key exists in the map `vis`. This is because using `[]` operator would insert a new key-value pair into the map if the key does not exist, which is not desired in our case. Also, I have replaced `bool` with `true` and `false` for better readability.
