```cpp
#include<bits/stdc++.h>
using namespace std;
#define ll long long
const ll MAXN = 1e9 + 5;

vector<ll> row(MAXN), col(MAXN), diag1(2 * MAXN - 1), diag2(2 * MAXN - 1);

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    ll n, m; cin >> n >> m;
    for(ll i = 1; i <= m; ++i) {
        ll x, y; cin >> x >> y;
        row[x] = col[y] = diag1[x + y] = diag2[x - y + n] = 1;
    }
    
    ll ans = n * n;
    for(ll i = 1; i <= n; ++i) {
        if(row[i]) --ans;
        if(col[i]) --ans;
        if(diag1[i]) --ans;
        if(diag2[i]) --ans;
    }
    
    cout << ans << '\n';
    return 0;
}
```
