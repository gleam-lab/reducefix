#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main(){
    ll n, m;
    cin >> n >> m;
    
    set<ll> rows, cols, diags, anti_diags;
    map<pair<ll,ll>, bool> vis;
    
    for(ll i = 0; i < m; ++i){
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diags.insert(a + b);
        anti_diags.insert(a - b);
        
        // Mark all positions that can be captured by this piece
        for(ll j = 1; j <= n; ++j){
            if(j != a && !vis[{j, b}]){
                --rows.size();
                vis[{j, b}] = true;
            }
            if(j != b && !vis[{a, j}]){
                --cols.size();
                vis[{a, j}] = true;
            }
            if((a + j) != (a + b) && !vis[{a + j, b}]){
                --diags.size();
                vis[{a + j, b}] = true;
            }
            if((a - j) != (a - b) && !vis[{a, b - j}]){
                --anti_diags.size();
                vis[{a, b - j}] = true;
            }
        }
    }
    
    // Calculate the number of empty squares
    ll ans = (n - rows.size()) * (n - cols.size());
    for(ll d : diags){
        ans -= n - abs(d);
    }
    for(ll ad : anti_diags){
        ans -= n - abs(ad);
    }
    
    cout << ans + 1 << endl;
    return 0;
}
