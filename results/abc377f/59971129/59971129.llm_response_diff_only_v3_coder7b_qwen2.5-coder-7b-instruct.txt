#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

#define N 1005

int main(){
    ll n, m;
    scanf("%lld%lld", &n, &m);
    set<ll> s, t, u, v; // s for selected rows, t for selected columns
    map<pair<ll, ll>, bool> vis; // visited map for diagonals

    for(ll i = 1; i <= m; ++i){
        ll a, b;
        scanf("%lld%lld", &a, &b);
        s.insert(a);
        t.insert(b);
        u.insert(a - b);
        v.insert(a + b);
    }

    // Calculate initial count of valid positions based on non-selected rows and columns
    ll ans = (n - s.size()) * (n - t.size());

    // Iterate over each diagonal and adjust the answer based on intersections with other pieces
    for(auto x : u){
        ll tmp = n - abs(x);
        for(auto y : s){
            if(abs(y - x) >= 1 && abs(y - x) <= n){
                if(!vis[{y, y - x}]){
                    --tmp;
                    vis[{y, y - x}] = true;
                }
            }
        }
        for(auto y : t){
            if(abs(x + y) >= 1 && abs(x + y) <= n){
                if(!vis[{x + y, y}]){
                    --tmp;
                    vis[{x + y, y}] = true;
                }
            }
        }
        ans -= tmp;
        vis.clear();
    }

    for(auto x : v){
        ll tmp = n - abs(x - n);
        for(auto y : s){
            if(abs(x - y) >= 1 && abs(x - y) <= n){
                if(!vis[{y, x - y}]){
                    --tmp;
                    vis[{y, x - y}] = true;
                }
            }
        }
        for(auto y : t){
            if(abs(x - y) >= 1 && abs(x - y) <= n){
                if(!vis[{x - y, y}]){
                    --tmp;
                    vis[{x - y, y}] = true;
                }
            }
        }
        for(auto y : u){
            ll p = x + y, q = x - y;
            if((p % 2 == 0 || q % 2 == 0) && p / 2 >= 1 && p / 2 <= n && q / 2 >= 1 && q / 2 <= n){
                p /= 2; q /= 2;
                if(!vis[{p, q}]){
                    --tmp;
                    vis[{p, q}] = true;
                }
            }
        }
        ans -= tmp;
        vis.clear();
    }

    printf("%lld\n", ans + 1); // Add one because we need to place a piece
    return 0;
}
