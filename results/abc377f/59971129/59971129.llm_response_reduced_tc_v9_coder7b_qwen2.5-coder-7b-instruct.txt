```cpp
#include <bits/stdc++.h>
using namespace std;

#define N 1000000000

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);

    long long n, m;
    cin >> n >> m;

    set<long long> s, t, u, v;
    unordered_map<long long, bool> vis;

    for(long long i = 0; i < m; ++i) {
        long long a, b;
        cin >> a >> b;
        s.insert(a);
        t.insert(b);
        u.insert(a - b);
        v.insert(a + b);
    }

    long long ans = (n - s.size()) * (n - t.size());

    for(auto x : u) {
        long long tmp = n - abs(x);
        for(auto y : s) {
            if(y - x >= 1 && y - x <= n) {
                if(!vis[y - x]) {
                    --tmp;
                    vis[y - x] = true;
                }
            }
        }
        for(auto y : t) {
            if(x + y >= 1 && x + y <= n) {
                if(!vis[x + y]) {
                    --tmp;
                    vis[x + y] = true;
                }
            }
        }
        ans -= tmp;
        vis.clear();
    }

    for(auto x : v) {
        long long tmp = n - abs(x - n);
        for(auto y : s) {
            if(x - y >= 1 && x - y <= n) {
                if(!vis[y - (x - y)]) {
                    --tmp;
                    vis[y - (x - y)] = true;
                }
            }
        }
        for(auto y : t) {
            if(x - y >= 1 && x - y <= n) {
                if(!vis[(x - y) - y]) {
                    --tmp;
                    vis[(x - y) - y] = true;
                }
            }
        }
        for(auto y : u) {
            long long p = x + y, q = x - y;
            if(p % 2 == 0 && p / 2 >= 1 && p / 2 <= n && q / 2 >= 1 && q / 2 <= n) {
                p /= 2; q /= 2;
                if(!vis[p - q]) {
                    --tmp;
                    vis[p - q] = true;
                }
            }
        }
        ans -= tmp;
        vis.clear();
    }

    cout << ans + 1 << '\n';

    return 0;
}
```
