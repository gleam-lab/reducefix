#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;

ll n, m;
set<ll> rows, cols;
set<ll> diag1, diag2; // diag1: a-b, diag2: a+b

map<pll, bool> occupied;

int main() {
    scanf("%lld%lld", &n, &m);
    
    for (ll i = 0; i < m; ++i) {
        ll a, b;
        scanf("%lld%lld", &a, &b);
        
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
        occupied[{a, b}] = true;
    }
    
    // Total empty squares
    ll total_empty = n * n - m;
    
    // Squares attacked by row and column
    ll attacked_by_row_or_col = (rows.size() * n + cols.size() * n - rows.size() * cols.size());
    
    // Now handle diagonals
    set<pll> attacked_squares;
    
    for (ll d : diag1) {
        for (ll r : rows) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                attacked_squares.insert({r, c});
            }
        }
        for (ll c : cols) {
            ll r = d + c;
            if (1 <= r && r <= n) {
                attacked_squares.insert({r, c});
            }
        }
        
        // Main diagonal line: i-j = d
        ll start_r = max(1LL, d + 1);
        ll end_r = min(n + d, n);
        
        for (ll r = start_r; r <= end_r; ++r) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                attacked_squares.insert({r, c});
            }
        }
    }
    
    for (ll d : diag2) {
        for (ll r : rows) {
            ll c = d - r;
            if (1 <= c && c <= n) {
                attacked_squares.insert({r, c});
            }
        }
        for (ll c : cols) {
            ll r = d - c;
            if (1 <= r && r <= n) {
                attacked_squares.insert({r, c});
            }
        }
        
        // Anti-diagonal line: i+j = d
        ll start_r = max(1LL, d - n);
        ll end_r = min(n, d - 1);
        
        for (ll r = start_r; r <= end_r; ++r) {
            ll c = d - r;
            if (1 <= c && c <= n) {
                attacked_squares.insert({r, c});
            }
        }
    }
    
    // Remove already occupied squares from attacked count
    ll total_attacked = attacked_squares.size();
    for (auto pr : occupied) {
        if (attacked_squares.count(pr.first)) {
            total_attacked--;
        }
    }
    
    // Calculate valid positions
    ll result = total_empty - total_attacked;
    printf("%lld\n", result);
    
    return 0;
}
