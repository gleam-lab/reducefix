```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main() {
    ll N, M;
    cin >> N >> M;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: a-b, diag2: a+b

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total squares not in any attacked row or column
    ll ans = (N - (ll)rows.size()) * (N - (ll)cols.size());

    // Now subtract the number of squares that lie on diagonals and are not already occupied
    set<pair<ll, ll>> attacked_cells;

    // Add all cells on the same diagonal/diagonal2 as existing pieces
    for (ll i = 1; i <= N; ++i) {
        for (ll d : diag1) {
            ll j = i - d;
            if (1 <= j && j <= N) {
                attacked_cells.insert({i, j});
            }
        }
        for (ll d : diag2) {
            ll j = d - i;
            if (1 <= j && j <= N) {
                attacked_cells.insert({i, j});
            }
        }
    }

    // Subtract the attacked cells that are not already occupied by pieces
    for (auto& p : attacked_cells) {
        if (rows.count(p.first) || cols.count(p.second)) continue;
        --ans;
    }

    // Finally, remove the cells where there are existing pieces
    for (ll i = 0; i < M; ++i) {
        ll a = p[i].first, b = p[i].second;
        if (rows.count(a) == 0 && cols.count(b) == 0) {
            // This cell was counted in ans, but it's actually occupied
            --ans;
        }
    }

    cout << ans << endl;
    return 0;
}
```
