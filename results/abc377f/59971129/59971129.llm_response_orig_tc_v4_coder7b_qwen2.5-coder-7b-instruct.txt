#include<bits/stdc++.h>
using namespace std;

#define rep(i,a,b) for(int i=a;i<b;i++)
#define all(x) begin(x),end(x)

int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);
    long long n,m,i,j,k,l,r,s,t,u,v,w,x,y,z;
    cin>>n>>m;
    set<long long>rows,cols,d1,d2;
    vector<vector<bool>>grid(n,vector<bool>(n));
    rep(i,0,m){
        cin>>x>>y;
        --x,--y;
        grid[x][y]=true;
        rows.insert(x);
        cols.insert(y);
        d1.insert(x+y);
        d2.insert(x-y);
    }
    long long ans=n*n-rows.size()-cols.size();
    rep(i,0,n){
        if(rows.find(i)!=end(rows)) continue;
        rep(j,0,n){
            if(grid[i][j]) continue;
            if(cols.find(j)!=end(cols)) continue;
            if(d1.find(i+j)!=end(d1)) continue;
            if(d2.find(i-j)!=end(d2)) continue;
            ++ans;
        }
    }
    cout<<ans<<'\n';
    return 0;
}
