#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: i-j, diag2: i+j

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total empty squares
    ll total_empty = N * N - M;

    // Squares attacked via row or column
    ll attacked_by_row_or_col = (N * rows.size() + N * cols.size() - rows.size() * cols.size());

    // Squares attacked via diagonals
    set<pair<ll, ll>> attacked_diags;
    for (ll d : diag1) {
        // For diagonal i-j == d
        // i = j + d
        // Valid if 1 <= j <= N and 1 <= i <= N => max(1, 1-d) <= j <= min(N, N-d)
        ll low = max(1LL, 1LL - d);
        ll high = min(N, N - d);
        if (low <= high) {
            ll cnt = high - low + 1;
            for (ll j = low; j <= high; ++j) {
                ll i = j + d;
                if (!rows.count(i) && !cols.count(j)) {
                    attacked_diags.insert({i, j});
                }
            }
        }
    }

    for (ll d : diag2) {
        // For diagonal i+j == d
        // i = d - j
        // Valid if 1 <= j <= N and 1 <= i <= N => max(1, d - N) <= j <= min(N, d - 1)
        ll low = max(1LL, d - N);
        ll high = min(N, d - 1LL);
        if (low <= high) {
            ll cnt = high - low + 1;
            for (ll j = low; j <= high; ++j) {
                ll i = d - j;
                if (!rows.count(i) && !cols.count(j)) {
                    attacked_diags.insert({i, j});
                }
            }
        }
    }

    // Subtract already counted attacked cells from row/col attack count
    ll attacked_by_diag_only = attacked_diags.size();

    // Total attacked squares
    ll total_attacked = attacked_by_row_or_col - rows.size() * cols.size() + attacked_by_diag_only;

    // The final answer
    ll answer = total_empty - total_attacked;
    cout << answer << endl;

    return 0;
}
