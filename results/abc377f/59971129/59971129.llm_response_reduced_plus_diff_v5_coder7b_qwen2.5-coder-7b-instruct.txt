    rep(i,n){
        rep(j,n){
            if(s.find({i+1,j+1}) == s.end()){
                bool ok = true;
                for(int k=1;k<=n;k++){
                    if(k != i+1 && s.find({k,j+1}) != s.end()){
                        ok = false; break;
                    }
                    if(k != j+1 && s.find({i+1,k}) != s.end()){
                        ok = false; break;
                    }
                }

                        if(abs(k-i) == d && s.find({k,j+1}) != s.end()){
                            ok = false; break;
                        }
                        if(abs(k-j) == d && s.find({i+1,k}) != s.end()){
                            ok = false; break;
                        }
                    }
                }
                if(ok) ans++;
            }
        }
    }
    cout<<ans<<endl;

    return 0;
}
