#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll INF = 1e18;

vector<vector<bool>> board;
vector<ll> rows, cols, diags1, diags2;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0);

    ll n, m; cin >> n >> m;
    board.resize(n + 1, vector<bool>(n + 1));
    for(ll i = 0; i < m; ++i) {
        ll r, c; cin >> r >> c;
        board[r][c] = true;
        rows[r]++; cols[c]++;
        diags1[r+c]++; diags2[r-c+n-1]++;
    }

    ll ans = 0;
    for(ll i = 1; i <= n; ++i) {
        for(ll j = 1; j <= n; ++j) {
            if(!board[i][j] && !rows[i] && !cols[j] && !diags1[i+j] && !diags2[i-j+n-1]) {
                ans++;
            }
        }
    }

    cout << ans << endl;

    return 0;
}
