#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <cmath>

using namespace std;

typedef long long ll;

int main() {
    ll n, m;
    cin >> n >> m;
    vector<pair<ll, ll>> pieces;
    
    // Reading the pieces and storing them in a set for unique rows and columns
    set<ll> rows, cols;
    for (ll i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        pieces.push_back({a, b});
        rows.insert(a);
        cols.insert(b);
    }
    
    // Calculate the number of empty squares
    ll emptySquares = n * n;
    
    // Subtract squares in the selected rows and columns
    emptySquares -= rows.size() * n;
    emptySquares -= cols.size() * n;
    
    // Subtract the squares that are at the intersections of selected rows and columns
    set<pair<ll, ll>> intersections;
    for (auto& p : pieces) {
        ll x = p.first, y = p.second;
        // Check all intersections related to row x and column y
        for (ll i = 1; i <= n; ++i) {
            ll inter1 = x + i - y;
            ll inter2 = x - i + y;
            // Check if the intersections are within bounds
            if (inter1 > 0 && inter1 <= n && y - i > 0 && y - i <= n) {
                intersections.insert({inter1, y - i});
            }
            if (inter2 > 0 && inter2 <= n && y + i > 0 && y + i <= n) {
                intersections.insert({inter2, y + i});
            }
        }
    }
    
    // Subtract the intersections from the count of empty squares
    emptySquares -= intersections.size();
    
    // Output the result
    cout << emptySquares << endl;
    
    return 0;
}

This code reads the input values, calculates the number of empty squares that cannot be captured by existing pieces, and prints the result. It uses sets to keep track of the selected rows and columns, and it checks for intersections of selected rows and columns with existing pieces to determine additional squares that cannot be used. The result is the number of empty squares that can be used to place a new piece without it being captured.
