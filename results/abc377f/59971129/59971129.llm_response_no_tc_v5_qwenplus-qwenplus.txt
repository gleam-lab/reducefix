```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main() {
    ll N, M;
    cin >> N >> M;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: i-j, diag2: i+j

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total empty squares = total squares - occupied squares
    ll total_empty = N * N - M;

    // Squares that are either in occupied row or column
    ll unsafe_row_col = (N * rows.size() + N * cols.size() - rows.size() * cols.size());

    // Squares on diagonals but not already counted in row/col
    ll extra_unsafe = 0;

    // Process diag1: i - j
    for (ll d : diag1) {
        // Number of cells on diagonal i-j = d
        ll count;
        if (d >= -(N-1) && d <= N-1) {
            if (N + d <= N)
                count = N - abs(d);
            else
                count = 2 * N - 1 - abs(d);
        } else {
            continue;
        }

        // Subtract cells already counted in row/col
        ll overlap = 0;
        for (ll r : rows) {
            ll c = r - d;
            if (c >= 1 && c <= N && !cols.count(c)) {
                overlap++;
            }
        }
        for (ll c : cols) {
            ll r = d + c;
            if (r >= 1 && r <= N && !rows.count(r)) {
                overlap++;
            }
        }
        extra_unsafe += count - overlap;
    }

    // Process diag2: i + j
    for (ll d : diag2) {
        // Number of cells on diagonal i+j = d
        ll count;
        if (d < 2 || d > 2 * N)
            continue;
        if (d <= N + 1)
            count = d - 1;
        else
            count = 2 * N + 1 - d;

        // Subtract cells already counted in row/col
        ll overlap = 0;
        for (ll r : rows) {
            ll c = d - r;
            if (c >= 1 && c <= N && !cols.count(c)) {
                overlap++;
            }
        }
        for (ll c : cols) {
            ll r = d - c;
            if (r >= 1 && r <= N && !rows.count(r)) {
                overlap++;
            }
        }
        extra_unsafe += count - overlap;
    }

    ll answer = total_empty - unsafe_row_col - extra_unsafe;
    cout << max(0LL, answer);
    return 0;
}
```
