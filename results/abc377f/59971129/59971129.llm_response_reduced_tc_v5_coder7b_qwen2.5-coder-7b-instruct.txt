#include<bits/stdc++.h>
using namespace std;
#define N 1000000000
long long n,m,a[1003],b[1003];
bool v[2000000002]={false};
int main(){
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=m;++i){
        scanf("%lld%lld",&a[i],&b[i]);
        v[a[i]]|=1<<b[i];
        v[b[i]+1000000001]|=1<<(a[i]-b[i]+1000000000);
        v[a[i]+b[i]]|=1<<(b[i]+a[i]);
        v[b[i]-a[i]+1000000001]|=1<<(b[i]-a[i]+1000000000);
    }
    long long ans=n*n;
    for(int i=1;i<=n;++i){
        if(v[i]||v[i+1000000001]||v[i+2000000000]||v[i+3000000000])
            ans--;
        else{
            int c=__builtin_popcount(v[i])+
                __builtin_popcount(v[i+1000000001])+
                __builtin_popcount(v[i+2000000000])+
                __builtin_popcount(v[i+3000000000]);
            ans-=(c+1)*(c+2)/2-c;
        }
    }
    printf("%lld\n",ans);
    return 0;
}
