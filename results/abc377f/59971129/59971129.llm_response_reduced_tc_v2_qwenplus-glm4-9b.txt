#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <algorithm>
using namespace std;

using ll = long long;

int main() {
    ll n, m;
    cin >> n >> m;
    set<ll> rows, cols, diag1, diag2;
    vector<pair<ll, ll>> points(m);

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        points[i] = {x, y};
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    ll total_empty_squares = n * n - m;
    ll captured_by_rows = rows.size();
    ll captured_by_cols = cols.size();
    ll captured_by_diag1 = diag1.size();
    ll captured_by_diag2 = diag2.size();

    // Calculate the number of squares that are captured by lines
    set<pair<ll, ll>> affected_squares;
    for (ll x : rows) {
        for (ll y : cols) {
            affected_squares.insert({x, y});
            affected_squares.insert({x, n - y + 1});
        }
    }
    for (ll y : cols) {
        for (ll x : rows) {
            affected_squares.insert({x, y});
            affected_squares.insert({n - x + 1, y});
        }
    }
    for (ll d : diag1) {
        for (ll y = max(1LL, d - n + 1); y <= min(n, d); ++y) {
            affected_squares.insert({d - y, y});
            affected_squares.insert({n - d + y, y});
        }
    }
    for (ll d : diag2) {
        for (ll y = max(1LL, d); y <= min(n, 2 * n - d); ++y) {
            affected_squares.insert({d - (n - y), y});
            affected_squares.insert({d - y, n - y});
        }
    }

    // Subtract the number of affected squares from the total
    total_empty_squares -= affected_squares.size();

    cout << total_empty_squares << endl;
    return 0;
}
