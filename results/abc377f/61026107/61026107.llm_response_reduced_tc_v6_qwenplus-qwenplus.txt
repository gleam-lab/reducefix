#include <bits/stdc++.h>

using namespace std;
using ll = long long;

struct Point {
    ll x, y;
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: x - y, diag2: x + y

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    ll total = n * n;
    
    // Subtract all squares attacked by existing pieces
    // Rows and columns
    total -= rows.size() * n;
    total -= cols.size() * n;

    // Diagonals diag1 (x - y)
    for (auto d : diag1) {
        ll count;
        if (d >= 0) {
            count = n - d;
        } else {
            count = n + d;
        }
        total -= count;
    }

    // Diagonals diag2 (x + y)
    for (auto d : diag2) {
        ll count;
        if (d <= n + 1) {
            count = d - 1;
        } else {
            count = 2 * n + 1 - d;
        }
        total -= count;
    }

    // Add back intersections (over-subtracted)

    // Row-Column intersections (already counted in grid)
    for (auto r : rows) {
        for (auto c : cols) {
            total++;
        }
    }

    // Row-Diag1 intersections
    for (auto r : rows) {
        for (auto d : diag1) {
            ll c = r - d;
            if (c >= 1 && c <= n) {
                total++;
            }
        }
    }

    // Row-Diag2 intersections
    for (auto r : rows) {
        for (auto d : diag2) {
            ll c = d - r;
            if (c >= 1 && c <= n) {
                total++;
            }
        }
    }

    // Col-Diag1 intersections
    for (auto c : cols) {
        for (auto d : diag1) {
            ll r = d + c;
            if (r >= 1 && r <= n) {
                total++;
            }
        }
    }

    // Col-Diag2 intersections
    for (auto c : cols) {
        for (auto d : diag2) {
            ll r = d - c;
            if (r >= 1 && r <= n) {
                total++;
            }
        }
    }

    // Diag1-Diag2 intersections
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            // Solve:
            // x - y = d1
            // x + y = d2
            // => 2x = d1 + d2 => x = (d1 + d2)/2
            // => 2y = d2 - d1 => y = (d2 - d1)/2
            if ((d1 + d2) % 2 == 0) {
                ll x = (d1 + d2) / 2;
                ll y = (d2 - d1) / 2;
                if (x >= 1 && x <= n && y >= 1 && y <= n) {
                    total++;
                }
            }
        }
    }

    // Remove occupied cells (they are not empty)
    total += m;

    cout << total << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();
    return 0;
}
