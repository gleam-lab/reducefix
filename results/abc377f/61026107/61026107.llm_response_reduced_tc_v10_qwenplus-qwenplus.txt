#include <bits/stdc++.h>

using namespace std;
using ll = long long;

struct Point {
    ll x, y;
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    // Total squares = all squares minus those attacked or occupied
    ll total = n * n;

    // Subtract all squares in same row
    total -= rows.size() * n;

    // Subtract all squares in same column
    total -= cols.size() * n;

    // Subtract all squares in same diagonal (\)
    for (auto d : diag1) {
        // Number of points on diagonal x - y = d is n - abs(d)
        total -= (n - abs(d));
    }

    // Subtract all squares in same diagonal (/)
    for (auto d : diag2) {
        // Number of points on diagonal x + y = d is n - abs(n - d)
        total -= (n - abs(n - d));
    }

    // Now add back overcounted intersections

    // Rows and columns intersections
    for (auto r : rows) {
        for (auto c : cols) {
            ++total; // square (r,c) was subtracted twice
        }
    }

    // Row and diagonal1 intersections
    for (auto r : rows) {
        for (auto d : diag1) {
            ll c = r - d;
            if (c >= 1 && c <= n) {
                ++total;
            }
        }
    }

    // Row and diagonal2 intersections
    for (auto r : rows) {
        for (auto d : diag2) {
            ll c = d - r;
            if (c >= 1 && c <= n) {
                ++total;
            }
        }
    }

    // Column and diagonal1 intersections
    for (auto c : cols) {
        for (auto d : diag1) {
            ll r = d + c;
            if (r >= 1 && r <= n) {
                ++total;
            }
        }
    }

    // Column and diagonal2 intersections
    for (auto c : cols) {
        for (auto d : diag2) {
            ll r = d - c;
            if (r >= 1 && r <= n) {
                ++total;
            }
        }
    }

    // Diagonal1 and diagonal2 intersections
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 == 0) {
                ll r = (d1 + d2) / 2;
                ll c = d2 - r;
                if (r >= 1 && r <= n && c >= 1 && c <= n) {
                    ++total;
                }
            }
        }
    }

    // Subtract squares that are already occupied by pieces
    total -= m;

    cout << max(0LL, total) << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int NT = 1;
    // cin >> NT;
    while (NT--) solve();
    
    return 0;
}
