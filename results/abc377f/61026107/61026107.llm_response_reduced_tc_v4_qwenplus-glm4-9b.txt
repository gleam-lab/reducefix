#include <iostream>
#include <set>
#include <vector>

using namespace std;

int main() {
    long long n, m;
    cin >> n >> m;

    vector<pair<long long, long long>> a1, a2, a3, a4; // a1: rows, a2: cols, a3: a+b, a4: a-b
    long long all = n * n; // Total number of squares

    for (long long i = 1; i <= n; ++i) {
        a1.push_back({i, i}); // Row i
        a2.push_back({i, i}); // Column i
    }

    for (long long i = 1; i <= m; ++i) {
        long long x, y;
        cin >> x >> y;

        a1.emplace_back(x, x + y); // Row x + y
        a2.emplace_back(y, x + y); // Column x + y
        a3.emplace_back(x, y); // Diagonal a+b
        a4.emplace_back(x, n - y + 1); // Diagonal a-b
    }

    set<pair<long long, long long>> a1set, a2set, a3set, a4set;
    for (auto& p : a1) a1set.insert(p);
    for (auto& p : a2) a2set.insert(p);
    for (auto& p : a3) a3set.insert(p);
    for (auto& p : a4) a4set.insert(p);

    // Calculate the number of captures for each square
    for (long long i = 1; i <= n; ++i) {
        for (long long j = 1; j <= n; ++j) {
            long long captures = 0;
            if (a1set.count({i, i})) captures++; // Captured by row
            if (a2set.count({j, j})) captures++; // Captured by column
            if (a3set.count({i, j})) captures++; // Captured by diagonal a+b
            if (a4set.count({i, j})) captures++; // Captured by diagonal a-b

            all -= captures;
        }
    }

    cout << all << endl;
    return 0;
}

It iterates through all possible positions, checks for captures, and subtracts the number of captures for each position from the total number of squares. The use of sets ensures that the check for captures is efficient even when there are many pieces.
