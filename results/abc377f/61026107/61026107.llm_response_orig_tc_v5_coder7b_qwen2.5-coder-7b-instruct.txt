To solve this problem efficiently, we need to determine how many empty squares remain after placing pieces such that no piece is captured by another. The key observation here is that a piece at position \((i, j)\) captures all pieces in the same row \(i\), column \(j\), and diagonals \(i+j\) and \(i-j\).

Given these constraints, let's outline a step-by-step approach to solve the problem:

1. **Identify Captured Squares**: For each piece, identify all squares it can capture.
2. **Mark Captured Squares**: Use sets to keep track of all captured squares.
3. **Count Empty Squares**: Subtract the number of captured squares from the total number of squares to get the number of empty squares available.

struct Point {
    ll x, y;
    bool operator < (const Point& rhs) const {
        if (x == rhs.x) {
            return y < rhs.y;
        }
        return x < rhs.x;
    };
};

void solve() {
    ll n, m; cin >> n >> m;
    set<ll> a, b, c, d;
    for (int i = 1; i <= m; i++) {
        ll x, y; cin >> x >> y;
        a.insert(x);
        b.insert(y);
        c.insert(x - y);
        d.insert(x + y);
    }

    // Calculate the number of captured squares
    ll captured_squares = 0;
    for (auto v : a) captured_squares += n;
    for (auto v : b) captured_squares += n;
    for (auto v : c) captured_squares += n - abs(v);
    for (auto v : d) captured_squares += n - abs(n - v);

    // Deduplicate captured squares
    captured_squares -= a.size() * n;
    captured_squares -= b.size() * n;
    captured_squares -= c.size() * (n - abs(*c.begin()));
    captured_squares -= d.size() * (n - abs(n - *d.rbegin()));

    cout << n * n - captured_squares << "\n";
}
