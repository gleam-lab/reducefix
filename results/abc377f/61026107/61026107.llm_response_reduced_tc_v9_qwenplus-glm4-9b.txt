#include <iostream>
#include <set>
#include <vector>

using namespace std;

using ll = long long;

int main() {
    ll n, m;
    cin >> n >> m;

    vector<pair<ll, ll>> pieces(m);
    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        pieces[i] = {x, y};
    }

    // Sort pieces based on their coordinates
    sort(pieces.begin(), pieces.end());

    // Calculate the number of initially safe squares
    ll safe_squares = n * n - m;

    // Check horizontal and vertical lines
    set<ll> rows, cols;
    for (const auto& p : pieces) {
        rows.insert(p.first);
        cols.insert(p.second);
    }

    for (ll r : rows) {
        safe_squares -= (n - rows.count(r)); // Safe squares not in this row
    }

    for (ll c : cols) {
        safe_squares -= (n - cols.count(c)); // Safe squares not in this column
    }

    // Check diagonals
    set<ll> diag1, diag2;
    for (ll i = 1; i <= n; ++i) {
        diag1.insert(i - 1 + n - 1); // i-j = 1
        diag2.insert(i - 1 + i);      // i+j = 1
    }

    for (const auto& p : pieces) {
        diag1.erase(p.first - p.second + n - 1);
        diag2.erase(p.first + p.second - 1);
    }

    for (ll d : diag1) {
        safe_squares -= (n - diag1.count(d)); // Safe squares not in this diagonal
    }

    for (ll d : diag2) {
        safe_squares -= (n - diag2.count(d)); // Safe squares not in this diagonal
    }

    cout << safe_squares << endl;

    return 0;
}
