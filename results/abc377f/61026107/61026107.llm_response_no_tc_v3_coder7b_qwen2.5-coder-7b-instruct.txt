struct Point {
    ll x, y;
    bool operator < (const Point& rhs) const {
        if (x == rhs.x) {
            return y < rhs.y;
        }
        return x < rhs.x;
    };
};

void solve() {
    ll n, m; cin >> n >> m;
    set<ll> a, b, c, d;
    for (int i = 1; i <= m; i++) {
        ll x, y; cin >> x >> y;
        a.insert(x);
        b.insert(y);
        c.insert(x - y);
        d.insert(x + y);
    }

    // Calculate the total number of empty squares initially
    ll ans = n * n;

    // Subtract the squares occupied by pieces
    ans -= a.size();
    ans -= b.size();
    ans -= c.size();
    ans -= d.size();

    // Add back the squares that were counted multiple times
    for (auto v : a) {
        if (c.count(v - v)) ans += 1;
        if (d.count(v + v)) ans += 1;
    }
    for (auto v : b) {
        if (c.count(b - v)) ans += 1;
        if (d.count(b + v)) ans += 1;
    }
    for (auto v : c) {
        if (d.count(c + c)) ans += 1;
    }

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false), cin.tie(nullptr);
    cout.precision(12);

    return 0;

### Explanation of Fixes:
1. **Initialization of `ans`**: The initial count of empty squares should be `n * n`.
2. **Subtraction of Occupied Squares**: We subtract the sizes of sets `a`, `b`, `c`, and `d` which represent the unique rows, columns, diagonals, and anti-diagonals occupied by pieces.
3. **Add Back Overcounted Squares**: 
   - A piece at `(x, y)` can be counted multiple times if it occupies both a row and a diagonal (or column and anti-diagonal).
   - We add back these overcounted squares to get the correct count of empty squares.

This approach ensures that we correctly account for all possible placements of new pieces without capturing any existing ones.
