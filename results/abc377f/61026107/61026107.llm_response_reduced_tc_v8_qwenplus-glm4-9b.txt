
```cpp
#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

using ll = long long;

int main() {
    ll n, m;
    cin >> n >> m;
    
    vector<pair<ll, ll>> pieces;
    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        pieces.emplace_back(x, y);
    }
    
    sort(pieces.begin(), pieces.end());
    
    // Calculate the number of squares that are already occupied or cannot place a piece
    ll occupied = 0;
    
    // Check rows and columns
    for (ll x : pieces[0].first) {
        occupied++;
        if (x > 1) {
            occupied += x - 2;
        }
    }
    
    for (ll y : pieces[0].second) {
        occupied++;
        if (y > 1) {
            occupied += y - 2;
        }
    }
    
    // Check diagonals
    for (ll i = 1; i < m; ++i) {
        auto [x1, y1] = pieces[i - 1];
        auto [x2, y2] = pieces[i];
        
        if (x1 == x2) { // Same row
            occupied++;
        } else if (y1 == y2) { // Same column
            occupied++;
        } else if (x1 - y1 == x2 - y2) { // Same main diagonal
            occupied++;
        } else if (x1 + y1 == x2 + y2) { // Same secondary diagonal
            occupied++;
        }
    }
    
    cout << n * n - occupied << endl;
    
    return 0;
}
```

This code calculates the number of safe squares where a new piece can be placed by considering all the occupied squares (either directly by existing pieces or by the lines they form). It uses sorting to simplify the checking process and counts the occupied cells accordingly. After counting the occupied squares, it subtracts this number from the total number of squares in the grid to get the number of safe squares.
