#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Point {
    ll x, y;
    bool operator<(const Point& rhs) const {
        if (x == rhs.x) return y < rhs.y;
        return x < rhs.x;
    }
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: x - y, diag2: x + y

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    ll total_captured = 0;

    // Count cells on same row
    total_captured += rows.size() * n;

    // Count cells on same column
    total_captured += cols.size() * n;

    // Count cells on same diagonal1 (x - y)
    for (auto d : diag1) {
        ll count = min(d + n, n) - max(1LL, d + 1) + 1;
        total_captured += count;
    }

    // Count cells on same diagonal2 (x + y)
    for (auto d : diag2) {
        ll count;
        if (d <= n + 1) {
            count = min(d - 1, n);
        } else {
            count = 2 * n + 1 - d;
            count = min(count, n);
        }
        total_captured += count;
    }

    // Subtract overlaps (double counted cells)

    // Row-Column overlap
    for (auto r : rows) {
        for (auto c : cols) {
            total_captured--;
        }
    }

    // Row-Diagonal1 overlap
    for (auto r : rows) {
        for (auto d : diag1) {
            ll c = r - d;
            if (c >= 1 && c <= n) {
                total_captured--;
            }
        }
    }

    // Row-Diagonal2 overlap
    for (auto r : rows) {
        for (auto d : diag2) {
            ll c = d - r;
            if (c >= 1 && c <= n) {
                total_captured--;
            }
        }
    }

    // Column-Diagonal1 overlap
    for (auto c : cols) {
        for (auto d : diag1) {
            ll r = d + c;
            if (r >= 1 && r <= n) {
                total_captured--;
            }
        }
    }

    // Column-Diagonal2 overlap
    for (auto c : cols) {
        for (auto d : diag2) {
            ll r = d - c;
            if (r >= 1 && r <= n) {
                total_captured--;
            }
        }
    }

    // Diagonal1-Diagonal2 overlap
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            ll x = (d1 + d2) / 2;
            ll y = d2 - x;
            if ((d1 + d2) % 2 == 0 && x >= 1 && x <= n && y >= 1 && y <= n) {
                total_captured--;
            }
        }
    }

    // Subtract the M occupied squares themselves
    total_captured -= m;

    ll total_squares = n * n;
    ll ans = total_squares - total_captured;

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
