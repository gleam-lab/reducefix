```cpp
#include<bits/stdc++.h>

using namespace std;
using ll = long long;

#ifdef HOME
#include<debug.h>
#else
#define dump(...)
#endif

struct Point {
    ll x, y;
    bool operator < (const Point& rhs) const {
        if (x == rhs.x) {
            return y < rhs.y;
        }
        return x < rhs.x;
    };
};

void solve() {
    ll n, m; cin >> n >> m;
    set<ll> a, b, c, d;
    for (int i = 1; i <= m; i++) {
        ll x, y; cin >> x >> y;
        a.insert(x);
        b.insert(y);
        c.insert(x - y);
        d.insert(x + y);
    }
    ll ans = n * n;

    // Subtract the number of squares occupied by pieces along rows, columns, diagonals
    for (auto v : a) ans -= n;
    for (auto v : b) ans -= n;
    for (auto v : c) ans -= n - abs(v);
    for (auto v : d) ans -= n - abs(n - v);

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false), cin.tie(nullptr);
    cout.precision(12);
    int NT = 1;
    while(NT--) solve();
    return 0;
}
```

Explanation:
- The original solution was attempting to count the number of squares that could not be captured by adding back the overcounted squares. However, this approach led to complexity and incorrect results.
- The revised solution directly calculates the total number of squares (`n * n`) and subtracts the number of squares occupied by pieces along rows, columns, and diagonals. This avoids the need to overcompensate and provides a more straightforward calculation.
