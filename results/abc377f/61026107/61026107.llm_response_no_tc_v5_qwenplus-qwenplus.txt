#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Point {
    ll x, y;
    bool operator<(const Point& o) const {
        return x != o.x ? x < o.x : y < o.y;
    }
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2; // diag1: x - y, diag2: x + y

    vector<Point> pieces;

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
        pieces.push_back({x, y});
    }

    // Total squares under threat from row, col, and diagonals
    ll totalThreatened = 0;

    // Count unique threatened positions
    set<Point> threatened;

    // Threat from rows
    for (ll r : rows) {
        for (ll j = 1; j <= n; ++j) {
            threatened.insert({r, j});
        }
    }

    // Threat from columns
    for (ll c : cols) {
        for (ll i = 1; i <= n; ++i) {
            threatened.insert({i, c});
        }
    }

    // Threat from diagonal 1 (x - y = k)
    for (ll d : diag1) {
        for (ll x = 1; x <= n; ++x) {
            ll y = x - d;
            if (y >= 1 && y <= n) {
                threatened.insert({x, y});
            }
        }
    }

    // Threat from diagonal 2 (x + y = s)
    for (ll s : diag2) {
        for (ll x = max(1LL, s - n); x <= min(n, s - 1); ++x) {
            ll y = s - x;
            threatened.insert({x, y});
        }
    }

    // Now subtract the number of threatened squares
    ll totalEmpty = n * n - m;
    ll ans = totalEmpty - threatened.size();

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
