#include <iostream>
#include <vector>
#include <unordered_set>
using namespace std;

using ll = long long;

int main() {
    ll n, m, a, b;
    cin >> n >> m;

    vector<pair<ll, ll>> pieces(m);
    unordered_set<ll> captured_lines;

    for (ll i = 0; i < m; ++i) {
        cin >> a >> b;
        pieces.push_back({a, b});
        // Line capturing by row
        captured_lines.insert(a);
        // Line capturing by column
        captured_lines.insert(b);
        // Line capturing by diagonal (a+b)
        captured_lines.insert(a + b);
        // Line capturing by diagonal (a-b)
        captured_lines.insert(a - b);
    }

    // Calculate the number of free squares
    ll free_squares = n * n;

    // Subtract squares in captured lines
    for (ll line : captured_lines) {
        if (line > n) continue; // The line is out of bounds
        free_squares -= n; // Entire row or column is captured
    }

    // Check pairs of pieces for shared capture areas
    for (ll i = 0; i < m; ++i) {
        for (ll j = i + 1; j < m; ++j) {
            auto p1 = pieces[i];
            auto p2 = pieces[j];

            // Check if they are in the same row or column
            if (p1.first == p2.first) {
                free_squares--;
                continue;
            }
            if (p1.second == p2.second) {
                free_squares--;
                continue;
            }

            // Check for diagonal captures
            if (p1.first + p1.second == p2.first + p2.second) {
                free_squares--;
            }
            if (p1.first - p1.second == p2.first - p2.second) {
                free_squares--;
            }
        }
    }

    cout << free_squares << endl;
    return 0;
}
