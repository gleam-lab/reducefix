#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Point {
    ll x, y;
    bool operator<(const Point& rhs) const {
        if (x == rhs.x) return y < rhs.y;
        return x < rhs.x;
    }
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    // Total squares = total - attacked + occupied
    ll total = n * n;
    ll attacked = 0;

    attacked += rows.size() * n;
    attacked += cols.size() * n;

    for (auto d : diag1) {
        ll count = 0;
        if (d >= 0) {
            count = n - d;
        } else {
            count = n + d;
        }
        attacked += count;
    }

    for (auto d : diag2) {
        ll count = 0;
        if (d > n) {
            count = 2 * n - d;
        } else {
            count = d - 1;
        }
        attacked += count;
    }

    // Inclusion-Exclusion corrections

    // Intersections of rows and columns
    attacked -= rows.size() * cols.size();

    // Intersections of row and diag1
    for (auto r : rows) {
        for (auto d : diag1) {
            ll c = r - d;
            if (c >= 1 && c <= n)
                attacked -= 1;
        }
    }

    // Intersections of row and diag2
    for (auto r : rows) {
        for (auto d : diag2) {
            ll c = d - r;
            if (c >= 1 && c <= n)
                attacked -= 1;
        }
    }

    // Intersections of col and diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            ll r = d + c;
            if (r >= 1 && r <= n)
                attacked -= 1;
        }
    }

    // Intersections of col and diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            ll r = d - c;
            if (r >= 1 && r <= n)
                attacked -= 1;
        }
    }

    // Intersections of diag1 and diag2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 != 0) continue;
            ll r = (d2 + d1) / 2;
            ll c = (d2 - d1) / 2;
            if (r >= 1 && r <= n && c >= 1 && c <= n)
                attacked -= 1;
        }
    }

    // Subtract already occupied positions
    ll occupied = m;

    ll result = total - attacked + occupied;
    cout << result << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
