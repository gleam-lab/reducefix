#include <bits/stdc++.h>

using namespace std;
using ll = long long;

struct Point {
    ll x, y;
};

void solve() {
    ll n, m;
    cin >> n >> m;

    set<pair<ll, ll>> pieces;
    set<ll> row, col, diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        pieces.insert({x, y});
        row.insert(x);
        col.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    // Total number of squares
    ll total = n * n;

    // Subtract all squares attacked by existing pieces
    ll attacked = 0;

    // Rows and columns
    attacked += (ll)row.size() * n;
    attacked += (ll)col.size() * n;

    // Diagonals: x - y = constant
    for (auto& d : diag1) {
        ll count = 0;
        if (d >= 0) {
            count = min(n, n - d);
        } else {
            count = min(n, n + d);
        }
        attacked += count;
    }

    // Anti-diagonals: x + y = constant
    for (auto& d : diag2) {
        ll count = 0;
        if (d <= n) {
            count = d;
            if (count > n) count = n;
        } else {
            count = 2 * n - d;
        }
        attacked += max(0LL, count);
    }

    // Add back over-counted cells (intersections of rows)
    for (auto r : row) {
        for (auto c : col) {
            if (pieces.count({r, c}) == 0) {
                attacked--;
            }
        }
    }

    // Intersections involving diagonals
    auto add_if_in_bounds = [&](ll x, ll y) {
        if (x >= 1 && x <= n && y >= 1 && y <= n) {
            if (pieces.count({x, y}) == 0) {
                attacked--;
            }
        }
    };

    // Row-diag1 intersections
    for (auto r : row) {
        for (auto d : diag1) {
            ll c = r - d;
            add_if_in_bounds(r, c);
        }
    }

    // Row-diag2 intersections
    for (auto r : row) {
        for (auto d : diag2) {
            ll c = d - r;
            add_if_in_bounds(r, c);
        }
    }

    // Col-diag1 intersections
    for (auto c : col) {
        for (auto d : diag1) {
            ll r = d + c;
            add_if_in_bounds(r, c);
        }
    }

    // Col-diag2 intersections
    for (auto c : col) {
        for (auto d : diag2) {
            ll r = d - c;
            add_if_in_bounds(r, c);
        }
    }

    // Diag1-diag2 intersections
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 == 0) {
                ll x = (d2 + d1) / 2;
                ll y = (d2 - d1) / 2;
                add_if_in_bounds(x, y);
            }
        }
    }

    // Total empty non-attacked cells
    ll empty = n * n - m;
    ll safe = empty - attacked;

    cout << safe << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();
    return 0;
}
