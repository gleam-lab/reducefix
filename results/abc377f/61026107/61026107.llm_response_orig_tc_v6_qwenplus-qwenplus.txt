#include <bits/stdc++.h>

using namespace std;
using ll = long long;

struct Point {
    ll x, y;
};

void solve() {
    ll N, M;
    cin >> N >> M;
    set<ll> rows, cols, diag1, diag2;
    vector<Point> pieces;

    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        pieces.push_back({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total squares attacked by existing pieces
    set<pair<ll, ll>> attacked;

    // For each piece, mark all squares in its row, column, and diagonals as attacked
    for (const auto& p : pieces) {
        ll x = p.x, y = p.y;

        // Row
        for (ll j = 1; j <= N; ++j)
            attacked.insert({x, j});

        // Column
        for (ll i = 1; i <= N; ++i)
            attacked.insert({i, y});

        // Diagonal 1 (x - y = const)
        ll d1 = x - y;
        for (ll i = max(1LL, d1 + 1), j = max(1LL, 1 - d1); i <= N && j <= N; ++i, ++j)
            attacked.insert({i, j});
        for (ll i = min(N, x), j = min(N, y); i >= 1 && j >= 1; --i, --j)
            attacked.insert({i, j});

        // Diagonal 2 (x + y = const)
        ll d2 = x + y;
        for (ll i = max(1LL, d2 - N), j = min(N, d2 - 1); i <= N && j >= 1; ++i, --j)
            attacked.insert({i, j});
        for (ll i = min(N, x), j = max(1LL, d2 - N); i >= 1 && j <= N; --i, ++j)
            attacked.insert({i, j});
    }

    // Count total empty safe squares
    ll safe = 0;
    for (ll i = 1; i <= N; ++i) {
        for (ll j = 1; j <= N; ++j) {
            if (attacked.find({i, j}) == attacked.end()) {
                safe++;
            }
        }
    }

    cout << safe << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
