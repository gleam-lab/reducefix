#include<bits/stdc++.h>

using namespace std;
using ll = long long;

#ifdef HOME
#include<debug.h>

struct Point {
    ll x, y;
    bool operator < (const Point& rhs) const {
        if (x == rhs.x) {
            return y < rhs.y;
        }
        return x < rhs.x;
    };
};

void solve() {
    ll n, m; cin >> n >> m;
    set<ll> a, b, c, d;
    for (int i = 1; i <= m; i++) {
        ll x, y; cin >> x >> y;
        a.insert(x);
        b.insert(y);
        c.insert(x - y);
        d.insert(x + y);
    }

    // Calculate initial count of all possible positions
    ll ans = n * n;

    // Subtract positions where pieces can capture
    for (auto v : a) ans -= n - 1; // Horizontal captures
    for (auto v : b) ans -= n - 1; // Vertical captures
    for (auto v : c) ans -= max(0LL, n - abs(v)); // Diagonal captures along x-y line
    for (auto v : d) ans -= max(0LL, n - abs(v)); // Diagonal captures along x+y line

    // Adjust for over-subtraction due to overlap
    for (auto v1 : a) {
        for (auto v2 : b) {
            int x = v1;
            int y = v2;
            if (c.count(x - y) && d.count(x + y)) {
                ans += 1; // Add back subtracted twice
            } else if (c.count(x - y) || d.count(x + y)) {
                ans += 1; // Add back subtracted once
            }
        }
    }

    for (auto v1 : a) {
        for (auto v2 : c) {
            ll x = v1;
            ll y = v1 - v2;
            if (x < 1 || x > n || y < 1 || y > n) continue;
            if (d.count(x + y)) {
                ans -= 1; // Subtract once if already counted
            }
        }
    }

    for (auto v1 : a) {
        for (auto v2 : d) {
            ll x = v1;
            ll y = v2 - v1;
            if (x < 1 || x > n || y < 1 || y > n) continue;
        }
    }

    for (auto v1 : b) {
        for (auto v2 : c) {
            ll x = v1 + v2;
            ll y = v1;
            if (x < 1 || x > n || y < 1 || y > n) continue;
            if (d.count(x + y)) {
                ans -= 1; // Subtract once if already counted
            }
        }
    }

    for (auto v1 : b) {
        for (auto v2 : d) {
            ll x = v2 - v1;
            ll y = v1;
            if (x < 1 || x > n || y < 1 || y > n) continue;
        }
    }

    for (auto v1 : c) {
        for (auto v2 : d) {
            if ((v1 + v2) % 2 != 0) continue;
            ll x = (v1 + v2) / 2;
            ll y = v2 - x;
            if (x < 1 || x > n || y < 1 || y > n) continue;
        }
    }

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false), cin.tie(nullptr);
    cout.precision(12);
    int NT = 1;
    // cin >> NT;
    while(NT--) solve();
    return 0;
}

### Explanation:
1. **Initialization**: We initialize sets `a`, `b`, `c`, and `d` to store the unique rows, columns, and diagonals occupied by the pieces.
2. **Counting Initial Positions**: We calculate the total number of positions as \(N \times N\).
3. **Subtracting Capturable Positions**: For each type of capture (horizontal, vertical, diagonal along both lines), we subtract the number of positions that can be captured.
4. **Adjusting Over-Subtraction**: We adjust for positions that were subtracted multiple times due to overlaps.
5. **Output**: Finally, we print the result which represents the number of empty squares that cannot be captured by any existing pieces.

This solution ensures that we correctly account for all possible captures and avoid double-counting, providing an efficient and accurate answer within the constraints.
