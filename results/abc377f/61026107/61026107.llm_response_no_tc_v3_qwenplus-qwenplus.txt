#include <bits/stdc++.h>

using namespace std;
using ll = long long;

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2;  // diag1: x - y, diag2: x + y

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    ll total_captured = 0;

    // Count squares captured by row, column or diagonal
    set<pair<ll, ll>> captured;

    // Rows
    for (auto r : rows) {
        for (ll c = 1; c <= n; ++c) {
            captured.insert({r, c});
        }
    }

    // Columns
    for (auto c : cols) {
        for (ll r = 1; r <= n; ++r) {
            if (!captured.count({r, c})) {
                captured.insert({r, c});
            }
        }
    }

    // Diag1 (x - y)
    for (auto d : diag1) {
        // All (x, y) such that x - y = d => y = x - d
        ll min_x = max(1LL, d + 1);
        ll max_x = min(n, n + d);
        for (ll x = min_x; x <= max_x; ++x) {
            ll y = x - d;
            if (y >= 1 && y <= n && !captured.count({x, y})) {
                captured.insert({x, y});
            }
        }
    }

    // Diag2 (x + y)
    for (auto d : diag2) {
        // All (x, y) such that x + y = d => y = d - x
        ll min_x = max(1LL, d - n);
        ll max_x = min(n, d - 1);
        for (ll x = min_x; x <= max_x; ++x) {
            ll y = d - x;
            if (y >= 1 && y <= n && !captured.count({x, y})) {
                captured.insert({x, y});
            }
        }
    }

    // Total empty squares not captured
    ll answer = n * n - captured.size();
    cout << answer << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
