struct Point {
    ll x, y;
    bool operator < (const Point& rhs) const {
        if (x == rhs.x) {
            return y < rhs.y;
        }
        return x < rhs.x;
    };
};

void solve() {
    ll n, m; cin >> n >> m;
    set<ll> a, b, c, d;
    for (int i = 1; i <= m; i++) {
        ll x, y; cin >> x >> y;
        a.insert(x);
        b.insert(y);
        c.insert(x - y);
        d.insert(x + y);
    }

    // Calculate the total number of empty squares
    ll ans = n * n;

    // Subtract the number of squares occupied by pieces
    ans -= a.size();
    ans -= b.size();
    ans -= c.size();
    ans -= d.size();

    // Add back the over-subtracted corners
    for (auto v1 : a) {
        for (auto v2 : b) {
            int x = v1;
            int y = v2;
            if (c.count(x - y) && d.count(x + y)) {
                ans += 1;
            } else if (c.count(x - y) || d.count(x + y)) {
                ans += 1;
            }
        }
    }

    for (auto v1 : a) {
        for (auto v2 : c) {
            ll x = v1;
            ll y = v1 - v2;
            if (x < 1 || x > n || y < 1 || y > n) continue;
            if (d.count(x + y)) {
                ans--;
            }
        }
    }

    for (auto v1 : a) {
        for (auto v2 : d) {
            ll x = v1;

            if (x < 1 || x > n || y < 1 || y > n) continue;
        }
    }

            ll x = v1 + v2;
            ll y = v1;
            if (x < 1 || x > n || y < 1 || y > n) continue;
            if (d.count(x + y)) {
                ans--;
            }
        }
    }
