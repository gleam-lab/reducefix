#include <bits/stdc++.h>

using namespace std;
using ll = long long;

void solve() {
    ll n, m;
    cin >> n >> m;

    set<ll> rows, cols;
    set<ll> diag1, diag2;  // diag1: x - y, diag2: x + y

    for (ll i = 0; i < m; ++i) {
        ll x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x - y);
        diag2.insert(x + y);
    }

    ll totalAttacked = 0;

    // Count squares attacked by row, column, diagonal 1, diagonal 2
    totalAttacked += rows.size() * n;
    totalAttacked += cols.size() * n;
    for (ll d : diag1) {
        ll count = min(d + n, n - max(1LL, d + 1)) + 1;
        if (-n < d && d < n) {
            count = n - abs(d);
        } else {
            count = abs(n - abs(d));
        }
        totalAttacked += count;
    }
    for (ll d : diag2) {
        ll count = min(d, 2 * n + 1 - d);
        if (1 < d && d < n + n) {
            count = n - abs(d - n + 1);
        } else {
            count = 1;
        }
        totalAttacked += count;
    }

    // Subtract overlaps
    set<pair<ll, ll>> pointSet;
    for (ll r : rows) {
        for (ll c : cols) {
            pointSet.emplace(r, c);
        }
    }
    for (ll r : rows) {
        for (ll d : diag1) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                pointSet.emplace(r, c);
            }
        }
    }
    for (ll r : rows) {
        for (ll d : diag2) {
            ll c = d - r;
            if (1 <= c && c <= n) {
                pointSet.emplace(r, c);
            }
        }
    }
    for (ll c : cols) {
        for (ll d : diag1) {
            ll r = d + c;
            if (1 <= r && r <= n) {
                pointSet.emplace(r, c);
            }
        }
    }
    for (ll c : cols) {
        for (ll d : diag2) {
            ll r = d - c;
            if (1 <= r && r <= n) {
                pointSet.emplace(r, c);
            }
        }
    }
    for (ll d1 : diag1) {
        for (ll d2 : diag2) {
            ll x = (d1 + d2) / 2;
            ll y = d2 - x;
            if (x >= 1 && x <= n && y >= 1 && y <= n && (d1 == x - y) && (d2 == x + y)) {
                pointSet.emplace(x, y);
            }
        }
    }

    ll ans = n * n - totalAttacked + pointSet.size();

    cout << ans << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    solve();
    return 0;
}
