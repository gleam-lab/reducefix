#include<bits/stdc++.h>
#define int long long
#define PII pair<int,int>
#define inf 1e9+7

using namespace std;

int qpowm(int a, int b, int m) {
	int r = 1;
	while (b) {
		if (b & 1)(r *= a) %= m;
		(a *= a) %= m, b >>= 1;
	}
	return r;
}

int qpow(int a, int b) {
	int r = 1;
	while (b) {
		if (b & 1)r *= a;
		a *= a;
		b >>= 1;
	}
	return r;
}

void solve() {
	int n, q;
	cin >> n >> q;
	set<int>h, v, d1, d2;//d1 存 i+j=d 的对角线 d2 存i-j=d的对角线

	for (int i = 1, x, y; i <= q; i++) {
		cin >> x >> y;
		h.insert(x);
		v.insert(y);
		d1.insert(x + y);
		d2.insert(x - y);
	}

	int ans = n * n - h.size() - v.size();

	for (auto d : d1) {
		//i+j=d 的对角线
		int cnt = 0;
		for (auto x : h) {
			if (1 <= d - x && d - x <= n) {
				cnt++;
			}
		}
		ans -= min(cnt, d1.count(d));
	}

	for (auto d : d2) {
		//i-j=d 的对角线
		int cnt = 0;
		for (auto x : h) {
			if (1 <= x - d && x - d <= n) {
				cnt++;
			}
		}
		ans -= min(cnt, d2.count(d));
	}

	cout << ans << endl;
}

signed main() {

	int _ = 1;
//	cin>>_;
	while (_--) {
		solve();
	}

	return 0;
}
