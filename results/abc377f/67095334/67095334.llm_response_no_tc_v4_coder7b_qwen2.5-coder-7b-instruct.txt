#include<bits/stdc++.h>
#define int long long
#define PII pair<int,int>
#define inf 1e9+7

using namespace std;

int qpowm(int a, int b, int m) {
	int r = 1;
	while (b) {
		if (b & 1)(r *= a) %= m;
		(a *= a) %= m, b >>= 1;
	}
	return r;
}

int qpow(int a, int b) {
	int r = 1;
	while (b) {
		if (b & 1)r *= a;
		a *= a;
		b >>= 1;
	}
	return r;
}

void solve() {
	int n, q;
	cin >> n >> q;
	vector<PII>pieces(q);
	for (int i = 0; i < q; i++) {
		cin >> pieces[i].first >> pieces[i].second;
	}

	set<int>rows, cols, diag1, diag2;

	for (const auto& p : pieces) {
		rows.insert(p.first);
		cols.insert(p.second);
		diag1.insert(p.first + p.second);
		diag2.insert(p.first - p.second);
	}

	int ans = (n - rows.size()) * (n - cols.size());

	for (const auto& d : diag1) {
		set<int>coveredRows, coveredCols;
		for (const auto& r : rows) {
			if (1 <= d - r && d - r <= n) {
				coveredRows.insert(r);
			}
		}
		for (const auto& c : cols) {
			if (1 <= d - c && d - c <= n) {
				coveredCols.insert(c);
			}
		}
		int validCells = min(n, d) - max(1LL, d - n) + 1;
		validCells -= coveredRows.size() + coveredCols.size() - coveredRows.size() * coveredCols.size();
		ans += validCells;
	}

	for (const auto& d : diag2) {
		set<int>coveredRows, coveredCols;
		for (const auto& r : rows) {
			if (1 <= r - d && r - d <= n) {
				coveredRows.insert(r);
			}
		}
		for (const auto& c : cols) {
			if (1 <= d + c && d + c <= n) {
				coveredCols.insert(c);
			}
		}
		int validCells = min(n, abs(d)) - max(1LL, abs(d) - n) + 1;
		validCells -= coveredRows.size() + coveredCols.size() - coveredRows.size() * coveredCols.size();
		ans += validCells;
	}

	cout << ans << endl;
}

signed main() {
	int _ = 1;
//	cin>>_;
	while (_) {
		solve();
	}
	return 0;
}
