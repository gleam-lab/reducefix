#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols;
    set<int> diag1, diag2;  // diag1: i+j, diag2: i-j

    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x + y);
        diag2.insert(x - y);
    }

    // Total safe cells = total not in row or col
    int ans = (n - rows.size()) * (n - cols.size());

    // Subtract cells that are on diagonals with pieces but already removed by row/col
    for (int d : diag1) {
        int count = 0;
        // Count how many of these intersections are already excluded by row or col
        for (int r : rows) {
            int c = d - r;
            if (c >= 1 && c <= n)
                count++;
        }
        for (int c : cols) {
            int r = d - c;
            if (r >= 1 && r <= n)
                count++;
        }
        ans -= count;
    }

    for (int d : diag2) {
        int count = 0;
        // Count how many of these intersections are already excluded by row or col
        for (int r : rows) {
            int c = r - d;
            if (c >= 1 && c <= n)
                count++;
        }
        for (int c : cols) {
            int r = d + c;
            if (r >= 1 && r <= n)
                count++;
        }
        ans -= count;
    }

    cout << ans << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    return 0;
}
