#include <bits/stdc++.h>
#define int long long
#define PII pair<int, int>

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols;
    set<int> diag1, diag2; // diag1: i+j, diag2: i-j

    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x + y);
        diag2.insert(x - y);
    }

    // Total empty squares not attacked by row or column
    int ans = (n - rows.size()) * (n - cols.size());

    // Subtract the cells on diagonal lines that are still in the valid rows/cols
    auto count_intersection = [&](const set<int>& lines, function<int(int)> get_len, function<bool(int, int)> is_valid) {
        for (int d : lines) {
            set<int> intersected;

            // Check intersections with rows
            for (int r : rows) {
                int c;
                if (is_valid(r, d)) {
                    c = (d - r); // for diag1: r + c = d => c = d - r
                    if (1 <= c && c <= n)
                        intersected.insert(r);
                }
            }

            // Check intersections with columns
            for (int c : cols) {
                int r;
                if (is_valid(c, d)) {
                    r = (d - c); // for diag1: r + c = d => r = d - c
                    if (1 <= r && r <= n)
                        intersected.insert(r);
                }
            }

            int len = get_len(d);
            ans -= (len - intersected.size());
        }
    };

    // For diag1: i+j = d
    count_intersection(diag1,
                       [&](int d) {
                           if (d <= n + 1) return d - 1;
                           else return 2 * n - d + 1;
                       },
                       [&](int x, int d) {
                           return (1 <= d - x && d - x <= n);
                       });

    // For diag2: i-j = d
    count_intersection(diag2,
                       [&](int d) {
                           if (d >= 0) return n - d;
                           else return n + d;
                       },
                       [&](int x, int d) {
                           return (1 <= x + d && x + d <= n);
                       });

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
}
