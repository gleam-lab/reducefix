#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int N, M;
    cin >> N >> M;

    set<int> rows, cols;
    set<int> diag1, diag2;  // i+j and i-j

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // First subtract all squares attacked by existing pieces
    int total_attacked = 0;

    // For each row with a piece, it attacks N columns
    total_attacked += rows.size() * N;

    // For each column with a piece, it attacks N - |rows| squares (excluding already counted row attacks)
    total_attacked += cols.size() * (N - rows.size());

    // Process diagonal 1: i + j = d
    for (int d : diag1) {
        // Calculate number of cells on this diagonal
        int cnt = 0;
        if (d >= 1 && d <= N+1) {
            cnt = d - 1;
        } else if (d > N+1 && d <= 2*N) {
            cnt = 2*N + 1 - d;
        }

        // Subtract the cells that are already counted in rows or cols
        int already_counted = 0;
        for (int r : rows) {
            int c = d - r;
            if (c >= 1 && c <= N) {
                already_counted++;
            }
        }
        for (int c : cols) {
            int r = d - c;
            if (r >= 1 && r <= N && rows.find(r) == rows.end()) {
                already_counted++;
            }
        }

        total_attacked += max(0LL, cnt - already_counted);
    }

    // Process diagonal 2: i - j = d
    for (int d : diag2) {
        // Calculate number of cells on this diagonal
        int cnt = 0;
        if (d >= -N+1 && d <= -1) {
            cnt = N + d;
        } else if (d >= 0 && d <= N-1) {
            cnt = N - d;
        }

        // Subtract the cells that are already counted in rows or cols
        int already_counted = 0;
        for (int r : rows) {
            int c = r - d;
            if (c >= 1 && c <= N) {
                already_counted++;
            }
        }
        for (int c : cols) {
            int r = d + c;
            if (r >= 1 && r <= N && rows.find(r) == rows.end()) {
                already_counted++;
            }
        }

        total_attacked += max(0LL, cnt - already_counted);
    }

    // Total available positions = total squares - occupied - attacked
    int total_squares = N * N;
    int occupied = M;
    int result = total_squares - occupied - total_attacked;
    cout << result << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    solve();
    return 0;
}
