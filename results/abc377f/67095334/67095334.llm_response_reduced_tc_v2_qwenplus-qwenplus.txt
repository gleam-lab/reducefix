#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols, diag1, diag2;

    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        rows.insert(x);
        cols.insert(y);
        diag1.insert(x + y);
        diag2.insert(x - y);
    }

    // Total safe cells = total cells not in any attacked row or column
    int ans = (n - rows.size()) * (n - cols.size());

    // Subtract overcounted cells on diagonals that intersect both unsafe rows and columns

    // For diagonal x + y = d (i.e., diag1)
    for (int d : diag1) {
        // Count how many rows intersect this diagonal within bounds
        int cnt = 0;
        for (int r : rows) {
            int c = d - r;
            if (1 <= c && c <= n) {
                cnt++;
            }
        }

        // Count how many cols intersect this diagonal within bounds
        for (int c : cols) {
            int r = d - c;
            if (1 <= r && r <= n && rows.count(r) == 0) {
                cnt++;
            }
        }

        // Calculate number of points on the diagonal
        int len;
        if (d <= n + 1) {
            len = d - 1;
        } else {
            len = 2 * n + 1 - d;
        }

        ans -= max(0LL, len - cnt);
    }

    // For diagonal x - y = d (i.e., diag2)
    for (int d : diag2) {
        // Count how many rows intersect this diagonal within bounds
        int cnt = 0;
        for (int r : rows) {
            int c = r - d;
            if (1 <= c && c <= n) {
                cnt++;
            }
        }

        // Count how many cols intersect this diagonal within bounds
        for (int c : cols) {
            int r = c + d;
            if (1 <= r && r <= n && rows.count(r) == 0) {
                cnt++;
            }
        }

        // Calculate number of points on the diagonal
        int len;
        if (d >= 0) {
            len = n - d;
        } else {
            len = n + d;
        }

        ans -= max(0LL, len - cnt);
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int t = 1;
    while (t--) {
        solve();
    }

    return 0;
}
