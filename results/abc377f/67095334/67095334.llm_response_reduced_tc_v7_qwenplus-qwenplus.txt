#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols;
    set<int> diag1, diag2; // i+j and i-j respectively

    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total empty squares not on any attacked row or column
    int ans = (n - rows.size()) * (n - cols.size());

    // Subtract the intersections with diagonals that are not already excluded by rows/cols
    auto process_diag1 = [&](int d) {
        // For i+j=d, valid range is from max(1, d-n) to min(n, d-1)
        int low = max(1LL, d - n);
        int high = min(n, d - 1);
        if (low > high) return;

        // Count how many of these are NOT in rows
        int total = high - low + 1;

        // Now subtract those whose row is not eliminated but column is eliminated
        for (auto x : rows) {
            int y = d - x;
            if (1 <= y && y <= n && cols.count(y)) {
                --total;
            }
        }

        ans -= total;
    };

    auto process_diag2 = [&](int d) {
        // For i-j=d => j = i-d
        // Valid i must satisfy 1 <= i <= n and 1 <= i-d <= n
        // => max(1, d+1) <= i <= min(n, d+n)
        int low = max(1LL, d + 1);
        int high = min(n, n + d);
        if (low > high) return;

        int total = high - low + 1;

        for (auto x : rows) {
            int y = x - d;
            if (1 <= y && y <= n && cols.count(y)) {
                --total;
            }
        }

        ans -= total;
    };

    // Process all affected diagonals
    for (int d : diag1) process_diag1(d);
    for (int d : diag2) process_diag2(d);

    cout << ans << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();
}
