#include<bits/stdc++.h>
#define long long long
using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;
    vector<pair<long long, long long>> pieces(M);

    for (long long i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    set<long long> rows, cols, diag1, diag2;
    for (auto &p : pieces) {
        rows.insert(p.first);
        cols.insert(p.second);
        diag1.insert(p.first + p.second);
        diag2.insert(p.first - p.second);
    }

    // Calculate number of free rows and columns
    long long free_rows = N - rows.size();
    long long free_cols = N - cols.size();

    // Calculate number of free diagonals
    long long free_diag1 = 0;
    long long free_diag2 = 0;
    for (long long d = 1; d <= 2 * N; ++d) {
        set<long long> s;
        for (auto r : rows) {
            if (r + d > 0 && r + d <= N) {
                s.insert(r + d);
            }
        }
        for (auto c : cols) {
            if (c + d > 0 && c + d <= N) {
                s.insert(c + d);
            }
            if (d - c > 0 && d - c <= N) {
                s.insert(d - c);
            }
        }
        if (d + 1 > 0 && d + 1 <= 2 * N) {
            for (auto d1 : diag1) {
                if (d1 >= d && d1 <= d + N) {
                    s.insert(d1);
                }
            }
            for (auto d2 : diag2) {
                if (d2 >= d && d2 <= d + N) {
                    s.insert(d2);
                }
            }
        }
        free_diag1 += N - s.size();
    }

    // Calculate number of free anti-diagonals
    long long free_anti_diag1 = 0;
    long long free_anti_diag2 = 0;
    for (long long d = -N + 1; d <= N; ++d) {
        set<long long> s;
        for (auto r : rows) {
            if (r - d >= 1 && r - d <= N) {
                s.insert(r - d);
            }
        }
        for (auto c : cols) {
            if (c + d >= 1 && c + d <= N) {
                s.insert(c + d);
            }
            if (c - d >= 1 && c - d <= N) {
                s.insert(c - d);
            }
        }
        if (d + 1 >= -N && d + 1 <= N) {
            for (auto d1 : diag1) {
                if (d1 >= -d && d1 <= N - d) {
                    s.insert(d1);
                }
            }
            for (auto d2 : diag2) {
                if (d2 >= -d && d2 <= N - d) {
                    s.insert(d2);
                }
            }
        }
        free_anti_diag1 += N - s.size();
    }

    // Total free squares
    long long total_free = N * N - free_rows * N - free_cols * N + free_diag1 + free_diag2 + free_anti_diag1 + free_anti_diag2;

    cout << total_free << endl;

    return 0;
}
