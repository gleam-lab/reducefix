#include<bits/stdc++.h>
#define int long long
#define PII pair<int,int>

using namespace std;

const int MAXN=1e9+7;

int n,m;
set<int>row,col,dia1,dia2;

void add(int x,int y){
    row.insert(x);
    col.insert(y);
    dia1.insert(x+y);
    dia2.insert(x-y);
}

bool check(int x,int y){
    return !row.count(x)&&!col.count(y)&&!dia1.count(x+y)&&!dia2.count(x-y);
}

int solve(){
    cin>>n>>m;
    for(int i=1;i<=m;++i){
        int x,y;
        cin>>x>>y;
        add(x,y);
    }
    int ans=n*(n-row.size())-(n-col.size());
    for(auto d:dia1){
        set<int>tmp;
        for(auto x:row){
            if(1<=d-x&&d-x<=n)tmp.insert(x);
        }
        for(auto y:col){
            if(1<=d-y&&d-y<=n)tmp.insert(d-y);
        }
        if(d<=n+1){
            ans-=max(0LL,n-d-sz(tmp));
        }else{
            ans-=max(0LL,d-n-sz(tmp));
        }
    }
    for(auto d:dia2){
        set<int>tmp;
        for(auto x:row){
            if(1<=x-d&&x-d<=n)tmp.insert(x);
        }
        for(auto y:col){
            if(1<=d+y&&d+y<=n)tmp.insert(d+y);
        }
        if(d<=0){
            ans-=max(0LL,n+1+d-sz(tmp));
        }else{
            ans-=max(0LL,n-d+1-sz(tmp));
        }
    }
    return ans;
}

signed main(){
    ios::sync_with_stdio(false);cin.tie(nullptr);cout.tie(nullptr);
    cout<<solve()<<endl;
    return 0;
}
