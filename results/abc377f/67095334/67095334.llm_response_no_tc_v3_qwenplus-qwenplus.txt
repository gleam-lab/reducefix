#include <bits/stdc++.h>
#define int long long

using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;

    set<int> rows, cols;
    set<int> diag1, diag2; // diag1: i+j, diag2: i-j

    for (int i = 0; i < m; ++i) {
        int r, c;
        cin >> r >> c;
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r + c);
        diag2.insert(r - c);
    }

    // Total safe cells = total cells - attacked cells
    // First calculate all cells not in any occupied row or column
    int safe_cells = (n - rows.size()) * (n - cols.size());

    // Now subtract the cells that are still attacked by pieces on diagonals
    // Each diagonal may intersect some of these safe rectangles

    // Store already excluded cells to avoid double-counting
    set<pair<int, int>> excluded;

    for (int k = 0; k < m; ++k) {
        int r, c;
        cin >> r >> c; // We need to re-read the inputs to process diagonals
        
        // Check all four lines this piece attacks

        // Row r
        if (!rows.count(r) && !cols.count(c)) {
            excluded.insert({r, c});
        }
        
        // Column c
        if (!rows.count(r) && !cols.count(c)) {
            excluded.insert({r, c});
        }
        
        // Diagonal 1: i+j = r+c
        int len_d1;
        if (r + c <= n + 1) {
            len_d1 = r + c - 1;
        } else {
            len_d1 = 2 * n - (r + c) + 1;
        }
        
        // Diagonal 2: i-j = r-c
        int len_d2 = n - abs(r - c);
        
        // We'll subtract these later from our safe_cells
        // But we need to track which ones we've already counted
    }

    // Instead of tracking intersections with rows/columns,
    // use a more direct approach:
    
    // For each diagonal line, count how many of its cells are in the safe rectangle
    
    // Process diag1: i+j=d
    for (int d : diag1) {
        // The number of cells on this diagonal is:
        int start_i = max(1LL, d - n);
        int end_i = min(n, d - 1);
        int diag_cells = end_i - start_i + 1;
        
        // These cells are in rows [start_i, end_i] and columns [d-end_i, d-start_i]
        
        // Subtract the number of these cells that are NOT in any occupied row or column
        int attacked_cells = 0;
        for (int i = start_i; i <= end_i; ++i) {
            int j = d - i;
            if (!rows.count(i) && !cols.count(j)) {
                attacked_cells++;
            }
        }
        
        safe_cells -= attacked_cells;
    }

    // Process diag2: i-j=d
    for (int d : diag2) {
        // The number of cells on this diagonal is:
        int start_i = max(1LL, d + 1);
        int end_i = min(n, n + d);
        int diag_cells = end_i - start_i + 1;
        
        // These cells are in rows [start_i, end_i] and columns [i-d for i in start_i..end_i]
        
        // Subtract the number of these cells that are NOT in any occupied row or column
        int attacked_cells = 0;
        for (int i = start_i; i <= end_i; ++i) {
            int j = i - d;
            if (!rows.count(i) && !cols.count(j)) {
                attacked_cells++;
            }
        }
        
        safe_cells -= attacked_cells;
    }

    cout << safe_cells << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    solve();
    
    return 0;
}
