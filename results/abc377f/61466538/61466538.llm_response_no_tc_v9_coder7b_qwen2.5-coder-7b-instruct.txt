#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

#define all(x) (x).begin(), (x).end()

// Function to check if a position is within the bounds of the board
bool inBounds(const pii &pos, const ll n) {
    return 1 <= pos.first && pos.first <= n && 1 <= pos.second && pos.second <= n;
}

// Main function to solve the problem
void solve() {
    ll n, m; cin >> n >> m;
    vector<pii> pieces(m); 
    for(auto &p : pieces) { cin >> p.first >> p.second; }

    // Sets to keep track of occupied positions in rows, columns, and diagonals
    set<ll> rows, cols, diags1, diags2;

    // Populate sets with occupied positions
    for(auto &[i, j] : pieces) {
        rows.insert(i); 
        cols.insert(j); 
        diags1.insert(i - j); 
        diags2.insert(i + j);
    }

    // Calculate the total number of possible positions
    ll total_positions = n * n;

    // Subtract positions occupied directly by pieces
    total_positions -= rows.size() * n;
    total_positions -= cols.size() * n;

    // Add back positions that were subtracted twice (where pieces occupy multiple lines/diagonals)
    total_positions += rows.size() * cols.size();

    // Iterate over each diagonal to adjust the count
    for(auto &d1 : diags1) {
        set<pii> common_positions;
        for(auto &r : rows) {
            pii pos = {r, r - d1};
            if(inBounds(pos, n)) common_positions.insert(pos);
        }
        for(auto &c : cols) {
            pii pos = {c + d1, c};
            if(inBounds(pos, n)) common_positions.insert(pos);
        }
        ll diag_length = n - abs(d1);
        total_positions -= diag_length;
        total_positions += common_positions.size();
    }

    // Iterate over each anti-diagonal to adjust the count
    for(auto &d2 : diags2) {
        set<pii> common_positions;
        for(auto &r : rows) {
            pii pos = {r, d2 - r};
            if(inBounds(pos, n)) common_positions.insert(pos);
        }
        for(auto &c : cols) {
            pii pos = {d2 - c, c};
            if(inBounds(pos, n)) common_positions.insert(pos);
        }
        for(auto &d1 : diags1) {
            // Check if both diagonals are of the same parity
            if(((d2 % 2) + 2) % 2 == ((d1 % 2) + 2) % 2) {
                pii pos = {(d2 + d1) / 2, (d2 - d1) / 2};
                if(inBounds(pos, n)) common_positions.insert(pos);
            }
        }
        ll diag_length = d2 < n ? (d2 - 1) : (2 * n - d2 + 1);
        total_positions -= diag_length;
        total_positions += common_positions.size();
    }

    // Output the result
    cout << total_positions;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t = 1;
    // Uncomment the line below to read multiple test cases
    // cin >> t;
    while(t--) solve();
    return 0;
}
