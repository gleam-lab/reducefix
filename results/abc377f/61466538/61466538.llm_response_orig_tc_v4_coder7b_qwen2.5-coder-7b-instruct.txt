#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e9 + 7;

long long countSafeSquares(long long N, long long M, vector<pair<long long, long long>>& pieces) {
    unordered_set<long long> rows, cols, diag1, diag2;

    for(auto& [r, c] : pieces) {
        rows.insert(r);
        cols.insert(c);
        diag1.insert(r - c);
        diag2.insert(r + c);
    }

    long long safe_squares = 0;

    for(long long r = 1; r <= N; ++r) {
        for(long long c = 1; c <= N; ++c) {
            if(rows.find(r) == rows.end() && cols.find(c) == cols.end() &&
               diag1.find(r - c) == diag1.end() && diag2.find(r + c) == diag2.end()) {
                ++safe_squares;
            }
        }
    }

    return safe_squares;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    long long N, M;
    cin >> N >> M;

    vector<pair<long long, long long>> pieces(M);
    for(long long i = 0; i < M; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }

    cout << countSafeSquares(N, M, pieces) << '\n';

    return 0;
}
