bool inBounds(const pll &a, const ll n) {
    return 1 <= a.fst && a.fst <= n && 1 <= a.snd && a.snd <= n;
}

// F - Avoid Queen Attack
void solve() {
    ll n, m; re(n, m);
    vector<pii> q(m); FOR(i, 0, m) { re(q[i].fst, q[i].snd); }
    
    set<ll> rows, cols;
    set<pll> positions;
    for (auto [i, j] : q) {
        rows.insert(i);
        cols.insert(j);
        positions.insert({i, j});
    }

    // Subtract rows and columns
    ll ans = n * n;
    ans -= static_cast<ll>(rows.size()) * n;
    ans -= static_cast<ll>(cols.size()) * n;
    ans += static_cast<ll>(rows.size()) * static_cast<ll>(cols.size()); // Overlap of rows and cols

    // Handle diagonals (i-j)
    set<ll> diag1;
    for (auto [i, j] : q) diag1.insert(i - j);

    for (auto d : diag1) {
        ll cnt = n - abs(d);
        ans -= cnt;
    }

    // Handle anti-diagonals (i+j)
    set<ll> diag2;
    for (auto [i, j] : q) diag2.insert(i + j);

    for (auto d : diag2) {
        ll cnt = d <= n ? (d - 1) : (2 * n - d + 1);
        ans -= cnt;
    }

    // Now add back overlaps between attacked lines
    // Intersection between row and diag1
    set<pll> overlap;
    for (auto r : rows) {
        for (auto d : diag1) {
            ll j = r - d;
            pll pt = {r, j};
            if (inBounds(pt, n) && positions.count(pt)) {
                overlap.insert(pt);
            }
        }
    }

    // Intersection between col and diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            ll i = d + c;
            pll pt = {i, c};
            if (inBounds(pt, n) && positions.count(pt)) {
                overlap.insert(pt);
            }
        }
    }

    ans += overlap.size();

    // Intersection between row and diag2
    overlap.clear();
    for (auto r : rows) {
        for (auto d : diag2) {
            ll j = d - r;
            pll pt = {r, j};
            if (inBounds(pt, n) && positions.count(pt)) {
                overlap.insert(pt);
            }
        }
    }

    // Intersection between col and diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            ll i = d - c;
            pll pt = {i, c};
            if (inBounds(pt, n) && positions.count(pt)) {
                overlap.insert(pt);
            }
        }
    }

    // Intersection between diag1 and diag2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            // Solve:
            // i - j = d1
            // i + j = d2
            if ((d1 + d2) % 2 != 0) continue;
            ll i = (d1 + d2) / 2;
            ll j = (d2 - d1) / 2;
            pll pt = {i, j};
            if (inBounds(pt, n) && positions.count(pt)) {
                overlap.insert(pt);
            }
        }
    }

    ans += overlap.size();

    // Finally subtract overlaps counted thrice (triple intersections)
    // Only intersection of all three: row, col, and diagonal
    for (auto [i, j] : positions) {
        if (rows.count(i) && cols.count(j) && diag1.count(i - j) && diag2.count(i + j)) {
            ans += 2; // Add back two because we subtracted thrice earlier
        }
    }

    cout << ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t = 1;
    //cin >> t;
    while(t--) solve();
    return 0;
}
