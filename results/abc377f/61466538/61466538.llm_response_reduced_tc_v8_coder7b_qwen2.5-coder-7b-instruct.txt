bool isSafe(long long i, long long j) {
    for (auto [x, y] : positions) {
        if (x == i || y == j || x + y == i + j || x - y == i - j) {
            return false;
        }
    }

}

    long long count = 0;
    for (long long i = 1; i <= n; i++) {
        for (long long j = 1; j <= n; j++) {
