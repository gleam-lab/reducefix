#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

// Function to check if a position is within bounds
bool inBounds(const pii &a, const ll n) {
    return 1 <= a.first && a.first <= n && 1 <= a.second && a.second <= n;
}

// Main function to solve the problem
void solve() {
    ll n, m; cin >> n >> m;
    vector<pii> q(m); 
    for(auto &p : q) { cin >> p.first >> p.second; }
    
    set<ll> row, col, d1, d2;
    for(auto &[i, j] : q) {
        row.insert(i); 
        col.insert(j); 
        d1.insert(i - j); 
        d2.insert(i + j);
    }
    
    // Calculate the total number of positions
    ll ans = n * n;
    
    // Subtract the number of positions covered by each row, column, and diagonal
    ans -= (ll)row.size() * n;
    ans -= (ll)col.size() * n;
    ans += (ll)row.size() * (ll)col.size();
    
    // Set to store common positions
    set<pii> com;
    
    // Check diagonals of type i - j
    for(auto &d: d1) {
        com.clear();
        for(auto &r : row) {
            pii ints = {r, r - d};
            if(inBounds(ints, n)) com.insert(ints);
        }
        for(auto &c : col) {
            pii ints = {c + d, c};
            if(inBounds(ints, n)) com.insert(ints);
        }
        ll cnt = n - abs(d);
        ans -= cnt;
        ans += (ll)com.size();
    }
    
    // Check diagonals of type i + j
    for(auto &d: d2) {
        com.clear();
        for(auto &r : row) {
            pii ints = {r, d - r};
            if(inBounds(ints, n)) com.insert(ints);
        }
        for(auto &c : col) {
            pii ints = {d - c, c};
            if(inBounds(ints, n)) com.insert(ints);
        }
        for(auto &od : d1) {
            // od = (i - j)
            // d = (i + j)
            ll x = ((d % 2) + 2) % 2;
            ll y = ((od % 2) + 2) % 2;
            if(x != y) continue;
            pii ints = {(d + od) / 2, (d - od) / 2};
            if(inBounds(ints, n)) com.insert(ints);
        }
        ll cnt = d < n ? (d - 1) : (2 * n - d + 1);
        ans -= cnt;
        ans += (ll)com.size();
    }
    
    // Output the result
    cout << ans << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t = 1;
    cin >> t;
    while(t--) solve();
    return 0;
}
