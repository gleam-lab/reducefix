bool inBounds(const pll &a, const ll n) {
    return 1 <= a.first && a.first <= n && 1 <= a.second && a.second <= n;
}

// F - Avoid Queen Attack
void solve() {
    ll n, m;
    re(n, m);
    vector<pii> q(m);
    for (int i = 0; i < m; ++i) {
        re(q[i].first, q[i].second);
    }

    set<ll> row, col, diag1, diag2;
    for (auto [i, j] : q) {
        row.insert(i);
        col.insert(j);
        diag1.insert(i - j);
        diag2.insert(i + j);
    }

    ll total = n * n;
    // Subtract rows and columns
    total -= (ll)row.size() * n;
    total -= (ll)col.size() * n;
    // Add intersection of rows and columns back
    total += (ll)row.size() * (ll)col.size();

    // Now subtract diagonals: diag1 = i-j
    set<pll> overlap;
    for (auto d : diag1) {
        overlap.clear();
        // Intersections with rows and cols on this diagonal
        for (auto r : row) {
            pll pt(r, r - d);
            if (inBounds(pt, n)) overlap.insert(pt);
        }
        for (auto c : col) {
            pll pt(c + d, c);
            if (inBounds(pt, n)) overlap.insert(pt);
        }
        // Number of squares on this diagonal
        ll cnt;
        if (d >= 0) {
            cnt = min(n, n - d);
        } else {
            cnt = min(n, n + d);
        }
        total -= cnt;
        total += (ll)overlap.size();
    }

    // Now subtract diagonals: diag2 = i+j
    for (auto d : diag2) {
        overlap.clear();
        // Intersections with rows and cols on this diagonal
        for (auto r : row) {
            pll pt(r, d - r);
            if (inBounds(pt, n)) overlap.insert(pt);
        }
        for (auto c : col) {
            pll pt(d - c, c);
            if (inBounds(pt, n)) overlap.insert(pt);
        }
        // Number of squares on this diagonal
        ll cnt;
        if (d <= n + n && d >= 2) {
            if (d <= n + 1)
                cnt = d - 1;
            else
                cnt = 2 * n + 1 - d;
        } else {
            cnt = 0;
        }
        total -= cnt;
        total += (ll)overlap.size();
    }

    cout << max(total, 0LL) << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t = 1;
    //cin >> t;
    while(t--) solve();
    return 0;
}
