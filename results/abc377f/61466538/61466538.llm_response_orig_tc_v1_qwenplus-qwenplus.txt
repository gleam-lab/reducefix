bool inBounds(const pll &a, const ll n) {
    return 1 <= a.first && a.first <= n && 1 <= a.second && a.second <= n;
}

void solve() {
    ll n, m;
    re(n, m);

    set<pll> pieces;
    set<ll> rows, cols, diag1, diag2;

    for (ll i = 0; i < m; ++i) {
        ll a, b;
        re(a, b);
        pieces.insert({a, b});
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    ll total = n * n;
    ll attacked = 0;

    // Rows and columns
    attacked += (ll)rows.size() * n;
    attacked += (ll)cols.size() * n;

    // Remove overlaps between rows and columns
    for (auto r : rows) {
        for (auto c : cols) {
            if (pieces.count({r, c}) == 0) {
                attacked--;
            }
        }
    }

    // Diagonal 1: i-j = constant
    for (auto d : diag1) {
        ll count;
        if (d < 0) {
            ll len = min(n, n + d);
            count = len;
        } else {
            ll len = min(n, n - d);
            count = len;
        }
        attacked += count;
        // remove overlaps with rows and cols
        set<pll> overlap;
        for (auto r : rows) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                overlap.insert({r, c});
            }
        }
        for (auto c : cols) {
            ll r = d + c;
            if (1 <= r && r <= n) {
                overlap.insert({r, c});
            }
        }
        attacked -= (ll)overlap.size();
    }

    // Diagonal 2: i+j = constant
    for (auto d : diag2) {
        ll count;
        if (d <= n + 1) {
            count = d - 1;
        } else {
            count = 2 * n + 1 - d;
        }
        attacked += count;
        // remove overlaps with rows and cols and diagonal1
        set<pll> overlap;
        for (auto r : rows) {
            ll c = d - r;
            if (1 <= c && c <= n) {
                overlap.insert({r, c});
            }
        }
        for (auto c : cols) {
            ll r = d - c;
            if (1 <= r && r <= n) {
                overlap.insert({r, c});
            }
        }
        for (auto pd : diag1) {
            ll x = (d + pd) / 2;
            ll y = (d - pd) / 2;
            if ((d + pd) % 2 == 0 && (d - pd) % 2 == 0 && inBounds({x, y}, n)) {
                overlap.insert({x, y});
            }
        }
        attacked -= (ll)overlap.size();
    }

    ll ans = total - attacked;
    cout << ans << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t = 1;
    //cin >> t;
    while(t--) solve();
    return 0;
}
