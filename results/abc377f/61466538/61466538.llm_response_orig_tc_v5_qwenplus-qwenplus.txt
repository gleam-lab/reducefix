// F - Avoid Queen Attack
void solve() {
    ll n, m;
    re(n, m);
    vector<pii> pieces(m);
    for (auto &[a, b] : pieces) {
        re(a, b);
    }

    // Sets to track occupied rows, columns, and diagonals
    set<ll> rows, cols, diag1, diag2;
    for (auto &[a, b] : pieces) {
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Count of squares attacked by queens directly
    ll attacked = 0;

    // Row and column attacks
    attacked += rows.size() * n;
    attacked += cols.size() * n;

    // Remove overlapping row-col intersections
    set<pair<ll, ll>> overlap;
    for (auto r : rows) {
        for (auto c : cols) {
            overlap.insert({r, c});
        }
    }

    // Subtract overlaps between rows and columns
    attacked -= overlap.size();

    // Diagonal 1: a - b = constant
    for (auto d : diag1) {
        ll cnt;
        if (n >= abs(d)) {
            cnt = n - abs(d);
        } else {
            cnt = 0;
        }

        attacked += cnt;

        // Subtract overlaps with rows and columns on this diagonal
        set<pair<ll, ll>> diag_overlap;
        for (auto r : rows) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                diag_overlap.insert({r, c});
            }
        }
        for (auto c : cols) {
            ll r = c + d;
            if (1 <= r && r <= n) {
                diag_overlap.insert({r, c});
            }
        }

        attacked -= diag_overlap.size();
    }

    // Diagonal 2: a + b = constant
    for (auto d : diag2) {
        ll cnt;
        if (d <= n + 1) {
            cnt = d - 1;
        } else {
            cnt = 2 * n - d + 1;
        }

        attacked += cnt;

        // Subtract overlaps with rows and columns on this diagonal
        set<pair<ll, ll>> diag_overlap;
        for (auto r : rows) {
            ll c = d - r;
            if (1 <= c && c <= n) {
                diag_overlap.insert({r, c});
            }
        }
        for (auto c : cols) {
            ll r = d - c;
            if (1 <= r && r <= n) {
                diag_overlap.insert({r, c});
            }
        }

        // Also check intersection with diag1
        for (auto od : diag1) {
            ll x = (d + od);
            if (x % 2 != 0) continue;
            ll i = x / 2;
            ll j = d - i;
            if (1 <= i && i <= n && 1 <= j && j <= n) {
                diag_overlap.insert({i, j});
            }
        }

        attacked -= diag_overlap.size();
    }

    // Total safe positions = total squares - attacked squares - occupied squares
    ll total = n * n;
    ll occupied = m;
    ll result = total - attacked - occupied;
    cout << result << "\n";
}

int main() {

    //cin >> t;
    while(t--) solve();
    return 0;
}
