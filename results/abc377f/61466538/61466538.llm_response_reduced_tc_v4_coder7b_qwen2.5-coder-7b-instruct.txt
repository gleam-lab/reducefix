#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

bool inBounds(const ll &a, const ll n) {
    return 1 <= a && a <= n;
}

void solve() {
    ll n, m; cin >> n >> m;
    unordered_set<ll> row, col, d1, d2;
    for(ll i = 0; i < m; ++i) {
        ll a, b; cin >> a >> b;
        row.insert(a); col.insert(b); d1.insert(a - b); d2.insert(a + b);
    }
    ll ans = n * n;
    ans -= row.size() * n;
    ans -= col.size() * n;
    ans += row.size() * col.size();
    for(auto &d: d1) {
        ll cnt = n - abs(d);
        ans -= cnt;
    }
    for(auto &d: d2) {
        ll cnt = d < n ? (d - 1) : (2 * n - d + 1);
        ans -= cnt;
    }
    cout << ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t = 1;
    cin >> t;
    while(t--) solve();
    return 0;
}
