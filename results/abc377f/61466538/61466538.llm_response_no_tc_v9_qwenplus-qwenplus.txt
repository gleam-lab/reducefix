#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;

bool inBounds(const pll &a, const ll n) {
    return 1 <= a.first && a.first <= n && 1 <= a.second && a.second <= n;
}

// Function to compute the number of cells on diagonal with i - j = d and within an N x N grid
ll diag1_cells(ll d, ll n) {
    if (d >= 0)
        return min(n, n - d);
    else
        return min(n, n + d);
}

// Function to compute the number of cells on diagonal with i + j = s and within an N x N grid
ll diag2_cells(ll s, ll n) {
    if (s <= n + 1)
        return s - 1;
    else
        return 2 * n + 1 - s;
}

void solve() {
    ll n, m;
    cin >> n >> m;
    
    set<ll> rows, cols;
    set<ll> diag1, diag2; // i-j and i+j

    for (ll i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total squares attacked by existing pieces
    ll attacked = 0;

    // Add all squares in attacked rows
    attacked += rows.size() * n;

    // Add all squares in attacked columns
    attacked += cols.size() * n;

    // Subtract overcounted intersections of rows and columns
    for (ll r : rows) {
        for (ll c : cols) {
            // Each intersection (r,c) is counted twice
            attacked--;
        }
    }

    // Add all squares on diagonal i-j=d
    for (ll d : diag1) {
        attacked += diag1_cells(d, n);
    }

    // Subtract overcounted intersections with rows
    for (ll d : diag1) {
        for (ll r : rows) {
            ll c = r - d;
            if (1 <= c && c <= n) {
                attacked--;
            }
        }
    }

    // Subtract overcounted intersections with columns
    for (ll d : diag1) {
        for (ll c : cols) {
            ll r = c + d;
            if (1 <= r && r <= n) {
                attacked--;
            }
        }
    }

    // Add all squares on diagonal i+j=s
    for (ll s : diag2) {
        attacked += diag2_cells(s, n);
    }

    // Subtract overcounted intersections with rows
    for (ll s : diag2) {
        for (ll r : rows) {
            ll c = s - r;
            if (1 <= c && c <= n) {
                attacked--;
            }
        }
    }

    // Subtract overcounted intersections with columns
    for (ll s : diag2) {
        for (ll c : cols) {
            ll r = s - c;
            if (1 <= r && r <= n) {
                attacked--;
            }
        }
    }

    // Subtract overcounted intersections between diagonals i-j=d and i+j=s
    for (ll d : diag1) {
        for (ll s : diag2) {
            // Solve:
            // i - j = d
            // i + j = s
            ll i = (s + d) / 2;
            ll j = (s - d) / 2;
            if ((s + d) % 2 == 0 && (s - d) % 2 == 0 && inBounds({i, j}, n)) {
                attacked--;
            }
        }
    }

    // Total empty safe squares = total squares - attacked squares - already occupied squares
    ll total_empty_safe = n * n - attacked - m;

    cout << total_empty_safe << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    solve();
    return 0;
}
