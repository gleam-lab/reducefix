The problem requires counting the number of empty squares on an N x N grid where a new piece can be placed without being captured by any existing pieces. The existing pieces can capture along rows, columns, and both diagonals (i + j and i - j).

### Approaches
1. **Identify Forbidden Squares**: The squares that are under attack by any existing piece are those in the same row, same column, or either diagonal (i + j or i - j) as any existing piece.
2. **Calculate Safe Squares**: The total number of safe squares is the total squares (N^2) minus the squares under attack plus the squares that were double-counted (intersection of rows, columns, and diagonals).
3. **Use Sets for Efficient Lookup**: Utilize sets to keep track of rows, columns, and diagonals that are already occupied by existing pieces to efficiently determine the under-attack squares.

### Solution Code
```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    ll N, M;
    cin >> N >> M;
    vector<pll> queens(M);
    set<ll> rows, cols, diag1, diag2;
    set<pll> queen_positions;
    
    for (ll i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        queens[i] = {a, b};
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
        queen_positions.insert({a, b});
    }
    
    ll unsafe = 0;
    
    // Rows and columns
    unsafe += rows.size() * N;
    unsafe += cols.size() * N;
    unsafe -= rows.size() * cols.size();
    
    // Diagonals (i - j)
    for (auto d : diag1) {
        ll cnt = 0;
        if (d >= 0) {
            cnt = N - d;
        } else {
            cnt = N + d;
        }
        unsafe += cnt;
        
        // Subtract intersections with rows and columns
        for (auto r : rows) {
            ll c = r - d;
            if (c >= 1 && c <= N) {
                if (queen_positions.find({r, c}) == queen_positions.end()) {
                    unsafe--;
                }
            }
        }
        for (auto c : cols) {
            ll r = c + d;
            if (r >= 1 && r <= N) {
                if (queen_positions.find({r, c}) == queen_positions.end()) {
                    unsafe--;
                }
            }
        }
    }
    
    // Diagonals (i + j)
    for (auto d : diag2) {
        ll cnt = 0;
        if (d <= N + 1) {
            cnt = d - 1;
        } else {
            cnt = 2 * N - d + 1;
        }
        unsafe += cnt;
        
        // Subtract intersections with rows and columns
        for (auto r : rows) {
            ll c = d - r;
            if (c >= 1 && c <= N) {
                if (queen_positions.find({r, c}) == queen_positions.end()) {
                    unsafe--;
                }
            }
        }
        for (auto c : cols) {
            ll r = d - c;
            if (r >= 1 && r <= N) {
                if (queen_positions.find({r, c}) == queen_positions.end()) {
                    unsafe--;
                }
            }
        }
        
        // Subtract intersections with diag1 (i - j)
        for (auto d1 : diag1) {
            ll i = (d + d1) / 2;
            ll j = (d - d1) / 2;
            if ((i + j) == d && (i - j) == d1 && i >= 1 && i <= N && j >= 1 && j <= N) {
                if (queen_positions.find({i, j}) == queen_positions.end()) {
                    unsafe--;
                }
            }
        }
    }
    
    ll total = N * N;
    ll safe = total - unsafe - M; // Subtract M because queen positions are initially counted as unsafe but are not empty
    safe = max(safe, 0LL); // Ensure non-negative
    cout << safe << "\n";
    
    return 0;
}
```

### Explanation
1. **Input Handling**: The code reads the grid size N and the number of queens M, followed by their positions.
2. **Tracking Rows, Columns, and Diagonals**: Sets are used to store occupied rows, columns, and diagonals (i - j and i + j) for quick lookup.
3. **Calculating Unsafe Squares**: The unsafe squares are those under attack by any queen. This includes squares in the same row, column, or diagonal as any queen. The calculation accounts for overlaps to avoid double-counting.
4. **Adjusting for Queen Positions**: The existing queen positions are subtracted from the unsafe count since they are occupied and not empty.
5. **Output**: The total safe squares is computed as the total squares minus unsafe squares minus occupied squares, ensuring a non-negative result.

This approach efficiently computes the number of safe squares by leveraging set operations to manage and query occupied rows, columns, and diagonals.
