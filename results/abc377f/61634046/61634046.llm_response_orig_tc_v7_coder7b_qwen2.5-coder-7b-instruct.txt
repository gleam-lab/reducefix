```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX_N = 1e9 + 5;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<int> rows(n+1), cols(n+1), diags1(n+1), diags2(n+1);
    vector<vector<int>> pos(4, vector<int>(n+1));

    for(int i=0; i<m; ++i){
        int x, y;
        cin >> x >> y;
        rows[x]++;
        cols[y]++;
        diags1[x+y]++;
        diags2[x-y+n]++;
        pos[0][x]++;
        pos[1][y]++;
        pos[2][x+y]++;
        pos[3][x-y+n]++;
    }

    long long ans = 0;
    for(int i=1; i<=n; ++i){
        for(int j=1; j<=n; ++j){
            if(!rows[i] && !cols[j] && !diags1[i+j] && !diags2[i-j+n]){
                ans++;
            }
        }
    }

    cout << ans << '\n';

    return 0;
}
```
