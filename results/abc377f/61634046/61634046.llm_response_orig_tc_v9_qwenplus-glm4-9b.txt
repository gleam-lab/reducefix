### 忙颅楼茅陋陇 3茂录職氓庐聻莽聨掳莽庐聴忙鲁聲

盲禄楼盲赂聥忙聵炉猫搂拢氓聠鲁猫驴聶盲赂陋茅聴庐茅垄聵莽職聞 Python 盲禄拢莽聽聛茂录職

```python
def count_safe_squares(N, M, positions):
    # 猫庐隆莽庐聴忙聣聙忙聹聣莽漏潞盲陆聧莽職聞忙聙禄忙聲掳
    total_squares = N * N
    
    # 氓聢聺氓搂聥氓聦聳猫垄芦忙聧聲猫聨路莽職聞莽漏潞盲陆聧忙聲掳茅聡聫
    captured_squares = 0
    
    # 氓颅聵氓聜篓忙聣聙忙聹聣忙拢聥氓颅聬莽職聞氓聺聬忙聽聡
    pieces = [(a - 1, b - 1) for a, b in positions]
    
    # 茅聛聧氓聨聠忙聣聙忙聹聣莽漏潞盲陆聧
    for i in range(N):
        for j in range(N):
            x, y = i, j
            
            # 忙拢聙忙聼楼忙聵炉氓聬娄猫垄芦猫隆聦忙聧聲猫聨路
            if any(x == px or y == py for px, py in pieces):
                captured_squares += 1
                continue
            
            # 忙拢聙忙聼楼忙聵炉氓聬娄猫垄芦盲赂禄氓炉鹿猫搂聮莽潞驴忙聧聲猫聨路
            if any(x + y == px + py for px, py in pieces):
                captured_squares += 1
                continue
            
            # 忙拢聙忙聼楼忙聵炉氓聬娄猫垄芦氓聣炉氓炉鹿猫搂聮莽潞驴忙聧聲猫聨路
            if any(x - y == px - py for px, py in pieces):
                captured_squares += 1
                continue
    
    # 猫庐隆莽庐聴盲赂聧氓聫聴忙聧聲猫聨路莽職聞莽漏潞盲陆聧忙聲掳茅聡聫
    safe_squares = total_squares - captured_squares
    return safe_squares

# 猫炉禄氓聫聳
N, M = map(int, input().split())
positions = [tuple(map(int, input().split())) for _ in range(M)]

# 猫庐隆莽庐聴盲赂聧氓聫聴忙聧聲猫聨路莽職聞莽漏潞盲陆聧忙聲掳茅聡聫
safe_squares = count_safe_squares(N, M, positions)
print(safe_squares)
```
