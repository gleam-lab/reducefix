```cpp
#include <bits/stdc++.h>

using namespace std;

const int MOD = 1e9 + 7;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<int> rows(n + 1), cols(n + 1), ups[n + 1], downs[n + 1];
    for (int i = 1; i <= n; ++i) {
        ups[i].resize(n + 1);
        downs[i].resize(n + 1);
    }

    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        rows[y]++;
        cols[x]++;
        ups[x + y][y]++;
        downs[x - y + n][y]++;
    }

    long long total = 1LL * n * n;
    for (int i = 1; i <= n; ++i) {
        total -= rows[i] + cols[i] - 1;
        total -= ups[i].back();
        total -= downs[i].back();
    }

    cout << (total + MOD) % MOD << endl;

    return 0;
}
```

This solution re-evaluates the problem by focusing on the counts of pieces in each row, column, upward diagonal, and downward diagonal. It then calculates the total number of valid positions by subtracting the over-counted positions and ensures the result is within bounds using modulo arithmetic. This approach avoids the need for complex sorting and binary search operations, making it more efficient and easier to understand.
