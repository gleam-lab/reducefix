```cpp
#include <bits/stdc++.h>

using namespace std;

using ll = long long;
constexpr int MAX_M = 1000 + 5;

int N, M;
set<int> rows, cols;
set<ll> diag1, diag2;

int main() {
    scanf("%d %d", &N, &M);
    for (int i = 0; i < M; ++i) {
        int x, y;
        scanf("%d %d", &x, &y);
        rows.insert(x);
        cols.insert(y);
        diag1.insert((ll)x - y);
        diag2.insert((ll)x + y);
    }

    // Total number of squares not attacked by any piece
    ll total = (ll)N * N;
    
    // Remove all squares in the same row
    total -= (ll)rows.size() * N;
    
    // Remove all squares in the same column
    total -= (ll)cols.size() * N;
    
    // Remove all squares on the same diagonal/diagonal'
    for (auto d : diag1)
        total -= (ll)(min(N + N - d, N + d) - max(1 - d, 1) + 1);
    for (auto d : diag2)
        total -= (ll)(min(N, d - 1) - max(1, d - N) + 1);

    // Add back over-subtracted intersections

    // Row-Column intersections
    total += (ll)rows.size() * cols.size();

    // Row-Diag1 intersections: Points where row=r and diag1=d => (r, r-d)
    for (auto r : rows)
        for (auto d : diag1) {
            int c = r - d;
            if (1 <= c && c <= N)
                ++total;
        }

    // Row-Diag2 intersections: Points where row=r and diag2=d => (r, d-r)
    for (auto r : rows)
        for (auto d : diag2) {
            int c = d - r;
            if (1 <= c && c <= N)
                ++total;
        }

    // Col-Diag1 intersections: Points where col=c and diag1=d => (d+c, c)
    for (auto c : cols)
        for (auto d : diag1) {
            int r = d + c;
            if (1 <= r && r <= N)
                ++total;
        }

    // Col-Diag2 intersections: Points where col=c and diag2=d => (d-c, c)
    for (auto c : cols)
        for (auto d : diag2) {
            int r = d - c;
            if (1 <= r && r <= N)
                ++total;
        }

    // Diag1-Diag2 intersections: Solve r - c = d1 and r + c = d2 => r = (d1 + d2)/2, c = (d2 - d1)/2
    for (auto d1 : diag1)
        for (auto d2 : diag2) {
            if ((d1 + d2) % 2 == 0) {
                int r = (d1 + d2) / 2;
                int c = (d2 - d1) / 2;
                if (1 <= r && r <= N && 1 <= c && c <= N)
                    ++total;
            }
        }

    // Subtract triple overlaps

    // Row-Col-Diag1 intersections
    for (auto r : rows)
        for (auto c : cols)
            if (diag1.count(r - c))
                ++total;

    // Row-Col-Diag2 intersections
    for (auto r : rows)
        for (auto c : cols)
            if (diag2.count(r + c))
                ++total;

    // Row-Diag1-Diag2 intersections
    for (auto r : rows)
        for (auto d1 : diag1)
            for (auto d2 : diag2) {
                int c1 = r - d1;
                int c2 = d2 - r;
                if (c1 == c2 && 1 <= c1 && c1 <= N)
                    ++total;
            }

    // Col-Diag1-Diag2 intersections
    for (auto c : cols)
        for (auto d1 : diag1)
            for (auto d2 : diag2) {
                int r1 = d1 + c;
                int r2 = d2 - c;
                if (r1 == r2 && 1 <= r1 && r1 <= N)
                    ++total;
            }

    // Add back quadruple overlaps
    // Row-Col-Diag1-Diag2 intersections
    for (auto r : rows)
        for (auto c : cols)
            if (diag1.count(r - c) && diag2.count(r + c))
                ++total;

    printf("%lld\n", total);
    return 0;
}
```
