#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MAXN = 1005;
ll dp[MAXN][MAXN];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    cin >> n >> m;

    memset(dp, 0, sizeof(dp));

    for(int i = 0; i < m; i++) {
        int x, y;
        cin >> x >> y;
        dp[x][y] = -1; // Marking the position as occupied
    }

    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(dp[i][j] == -1) continue; // Skip occupied positions
            dp[i][j] = 1; // This position is safe initially
            // Check row
            for(int k = 1; k <= n; k++) {
                if(k != i && dp[k][j] >= 0) dp[k][j] = -1; // Capture all pieces in the same row
            }
            // Check column
            for(int k = 1; k <= n; k++) {
                if(k != j && dp[i][k] >= 0) dp[i][k] = -1; // Capture all pieces in the same column
            }
            // Check diagonals
            for(int k = 1; k <= n; k++) {
                if((i+k <= n && j+k <= n) && (i+k != i || j+k != j) && dp[i+k][j+k] >= 0) dp[i+k][j+k] = -1; // Capture all pieces in the same diagonal
                if((i-k >= 1 && j-k >= 1) && (i-k != i || j-k != j) && dp[i-k][j-k] >= 0) dp[i-k][j-k] = -1; // Capture all pieces in the same anti-diagonal
            }
        }
    }

    ll ans = 0;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(dp[i][j] == 0) ans++; // Count all remaining empty positions
        }
    }

    cout << ans << endl;

    return 0;
}
