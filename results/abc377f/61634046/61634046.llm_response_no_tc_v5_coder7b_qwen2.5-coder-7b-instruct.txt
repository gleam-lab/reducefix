```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    
    vector<pair<int, int>> positions(M);
    for(int i = 0; i < M; ++i) {
        cin >> positions[i].first >> positions[i].second;
    }
    
    // Sort positions based on their sum and difference
    sort(positions.begin(), positions.end(), [](const pair<int, int>& p1, const pair<int, int>& p2) {
        if(p1.first + p1.second != p2.first + p2.second) return p1.first + p1.second < p2.first + p2.second;
        return p1.first - p1.second < p2.first - p2.second;
    });
    
    int lastSum = positions[0].first + positions[0].second;
    int lastDiff = positions[0].first - positions[0].second;
    int sameSumCount = 1, sameDiffCount = 1;
    
    long long result = 0;
    for(int i = 1; i < M; ++i) {
        int currentSum = positions[i].first + positions[i].second;
        int currentDiff = positions[i].first - positions[i].second;
        
        if(currentSum == lastSum) {
            ++sameSumCount;
        } else {
            result += (long long)(N - sameSumCount) * (N - sameDiffCount);
            sameSumCount = 1;
        }
        
        if(currentDiff == lastDiff) {
            ++sameDiffCount;
        } else {
            result += (long long)(N - sameSumCount) * (N - sameDiffCount);
            sameDiffCount = 1;
        }
        
        lastSum = currentSum;
        lastDiff = currentDiff;
    }
    
    result += (long long)(N - sameSumCount) * (N - sameDiffCount);
    
    cout << result << endl;
    
    return 0;
}
```

This solution first sorts the positions based on their sum and difference. Then it iterates through the sorted list to calculate the number of valid positions where no piece can be captured. The result is printed at the end.
