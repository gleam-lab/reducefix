#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

#define ll long long
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;

    vector<pair<int, int>> pieces(M);
    for(int i = 0; i < M; ++i) {
        int x, y;
        cin >> x >> y;
        pieces[i] = {x, y};
    }

    // Sort pieces by row and column to use binary search
    sort(pieces.begin(), pieces.end());

    // Maps to store the positions of pieces
    map<int, vector<int>> row_map;
    map<int, vector<int>> col_map;
    map<int, vector<int>> diag1_map; // i + j
    map<int, vector<int>> diag2_map; // i - j

    for(int i = 0; i < M; ++i) {
        int x = pieces[i].first;
        int y = pieces[i].second;
        row_map[x].push_back(y);
        col_map[y].push_back(x);
        diag1_map[x + y].push_back(i);
        diag2_map[x - y].push_back(i);
    }

    ll total_empty_squares = N * N;

    // Subtract squares that are in the same row or column as any piece
    for(int r = 1; r <= N; ++r) {
        total_empty_squares -= row_map[r].size();
    }
    for(int c = 1; c <= N; ++c) {
        total_empty_squares -= col_map[c].size();
    }

    // Subtract squares that are on the same diagonals as any piece
    for(int d = 1; d <= 2 * N; ++d) {
        total_empty_squares -= diag1_map[d].size();
        total_empty_squares -= diag2_map[d].size();
    }

    cout << total_empty_squares << endl;
    return 0;
}
