#include <bits/stdc++.h>
using namespace std;

#define ll long long

const int MOD = 1e9 + 7;

ll power(ll base, ll exp) {
    ll res = 1;
    while(exp) {
        if(exp & 1) res = (res * base) % MOD;
        base = (base * base) % MOD;
        exp >>= 1;
    }
    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    cin >> n >> m;

    vector<int> row(n+1), col(n+1), diag1(n+1), diag2(n+1);
    memset(row, 0, sizeof(row));
    memset(col, 0, sizeof(col));
    memset(diag1, 0, sizeof(diag1));
    memset(diag2, 0, sizeof(diag2));

    for(int i=1;i<=m;i++) {
        int a, b;
        cin >> a >> b;
        row[a]++;
        col[b]++;
        diag1[a+b]++;
        diag2[a-b+n]++;
    }

    ll total_squares = 1LL * n * n;
    ll occupied_squares = 1LL * m;

    for(int i=1;i<=n;i++) {
        occupied_squares += 1LL * (n-row[i]);
        occupied_squares += 1LL * (n-col[i]);
        occupied_squares += 1LL * (n-diag1[i]);
        occupied_squares += 1LL * (n-diag2[i]);
    }

    occupied_squares /= 2;

    cout << (total_squares - occupied_squares) % MOD << endl;

    return 0;
}
