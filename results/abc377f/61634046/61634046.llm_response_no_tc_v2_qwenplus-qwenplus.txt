```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    long long N;
    int M;
    cin >> N >> M;

    set<long long> rows, cols, diag1, diag2;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total empty cells = N*N - M (since M cells have pieces)
    // From these, we subtract the number of dangerous empty cells

    long long total_safe = N * N - M;

    // Dangerous cells are those that are in any row, column, or diagonal with a piece
    // Use inclusion-exclusion to count the number of dangerous cells

    // Count cells covered by any line
    long long dangerous = 0;

    // Number of cells on rows with pieces (excluding where pieces are already placed)
    dangerous += (long long)rows.size() * N;

    // Number of cells on columns with pieces (excluding where pieces are already placed)
    dangerous += (long long)cols.size() * N;

    // Number of cells on diagonals (a+b)=const with pieces
    dangerous += (long long)diag1.size() * N;

    // Number of cells on diagonals (a-b)=const with pieces
    dangerous += (long long)diag2.size() * N;

    // Subtract overlaps (Inclusion-Exclusion)

    // Rows and Columns
    dangerous -= (long long)rows.size() * cols.size();

    // Rows and Diagonal1 (a+b)
    for (auto r : rows) {
        for (auto d : diag1) {
            long long c = d - r;
            if (1 <= c && c <= N) {
                dangerous -= 1;
            }
        }
    }

    // Rows and Diagonal2 (a-b)
    for (auto r : rows) {
        for (auto d : diag2) {
            long long c = r - d;
            if (1 <= c && c <= N) {
                dangerous -= 1;
            }
        }
    }

    // Columns and Diagonal1 (a+b)
    for (auto c : cols) {
        for (auto d : diag1) {
            long long r = d - c;
            if (1 <= r && r <= N) {
                dangerous -= 1;
            }
        }
    }

    // Columns and Diagonal2 (a-b)
    for (auto c : cols) {
        for (auto d : diag2) {
            long long r = c + d;
            if (1 <= r && r <= N) {
                dangerous -= 1;
            }
        }
    }

    // Diagonal1 and Diagonal2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            long long r = (d1 + d2) / 2;
            long long c = (d1 - d2) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N) {
                dangerous -= 1;
            }
        }
    }

    // Add back triple overlaps

    // Rows, Columns, and Diagonals at intersections
    for (auto r : rows) {
        for (auto c : cols) {
            if ((diag1.count(r + c) + diag2.count(r - c)) > 0) {
                dangerous += 1;
            }
        }
    }

    for (auto r : rows) {
        for (auto d1 : diag1) {
            long long c = d1 - r;
            if (1 <= c && c <= N && diag2.count(r - c)) {
                dangerous += 1;
            }
        }
    }

    for (auto c : cols) {
        for (auto d1 : diag1) {
            long long r = d1 - c;
            if (1 <= r && r <= N && diag2.count(r - c)) {
                dangerous += 1;
            }
        }
    }

    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            long long r = (d1 + d2) / 2;
            long long c = (d1 - d2) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N) {
                if (rows.count(r) + cols.count(c) > 0) {
                    dangerous += 1;
                }
            }
        }
    }

    // Subtract quadruple overlaps
    for (auto r : rows) {
        for (auto c : cols) {
            if (diag1.count(r + c) && diag2.count(r - c)) {
                dangerous -= 1;
            }
        }
    }

    // The final answer is total empty cells minus dangerous ones
    cout << total_safe - dangerous << endl;

    return 0;
}
```
