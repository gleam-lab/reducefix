#include <bits/stdc++.h>
using namespace std;

const int MAX_N = 1e9 + 5;
const int MOD = 1e9 + 7;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<pair<int, int>> pieces(m);
    for (auto& [x, y] : pieces) {
        cin >> x >> y;
    }

    unordered_set<int> rows, cols, diags1, diags2;
    for (auto& [x, y] : pieces) {
        rows.insert(x);
        cols.insert(y);
        diags1.insert(x - y);
        diags2.insert(x + y);
    }

    long long total_squares = static_cast<long long>(n) * n;
    long long occupied_squares = rows.size() + cols.size() + diags1.size() + diags2.size();
    long long valid_squares = total_squares - occupied_squares + 1;

    cout << valid_squares % MOD << endl;

    return 0;
}
