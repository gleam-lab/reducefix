#include <bits/stdc++.h>

typedef long long ll;

int main() {
    ll N, M;
    cin >> N >> M;
    
    set<ll> rows, cols, diag1, diag2;
    
    for (int i = 0; i < M; ++i) {
        ll a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);

    ll unsafe_rows = rows.size();
    ll unsafe_cols = cols.size();
    ll unsafe_diag1 = diag1.size();
    ll unsafe_diag2 = diag2.size();
    
    // Total unsafe squares: rows + cols + diag1 + diag2 - intersections
    // Using inclusion-exclusion principle
    ll total_unsafe = unsafe_rows * N + unsafe_cols * N;
    
    for (ll d : diag1) {
        ll count = 0;
        // Diagonal i+j = d
        // i ranges from max(1, d - N) to min(N, d - 1)

        ll i_max = min(N, d - 1);
        if (i_min > i_max) continue;
        count += i_max - i_min + 1;

                }
            }

    ll col_diag2 = 0;
    for (ll c : cols) {

            if (r >= 1 && r <= N) {
                if (cols.count(c) && diag2.count(r - c)) {
                    col_diag2++;
                }
            }
        }

    total_unsafe -= col_diag2;
    
    ll diag1_diag2 = 0;
    for (ll d1 : diag1) {
        for (ll d2 : diag2) {
            ll i = (d1 + d2) / 2;
            ll j = (d1 - d2) / 2;
            if ((d1 + d2) % 2 == 0 && (d1 - d2) % 2 == 0 && i >= 1 && i <= N && j >= 1 && j <= N) {
                if (diag1.count(i + j) && diag2.count(i - j)) {
                    diag1_diag2++;
                }
            }
        }
    }
    total_unsafe -= diag1_diag2;
    
    // Add back triple intersections

    // Subtract quadruple intersection
    ll row_col_diag1_diag2 = 0;
    for (ll r : rows) {
        for (ll c : cols) {
            if (diag1.count(r + c) && diag2.count(r - c)) {
                row_col_diag1_diag2++;
            }
        }
    }
    total_unsafe -= row_col_diag1_diag2;
    
    ll total_safe = N * N - total_unsafe + M; // Since existing pieces are already counted as unsafe but are not empty
    cout << total_safe << endl;
    
    return 0;
}

### Explanation
1. **Initialization and Input Handling**: The code reads the grid size `N` and the number of existing pieces `M`. It initializes sets to store the rows, columns, and diagonals that are under attack by existing pieces.
2. **Counting Unsafe Squares**: The total unsafe squares are calculated by summing the squares in attacked rows, columns, and diagonals. The inclusion-exclusion principle is applied to avoid double-counting squares that lie at the intersections of these attack lines.
3. **Adjustments for Existing Pieces**: The existing pieces are already counted in the unsafe squares but are not empty, so the final count of safe squares is adjusted by adding `M` back to the result.
4. **Output**: The total number of safe squares is printed, which is the difference between the total grid squares and the unsafe squares, adjusted for existing pieces.
