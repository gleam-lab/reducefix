```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    long long N;
    int M;
    cin >> N >> M;

    set<long long> rows, cols, diag1, diag2;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a + b);
        diag2.insert(a - b);
    }

    // Total squares not attacked by any piece
    long long total = N * N;
    total -= (long long)rows.size() * N; // Rows
    total -= (long long)cols.size() * N; // Columns
    total -= (long long)diag1.size() * N; // Diagonal 1 (\)
    total -= (long long)diag2.size() * N; // Diagonal 2 (/)

    // Add back over-subtracted intersections

    // Rows x Columns
    total += (long long)rows.size() * cols.size();

    // Rows x Diag1
    for (auto r : rows) {
        for (auto d : diag1) {
            long long c = d - r;
            if (c >= 1 && c <= N) total++;
        }
    }

    // Rows x Diag2
    for (auto r : rows) {
        for (auto d : diag2) {
            long long c = r + d;
            if (c >= 1 && c <= N) total++;
        }
    }

    // Cols x Diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            long long r = d - c;
            if (r >= 1 && r <= N) total++;
        }
    }

    // Cols x Diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            long long r = c + d;
            if (r >= 1 && r <= N) total++;
        }
    }

    // Diag1 x Diag2 -> Each intersection is a unique point (r,c)
    // So we check how many such intersections exist within grid bounds
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            long long r = (d1 + d2) / 2;
            long long c = (d1 - d2) / 2;
            if ((d1 + d2) % 2 == 0 && (d1 - d2) % 2 == 0 && r >= 1 && r <= N && c >= 1 && c <= N) {
                total++;
            }
        }
    }

    // Subtract again the 3-way overlaps
    for (auto r : rows) {
        for (auto c : cols) {
            long long d1 = r + c;
            long long d2 = r - c;
            if (diag1.count(d1)) total--;
            if (diag2.count(d2)) total--;
        }
    }

    // Add back 4-way overlaps
    for (auto r : rows) {
        for (auto c : cols) {
            long long d1 = r + c;
            long long d2 = r - c;
            if (diag1.count(d1) && diag2.count(d2)) total++;
        }
    }

    cout << total << endl;

    return 0;
}
```
