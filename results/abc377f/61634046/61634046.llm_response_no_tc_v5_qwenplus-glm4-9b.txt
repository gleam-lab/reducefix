#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ll = long long;
const ll MOD = 1e9 + 7;

int n, m;
vector<pair<ll, ll>> pieces;

// Function to calculate the number of empty squares
ll countEmptySquares() {
    vector<ll> dr = {1, 1, 0, -1};
    vector<ll> dc = {0, -1, -1, 0};
    vector<vector<bool>> valid(n + 1, vector<bool>(n + 1, true));
    ll emptyCount = n * n - m;

    // Mark invalid positions based on captured ranges
    for (auto& p : pieces) {
        ll r = p.first, c = p.second;
        for (int d = 0; d < 4; ++d) {
            ll dr_ = dr[d], dc_ = dc[d];
            ll r1 = r + dr_, c1 = c + dc_;
            ll r2 = r - dr_, c2 = c - dc_;
            while (r1 <= n && c1 <= n && r2 >= 1 && c2 >= 1) {
                if (valid[r1][c1]) {
                    valid[r1][c1] = false;
                    emptyCount--;
                }
                if (valid[r2][c2]) {
                    valid[r2][c2] = false;
                    emptyCount--;
                }
                r1 += dr_;
                c1 += dc_;
                r2 -= dr_;
                c2 -= dc_;
            }
        }
    }

    return emptyCount;
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        ll a, b;
        cin >> a >> b;
        pieces.push_back({a, b});
        // Mark the square as occupied
        pieces.push_back({b, a});
    }

    // Sort pieces to make the range checking easier
    sort(pieces.begin(), pieces.end());

    // Find duplicate pieces for diagonal capturing
    vector<pair<ll, ll>> duplicates;
    for (int i = 0; i + 1 < m; ++i) {
        if (pieces[i].first == pieces[i + 1].first && pieces[i].second == pieces[i + 1].second) {
            duplicates.push_back({pieces[i].first, pieces[i].second});
            m--;
        }
    }

    cout << countEmptySquares() << endl;
    return 0;
}

This code makes several improvements:
- It uses a 2D vector `valid` to track valid positions in the grid.
- It marks all positions that can be captured by each piece.
- It calculates the number of empty squares that cannot be captured by any piece.
- It considers diagonal capturing by adding duplicate pieces to the list.
- It uses a simple and efficient approach to calculate the result.
