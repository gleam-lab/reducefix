```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    set<long long> rows, cols, diag1, diag2;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;

        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total number of squares
    long long total = N * N;

    // Number of occupied squares
    long long occupied = M;

    // Number of squares attacked by existing pieces
    // = all squares in the same row, column, or diagonal as any piece

    // Rows and columns with at least one piece
    long long attacked = (rows.size() + cols.size()) * N;

    // Diagonals d1: i-j = constant, d2: i+j = constant
    attacked += diag1.size() * N;
    attacked += diag2.size() * N;

    // Subtract overlaps:
    // Squares that are counted more than once because they're on intersections

    // Intersections: row & col
    attacked -= 1LL * rows.size() * cols.size();

    // Intersections: row & diag1
    for (auto r : rows) {
        for (auto d : diag1) {
            if ((r - d) >= 1 && (r - d) <= N) ++attacked;
        }
    }

    // Intersections: row & diag2
    for (auto r : rows) {
        for (auto d : diag2) {
            if ((d - r) >= 1 && (d - r) <= N) ++attacked;
        }
    }

    // Intersections: col & diag1
    for (auto c : cols) {
        for (auto d : diag1) {
            if ((d + c) >= 1 && (d + c) <= N) ++attacked;
        }
    }

    // Intersections: col & diag2
    for (auto c : cols) {
        for (auto d : diag2) {
            if ((d - c) >= 1 && (d - c) <= N) ++attacked;
        }
    }

    // Intersections: diag1 & diag2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            long long x = (d2 - d1) / 2;
            long long y = (d2 + d1) / 2;
            if (x >= 1 && x <= N && y >= 1 && y <= N) ++attacked;
        }
    }

    // Intersections: row & col & diag1/diag2
    for (auto r : rows) {
        for (auto c : cols) {
            if (diag1.count(r - c)) ++attacked;
            if (diag2.count(r + c)) ++attacked;
        }
    }

    // Intersections: row & diag1 & diag2
    for (auto r : rows) {
        for (auto d1 : diag1) {
            long long c = r - d1;
            if (c >= 1 && c <= N && diag2.count(r + c)) ++attacked;
        }
    }

    // Intersections: col & diag1 & diag2
    for (auto c : cols) {
        for (auto d1 : diag1) {
            long long r = d1 + c;
            if (r >= 1 && r <= N && diag2.count(r + c)) ++attacked;
        }
    }

    // Intersections: diag1 & diag2 & row/col
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            long long x = (d2 - d1) / 2;
            long long y = (d2 + d1) / 2;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                if (rows.count(x) || cols.count(y)) ++attacked;
            }
        }
    }

    // Intersections: row & col & diag1 & diag2
    for (auto r : rows) {
        for (auto c : cols) {
            if (diag1.count(r - c) && diag2.count(r + c)) ++attacked;
        }
    }

    // Now subtract all attacked and occupied squares from total
    long long answer = total - attacked - occupied;
    cout << answer << endl;

    return 0;
}
```
