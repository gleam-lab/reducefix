#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

#define MAX_N 1000

using namespace std;

int n, m;
vector<int> pieces[MAX_N + 1][MAX_N + 1];

int main() {
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        pieces[x][y].push_back(i + 1);
    }

    long long ans = n * n - m;
    set<pair<int, int>> row_captures, col_captures, diag1_captures, diag2_captures;

    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (!pieces[i][j].empty()) {
                // Check if any piece can capture this square
                if (contains(row_captures, {i, j}) || contains(col_captures, {i, j}) ||
                    contains(diag1_captures, {i + j, j - i}) || contains(diag2_captures, {i - j, i + j})) {
                    ans--;
                }

                // Capture other squares
                row_captures.insert({i, j});
                col_captures.insert({i, j});
                diag1_captures.insert({i + j, j - i});
                diag2_captures.insert({i - j, i + j});

                // Capture opposite direction diagonals
                diag1_captures.insert({i + j, n + 1 - (j - i)});
                diag2_captures.insert({i - j, n + 1 - (i + j)});
            }
        }
    }

    cout << ans << endl;
    return 0;
}
