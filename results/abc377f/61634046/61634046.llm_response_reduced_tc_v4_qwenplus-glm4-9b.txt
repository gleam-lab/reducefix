#include <iostream>
#include <algorithm>
#include <vector>
#include <map>

using namespace std;

long long countEmptyPositions(int n, int m, const vector<pair<int, int>>& pieces) {
    // Maps to store unique positions for rows, columns, and diagonals
    map<int, bool> row_map, col_map, diag1_map, diag2_map;
    
    // Fill the maps with piece positions
    for (const auto& piece : pieces) {
        row_map[piece.second] = true;
        col_map[piece.first] = true;
        diag1_map[piece.first + piece.second] = true;
        diag2_map[piece.first - piece.second] = true;
    }
    
    // Calculate the number of unique rows, columns, and diagonals
    int unique_rows = row_map.size();
    int unique_cols = col_map.size();
    int unique_diag1 = diag1_map.size();
    int unique_diag2 = diag2_map.size();
    
    // Calculate the number of positions captured by pieces
    int captured_positions = 0;
    // Rows and columns
    captured_positions += unique_rows + unique_cols;
    // Diagonals (considering both directions: 1+1 and 2+2)
    captured_positions += 2 * (unique_diag1 + unique_diag2);
    
    // Total number of squares on the grid
    long long total_squares = (long long)n * n;
    
    // The number of positions where a piece can be placed without being captured
    return total_squares - captured_positions;
}

int main() {
    int n, m;
    cin >> n >> m;
    vector<pair<int, int>> pieces(m);
    for (int i = 0; i < m; ++i) {
        cin >> pieces[i].first >> pieces[i].second;
    }
    cout << countEmptyPositions(n, m, pieces) << endl;
    return 0;
}
