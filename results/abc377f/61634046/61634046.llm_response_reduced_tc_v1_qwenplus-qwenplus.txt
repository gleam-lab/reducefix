```cpp
#include <bits/stdc++.h>

using namespace std;

#define MAX_M 1000

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    int M;
    cin >> N >> M;

    set<int> rows, cols;
    set<long long> diag1, diag2;

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(1LL * a + b);
        diag2.insert(1LL * a - b);
    }

    long long totalEmpty = N * N - M;
    long long attackCount = 0;

    // Each piece attacks its row: |rows| * N
    attackCount += 1LL * rows.size() * N;

    // Each piece attacks its column: |cols| * N
    attackCount += 1LL * cols.size() * N;

    // Each diagonal of type x+y attacks N points
    attackCount += 1LL * diag1.size() * N;

    // Each diagonal of type x-y attacks N points
    attackCount += 1LL * diag2.size() * N;

    // Inclusion-Exclusion correction

    // Row and Column overlaps: each pair (r,c) is counted twice
    attackCount -= 1LL * rows.size() * cols.size();

    // Row and Diag1 overlaps: For each row r and diag s = x+y, the intersection is (r, s-r)
    for (int r : rows) {
        for (auto d : diag1) {
            int c = d - r;
            if (c >= 1 && c <= N) {
                if (rows.find(r) != rows.end() || cols.find(c) != cols.end() ||
                    diag1.find(1LL * r + c) != diag1.end() ||
                    diag2.find(1LL * r - c) != diag2.end()) {
                    continue;
                }
                --attackCount;
            }
        }
    }

    // Row and Diag2 overlaps: For each row r and diag d = x-y, the intersection is (r, r-d)
    for (int r : rows) {
        for (auto d : diag2) {
            int c = r - d;
            if (c >= 1 && c <= N) {
                if (rows.find(r) != rows.end() || cols.find(c) != cols.end() ||
                    diag1.find(1LL * r + c) != diag1.end() ||
                    diag2.find(1LL * r - c) != diag2.end()) {
                    continue;
                }
                --attackCount;
            }
        }
    }

    // Col and Diag1 overlaps: For each col c and diag s = x+y, the intersection is (s-c, c)
    for (int c : cols) {
        for (auto s : diag1) {
            int r = s - c;
            if (r >= 1 && r <= N) {
                if (rows.find(r) != rows.end() || cols.find(c) != cols.end() ||
                    diag1.find(1LL * r + c) != diag1.end() ||
                    diag2.find(1LL * r - c) != diag2.end()) {
                    continue;
                }
                --attackCount;
            }
        }
    }

    // Col and Diag2 overlaps: For each col c and diag d = x-y, the intersection is (d+c, c)
    for (int c : cols) {
        for (auto d : diag2) {
            int r = d + c;
            if (r >= 1 && r <= N) {
                if (rows.find(r) != rows.end() || cols.find(c) != cols.end() ||
                    diag1.find(1LL * r + c) != diag1.end() ||
                    diag2.find(1LL * r - c) != diag2.end()) {
                    continue;
                }
                --attackCount;
            }
        }
    }

    // Diag1 and Diag2 overlaps: For each pair of diagonals, solve system:
    // x+y = s
    // x-y = d
    // => x = (s+d)/2, y = (s-d)/2
    for (auto s : diag1) {
        for (auto d : diag2) {
            if ((s + d) % 2 != 0) continue;
            int x = (s + d) / 2;
            int y = s - x;
            if (x >= 1 && x <= N && y >= 1 && y <= N) {
                if (rows.find(x) != rows.end() || cols.find(y) != cols.end() ||
                    diag1.find(1LL * x + y) != diag1.end() ||
                    diag2.find(1LL * x - y) != diag2.end()) {
                    continue;
                }
                --attackCount;
            }
        }
    }

    // Subtract overlapping regions that were double-counted
    cout << totalEmpty - attackCount << "\n";
    return 0;
}
```
