#include <bits/stdc++.h>

using namespace std;

int main() {
    long long N, M;
    cin >> N >> M;

    set<long long> rows, cols, diag1, diag2;

    for (int i = 0; i < M; ++i) {
        long long a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);       // direction: top-left to bottom-right
        diag2.insert(a + b);       // direction: bottom-left to top-right
    }

    // Total empty cells = N*N - M (since M cells have pieces)
    long long totalEmpty = N * N - M;

    // For inclusion-exclusion principle
    long long attackCount = 0;

    // Count how many cells are attacked due to row, column, diagonal 1 and diagonal 2
    attackCount += rows.size() * N; // all cells in these rows
    attackCount += cols.size() * N; // all cells in these columns
    attackCount += diag1.size() * N; // all cells in these diagonals (a-b)
    attackCount += diag2.size() * N; // all cells in these diagonals (a+b)

    // Subtract overlaps: row & column
    for (auto r : rows) {
        for (auto c : cols) {
            // (r,c) is counted both in row and column, so subtract 1
            attackCount -= 1;
        }
    }

    // Subtract overlaps: row & diagonal1
    for (auto r : rows) {
        for (auto d : diag1) {
            // Solve for c such that r - c == d => c = r - d
            long long c = r - d;
            if (1 <= c && c <= N) attackCount -= 1;
        }
    }

    // Subtract overlaps: row & diagonal2
    for (auto r : rows) {
        for (auto d : diag2) {
            // Solve for c such that r + c == d => c = d - r
            long long c = d - r;
            if (1 <= c && c <= N) attackCount -= 1;
        }
    }

    // Subtract overlaps: column & diagonal1
    for (auto c : cols) {
        for (auto d : diag1) {
            // Solve for r such that r - c == d => r = d + c
            long long r = d + c;
            if (1 <= r && r <= N) attackCount -= 1;
        }
    }

    // Subtract overlaps: column & diagonal2
    for (auto c : cols) {
        for (auto d : diag2) {
            // Solve for r such that r + c == d => r = d - c
            long long r = d - c;
            if (1 <= r && r <= N) attackCount -= 1;
        }
    }

    // Subtract overlaps: diagonal1 & diagonal2
    for (auto d1 : diag1) {
        for (auto d2 : diag2) {
            // Solve for (r, c) such that:
            // r - c == d1
            // r + c == d2
            // => r = (d1 + d2)/2
            // => c = (d2 - d1)/2
            if ((d1 + d2) % 2 != 0) continue;
            long long r = (d1 + d2) / 2;
            long long c = (d2 - d1) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N) attackCount -= 1;
        }
    }

    // Subtract overlaps: row & column & diagonal1/diag2
    // This is already handled by previous steps

    cout << totalEmpty - attackCount << endl;

    return 0;
}
