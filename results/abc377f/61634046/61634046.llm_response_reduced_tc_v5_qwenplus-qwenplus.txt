#include <bits/stdc++.h>

using namespace std;

#define int long long

const int MAX_M = 2000;

int N, M;
set<int> rows, cols;
set<int> diag1, diag2; // diag1: i-j, diag2: i+j

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> N >> M;

    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        rows.insert(a);
        cols.insert(b);
        diag1.insert(a - b);
        diag2.insert(a + b);
    }

    // Total squares not attacked by any row, column, or diagonal
    int total = N * N;

    // Subtract squares attacked by rows and columns
    total -= (rows.size() + cols.size()) * N;

    // Add back intersection of rows and columns (double counted)
    total += rows.size() * cols.size();

    // Subtract diagonals
    for (int d : diag1) {
        int count;
        if (N <= 2 * N - 1) {
            // Number of points on diagonal a-b = d in N x N grid
            int l = max(1LL, d + 1), r = min(N, N + d);
            count = r - l + 1;
        } else {
            count = 0;
        }
        total -= count;
    }

    for (int d : diag2) {
        int count;
        if (2 <= d && d <= 2 * N) {
            // Number of points on diagonal a+b = d in N x N grid
            int l = max(1LL, d - N), r = min(N, d - 1);
            count = r - l + 1;
        } else {
            count = 0;
        }
        total -= count;
    }

    // Add back intersections of rows with diag1 and diag2 (each subtracted twice)
    for (int r : rows) {
        for (int d : diag1) {
            int c = r - d;
            if (1 <= c && c <= N) total += 1;
        }
        for (int d : diag2) {
            int c = d - r;
            if (1 <= c && c <= N) total += 1;
        }
    }

    // Add back intersections of columns with diag1 and diag2 (each subtracted twice)
    for (int c : cols) {
        for (int d : diag1) {
            int r = d + c;
            if (1 <= r && r <= N) total += 1;
        }
        for (int d : diag2) {
            int r = d - c;
            if (1 <= r && r <= N) total += 1;
        }
    }

    // Subtract intersections of diag1 and diag2 (each subtracted twice)
    for (int d1 : diag1) {
        for (int d2 : diag2) {
            int r = (d2 + d1) / 2;
            int c = (d2 - d1) / 2;
            if (1 <= r && r <= N && 1 <= c && c <= N) total -= 1;
        }
    }

    // Now remove the cells that are already occupied
    for (int i = 0; i < M; ++i) {
        int a, b;
        cin >> a >> b;
        bool attacked = false;
        if (rows.count(a) || cols.count(b) || diag1.count(a - b) || diag2.count(a + b)) {
            attacked = true;
        }
        if (attacked) total -= 1;
    }

    cout << total << endl;

    return 0;
}
