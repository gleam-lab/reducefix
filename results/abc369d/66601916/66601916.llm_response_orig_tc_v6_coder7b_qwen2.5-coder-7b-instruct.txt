```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    long long dp0 = 0, dp1 = -2e18;
    cin >> n;
    
    for(int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        dp0 = max(dp1 + 2 * x, dp0);
        dp1 = max(dp0 - x, dp1);
    }
    
    cout << dp0 << endl;
    return 0;
}
```
