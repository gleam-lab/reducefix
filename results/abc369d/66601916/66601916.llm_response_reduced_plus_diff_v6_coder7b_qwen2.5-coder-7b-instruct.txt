#include <vector>
#include <iostream>
#include <algorithm>
#include <limits>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

int main() {
    int n;
    long long dp0 = 0, dp1 = numeric_limits<long long>::min();
    cin >> n;

    for (int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        dp0 = max(dp1 + 2 * x, dp0);
        dp1 = max(dp0 - x, dp1);
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
