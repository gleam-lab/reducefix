```cpp
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e18

int main(){
    int n;
    long long dp0, dp1, x, temp;
    dp0 = 0;
    dp1 = -INF;
    cin >> n;

    for ( int i = 0; i < n; i++) {
        cin >> x;
        temp = dp0;
        dp0 = max(dp1 + 2 * x, dp0); // Corrected the addition of 2x for even-numbered monsters
        dp1 = max(temp + x, dp1);
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
```

**Explanation of Changes:**
- The issue in the original code was that the logic for adding 2X when defeating an even-numbered monster was incorrect. It should have been `dp1 + 2 * x` instead of just `dp1 + x`.
- This change ensures that the correct calculation is made for the experience points gained when defeating even-numbered monsters.
