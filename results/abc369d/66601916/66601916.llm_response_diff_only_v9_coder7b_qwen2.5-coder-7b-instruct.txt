#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e15

int main(){
    int n;
    long long dp0, dp1, x, temp;
    dp0 = 0;
    dp1 = -INF;
    cin >> n;

    for ( int i = 0; i < n; i++) {
        cin >> x;
        temp = dp0;
        dp0 = max(dp1 + x, dp0); // Corrected the coefficient to 1 instead of 2
        dp1 = max(temp, dp1 + x); // Fixed the order of operations
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
