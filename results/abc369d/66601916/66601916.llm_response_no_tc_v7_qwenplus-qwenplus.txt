```cpp
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e18  // Fix: Was too small, caused incorrect comparisons with large values

int main(){
    int n;
    long long dp0 = 0;          // dp0: max profit with no stock at hand
    long long dp1 = -INF;       // dp1: max profit with holding a stock
    long long x;

    cin >> n;

    for (int i = 0; i < n; ++i) {
        cin >> x;
        long long prev_dp0 = dp0;
        dp0 = max(dp0, dp1 + 2 * x);  // Sell a stock we held
        dp1 = max(dp1, prev_dp0 - x); // Buy a stock
    }

    cout << dp0 << endl;  // Final profit must come from not holding any stock
    return 0;
}
```
