```cpp
#include <iostream>
#include <algorithm>
#include <climits>

using namespace std;

int main() {
    int n;
    long long dp[2] = {0, -LLONG_MAX};
    cin >> n;

    for (int i = 0; i < n; ++i) {
        int x;
        cin >> x;
        dp[i % 2] = max(dp[(i + 1) % 2], dp[i % 2] + 2 * x);
    }

    cout << max(dp[0], dp[1]) << endl;
    return 0;
}
```
