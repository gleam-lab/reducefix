#include <vector>
#include <iostream>
#include <algorithm>
#include <climits>

using namespace std;

#define INF LONG_LONG_MIN

int main() {
    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    vector<long long> dp(n + 2, 0);
    dp[0] = 0;
    dp[1] = a[0];
    if (n >= 2) {
        dp[2] = max(a[0], a[1]);
    }
    for (int i = 3; i <= n; ++i) {
        dp[i] = max(dp[i - 1], max(dp[i - 2] + a[i - 1], dp[i - 3] + a[i - 1]));

    cout << dp[n] << endl;
    return 0;
}

1. **Dynamic Programming Array Setup**: We initialize a DP array `dp` where `dp[i]` represents the maximum sum achievable up to the i-th element (1-based index).
2. **Base Cases**:

   - `dp[1] = a[0]`: Only the first element selected.

3. **DP Transitions**:

     - Not selecting the current element: `dp[i - 1]`.
     - Selecting the current element and skipping the previous one: `dp[i - 2] + a[i - 1]`.
     - Selecting the current element and skipping the previous two: `dp[i - 3] + a[i - 1]`.
4. **Result**: The value `dp[n]` gives the maximum sum achievable under the given constraints.
