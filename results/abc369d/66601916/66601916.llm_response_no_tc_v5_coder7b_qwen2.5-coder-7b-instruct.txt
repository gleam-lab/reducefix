#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e18

int main(){
    int n;
    long long dp0 = 0, dp1 = 0; // Initialize dp1 to 0 instead of -INF
    long long x, temp;
    cin >> n;

    for ( int i = 0; i < n; i++) {
        cin >> x;
        temp = dp0;
        dp0 = max(dp1 + 2 * x, dp0); // Correctly update dp0 based on dp1 and current monster's strength
        dp1 = max(temp + x, dp1);   // Correctly update dp1 based on previous dp0 and current monster's strength
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
