```cpp
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e18  // Fix: Previously too small, caused incorrect comparisons

int main(){
    int n;
    long long dp0 = 0;        // dp0: max profit with no stock held
    long long dp1 = -INF;     // dp1: max profit with stock held
    long long x;

    cin >> n;
    vector<long long> prices(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> prices[i];
    }

    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }

    dp1 = -prices[0];  // Correct initial state: bought at first price

    for (int i = 1; i < n; ++i) {
        long long prev_dp0 = dp0;
        dp0 = max(dp0, dp1 + 2 * prices[i]);  // Sell with cooldown (gain 2*price)
        dp1 = max(dp1, prev_dp0 - prices[i]); // Buy new stock
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
```
