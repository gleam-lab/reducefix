#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <numeric>

using namespace std;

#define INF (long long)1e18  // Fixed: Previously too small, leading to incorrect comparisons

int main(){
    int n;
    long long dp0 = 0;           // dp0: max profit with no stock at day i
    long long dp1 = -1e18;       // dp1: max profit with holding a stock at day i
    long long x;

    cin >> n;

    for (int i = 0; i < n; ++i) {
        cin >> x;
        long long prev_dp0 = dp0;
        dp0 = max(dp0, dp1 + 2 * x);  // Sell stock bought earlier
        dp1 = max(dp1, prev_dp0 - x); // Buy stock, spend x
    }

    cout << dp0 << endl;  // Final profit is when we end with no stock held
    return 0;
}
