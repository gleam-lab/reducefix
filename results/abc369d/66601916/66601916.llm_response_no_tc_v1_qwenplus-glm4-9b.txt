
```cpp
#include <iostream>

using namespace std;

#define INF (long long)1e9+5

int main() {
    int n;
    long long dp0 = 0, dp1 = -INF;
    cin >> n;

    for (int i = 0; i < n; i++) {
        long long x;
        cin >> x;
        long long temp = max(dp0, dp1 + 2 * x);
        dp1 = max(dp0, dp1);
        dp0 = temp;
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
```
