#include <bits/stdc++.h>

using namespace std;
using ll = long long;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max total when the i-th element is NOT flipped
    // dp[i][1] = max total when the i-th element IS flipped
    vector<vector<ll>> dp(n + 1, vector<ll>(2, 0));

    // Base cases:
    // For 0 elements, both states have 0 value
    dp[0][0] = dp[0][1] = 0;

    // For 1 element (index 0 in array)
    dp[1][0] = v[0]; // not flipped
    dp[1][1] = v[0]; // flipped (only one element, flipping has no effect)

    for (int i = 1; i < n; ++i)
    {
        // Transition:
        // At step i+1 (i-th element in array), either previous was flipped or not
        // If current is NOT flipped: previous could be flipped and we add v[i]
        // If current is flipped: previous could be not flipped and we add v[i]
        dp[i + 1][0] = max(dp[i][0], dp[i][1] + v[i]);
        dp[i + 1][1] = max(dp[i][1], dp[i][0] + v[i]);
    }

    // The answer is the max of the two states at the end
    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--)
        solve();

    return 0;
}
