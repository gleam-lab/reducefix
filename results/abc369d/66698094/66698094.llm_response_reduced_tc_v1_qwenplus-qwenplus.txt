#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define rall(v) v.rbegin(), v.rend()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max profit considering first i elements when last operation is sell
    // dp[i][1] = max profit considering first i elements when last operation is hold/buy
    ll dp[n + 1][2];
    dp[0][0] = 0;            // No profit at start
    dp[0][1] = -v[0];        // Bought the first item

    for (int i = 1; i < n; ++i)
    {
        dp[i][0] = max(dp[i - 1][1] + v[i], dp[i - 1][0]);
        dp[i][1] = max(dp[i - 1][0] - v[i], dp[i - 1][1]);
    }

    cout << dp[n - 1][0] << '\n';
}

int main()
{
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
