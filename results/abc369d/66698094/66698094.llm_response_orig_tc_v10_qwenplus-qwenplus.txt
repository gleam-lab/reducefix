#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (int i = 0; i < n; ++i)
        cin >> v[i];

    // dp[i][0] = max sum ending at i with even number of elements selected
    // dp[i][1] = max sum ending at i with odd number of elements selected
    vector<vector<ll>> dp(n + 1, vector<ll>(2, -1));
    dp[0][0] = dp[0][1] = 0;

    if (n > 0)
    {
        dp[1][0] = 0;          // not taking first element
        dp[1][1] = v[0];       // taking first element
    }

    for (int i = 2; i <= n; ++i)
    {
        // Case: current element is not taken
        dp[i][0] = max(dp[i - 1][0], dp[i - 1][1]);

        // Case: current element is taken -> it alternates the selection state
        dp[i][1] = dp[i - 1][0] + v[i - 1];
    }

    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main()
{
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
