#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (auto &x : v)
        cin >> x;
    
    // Initialize DP table
    vector<vector<ll>> dp(n + 1, vector<ll>(2));
    dp[0][0] = dp[0][1] = 0;
    
    // Fill DP table
    for (int i = 1; i <= n; ++i) {
        dp[i][0] = max(dp[i][0], dp[i-1][1] + 2 * v[i-1]);
        dp[i][1] = max(dp[i][1], dp[i-1][0] + v[i-1]);
    }
    
    // Output the result
    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    int t = 1;
    cin >> t;
    while (t--) solve();
    return 0;
}
