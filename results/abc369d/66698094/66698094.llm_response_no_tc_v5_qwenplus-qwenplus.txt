```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    if (n == 0) {
        cout << 0 << '\n';
        return;
    }

    vector<vector<ll>> dp(n + 1, vector<ll>(2, 0));

    // Base case: 0 elements
    dp[0][0] = dp[0][1] = 0;

    // First element
    dp[1][0] = v[0]; // not flipped
    dp[1][1] = v[0]; // flipped

    for (int i = 1; i < n; ++i)
    {
        // Transition:
        // dp[i+1][0] -> no flip at i-th element
        // dp[i+1][1] -> flip at i-th element
        dp[i + 1][0] = max(dp[i][0], dp[i][1]);
        dp[i + 1][1] = dp[i][0] + v[i];
    }

    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--)
        solve();

    return 0;
}
```
