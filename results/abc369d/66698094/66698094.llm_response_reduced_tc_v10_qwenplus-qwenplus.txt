#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define rall(v) v.rbegin(), v.rend()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max profit ending at i with even number of elements picked
    // dp[i][1] = max profit ending at i with odd number of elements picked
    ll dp[N][2];
    dp[0][0] = 0;
    dp[0][1] = -1e18;  // Invalid state initially

    for (int i = 1; i <= n; ++i) {
        // Current element is v[i-1]
        // Transition from i-1 to i
        // Case 1: continue previous parity
        // Case 2: start new subarray with current element
        dp[i][0] = max(dp[i-1][0], dp[i-1][1] + v[i-1]);
        dp[i][1] = max(dp[i-1][1], dp[i-1][0] + v[i-1]);
    }

    // The answer is the maximum value over both parities at the end
    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
