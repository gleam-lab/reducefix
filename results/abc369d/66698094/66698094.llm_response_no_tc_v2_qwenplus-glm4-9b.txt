#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;

#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);
const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (int i = 0; i < n; ++i) {
        cin >> v[i];
    }
    vector<ll> dp(2, 0);
    for (ll x : v) {
        vector<ll> new_dp(2, 0);
        new_dp[0] = max(dp[0] + x, dp[1] + 2 * x);
        new_dp[1] = max(dp[1], dp[0] + x);
        dp = new_dp;
    }
    cout << max(dp[0], dp[1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
