#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    ll dp[n + 1];
    dp[0] = 0;
    for (int i = 0; i < n; ++i) {
        cin >> v[i];
        dp[i + 1] = max(dp[i], v[i]);
    }
    for (int i = 1; i < n; ++i) {
        dp[i + 1] = max(dp[i + 1], dp[i] + v[i]);
    }
    cout << dp[n] << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
