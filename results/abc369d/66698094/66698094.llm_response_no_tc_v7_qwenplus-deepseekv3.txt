```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v) cin >> x;
    
    if (n == 1) {
        cout << v[0] << '\n';
        return;
    }
    
    // dp[i][0] = max sum up to i-th element without doubling
    // dp[i][1] = max sum up to i-th element with last element doubled
    vector<vector<ll>> dp(n, vector<ll>(2, -1));
    
    dp[0][0] = v[0];
    dp[0][1] = 2 * v[0];
    
    if (n > 1) {
        dp[1][0] = max(v[0] + v[1], 2 * v[0] + v[1]);
        dp[1][1] = max(v[0] + 2 * v[1], 2 * v[0] + 2 * v[1]);
    }
    
    for (int i = 2; i < n; ++i) {
        dp[i][0] = max(dp[i-1][0] + v[i], dp[i-1][1] + v[i]);
        dp[i][1] = max(dp[i-1][0] + 2 * v[i], dp[i-1][1] + 2 * v[i]);
    }
    
    cout << max(dp[n-1][0], dp[n-1][1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
```

2. Correct state transitions considering both cases (current element doubled or not)
