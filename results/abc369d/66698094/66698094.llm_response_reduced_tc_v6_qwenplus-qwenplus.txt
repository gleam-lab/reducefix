#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max sum considering i elements and picking even number of elements
    // dp[i][1] = max sum considering i elements and picking odd number of elements
    ll dp[n + 1][2];
    dp[0][0] = 0;           // No elements picked, sum is 0
    dp[0][1] = LLONG_MIN;   // Impossible to pick odd number of elements from 0 elements

    for (int i = 1; i <= n; ++i)
    {
        // If we don't take current element, carry forward the best values
        dp[i][0] = dp[i - 1][0];
        dp[i][1] = dp[i - 1][1];

        // If we take current element
        if (dp[i - 1][0] != LLONG_MIN)
            dp[i][1] = max(dp[i][1], dp[i - 1][0] + v[i - 1]);
        if (dp[i - 1][1] != LLONG_MIN)
            dp[i][0] = max(dp[i][0], dp[i - 1][1] + v[i - 1]);
    }

    cout << dp[n][0] << '\n';  // We want maximum sum with even number of elements
}

int main()
{
    fast;
    int t = 1;
    cin >> t;
    while (t--)
        solve();
    return 0;
}
