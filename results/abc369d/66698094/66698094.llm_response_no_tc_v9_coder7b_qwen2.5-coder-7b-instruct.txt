#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define rall(v) v.rbegin(), v.rend()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);
const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (auto &x : v)
        cin >> x;
    
    vector<vector<ll>> dp(n + 1, vector<ll>(2, 0));
    dp[1][0] = v[0]; // Initialize dp for the first element when not taking it
    dp[1][1] = 0;   // Initialize dp for the first element when taking it
    
    for (int i = 1; i < n; ++i) {
        dp[i + 1][0] = max(dp[i][0], dp[i][1] + 2 * v[i]); // Not taking the current element
        dp[i + 1][1] = max(dp[i][1], dp[i][0] + v[i]);    // Taking the current element
    }

    cout << max(dp[n][0], dp[n][1]) << '\n'; // Output the maximum value between not taking or taking the last element
}

int main() {
    fast;
    int t = 1;
    // cin >> t;
    while (t--) 
        solve();
    return 0;
}
