#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max score considering i elements, last element not flipped
    // dp[i][1] = max score considering i elements, last element flipped
    vector<vector<ll>> dp(n + 1, vector<ll>(2, 0));

    // Base cases
    dp[0][0] = dp[0][1] = 0;
    if (n > 0) {
        dp[1][0] = v[0];      // Not flipped
        dp[1][1] = v[0] * 2;  // Flipped
    }

    for (int i = 1; i < n; ++i) {
        // If current element is not flipped, previous must be flipped
        dp[i + 1][0] = dp[i][1] + v[i];
        // If current element is flipped, previous must not be flipped
        dp[i + 1][1] = dp[i][0] + 2 * v[i];
    }

    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main()
{
    fast;
    int t = 1;
    cin >> t;
    while (t--)
        solve();
    return 0;
}
