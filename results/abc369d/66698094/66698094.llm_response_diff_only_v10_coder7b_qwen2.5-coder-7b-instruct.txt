void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    ll dp[n + 1][2];
    for (ll &x : v)
        cin >> x;
    for (int i = 0; i <= n; ++i)
    {
        dp[i][0] = dp[i][1] = 0; // Initialize to 0 instead of -1
    }
    dp[1][1] = v[0]; // Base case for the first element
    dp[1][0] = 0;
    for (int i = 1; i < n; ++i)
    {
        dp[i + 1][0] = max(dp[i + 1][0], dp[i][1] + 2 * v[i]); // Correct the index access
        dp[i + 1][1] = max(dp[i + 1][1], dp[i][0] + v[i]);
    }
    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main()
{
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}

### Explanation of Changes
1. **Initialization**: Changed `dp[i][0]` and `dp[i][1]` initialization from `-1` to `0`. This is because we are looking for maximum values, and starting with `0` allows us to add values directly without underflow issues.
2. **Base Case Correction**: The base case for `dp[1][1]` was incorrectly set to `v[0]`, which is correct as it represents choosing the first element. However, `dp[1][0]` should be initialized to `0` since not choosing any elements at the start leads to zero value.
3. **Index Access Fix**: Ensured that the loop correctly accesses the array indices by removing unnecessary type casting and ensuring proper indexing.
