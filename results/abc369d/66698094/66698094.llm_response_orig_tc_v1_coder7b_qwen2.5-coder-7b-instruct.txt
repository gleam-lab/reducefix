```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using vi = vector<int>;

void solve() {
    int n;
    cin >> n;
    vi v(n);
    for(int& x : v) cin >> x;
    
    // Initialize DP table with two states: whether the last element was taken or not
    vector<vector<ll>> dp(n+1, vector<ll>(2, 0));
    
    for(int i = 1; i <= n; ++i) {
        dp[i][0] = max(dp[i-1][0], dp[i-1][1] + v[i-1]); // Not taking the current element
        dp[i][1] = dp[i-1][0] + 2 * v[i-1]; // Taking the current element
    }
    
    // The answer is the maximum of either taking or not taking the last element
    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
    int t = 1;
    cin >> t;
    while(t--) solve();
    return 0;
}
```

This solution corrects the logic of the dynamic programming approach by properly handling the states and transitions. It ensures that the maximum value is calculated correctly by considering both possibilities at each step (taking or not taking the current element). The time complexity remains O(n), which is efficient enough for large inputs as specified.
