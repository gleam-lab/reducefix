```cpp
#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<ll>;
#define all(v) v.begin(), v.end()
#define rall(v) v.rbegin(), v.rend()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

void solve() {
    int n;
    cin >> n;
    vi v(n);
    for (ll &x : v) cin >> x;

    if (n == 1) {
        cout << v[0] << '\n';
        return;
    }

    vector<vector<ll>> dp(n + 1, vector<ll>(2, 0));
    dp[1][0] = v[0];
    dp[1][1] = 0;

    for (int i = 2; i <= n; ++i) {
        dp[i][0] = max(dp[i-1][0], dp[i-1][1]) + v[i-1];
        dp[i][1] = dp[i-1][0] + 2 * v[i-1];
    }

    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    while (t--)
        solve();
    return 0;
}
```

   - For `dp[i][1]`, the sum is derived by doubling the current element and adding it to the sum where the previous element was not doubled (to avoid consecutive doublings).
4. **Result**: The final result is the maximum value between the two states after processing all elements.

This approach ensures that all possible sequences are considered, and the maximum sum is achieved without violating the constraints. The time complexity is O(N), which is efficient for large inputs.
