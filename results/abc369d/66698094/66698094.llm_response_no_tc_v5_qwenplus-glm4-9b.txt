#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
using ll = long long;
using vi = vector<int>;

#define all(v) v.begin(), v.end()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);
const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    for (int i = 0; i < n; ++i) {
        cin >> v[i];
    }
    vector<ll> dp(n + 1);
    dp[0] = 0;
    dp[1] = v[0];
    for (int i = 1; i < n; ++i) {
        dp[i + 1] = max(dp[i], v[i] + 2 * dp[i - 1]);
    }
    cout << max(dp[n], dp[n - 1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
