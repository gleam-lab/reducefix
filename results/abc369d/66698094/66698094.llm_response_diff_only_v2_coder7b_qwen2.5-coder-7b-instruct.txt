#include <bits/stdc++.h>

#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve() {
    int n;
    cin >> n;
    vector<ll> v(n);
    ll dp[n + 1][2];
    
    for (auto &x : v) 
        cin >> x;
    
    for (int i = 0; i <= n; ++i) 
        dp[i][0] = dp[i][1] = 0;
    
    dp[1][0] = v[0];
    dp[1][1] = 0;
    
    for (int i = 1; i < n; ++i) {
        dp[i + 1][0] = max(dp[i][0], dp[i][1] + v[i]);
        dp[i + 1][1] = max(dp[i][1], dp[i][0] + 2 * v[i]);
    }

    cout << max(dp[n][0], dp[n][1]) << '\n';
}

int main() {
    fast;
    int t = 1;
    while (t--) 
        solve();
    return 0;
}
