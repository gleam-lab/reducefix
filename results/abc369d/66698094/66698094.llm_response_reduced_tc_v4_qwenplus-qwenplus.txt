#include <bits/stdc++.h>

using namespace std;
using ll = long long;
using vi = vector<int>;
#define all(v) v.begin(), v.end()
#define fast ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0);

const int MOD = 1e9 + 7, N = 2e5 + 5;

void solve()
{
    int n;
    cin >> n;
    vector<ll> v(n);
    for (ll &x : v)
        cin >> x;

    // dp[i][0] = max profit if we buy/sell up to i days and end with no stock
    // dp[i][1] = max profit if we end with 1 stock
    ll dp_prev[2] = {0, -v[0]};
    for (int i = 1; i < n; ++i)
    {
        ll new_dp0 = max(dp_prev[0], dp_prev[1] + v[i]);
        ll new_dp1 = max(dp_prev[1], dp_prev[0] - v[i]);
        dp_prev[0] = new_dp0;
        dp_prev[1] = new_dp1;
    }
    cout << dp_prev[0] << '\n';
}

int main()
{
    fast;
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
