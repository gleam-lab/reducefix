#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i = 0; i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    ll dp_even = 0, dp_odd = 0, temp_even, temp_odd;

    rep(i, n)
    {
        if (i % 2 == 0) // process even index
        {
            temp_even = dp_even;
            dp_even = max(dp_odd + 2 * a[i], dp_even);
            dp_odd = max(temp_even + a[i], dp_odd);
        }
        else // process odd index
        {
            temp_odd = dp_odd;
            dp_odd = max(dp_even + 2 * a[i], dp_odd);
            dp_even = max(temp_odd + a[i], dp_even);
        }
    }

    cout << max(dp_even, dp_odd) << endl;
}
