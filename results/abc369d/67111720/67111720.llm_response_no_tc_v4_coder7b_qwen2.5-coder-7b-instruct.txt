#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (long long i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // Initialize variables to store the maximum experience points if the last defeated monster was even or odd
    ll dp_even = 0, dp_odd = -1;
    ll temp;

    rep(i,n)
    {
        // Store the current value of dp_even before updating
        temp = dp_even;
        
        // Update dp_even: take the maximum of either not defeating the current monster or defeating it and adding its strength twice (for even index)
        dp_even = max(dp_odd + 2 * a[i], dp_even);

        // Update dp_odd: take the maximum of either not defeating the current monster or defeating it and adding its strength once (for odd index)
        dp_odd = max(temp + a[i], dp_odd);
    }

    // Output the maximum experience points possible
    cout << max(dp_even, dp_odd) << endl;
}
