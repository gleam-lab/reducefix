```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // dp_even: max sum ending at even index
    // dp_odd: max sum ending at odd index
    ll dp_even = 0, dp_odd = -1e18;

    rep(i,n)
    {
        ll temp_even = dp_even, temp_odd = dp_odd;
        dp_even = max(temp_even + a[i], temp_odd + 2*a[i]);
        dp_odd = max(temp_odd + a[i], temp_even);
    }

    cout << max(dp_even, dp_odd) << endl;
}
```
