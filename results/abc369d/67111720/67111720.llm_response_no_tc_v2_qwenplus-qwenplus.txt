```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // dp_e: max sum ending at even number of elements picked
    // dp_o: max sum ending at odd number of elements picked
    ll dp_e = 0, dp_o = -1, temp;

    rep(i,n)
    {
        temp = dp_e;
        // If we don't pick current element: dp_e remains same
        // If we pick current element:
        // - if previous was even: still even (add 2*a[i])
        // - if previous was odd: becomes even (add a[i])
        dp_e = max(dp_e, max(dp_o + a[i], temp + 2*a[i]));

        // If we don't pick current element: dp_o remains same
        // If we pick current element:
        // - if previous was even: becomes odd (add a[i])
        // - if previous was odd: can't pick again unless even count
        dp_o = max(dp_o, temp + a[i]);
    }

    cout << max(dp_e, dp_o) << endl;
}
```
