```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    ll dp_e = 0, dp_o = -1e18; // Initialize dp_o to negative infinity to handle cases where taking odd number of elements isn't possible yet

    rep(i,n)
    {
        ll temp = dp_e;
        dp_e = max(dp_e + 2 * a[i], dp_e); // We can either take the current element in even count or not
        dp_o = max(temp + a[i], dp_o);     // We can start with odd count using this element or continue with previous odd count
    }

    cout << max(dp_e, dp_o) << endl;
}
```
