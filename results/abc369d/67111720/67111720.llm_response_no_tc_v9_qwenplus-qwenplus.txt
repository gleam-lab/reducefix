```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0; i < (n); i++)
using ll = long long;

int main()
{
    int n; 
    cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // dp_e: max sum ending at even number of elements taken
    // dp_o: max sum ending at odd number of elements taken
    ll dp_e = 0, dp_o = -1, temp;

    rep(i,n)
    {
        temp = dp_e;
        // If we don't take current element, dp_e remains same
        // If we take current element and were previously at odd length, then we add it to make even
        dp_e = max(dp_e, dp_o + 2 * a[i]);
        
        // If we don't take current element, dp_o remains same
        // If we take current element and were previously at even length, then we add it as odd length
        dp_o = max(dp_o, temp + a[i]);
    }

    // Final result is the maximum of sums ending in even or odd length
    cout << max(dp_e, dp_o) << endl;
}
```
