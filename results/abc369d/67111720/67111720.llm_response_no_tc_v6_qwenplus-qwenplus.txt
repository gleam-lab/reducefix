```cpp
#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // dp_e: max sum ending at even number of elements picked
    // dp_o: max sum ending at odd number of elements picked
    ll dp_e = 0, dp_o = -1, temp;

    rep(i,n)
    {
        temp = dp_e;
        // If we don't take current element: dp_e remains same
        // If we take current element:
        // - if previous was odd count: dp_o + 2*a[i] (since last element was added once, now it's doubled)
        dp_e = max(dp_e, dp_o + 2*a[i]);
        
        // If we don't take current element: dp_o remains same
        // If we take current element:
        // - and previous was even count: dp_e (before update) + a[i]
        dp_o = max(dp_o, temp + a[i]);
    }

    cout << max(dp_e, dp_o) << endl;
}
```
