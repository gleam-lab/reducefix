#include<bits/stdc++.h>
using namespace std;

#define rep(i,n) for (ll i=0;i < (n); i++)
using ll = long long;

int main()
{
    int n; cin >> n;
    vector<ll> a(n);
    rep(i,n) cin >> a[i];

    // Initialize dp arrays to handle even and odd indexed elements separately
    vector<ll> dp_e(n, 0), dp_o(n, -1);

    rep(i,n)
    {
        if (i % 2 == 0)
        {
            dp_e[i] = max((i > 0 ? dp_o[i-1] : -1) + 2 * a[i], dp_e[i]);
            dp_o[i] = max(dp_e[i], (i > 0 ? dp_o[i-1] : -1) + a[i]);
        }
        else
        {
            dp_o[i] = max((i > 0 ? dp_e[i-1] : -1) + 2 * a[i], dp_o[i]);
            dp_e[i] = max(dp_o[i], (i > 0 ? dp_e[i-1] : -1) + a[i]);
        }
    }

    // The result is the maximum value between the last elements of both dp arrays
    cout << max(dp_e[n-1], dp_o[n-1]) << endl;
}
