ll exp(int i = 0, bool parity = 1) {
    if (i == n)
        return 0;

    if (~dp[i][parity])
        return dp[i][parity];

    ll take = exp(i + 1, !parity) + (parity ? v[i] : 2 * v[i]);
    ll leave = exp(i + 1, parity);

    return dp[i][parity] = max(take, leave);
}

    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    memset(dp, -1, sizeof dp);

}
