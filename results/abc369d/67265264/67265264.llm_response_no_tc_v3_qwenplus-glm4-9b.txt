
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
#define ll long long
#define mod 1000000007
const int N = 1e5 + 5;

int n;
vector<ll> v;
ll dp[N][2];

ll exp(int i = 0, bool parity = 1) {
    if (i == n)
        return 0;
    if (dp[i][parity] != -1)
        return dp[i][parity];
    ll take = exp(i + 1, !parity) + (parity ? v[i] : 2 * v[i]);
    ll leave = exp(i + 1, parity);
    dp[i][parity] = max(take, leave);
    return dp[i][parity];
}

void solve() {
    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    memset(dp, -1, sizeof dp);
    cout << exp();
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int q = 1;
    cin >> q;
    while (q--) {
        solve();
    }
    return 0;
}
```

### Explanation:
1. **Initialization of dp array**: The dp array is initialized with `-1` to signify that the subproblems have not been solved yet.
2. **Base case**: When `i == n`, the recursion returns `0` since there are no more elements to process.
3. **Memoization**: Before computing the result for `dp[i][parity]`, the function checks if it has already been computed and stored in `dp`. If so, it returns the stored value to avoid redundant calculations.
4. **Recursive calls**: The function computes two scenarios - taking the current element or leaving it (hence the `parity` parameter). The `parity` is used to decide whether to multiply by 2 or not.
5. **Final output**: The result of `exp()` is printed inside the loop in `solve()` function.
6. **Input and output optimization**: The `ios_base::sync_with_stdio(0);` and `cin.tie(0);cout.tie(0);` lines are used to make the I/O operations faster.
