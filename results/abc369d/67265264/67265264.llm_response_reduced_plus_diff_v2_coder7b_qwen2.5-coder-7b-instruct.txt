#define ll long long

const int N = 1e5 + 5;
int n;
vector<ll> v;
ll dp[N][2];

ll exp(int i = 0, bool parity = 1) {
    if (i == n)
        return 0;

    if (~dp[i][parity])
        return dp[i][parity];

    ll take = exp(i + 1, !parity) + (parity ? v[i] : 2 * v[i]);
    ll leave = exp(i + 1, parity);

    return dp[i][parity] = max(take, leave);
}

void solve() {
    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    memset(dp, -1, sizeof dp);
    cout << exp() << endl;
}

This solution uses dynamic programming to solve the problem efficiently. The function `exp` is used to calculate the maximum sum by either taking or leaving each element based on its parity. The use of memoization (`dp` array) ensures that each subproblem is solved only once, resulting in an overall time complexity of O(N). This approach handles all given test cases, including large inputs.
