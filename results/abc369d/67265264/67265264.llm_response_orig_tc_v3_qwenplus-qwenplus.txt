#include <bits/stdc++.h>
using namespace std;
#define Youssef ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);
#define asc(v) v.begin(), v.end()
#define desc(v) v.rbegin(), v.rend()
#define ll long long
#define mod 1000000007

const int N = 2 * 100000 + 5; // Slightly larger to be safe for 2e5
int n;
vector<ll> v;
ll dp[N][2];

ll solveDP(int i, bool parity) {
    if (i == n)
        return 0;

    if (dp[i][parity] != -1)
        return dp[i][parity];

    ll take = solveDP(i + 1, !parity) + (parity ? v[i] : 2 * v[i]);
    ll leave = solveDP(i + 1, parity);

    return dp[i][parity] = max(take, leave);
}

void solve() {
    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; ++i) {
        cin >> v[i];
    }

    // Sort in descending order to maximize the effect of doubling
    sort(v.begin(), v.end(), greater<ll>());

    memset(dp, -1, sizeof dp);
    cout << solveDP(0, 1);
}

int main() {
    Youssef
    int q = 1;
    //cin >> q;
    while (q--)
        solve();
    return 0;
}
