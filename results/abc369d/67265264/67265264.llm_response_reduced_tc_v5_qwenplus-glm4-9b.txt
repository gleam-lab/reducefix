#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define ll long long
#define mod 1000000007
const int N = 1e5 + 5;

int n;
vector<ll> v;
vector<vector<ll>> dp(N + 1, vector<ll>(2, -1)); // 2 states: parity even or odd

ll exp(int i, bool parity) {
    if (i == n)
        return 0;

    if (dp[i][parity] != -1)
        return dp[i][parity];

    ll take = exp(i + 1, !parity) + (parity ? v[i] : 2 * v[i]);
    ll leave = exp(i + 1, parity);

    return dp[i][parity] = max(take, leave);
}

void solve() {
    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    cout << exp(0, 1) << endl; // Start the recursion with index 0 and parity 1
}

int main() {
    int q;
    cin >> q;
    while (q--) {
        solve();
    }
    return 0;
}
