#include <bits/stdc++.h>

#define ll long long
#define mod 1000000007
const int N = 1e5 + 5;
int n;
vector<ll> v;
ll dp[N][2];

ll exp(int i = 0, bool parity = 0) { // corrected initial parity (0 for even index count starting from 0)
    if (i == n)
        return 0;

    if (~dp[i][parity])
        return dp[i][parity];

    ll take = exp(i + 1, !parity) + (parity ? v[i] : -v[i]); // corrected the reward logic
    ll leave = exp(i + 1, parity);

    return dp[i][parity] = max(take, leave);
}

void solve() {
    cin >> n;
    v.assign(n, 0);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    memset(dp, -1, sizeof dp);
    cout << exp(); // this will now correctly compute the maximum alternating sum
}

    //cin >> q;
    while (q--)
        solve();
