#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;

ll a[MAXN], f[MAXN][2];
int cnt[MAXN][2]; // count of take operations in each state

int main(){
    ll n;
    scanf("%lld", &n);
    for(ll i = 1; i <= n; ++i)
        scanf("%lld", &a[i]);

    for(ll i = 1; i <= n; ++i){
        // State transition: either don't take the i-th element
        if(f[i-1][0] > f[i-1][1]){
            f[i][0] = f[i-1][0];
            cnt[i][0] = cnt[i-1][0];
        } else {
            f[i][0] = f[i-1][1];
            cnt[i][0] = cnt[i-1][1];
        }

        // State transition: take the i-th element
        int prev_cnt_odd = cnt[i-1][0] % 2;
        ll val1 = (prev_cnt_odd == 0) ? a[i]*2 : a[i];
        ll total1 = f[i-1][0] + val1;

        prev_cnt_odd = cnt[i-1][1] % 2;
        ll val2 = (prev_cnt_odd == 0) ? a[i]*2 : a[i];
        ll total2 = f[i-1][1] + val2;

        if(total1 > total2){
            f[i][1] = total1;
            cnt[i][1] = cnt[i-1][0] + 1;
        } else {
            f[i][1] = total2;
            cnt[i][1] = cnt[i-1][1] + 1;
        }
    }

    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
