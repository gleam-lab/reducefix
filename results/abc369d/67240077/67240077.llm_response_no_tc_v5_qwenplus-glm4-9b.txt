```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
ll a[MAXN], f[MAXN][2], cnt[MAXN][2];
int main(){
    ll n;
    scanf("%lld",&n);
    for(ll i = 1; i <= n; i++){
        scanf("%lld",&a[i]);
    }
    f[0][0] = 0; // Initialize for the first element
    cnt[0][0] = 0;
    f[0][1] = a[1]; // Initialize for the first element assuming single element groups
    cnt[0][1] = 1;
    
    for(ll i = 1; i <= n; i++){
        f[i][0] = max(f[i-1][0], f[i-1][1]);
        cnt[i][0] = f[i-1][0] > f[i-1][1] ? cnt[i-1][0] : cnt[i-1][1];
        
        f[i][1] = max(f[i-1][0] + 2 * a[i], f[i-1][1] + a[i]);
        cnt[i][1] = f[i-1][0] + 2 * a[i] > f[i-1][1] + a[i] ? cnt[i-1][0] + 1 : cnt[i-1][1] + 1;
    }
    
    printf("%lld", max(f[n][0], f[n][1]));
    return 0;
}
```

### Explanation:
1. **Initialization**: Initially, `f[0][0]` is set to 0 since there's no group, and `f[0][1]` is set to the value of the first element since it forms a single-element group.
2. **Edge Case**: When considering transitions from `f[i-1][0]` to `f[i][1]`, the calculation `f[i-1][0] + 2 * a[i]` vs `f[i-1][1] + a[i]` is crucial. This reflects whether adding the current element to a larger group (doubled) or continuing with the smaller group (not doubled) gives a better result.
3. **Correctness and Efficiency**: The code ensures that each state transition is clear and correctly implements the logic for grouping elements either in pairs or singly based on the previously computed values.
