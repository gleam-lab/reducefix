#include <bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN = 2e5 + 10;

ll a[MAXN], f[MAXN][2];
ll cnt[MAXN][2]; // cnt[i][j] stores the number of selected elements in the optimal solution up to i ending with state j

int main() {
    ll n;
    scanf("%lld", &n);
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }

    cnt[0][0] = 0; // base case: no elements selected
    cnt[0][1] = 1; // base case: one element selected (hypothetically)

    for (ll i = 1; i <= n; i++) {
        // State 0: not taking the current element
        if (f[i - 1][0] > f[i - 1][1]) {
            f[i][0] = f[i - 1][0];
            cnt[i][0] = cnt[i - 1][0];
        } else {
            f[i][0] = f[i - 1][1];
            cnt[i][0] = cnt[i - 1][1];
        }

        // State 1: taking the current element
        ll prev_cnt0 = cnt[i - 1][0], prev_cnt1 = cnt[i - 1][1];
        ll num0 = (prev_cnt0 % 2 == 0) ? a[i] * 2 : a[i];
        ll num1 = (prev_cnt1 % 2 == 0) ? a[i] * 2 : a[i];

        ll val0 = f[i - 1][0] + num0;
        ll val1 = f[i - 1][1] + num1;

        if (val0 > val1) {
            f[i][1] = val0;
            cnt[i][1] = prev_cnt0 + 1;
        } else {
            f[i][1] = val1;
            cnt[i][1] = prev_cnt1 + 1;
        }
    }

    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
