#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
ll a[MAXN], f[MAXN][2], cnt[MAXN][2];
int main(){
    ll n;
    scanf("%lld",&n);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
    }
    f[1][0] = a[1]; // Initialize the first element correctly
    cnt[1][0] = 1;  // Initialize the count for the first element
    for(ll i=2; i<=n; i++){
        // Calculate the maximum of previous two or one from previous day
        f[i][0] = max(f[i-1][0], f[i-1][1]);
        // If using previous two, carry forward the count of the larger of the two
        if(f[i-1][0] > f[i-1][1]) cnt[i][0] = cnt[i-1][0];
        else cnt[i][0] = cnt[i-1][1];
        // Calculate the total cost if choosing to use the current element
        ll cost1 = f[i-1][0] + (cnt[i-1][0] % 2 == 0 ? a[i] * 2 : a[i]);
        // Calculate the total cost if choosing to use the current element and the previous one
        ll cost2 = f[i-1][1] + (cnt[i-1][1] % 2 == 0 ? a[i] * 2 : a[i]);
        // Choose the maximum of the two costs
        f[i][1] = max(cost1, cost2);
        // Update the count based on which choice gives the maximum cost
        cnt[i][1] = (cost1 > cost2) ? cnt[i-1][0] + 1 : cnt[i-1][1] + 1;
    }
    // Output the maximum of the final two choices
    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
