typedef long long ll;
const int MAXN=2e5+10;
ll a[MAXN], f[MAXN][2], cnt[MAXN][2];

int main() {
    ll n;
    scanf("%lld", &n);
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }

    for (ll i = 1; i <= n; i++) {
        f[i][0] = max(f[i - 1][0], f[i - 1][1]);
        cnt[i][0] = f[i][0] == f[i - 1][0] ? cnt[i - 1][0] : cnt[i - 1][1];
        
        ll cnt1 = cnt[i - 1][0] + 1;
        ll cnt2 = cnt[i - 1][1] + 1;
        ll num1 = cnt1 % 2 == 0 ? a[i] * 2 : a[i];
        ll num2 = cnt2 % 2 == 0 ? a[i] * 2 : a[i];
        
        f[i][1] = max(f[i - 1][0] + num1, f[i - 1][1] + num2);
        cnt[i][1] = f[i - 1][0] + num1 > f[i - 1][1] + num2 ? cnt1 : cnt2;
    }

    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
