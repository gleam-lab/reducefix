#define INF 0x3f3f3f3f3f3f3f3f

typedef long long ll;
const int MAXN=2e5+10;
ll a[MAXN],f[MAXN][2],last_parity[MAXN][2];

    scanf("%lld",&n);
    for(ll i=1;i<=n;i++){

        f[i][0] = max(f[i-1][0], f[i-1][1]);
        if (f[i-1][0] > f[i-1][1]) {
