#include <bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN = 2e5 + 10;

ll a[MAXN], f[MAXN][2];
int cnt[MAXN][2]; // count of selected elements in each state

int main() {
    ll n;
    scanf("%lld", &n);
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
    }

    for (ll i = 1; i <= n; i++) {
        // Option: not taking the current element
        if (f[i-1][0] > f[i-1][1]) {
            f[i][0] = f[i-1][0];
            cnt[i][0] = cnt[i-1][0];
        } else {
            f[i][0] = f[i-1][1];
            cnt[i][0] = cnt[i-1][1];
        }

        // Option: taking the current element
        ll num1, num2;
        ll cnt1 = cnt[i-1][0] + 1;
        ll cnt2 = cnt[i-1][1] + 1;

        if (cnt1 % 2 == 0) num1 = a[i] * 2;
        else num1 = a[i];

        if (cnt2 % 2 == 0) num2 = a[i] * 2;
        else num2 = a[i];

        ll val1 = f[i-1][0] + num1;
        ll val2 = f[i-1][1] + num2;

        if (val1 > val2) {
            f[i][1] = val1;
            cnt[i][1] = cnt1;
        } else {
            f[i][1] = val2;
            cnt[i][1] = cnt2;
        }
    }

    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
