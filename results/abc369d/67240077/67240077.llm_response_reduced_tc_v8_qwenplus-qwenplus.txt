#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;

ll a[MAXN];
ll f[MAXN][2];      // f[i][0]: max sum up to i without selecting i as doubled
                    // f[i][1]: max sum up to i selecting i as doubled
ll cnt[MAXN][2];    // count of selected elements in each case

int main(){
    ll n;
    scanf("%lld",&n);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
    }

    // Initialization for the first element
    f[1][0] = a[1];       // not doubling the first element
    f[1][1] = a[1] * 2;   // doubling the first element
    cnt[1][0] = 1;        // count of elements chosen without doubling
    cnt[1][1] = 1;        // count of elements chosen with doubling

    for(ll i=2;i<=n;i++){
        // Case: not doubling current element
        if(f[i-1][0] > f[i-1][1]){
            f[i][0] = f[i-1][0];
            cnt[i][0] = cnt[i-1][0];
        } else {
            f[i][0] = f[i-1][1];
            cnt[i][0] = cnt[i-1][1];
        }

        // Case: doubling current element
        ll val1 = f[i-1][0] + a[i];
        ll val2 = f[i-1][1] + (cnt[i-1][1] % 2 == 0 ? a[i] : a[i] * 2);
        ll new_cnt1 = cnt[i-1][0] + 1;
        ll new_cnt2 = cnt[i-1][1] + 1;

        if(val1 > val2 || (val1 == val2 && new_cnt1 < new_cnt2)){
            f[i][1] = val1;
            cnt[i][1] = new_cnt1;
        } else {
            f[i][1] = val2;
            cnt[i][1] = new_cnt2;
        }
    }

    printf("%lld\n", max(f[n][0], f[n][1]));
    return 0;
}
