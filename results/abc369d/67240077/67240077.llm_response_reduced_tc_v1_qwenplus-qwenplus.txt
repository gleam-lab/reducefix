#include<bits/stdc++.h>
#define INF 0x3f3f3f3f3f3f3f3f
using namespace std;
typedef long long ll;
const int MAXN=2e5+10;
ll a[MAXN];
struct State {
    ll val[2];  // 0: not using current, 1: using current
    ll cnt[2];  // count of operations for each state
} dp[MAXN];

int main(){
    ll n;
    scanf("%lld",&n);
    for(ll i=1;i<=n;i++){
        scanf("%lld",&a[i]);
    }
    
    dp[0].val[0] = 0;
    dp[0].cnt[0] = 0;
    dp[0].val[1] = -INF;  // Impossible state
    dp[0].cnt[1] = 0;
    
    for(ll i=1;i<=n;i++){
        // Case 1: take the previous best without current operation
        dp[i].val[0] = max(dp[i-1].val[0], dp[i-1].val[1]);
        
        // Transfer count based on which value is greater
        if(dp[i-1].val[0] > dp[i-1].val[1]) {
            dp[i].cnt[0] = dp[i-1].cnt[0];
        } else {
            dp[i].cnt[0] = dp[i-1].cnt[1];
        }
        
        // Case 2: extend the operation sequence
        ll choice1 = dp[i-1].val[0] + (dp[i-1].cnt[0] % 2 == 0 ? 2 : 1) * a[i];
        ll choice2 = dp[i-1].val[1] + (dp[i-1].cnt[1] % 2 == 0 ? 2 : 1) * a[i];
        
        if(choice1 > choice2) {
            dp[i].val[1] = choice1;
            dp[i].cnt[1] = dp[i-1].cnt[0] + 1;
        } else {
            dp[i].val[1] = choice2;
            dp[i].cnt[1] = dp[i-1].cnt[1] + 1;
        }
    }
    
    printf("%lld\n", max(dp[n].val[0], dp[n].val[1]));
    return 0;
}
