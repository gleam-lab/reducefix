const int MAXN=2e5+10;
ll a[MAXN],f[MAXN][3];

    f[0][0] = 0;

        f[i][1] = f[i-1][0] + a[i];
        f[i][2] = f[i-1][1] + a[i] * 2;

    printf("%lld",max(f[n][0], max(f[n][1], f[n][2])));
    return 0;
