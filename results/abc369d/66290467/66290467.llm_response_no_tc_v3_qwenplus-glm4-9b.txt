#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

#define rep(i, n) for (int i = 0; i < n; ++i)

int main() {
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    } else if (n == 2) {
        cout << a[0] + 2 * a[1] << endl;
        return 0;
    }

    vector<ll> dp(n, 0); // i盲陆聯莽聸庐茫聜聮氓聙聮茫聛聶茫聛戮茫聛搂茫聛庐忙聹聙氓陇搂茫聛庐莽碌聦茅篓聯氓聙陇
    dp[0] = a[0];
    dp[1] = dp[0] + 2 * a[1];

    for (int i = 2; i < n; ++i) {
        dp[i] = max(dp[i - 2] + 2 * a[i], dp[i - 1] + a[i]);
    }
    cout << dp[n - 1] << endl;

    return 0;
}
