#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (int(i) = 0; (i) < (n); ++(i))
#define drep(i, n) for (int(i) = (n) - 1; (i) >= 0; --(i))

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> a(n, 0);
    rep(i, n) cin >> a[i];

    if (n <= 2) {
        cout << accumulate(a.begin(), a.end(), 0LL) << '\n';
        return 0;
    }

    vector<ll> dp(n, 0);
    dp[0] = a[0];
    dp[1] = dp[0] + 2 * a[1];

    rep(i, 2, n) {
        dp[i] = max(dp[i-2] + 2 * a[i], dp[i-1]);
    }

    cout << dp[n-1] << '\n';

    return 0;
}
