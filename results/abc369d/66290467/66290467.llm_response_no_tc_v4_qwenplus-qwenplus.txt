```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (int(i) = 0; (i) < (n); ++(i))
#define drep(i, n) for (int(i) = (n) - 1; (i) >= 0; --(i))

int main() {
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    // Early return for small cases
    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    }

    vector<ll> dp(n, 0); // dp[i] = max experience up to monster i

    dp[0] = a[0]; // Only one monster
    dp[1] = max(a[0] + a[1], a[1]); // Can skip first or attack both

    for (int i = 2; i < n; ++i) {
        // Option 1: Skip current monster, add normal exp from previous
        // Option 2: Attack current and previous, add double exp
        dp[i] = max(dp[i - 1] + a[i], dp[i - 2] + 2 * a[i]);
    }

    cout << dp[n - 1] << endl;

    return 0;
}
```
