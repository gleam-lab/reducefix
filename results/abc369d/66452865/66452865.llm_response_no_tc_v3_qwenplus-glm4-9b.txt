
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N);
    for (int i = 0; i < N; i++) cin >> A[i];
    vector<long long> dp(2);
    dp[0] = 0;
    dp[1] = 0;
    for (int i = 0; i < N; i++) {
        vector<long long> p(2);
        swap(dp, p);
        for (int j = 0; j < 2; j++) {
            // 茅聙聝茫聛聶茫聛篓茫聛聧
            dp[j] = max(dp[j], p[j]);
            // 氓聙聮茫聛聶茫聛篓茫聛聧
            dp[j ^ 1] = max(dp[j ^ 1], p[j] + A[i]);
            if (j == 1 && i > 1) dp[j ^ 1] = max(dp[j ^ 1], p[j] + 2 * A[i]);  // 盲驴庐忙颅拢: i>0 <- i>1
        }
    }
    cout << max(dp[0], dp[1]) << endl;
    return 0;
}
```
