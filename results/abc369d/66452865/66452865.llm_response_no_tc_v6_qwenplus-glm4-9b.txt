#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<long long> A(N);
    for (int i = 0; i < N; i++) cin >> A[i];
    vector<long long> dp(2);
    dp[0] = 0;
    dp[1] = 1;  // Initialize dp[1] to 1 since the minimum cost when choosing is 1
    for (int i = 0; i < N; i++) {
        vector<long long> next_dp(2);
        next_dp[0] = max(dp[0], dp[1]);
        next_dp[1] = max(dp[1], dp[0] + A[i]);
        if (i > 0) next_dp[1] = max(next_dp[1], dp[1] + 2 * A[i]);  // Allow double the cost if i > 1
        dp = next_dp;
    }
    cout << max(dp[0], dp[1]) << endl;
    return 0;
}
