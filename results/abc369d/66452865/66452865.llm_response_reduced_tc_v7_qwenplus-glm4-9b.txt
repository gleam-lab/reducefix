
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  vector<long long> A(N);
  for (int i = 0; i < N; i++) cin >> A[i];
  vector<long long> dp(2);
  dp[0] = 0;
  dp[1] = 0;
  for (int i = 0; i < N; i++) {
    vector<long long> next_dp(2);
    for (int j = 0; j < 2; j++) {
      // 茅聙聝茫聛聶茫聛篓茫聛聧
      next_dp[j] = max(next_dp[j], dp[j]);
      // 氓聙聮茫聛聶茫聛篓茫聛聧
      next_dp[j ^ 1] = max(next_dp[j ^ 1], dp[j] + A[i]);
      if (i > 1) next_dp[j ^ 1] = max(next_dp[j ^ 1], dp[j] + 2 * A[i]);  // 盲驴庐忙颅拢: i>1 <- i>0
    }
    dp = next_dp;
  }
  long long ans = max(dp[0], dp[1]);
  cout << ans << endl;
  return 0;
}
```
