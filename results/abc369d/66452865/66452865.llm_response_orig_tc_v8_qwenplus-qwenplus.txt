```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int N;
  cin >> N;
  vector<long long> A(N);
  for (int i = 0; i < N; i++) cin >> A[i];

  // dp[j]: max score when we have parity j (0: even, 1: odd) of special actions used
  vector<long long> dp(2);
  dp[0] = 0;

  for (int i = 0; i < N; i++) {
    vector<long long> ndp(2, -1e18); // Initialize with very small value
    // Case: skip current monster
    for (int j = 0; j < 2; j++) {
      ndp[j] = max(ndp[j], dp[j]);
    }
    // Case: fight current monster
    for (int j = 0; j < 2; j++) {
      int nj = j ^ 1; // parity flips
      if (i > 0 && j == 1) {
        // If last action was special (j==1), this is second consecutive special
        // So double reward
        ndp[nj] = max(ndp[nj], dp[j] + 2 * A[i]);
      } else {
        ndp[nj] = max(ndp[nj], dp[j] + A[i]);
      }
    }
    dp = ndp;
  }

  long long ans = max(dp[0], dp[1]);
  cout << ans << endl;
  return 0;
}
```
