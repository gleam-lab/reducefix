- When choosing to kill the current monster:

#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;

  vector<long long> A(N);
  for (int i = 0; i < N; ++i) cin >> A[i];

  vector<long long> dp(2, 0);  // dp[0]: last was skip, dp[1]: last was kill

  for (int i = 0; i < N; ++i) {
    vector<long long> ndp(2);

    dp = ndp;

  cout << max(dp[0], dp[1]) << endl;
  return 0;
}
