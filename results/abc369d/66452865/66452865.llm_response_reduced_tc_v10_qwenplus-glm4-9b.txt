#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  vector<long long> A(N);
  for (int i = 0; i < N; i++) cin >> A[i];
  vector<long long> dp(2);
  dp[0] = 0;
  dp[1] = 0;
  for (int i = 0; i < N; i++) {
    vector<long long> p(2);
    p[0] = dp[0];
    p[1] = dp[1];
    dp[0] = max(p[0], p[1]);
    dp[1] = max(p[1], p[0] + A[i]);
    if (i > 1) {
      dp[1] = max(dp[1], p[1] + 2 * A[i - 1]);  // Update the condition i>1 to i>1
    }
  }
  long long ans = max(dp[0], dp[1]);
  cout << ans << endl;
  return 0;
}
