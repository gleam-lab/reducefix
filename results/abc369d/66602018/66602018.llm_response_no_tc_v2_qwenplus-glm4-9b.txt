#include <iostream>
#include <algorithm>
using namespace std;

#define INF (long long)(1e18)

int main() {
    int n;
    long long dp0, dp1, x;
    dp0 = 0;
    dp1 = -INF;
    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> x;
        long long prev_dp0 = dp0;
        dp0 = max(dp1 + (long long)(2) * x, dp0);
        dp1 = max(prev_dp0 + x, dp1);
    }

    cout << max(dp0, dp1) << endl;
    return 0;
}
