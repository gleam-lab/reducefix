```cpp
#include <bits/stdc++.h>
using namespace std;

#define INF (long long)1e18  // Fixed: Previously too small, causing potential overflow/incorrect comparisons

int main() {
    int n;
    long long dp0 = 0, dp1 = -INF, x;
    
    cin >> n;

    for (int i = 0; i < n; ++i) {
        cin >> x;
        long long new_dp0 = max(dp0, dp1 + 2 * x);  // Max of not taking or taking double
        long long new_dp1 = max(dp1, dp0 + x);      // Max of not taking or taking single
        dp0 = new_dp0;
        dp1 = new_dp1;
    }

    cout << max(dp0, dp1) << endl;  // Final result is max of ending with either state
    return 0;
}
```
