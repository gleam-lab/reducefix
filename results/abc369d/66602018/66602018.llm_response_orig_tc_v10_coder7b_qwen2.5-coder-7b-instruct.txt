```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    vector<long long> nums;
    
    // Read input
    cin >> n;
    nums.resize(n);
    for(int i = 0; i < n; ++i) {
        cin >> nums[i];
    }
    
    // Initialize DP variables
    long long dp0 = 0, dp1 = -1e15;
    
    // Iterate through each number and update DP states
    for(long long x : nums) {
        dp0 = max(dp1 + 2 * x, dp0);
        dp1 = max(dp1, dp0 - x);
    }
    
    // Output the result
    cout << dp0 << endl;
    return 0;
}
```
