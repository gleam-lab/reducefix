#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long maxProfit(vector<int>& prices) {
    int n = prices.size();
    if (n == 0) return 0;
    
    vector<long long> dp(n, 0);
    dp[0] = 0;
    long long minPrice = prices[0];
    
    for (int i = 1; i < n; ++i) {
        minPrice = min(minPrice, static_cast<long long>(prices[i]));
        dp[i] = max(dp[i-1], prices[i] - minPrice);
    }
    
    return dp[n-1];
}

int main() {
    int n;
    cin >> n;
    vector<int> prices(n);
    for (int i = 0; i < n; ++i) {
        cin >> prices[i];
    }
    
    cout << maxProfit(prices) << endl;
    return 0;
}
