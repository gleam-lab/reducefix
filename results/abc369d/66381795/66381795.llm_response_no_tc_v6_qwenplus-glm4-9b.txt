#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    cin >> n;
    vector<int> data(n);
    rep(i, n) cin >> data[i];

    ll sum = 0;
    for (int i = 0; i < n; i++) {
        sum += (i % 2 == 0) ? 2 * data[i] : data[i];
    }

    cout << sum << endl;
    return 0;
}
