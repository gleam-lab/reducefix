#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    for (int i = 0; i < n; i++) {
        cin >> data[i];
    }

    long long sum = 0;
    bool evenCount = false;

    for (int i = 0; i < n; i++) {
        if (evenCount) {
            sum += data[i] * 2;
        } else {
            if (i + 1 < n && data[i + 1] * 2 > data[i] + data[i + 1]) {
                sum += data[i] + data[i + 1];
                i++; // Skip the next element as it's already included in the sum
            } else {
                sum += data[i];
            }
        }
        evenCount = !evenCount; // Toggle flag to alternate between even and odd counts
    }

    cout << sum << endl;
    return 0;
}
