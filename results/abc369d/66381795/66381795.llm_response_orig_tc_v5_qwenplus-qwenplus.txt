#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> data(n);
    rep(i, n) {
        cin >> data[i];
    }

    ll sum = 0;
    int i = 0;
    while (i < n) {
        // If we're at the end or cannot form a triplet with next two elements
        if (i + 1 >= n || i + 2 >= n) {
            sum += data[i];
            i++;
        } else {
            // Check if it's beneficial to group current and next two elements
            if (data[i+1] * 2 > data[i] + data[i+1]) {
                sum += data[i+1] * 2;
                i += 2;
            } else {
                sum += data[i];
                i++;
            }
        }
    }

    cout << sum << endl;
    return 0;
}
