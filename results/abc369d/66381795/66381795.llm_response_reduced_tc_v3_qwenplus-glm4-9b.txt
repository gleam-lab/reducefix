#include <iostream>
#include <vector>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int n;
int main() {
    cin >> n;
    vector<int> data(n);
    for (int i = 0; i < n; ++i) {
        cin >> data[i];
    }

    long long sum = 0;
    int cnt = 0;
    bool even = true; // Use a boolean to keep track of even and odd positions

    for (int i = 0; i < n; ++i) {
        cnt++;
        if ((cnt % 2 == 0 && even) || (cnt % 2 == 1 && !even)) {
            sum += data[i] * 2;
            if (i + 1 < n) {
                sum += data[i + 1] * 2;
                even = !even; // Toggle even flag for the next pair
            }
        } else {
            sum += data[i];
        }
    }

    cout << sum << endl;
    return 0;
}
