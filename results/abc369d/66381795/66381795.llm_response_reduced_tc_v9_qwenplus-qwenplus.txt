```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> data(n);
    rep(i, n) {
        cin >> data[i];
    }

    // dp[i] = max total sum using first i elements
    vector<ll> dp(n + 1, 0);
    
    // Base cases
    if (n >= 1) dp[1] = data[0];  // Only one element - take it normally
    
    // Fill dp array
    for (int i = 2; i <= n; i++) {
        // Either take current element as normal value or pair it with previous
        dp[i] = max(dp[i-1] + data[i-1], dp[i-2] + 2 * data[i-1]);
    }
    
    cout << dp[n] << endl;
    return 0;
}
```
