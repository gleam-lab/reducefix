#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    rep(i, n) {
        cin >> data[i];
    }

    ll sum = 0;
    int i = 0;
    while (i < n) {
        // If we're not at the end and it's beneficial to pair
        if (i + 1 < n && data[i+1] > data[i]) {
            sum += 2 * data[i+1];
            i += 2; // Skip both elements
        } else {
            sum += data[i];
            i++;
        }
    }

    cout << sum << endl;
    return 0;
}
