```cpp
#include <iostream>
#include <vector>

int main() {
    int n;
    std::cin >> n;
    std::vector<int> data(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> data[i];
    }

    long long sum = 0;
    for (size_t i = 0; i < data.size(); ++i) {
        if ((i + 1) % 2 == 0 && i != data.size() - 1 && data[i] * 2 > data[i + 1]) {
            sum += data[i] * 2;
        } else {
            sum += data[i];
        }
    }

    std::cout << sum << std::endl;
    return 0;
}
```
