#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    cin >> n;
    vector<int> data(n);
    rep(i, n) {
        cin >> data[i];
    }
    ll sum = 0;
    int i = 0;
    while (i < n) {
        if (i + 1 < n && data[i + 1] * 2 > data[i] + data[i + 1] && i % 2 == 0) {
            sum += (ll)data[i + 1] * 2 + (ll)data[i] * 2;
            i += 2;
        } else if (i % 2 == 0) {
            sum += (ll)data[i] * 2;
            i++;
        } else {
            sum += (ll)data[i];
            i++;
        }
    }
    cout << sum << endl;
    return 0;
}
