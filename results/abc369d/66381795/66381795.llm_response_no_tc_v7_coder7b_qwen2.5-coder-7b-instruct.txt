#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    
    rep(i, n) {
        cin >> data[i];
    }
    
    int cnt = 0;
    ll sum = 0;
    
    rep(i, n - 1) { // Adjust loop to avoid out-of-bounds access
        cnt++;
        if (cnt % 2 == 0 && data[i + 1] * 2 > data[i] + data[i + 1]) {
            sum += data[i + 1] * 2;
        } else if (cnt % 2 != 0) {
            sum += data[i];
        }
    }
    
    // Add the last element if it's not processed yet
    if (n % 2 != 0) {
        sum += data[n - 1];
    }
    
    cout << sum << endl;
    return 0;
}
