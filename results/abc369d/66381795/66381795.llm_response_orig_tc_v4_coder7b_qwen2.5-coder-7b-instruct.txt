```cpp
#include <bits/stdc++.h>
using namespace std;

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;
int n;
vector<int> data;

int main() {
    cin >> n;
    data.resize(n);
    rep(i, n) {
        cin >> data[i];
    }

    ll sum = 0;
    rep(i, n) {
        if ((i + 1) % 2 == 1) {
            sum += data[i];
        } else {
            sum += min(data[i], data[i + 1]);
        }
    }

    cout << sum << endl;
    return 0;
}
```
