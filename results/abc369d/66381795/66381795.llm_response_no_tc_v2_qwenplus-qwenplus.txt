```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    rep(i, n) {
        cin >> data[i];
    }

    ll sum = 0;
    int i = 0;

    while (i < n) {
        // If there is at least one more element
        if (i + 1 < n) {
            // Compare current and next to decide pairing
            if (2 * data[i + 1] > data[i]) {
                sum += 2 * data[i + 1];
                i += 2; // skip both elements
            } else {
                sum += data[i];
                i++; // process next single element
            }
        } else {
            // Odd number of remaining elements
            sum += data[i];
            i++;
        }
    }

    cout << sum << endl;
    return 0;
}
```
