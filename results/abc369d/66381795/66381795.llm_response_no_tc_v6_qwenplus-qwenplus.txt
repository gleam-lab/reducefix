```cpp
#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    rep(i, n) cin >> data[i];

    ll sum = 0;
    int i = 0;

    while (i < n) {
        // If we are not at the last element and it's beneficial to pair
        if (i + 1 < n && data[i] < data[i + 1]) {
            sum += 2 * data[i + 1];
            i += 2;  // skip both elements as they are paired
        } else {
            sum += data[i];
            i++;
        }
    }

    cout << sum << endl;
    return 0;
}
```
