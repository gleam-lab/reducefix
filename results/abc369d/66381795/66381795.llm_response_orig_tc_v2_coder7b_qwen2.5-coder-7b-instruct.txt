```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<long long> data(n);
    for(int i = 0; i < n; ++i) {
        cin >> data[i];
    }

    long long sum = 0;
    for(int i = 0; i < n-1; i += 2) {
        sum += max(data[i], data[i+1]);
    }
    if(n % 2 == 1) {
        sum += data[n-1];
    }

    cout << sum << endl;
    return 0;
}
```
