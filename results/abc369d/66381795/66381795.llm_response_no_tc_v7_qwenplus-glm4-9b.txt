#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;
int n, a;
int main (){
    cin >> n;
    vector<int> data(n);
    rep(i, n){
        cin >> data[i];
    }
    ll sum = 0;
    int even_index = 0, odd_index = 1; // Use two pointers to track even and odd indexed elements
    while (even_index < n && odd_index < n) {
        if (data[odd_index] * 2 > data[even_index] + data[odd_index]) {
            sum += data[even_index] * 2; // Add twice the even-indexed element
            even_index += 2; // Move to the next even-indexed element
        } else {
            sum += data[odd_index] + data[even_index]; // Add sum of current even and odd indexed elements
            if (even_index < n) even_index += 2; // Move to the next even-indexed element if exists
            odd_index += 2; // Always move to the next odd-indexed element
        }
    }
    // Check if the last pair of elements is left unprocessed (if n is odd)
    if (even_index < n) {
        sum += data[even_index] * 2;
    }
    cout << sum << endl;
    return 0;
}
