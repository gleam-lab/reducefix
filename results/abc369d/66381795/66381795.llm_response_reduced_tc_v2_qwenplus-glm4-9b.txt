    for (int i = 0; i < n; i++) {
        cin >> data[i];
    }
    ll sum = 0;
    for (int i = 0; i < n; i++) {
        if (i + 1 < n && data[i + 1] * 2 > data[i] + data[i + 1]) {
            sum += data[i + 1] * 2;
            i++; // Move to the next index since the current index has been processed
        } else {
            sum += data[i];
        }
    }
    cout << sum << endl;
    return 0;
}
