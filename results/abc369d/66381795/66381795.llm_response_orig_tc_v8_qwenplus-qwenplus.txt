#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<ll> data(n);
    rep(i, n) {
        cin >> data[i];
    }

    ll sum = 0;
    int i = 0;
    while (i < n) {
        if (i + 1 < n && data[i+1] > (data[i] + data[i+1]) / 2) {
            // Use pair optimization: pick larger of two
            sum += data[i+1] * 2;
            i += 2;
        } else {
            // Single element
            sum += data[i];
            i++;
        }
    }

    cout << sum << endl;
    return 0;
}
