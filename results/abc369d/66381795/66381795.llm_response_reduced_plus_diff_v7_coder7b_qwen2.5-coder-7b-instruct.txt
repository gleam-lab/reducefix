```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<long long> data(n);
    for(int i = 0; i < n; ++i) {
        cin >> data[i];
    }

    long long sum = 0;
    for(int i = 0; i < n; ++i) {
        if((i + 1) % 2 == 0) {
            sum += data[i] * 2;
        } else {
            sum += data[i];
        }
    }

    cout << sum << '\n';
    return 0;
}
```
