#include <bits/stdc++.h>

#define rep(i, n) for (int i = 0; i < (int)(n); i++)
using ll = long long;

int main() {
    int n;
    cin >> n;
    vector<int> data(n);
    rep(i, n) {
        cin >> data[i];

    ll sum = 0;
    rep(i, n) {

    return 0;
}
