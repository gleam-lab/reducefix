```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll(i) = 0; (i) < (n); ++(i))

int main() {
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    }

    // dp_odd[i]: i体目まで倒して、奇数体倒したときの最大経験値
    // dp_even[i]: i体目まで倒して、偶数体倒したときの最大経験値
    vector<ll> dp_odd(n, 0), dp_even(n, 0);
    
    dp_odd[0] = a[0];
    dp_even[0] = 0;

    for (int i = 1; i < n; ++i) {
        // i体目を倒す -> 倒した総数が奇数になる
        dp_odd[i] = max(dp_even[i-1] + 2 * a[i], dp_odd[i-1] + a[i]);
        
        // i体目を倒さない -> 倒した総数は偶数のまま
        dp_even[i] = max(dp_odd[i-1] + a[i], dp_even[i-1] + 2 * a[i]);
    }

    cout << max(dp_odd[n-1], dp_even[n-1]) << endl;
    return 0;
}
```
