// Function to calculate the maximum experience points
ll maxExperiencePoints(const vector<ll>& a, int index, bool oddTurn, vector<vector<ll>>& dp) {
    int n = a.size();

    if (dp[index][oddTurn] != -1) return dp[index][oddTurn];
