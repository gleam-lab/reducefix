#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll(i) = 0; (i) < (n); ++(i))
#define drep(i, n) for (ll(i) = (n) - 1; (i) >= 0; --(i))

int main() {
    int n;
    cin >> n;
    vector<ll> a(n, 0);
    rep(i, n) cin >> a[i];

    // Early return
    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    } else if (n == 2) {
        cout << a[0] + 2 * a[1] << endl;
        return 0;
    }

    // dp[i][0]: Maximum experience points with an even number of monsters defeated up to i-th monster
    // dp[i][1]: Maximum experience points with an odd number of monsters defeated up to i-th monster
    vector<vector<ll>> dp(n, vector<ll>(2, 0));
    dp[0][0] = 0;
    dp[0][1] = a[0];
    dp[1][0] = a[0];
    dp[1][1] = a[0] + 2 * a[1];

    for (int i = 2; i < n; ++i) {
        dp[i][0] = max(dp[i - 1][0], dp[i - 2][1] + a[i]); // Either not defeating the current monster or defeating the previous one as odd
        dp[i][1] = max(dp[i - 1][1], dp[i - 2][0] + 2 * a[i]); // Either not defeating the current monster or defeating the previous one as even
    }
    cout << max(dp[n - 1][0], dp[n - 1][1]) << endl;

    return 0;
}
