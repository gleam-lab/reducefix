#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// Function to calculate the maximum experience points using dynamic programming
long long maxExperiencePoints(const vector<int>& a) {
    int n = a.size();
    if (n == 0) return 0;
    if (n == 1) return a[0];
    
    // dp[i][j] represents the maximum experience points up to the i-th element with j flips
    vector<vector<long long>> dp(n, vector<long long>(2, 0));
    dp[0][0] = a[0]; // No flip at the first element
    dp[0][1] = 0;   // Flip at the first element
    
    for (int i = 1; i < n; ++i) {
        dp[i][0] = max(dp[i-1][0], dp[i-1][1] + a[i]); // No flip at the i-th element
        dp[i][1] = max(dp[i-1][1], dp[i-1][0] + 2*a[i]); // Flip at the i-th element
    }
    
    // The answer is the maximum of no flip or one flip at the last element
    return max(dp[n-1][0], dp[n-1][1]);
}

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    cout << maxExperiencePoints(a) << endl;
    return 0;
}
