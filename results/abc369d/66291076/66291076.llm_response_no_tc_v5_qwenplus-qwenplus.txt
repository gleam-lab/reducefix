#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll(i) = 0; (i) < (n); ++(i))

int main() {
    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    // Early return for small cases
    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    } else if (n == 2) {
        cout << max(a[0] + 2 * a[1], 2 * a[0] + a[1]) << endl;
        return 0;
    }

    // dp_odd[i]: max exp if we defeat odd number of monsters up to i
    // dp_even[i]: max exp if we defeat even number of monsters up to i
    vector<ll> dp_odd(n, 0), dp_even(n, 0);

    dp_odd[0] = a[0];     // defeat 1st monster
    dp_even[0] = 0;       // defeat 0 monsters

    dp_odd[1] = max(dp_even[0] + a[1], dp_odd[0] + 2 * a[1]);
    dp_even[1] = max(dp_odd[0] + 2 * a[1], dp_even[0] + a[1]);

    for (int i = 2; i < n; ++i) {
        // If current count is odd: previous must be even or odd+1
        dp_odd[i] = max(dp_even[i - 1] + a[i], dp_odd[i - 2] + 2 * a[i]);

        // If current count is even: previous must be odd or even+1
        dp_even[i] = max(dp_odd[i - 1] + 2 * a[i], dp_even[i - 2] + a[i]);
    }

    cout << max(dp_odd[n - 1], dp_even[n - 1]) << endl;

    return 0;
}
