#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll i = 0; (i) < (n); ++(i))
#define drep(i, n) for (ll i = (n)-1; (i) >= 0; --(i))

int main() {
    cin.tie(nullptr);
    ios::sync_with_stdio(false);

    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    // Early returns for small n
    if (n == 1) {
        cout << a[0] << '\n';
        return 0;
    }

    // Initialize dp arrays for odd and even selections
    vector<ll> dp_odd(n, 0), dp_even(n, 0);
    dp_odd[0] = a[0];
    dp_even[0] = 0;

    // For n == 2
    if (n == 2) {
        cout << a[0] + 2 * a[1] << '\n';
        return 0;
    }

    // Initialize first two states
    dp_odd[1] = a[1];                     // one selected: pick second
    dp_even[1] = a[0] + 2 * a[1];         // two selected: pick both

    // Fill dp arrays
    for (int i = 2; i < n; ++i) {
        dp_odd[i] = max(dp_odd[i - 2] + 2 * a[i], dp_even[i - 1] + a[i]);
        dp_even[i] = max(dp_even[i - 2] + a[i], dp_odd[i - 1] + 2 * a[i]);
    }

    // Final result is the maximum of last states
    cout << max(dp_odd[n - 1], dp_even[n - 1]) << '\n';

    return 0;
}
