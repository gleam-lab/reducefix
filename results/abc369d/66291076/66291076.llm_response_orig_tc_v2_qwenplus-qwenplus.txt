#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll(i) = 0; (i) < (n); ++(i))
#define drep(i, n) for (ll(i) = (n) - 1; (i) >= 0; --(i))

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<ll> a(n);
    rep(i, n) cin >> a[i];

    if (n == 1) {
        cout << a[0] << "\n";
        return 0;
    }

    // dp_odd[i]: max exp when selecting odd number of monsters up to i
    // dp_even[i]: max exp when selecting even number of monsters up to i
    vector<ll> dp_odd(n), dp_even(n);

    dp_odd[0] = a[0];
    dp_even[0] = 0;

    dp_odd[1] = a[1];
    dp_even[1] = a[0] + 2 * a[1];

    for (int i = 2; i < n; ++i) {
        dp_odd[i] = max(dp_even[i - 1] + a[i], dp_odd[i - 2] + 2 * a[i]);
        dp_even[i] = max(dp_odd[i - 1] + 2 * a[i], dp_even[i - 2] + a[i]);
    }

    cout << max(dp_odd[n - 1], dp_even[n - 1]) << "\n";

    return 0;
}
