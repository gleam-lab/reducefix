```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i, n) for (ll(i) = 0; (i) < (n); ++(i))
#define drep(i, n) for (ll(i) = (n) - 1; (i) >= 0; --(i))

int main() {
    int n;
    cin >> n;
    vector<ll> a(n, 0);
    rep(i, n) cin >> a[i];

    // 忙聴漏忙聹聼茫聝陋茫聜驴茫聝录茫聝鲁
    if (n == 1) {
        cout << a[0] << endl;
        return 0;
    } else if (n == 2) {
        cout << a[0] + 2 * a[1] << endl;
        return 0;
    }

    // i盲陆聯莽聸庐茫聛戮茫聛搂茫聛庐忙聹聙氓陇搂茫聛庐莽碌聦茅篓聯氓聙陇茫聙聛i盲陆聯莽聸庐茫聛戮茫聛搂茫聛搂茫聙聛odd/even氓聸聻氓聙聮茫聛聴茫聛聼氓聽麓氓聬聢
    vector<ll> dp_odd(n, 0), dp_even(n, 0);
    dp_odd[0] = a[0];
    dp_even[0] = 0;
    dp_odd[1] = a[1];
    dp_even[1] = a[0] + 2 * a[1];

    for (int i = 2; i < n; ++i) {
        dp_odd[i] = max(dp_odd[i - 2] + 2 * a[i], dp_even[i - 1] + a[i]);
        dp_even[i] = max(dp_even[i - 2] + a[i], dp_odd[i - 1] + 2 * a[i]);
    }
    cout << max(dp_odd[n - 1], dp_even[n - 1]) << endl;

    return 0;
}
```
