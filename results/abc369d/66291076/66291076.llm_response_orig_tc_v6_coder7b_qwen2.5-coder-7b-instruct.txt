#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long maxExperience(int n, const vector<long long>& a) {
    if (n == 1) return a[0];
    if (n == 2) return a[0] + 2 * a[1];

    vector<long long> dp(n, 0);
    dp[0] = a[0];
    dp[1] = a[1];
    for (int i = 2; i < n; ++i) {
        dp[i] = max(dp[i-1], dp[i-2] + 2 * a[i]);
    }
    return dp[n-1];
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    cout << maxExperience(n, a) << '\n';

    return 0;
}
