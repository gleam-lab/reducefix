#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using ll = long long;

int main() {
    int n;
    cin >> n;
    if (n == 0) return 0;

    vector<ll> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // dp[i][0] - odd round, dp[i][1] - even round
    vector<vector<ll>> dp(n, vector<ll>(2, 0));
    dp[0][0] = a[0];
    dp[0][1] = 0;
    if (n > 1) {
        dp[1][0] = a[1];
        dp[1][1] = a[0] + 2 * a[1];
    }

    for (int i = 2; i < n; ++i) {
        dp[i][0] = max(dp[i - 1][0] + 2 * a[i], dp[i - 1][1] + a[i]);
        dp[i][1] = max(dp[i - 1][0] + a[i], dp[i - 2][1] + 2 * a[i]);
    }

    cout << max(dp[n - 1][0], dp[n - 1][1]) << endl;

    return 0;
}
