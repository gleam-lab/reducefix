```python
N = int(input())
S = input().strip()

# Map opponent's move to the winning move for Takahashi
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] will store the maximum wins up to game i, where j represents Takahashi's move in game i
# 0: Rock, 1: Paper, 2: Scissors
dp = [[-1] * 3 for _ in range(N)]

def get_move_index(move):
    return {'R': 0, 'P': 1, 'S': 2}[move]

def get_move_char(idx):
    return ['R', 'P', 'S'][idx]

# Initialize first game
opp_move = S[0]
winning_move = win_move[opp_move]
losing_move = lose_move[opp_move]

# Can choose any move that doesn't lose
for m in range(3):
    move_char = get_move_char(m)
    if move_char == losing_move:
        continue  # This would be a loss, not allowed
    dp[0][m] = 1 if move_char == winning_move else 0

# Fill DP table
for i in range(1, N):
    opp_move = S[i]
    winning_move = win_move[opp_move]
    losing_move = lose_move[opp_move]
    
    for curr in range(3):
        curr_move = get_move_char(curr)
        
        # Cannot have same move consecutively
        if dp[i-1][curr] != -1:
            continue
            
        # Cannot play a losing move
        if curr_move == losing_move:
            dp[i][curr] = -1
            continue
            
        best_prev = -1
        for prev in range(3):
            if prev == curr:  # consecutive same move not allowed
                continue
            if dp[i-1][prev] == -1:
                continue
            best_prev = max(best_prev, dp[i-1][prev])
        
        if best_prev == -1:
            dp[i][curr] = -1
        else:
            add_win = 1 if curr_move == winning_move else 0
            dp[i][curr] = best_prev + add_win

print(max(dp[N-1]))
```
