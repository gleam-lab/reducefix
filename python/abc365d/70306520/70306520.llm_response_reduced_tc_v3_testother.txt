```python
N = int(input())
S = input().strip()

# Map opponent's move to the winning move for Takahashi
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] will store the maximum wins up to game i, where j represents Takahashi's move in game i
# 0: Rock, 1: Paper, 2: Scissors
dp = [[-1] * 3 for _ in range(N)]

def move_to_idx(m):
    return {'R': 0, 'P': 1, 'S': 2}[m]

def idx_to_move(idx):
    return ['R', 'P', 'S'][idx]

# Initialize first game
aoki_first = S[0]
takahashi_win = win_move[aoki_first]
takahashi_draw = aoki_first

# Can choose winning move (win) or drawing move (not lose)
win_idx = move_to_idx(takahashi_win)
draw_idx = move_to_idx(takahashi_draw)

dp[0][win_idx] = 1  # Win
dp[0][draw_idx] = 0  # Draw

# For moves that are losing, we leave them as -1 (invalid)

for i in range(1, N):
    aoki_curr = S[i]
    win_curr = win_move[aoki_curr]
    draw_curr = aoki_curr
    
    win_idx = move_to_idx(win_curr)
    draw_idx = move_to_idx(draw_curr)
    
    # Try all three possible current moves
    for curr in range(3):
        if curr == win_idx:
            base = 1  # win
        elif curr == draw_idx:
            base = 0  # draw
        else:
            continue  # losing move, not allowed
        
        # Check all previous moves that are valid and different from current
        best_prev = -1
        for prev in range(3):
            if prev == curr:
                continue  # consecutive same move not allowed
            if dp[i-1][prev] != -1:
                if dp[i-1][prev] > best_prev:
                    best_prev = dp[i-1][prev]
        
        if best_prev != -1:
            dp[i][curr] = best_prev + base
        # else remains -1

result = max(dp[N-1])
print(result)
```
