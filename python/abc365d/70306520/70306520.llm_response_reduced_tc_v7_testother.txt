```python
N = int(input())
S = input().strip()

# Map moves to indices: R->0, P->1, S->2
# Winning move against each of Aoki's moves:
# If Aoki plays R (rock), Takahashi must play P (paper) to win -> index 1
# If Aoki plays P (paper), Takahashi must play S (scissors) to win -> index 2
# If Aoki plays S (scissors), Takahashi must play R (rock) to win -> index 0
win_move = {'R': 1, 'P': 2, 'S': 0}

# Losing move against each of Aoki's moves (so Takahashi would lose if he played this)
lose_move = {'R': 2, 'P': 0, 'S': 1}

# dp[i][j] = maximum wins up to game i when Takahashi plays move j in game i
# We'll use a list of 3 values for the current state
prev = [-1] * 3

# Initialize first game
aoki_first = S[0]
w = win_move[aoki_first]
l = lose_move[aoki_first]

# Takahashi can choose any move except the losing one
for move in range(3):
    if move == l:
        prev[move] = -float('inf')  # invalid: leads to loss
    else:
        prev[move] = 1 if move == w else 0  # win if it's winning move, else draw

# Process remaining games
for i in range(1, N):
    curr = [-float('inf')] * 3
    aoki_move = S[i]
    w = win_move[aoki_move]
    l = lose_move[aoki_move]
    
    for cur_move in range(3):
        if cur_move == l:
            continue  # cannot play losing move
        for prev_move in range(3):
            if prev_move == cur_move:
                continue  # consecutive moves must be different
            if prev[prev_move] == -float('inf'):
                continue
            score = prev[prev_move] + (1 if cur_move == w else 0)
            if score > curr[cur_move]:
                curr[cur_move] = score
    
    prev = curr

print(max(prev))
```
