```python
N = int(input())
S = input().strip()

# Map opponent's move to the move that beats it
win_move = {'R': 'P', 'P': 'S', 'S': 'R'}
lose_move = {'R': 'S', 'P': 'R', 'S': 'P'}

# dp[i][j] will store the maximum wins up to game i, where j represents the move used in game i
# 0: Rock, 1: Paper, 2: Scissors
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for move_idx, move_char in enumerate(['R', 'P', 'S']):
    if move_char == win_move[S[0]]:
        dp[0][move_idx] = 1  # Win
    elif move_char == S[0]:
        dp[0][move_idx] = 0  # Draw
    # Cannot lose, so no other option

# Fill DP table
for i in range(1, N):
    current_opponent = S[i]
    for curr in range(3):  # current move
        curr_char = ['R', 'P', 'S'][curr]
        # Check if this move is valid (not losing)
        if curr_char == lose_move[current_opponent]:
            continue  # This would be a loss, not allowed
        
        # Determine score for this move
        win_bonus = 1 if curr_char == win_move[current_opponent] else 0
        
        # Transition from previous state with different move
        best_prev = -1
        for prev in range(3):
            if prev != curr and dp[i-1][prev] != -1:
                best_prev = max(best_prev, dp[i-1][prev])
        
        if best_prev != -1:
            dp[i][curr] = best_prev + win_bonus

print(max(dp[N-1]))
```
