```python
N = int(input())
S = input().strip()

# Map moves to indices: R->0, P->1, S->2
move_to_idx = {'R': 0, 'P': 1, 'S': 2}
# Winning move against each of Aoki's moves:
# Against Rock (R), Takahashi should play Paper (P) -> index 1
# Against Paper (P), Takahashi should play Scissors (S) -> index 2
# Against Scissors (S), Takahashi should play Rock (R) -> index 0
win_move = {'R': 1, 'P': 2, 'S': 0}

# dp[i][j] = maximum wins up to game i, where j is Takahashi's move in game i (0=R,1=P,2=S)
dp = [[-1] * 3 for _ in range(N)]

# Initialize first game
for j in range(3):
    if j == win_move[S[0]]:
        dp[0][j] = 1  # Win
    elif j != move_to_idx[S[0]]:  # Not losing means it's a draw (same move would be loss only if opponent beats us, but we can't lose)
        dp[0][j] = 0  # Draw

# Fill DP table
for i in range(1, N):
    for curr in range(3):  # current move by Takahashi
        # Check if this move loses to Aoki's move
        aoki_move = S[i]
        aoki_idx = move_to_idx[aoki_move]
        if (curr == 0 and aoki_idx == 1) or (curr == 1 and aoki_idx == 2) or (curr == 2 and aoki_idx == 0):
            continue  # This move loses, not allowed
        
        # Determine outcome: win or draw
        if curr == win_move[aoki_move]:
            outcome = 1  # Win
        else:
            outcome = 0  # Draw
        
        # Transition from previous state with different move
        for prev in range(3):
            if prev == curr:
                continue  # Consecutive same moves not allowed
            if dp[i-1][prev] != -1:
                if dp[i][curr] < dp[i-1][prev] + outcome:
                    dp[i][curr] = dp[i-1][prev] + outcome

print(max(dp[N-1]))
```
