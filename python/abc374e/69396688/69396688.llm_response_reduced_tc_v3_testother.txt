def can_achieve(capacity):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Try different numbers of S_i machines
        max_s = capacity // a if a > 0 else 0
        for s in range(max_s + 1):
            remaining = capacity - s * a
            t = (remaining + b - 1) // b if b > 0 and remaining > 0 else 0
            if remaining < 0:
                continue
            cost = s * p + t * q
            min_cost = min(min_cost, cost)
            
        # Also try using only T_i machines
        t_only = (capacity + b - 1) // b if b > 0 else 0
        min_cost = min(min_cost, t_only * q)
        
        total_cost += min_cost
        if total_cost > x:
            return False
            
    return total_cost <= x

n, x = map(int, input().split())
machines = []
for _ in range(n):
    machines.append(list(map(int, input().split())))

# Binary search on the production capacity
lo = 0
hi = 2 * 10**7  # Upper bound: even with worst machines, we won't exceed this

while lo < hi:
    mid = (lo + hi + 1) // 2
    if can_achieve(mid):
        lo = mid
    else:
        hi = mid - 1

print(lo)
