def min_cost_for_capacity(cap):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Try all possible numbers of S_i machines (from 0 up to cap)
        # We need at least ceil(cap / a) if we were to use only S_i, but we can mix
        max_s = (cap + a - 1) // a if a > 0 else 0
        for s in range(max_s + 1):
            products_from_s = s * a
            remaining = max(0, cap - products_from_s)
            t = (remaining + b - 1) // b if b > 0 else 0
            cost = s * p + t * q
            min_cost = min(min_cost, cost)
            
            # Also try using only T machines or mixed, but bounded reasonably
            # Alternatively, we can iterate on T side too, but let's optimize:
            # Instead, we can break early if s*a >= cap (no T needed)
            if products_from_s >= cap:
                min_cost = min(min_cost, s * p)
        
        # Similarly try iterating over T first
        max_t = (cap + b - 1) // b if b > 0 else 0
        for t in range(max_t + 1):
            products_from_t = t * b
            remaining = max(0, cap - products_from_t)
            s = (remaining + a - 1) // a if a > 0 else 0
            cost = t * q + s * p
            min_cost = min(min_cost, cost)
            
            if products_from_t >= cap:
                min_cost = min(min_cost, t * q)
                
        if min_cost == float('inf'):
            return float('inf')
        total_cost += min_cost
        if total_cost > x:
            return float('inf')
    return total_cost

n, x = map(int, input().split())
machines = []
for _ in range(n):
    a, p, b, q = map(int, input().split())
    machines.append((a, p, b, q))

# Binary search on the production capacity
lo, hi = 0, 20000000  # upper bound: even with worst machines and full budget

while lo < hi:
    mid = (lo + hi + 1) // 2
    if min_cost_for_capacity(mid) <= x:
        lo = mid
    else:
        hi = mid - 1

print(lo)
