def min_cost_for_capacity(target):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Try all possible numbers of S_i machines
        # We need at least `target` processing power from this process
        max_s = (target + a - 1) // a if a > 0 else 0
        for s in range(max_s + 1):
            need = target - s * a
            t = 0 if need <= 0 else (need + b - 1) // b
            
            cost = s * p + t * q
            min_cost = min(min_cost, cost)
            
        total_cost += min_cost
        if total_cost > x:  # Early termination if already over budget
            return float('inf')
            
    return total_cost

n, x = map(int, input().split())
machines = []
for _ in range(n):
    a, p, b, q = map(int, input().split())
    machines.append((a, p, b, q))

# Binary search on the production capacity
lo = 0
hi = 2 * 10**7  # Maximum possible capacity: max(A_i,B_i) * (X / min(P_i,Q_i)) -> bounded by ~10^7 * 10^7 / 1 = 10^14 but we can use safe upper bound

while hi - lo > 1:
    mid = (lo + hi) // 2
    if min_cost_for_capacity(mid) <= x:
        lo = mid
    else:
        hi = mid

print(lo)
