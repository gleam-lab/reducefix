def min_cost_for_capacity(capacity):
    total_cost = 0
    for i in range(n):
        a, p, b, q = machines[i]
        min_cost = float('inf')
        
        # Try different numbers of S_i machines
        max_s = (capacity + a - 1) // a if a > 0 else 0
        for s in range(max_s + 1):
            products_from_s = s * a
            remaining = max(0, capacity - products_from_s)
            t = (remaining + b - 1) // b if b > 0 else 0
            
            cost = s * p + t * q
            min_cost = min(min_cost, cost)
            
            # Early termination if s is already too expensive
            if s * p >= min_cost:
                break
                
        total_cost += min_cost
        if total_cost > x:
            return float('inf')
            
    return total_cost

n, x = map(int, input().split())
machines = []
for _ in range(n):
    a, p, b, q = map(int, input().split())
    machines.append((a, p, b, q))

# Binary search on the production capacity
left, right = 0, 2 * 10**7
answer = 0

while left <= right:
    mid = (left + right) // 2
    if min_cost_for_capacity(mid) <= x:
        answer = mid
        left = mid + 1
    else:
        right = mid - 1

print(answer)
