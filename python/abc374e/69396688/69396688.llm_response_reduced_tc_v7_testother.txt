def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    x = int(data[1])
    machines = []
    index = 2
    for _ in range(n):
        a = int(data[index])
        p = int(data[index+1])
        b = int(data[index+2])
        q = int(data[index+3])
        index += 4
        machines.append((a, p, b, q))
    
    # Binary search on the production capacity
    left, right = 0, 10**9
    ans = 0
    
    while left <= right:
        mid = (left + right) // 2
        total_cost = 0
        
        for a, p, b, q in machines:
            min_cost = float('inf')
            
            # If this process cannot achieve mid capacity with any machine
            if a == 0 and b == 0 and mid > 0:
                min_cost = float('inf')
                break
            
            # Calculate minimum cost to achieve at least 'mid' units in this process
            # Using machine S_i: need ceil(mid / a) units
            if a > 0:
                units_s = (mid + a - 1) // a
                cost_s = units_s * p
                min_cost = min(min_cost, cost_s)
            
            # Using machine T_i: need ceil(mid / b) units
            if b > 0:
                units_t = (mid + b - 1) // b
                cost_t = units_t * q
                min_cost = min(min_cost, cost_t)
            
            # Try combinations only if beneficial (avoid TLE)
            # We try up to mid//min(a,b) but cap it for performance
            if a > 0 and b > 0:
                # Try using some of S and some of T
                max_s_units = (mid + a - 1) // a
                for s_units in range(max_s_units + 1):
                    remaining = mid - s_units * a
                    if remaining <= 0:
                        cost = s_units * p
                        min_cost = min(min_cost, cost)
                    else:
                        t_units = (remaining + b - 1) // b
                        cost = s_units * p + t_units * q
                        min_cost = min(min_cost, cost)
                        
                    # Early break if cost already too high
                    if s_units * p >= min_cost:
                        break
            
            if min_cost == float('inf'):
                total_cost = float('inf')
                break
                
            total_cost += min_cost
            if total_cost > x:
                break
        
        if total_cost <= x:
            ans = mid
            left = mid + 1
        else:
            right = mid - 1
    
    print(ans)

solve()
