import heapq

N = int(input())
A = sorted(map(int, input().split()), reverse=True)  # おもちゃのサイズを降順
B = sorted(map(int, input().split()), reverse=True)  # 箱のサイズを降順

# 貪欲法: 大きいおもちゃから、大きめの箱を使っていく
# ヒープは最小ヒープなので、負値で管理（最大ヒープとして使う）
box_heap = [-b for b in B]
heapq.heapify(box_heap)

# 新しい箱の必要サイズ（初期は無限大のように大きい）
required_x = 0
possible = True

for a in A:
    if box_heap and -box_heap[0] >= a:
        # 既存の箱で収まる
        heapq.heappop(box_heap)
    else:
        # 既存の箱では収まらない → 新しい箱が必要
        required_x = a
        break
else:
    # 全てのおもちゃが既存の箱に入った → 新しい箱は不要だが、1つは買う必要があるので x=0 でもいいが、正の整数なので最小は1?
    # しかし問題: 「正の整数 x」 を購入 → 但し、全て既存で賄えるなら x は 1 でも良いが、最小 x は 1?
    # だが、この場合新しい箱を使わないので、x は「使われない」。条件は「各玩具はサイズ以上の箱に入れる」。
    # 新しい箱のサイズ x が a より小さければその玩具は入れられないが、新しい箱を使わない限り問題ない。
    # よって、新しい箱を使わないなら、どんな x>0 でも理論上は可能。ただし最小の正の整数は 1。
    # しかし、上のループで break しなかった = 新しい箱を使わないですむ → 最小の x は 1。
    required_x = 1

# しかし、これは正しくない。なぜなら、上の貪欲は「新しい箱をいつ使うか」ではなく、「どの順番で箱を使うか」を間違えている。

# 正しいアプローチ:
# 新しい箱のサイズ x を決めると、残り N-1 個のおもちゃを N-1 個の既存の箱に割り当てる必要がある。
# つまり、ある1つのおもちゃだけが新しい箱に入り、残りが既存の箱にフィットするようにしたい。
# 貪欲戦略: 各おもちゃ i を新しい箱に入れたと仮定して、残り N-1 個のおもちゃが既存の箱にフィットするかチェック。
# しかし N=2e5 なので O(N^2) は不可。

# 別のアプローチ:
# まず、すべてのおもちゃをソート、すべての箱をソート。
# 新しい箱に入るおもちゃは1つだけ。それを選ぶ。残りの N-1 個のおもちゃを N-1 個の既存の箱に割り当てる。
# これは「マッチング可能か？」という問題。

# 高速な方法:
# 1. おもちゃを昇順、箱を昇順にする。
# 2. 「どのおもちゃを新しい箱に入れるか」を決める。
# 3. 残りのおもちゃたちを、箱たちに貪欲に割り当てる（小さいおもちゃから、そのおもちゃ以上で最も小さい箱）。

# これを全探索すると O(N^2 log N) となり、TLE.

# 効率的な方法:
# - 事前に、左側からのマッチング可能性（prefix）と右側からのマッチング可能性（suffix）を前処理。

# アイデア:
# 1. おもちゃを昇順にソート: a0 <= a1 <= ... <= a_{N-1}
# 2. 箱を昇順にソート: b0 <= b1 <= ... <= b_{N-2}
# 3. 新しい箱に入れる候補のおもちゃを決める。
# 4. そのおもちゃ以外のおもちゃを、箱に割り当てる。

# 前処理:
# - prefix[i]: おもちゃ[0:i] が 箱[0:i-1] までで完全にカバーできるか？ → 実際には、i個のおもちゃに対してi個の箱が必要。
#   ここでは、prefix_ok[i] := おもちゃ[0..i-1] を 箱[0..i-1] に割り当てられるか？
#   だが、おもちゃが1つ抜けるので、全体としては N-1 個の割り当て。

# 正しい方法（公式解法風）:
# - おもちゃを昇順: A_sorted
# - 箱を昇順: B_sorted
# - 左から貪欲: left[i] := おもちゃ[0:i] が 箱[0:i-1] に割り当てられるための最小の追加条件？ いや違う。

# 代わりに:
# - left[i] := おもちゃ[0..i-1] を 箱[0..i-1] に割り当てるとき、最大のおもちゃサイズと箱の対応の余裕？
# 実際には、left[i] を「おもちゃ[0..i] を箱[0..i] に割り当てるとき、失敗するか？」ではなく。

# 標準的手法:
# - まず、すべてのおもちゃを既存の箱に割り当てようとすると、1つだけ余る。
# - なので、どの1つをおもちゃを外すか（＝新しい箱に入れるか）を決める。
# - 外したときに、残りが完全マッチするか？

# 前処理:
# - left[i]: おもちゃ[0..i-1] が 箱[0..i-1] に割り当てられるか？ → だが、おもちゃの数と箱の数が異なる。

# 正しい前処理:
# おもちゃを昇順: A
# 箱を昇順: B (長さ N-1)

# 定義:
# - prefix[i]: おもちゃ[0:i] と 箱[0:i] を使って、おもちゃ[0:i] のうち i 個を割り当てられるか？ → 箱は i 個しかないので、i+1 個以上のおもちゃは無理。

# 実際には:
# - おもちゃを1つ除く → 除いた後、N-1個のおもちゃとN-1個の箱で完全マッチ。
# - 貪欲に、小さいおもちゃから、そのおもちゃ以上で最小の箱を割り当てる。

# 効率的にやるために:
# - prefix_match[i]: おもちゃ[0..i-1] を 箱[0..i-1] に割り当てるとき、成功するかどうか？
#   → 貪欲に i=0 から見て、A[i] <= B[i] が必要？ いや、貪欲に割り当てればよい。

# アルゴリズム:
# 1. A.sort(), B.sort()
# 2. prefix[i]: おもちゃ[0:i] を 箱[0:i-1] に割り当てられるか？ → 数が合わない。

# 改めて:
# - おもちゃを1つ除外する。除外した結果、残りのN-1個のおもちゃを、N-1個の箱に1対1で割り当てる。
# - これは、二部マッチングだが、貪欲で可能。

# 貪欲マッチング（標準）:
# おもちゃを昇順、箱を昇順。
# 両方のポインタを進めて、おもちゃ <= 箱ならマッチ、そうでなければ失敗。

# 今、1つのおもちゃをスキップできる。
# → どの時点でスキップすればよいか？

# 解法:
# - 左から貪欲にマッチングを試みるが、1回だけスキップできる。
# - skip_count = 0 で始める。
# - i: おもちゃインデックス, j: 箱インデックス
# - while i < N and j < N-1:
#       if A[i] <= B[j]:
#           i++, j++
#       else:
#           if skip_count == 0:
#               skip_count++ (and i++)
#           else:
#               失敗
# - 最後に、j == N-1 かつ i == N or (i==N-1 and skip_count==0 で最後をスキップ) など。

# しかし、スキップすべきはいつでもよいが、A[i] > B[j] のときスキップするのが最適か？

# 例: 
# A = [1,2,3], B = [2,3]
# - i=0, A[0]=1<=B[0]=2 → match → i=1,j=1
# - i=1, A[1]=2<=B[1]=3 → match → i=2,j=2 → 終了。スキップなし。→ 新しい箱のサイズは1? いや、全部マッチしたので、どれか1つを新しい箱に入れる。最小のxは max(使わなかったおもちゃ) の最小値。

# ここで重要な点:
# 新しい箱に入れるおもちゃは1つだけ。残りが既存の箱に完全マッチすればよい。
# スキップできるのは1回だけ。

# アルゴリズム:
# A.sort()
# B.sort()
# 
# i = 0  # Aのインデックス
# j = 0  # Bのインデックス
# skip_count = 0
# skipped_toy = None
#
# while i < N:
#   if j < N-1 and A[i] <= B[j]:
#       i += 1
#       j += 1
#   elif skip_count == 0:
#       skipped_toy = A[i]
#       skip_count += 1
#       i += 1
#   else:
#       # スキップできない
#       print(-1)
#       exit()
#
# if j != N-1:
#   # 箱が余っているが、おもちゃはもうない → 問題ない（スキップした分）
#   # ただし、j < N-1 なら箱が余っているが、おもちゃは全部処理済み → OK
#   pass
#
# このとき、skipped_toy が新しい箱に入る。x の最小値は skipped_toy。
# しかし、これが最適か？

# 例: 
# Input: 
# 2
# 999993332 999997752
# 662
# A = [999993332, 999997752], B = [662]
#
# i=0: A[0]=999... > B[0]=662 → スキップ（skip_count=0）→ skipped_toy = 999993332, i=1
# i=1: j=0 < 1, A[1]=999... > B[0]=662 → skip_count=1 → もうスキップできない → 失敗 → print -1
# → 正解！

# しかし、先に大きい方をスキップした方が良いか？

# 例: A=[1,10], B=[5]
# - i=0: A[0]=1 <= B[0]=5 → マッチ → i=1, j=1
# - i=1: j=1 >= len(B)=1 → j out → スキップ（skip_count=0）→ skipped_toy=10
# → x=10

# でも、もし先に10をスキップしていたら？
# - i=0: A[0]=1 <= B[0]=5 → マッチ → i=1, j=1 → 残りの10をスキップ → 同じ。

# しかし、順序が重要。常に小さいおもちゃから見て、マッチできればマッチ、できなければスキップ（1回だけ）。

# でも、最適なスキップはいつか？
# → 貪欲に、マッチできるならマッチ、できないならスキップ（1回まで）が最適。

# なぜなら、小さいおもちゃほどマッチしやすいので、スキップするのは大きなおもちゃの方が良い。
# しかし、小さなおもちゃが箱にマッチできない（A[i] > B[j]）なら、それより大きなおもちゃは絶対にマッチしないので、スキップせざるを得ない。

# よって、アルゴリズム:
# - Aを昇順、Bを昇順
# - i=0, j=0, skip_count=0, skipped_toy=None
# - while i < N:
#     if j < N-1 and A[i] <= B[j]:
#         i += 1
#         j += 1
#     elif skip_count == 0:
#         skipped_toy = A[i]
#         skip_count += 1
#         i += 1
#     else:
#         print(-1)
#         exit()
# - print(skipped_toy)

# しかし、これで十分か？

# 例: A=[1,2,3,4], B=[2,3,5]
# - i=0: A[0]=1<=B[0]=2 → match → i=1,j=1
# - i=1: A[1]=2<=B[1]=3 → match → i=2,j=2
# - i=2: A[2]=3<=B[2]=5 → match → i=3,j=3 → j==3==len(B), so j>=len(B)
# - i=3: j>=len(B) → skip_count==0 → skipped_toy=4 → x=4
# → 正しい。

# 別の例: A=[1,2,3,4], B=[1,2,3]
# - i=0: 1<=1 → match → i=1,j=1
# - i=1: 2<=2 → match → i=2,j=2
# - i=2: 3<=3 → match → i=3,j=3 → j=len(B)
# - i=3: j>=len(B) → skip_count==0 → skipped_toy=4 → x=4

# でも、もし3をスキップして4をB[2]=3に入れる？ → 4>3 → 無理。なので、4をスキップするしかない。

# 別の例: A=[1,2,4,5], B=[2,3,6]
# - i=0: 1<=2 → match → i=1,j=1
# - i=1: 2<=3 → match → i=2,j=2
# - i=2: 4<=6 → match → i=3,j=3 → j=len(B)
# - i=3: j>=len(B) → skip → skipped_toy=5 → x=5

# でも、もし5をスキップせず4をスキップしたら？
# - 4をスキップ → 残り: [1,2,5] と B=[2,3,6]
# - 1<=2 → match
# - 2<=3 → match
# - 5<=6 → match → 可能。x=4 で済む。
# → しかし、上のアルゴリズムは x=5 と出力してしまう！ → 間違い！

# 問題: スキップのタイミング。貪欲に「マッチできなければスキップ」ではなく、「マッチできるときはマッチ、マッチできなくてもスキップできるならスキップしてもよい」が、最適なスキップを選ぶ必要がある。

# つまり、複数のスキップの選択肢があり得る。

# 解決策: すべてのおもちゃについて、それをスキップしたときに残りがマッチするかをチェック。O(N) 回の貪欲マッチングで O(N^2) → TLE.

# 効率的な方法:
# - prefix[i]: おもちゃ[0:i] と 箱[0:i-1] までで完全マッチできるか？
# - suffix[i]: おもちゃ[i+1:N] と 箱[i:N-1] で完全マッチできるか？
# そして、おもちゃ i をスキップしたときに、prefix[i] と suffix[i] がともに成立すればOK。

# 具体的には:
# - A.sort(), B.sort()
# - prefix[i]: おもちゃ[0:i] を 箱[0:i-1] に割り当てられるか？ → おもちゃ i 個、箱 i-1 個 → 無理。
# 
# 正しくは:
# - おもちゃ i をスキップすると、残りは：
#   - 左: おもちゃ[0:i] （i個） → 箱[0:i-1] （i-1個）？ いや、箱は連続していない。

# 修正:
# - おもちゃを昇順: A[0..N-1]
# - 箱を昇順: B[0..N-2]
# - skip[i
