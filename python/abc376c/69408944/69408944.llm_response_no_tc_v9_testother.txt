```python
import bisect

N = int(input())
A = list(map(int, input().split()))
B = list(map(int, input().split()))

# Sort toys in descending order to handle largest toys first
A.sort(reverse=True)
B.sort()

# We are going to try to assign the largest unassigned toy to a box.
# Use binary search on the answer x (size of new box)

def can_store(x):
    # Available boxes: original B and one new box of size x
    unused_boxes = B[:]  # we'll simulate using boxes
    
    # Try to assign each toy (from largest to smallest)
    for a in A:
        # Check if we can use the new box
        if x >= a:
            # Mark this new box as used (only one such box)
            x = -1  # consume the new box
            continue
        
        # Otherwise, find the smallest box in B that can fit toy `a`
        pos = bisect.bisect_left(unused_boxes, a)
        if pos < len(unused_boxes):
            # Use this box
            unused_boxes.pop(pos)
        else:
            return False  # Cannot fit this toy
    return True

# Binary search on x
lo, hi = 1, 10**9 + 1
ans = -1

while lo <= hi:
    mid = (lo + hi) // 2
    if can_store(mid):
        ans = mid
        hi = mid - 1
    else:
        lo = mid + 1

print(ans)
```
