```python
import sys

def main():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    idx = 0
    N = int(data[idx])
    idx += 1
    Q = int(data[idx])
    idx += 1
    
    # Store the adjacency list for each connected component root
    comp = [set() for _ in range(N + 1)]
    for i in range(1, N + 1):
        comp[i].add(i)
    
    parent = list(range(N + 1))
    size = [1] * (N + 1)
    
    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])
        return parent[x]
    
    def union(x, y):
        rx = find(x)
        ry = find(y)
        if rx == ry:
            return
        if size[rx] < size[ry]:
            rx, ry = ry, rx
        # Merge ry into rx
        parent[ry] = rx
        size[rx] += size[ry]
        # Move all vertices from ry's component to rx's component
        comp[rx].update(comp[ry])
        comp[ry].clear()
    
    output = []
    for _ in range(Q):
        t = int(data[idx])
        idx += 1
        if t == 1:
            u = int(data[idx])
            v = int(data[idx + 1])
            idx += 2
            if find(u) != find(v):
                union(u, v)
        else:  # t == 2
            v = int(data[idx])
            k = int(data[idx + 1])
            idx += 2
            root = find(v)
            component = sorted(comp[root], reverse=True)
            if len(component) < k:
                output.append("-1")
            else:
                output.append(str(component[k - 1]))
    
    print('\n'.join(output))

if __name__ == "__main__":
    main()
```
