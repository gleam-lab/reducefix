class UnionFind:
    def __init__(self, n):
        self.par = list(range(n))
        self.size = [1] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]

    def unite(self, x, y):
        x = self.find(x)
        y = self.find(y)
        if x == y:
            return False
        if self.size[x] < self.size[y]:
            x, y = y, x
        self.par[y] = x
        self.size[x] += self.size[y]
        return True

    def same(self, x, y):
        return self.find(x) == self.find(y)

N, Q = map(int, input().split())

# Store the vertices in each connected component (as sorted lists in descending order)
components = [list() for _ in range(N+1)]
for i in range(1, N+1):
    components[i].append(i)

ans = []
uf = UnionFind(N+1)

for _ in range(Q):
    query = list(map(int, input().split()))
    if query[0] == 1:
        u, v = query[1], query[2]
        if not uf.same(u, v):
            root_u = uf.find(u)
            root_v = uf.find(v)
            # Always merge smaller into larger (by root index or by size, here we use union by size in UF)
            if uf.size[root_u] < uf.size[root_v]:
                u, v = v, u
                root_u, root_v = root_v, root_u
            
            # Merge components: move all vertices from root_v to root_u
            uf.unite(u, v)
            new_root = uf.find(u)
            
            # Combine the two lists and sort in descending order
            merged = components[root_u] + components[root_v]
            merged.sort(reverse=True)
            # Keep only top 10 largest (since k <= 10)
            components[new_root] = merged[:10]
            
            # Clear the old component
            components[root_v] = []
    else:
        v, k = query[1], query[2]
        root = uf.find(v)
        if k <= len(components[root]):
            ans.append(components[root][k-1])
        else:
            ans.append(-1)

for a in ans:
    print(a)
