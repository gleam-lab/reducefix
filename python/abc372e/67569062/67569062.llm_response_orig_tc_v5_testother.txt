class UnionFind:
    def __init__(self, n):
        self.par = list(range(n))
        self.size = [1] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]

    def unite(self, x, y):
        x = self.find(x)
        y = self.find(y)
        if x == y:
            return False
        if self.size[x] < self.size[y]:
            x, y = y, x
        self.par[y] = x
        self.size[x] += self.size[y]
        return True

    def same(self, x, y):
        return self.find(x) == self.find(y)

N, Q = map(int, input().split())

# For each component root, maintain a sorted list (descending) of the largest up to 10 vertex numbers
component_vertices = {}
for i in range(1, N+1):
    component_vertices[i] = [i]

uf = UnionFind(N+1)
ans = []

for _ in range(Q):
    query = list(map(int, input().split()))
    if query[0] == 1:
        u, v = query[1], query[2]
        if not uf.same(u, v):
            root_u = uf.find(u)
            root_v = uf.find(v)
            # Ensure root_u is the new root (larger size or u >= v by union strategy)
            if uf.size[root_u] < uf.size[root_v] or (uf.size[root_u] == uf.size[root_v] and root_u > root_v):
                root_u, root_v = root_v, root_u
            
            # Merge the vertices from root_v into root_u
            merged = component_vertices[root_u] + component_vertices[root_v]
            # Sort in descending order and keep at most 10 largest
            merged.sort(reverse=True)
            component_vertices[root_u] = merged[:10]
            # Remove the old component
            del component_vertices[root_v]
            # Perform union
            uf.unite(u, v)
    else:  # query[0] == 2
        v, k = query[1], query[2]
        root = uf.find(v)
        if root in component_vertices and len(component_vertices[root]) >= k:
            ans.append(component_vertices[root][k-1])
        else:
            ans.append(-1)

for a in ans:
    print(a)
