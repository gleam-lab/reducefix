```python
import math

N = int(input())
X = 0
Y = 0
total_cost = 0.0

# Start from origin (0, 0)
prev_x, prev_y = 0, 0

for i in range(N):
    x, y = map(int, input().split())
    # Distance from previous point to current point
    dx = prev_x - x
    dy = prev_y - y
    total_cost += math.sqrt(dx*dx + dy*dy)
    prev_x, prev_y = x, y

# Distance from last point back to origin
dx = prev_x - 0
dy = prev_y - 0
total_cost += math.sqrt(dx*dx + dy*dy)

print(total_cost)
```
