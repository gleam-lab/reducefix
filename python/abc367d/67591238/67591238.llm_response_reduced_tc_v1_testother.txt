from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和（時計回りの距離）
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# 全体の周長
total = cum[N]

# mod M の値の出現回数をカウント
count_mod = defaultdict(int)
answer = 0

for val in cum:
    mod = val % M
    answer += count_mod[mod]
    count_mod[mod] += 1

# 同じ地点同士のペア（s == t）は除く必要があるが、
# cumの中の同じmodを持つ異なるインデックス間の距離がMの倍数 → s != tなら有効
# ただし、一周以上回る経路は想定されておらず、s→tは一方向かつ最短ではない可能性がある。

# 問題：時計回りのs→tの距離がMの倍数となる(s,t)ペア（s≠t）の数

# 上のコードは「累積和のmodが等しい」→「その間の距離がMの倍数」と正しく捉えている。
# しかし、問題文では「最小ステップ数」がMの倍数と言っているが、
# 時計回りに進む場合、s→tの距離は通常 min(時計回り, 反時計回り) だが、
# ここでは明確に「時計回りに進む」と指定されている。

# 実際には、「時計回りにsからtまで進む」＝ cum[t] - cum[s] （ただしt>s）
# よって、cum[i] ≡ cum[j] (mod M) なら、i→jの時計回り距離はMの倍数。

# しかし、i=j のときも cum[i] ≡ cum[j] なので、s=t のケースが含まれてしまう。
# そして、s≠t という条件があるので、s=t は除外しなければならない。

# 一方で、上のループではすべてのi<jについて cum[i]%M == cum[j]%M をカウントしている。
# これはs=i, t=j（ただしi≠j）に対応し、s≠tである。

# つまり、rest area 1 から rest area 1 へ時計回りに行く → 一周する
# 距離 = A_1 = 136872, これはM=1の倍数 → 条件満たす
# しかし、問題文：「s ≠ t」
# つまり、sとtが同じ番号のrest areaならダメ。

# したがって、s=1, t=1 は s≠t を満たさない → 不正なペア

# では、このペアは何を表しているか？
# cum[0] は rest area 1 開始前（または位置0）
# cum[1] は rest area 1 を出発して一周した後（再び1）

# つまり、i=0 → j=1 は 「rest area 1 から出発して、rest area 1 に戻る」
# → s=1, t=1

# よって、s≠t という条件から除外すべき。

# しかし、他のペアも同様に、sとtは rest area の番号で考えるべき。

# したがって、cum[i] と cum[j] (i<j) の差は、rest area i+1 から rest area j+1 へ時計回りに進む距離

# つまり、s = i+1, t = j+1

# ここで、s≠t とは、i+1 ≠ j+1 → i ≠ j

# cum = [0, 136872]
# ペア: i=0, j=1 → j=1=N → i=0, j=N → s = (0%1)+1 = 1, t=(1%1)+1=1 → s=t
# よって、このペアは除外すべき。

# しかし、上記のコードではこのような除外をしていない。

# 修正：同じmodの累積和のペアを数えるが、s=tになるようなペア（つまりi≡j mod N）を除外する？

# いや、問題は「s≠t」なので、s=tのペアだけを除けばよい。

# しかし、s=tになるのは、i≡j (mod N) のときのみで、しかもi<j、i,j∈[0,N] → それは i=0, j=N のみ。

# したがって、もし cum[0] % M == cum[N] % M ならば、そのペア(i=0,j=N)はs=tなのでカウントしない。

# しかし、現在のコードでは、defaultdictを使ってすべてのi<jで cum[i]%M == cum[j]%M をカウントしている。
# 特に、i=0,j=N のペアも含まれる。

# 解決法：まず全ペアを数えて、もし cum[0] % M == cum[N] % M なら、その分（1ペア）を引く？

# いや、それだと不十分。なぜなら、cum[0]とcum[N]が同じmodでも、他に cum[k]%M==x となるkがあれば、
# そのkとcum[0]やcum[N]のペアも存在する。その中でs=tになるのは、i=0,j=Nのペアだけ。

# したがって、全体のカウントから、i=0,j=N で s=t になるペアが含まれていたらそれを引けばよい。

# しかし、そもそも i=0,j=N のペアがカウントされるのは、cum[0]%M == cum[N]%M のときのみ。

# そして、そのペアはちょうど1つ。

# したがって、最終的な答えから、もし cum[0] % M == cum[N] % M なら 1 を引く？

# いや、待て。cum[0] と cum[N] は同じ値とは限らないが、mod Mで一致すればペアとしてカウントされる。

# そして、そのペア(i=0,j=N)だけがs=tとなる。

# 他のペア（例：i=0,j=k<N）→ s=1, t=k+1 ≠1（k≥1なら）→ s≠t
# i=k>0,j=N → s=k+1, t=1 → k+1≠1（k≥1）→ s≠t

# よって、唯一のs=tペアは i=0,j=N
