from collections import defaultdict

N, M = map(int, input().split())
A = list(map(int, input().split()))

# 累積和
cum = [0] * (N + 1)
for i in range(N):
    cum[i + 1] = cum[i] + A[i]

# mod M の値の出現回数
count_mod = defaultdict(int)
answer = 0

for val in cum:
    mod = val % M
    answer += count_mod[mod]
    count_mod[mod] += 1

# 全体の合計が M の倍数の場合、0 から始まるループが存在するが、
# s ≠ t なので、同じ位置同士のペアは除く必要がある。
# 上のループでは累積和の同じ mod を持つ異なるインデックス間が valid な (s,t) ペアに対応。
# ただし、元の問題では時計回りに s から t へ移動する距離が cum[t] - cum[s] だが、
# ここで扱っているのは cum[i] % M の一致 → 差が M の倍数。
# よってアルゴリズム自体は正しい。

# しかし、上のコードは N=1 の場合でも cum[0]=0, cum[1]=A[0] を見て、
# もし A[0] % M == 0 なら、0 と A[0] で mod 0 が2回出て answer=1 になる。
# しかし、このとき s=1, t=1？ → s≠t より無効。または s=1→t=2(=1) だが、N=1 なので rest area 1 しかなく、
# s≠t なのに t が 1 しかない → 実際には valid なペアなし。

# 問題は：N=1 のとき、s と t は異なり得ない（rest area が1つしかない）ので、ペア数は0。
# しかし、上のコードは累積和配列が長さ2 (index 0 と 1) なので、同じ mod ならカウントしてしまう。

# つまり、このアルゴリズムは「異なる i < j に対して cum[j] - cum[i] ≡ 0 (mod M) となる組の数」を数えている。
# これは「i から j への時計回りの距離が M の倍数」という意味。
# このとき、rest area i+1 から rest area j への移動（i+1 を s, j を t）に対応する。
# ただし、j=N のとき次は1に戻るが、ここでは累積和が全体として対応している。

# しかし、N=1 のとき：
#   cum[0] = 0
#   cum[1] = A[0]
#   ペア (0,1): cum[1]-cum[0] = A[0] → A[0] % M == 0 なら有効？
#   これは rest area 1 から rest area 1 への移動（一周）を意味する。
#   しかし問題文：「minimum number of steps required to walk clockwise from rest area s to rest area t (s ≠ t)」
#   s ≠ t なので、s=1, t=1 は許されない。
#   よって、一周して戻るような s→s のパスはすべて無視しなければならない。

# したがって、累積和のインデックス i と j (i < j) について、
# 「j - i >= N」のような条件が必要か？いや、違う。

# 実際には、時計回りの距離は単純に cum[j] - cum[i] だが、
# これが一周以上を含む場合も最小距離とは言えない。

# 重要な点：問題文の「minimum number of steps」は、時計回り or 反時計回りのうち短い方？
# しかし問題文：「It takes A_i steps to walk clockwise from rest area i to rest area i+1」
# そして「minimum number of steps required to walk clockwise from rest area s to rest area t」
# 「clockwise from s to t」とあるが、「minimum」なので反時計回りの方が短ければそちらを使う？

# 再読：「The minimum number of steps required to walk clockwise from rest area s to rest area t (s ≠ t) is a multiple of M.」

# 「walk clockwise from s to t」の最小ステップ？ 時計回り固定なのか？
# しかし「minimum」があるので、おそらく「s から t までの最短距離（時計回りまたは反時計回りのうち短い方）」の意。

# 実際、サンプルを考えると、そうでないと整合性が取れない。

# 例えば N=2, A=[1,1], M=1 なら、全ペアの最短距離は1（M=1の倍数）、ペア数は2: (1,2), (2,1)

# しかし、時計回り固定だと、(2,1) は時計回りに一周 → 1+1=2 歩。最小ではなくなる。

# よって、「minimum number of steps」は双方向のうち短い方を指す。

# したがって、s から t への距離は min(cum[t]-cum[s], total-cum[t]+cum[s]) （ただし cum[s] <= cum[t] を仮定）

# しかし、累積和の差分だけで処理するのは難しくなる。

# ところが、元のコードは「時計回りの距離が M の倍数」と解釈して cum[t]-cum[s] ≡ 0 (mod M) を数えている。
# これは誤り。

# 正しい解釈：最短距離 d(s,t) = min(clockwise_dist, counter_clockwise_dist)
# そして d(s,t) ≡ 0 (mod M) であるような (s,t) (s≠t) の数を求めよ。

# しかし、d(s,t) が M の倍数という条件は、
# d(s,t) ∈ {0, M, 2M, ...} だが、d(s,t)=0 は s=t のみ → s≠t より不可。
# よって d(s,t) は M, 2M, ..., かつ d(s,t) ≤ total//2 （最短なので半周以下）

# したがって、各 s,t に対して d(s,t) を計算し、それが M の倍数か判定すればよいが、O(N^2) になり TLE。

# 別のアプローチ：

# 全体の距離を S = sum(A) とする。
# s から t への時計回り距離を D とすると、反時計回りは S-D。
# 最短距離 d = min(D, S-D)
# 条件：d > 0 かつ d % M == 0.

# d = min(D, S-D) なので、d <= S/2.
# よって d は M の倍数で、かつ 1 <= d <= floor(S/2).

# また、D は累積和の差分で表せる。

# しかし、d = D または d = S-D のいずれか。
# よって、d ≡ 0 (mod M) とは、
#   (i) D ≡ 0 (mod M) かつ D <= S-D → D <= S/2
#   or
#   (ii) S-D ≡ 0 (mod M) かつ S-D <= D → D >= S/2
#       → S ≡ D (mod M) かつ D >= S/2

# つまり、s から t への時計回り距離 D が、
#   (a) D ≡ 0 (mod M) かつ D <= S/2
#   or
#   (b) D ≡ S (mod M) かつ D >= S/2
# を満たす。

# 注意：D=0 は s=t なので除外。また、D=S も s=t と同じ（一周）なので除外。
# しかし、s≠t なので D は 1 以上 S-1 以下の値を取る。

# さらに、D >= S/2 かつ D ≡ S (mod M) のとき、S-D ≡ 0 (mod M) かつ S-D <= S/2.

# したがって、条件は「D ≡ 0 (mod M) かつ 1 <= D <= S/2」または「D ≡ S (mod M) かつ ceil(S/2) <= D <= S-1」
# ただし、S が偶数で D = S/2 のとき、D = S-D なので両方同じ。これを重複カウントしないよう注意。

# そこで、D = S/2 かつ S ≡ 0 (mod M) のとき、D ≡ 0 (mod M) かつ D ≡ S (mod M) なので、
# 同じパスに対して二重にカウントされる可能性がある。
# しかし、この場合、最短距離は D = S/2 であり、条件は一つだけ満たすべき。
# よって、片方の条件のみでカウントする。

# しかし、実際には D = S/2 のとき、(a) と (b) の両方に該当するが、
# 一つの (s,t) ペアに対して一つの D なので、二重カウントは発生しない。
# なぜなら、(a) と (b) は互いに排他的ではないが、同じ D に対してどちらか一方でしかカウントしないようにはなっていない。
# つまり、一つの (s,t) が (a) と (b) の両方に該当する場合、二重にカウントされる。

# 例：S=4, M=2, D=2 → D≡0 mod 2, かつ D≡S=4≡0 mod 2 → 両方満たす。
# しかし、d=min(2,2)=2 で、一つのペアに対して条件を一度だけ満たすべき。
# よって、(a) と (b) を足すと二重カウント。

# 対策：D = S/2 かつ S ≡ 0 (mod M) のとき、そのような D を持つペアを (a) でカウントし、(b) ではカウントしない。
# つまり、(b) の条件を D > S/2 に変更する。

# まとめ：
# 各 (s,t) (s≠t) について、時計回り距離 D を考える。
# その D が次のいずれかを満たすときカウント：
#   (1) 1 <= D <= S/2 かつ D ≡ 0 (mod M)
#   (2) S/2 < D <= S-1 かつ D ≡ S (mod M)

# これにより、D = S/2 の場合は (1) でのみカウント。

# あとは、すべての s≠t に対する時計回り距離 D を生成し、上記条件を満たすものを数える。
# しかし O(N^2) なので無理。

# 代わりに、累積和の配列を使って、距離 D として現れる値を集計。

# cum[0..N] を用意。cum[i] は rest area 1 から i+1 への時計回り距離（cum[0]=0, cum[1]=A0, ...）
# 任意の s,t (s≠t) について、s から t への時計回り距離 D は？
# s が i, t が j のとき（rest area番号）、
#   if i < j: D = cum[j-1] - cum[i-1]
#   if i > j: D = (cum[N] - cum[i-1]) + cum[j-1] = S - cum[i-1] + cum[j-1]
# これは、cum の差分で表すのが難しい。

# 代わりに、円形なので、すべての i<j に対して cum[j]-cum[i] を距離として考える。
# これは、rest area i+1 から j+1 への時計回り距離。
# また、i>j の場合は cum[i]-cum[j] が逆方向距離なので、時計回り距離は S - (cum[i]-cum[j])。

# よって、すべての i≠j に対して、
#   if cum[i] < cum[j]: D = cum[j] - cum[i]
#   else: D = S - (cum[i] - cum[j]) = S - cum[i] + cum[j]

# しかし、これは非効率。

# 別の方法：すべての異なる i,j に対して、D = (cum[j] - cum[i]) % S とすると、
# 0 < D < S で、D が時計回り距離。
# そして、最短距離 d = min(D, S-D)

# したがって、条件は d % M == 0 and d > 0.

# これをすべての i≠j でチェックすると O(N^2) で TLE。

# 高速化：累積和の mod M ではなく、距離 D 自体の性質を使う。

# 考察：条件を満たす (s,t) の数 = 
#   (時計回り距離 D が 1<=D<=S/2 かつ D≡0 mod M となる (s,t) 数)
#   + (時計回り距離 D が S/2<D<S かつ D≡S mod M となる (s,t) 数)

# ここで、D = cum[j] - cum[i] (mod S) だが、S が大きい。

# 代わりに、累積和をソートして、尺取り法や二分探索で D の範囲を調べる。

# しかし、D は cum[j] - cum[i] または S - cum[i] + cum[j] の両方があり、複雑。

# 実際、多くの提出では、最初のコードのように「cum[i] ≡ cum[j] (mod M) かつ i≠j」を数えている。
# しかし、それは「時計回り距離が M の倍数」という意味で、最短距離が M の倍数とは限らない。

# なぜなら、最短距離は min(D, S-D) だから。

# したがって、最初のアプローチは誤り。

# 正しい方法は、次の通り（参考：類似問題の解法）：

# 最短距離が M の倍数 ⇔ min(D, S-D) ≡ 0 (mod M)
# ⇔ D ≡ 0 (mod M) または D ≡ S (mod M) だが、min(D,S-D) が M の倍数になる条件。

# しかし、min(D,S-D) ≡ 0 (mod M) は、
#   min(D, S-D) = k*M となる整数 k>=1 が存在し、かつ k*M <= S/2.

# したがって、可能な d = M, 2M, 3M, ..., floor(S/2)//M * M の各 d に対して、
# その d が最短距離となるような (s,t) の数を数える。

# しかし、d が与えられたとき、D=d または D=S-d となる (s,t) の数を数える。
# そして、D は累積和の差分。

# そこで、累積和の配列 cum[0..N-1] を作り（cum[0]=0, cum[i]=A0+...+A{i-1}）、
# すべての i<j に対して、D = cum[j]-cum[i] を考える。
# これは rest area i+1 から j+1 への時計回り距離。

# また、i>j の場合は D = cum[i]-cum[j] が逆方向なので、時計回り距離は S-D.

# しかし、対称性より、すべての i≠j に対して、D = |cum[i] - cum[j]| ではない（円環だから）。

# 正確には、円環上の距離は min(|c_i - c_j|, S - |c_i - c_j|) ではない。
# なぜなら cum はソートされていないかもしれないが、順序は保たれている。

# 実際、cum は単調増加：0 = cum[0] < cum[1] < ... < cum[N] = S.

# したがって、i<j ならば cum[j] > cum[i] なので、D = cum[j] - cum[i].

# この D は rest area i+1 から j+1 への時計回り距離。

# また、j<i のときは、cum[i] - cum[j] が時計回り距離ではなく、逆。
# j<i のとき、rest area i+1 から j+1 への時計回り距離は (S - cum[i]) + cum[j].

# しかし、i+1 から j+1 の時計回り距離は cum[j] - cum[i] + S? いや、
# cum[i] から cum[j] まで、i>j なら、cum[j] + S - cum[i].

# したがって、任意の i,j (i≠j) に対して、時計回り距離 D_ij = (cum[j] - cum[i]) % S.
# ただし、正の値として、
#   if j > i:
