import sys

data = sys.stdin.read().splitlines()
if not data: 
    print(0)

n, m = map(int, data[0].split())
pieces = []
for i in range(1, m + 1):
    a, b = map(int, data[i].split())
    pieces.append((a, b))

# Set of squares that are either occupied or can be captured by opponent pieces
occupied = set(pieces)
captured = set()  # Squares that are under threat (can be captured)

# Knight move offsets
knight_moves = [
    (2, 1), (1, 2), (-1, 2), (-2, 1),
    (-2, -1), (-1, -2), (1, -2), (2, -1)
]

# For each existing piece, compute which squares it can capture
for a, b in pieces:
    for dx, dy in knight_moves:
        nx, ny = a + dx, b + dy
        if 1 <= nx <= n and 1 <= ny <= n:
            captured.add((nx, ny))

# Safe placement: must be empty and not capturable
# Empty squares: total - occupied
# Among empty squares, remove those that are captured
safe_squares = n * n - len(occupied) - len(captured - occupied)
