```python
n, m = map(int, input().split())
s = set()

# Define the 8 possible knight moves (relative positions)
knight_moves = [
    (2, 1), (1, 2), (-1, 2), (-2, 1),
    (-2, -1), (-1, -2), (1, -2), (2, -1)
]

for _ in range(m):
    a, b = map(int, input().split())
    # Add the piece's position to the set of occupied squares
    s.add((a, b))
    # For each knight move, add the square that would be captured by this piece
    for dx, dy in knight_moves:
        x, y = a + dx, b + dy
        if 1 <= x <= n and 1 <= y <= n:
            s.add((x, y))

# Total empty squares minus squares where we would be captured
ans = n * n - len(s)

print(ans)
```
