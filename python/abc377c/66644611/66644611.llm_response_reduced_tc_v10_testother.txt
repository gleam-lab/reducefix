```python
n, m = map(int, input().split())
captured = set()

# Define the 8 possible capture moves (knight-like moves)
moves = [(2,1), (1,2), (-1,2), (-2,1), (-2,-1), (-1,-2), (1,-2), (2,-1)]

for _ in range(m):
    a, b = map(int, input().split())
    # Add the piece itself (though not strictly needed for capturing logic)
    captured.add((a, b))
    # For each of the 8 directions, mark the square that would be captured
    for dx, dy in moves:
        nx, ny = a + dx, b + dy
        if 1 <= nx <= n and 1 <= ny <= n:
            captured.add((nx, ny))

# Total empty squares minus those that are under capture
ans = n * n - len(captured)

print(ans)
```
