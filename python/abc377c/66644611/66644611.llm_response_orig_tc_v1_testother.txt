```python
n, m = map(int, input().split())
captured = set()

# Define the 8 possible capture moves (knight moves)
moves = [(2,1), (1,2), (-1,2), (-2,1), (-2,-1), (-1,-2), (1,-2), (2,-1)]

for _ in range(m):
    a, b = map(int, input().split())
    # For each existing piece, mark all squares it can capture
    for dx, dy in moves:
        nx, ny = a + dx, b + dy
        if 1 <= nx <= n and 1 <= ny <= n:
            captured.add((nx, ny))

# Count empty squares that are not captured by any piece
# Total squares: n*n
# Subtract squares with existing pieces (m) and captured squares
ans = n * n - m - len(captured)

print(ans)
```
