n, m = map(int, input().split())
captured = set()

# Define the 8 possible capture moves (knight moves)
moves = [
    (2, 1), (1, 2), (-1, 2), (-2, 1),
    (-2, -1), (-1, -2), (1, -2), (2, -1)
]

# For each existing piece, mark all squares it can capture
for _ in range(m):
    a, b = map(int, input().split())
    for dx, dy in moves:
        x, y = a + dx, b + dy
        if 1 <= x <= n and 1 <= y <= n:
            captured.add((x, y))

# Count empty squares that are not captured
total_squares = n * n
occupied_or_captured = captured.copy()

# Add all existing pieces to the occupied/captured set
for i in range(m):
    # We need to re-read or store the pieces
    pass

# Since we already processed the input, we need to store the pieces
pieces = []
captured.clear()

# Re-process input
import sys
data = sys.stdin.read().split()
if data:
    n = int(data[0])
    m = int(data[1])
    pieces = []
    index = 2
    for i in range(m):
        a = int(data[index])
        b = int(data[index + 1])
        index += 2
        pieces.append((a, b))
        for dx, dy in moves:
            x, y = a + dx, b + dy
            if 1 <= x <= n and 1 <= y <= n:
                captured.add((x, y))

# Add all piece positions to occupied_or_captured
for a, b in pieces:
    captured.add((a, b))

# The answer is total squares minus occupied/captured squares
ans = total_squares - len(captured)
print(ans)
