```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        print(1)
        return
    
    n = int(data[0])
    m = int(data[1])
    pieces = []
    index = 2
    for i in range(m):
        a = int(data[index])
        b = int(data[index+1])
        index += 2
        pieces.append((a, b))
    
    # Set of all squares that are either occupied or threatened
    forbidden = set()
    
    # The 8 directions where a piece at (x,y) can attack
    attacks = [(2,1), (1,2), (-1,2), (-2,1), (-2,-1), (-1,-2), (1,-2), (2,-1)]
    
    # Add all occupied squares
    for a, b in pieces:
        forbidden.add((a, b))
    
    # For each existing piece, mark all squares it can attack
    for a, b in pieces:
        for dx, dy in attacks:
            nx, ny = a + dx, b + dy
            if 1 <= nx <= n and 1 <= ny <= n:
                forbidden.add((nx, ny))
    
    # Safe placements = total squares - forbidden squares
    total_squares = n * n
    safe_squares = total_squares - len(forbidden)
    print(safe_squares)

main()
```
