```python
import sys

def main():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    k = int(data[1])
    A = list(map(int, data[2:2+n]))
    
    # We need to remove exactly K elements, so we keep (n - k) elements.
    m = n - k
    if m == 1:
        print(0)
        return
    
    # Sort the array along with original indices to track order
    sorted_A = sorted((val, idx) for idx, val in enumerate(A))
    
    # The idea is that after removing K elements, the remaining sequence B
    # must be a contiguous subsequence in the original array? 
    # But note: we can remove arbitrary elements, not necessarily contiguous.
    # However, the final sequence preserves the original order.
    
    # Important insight:
    # To minimize (max(B) - min(B)), we want B to have values as close as possible.
    # So we should consider a window of m = n-k values from the sorted array.
    # But they must appear in the original array in increasing index order?
    
    # Alternate approach:
    # Since we can choose any K elements to remove, we are free to pick any set of m = n-k
    # elements that remain, as long as their relative order is preserved.
    # But we cannot reorder them.
    
    # However, we can think greedily:
    # Suppose we fix the smallest and largest value in B. Then all values between them (in sorted order)
    # can potentially be included, but only if their indices allow forming a valid subsequence.
    
    # Simpler idea: try all possible windows of size m in the sorted array.
    # For each window [i, i+m-1], check whether the corresponding elements can appear in increasing index order
    # (i.e., form a valid subsequence in the original array).
    # If yes, then candidate answer = sorted_A[i+m-1][0] - sorted_A[i][0]
    
    # How to check if m elements (from sorted_A[i] to sorted_A[i+m-1]) appear in increasing index order?
    # Extract their original indices and check if they are strictly increasing.
    
    # But note: we don't require consecutive in original array, just preserved order.
    # So if the indices of selected elements are in increasing order, then they can form B.
    
    ans = float('inf')
    
    # Try every window of size m in the sorted array
    for i in range(n - m + 1):
        window = sorted_A[i:i+m]
        indices = [idx for val, idx in window]
        # Check if indices are in increasing order
        if indices == sorted(indices):
            candidate = window[-1][0] - window[0][0]
            if candidate < ans:
                ans = candidate
    
    print(ans)

if __name__ == "__main__":
    main()
```
