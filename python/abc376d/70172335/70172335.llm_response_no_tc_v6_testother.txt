from collections import deque

def main():
    n, m = map(int, input().split())
    graph = [[] for _ in range(n)]
    
    for _ in range(m):
        a, b = map(int, input().split())
        a -= 1
        b -= 1
        graph[a].append(b)
    
    # BFS from vertex 0 (vertex 1 in 1-indexed) to find shortest cycle containing it
    # We want the shortest path from 0 back to 0
    dist = [-1] * n
    q = deque()
    q.append(0)
    dist[0] = 0
    
    while q:
        u = q.popleft()
        for v in graph[u]:
            if dist[v] == -1:  # Not visited yet
                dist[v] = dist[u] + 1
                q.append(v)
                # Don't stop here, we need to explore all paths
            elif v == 0:  # Found a way back to start
                # This handles the case where we can return to 0
                # But we need to make sure this forms a valid cycle
                pass
    
    # Actually, we need to find shortest cycle containing vertex 0
    # We can do: for each edge that goes back to 0, consider path length + 1
    # Or better: run BFS and when we reach 0 again through a path, that's a cycle
    
    # Let's restart with proper approach
    # Find shortest cycle containing vertex 0 using BFS
    if n == 1:
        # Check if there's a self-loop
        for neighbor in graph[0]:
            if neighbor == 0:
                print(1)
                return
        print(-1)
        return
    
    # Standard approach: remove edges going to vertex 0 temporarily,
    # then find shortest path from neighbors of 0 to 0
    min_cycle = float('inf')
    
    # Find all vertices that have direct edge to vertex 0
    # Instead, we can do: for each outgoing edge from 0, 
    # find shortest path from that neighbor back to 0
    
    for neighbor in graph[0]:
        if neighbor == 0:  # self-loop
            min_cycle = min(min_cycle, 1)
    
    # Now find shortest path from any direct successor of 0 back to 0
    # But we must avoid using the reverse of the first edge immediately
    # Do BFS starting from all direct neighbors of 0 (except 0 itself)
    
    if not graph[0]:  # No outgoing edges from 0
        print(-1)
        return
    
    dist = [-1] * n
    q = deque()
    
    # Initialize: add all direct neighbors of 0
    for neighbor in graph[0]:
        if neighbor != 0:  # skip self-loop (already handled)
            dist[neighbor] = 1
            q.append(neighbor)
    
    # Mark vertex 0 as unvisited (we want to reach it)
    # But don't start from it
    dist[0] = -1  # Ensure we can "visit" it when we come back
    
    while q:
        u = q.popleft()
        for v in graph[u]:
            if dist[v] == -1:  # Not visited in this BFS
                dist[v] = dist[u] + 1
                if v == 0:
                    min_cycle = min(min_cycle, dist[u] + 1)
                else:
                    q.append(v)
    
    print(min_cycle if min_cycle != float('inf') else -1)

if __name__ == "__main__":
    main()
