from collections import defaultdict, deque

def main():
    n, m = map(int, input().split())
    graph = defaultdict(list)
    
    for _ in range(m):
        a, b = map(int, input().split())
        a -= 1
        b -= 1
        graph[a].append(b)
    
    # BFS from vertex 0 (vertex 1 in 1-indexed)
    INF = float('inf')
    dist = [INF] * n
    dist[0] = 0
    queue = deque([0])
    
    while queue:
        u = queue.popleft()
        for v in graph[u]:
            if dist[v] == INF:
                dist[v] = dist[u] + 1
                queue.append(v)
    
    # Find minimum cycle containing vertex 0
    min_cycle = INF
    
    # Check all edges that can form a cycle with vertex 0
    for u in range(n):
        if dist[u] != INF:  # u is reachable from 0
            for v in graph[u]:
                if v == 0:  # Edge from u to 0 forms a cycle
                    min_cycle = min(min_cycle, dist[u] + 1)
    
    print(min_cycle if min_cycle != INF else -1)

if __name__ == "__main__":
    main()
